45 /*--- Detect preferred SIMD size (bytes). This only covers x86 architectures. ---*/
46 #if defined(__AVX512F__)
47 constexpr size_t PREFERRED_SIZE = 64;
48 #elif defined(__AVX__)
49 constexpr size_t PREFERRED_SIZE = 32;
50 #elif defined(__SSE2__)
51 constexpr size_t PREFERRED_SIZE = 16;
52 #else
53 constexpr size_t PREFERRED_SIZE = 8;
54 #endif
55 
56 /*!
57  * \brief Convert the SIMD size (bytes) to a lenght (num elems).
58  */
59 template<class T>
60 constexpr size_t preferredLen() { return PREFERRED_SIZE / sizeof(T); }
61 
62 template<>
63 constexpr size_t preferredLen<su2double>() {
64 #ifdef CODI_REVERSE_TYPE
65  /*--- Use a SIMD size of 1 for reverse AD, larger sizes increase
66  * the pre-accumulation time with no performance benefit. ---*/
67  return 1;
68 #else
69  /*--- For forward AD there is a performance benefit. This covers
70  * forward AD and primal mode (su2double == passivedouble). ---*/
71  return PREFERRED_SIZE / sizeof(passivedouble);
72 #endif
73 }

75 /*!
76  * \class Array
77  * \brief A simple SIMD type relying on implicit vectorization, i.e. done by
78  * the compiler, explicitly vectorized specializations are defined after.
79  * \note This class gets its math operator overloads from CVecExpr, the
80  * specializations do not use expression templates, IF YOU NEED A NEW FUNCTION,
81  * define it both in vector_expressions.hpp and in special_vectorization.hpp.
82  */
83 template<class Scalar_t, size_t N = preferredLen<Scalar_t>()>
84 class Array : public CVecExpr<Array<Scalar_t,N>, Scalar_t> {
85 #define FOREACH for(size_t k=0; k<N; ++k)
86  static_assert(N > 0, "Invalid SIMD size");
87 public:
88  using Scalar = Scalar_t;
89  enum : size_t {Size = N};
90  enum : size_t {Align = Size*sizeof(Scalar)};
91  static constexpr bool StoreAsRef = true;
92 
93 private:
94  alignas(Size*sizeof(Scalar)) Scalar x_[N];
95 
96 public:
97 #define ARRAY_BOILERPLATE \
98  /*!--- Access elements ---*/ \
99  FORCEINLINE Scalar& operator[] (size_t k) { return x_[k]; } \
100  FORCEINLINE const Scalar& operator[] (size_t k) const { return x_[k]; } \
101  /*!--- Constructors ---*/ \
102  FORCEINLINE Array() = default; \
103  FORCEINLINE Array(Scalar x) { bcast(x); } \
104  FORCEINLINE Array(std::initializer_list<Scalar> vals) { \
105  auto it = vals.begin(); FOREACH { x_[k] = *it; ++it; } \
106  } \
107  FORCEINLINE Array(Scalar x0, Scalar dx) { FOREACH x_[k] = x0 + k*dx; } \
108  FORCEINLINE Array(const Scalar* ptr) { load(ptr); } \
109  template<class T> \
110  FORCEINLINE Array(const Scalar* beg, const T& off) { gather(beg,off); } \
111  /*!--- Reduction operations ---*/ \
112  FORCEINLINE Scalar sum() const { Scalar s(0); FOREACH { s+=x_[k]; } return s; } \
113  FORCEINLINE Scalar dot(const Array& other) const { \
114  Scalar s(0); FOREACH { s += x_[k] * other[k]; } return s; \
115  }
116 
117 #if defined(CODI_REVERSE_TYPE) || defined(CODI_FORWARD_TYPE)
118  /*--- These are not very nice but without them it would not be
119  * possible to assign literals to Arrays of active types. ---*/
120  template<class U = Scalar, su2enable_if<std::is_same<U,su2double>::value> = 0>
121  FORCEINLINE Array(passivedouble x) { bcast(x); }
122  template<class U = Scalar, su2enable_if<std::is_same<U,su2double>::value> = 0>
123  FORCEINLINE Array& operator= (passivedouble x) { bcast(x); return *this; }
124 #endif
125 
126  ARRAY_BOILERPLATE
127 
128  /*! \brief Copy construct from expression. */
129  template<class U>
130  FORCEINLINE Array(const CVecExpr<U,Scalar>& expr) {
131  FOREACH x_[k] = expr.derived()[k];
132  }
133 
134  /*--- Implementation of the construction primitives. ---*/
135 
136  FORCEINLINE void bcast(Scalar x) { FOREACH x_[k] = x; }
137  FORCEINLINE void load(const Scalar* ptr) { FOREACH x_[k] = ptr[k]; }
138  FORCEINLINE void loada(const Scalar* ptr) { load(ptr); }
139  FORCEINLINE void store(Scalar* ptr) const { FOREACH ptr[k] = x_[k]; }
140  FORCEINLINE void storea(Scalar* ptr) const { store(ptr); }
141  FORCEINLINE void stream(Scalar* ptr) const { store(ptr); }
142  template<class T>
143  FORCEINLINE void gather(const Scalar* begin, const T& offsets) { FOREACH x_[k] = begin[offsets[k]]; }
144 
145  /*--- Compound assignment operators. ---*/
146 
147 #define MAKE_COMPOUND(OP) \
148  FORCEINLINE Array& operator OP (Scalar x) { FOREACH { x_[k] OP x; } return *this; } \
149  template<class U> \
150  FORCEINLINE Array& operator OP (const CVecExpr<U,Scalar>& expr) { \
151  FOREACH { x_[k] OP expr.derived()[k]; } return *this; \
152  }
153  MAKE_COMPOUND(=)
154  MAKE_COMPOUND(+=)
155  MAKE_COMPOUND(-=)
156  MAKE_COMPOUND(*=)
157  MAKE_COMPOUND(/=)
158 #undef MAKE_COMPOUND
159 
160 #undef FOREACH
161 };
