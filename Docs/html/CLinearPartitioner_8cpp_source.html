31 void CLinearPartitioner::Initialize(unsigned long global_count,
32  unsigned long offset,
33  bool isDisjoint) {
34 
35  /*--- Store MPI size ---*/
36 
37  size = SU2_MPI::GetSize();
38 
39  /*--- Resize the vectors containing our linear partitioning. ---*/
40 
41  firstIndex.resize(size);
42  lastIndex.resize(size);
43  sizeOnRank.resize(size);
44  cumulativeSizeBeforeRank.resize(size+1);
45 
46  /*--- Compute the number of points that will be on each processor.
47  This is a linear partitioning with the addition of a simple load
48  balancing for any remainder points. ---*/
49 
50  unsigned long quotient = 0;
51  if (global_count >= (unsigned long)size)
52  quotient = global_count/size;
53 
54  int remainder = int(global_count%size);
55  for (int ii = 0; ii < size; ii++) {
56  sizeOnRank[ii] = quotient + int(ii < remainder);
57  }
58 
59  /*--- Store the local number of nodes on each proc in the linear
60  partitioning, the beginning/end index, and the linear partitioning
61  within an array in cumulative storage format. ---*/
62 
63  unsigned long adjust = 0;
64  if (isDisjoint) adjust = 1;
65 
66  firstIndex[0] = offset;
67  lastIndex[0] = firstIndex[0] + sizeOnRank[0] - adjust;
68  cumulativeSizeBeforeRank[0] = 0;
69  for (int iProc = 1; iProc < size; iProc++) {
70  firstIndex[iProc] = lastIndex[iProc-1] + adjust;
71  lastIndex[iProc] = firstIndex[iProc] + sizeOnRank[iProc] - adjust;
72  cumulativeSizeBeforeRank[iProc] = (cumulativeSizeBeforeRank[iProc-1] +
73  sizeOnRank[iProc-1]);
74  }
75  cumulativeSizeBeforeRank[size] = global_count;
76 
77 }

79 unsigned long CLinearPartitioner::GetRankContainingIndex(unsigned long index) const {
80 
81  /*--- Initial guess ---*/
82 
83  unsigned long iProcessor = min<unsigned long>(index/sizeOnRank[0], size-1);
84 
85  /*--- Move up or down until we find the processor. ---*/
86 
87  if (index >= cumulativeSizeBeforeRank[iProcessor])
88  while(index >= cumulativeSizeBeforeRank[iProcessor+1])
89  iProcessor++;
90  else
91  while(index < cumulativeSizeBeforeRank[iProcessor])
92  iProcessor--;
93 
94  return iProcessor;
95 }
