36 /*!
37  * \brief OpenMP worksharing construct used in CSysVector for loops.
38  * \note The loop will only run in parallel if methods are called from a
39  * parallel region (if not the results will still be correct).
40  * Static schedule to reduce overhead, chunk size determined at initialization.
41  * "nowait" clause is safe when calling CSysVector methods after each other
42  * as the loop size is the same. Methods of other classes that operate on a
43  * CSysVector and do not have the same work scheduling must use a
44  * SU2_OMP_BARRIER before using the vector.
45  */

47 #ifdef HAVE_OMP_SIMD

51 #endif
52 #define END_CSYSVEC_PARFOR END_SU2_OMP_FOR
53 #else
54 #define CSYSVEC_PARFOR SU2_OMP_SIMD

87  /*!
88  * \brief Helper to unpack (transpose) a SIMD input block.
89  */
90  template <size_t N, size_t nVar, class VecTypeSIMD, class F>
91  FORCEINLINE static void UnpackBlock(const VecTypeSIMD& in, simd::Array<F, N> mask, ScalarType out[][nVar]) {
92  static_assert(VecTypeSIMD::StaticSize, "This method requires static size vectors.");
93  for (size_t i = 0; i < nVar; ++i) {
94  SU2_OMP_SIMD_IF_NOT_AD
95  for (size_t k = 0; k < N; ++k) out[k][i] = mask[k] * in[i][k];
96  }
97  }

178  /*!
179  * \brief Set our values (resizing if required) by copying from other, the derivative information is lost.
180  * \param[in] other - source CSysVector
181  */
182  template <class T>
183  void PassiveCopy(const CSysVector<T>& other) {
184  /*--- This is a method and not the overload of an operator to make sure who
185  * calls it knows the consequence to the derivative information (lost) ---*/
186 
187  /*--- check if self-assignment, otherwise perform deep copy ---*/
188  if ((const void*)this == (const void*)&other) return;
189 
190  SU2_OMP_SAFE_GLOBAL_ACCESS(Initialize(other.GetNBlk(), other.GetNBlkDomain(), other.GetNVar(), nullptr, true, false);)
191 
192  CSYSVEC_PARFOR
193  for (auto i = 0ul; i < nElm; i++) vec_val[i] = SU2_TYPE::GetValue(other[i]);
194  END_CSYSVEC_PARFOR
195  }

289  /*!
290  * \brief Dot product between "this" and an expression.
291  * \param[in] expr - Expression.
292  * \return Result of dot product
293  */
294  template <class T>
295  ScalarType dot(const VecExpr::CVecExpr<T, ScalarType>& expr) const {
296  static ScalarType dotRes;
297  /*--- All threads get the same "view" of the vectors and shared variable. ---*/
298  SU2_OMP_SAFE_GLOBAL_ACCESS(dotRes = 0.0;)
299 
300  /*--- Local dot product for each thread. ---*/
301  ScalarType sum = 0.0;
302 
303  CSYSVEC_PARFOR
304  for (auto i = 0ul; i < nElmDomain; ++i) {
305  sum += vec_val[i] * expr.derived()[i];
306  }
307  END_CSYSVEC_PARFOR
308 
309  /*--- Update shared variable with "our" partial sum. ---*/
310  atomicAdd(sum, dotRes);
311 
312 #ifdef HAVE_MPI
313  /*--- Reduce across all mpi ranks, only master thread communicates. ---*/
314  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
315  sum = dotRes;
316  const auto mpi_type = (sizeof(ScalarType) < sizeof(double)) ? MPI_FLOAT : MPI_DOUBLE;
317  SelectMPIWrapper<ScalarType>::W::Allreduce(&sum, &dotRes, 1, mpi_type, MPI_SUM, SU2_MPI::GetComm());
318  }
319  END_SU2_OMP_SAFE_GLOBAL_ACCESS
320 #else
321  /*--- Make view of result consistent across threads. ---*/
322  SU2_OMP_BARRIER
323 #endif
324 
325  return dotRes;
326  }

398  /*!
399  * \brief Vectorized version of SetBlock, sets multiple iPoint's.
400  * \param[in] iPoint - SIMD integer, the positions to update.
401  * \param[in] vector - Vector of SIMD scalars.
402  * \param[in] mask - Optional scale factor (axpy type operation).
403  * \note Nothing is updated if the mask is 0.
404  */
405  template <size_t N, class T, class VecTypeSIMD, class F = ScalarType>
406  FORCEINLINE void SetBlock(simd::Array<T, N> iPoint, const VecTypeSIMD& vector, simd::Array<F, N> mask = 1) {
407  /*--- "Transpose" and scale input vector. ---*/
408  constexpr size_t nVar = VecTypeSIMD::StaticSize;
409  assert(nVar == this->nVar);
410  ScalarType vec[N][nVar];
411  UnpackBlock(vector, mask, vec);
412 
413  /*--- Update one by one skipping if mask is 0. ---*/
414  for (size_t k = 0; k < N; ++k) {
415  if (mask[k] == 0) continue;
416  SU2_OMP_SIMD
417  for (size_t i = 0; i < nVar; ++i) vec_val[iPoint[k] * nVar + i] = vec[k][i];
418  }
419  }
420 
421  /*!
422  * \brief Vectorized version of UpdateBlocks, updates multiple i/jPoint's.
423  * \note See SIMD overload of SetBlock.
424  */
425  template <size_t N, class T, class VecTypeSIMD, class F = ScalarType>
426  FORCEINLINE void UpdateBlocks(simd::Array<T, N> iPoint, simd::Array<T, N> jPoint, const VecTypeSIMD& vector,
427  simd::Array<F, N> mask = 1) {
428  /*--- "Transpose" and scale input vector. ---*/
429  constexpr size_t nVar = VecTypeSIMD::StaticSize;
430  assert(nVar == this->nVar);
431  ScalarType vec[N][nVar];
432  UnpackBlock(vector, mask, vec);
433 
434  /*--- Update one by one skipping if mask is 0. ---*/
435  for (size_t k = 0; k < N; ++k) {
436  if (mask[k] == 0) continue;
437  SU2_OMP_SIMD
438  for (size_t i = 0; i < nVar; ++i) {
439  vec_val[iPoint[k] * nVar + i] += vec[k][i];
440  vec_val[jPoint[k] * nVar + i] -= vec[k][i];
441  }
442  }
443  }
