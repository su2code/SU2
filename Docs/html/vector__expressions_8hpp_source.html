42 /*!
43  * \brief Base vector expression class.
44  * \ingroup BLAS
45  * \param[in] Derived - The class that inherits from this one to use the expressions.
46  * \param[in] Scalar - Associated scalar type, prevents implicit conversions between exprs.
47  * \note Derived classes must implement operator[], and at least operator= with
48  * expressions (that is when they are evaluated). They must also contain a constexpr
49  * boolean "StoreAsRef", indicating whether to store them by value (false) or by
50  * reference (true), when composing expressions.
51  * Expression classes need to be stored by value to allow nested expressions to
52  * propagate correctly (i.e. if "Scalar" has its own expression templates).
53  * Vector classes should be stored by reference to avoid copies, especially if they
54  * allocate memory dynamically.
55  */
56 template<class Derived, class Scalar>
57 class CVecExpr {
58 public:
59  /*!
60  * \brief Cast the expression to Derived, usually to allow evaluation via operator[].
61  */
62  FORCEINLINE const Derived& derived() const { return static_cast<const Derived&>(*this); }
63 
64  // Allowed from C++14, allows nested expression propagation without
65  // manually calling derived() on the expression being evaluated.
66  //FORCEINLINE auto operator[] (size_t i) const { return derived()[i]; }
67 };

112 #define MAKE_UNARY_FUN(FUN, EXPR, IMPL) \
113 /*!--- Expression class. ---*/ \

138 /*!--- Expression class. ---*/ \
139 template<class U, class V, class Scalar> \
140 class EXPR : public CVecExpr<EXPR<U,V,Scalar>, Scalar> { \
141  store_t<const U> u; \
142  store_t<const V> v; \
143 public: \
144  static constexpr bool StoreAsRef = false; \
145  FORCEINLINE EXPR(const U& u_, const V& v_) : u(u_), v(v_) {} \
146  FORCEINLINE auto operator[] (size_t i) const RETURNS( IMPL(u[i], v[i]) ) \
147 }; \

165  * fmin and fmax return by value and thus are fine, but they would force
166  * conversions to double, to avoid that we provide integer overloads.
