31 void CAdjFluidIteration::Preprocess(COutput* output, CIntegration**** integration, CGeometry**** geometry,
32  CSolver***** solver, CNumerics****** numerics, CConfig** config,
33  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
34  CFreeFormDefBox*** FFDBox, unsigned short val_iZone, unsigned short val_iInst) {
35  unsigned short iMesh;
36  bool harmonic_balance = (config[ZONE_0]->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE);
37  bool dynamic_mesh = config[ZONE_0]->GetGrid_Movement();
38  unsigned long InnerIter = 0;
39  unsigned long TimeIter = config[ZONE_0]->GetTimeIter();
40 
41  /*--- For the unsteady adjoint, load a new direct solution from a restart file. ---*/
42 
43  if (((dynamic_mesh && TimeIter == 0) || (config[val_iZone]->GetTime_Marching() != TIME_MARCHING::STEADY)) && !harmonic_balance) {
44  int Direct_Iter = SU2_TYPE::Int(config[val_iZone]->GetUnst_AdjointIter()) - SU2_TYPE::Int(TimeIter) - 1;
45  if (rank == MASTER_NODE && val_iZone == ZONE_0 && (config[val_iZone]->GetTime_Marching() != TIME_MARCHING::STEADY))
46  cout << endl << " Loading flow solution from direct iteration " << Direct_Iter << "." << endl;
47  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->LoadRestart(
48  geometry[val_iZone][val_iInst], solver[val_iZone][val_iInst], config[val_iZone], Direct_Iter, true);
49  }
50 
51  /*--- Continuous adjoint Euler, Navier-Stokes or Reynolds-averaged Navier-Stokes (RANS) equations ---*/
52 
53  if ((InnerIter == 0) || (config[val_iZone]->GetTime_Marching() != TIME_MARCHING::STEADY)) {
54  const auto kind_solver = config[val_iZone]->GetKind_Solver();
55  switch (kind_solver) {
56  case MAIN_SOLVER::ADJ_EULER:
57  case MAIN_SOLVER::ADJ_NAVIER_STOKES:
58  case MAIN_SOLVER::ADJ_RANS:
59  config[val_iZone]->SetGlobalParam(kind_solver, RUNTIME_FLOW_SYS);
60  break;
61  default:
62  break;
63  }
64 
65  /*--- Solve the Euler, Navier-Stokes or Reynolds-averaged Navier-Stokes (RANS) equations (one iteration) ---*/
66 
67  if (rank == MASTER_NODE && val_iZone == ZONE_0)
68  cout << "Begin direct solver to store flow data (single iteration)." << endl;
69 
70  if (rank == MASTER_NODE && val_iZone == ZONE_0)
71  cout << "Compute residuals to check the convergence of the direct problem." << endl;
72 
73  integration[val_iZone][val_iInst][FLOW_SOL]->MultiGrid_Iteration(geometry, solver, numerics, config,
74  RUNTIME_FLOW_SYS, val_iZone, val_iInst);
75 
76  if (config[val_iZone]->GetKind_Solver() == MAIN_SOLVER::ADJ_RANS) {
77  /*--- Solve the turbulence model ---*/
78 
79  config[val_iZone]->SetGlobalParam(MAIN_SOLVER::ADJ_RANS, RUNTIME_TURB_SYS);
80  integration[val_iZone][val_iInst][TURB_SOL]->SingleGrid_Iteration(geometry, solver, numerics, config,
81  RUNTIME_TURB_SYS, val_iZone, val_iInst);
82 
83  /*--- Solve transition model ---*/
84 
85  if (config[val_iZone]->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
86  config[val_iZone]->SetGlobalParam(MAIN_SOLVER::ADJ_RANS, RUNTIME_TRANS_SYS);
87  integration[val_iZone][val_iInst][TRANS_SOL]->SingleGrid_Iteration(geometry, solver, numerics, config,
88  RUNTIME_TRANS_SYS, val_iZone, val_iInst);
89  }
90  }
91 
92  /*--- Output the residual (visualization purpouses to identify if
93  the direct solution is converged)---*/
94  if (rank == MASTER_NODE && val_iZone == ZONE_0)
95  cout << "log10[Maximum residual]: " << log10(solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->GetRes_Max(0))
96  << ", located at point " << solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->GetPoint_Max(0) << "." << endl;
97 
98  /*--- Compute gradients of the flow variables, this is necessary for sensitivity computation,
99  note that in the direct Euler problem we are not computing the gradients of the primitive variables ---*/
100 
101  if (config[val_iZone]->GetKind_Gradient_Method() == GREEN_GAUSS)
102  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->SetPrimitive_Gradient_GG(geometry[val_iZone][val_iInst][MESH_0],
103  config[val_iZone]);
104  if (config[val_iZone]->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES)
105  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->SetPrimitive_Gradient_LS(geometry[val_iZone][val_iInst][MESH_0],
106  config[val_iZone]);
107 
108  /*--- Set contribution from cost function for boundary conditions ---*/
109 
110  for (iMesh = 0; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
111  /*--- Set the value of the non-dimensional coefficients in the coarse levels, using the fine level solution ---*/
112 
113  solver[val_iZone][val_iInst][iMesh][FLOW_SOL]->SetTotal_CD(
114  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->GetTotal_CD());
115  solver[val_iZone][val_iInst][iMesh][FLOW_SOL]->SetTotal_CL(
116  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->GetTotal_CL());
117  solver[val_iZone][val_iInst][iMesh][FLOW_SOL]->SetTotal_CT(
118  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->GetTotal_CT());
119  solver[val_iZone][val_iInst][iMesh][FLOW_SOL]->SetTotal_CQ(
120  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->GetTotal_CQ());
121 
122  /*--- Compute the adjoint boundary condition on Euler walls ---*/
123 
124  solver[val_iZone][val_iInst][iMesh][ADJFLOW_SOL]->SetForceProj_Vector(
125  geometry[val_iZone][val_iInst][iMesh], solver[val_iZone][val_iInst][iMesh], config[val_iZone]);
126  }
127 
128  if (rank == MASTER_NODE && val_iZone == ZONE_0) cout << "End direct solver, begin adjoint problem." << endl;
129  }
130 }
131 void CAdjFluidIteration::Iterate(COutput* output, CIntegration**** integration, CGeometry**** geometry,
132  CSolver***** solver, CNumerics****** numerics, CConfig** config,
133  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
134  CFreeFormDefBox*** FFDBox, unsigned short val_iZone, unsigned short val_iInst) {
135  const auto kind_solver = config[val_iZone]->GetKind_Solver();
136  switch (kind_solver) {
137  case MAIN_SOLVER::ADJ_EULER:
138  case MAIN_SOLVER::ADJ_NAVIER_STOKES:
139  case MAIN_SOLVER::ADJ_RANS:
140  config[val_iZone]->SetGlobalParam(kind_solver, RUNTIME_ADJFLOW_SYS);
141  break;
142  default:
143  break;
144  }
145 
146  /*--- Iteration of the flow adjoint problem ---*/
147 
148  integration[val_iZone][val_iInst][ADJFLOW_SOL]->MultiGrid_Iteration(geometry, solver, numerics, config,
149  RUNTIME_ADJFLOW_SYS, val_iZone, val_iInst);
150 
151  /*--- Iteration of the turbulence model adjoint ---*/
152 
153  if ((config[val_iZone]->GetKind_Solver() == MAIN_SOLVER::ADJ_RANS) && (!config[val_iZone]->GetFrozen_Visc_Cont())) {
154  /*--- Adjoint turbulence model solution ---*/
155 
156  config[val_iZone]->SetGlobalParam(MAIN_SOLVER::ADJ_RANS, RUNTIME_ADJTURB_SYS);
157  integration[val_iZone][val_iInst][ADJTURB_SOL]->SingleGrid_Iteration(geometry, solver, numerics, config,
158  RUNTIME_ADJTURB_SYS, val_iZone, val_iInst);
159  }
160 }
161 void CAdjFluidIteration::Update(COutput* output, CIntegration**** integration, CGeometry**** geometry,
162  CSolver***** solver, CNumerics****** numerics, CConfig** config,
163  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
164  CFreeFormDefBox*** FFDBox, unsigned short val_iZone, unsigned short val_iInst) {
165  su2double Physical_dt, Physical_t;
166  unsigned short iMesh;
167  unsigned long TimeIter = config[ZONE_0]->GetTimeIter();
168 
169  /*--- Dual time stepping strategy ---*/
170 
171  if ((config[val_iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
172  (config[val_iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND)) {
173  /*--- Update dual time solver ---*/
174 
175  for (iMesh = 0; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
176  integration[val_iZone][val_iInst][ADJFLOW_SOL]->SetDualTime_Solver(
177  geometry[val_iZone][val_iInst][iMesh], solver[val_iZone][val_iInst][iMesh][ADJFLOW_SOL], config[val_iZone],
178  iMesh);
179 
180  integration[val_iZone][val_iInst][ADJFLOW_SOL]->SetDualTime_Geometry(
181  geometry[val_iZone][val_iInst][iMesh], solver[val_iZone][val_iInst][iMesh][MESH_SOL], config[val_iZone],
182  iMesh);
183 
184  integration[val_iZone][val_iInst][ADJFLOW_SOL]->SetConvergence(false);
185  }
186 
187  Physical_dt = config[val_iZone]->GetDelta_UnstTime();
188  Physical_t = (TimeIter + 1) * Physical_dt;
189  if (Physical_t >= config[val_iZone]->GetTotal_UnstTime())
190  integration[val_iZone][val_iInst][ADJFLOW_SOL]->SetConvergence(true);
191  }
192 }
