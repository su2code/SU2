39 void CCSVFileWriter::Write_Data(string val_filename){
40 
41  /*--- Routine to write the surface CSV files (ASCII). We
42  assume here that, as an ASCII file, it is safer to merge the
43  surface data onto the master rank for writing for 2 reasons:
44  (a) as a surface file, the amount of data should be much less
45  than the volume solution, and (b) writing ASCII files in parallel
46  requires serializing the IO calls with barriers, which ruins
47  the performance at moderate to high rank counts. ---*/
48 
49  unsigned short iVar;
50 
51  int iProcessor, nProcessor = size;
52 
53  unsigned long iPoint, index;
54  unsigned long Buffer_Send_nVertex[1], *Buffer_Recv_nVertex = nullptr;
55  unsigned long nLocalVertex_Surface = 0, MaxLocalVertex_Surface = 0;
56 
57  const vector<string> fieldNames = dataSorter->GetFieldNames();
58 
59  ofstream Surf_file;
60  Surf_file.precision(15);
61 
62  /*--- Find the max number of surface vertices among all
63  partitions so we can set up buffers. The master node will handle
64  the writing of the CSV file after gathering all of the data. ---*/
65 
66  nLocalVertex_Surface = dataSorter->GetnPoints();
67  Buffer_Send_nVertex[0] = nLocalVertex_Surface;
68  if (rank == MASTER_NODE) Buffer_Recv_nVertex = new unsigned long[nProcessor];
69 
70  /*--- Communicate the number of local vertices on each partition
71  to the master node with collective calls. ---*/
72 
73  SU2_MPI::Allreduce(&nLocalVertex_Surface, &MaxLocalVertex_Surface, 1,
74  MPI_UNSIGNED_LONG, MPI_MAX, SU2_MPI::GetComm());
75 
76  SU2_MPI::Gather(&Buffer_Send_nVertex, 1, MPI_UNSIGNED_LONG,
77  Buffer_Recv_nVertex, 1, MPI_UNSIGNED_LONG,
78  MASTER_NODE, SU2_MPI::GetComm());
79 
80  /*--- Allocate buffers for send/recv of the data and global IDs. ---*/
81 
82  su2double *bufD_Send = new su2double[MaxLocalVertex_Surface*fieldNames.size()]();
83  su2double *bufD_Recv = nullptr;
84 
85  unsigned long *bufL_Send = new unsigned long [MaxLocalVertex_Surface]();
86  unsigned long *bufL_Recv = nullptr;
87 
88  /*--- Load send buffers with the local data on this rank. ---*/
89 
90  index = 0;
91  for (iPoint = 0; iPoint < nLocalVertex_Surface; iPoint++) {
92 
93  /*--- Global index values. ---*/
94 
95  bufL_Send[iPoint] = dataSorter->GetGlobalIndex(iPoint);
96 
97  /*--- Solution data. ---*/
98 
99  for (iVar = 0; iVar < fieldNames.size(); iVar++){
100  bufD_Send[index] = dataSorter->GetData(iVar, iPoint);
101  index++;
102  }
103 
104  }
105 
106  /*--- Only the master rank allocates buffers for the recv. ---*/
107 
108  if (rank == MASTER_NODE) {
109  bufD_Recv = new su2double[nProcessor*MaxLocalVertex_Surface*fieldNames.size()]();
110  bufL_Recv = new unsigned long[nProcessor*MaxLocalVertex_Surface];
111  }
112 
113  /*--- Collective comms of the solution data and global IDs. ---*/
114 
115  SU2_MPI::Gather(bufD_Send, (int)MaxLocalVertex_Surface*fieldNames.size(), MPI_DOUBLE,
116  bufD_Recv, (int)MaxLocalVertex_Surface*fieldNames.size(), MPI_DOUBLE, MASTER_NODE, SU2_MPI::GetComm());
117 
118  SU2_MPI::Gather(bufL_Send, (int)MaxLocalVertex_Surface, MPI_UNSIGNED_LONG,
119  bufL_Recv, (int)MaxLocalVertex_Surface, MPI_UNSIGNED_LONG, MASTER_NODE, SU2_MPI::GetComm());
120 
121  /*--- The master rank alone writes the surface CSV file. ---*/
122 
123  if (rank == MASTER_NODE) {
124 
125  /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
126  val_filename.append(fileExt);
127 
128  /*--- Open the CSV file and write the header with variable names. ---*/
129  Surf_file.open(val_filename.c_str(), ios::out);
130  Surf_file << "\"Point\",";
131  for (iVar = 0; iVar < fieldNames.size()-1; iVar++) {
132  Surf_file << "\"" << fieldNames[iVar] << "\",";
133  }
134  Surf_file << "\"" << fieldNames[fieldNames.size()-1] << "\"" << endl;
135 
136  /*--- Loop through all of the collected data and write each node's values ---*/
137 
138  for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
139  for (iPoint = 0; iPoint < Buffer_Recv_nVertex[iProcessor]; iPoint++) {
140 
141  /*--- Current index position for global index access. ---*/
142 
143  index = iProcessor*MaxLocalVertex_Surface + iPoint;
144 
145  /*--- Write global index values. ---*/
146 
147  Surf_file << bufL_Recv[index] << ", ";
148 
149  /*--- Reset index for solution data access. ---*/
150 
151  index = (iProcessor*MaxLocalVertex_Surface*fieldNames.size() +
152  iPoint*fieldNames.size());
153 
154  /*--- Write the solution data for each field variable. ---*/
155 
156  for (iVar = 0; iVar < fieldNames.size(); iVar++){
157  Surf_file << scientific << bufD_Recv[index + iVar];
158  if (iVar != fieldNames.size() -1) Surf_file << ", ";
159  }
160  Surf_file << endl;
161 
162  }
163  }
164 
165  /*--- Close the file. ---*/
166 
167  Surf_file.close();
168 
169  }
170 
171  /*--- Free temporary memory. ---*/
172 
173  if (rank == MASTER_NODE) {
174  delete [] bufL_Recv;
175  delete [] bufD_Recv;
176  delete [] Buffer_Recv_nVertex;
177  }
178  delete [] bufL_Send;
179  delete [] bufD_Send;
180 }
