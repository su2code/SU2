162 void CFaceOfElement::CreateUniqueNumberingWithOrientation(void) {
163 
164  /*--- Determine the element type and create the unique numbering accordingly. ---*/
165  bool swapElements = false;
166  switch( nCornerPoints ) {
167 
168  case 2: {
169  /* Element is a line. Check if the node numbering must be swapped. If so
170  also the element information must be swapped, because element 0 is to
171  the left of the face and element 1 to the right. */
172  if(cornerPoints[1] < cornerPoints[0]) {
173  swap(cornerPoints[0], cornerPoints[1]);
174  swapElements = true;
175  }
176  break;
177  }
178 
179  case 3: {
180  /* Element is a triangle. The vertices are sorted in increasing order.
181  If the sequence of the new numbering is opposite to the current
182  numbering, the element information must be exchanged, because
183  element 0 is to the left of the face and element 1 to the right. */
184  unsigned long nn[] = {cornerPoints[0], cornerPoints[1], cornerPoints[2]};
185  unsigned short ind = 0;
186  if(nn[1] < nn[ind]) ind = 1;
187  if(nn[2] < nn[ind]) ind = 2;
188 
189  unsigned short indm1 = ind==0 ? 2 : ind - 1; // Next lower index.
190  unsigned short indp1 = ind==2 ? 0 : ind + 1; // Next upper index.
191 
192  if(nn[indp1] < nn[indm1]) {
193 
194  /* The orientation of the triangle remains the same.
195  Store the new sorted node numbering. */
196  cornerPoints[0] = nn[ind];
197  cornerPoints[1] = nn[indp1];
198  cornerPoints[2] = nn[indm1];
199  }
200  else {
201 
202  /* The orientation of the triangle changes. Store the new
203  sorted node numbering and set swapElements to true. */
204  cornerPoints[0] = nn[ind];
205  cornerPoints[1] = nn[indm1];
206  cornerPoints[2] = nn[indp1];
207  swapElements = true;
208  }
209 
210  break;
211  }
212 
213  case 4: {
214  /* Element is a quadrilateral. The vertices are sorted in increasing order
215  under the condition neighboring vertices remain neighbors. If the
216  sequence of the new numbering is opposite to the current
217  numbering, the element information must be exchanged, because
218  element 0 is to the left of the face and element 1 to the right. */
219  unsigned long nn[] = {cornerPoints[0], cornerPoints[1],
220  cornerPoints[2], cornerPoints[3]};
221  unsigned short ind = 0;
222  if(nn[1] < nn[ind]) ind = 1;
223  if(nn[2] < nn[ind]) ind = 2;
224  if(nn[3] < nn[ind]) ind = 3;
225 
226  unsigned short indm1 = ind==0 ? 3 : ind - 1; // Next lower index.
227  unsigned short indp1 = ind==3 ? 0 : ind + 1; // Next upper index.
228  unsigned short indp2 = ind>=2 ? ind - 2 : ind + 2; // Opposite index.
229 
230  if(nn[indp1] < nn[indm1]) {
231 
232  /* The orientation of the quadrilateral remains the same.
233  Store the new sorted node numbering. */
234  cornerPoints[0] = nn[ind];
235  cornerPoints[1] = nn[indp1];
236  cornerPoints[2] = nn[indp2];
237  cornerPoints[3] = nn[indm1];
238  }
239  else {
240 
241  /* The orientation of the quadrilateral changes. Store the new
242  sorted node numbering and set swapElements to true. */
243  cornerPoints[0] = nn[ind];
244  cornerPoints[1] = nn[indm1];
245  cornerPoints[2] = nn[indp2];
246  cornerPoints[3] = nn[indp1];
247  swapElements = true;
248  }
249 
250  break;
251  }
252 
253  default: {
254  ostringstream message;
255  message << "Unknown surface element type with " << nCornerPoints
256  << " corners." << endl;
257  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
258  }
259  }
260 
261  /* Swap the element information, if needed. */
262  if( swapElements ) {
263  swap(elemID0, elemID1);
264  swap(nPolyGrid0, nPolyGrid1);
265  swap(nPolySol0, nPolySol1);
266  swap(nDOFsElem0, nDOFsElem1);
267  swap(elemType0, elemType1);
268  swap(faceID0, faceID1);
269  }
270 }

297 bool CMatchingFace::operator<(const CMatchingFace &other) const {
298 
299  /* First compare the number of corner points. ---*/
300  if(nCornerPoints != other.nCornerPoints) return nCornerPoints < other.nCornerPoints;
301 
302  /*--- Determine the tolerance for comparing both objects. ---*/
303  const su2double tol = min(tolForMatching, other.tolForMatching);
304 
305  /*--- Loop over the number of corner points and dimensions and compare the
306  coordinates. If considered different, return true if the current face
307  is considered smaller and false otherwise. ---*/
308  for(unsigned short k=0; k<nCornerPoints; ++k) {
309  for(unsigned short l=0; l<nDim; ++l) {
310  if(fabs(cornerCoor[k][l] - other.cornerCoor[k][l]) > tol)
311  return cornerCoor[k][l] < other.cornerCoor[k][l];
312  }
313  }
314 
315  /*--- Both objects are considered the same. Return false. ---*/
316  return false;
317 }

319 void CMatchingFace::SortFaceCoordinates(void) {
320 
321  /*--- Determine the tolerance for a matching point for this face. This is
322  accomplished by computing the minimum distance between the points of
323  the face, multiplied by a relative tolerance. ---*/
324  for(unsigned short k=0; k<nCornerPoints; ++k) {
325  for(unsigned short j=(k+1); j<nCornerPoints; ++j) {
326  su2double dist = 0.0;
327  for(unsigned short l=0; l<nDim; ++l) {
328  su2double ds = cornerCoor[k][l] - cornerCoor[j][l];
329  dist += ds*ds;
330  }
331  dist = sqrt(dist);
332 
333  if(k==0 && j==1) tolForMatching = dist;
334  else tolForMatching = min(tolForMatching, dist);
335  }
336  }
337 
338  tolForMatching *= 1.e-6;
339 
340  /*--- Sort the points in increasing order based on the coordinates.
341  An insertion sort algorithm is used, which is quite efficient
342  for at most four corner points. ---*/
343  for(unsigned short k=1; k<nCornerPoints; ++k) {
344  for(unsigned short j=k; j>0; --j) {
345 
346  /* Check if cornerCoor[j] is considered less than cornerCoor[j-1]. */
347  bool lessThan = false;
348  for(unsigned short l=0; l<nDim; ++l) {
349  if(fabs(cornerCoor[j][l] - cornerCoor[j-1][l]) > tolForMatching) {
350  lessThan = cornerCoor[j][l] < cornerCoor[j-1][l];
351  break;
352  }
353  }
354 
355  /* If cornerCoor[j] is less than cornerCoor[j-1] they must be swapped.
356  Otherwise an exit can be made from the j-loop. */
357  if( lessThan ) {
358  for(unsigned short l=0; l<nDim; ++l)
359  swap(cornerCoor[j][l], cornerCoor[j-1][l]);
360  }
361  else break;
362  }
363  }
364 }

383 void CPhysicalGeometry::Read_SU2_Format_Parallel_FEM(CConfig *config,
384  string val_mesh_filename,
385  unsigned short val_iZone,
386  unsigned short val_nZone) {
387  string text_line, Marker_Tag;
388  ifstream mesh_file;
389  string::size_type position;
390  unsigned long nDOFsGrid_Local = 0, loc_element_count = 0;
391  bool domain_flag = false;
392  bool time_spectral = config->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE;
393  unsigned short nMarker_Max = config->GetnMarker_Max();
394  nZone = val_nZone;
395 
396  /*--- Initialize counters for local/global points & elements ---*/
397  Global_nPoint = 0; Global_nPointDomain = 0; Global_nElem = 0;
398  nelem_edge = 0; Global_nelem_edge = 0;
399  nelem_triangle = 0; Global_nelem_triangle = 0;
400  nelem_quad = 0; Global_nelem_quad = 0;
401  nelem_tetra = 0; Global_nelem_tetra = 0;
402  nelem_hexa = 0; Global_nelem_hexa = 0;
403  nelem_prism = 0; Global_nelem_prism = 0;
404  nelem_pyramid = 0; Global_nelem_pyramid = 0;
405 
406  /*--- Allocate memory for the linear partition of the elements of the mesh.
407  These arrays are the size of the number of ranks. ---*/
408 
409  beg_node = new unsigned long[size];
410  end_node = new unsigned long[size];
411  nPointLinear = new unsigned long[size];
412 
413  /*--- Open grid file ---*/
414 
415  mesh_file.open(val_mesh_filename.c_str(), ios::in);
416 
417  /*--- Check the grid ---*/
418 
419  if (mesh_file.fail())
420  SU2_MPI::Error(string("There is no mesh file (CPhysicalGeometry)!! ") + val_mesh_filename,
421  CURRENT_FUNCTION);
422 
423  /*--- If more than one, find the zone in the mesh file ---*/
424 
425  if (val_nZone > 1 || time_spectral) {
426  if (time_spectral) {
427  if (rank == MASTER_NODE) cout << "Reading time spectral instance " << val_iZone+1 << ":" << endl;
428  } else {
429  while (getline (mesh_file,text_line)) {
430  /*--- Search for the current domain ---*/
431  position = text_line.find ("IZONE=",0);
432  if (position != string::npos) {
433  text_line.erase (0,6);
434  unsigned short jDomain = atoi(text_line.c_str());
435  if (jDomain == val_iZone+1) {
436  if (rank == MASTER_NODE) cout << "Reading zone " << val_iZone+1 << " points:" << endl;
437  break;
438  }
439  }
440  }
441  }
442  }
443 
444  /*--- Read grid file with format SU2 ---*/
445 
446  while (getline (mesh_file, text_line)) {
447 
448  /*--- Read the dimension of the problem ---*/
449 
450  position = text_line.find ("NDIME=",0);
451  if (position != string::npos) {
452  if (domain_flag == false) {
453  text_line.erase (0,6); nDim = atoi(text_line.c_str());
454  if (rank == MASTER_NODE) {
455  if (nDim == 2) cout << "Two dimensional problem." << endl;
456  if (nDim == 3) cout << "Three dimensional problem." << endl;
457  }
458  domain_flag = true;
459  } else { break; }
460  }
461 
462  /*--- Read the information about inner elements ---*/
463 
464  position = text_line.find ("NELEM=",0);
465  if (position != string::npos) {
466  text_line.erase (0,6);
467  stringstream stream_line(text_line);
468  stream_line >> Global_nElem;
469 
470  if ((rank == MASTER_NODE) && (size > SINGLE_NODE))
471  cout << Global_nElem << " interior elements before parallel partitioning." << endl;
472 
473  /*--- Check if the number of cores used is larger than the number of
474  elements. Terminate if this is the case, because it does not make
475  sense to do this. ---*/
476  unsigned long nCores = size; // Correct for the number of cores per rank.
477  if(nCores > Global_nElem) {
478  ostringstream message;
479  message << "The number of cores, " << nCores;
480  message << ", is larger than the number of elements, " << Global_nElem << "." << endl;
481  message << "This is not an efficient use of the resources and therefore "
482  << "SU2 will terminate.";
483 
484  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
485  }
486 
487  /*--- Compute the number of elements that will be on each processor.
488  This is a linear partitioning with the addition of a simple load
489  balancing for any remainder elements. ---*/
490  unsigned long total_elem_accounted = 0;
491  for(unsigned long i = 0; i < (unsigned long)size; i++) {
492  nPointLinear[i] = Global_nElem/size;
493  total_elem_accounted = total_elem_accounted + nPointLinear[i];
494  }
495 
496  /*--- Get the number of remainder elements after the even division ---*/
497  unsigned long rem_elem = Global_nElem - total_elem_accounted;
498  for (unsigned long i = 0; i<rem_elem; i++) {
499  ++nPointLinear[i];
500  }
501 
502  /*--- Store the local number of elements and the beginning/end index ---*/
503  nElem = nPointLinear[rank];
504  beg_node[0] = 0;
505  end_node[0] = beg_node[0] + nPointLinear[0];
506  for (unsigned long i = 1; i < (unsigned long)size; i++) {
507  beg_node[i] = end_node[i-1];
508  end_node[i] = beg_node[i] + nPointLinear[i] ;
509  }
510 
511  /*--- Allocate space for elements ---*/
512  elem = new CPrimalGrid*[nElem] ();
513 
514  /*--- Loop over all the elements and store the elements to be stored on
515  this rank. Furthermore, determine the total amount of DOFs for
516  the solution (which may differ from the number of DOFS for the
517  grid). ---*/
518  unsigned long nDOFs_tot = 0;
519  for (unsigned long i = 0; i < Global_nElem; i++) {
520 
521  getline(mesh_file, text_line);
522  istringstream elem_line(text_line);
523 
524  /*--- Read the value that defines the element type, the polynomial
525  degree of the geometry and the polynomial degree of the
526  solution. Extract this info as well. ---*/
527  unsigned long typeRead; elem_line >> typeRead;
528  unsigned long typeReadErrorMessage = typeRead;
529  unsigned short nPolySol, nPolyGrid;
530  if(typeRead > 10000) {
531  nPolySol = typeRead/10000 -1;
532  typeRead = typeRead%10000;
533  nPolyGrid = typeRead/100 + 1;
534  }
535  else {
536  nPolyGrid = typeRead/100 + 1;
537  nPolySol = nPolyGrid;
538  }
539 
540  unsigned short VTK_Type = typeRead%100;
541 
542  unsigned short nDOFsGrid = CFEMStandardElementBase::GetNDOFsStatic(VTK_Type, nPolyGrid,
543  typeReadErrorMessage);
544  unsigned short nDOFsSol = CFEMStandardElementBase::GetNDOFsStatic(VTK_Type, nPolySol,
545  typeReadErrorMessage);
546 
547  /*--- Allocate the memory for a new primary grid FEM element if
548  this element must be stored on this rank. ---*/
549  if ((i >= beg_node[rank]) && (i < end_node[rank])) {
550 
551  elem[loc_element_count] = new CPrimalGridFEM(i, VTK_Type, nPolyGrid, nPolySol,
552  nDOFsGrid, nDOFsSol, nDOFs_tot,
553  elem_line);
554  nDOFsGrid_Local += nDOFsGrid;
555  loc_element_count++;
556  }
557 
558  /*--- Update the total value of the number of DOFs. ---*/
559  nDOFs_tot += nDOFsSol;
560  }
561 
562  /*--- Break from the loop to find the element information. ---*/
563  break;
564  }
565  }
566 
567  mesh_file.close();
568 
569  /*--- Create a vector, which contains the global node IDs of the local elements. ---*/
570  vector<unsigned long> nodeIDsElemLoc;
571  nodeIDsElemLoc.reserve(nDOFsGrid_Local);
572  for(unsigned long i=0; i<loc_element_count; i++)
573  {
574  unsigned short nDOFsElem = elem[i]->GetnNodes();
575  for(unsigned short j=0; j<nDOFsElem; ++j)
576  nodeIDsElemLoc.push_back(elem[i]->GetNode(j));
577  }
578 
579  sort(nodeIDsElemLoc.begin(), nodeIDsElemLoc.end());
580  vector<unsigned long>::iterator lastNode;
581  lastNode = unique(nodeIDsElemLoc.begin(), nodeIDsElemLoc.end());
582  nodeIDsElemLoc.erase(lastNode, nodeIDsElemLoc.end());
583 
584  /*--- Allocate the memory for the coordinates to be stored on this rank. ---*/
585  nPoint = nodeIDsElemLoc.size();
586  nodes = new CPoint(nPoint, nDim);
587 
588  /*--- Open the grid file again and go to the position where
589  the correct zone is stored. ---*/
590  mesh_file.open(val_mesh_filename.c_str(), ios::in);
591 
592  if (val_nZone > 1 && !time_spectral) {
593  while (getline (mesh_file,text_line)) {
594  position = text_line.find ("IZONE=",0);
595  if (position != string::npos) {
596  text_line.erase (0,6);
597  unsigned short jDomain = atoi(text_line.c_str());
598  if (jDomain == val_iZone+1) break;
599  }
600  }
601  }
602 
603  /*--- While loop to read the point information. ---*/
604  while (getline (mesh_file, text_line)) {
605 
606  position = text_line.find ("NPOIN=",0);
607  if (position != string::npos) {
608  text_line.erase (0,6);
609  stringstream stream_line(text_line);
610  stream_line >> Global_nPoint;
611 
612  /*--- Loop over the global number of points and store the
613  ones that are needed on this processor. ---*/
614  unsigned long ii = 0;
615  for(unsigned long i=0; i<Global_nPoint; ++i) {
616  getline(mesh_file, text_line);
617 
618  if( binary_search(nodeIDsElemLoc.begin(), nodeIDsElemLoc.end(), i) ) {
619  istringstream point_line(text_line);
620  su2double Coord[3] = {0.0};
621  point_line >> Coord[0];
622  point_line >> Coord[1];
623  if (nDim==3) point_line >> Coord[2];
624  nodes->SetCoord(ii, Coord);
625  nodes->SetGlobalIndex(ii, i);
626  ++ii;
627  }
628  }
629 
630  break;
631  }
632  }
633 
634  mesh_file.close();
635 
636  /*--- Determine the faces of the local elements. --- */
637  vector<CFaceOfElement> localFaces;
638  for(unsigned long k=0; k<loc_element_count; k++) {
639 
640  /*--- Get the global IDs of the corner points of all the faces of this elements. ---*/
641  unsigned short nFaces;
642  unsigned short nPointsPerFace[6];
643  unsigned long faceConn[6][4];
644 
645  elem[k]->GetCornerPointsAllFaces(nFaces, nPointsPerFace, faceConn);
646 
647  /*--- Loop over the faces and add them to localFaces. ---*/
648  for(unsigned short i=0; i<nFaces; ++i) {
649  CFaceOfElement thisFace;
650  thisFace.nCornerPoints = nPointsPerFace[i];
651  for(unsigned short j=0; j<nPointsPerFace[i]; ++j)
652  thisFace.cornerPoints[j] = faceConn[i][j];
653  thisFace.elemID0 = k + beg_node[rank];
654 
655  thisFace.CreateUniqueNumbering();
656  localFaces.push_back(thisFace);
657  }
658  }
659 
660  /*--- Sort localFaces in increasing order and remove the double entities,
661  such that the binary search later on is a bit more efficient. ---*/
662  sort(localFaces.begin(), localFaces.end());
663  vector<CFaceOfElement>::iterator lastFace;
664  lastFace = unique(localFaces.begin(), localFaces.end());
665  localFaces.erase(lastFace, localFaces.end());
666 
667 
668  /*--- Open the grid file again and go to the position where
669  the correct zone is stored. ---*/
670  mesh_file.open(val_mesh_filename.c_str(), ios::in);
671 
672  if (val_nZone > 1 && !time_spectral) {
673  while (getline (mesh_file,text_line)) {
674  position = text_line.find ("IZONE=",0);
675  if (position != string::npos) {
676  text_line.erase (0,6);
677  unsigned short jDomain = atoi(text_line.c_str());
678  if (jDomain == val_iZone+1) break;
679  }
680  }
681  }
682 
683  /*--- While loop to read the boundary information. ---*/
684  while (getline (mesh_file, text_line)) {
685  /*--- Read number of markers ---*/
686  position = text_line.find ("NMARK=",0);
687  if (position != string::npos) {
688  text_line.erase (0,6);
689  istringstream stream_line(text_line);
690  stream_line >> nMarker;
691 
692  if (rank == MASTER_NODE) cout << nMarker << " surface markers." << endl;
693  config->SetnMarker_All(nMarker);
694  bound = new CPrimalGrid**[nMarker];
695  nElem_Bound = new unsigned long [nMarker];
696  Tag_to_Marker = new string [nMarker_Max];
697 
698  for(unsigned short iMarker = 0 ; iMarker < nMarker; iMarker++) {
699  getline (mesh_file, text_line);
700  text_line.erase (0,11);
701  text_line.erase (remove(text_line.begin(), text_line.end(), ' '), text_line.end());
702  text_line.erase (remove(text_line.begin(), text_line.end(), '\r'), text_line.end());
703  text_line.erase (remove(text_line.begin(), text_line.end(), '\n'), text_line.end());
704 
705  Marker_Tag = text_line.c_str();
706 
707  /*--- Read the number of elements for this marker. ---*/
708  getline (mesh_file, text_line);
709  text_line.erase (0,13); istringstream nmark_line(text_line);
710  unsigned long nElem_Bound_Global;
711  nmark_line >> nElem_Bound_Global;
712  if (rank == MASTER_NODE)
713  cout << nElem_Bound_Global << " boundary elements in index "<< iMarker
714  <<" (Marker = " <<Marker_Tag<< ")." << endl;
715 
716  /*--- Define a vector of FEM boundary elements to store the local boundary faces. ---*/
717  vector<CBoundaryFace> boundElems;
718 
719  /*--- Loop over the global boundary faces. ---*/
720  for(unsigned long i=0; i<nElem_Bound_Global; ++i) {
721  getline (mesh_file, text_line);
722  istringstream bound_line(text_line);
723 
724  /*--- Determine the element type, its number of DOFs and read
725  its node IDs. ---*/
726  unsigned long typeRead; bound_line >> typeRead;
727  unsigned short nPolyGrid = typeRead/100 + 1;
728  unsigned short VTK_Type = typeRead%100;
729  unsigned short nDOFEdgeGrid = nPolyGrid + 1;
730 
731  unsigned short nDOFsGrid = 0;
732  CFaceOfElement thisFace;
733  thisFace.cornerPoints[0] = 0; thisFace.cornerPoints[1] = nPolyGrid;
734  switch( VTK_Type ) {
735  case LINE:
736  nDOFsGrid = nDOFEdgeGrid;
737  thisFace.nCornerPoints = 2;
738  break;
739 
740  case TRIANGLE:
741  nDOFsGrid = nDOFEdgeGrid*(nDOFEdgeGrid+1)/2;
742  thisFace.nCornerPoints = 3;
743  thisFace.cornerPoints[2] = nDOFsGrid -1;
744  break;
745 
746  case QUADRILATERAL:
747  nDOFsGrid = nDOFEdgeGrid*nDOFEdgeGrid;
748  thisFace.nCornerPoints = 4;
749  thisFace.cornerPoints[2] = nPolyGrid*nDOFEdgeGrid;
750  thisFace.cornerPoints[3] = nDOFsGrid -1;
751  break;
752 
753  default:
754  ostringstream message;
755  message << "Unknown FEM boundary element value, " << typeRead
756  << ", in " << val_mesh_filename;
757  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
758  }
759 
760  vector<unsigned long> nodeIDs(nDOFsGrid);
761  for(unsigned short j=0; j<nDOFsGrid; ++j) bound_line >> nodeIDs[j];
762 
763  /*--- Convert the local numbering of thisFace to global numbering
764  and create a unique numbering of these nodes. ---*/
765  for(unsigned short j=0; j<thisFace.nCornerPoints; ++j)
766  thisFace.cornerPoints[j] = nodeIDs[thisFace.cornerPoints[j]];
767  thisFace.CreateUniqueNumbering();
768 
769  /*--- Check if this boundary face must be stored on this rank.
770  If so, create an object of CBoundaryFace and add it
771  to boundElems. ---*/
772  vector<CFaceOfElement>::iterator low;
773  low = lower_bound(localFaces.begin(), localFaces.end(), thisFace);
774  if(low != localFaces.end()) {
775  if( !(thisFace < *low) ) {
776 
777  CBoundaryFace thisBoundFace;
778  thisBoundFace.VTK_Type = VTK_Type;
779  thisBoundFace.nPolyGrid = nPolyGrid;
780  thisBoundFace.nDOFsGrid = nDOFsGrid;
781  thisBoundFace.globalBoundElemID = i;
782  thisBoundFace.domainElementID = low->elemID0;
783  thisBoundFace.Nodes = nodeIDs;
784 
785  boundElems.push_back(thisBoundFace);
786  }
787  }
788  }
789 
790  /*--- Allocate space for the boundary elements and store the ones
791  whose parent element is stored on this rank. ---*/
792  nElem_Bound[iMarker] = boundElems.size();
793  bound[iMarker] = new CPrimalGrid* [nElem_Bound[iMarker]];
794 
795  for(unsigned long i=0; i<nElem_Bound[iMarker]; ++i)
796  bound[iMarker][i] = new CPrimalGridBoundFEM(boundElems[i].globalBoundElemID,
797  boundElems[i].domainElementID,
798  boundElems[i].VTK_Type,
799  boundElems[i].nPolyGrid,
800  boundElems[i].nDOFsGrid,
801  boundElems[i].Nodes);
802 
803  /*--- Update config information storing the boundary information in the right place ---*/
804  Tag_to_Marker[config->GetMarker_CfgFile_TagBound(Marker_Tag)] = Marker_Tag;
805  config->SetMarker_All_TagBound(iMarker, Marker_Tag);
806  config->SetMarker_All_KindBC(iMarker, config->GetMarker_CfgFile_KindBC(Marker_Tag));
807  config->SetMarker_All_Monitoring(iMarker, config->GetMarker_CfgFile_Monitoring(Marker_Tag));
808  config->SetMarker_All_GeoEval(iMarker, config->GetMarker_CfgFile_GeoEval(Marker_Tag));
809  config->SetMarker_All_Designing(iMarker, config->GetMarker_CfgFile_Designing(Marker_Tag));
810  config->SetMarker_All_Plotting(iMarker, config->GetMarker_CfgFile_Plotting(Marker_Tag));
811  config->SetMarker_All_Analyze(iMarker, config->GetMarker_CfgFile_Analyze(Marker_Tag));
812  config->SetMarker_All_ZoneInterface(iMarker, config->GetMarker_CfgFile_ZoneInterface(Marker_Tag));
813  config->SetMarker_All_DV(iMarker, config->GetMarker_CfgFile_DV(Marker_Tag));
814  config->SetMarker_All_Moving(iMarker, config->GetMarker_CfgFile_Moving(Marker_Tag));
815  config->SetMarker_All_SobolevBC(iMarker, config->GetMarker_CfgFile_SobolevBC(Marker_Tag));
816  config->SetMarker_All_PerBound(iMarker, config->GetMarker_CfgFile_PerBound(Marker_Tag));
817  config->SetMarker_All_SendRecv(iMarker, NONE);
818  }
819 
820  break;
821  }
822  }
823 
824  mesh_file.close();
825 }

827 void CPhysicalGeometry::Read_CGNS_Format_Parallel_FEM(CConfig *config,
828  string val_mesh_filename,
829  unsigned short val_iZone,
830  unsigned short val_nZone) {
831 #ifdef HAVE_CGNS
832 
833  /*--- For proper support of the high order elements, at least version 3.3
834  of CGNS must be used. Check this. ---*/
835 #if CGNS_VERSION >= 3300
836 
837  /*--- Check whether the supplied file is truly a CGNS file. ---*/
838  int file_type;
839  if (cg_is_cgns(val_mesh_filename.c_str(), &file_type) != CG_OK)
840  SU2_MPI::Error(val_mesh_filename + string(" is not a CGNS file that can be read."),
841  CURRENT_FUNCTION);
842 
843  /*--- Initialize counters for local/global points & elements ---*/
844  Global_nPoint = 0; Global_nPointDomain = 0; Global_nElem = 0;
845  nelem_edge = 0; Global_nelem_edge = 0;
846  nelem_triangle = 0; Global_nelem_triangle = 0;
847  nelem_quad = 0; Global_nelem_quad = 0;
848  nelem_tetra = 0; Global_nelem_tetra = 0;
849  nelem_hexa = 0; Global_nelem_hexa = 0;
850  nelem_prism = 0; Global_nelem_prism = 0;
851  nelem_pyramid = 0; Global_nelem_pyramid = 0;
852 
853  /*--------------------------------------------------------------------------*/
854  /*--- Checking of the file, determine the dimensions, etc. ---*/
855  /*--------------------------------------------------------------------------*/
856 
857  /*--- Allocate memory for the linear partition of the elements of the mesh.
858  These arrays are the size of the number of ranks. ---*/
859  beg_node = new unsigned long[size];
860  end_node = new unsigned long[size];
861  nPointLinear = new unsigned long[size];
862 
863  /* Open the CGNS file for reading and check if it went OK. */
864  int fn;
865  if(cg_open(val_mesh_filename.c_str(), CG_MODE_READ, &fn) != CG_OK) cg_error_exit();
866  if(rank == MASTER_NODE) {
867  cout << "Reading the CGNS file: " << val_mesh_filename << "." << endl;
868  }
869 
870  /* Get the number of databases. This is the highest node in the CGNS heirarchy.
871  The current implementation assumes that there is only one database. */
872  int nbases;
873  if(cg_nbases(fn, &nbases) != CG_OK) cg_error_exit();
874  if(nbases > 1) {
875  ostringstream message;
876  message << "CGNS file contains " << nbases << " databases." << endl;
877  message << "CGNS reader can handle only 1 at the moment." << endl;
878 
879  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
880  }
881 
882  /* Read the information of the base, especially the number of dimensions. */
883  char cgnsname[CGNS_STRING_SIZE];
884  int cellDim, physDim;
885  const int iBase = 1;
886  if(cg_base_read(fn, iBase, cgnsname, &cellDim, &physDim)) cg_error_exit();
887  nDim = physDim;
888 
889  if(cellDim != physDim) {
890  ostringstream message;
891  message << "The element dimension, " << cellDim
892  << ", differs from the physical dimension, " << physDim << "." << endl;
893  message << "These should be the same for the DG-FEM solver." << endl;
894 
895  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
896  }
897 
898  /* Write the info about the number of dimensions. */
899  if(rank == MASTER_NODE) {
900  if (nDim == 2) cout << "Two dimensional problem." << endl;
901  if (nDim == 3) cout << "Three dimensional problem." << endl;
902  }
903 
904  /* Set the zone ID for CGNS, CGNS indices starts at 1, and check
905  if the requested zone ID is a valid one. */
906  const int iZone = val_iZone + 1;
907 
908  int nzones;
909  if(cg_nzones(fn, iBase, &nzones) != CG_OK) cg_error_exit();
910  if(iZone < 1 || iZone > nzones) {
911  ostringstream message;
912  message << "Zone " << iZone << " requested for reading, but there are only "
913  << nzones << " zones present in the CGNS file." << endl;
914 
915  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
916  }
917 
918  /* Determine the zone type for the requested zone and check if it is
919  unstructured. */
920  ZoneType_t zoneType;
921  if(cg_zone_type(fn, iBase, iZone, &zoneType) != CG_OK) cg_error_exit();
922  if(zoneType != Unstructured)
923  SU2_MPI::Error("Structured CGNS zone found while unstructured expected.",
924  CURRENT_FUNCTION);
925 
926  /* Determine the number of sections for the connectivities in this zone. */
927  int nsections;
928  if(cg_nsections(fn, iBase, iZone, &nsections) != CG_OK) cg_error_exit();
929 
930  /*--------------------------------------------------------------------------*/
931  /*--- Reading and distributing the volume elements. ---*/
932  /*--------------------------------------------------------------------------*/
933 
934  /* Loop over the sections to store the meta data in CGNSElemTypes
935  and to determine the number of volume elements.
936  Note that the indices start at 1 in CGNS. */
937  vector<CCGNSElementType> CGNSElemTypes(nsections);
938 
939  for(int iConn=1; iConn<=nsections; ++iConn) {
940  CGNSElemTypes[iConn-1].DetermineMetaData(nDim, fn, iBase, iZone, iConn);
941  if( CGNSElemTypes[iConn-1].volumeConn )
942  Global_nElem += CGNSElemTypes[iConn-1].nElem;
943  }
944 
945  if((rank == MASTER_NODE) && (size > SINGLE_NODE))
946  cout << Global_nElem << " interior elements before parallel partitioning." << endl;
947 
948  /*--- Check if the number of cores used is larger than the number of
949  elements. Terminate if this is the case, because it does not make
950  sense to do this. ---*/
951  unsigned long nCores = size; // Correct for the number of cores per rank.
952  if(nCores > Global_nElem) {
953  ostringstream message;
954  message << "The number of cores, " << nCores;
955  message << ", is larger than the number of elements, " << Global_nElem << "." << endl;
956  message << "This is not an efficient use of the resources and therefore "
957  << "SU2 will terminate.";
958 
959  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
960  }
961 
962  /*--- Compute the number of elements that will be stored on each rank.
963  This is a linear partitioning with the addition of a simple load
964  balancing for any remainder elements. ---*/
965  unsigned long total_elem_accounted = 0;
966  for(unsigned long i = 0; i < (unsigned long)size; i++) {
967  nPointLinear[i] = Global_nElem/size;
968  total_elem_accounted = total_elem_accounted + nPointLinear[i];
969  }
970 
971  /*--- Get the number of remainder elements after the even division ---*/
972  const unsigned long rem_elem = Global_nElem - total_elem_accounted;
973  for (unsigned long i = 0; i<rem_elem; i++) {
974  ++nPointLinear[i];
975  }
976 
977  /*--- Store the local number of elements and the beginning/end index ---*/
978  nElem = nPointLinear[rank];
979  beg_node[0] = 0;
980  end_node[0] = beg_node[0] + nPointLinear[0];
981  for(int i=1; i<size; i++) {
982  beg_node[i] = end_node[i-1];
983  end_node[i] = beg_node[i] + nPointLinear[i] ;
984  }
985 
986  /*--- Allocate space for elements ---*/
987  elem = new CPrimalGrid*[nElem] ();
988 
989  /*--- Loop over over the connectivities and read the elements to be stored on
990  this rank. Furthermore, determine the local amount of DOFs for the
991  solution (which may differ from the number of DOFS for the grid). ---*/
992  unsigned long nDOFsLoc = 0, elemCount = 0, locElemCount = 0;
993 
994  for(int iConn=0; iConn<nsections; ++iConn) {
995  if( CGNSElemTypes[iConn].volumeConn ) {
996 
997  /* Determine the global volume element range for this connectivity. */
998  const unsigned long elemCountOld = elemCount;
999  elemCount += CGNSElemTypes[iConn].nElem;
1000 
1001  /* Check for overlap with the element range this rank is responsible for. */
1002  const unsigned long indBegOverlap = max(elemCountOld, beg_node[rank]);
1003  const unsigned long indEndOverlap = min(elemCount, end_node[rank]);
1004 
1005  if(indEndOverlap > indBegOverlap) {
1006 
1007  /* This rank must read element data from this connectivity section.
1008  Determine the offset relative to the start of this section and the
1009  number of elements to be read by this rank. */
1010  const unsigned long offsetRank = indBegOverlap - elemCountOld;
1011  const unsigned long nElemRank = indEndOverlap - indBegOverlap;
1012 
1013  /* Read the connectivity range determined above. */
1014  CGNSElemTypes[iConn].ReadConnectivityRange(fn, iBase, iZone, offsetRank,
1015  nElemRank, beg_node[rank],
1016  elem, locElemCount, nDOFsLoc);
1017  }
1018  }
1019  }
1020 
1021 #ifdef HAVE_MPI
1022  /* The global offset of the DOFs must be corrected when running in
1023  parallel. Therefore gather the number of DOFs of all the ranks. */
1024  vector<unsigned long> nDOFsPerRank(size);
1025  SU2_MPI::Allgather(&nDOFsLoc, 1, MPI_UNSIGNED_LONG, nDOFsPerRank.data(), 1,
1026  MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
1027 
1028  /* Determine the offset for the DOFs on this rank. */
1029  unsigned long offsetRank = 0;
1030  for(int i=0; i<rank; ++i)
1031  offsetRank += nDOFsPerRank[i];
1032 
1033  /* Loop over the local elements to correct the global offset of the DOFs. */
1034  for(unsigned long i=0; i<nElem; ++i)
1035  elem[i]->AddOffsetGlobalDOFs(offsetRank);
1036 #endif
1037 
1038  /*--------------------------------------------------------------------------*/
1039  /*--- Reading and distributing the coordinates. ---*/
1040  /*--------------------------------------------------------------------------*/
1041 
1042  /* Determine the global number of vertices in the requested zone.
1043  The other size information is not used. */
1044  cgsize_t sizes[3];
1045  if(cg_zone_read(fn, iBase, iZone, cgnsname, sizes) != CG_OK) cg_error_exit();
1046  Global_nPoint = sizes[0];
1047 
1048  /*--- Determine the number of points per rank in cumulative storage format.
1049  This is done to avoid that every rank reads all the coordinates.
1050  The required coordinates for each rank are later obtained via
1051  communication. ---*/
1052  unsigned long totalPointsAccounted = 0;
1053  vector<cgsize_t> nPointsPerRank(size+1);
1054  for(int i=1; i<=size; ++i) {
1055  nPointsPerRank[i] = Global_nPoint/size;
1056  totalPointsAccounted += nPointsPerRank[i];
1057  }
1058 
1059  const unsigned long nPointsRem = Global_nPoint - totalPointsAccounted;
1060  for(unsigned long i=1; i<=nPointsRem; ++i) ++nPointsPerRank[i];
1061 
1062  nPointsPerRank[0] = 0;
1063  for(int i=0; i<size; ++i)
1064  nPointsPerRank[i+1] += nPointsPerRank[i];
1065 
1066  /* Determine the number of points that must be read by this rank and
1067  allocate the memory for the coordinate buffers. */
1068  const cgsize_t nPointsRead = nPointsPerRank[rank+1] - nPointsPerRank[rank];
1069  vector<vector<su2double> > coorBuf(nDim, vector<su2double>(nPointsRead));
1070 
1071  /* Loop over the number of dimensions to read the coordinates. Note that
1072  the loop starts at 1 and ends at nDim because CGNS requires this. */
1073  for(unsigned short iDim=1; iDim<=nDim; ++iDim) {
1074 
1075  /* Determine the data type and name of the coordinate. Copy the name
1076  of the coordinate in a string for easier comparison. */
1077  DataType_t datatype;
1078  if(cg_coord_info(fn, iBase, iZone, iDim, &datatype, cgnsname) != CG_OK)
1079  cg_error_exit();
1080  string coorname = cgnsname;
1081 
1082  /* Check the name of the coordinate and determine the index in coorBuf
1083  where to store this coordinate. Normally this should be iDim-1. */
1084  unsigned short indC = 0;
1085  if( coorname == "CoordinateX") indC = 0;
1086  else if(coorname == "CoordinateY") indC = 1;
1087  else if(coorname == "CoordinateZ") indC = 2;
1088  else
1089  SU2_MPI::Error(string("Unknown coordinate name, ") + coorname +
1090  string(", encountered in the CGNS file."), CURRENT_FUNCTION);
1091 
1092  /* Easier storage of the range in the CGNS file. */
1093  cgsize_t range_min = nPointsPerRank[rank] + 1;
1094  cgsize_t range_max = nPointsPerRank[rank+1];
1095 
1096  /*--- Read the coordinate with the required precision and copy
1097  this data to the correct index in coorBuf. ---*/
1098  switch( datatype ) {
1099  case RealSingle: {
1100  /* Single precision used. */
1101  vector<float> buf(nPointsRead);
1102  if(cg_coord_read(fn, iBase, iZone, cgnsname, datatype, &range_min,
1103  &range_max, buf.data()) != CG_OK) cg_error_exit();
1104 
1105  for(cgsize_t i=0; i<nPointsRead; ++i)
1106  coorBuf[indC][i] = buf[i];
1107  break;
1108  }
1109 
1110  case RealDouble: {
1111  /* Double precision used. */
1112  vector<double> buf(nPointsRead);
1113  if(cg_coord_read(fn, iBase, iZone, cgnsname, datatype, &range_min,
1114  &range_max, buf.data()) != CG_OK) cg_error_exit();
1115 
1116  for(cgsize_t i=0; i<nPointsRead; ++i)
1117  coorBuf[indC][i] = buf[i];
1118  break;
1119  }
1120 
1121  default: {
1122  ostringstream message;
1123  message << "Datatype for coordinates must be RealSingle or RealDouble, "
1124  << "not " << datatype << endl;
1125  SU2_MPI::Error(message.str(), CURRENT_FUNCTION);
1126  }
1127  }
1128  }
1129 
1130  /* Make a distinction between sequential and parallel mode. */
1131 #ifdef HAVE_MPI
1132  /*--- Parallel mode. Create a vector, which contains the global
1133  node IDs of the local elements. ---*/
1134  vector<unsigned long> nodeIDsElemLoc;
1135  nodeIDsElemLoc.reserve(nDOFsLoc);
1136  for(unsigned long i=0; i<locElemCount; ++i) {
1137  unsigned short nDOFsElem = elem[i]->GetnNodes();
1138  for(unsigned short j=0; j<nDOFsElem; ++j)
1139  nodeIDsElemLoc.push_back(elem[i]->GetNode(j));
1140  }
1141 
1142  sort(nodeIDsElemLoc.begin(), nodeIDsElemLoc.end());
1143  vector<unsigned long>::iterator lastNode;
1144  lastNode = unique(nodeIDsElemLoc.begin(), nodeIDsElemLoc.end());
1145  nodeIDsElemLoc.erase(lastNode, nodeIDsElemLoc.end());
1146 
1147  /*--- Allocate the memory for the coordinates to be stored on this rank. ---*/
1148  nPoint = nodeIDsElemLoc.size();
1149  nodes = new CPoint(nPoint, nDim);
1150 
1151  /*--- Store the global ID's of the nodes in such a way that they can

4819  /* Loop over the markers and select the ones for which a wall function
4820  treatment must be carried out. */
4821  for(unsigned short iMarker=0; iMarker<nMarker; ++iMarker) {
4822

4824  case ISOTHERMAL:
4825  case HEAT_FLUX: {

4827  if(config->GetWallFunction_Treatment(Marker_Tag) != WALL_FUNCTIONS::NONE) {
4828

4830  The number of points in normal direction for the wall function
4831  treatment is the first element of this array. */
4832  const unsigned short *shortInfo = config->GetWallFunction_IntInfo(Marker_Tag);
4833 
4834  /* Loop over the local boundary elements for this marker. */
4835  for(unsigned long l=0; l<nElem_Bound[iMarker]; ++l) {
4836 
4837  /* Get the required data to define the standard boundary element. */
4838 
4839  /* Easier storage of the element type, the corresponding volume
4840  element, the polynomial degree for the solution and whether or
4841  not the Jacobian can be considered constant. */
4842  const unsigned short VTK_Type_Face = bound[iMarker][l]->GetVTK_Type();
4843  const unsigned long elemID = bound[iMarker][l]->GetDomainElement()
4844  - beg_node[rank];
4845  const unsigned short nPolySol = elem[elemID]->GetNPolySol();
4846  const unsigned short VTK_Type_Elem = elem[elemID]->GetVTK_Type();

4850  boundary faces. This entry must be found. */
4851  unsigned long ii;
4852  for(ii=0; ii<standardBoundaryFaces.size(); ++ii) {
4853  if(standardBoundaryFaces[ii].SameStandardBoundaryFace(VTK_Type_Face, JacIsConstant,
4854  VTK_Type_Elem, nPolySol, false) )
4855  break;
4856  }
4857 
4858  if(ii == standardBoundaryFaces.size())
4859  SU2_MPI::Error("No matching standard boundary element found", CURRENT_FUNCTION);
4860 
4861  /* Update the computational work for the corresponding volume element,
4862  i.e. the 1st vertex weight. */
4863  vwgt[2*elemID] += standardBoundaryFaces[ii].WorkEstimateMetisWallFunctions(config, shortInfo[0]);
4864  }
4865  }
4866 
4867  break;
4868  }
4869 
4870  default: /* Just to avoid a compiler warning. */
4871  break;
4872  }
4873  }
4874 
4875  /*--------------------------------------------------------------------------*/
4876  /*--- The final weight is obtained by taking the amount of work in time ---*/
4877  /*--- into account. Note that this correction is only relevant when time ---*/
4878  /*--- accurate local time stepping is employed. ---*/
4879  /*--------------------------------------------------------------------------*/
4880

4882  const unsigned short diffLevel = maxTimeLevel - elem[i]->GetTimeLevel();
4883  vwgt[2*i] *= pow(2, diffLevel);
4884  }
4885 
4886  /*--- Determine the minimum of the workload of the elements, i.e. 1st vertex
4887  weight, over the entire domain. ---*/
4888  su2double minvwgt = vwgt[0];
4889  for(unsigned long i=0; i<nElem; ++i) minvwgt = min(minvwgt, vwgt[2*i]);
4890

4893  SU2_MPI::Allreduce(&locminvwgt, &minvwgt, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
4894 #endif
4895 
4896  /*--- Scale the workload of the elements, the 1st vertex weight, with the
4897  minimum value and multiply by 100, such that the conversion to integer
4898  weights (ParMETIS is using integers for the weights) does not lead to
4899  a significant increase in the load imbalance. ---*/
4900  minvwgt = 100.0/minvwgt;
4901  for(unsigned long i=0; i<nElem; ++i) vwgt[2*i] *= minvwgt;
4902 
4903  /*--------------------------------------------------------------------------*/
4904  /* Step 2: Determine the adjacency weights, which are proportional to the */
4905  /* amount of communication needed when the two neighboring */
4906  /* elements are stored on different ranks. */
4907  /*--------------------------------------------------------------------------*/
4908 
4909  /*--- Determine the edge weights by a double loop over the local elements
4910  and its edges. The map mapElemIDsToFaceInd is used to determine
4911  the corresponding index in localFaces. ---*/
4912  for(unsigned long i=0; i<nElem; ++i) {
4913

4915  number of solution DOFs. */
4916  const unsigned short timeLevel0 = elem[i]->GetTimeLevel();

4920  for(unsigned long j=0; j<adjacency[i].size(); ++j) {

4923  DOFs for the neighboring element. */
4924  unsigned short timeLevel1, nDOFs1;
4925

4927  if(adjacency[i][j] >= beg_node[rank] &&
4928  adjacency[i][j] < beg_node[rank]+nElem) {

4931  time level and number of solution DOFs. */

4933  timeLevel1 = elem[elemID1]->GetTimeLevel();
4934  nDOFs1 = elem[elemID1]->GetNDOFsSol();
4935  }
4936  else {
4937 
4938  /* The neighbor is an external element. Find it in mapExternalElemIDToTimeLevel
4939  and set the time level and number of solution DOFs accordingly. */
4940  map<unsigned long,CUnsignedShort2T>::const_iterator MI;
4941  MI = mapExternalElemIDToTimeLevel.find(adjacency[i][j]);

4943  SU2_MPI::Error("Entry not found in mapExternalElemIDToTimeLevel", CURRENT_FUNCTION);
4944  timeLevel1 = MI->second.short0;
4945  nDOFs1 = MI->second.short1;
4946  }

4951  is employed. */

4955  to the sum of the number of DOFs on both sides, multiplied by the weight factor
4956  to account for different time levels. */
4957  adjwgt[i][j] = pow(2, diffLevel) *(nDOFs0 + nDOFs1);
4958  }
4959  }
