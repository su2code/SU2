244  /*!
245  * \brief Get the projected velocity in a unitary vector direction (compressible solver).
246  * \param[in] val_vector - Direction of projection.
247  * \return Value of the projected velocity.
248  */
249  inline su2double GetProjVel(unsigned long iPoint, const su2double *val_vector) const final {
250  su2double ProjVel = 0.0;
251  for (unsigned long iDim = 0; iDim < nDim; iDim++)
252  ProjVel += Primitive(iPoint,iDim+indices.Velocity())*val_vector[iDim];
253  return ProjVel;

256  /*!
257  * \brief Set the velocity vector from the solution.
258  * \param[in] val_velocity - Pointer to the velocity.
259  */
260  inline void SetVelocity(unsigned long iPoint) final {
261  Velocity2(iPoint) = 0.0;
262  for (unsigned long iDim = 0; iDim < nDim; iDim++) {
263  Primitive(iPoint,iDim+indices.Velocity()) = Solution(iPoint,iDim+1) / Solution(iPoint,0);
264  Velocity2(iPoint) += pow(Primitive(iPoint,iDim+indices.Velocity()),2);
265  }
266  }
