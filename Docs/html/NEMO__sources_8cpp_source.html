73 CNumerics::ResidualType<> CSource_NEMO::ComputeChemistry(const CConfig *config) {
74 
75  /*--- Nonequilibrium chemistry ---*/
76  unsigned short iSpecies, iVar;
77  unsigned short jVar;
78 
79  vector<su2double> rhos;
80  rhos.resize(nSpecies,0.0);
81 
82  /*--- Initialize residual and Jacobian arrays ---*/
83  for (iVar = 0; iVar < nVar; iVar++) {
84  residual[iVar] = 0.0;
85  }
86  if (implicit)
87  for (iVar = 0; iVar < nVar; iVar++)
88  for (jVar = 0; jVar < nVar; jVar++)
89  jacobian[iVar][jVar] = 0.0;
90 
91  /*--- Rename for convenience ---*/
92  su2double T = V_i[T_INDEX];
93  su2double Tve = V_i[TVE_INDEX];
94  for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
95  rhos[iSpecies]=V_i[RHOS_INDEX+iSpecies];
96 
97  /*--- Set mixture state ---*/
98  fluidmodel->SetTDStateRhosTTv(rhos, T, Tve);
99 
100  /*---Compute Prodcution/destruction terms ---*/
101  const auto& ws = fluidmodel->ComputeNetProductionRates(implicit, V_i, eve_i, Cvve_i,
102  dTdU_i, dTvedU_i, jacobian);
103 
104  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++){
105  residual[iSpecies] = ws[iSpecies] * Volume;}
106 
107  if (implicit) {
108  for (iVar = 0; iVar<nVar; iVar++) {
109  for (jVar = 0; jVar<nVar; jVar++) {
110  jacobian[iVar][jVar] *= Volume;
111  }
112  }
113  }
114 
115  return ResidualType<>(residual, jacobian, nullptr);

119 CNumerics::ResidualType<> CSource_NEMO::ComputeVibRelaxation(const CConfig *config) {
120 
121  /*--- Trans.-rot. & vibrational energy exchange via inelastic collisions ---*/
122  // Note: Electronic energy not implemented
123  // Note: Landau-Teller formulation
124  // Note: Millikan & White relaxation time (requires P in Atm.)
125  // Note: Park limiting cross section
126  unsigned short iSpecies, iVar;
127  unsigned short jVar;
128  su2double res_min = -1E6;
129  su2double res_max = 1E6;
130 
131  vector<su2double> rhos;
132  rhos.resize(nSpecies,0.0);
133 
134  /*--- Initialize residual and Jacobian arrays ---*/
135  for (iVar = 0; iVar < nVar; iVar++) {
136  residual[iVar] = 0.0;
137  }
138  if (implicit) {
139  for (iVar = 0; iVar < nVar; iVar++)
140  for (jVar = 0; jVar < nVar; jVar++)
141  jacobian[iVar][jVar] = 0.0;
142  }
143 
144  /*--- Rename for convenience ---*/
145  su2double T = V_i[T_INDEX];
146  su2double Tve = V_i[TVE_INDEX];
147  for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
148  rhos[iSpecies]=V_i[RHOS_INDEX+iSpecies];
149 
150  /*--- Set fluid state ---*/
151  fluidmodel->SetTDStateRhosTTv(rhos, T, Tve);
152 
153  /*--- Compute residual and jacobians ---*/
154  su2double VTterm = fluidmodel -> ComputeEveSourceTerm();
155  if (implicit) {
156  fluidmodel->GetEveSourceTermJacobian(V_i, eve_i, Cvve_i, dTdU_i,
157  dTvedU_i, jacobian);
158  }
159 
160  residual[nSpecies+nDim+1] = VTterm * Volume;
161 
162  if (implicit) {
163  for (iVar = 0; iVar<nVar; iVar++) {
164  for (jVar = 0; jVar<nVar; jVar++) {
165  jacobian[iVar][jVar] *= Volume;
166  }
167  }
168  }
169 
170  /*--- Relax/limit vt transfer ---*/
171  if(config->GetVTTransferResidualLimiting()){
172  if(residual[nSpecies+nDim+1]>res_max) residual[nSpecies+nDim+1]=res_max;
173  if(residual[nSpecies+nDim+1]<res_min) residual[nSpecies+nDim+1]=res_min;
174  }
175 
176  return ResidualType<>(residual, jacobian, nullptr);
177 }

179 CNumerics::ResidualType<> CSource_NEMO::ComputeAxisymmetric(const CConfig *config) {
180 
181  unsigned short iDim, iSpecies, iVar;
182  unsigned short jSpecies, jVar;
183 
184  /*--- Rename for convenience ---*/
185  auto Ds = Diffusion_Coeff_i;
186  auto GV = PrimVar_Grad_i;
187  su2double ktr = Thermal_Conductivity_i;
188  su2double kve = Thermal_Conductivity_ve_i;
189  su2double rho = V_i[RHO_INDEX];
190  su2double RuSI = UNIVERSAL_GAS_CONSTANT;
191  su2double Ru = 1000.0*RuSI;
192  su2double rhou = U_i[nSpecies];
193  su2double rhov = U_i[nSpecies+1];
194  su2double H = V_i[H_INDEX];
195  su2double rhoEve = U_i[nVar-1];
196  const auto& Ms = fluidmodel->GetSpeciesMolarMass();
197  const auto& hs = fluidmodel->ComputeSpeciesEnthalpy(V_i[T_INDEX], V_i[TVE_INDEX], eve_i );
198 
199  bool viscous = config->GetViscous();
200  bool rans = (config->GetKind_Turb_Model() != TURB_MODEL::NONE);
201 
202  /*--- Initialize residual and Jacobian arrays ---*/
203  for (iVar = 0; iVar < nVar; iVar++) {
204  residual[iVar] = 0.0;
205  }
206 
207  /*--- Calculate inverse of y coordinate ---*/
208  su2double yinv = 0.0;
209  if (Coord_i[1]!= 0.0) yinv = 1.0/Coord_i[1];
210  else yinv = 0.0;
211 
212  /*--- Rename for convenience ---*/
213  su2double vel2 = 0.0;
214  for (iDim = 0; iDim < nDim; iDim++)
215  vel2 += V_i[VEL_INDEX+iDim]*V_i[VEL_INDEX+iDim];
216  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
217  Y[iSpecies] = V_i[RHOS_INDEX+iSpecies] / rho;
218 
219  /*--- Compute residual for inviscid axisym flow---*/
220  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
221  residual[iSpecies] = yinv*rhov*Y[iSpecies]*Volume;
222  residual[nSpecies] = yinv*rhov*U_i[nSpecies]/rho*Volume;
223  residual[nSpecies+1] = yinv*rhov*U_i[nSpecies+1]/rho*Volume;
224  residual[nSpecies+2] = yinv*rhov*H*Volume;
225  residual[nSpecies+3] = yinv*rhov*U_i[nSpecies+nDim+1]/rho*Volume;
226 
227  /*---Compute Jacobian for inviscid axisym flow ---*/
228  if (implicit) {
229 
230  /*--- Initialize ---*/
231  for (iVar = 0; iVar < nVar; iVar++)
232  for (jVar = 0; jVar < nVar; jVar++)
233  jacobian[iVar][jVar] = 0.0;
234  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
235  for (jSpecies = 0; jSpecies < nSpecies; jSpecies++)
236  dYdr[iSpecies][jSpecies] = 0.0;
237 
238  /*--- Calculate additional quantities ---*/
239  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
240  for (jSpecies = 0; jSpecies < nSpecies; jSpecies++) {
241  dYdr[iSpecies][jSpecies] += -1/rho*Y[iSpecies];
242  }
243  dYdr[iSpecies][iSpecies] += 1/rho;
244  }
245 
246  /*--- Populate Jacobian ---*/
247 
248  // Species density
249  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
250  for (jSpecies = 0; jSpecies < nSpecies; jSpecies++) {
251  jacobian[iSpecies][jSpecies] = dYdr[iSpecies][jSpecies]*rhov;
252  }
253  jacobian[iSpecies][nSpecies+1] = Y[iSpecies];
254  }
255 
256  // X-momentum
257  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
258  jacobian[nSpecies][iSpecies] = -rhou*rhov/(rho*rho);
259  jacobian[nSpecies][nSpecies] = rhov/rho;
260  jacobian[nSpecies][nSpecies+1] = rhou/rho;
261 
262  // Y-momentum
263  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
264  jacobian[nSpecies+1][iSpecies] = -rhov*rhov/(rho*rho);
265  jacobian[nSpecies+1][nSpecies+1] = 2*rhov/rho;
266 
267  // Energy
268  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
269  jacobian[nSpecies+nDim][iSpecies] = -H*rhov/rho + dPdU_i[iSpecies]*rhov/rho;
270  jacobian[nSpecies+nDim][nSpecies] = dPdU_i[nSpecies]*rhov/rho;
271  jacobian[nSpecies+nDim][nSpecies+1] = H + dPdU_i[nSpecies+1]*rhov/rho;
272  jacobian[nSpecies+nDim][nSpecies+nDim] = (1+dPdU_i[nSpecies+nDim])*rhov/rho;
273  jacobian[nSpecies+nDim][nSpecies+nDim+1] = dPdU_i[nSpecies+nDim+1]*rhov/rho;
274 
275  // Vib-el energy
276  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
277  jacobian[nSpecies+nDim+1][iSpecies] = -rhoEve*rhov/(rho*rho);
278  jacobian[nSpecies+nDim+1][nSpecies+1] = rhoEve/rho;
279  jacobian[nSpecies+nDim+1][nSpecies+nDim+1] = rhov/rho;
280 
281  for (iVar = 0; iVar < nVar; iVar++)
282  for (jVar = 0; jVar < nVar; jVar++)
283  jacobian[iVar][jVar] *= yinv*Volume;
284  }
285 
286  /*--- Compute residual for viscous portion of axisym flow ---*/
287  if (viscous) {
288  if (!rans){ turb_ke_i = 0.0; }
289 
290  su2double Vector = 0.0;
291  for (iSpecies = 0; iSpecies < nHeavy; iSpecies++)
292  Vector += rho*Ds[iSpecies]*GV[RHOS_INDEX+iSpecies][1];
293 
294  su2double sumJhs_y = 0.0;
295  su2double sumJeve_y = 0.0;
296  su2double Mass = 0.0;
297 
298  for (iSpecies=0; iSpecies<nSpecies; iSpecies++)
299  Mass += V_i[iSpecies]/rho*Ms[iSpecies];
300 
301  su2double heat_capacity_cp_i = V_i[RHOCVTR_INDEX]/rho + Ru/Mass;
302  su2double total_viscosity_i = Laminar_Viscosity_i + Eddy_Viscosity_i;
303  su2double total_conductivity_i = ktr + kve + heat_capacity_cp_i*Eddy_Viscosity_i/Prandtl_Turb;
304  su2double u = V_i[VEL_INDEX];
305  su2double v = V_i[VEL_INDEX+1];
306  su2double qy_t = -total_conductivity_i*GV[T_INDEX][1];
307  su2double qy_ve = -kve*GV[TVE_INDEX][1];
308 
309  /*--- Enthalpy and vib-el energy transport due to y-direction diffusion---*/
310  for (iSpecies = 0; iSpecies < nHeavy; iSpecies++) {
311  sumJhs_y += -(rho*Ds[iSpecies]*GV[RHOS_INDEX+iSpecies][1] - V_i[RHOS_INDEX+iSpecies]*Vector) * hs[iSpecies];
312  sumJeve_y += -(rho*Ds[iSpecies]*GV[RHOS_INDEX+iSpecies][1] - V_i[RHOS_INDEX+iSpecies]*Vector) * eve_i[iSpecies];
313  }
314 
315  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
316  residual[iSpecies] -= 0.0;
317  residual[nSpecies] -= Volume*(yinv*total_viscosity_i*(GV[nSpecies+2][1]+GV[nSpecies+3][0])
318  -TWO3*AuxVar_Grad_i[0][0]);
319  residual[nSpecies+1] -= Volume*(yinv*total_viscosity_i*2*(GV[nSpecies+3][1]-v*yinv)
320  -TWO3*AuxVar_Grad_i[0][1]);
321  residual[nSpecies+2] -= Volume*(yinv*(-sumJhs_y + total_viscosity_i*(u*(GV[nSpecies+3][0]+GV[nSpecies+2][1])
322  +v*TWO3*(2*GV[nSpecies+3][1]-GV[nSpecies+2][0]
323  -v*yinv+rho*turb_ke_i))-qy_t)
324  -TWO3*(AuxVar_Grad_i[1][1]+AuxVar_Grad_i[2][0]));
325  residual[nSpecies+3] -= Volume*(yinv*(-sumJeve_y -qy_ve));
326  }
327 
328  return ResidualType<>(residual, jacobian, nullptr);
329 }
