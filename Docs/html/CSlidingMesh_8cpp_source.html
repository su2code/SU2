231

264

329  Coeff_Vect = tmp_Coeff_Vect;

331

342

344  donor_OldiPoint = donor_StartIndex;
345  }

555  * - The closest donor element becomes the core, at each iteration a new layer of elements around the core is taken into account
556  */

559

564

566 
567  nEdgeVisited = Donor_nLinkedNodes[vPoint];
568 
569  for (iEdgeVisited = 0; iEdgeVisited < nEdgeVisited; iEdgeVisited++){
570

572

574 
575  check = 0;
576 
577  for( jj = 0; jj < nAlreadyVisited; jj++ ){
578  if( donor_iPoint == alreadyVisitedDonor[jj] ){
579  check = 1;
580  break;
581  }
582  }
583

585  for( jj = 0; jj < nToVisit; jj++ )
586  if( donor_iPoint == ToVisit[jj] ){
587  check = 1;
588  break;
589  }
590  }
591 
592  if( check == 0 ){
593  /*--- If the node was not already visited, visit it and list it into data structure ---*/
594 
595  tmpVect = new unsigned long[ nToVisit + 1 ];
596 
597  for( jj = 0; jj < nToVisit; jj++ )
598  tmpVect[jj] = ToVisit[jj];
599  tmpVect[nToVisit] = donor_iPoint;
600 
601

603 
604  ToVisit = tmpVect;
605  tmpVect = nullptr;
606 
607  nToVisit++;
608

610

612

614  for (ii = 0; ii < 2*nEdges_donor + 2; ii++)
615  donor_element[ii] = new su2double[nDim];

624  donor_element[0], donor_element[jj], donor_element[jj+1], Normal);
625 
626  for (ii = 0; ii < 2*nEdges_donor + 2; ii++)
627  delete [] donor_element[ii];
628  delete [] donor_element;
629 
630  /*--- In case the element intersect the target cell update the auxiliary communication data structure ---*/
631 
632  tmp_Coeff_Vect = new su2double[ nDonorPoints + 1 ];
633  tmp_Donor_Vect = new unsigned long[ nDonorPoints + 1 ];
634  tmp_storeProc = new unsigned long[ nDonorPoints + 1 ];
635 
636  for( iDonor = 0; iDonor < nDonorPoints; iDonor++){

705  delete [] storeProc; storeProc = nullptr;
706  }

719 int CSlidingMesh::Build_3D_surface_element(const su2vector<unsigned long>& map, const su2vector<unsigned long>& startIndex,
720  const su2vector<unsigned long>& nNeighbor, su2activematrix const& coord,
721  unsigned long centralNode, su2double** element) {
722 
723  /*--- Given a node "centralNode", this routines reconstruct the vertex centered
724  * surface element around the node and store it into "element" ---*/
725 
726  constexpr unsigned short nDim = 3;
727 
728  const unsigned long *OuterNodes;
729 
730  /* --- Store central node as element first point --- */
731 
732  for (unsigned short iDim = 0; iDim < nDim; iDim++)
733  element[0][iDim] = coord(centralNode,iDim);
734 
735  unsigned long nOuterNodes = nNeighbor[centralNode];
736 
737  OuterNodes = &map[ startIndex[centralNode] ];
738 
739  // For each neighbor n of centralNode, store <=2 neighbors of centralNode that are neighbors of n.
740  su2matrix<int> OuterNodesNeighbour(nOuterNodes,2);
741  OuterNodesNeighbour = -1;
742  // Typically there are exactly 2 such neighbors, and all the neighbors of centralNode can be
743  // arranged into a closed chain. However at 1D boundaries of 2D markers, the neighbors might
744  // be an open chain, with the two ends having only 1 such neighbor.
745  // StartNode is a node where we can start to iterate through the chain. I.e. it's any node in
746  // case of a closed chain, or one of the two ends in case of an open chain.
747  int StartNode = 0;
748  for( unsigned long iNode = 0; iNode < nOuterNodes; iNode++ ){
749 
750  int count = 0; // number of neighboring outer nodes already found
751  const unsigned long iPoint = OuterNodes[ iNode ];
752  const unsigned long *ptr = &map[ startIndex[iPoint] ];
753 
754  for ( unsigned long jNode = 0; jNode < nNeighbor[iPoint]; jNode++ ){
755  const unsigned long jPoint = ptr[jNode];
756  for( unsigned long kNode = 0; kNode < nOuterNodes; kNode++ ){
757  if ( jPoint == OuterNodes[ kNode ] && jPoint != centralNode){
758  OuterNodesNeighbour(iNode,count) = static_cast<int>(kNode);
759  count++;
760  break;
761  }
762  }
763  }
764  if( count == 1 )
765  StartNode = static_cast<int>(iNode);
766  }
767 
768  /* --- Build element, starts from one outer node and loops along the external edges until the element is reconstructed --- */
769 
770  int CurrentNode = StartNode;
771  int NextNode = OuterNodesNeighbour(CurrentNode,0);
772  unsigned long iElementNode = 1;
773 
774  while( NextNode != -1 ){ // We finished iterating through the chain if it is an open chain and we reached the other end.
775 
776  for (unsigned short iDim = 0; iDim < nDim; iDim++)
777  element[ iElementNode ][iDim] = ( element[0][iDim] + coord(OuterNodes[ CurrentNode ], iDim) )/2.;
778 
779  iElementNode++;
780 
781  for (unsigned short iDim = 0; iDim < nDim; iDim++)
782  element[ iElementNode ][iDim] = ( element[0][iDim] + coord[ OuterNodes[ CurrentNode ] ][ iDim] +
783  coord(OuterNodes[ NextNode ], iDim) )/3.;
784  iElementNode++;
785 
786  // "Place the next domino piece in the correct orientation."
787  if( OuterNodesNeighbour(NextNode, 0) == CurrentNode){
788  CurrentNode = NextNode;
789  NextNode = OuterNodesNeighbour(NextNode, 1);
790  } else{
791  CurrentNode = NextNode;
792  NextNode = OuterNodesNeighbour(NextNode, 0);
793  }
794 
795  // We finished iterating through the chain if it is closed and we reached the beginning again.
796  if (CurrentNode == StartNode)
797  break;
798  }
799 
800  if( CurrentNode == StartNode ){ // This is a closed element, so add again element 1 to the end of the structure, useful later
801  for (unsigned short iDim = 0; iDim < nDim; iDim++)
802  element[ iElementNode ][iDim] = element[1][iDim];
803  iElementNode++;
804  } else{
805  for (unsigned short iDim = 0; iDim < nDim; iDim++)
806  element[ iElementNode ][iDim] = ( element[0][iDim] + coord(OuterNodes[ CurrentNode ], iDim) )/2.;
807  iElementNode++;
808  }
809 
810  return static_cast<int>(iElementNode);
811 
812 }

814 su2double CSlidingMesh::ComputeLineIntersectionLength(unsigned short nDim, const su2double* A1, const su2double* A2,
815  const su2double* B1, const su2double* B2, const su2double* Direction) {
816 
817  /*--- Given 2 segments, each defined by 2 points, it projects them along a given direction
818  * and it computes the length of the segment resulting from their intersection ---*/
819  /*--- The algorithm works for both 2D and 3D problems ---*/
820 
821  unsigned short iDim;
822 
823  su2double dotA2, dotB1, dotB2;
824 
825  dotA2 = 0;
826  for(iDim = 0; iDim < nDim; iDim++)
827  dotA2 += ( A2[iDim] - A1[iDim] ) * Direction[iDim];
828 
829  if( dotA2 >= 0 ){
830  dotB1 = 0;
831  dotB2 = 0;
832  for(iDim = 0; iDim < nDim; iDim++){
833  dotB1 += ( B1[iDim] - A1[iDim] ) * Direction[iDim];
834  dotB2 += ( B2[iDim] - A1[iDim] ) * Direction[iDim];
835  }
836  }
837  else{
838  dotA2 *= -1;
839 
840  dotB1 = 0;
841  dotB2 = 0;
842  for(iDim = 0; iDim < nDim; iDim++){
843  dotB1 -= ( B1[iDim] - A1[iDim] ) * Direction[iDim];
844  dotB2 -= ( B2[iDim] - A1[iDim] ) * Direction[iDim];
845  }
846  }
847 
848  if( dotB1 >= 0 && dotB1 <= dotA2 ){
849  if ( dotB2 < 0 )
850  return fabs( dotB1 );
851  if ( dotB2 > dotA2 )
852  return fabs( dotA2 - dotB1 );
853 
854  return fabs( dotB1 - dotB2 );
855  }
856 
857  if( dotB2 >= 0 && dotB2 <= dotA2 ){
858  if ( dotB1 < 0 )
859  return fabs(dotB2);
860  if ( dotB1 > dotA2 )
861  return fabs( dotA2 - dotB2 );
862  }
863 
864  if( ( dotB1 <= 0 && dotA2 <= dotB2 ) || ( dotB2 <= 0 && dotA2 <= dotB1 ) )
865  return fabs( dotA2 );
866 
867  return 0.0;
868 }

870 su2double CSlidingMesh::Compute_Triangle_Intersection(const su2double* A1, const su2double* A2, const su2double* A3,
871  const su2double* B1, const su2double* B2, const su2double* B3,
872  const su2double* Direction) {
873 
874  /* --- This routine is ONLY for 3D grids --- */
875  /* --- Projects triangle points onto a plane, specified by its normal "Direction", and calls the ComputeIntersectionArea routine --- */
876 
877  unsigned short iDim;
878  constexpr unsigned short nDim = 3;
879 
880  su2double I[3], J[3], K[3];
881  su2double a1[3], a2[3], a3[3];
882  su2double b1[3], b2[3], b3[3];
883  su2double m1, m2;
884 
885  /* --- Reference frame is determined by: x = A1A2 y = x ^ ( -Direction ) --- */
886 
887  for(iDim = 0; iDim < 3; iDim++){
888  a1[iDim] = 0;
889  a2[iDim] = 0;
890  a3[iDim] = 0;
891 
892  b1[iDim] = 0;
893  b2[iDim] = 0;
894  b3[iDim] = 0;
895  }
896 
897  m1 = 0;
898  for(iDim = 0; iDim < nDim; iDim++){
899  K[iDim] = Direction[iDim];
900 
901  m1 += K[iDim] * K[iDim];
902  }
903 
904  for(iDim = 0; iDim < nDim; iDim++)
905  K[iDim] /= sqrt(m1);
906 
907  m2 = 0;
908  for(iDim = 0; iDim < nDim; iDim++)
909  m2 += (A2[iDim] - A1[iDim]) * K[iDim];
910 
911  m1 = 0;
912  for(iDim = 0; iDim < nDim; iDim++){
913  I[iDim] = (A2[iDim] - A1[iDim]) - m2 * K[iDim];
914  m1 += I[iDim] * I[iDim];
915  }
916 
917  for(iDim = 0; iDim < nDim; iDim++)
918  I[iDim] /= sqrt(m1);
919 
920  // Cross product to find Y
921  J[0] = K[1]*I[2] - K[2]*I[1];
922  J[1] = -(K[0]*I[2] - K[2]*I[0]);
923  J[2] = K[0]*I[1] - K[1]*I[0];
924 
925  /* --- Project all points on the plane specified by Direction and change their reference frame taking A1 as origin --- */
926 
927  for(iDim = 0; iDim < nDim; iDim++){
928  a2[0] += (A2[iDim] - A1[iDim]) * I[iDim];
929  a2[1] += (A2[iDim] - A1[iDim]) * J[iDim];
930  a2[2] += (A2[iDim] - A1[iDim]) * K[iDim];
931 
932  a3[0] += (A3[iDim] - A1[iDim]) * I[iDim];
933  a3[1] += (A3[iDim] - A1[iDim]) * J[iDim];
934  a3[2] += (A3[iDim] - A1[iDim]) * K[iDim];
935 
936  b1[0] += (B1[iDim] - A1[iDim]) * I[iDim];
937  b1[1] += (B1[iDim] - A1[iDim]) * J[iDim];
938  b1[2] += (B1[iDim] - A1[iDim]) * K[iDim];
939 
940  b2[0] += (B2[iDim] - A1[iDim]) * I[iDim];
941  b2[1] += (B2[iDim] - A1[iDim]) * J[iDim];
942  b2[2] += (B2[iDim] - A1[iDim]) * K[iDim];
943 
944  b3[0] += (B3[iDim] - A1[iDim]) * I[iDim];
945  b3[1] += (B3[iDim] - A1[iDim]) * J[iDim];
946  b3[2] += (B3[iDim] - A1[iDim]) * K[iDim];
947  }
948 
949  /*--- Compute intersection area ---*/
950 
951  return ComputeIntersectionArea( a1, a2, a3, b1, b2, b3 );
952 }

954 su2double CSlidingMesh::ComputeIntersectionArea(const su2double* P1, const su2double* P2, const su2double* P3,
955  const su2double* Q1, const su2double* Q2, const su2double* Q3) {
956 
957  /* --- This routines computes the area of the polygonal element generated by the superimposition of 2 planar triangle --- */
958  /* --- The 2 triangle must lie on the same plane --- */
959 
960  unsigned short iDim, nPoints = 0, i, j, k;
961  unsigned short min_theta_index;
962 
963  su2double points[16][2], IntersectionPoint[2], theta[6];
964  su2double TriangleP[4][2], TriangleQ[4][2];
965  su2double Area, det, dot1, dot2, dtmp, min_theta;
966 
967  constexpr unsigned short nDim = 2;
968 
969  for(iDim = 0; iDim < nDim; iDim++){
970  TriangleP[0][iDim] = 0;
971  TriangleP[1][iDim] = P2[iDim] - P1[iDim];
972  TriangleP[2][iDim] = P3[iDim] - P1[iDim];
973  TriangleP[3][iDim] = 0;
974 
975  TriangleQ[0][iDim] = Q1[iDim] - P1[iDim];
976  TriangleQ[1][iDim] = Q2[iDim] - P1[iDim];
977  TriangleQ[2][iDim] = Q3[iDim] - P1[iDim];
978  TriangleQ[3][iDim] = Q1[iDim] - P1[iDim];
979  }
980 
981 
982  for( j = 0; j < 3; j++){
983  if( CheckPointInsideTriangle(TriangleP[j], TriangleQ[0], TriangleQ[1], TriangleQ[2]) ){
984 
985  // Then P1 is also inside triangle Q, so store it
986  for(iDim = 0; iDim < nDim; iDim++)
987  points[nPoints][iDim] = TriangleP[j][iDim];
988 
989  nPoints++;
990  }
991  }
992 
993  for( j = 0; j < 3; j++){
994  if( CheckPointInsideTriangle(TriangleQ[j], TriangleP[0], TriangleP[1], TriangleP[2]) ){
995 
996  // Then Q1 is also inside triangle P, so store it
997  for(iDim = 0; iDim < nDim; iDim++)
998  points[nPoints][iDim] = TriangleQ[j][iDim];
999 
1000  nPoints++;
1001  }
1002  }
1003 
1004 
1005  // Compute all edge intersections
1006 
1007  for( j = 0; j < 3; j++){
1008  for( i = 0; i < 3; i++){
1009 
1010  det = (TriangleP[j][0] - TriangleP[j+1][0]) * (TriangleQ[i][1] - TriangleQ[i+1][1]) -
1011  (TriangleP[j][1] - TriangleP[j+1][1]) * (TriangleQ[i][0] - TriangleQ[i+1][0]);
1012 
1013  if ( det != 0.0 ){
1014  ComputeLineIntersectionPoint( TriangleP[j], TriangleP[j+1], TriangleQ[i], TriangleQ[i+1], IntersectionPoint );
1015 
1016  dot1 = 0;
1017  dot2 = 0;
1018  for(iDim = 0; iDim < nDim; iDim++){
1019  dot1 += ( TriangleP[j][iDim] - IntersectionPoint[iDim] ) * ( TriangleP[j+1][iDim] - IntersectionPoint[iDim] );
1020  dot2 += ( TriangleQ[i][iDim] - IntersectionPoint[iDim] ) * ( TriangleQ[i+1][iDim] - IntersectionPoint[iDim] );
1021  }
1022 
1023  if( dot1 <= 0 && dot2 <= 0 ){ // It found one intersection
1024 
1025  // Store temporarily the intersection point
1026 
1027  for(iDim = 0; iDim < nDim; iDim++)
1028  points[nPoints][iDim] = IntersectionPoint[iDim];
1029 
1030  nPoints++;
1031  }
1032  }
1033  }
1034  }
1035 
1036  // Remove double points, if any
1037 
1038  for( i = 0; i < nPoints; i++){
1039  for( j = i+1; j < nPoints; j++){
1040  if(points[j][0] == points[i][0] && points[j][1] == points[i][1]){
1041  for( k = j; k < nPoints-1; k++){
1042  points[k][0] = points[k+1][0];
1043  points[k][1] = points[k+1][1];
1044  }
1045  nPoints--;
1046  j--;
1047  }
1048  }
1049  }
1050 
1051  // Re-order nodes
1052 
1053  for( i = 1; i < nPoints; i++){ // Change again reference frame
1054  for(iDim = 0; iDim < nDim; iDim++)
1055  points[i][iDim] -= points[0][iDim];
1056 
1057  // Compute polar azimuth for each node but the first
1058  theta[i] = atan2(points[i][1], points[i][0]);
1059  }
1060 
1061  for(iDim = 0; iDim < nDim; iDim++)
1062  points[0][iDim] = 0;
1063 
1064  for( i = 1; i < nPoints; i++){
1065 
1066  min_theta = theta[i];
1067  min_theta_index = 0;
1068 
1069  for( j = i + 1; j < nPoints; j++){
1070 
1071  if( theta[j] < min_theta ){
1072  min_theta = theta[j];
1073  min_theta_index = j;
1074  }
1075  }
1076 
1077  if( min_theta_index != 0 ){
1078  dtmp = theta[i];
1079  theta[i] = theta[min_theta_index];
1080  theta[min_theta_index] = dtmp;
1081 
1082  dtmp = points[i][0];
1083  points[i][0] = points[min_theta_index][0];
1084  points[min_theta_index][0] = dtmp;
1085 
1086  dtmp = points[i][1];
1087  points[i][1] = points[min_theta_index][1];
1088  points[min_theta_index][1] = dtmp;
1089  }
1090  }
1091 
1092  // compute area using cross product rule, points position are referred to the
1093  // 2-dimensional, local, reference frame centered in points[0]
1094 
1095  Area = 0;
1096 
1097  if (nPoints > 2){
1098  for( i = 1; i < nPoints-1; i++ ){
1099 
1100  // Ax*By
1101  Area += ( points[i][0] - points[0][0] ) * ( points[i+1][1] - points[0][1] );
1102 
1103  // Ay*Bx
1104  Area -= ( points[i][1] - points[0][1] ) * ( points[i+1][0] - points[0][0] );
1105  }
1106  }
1107 
1108  return fabs(Area)/2;
1109 }

1111 void CSlidingMesh::ComputeLineIntersectionPoint(const su2double* A1, const su2double* A2, const su2double* B1,
1112  const su2double* B2, su2double* IntersectionPoint ){
1113 
1114  /* --- Uses determinant rule to compute the intersection point between 2 straight segments --- */
1115  /* This works only for lines on a 2D plane, A1, A2 and B1, B2 are respectively the head and the tail points of each segment,
1116  * since they're on a 2D plane they are defined by a 2-elements array containing their coordinates */
1117

1119 
1120  det = (A1[0] - A2[0]) * (B1[1] - B2[1]) - (A1[1] - A2[1]) * (B1[0] - B2[0]);
1121 
1122  if ( det != 0.0 ){ // else there is no intersection point
1123  IntersectionPoint[0] = ( ( A1[0]*A2[1] - A1[1]*A2[0] ) * ( B1[0] - B2[0] ) - ( B1[0]*B2[1] - B1[1]*B2[0] ) * ( A1[0] - A2[0] ) ) / det;
1124  IntersectionPoint[1] = ( ( A1[0]*A2[1] - A1[1]*A2[0] ) * ( B1[1] - B2[1] ) - ( B1[0]*B2[1] - B1[1]*B2[0] ) * ( A1[1] - A2[1] ) ) / det;
1125  }

1128 bool CSlidingMesh::CheckPointInsideTriangle(const su2double* Point, const su2double* T1, const su2double* T2, const su2double* T3) {
1129 
1130  /* --- Check whether a point "Point" lies inside or outside a triangle defined by 3 points "T1", "T2", "T3" --- */
1131  /* For each edge it checks on which side the point lies:
1132  * - Computes the unit vector pointing at the internal side of the edge
1133  * - Comutes the vector that connects the point to a point along the edge
1134  * - If the dot product is positive it means that the point is on the internal side of the edge
1135  * - If the check is positive for all the 3 edges, then the point lies within the triangle
1136  */
1137 
1138  unsigned short iDim, check = 0;
1139 
1140  su2double vect1[2], vect2[2], r[2];
1141  su2double dot;
1142 
1143  constexpr unsigned short nDim = 2;
1144 
1145  /* --- Check first edge --- */
1146 
1147  dot = 0;
1148  for(iDim = 0; iDim < nDim; iDim++){
1149  vect1[iDim] = T3[iDim] - T1[iDim]; // vec 1 is aligned to the edge
1150  vect2[iDim] = T2[iDim] - T1[iDim]; // vect 2 is the vector connecting one edge point to the third triangle vertex
1151 
1152  r[iDim] = Point[iDim] - T1[iDim]; // Connects point to vertex T1
1153 
1154  dot += vect2[iDim] * vect2[iDim];
1155  }
1156  dot = sqrt(dot);
1157 
1158  for(iDim = 0; iDim < nDim; iDim++)
1159  vect2[iDim] /= dot;
1160 
1161  dot = 0;
1162  for(iDim = 0; iDim < nDim; iDim++)
1163  dot += vect1[iDim] * vect2[iDim];
1164 
1165  for(iDim = 0; iDim < nDim; iDim++)
1166  vect1[iDim] = T3[iDim] - (T1[iDim] + dot * vect2[iDim]); // Computes the inward unit vector
1167 
1168  dot = 0;
1169  for(iDim = 0; iDim < nDim; iDim++) // Checs that the point lies on the internal plane
1170  dot += vect1[iDim] * r[iDim];
1171 
1172  if (dot >= 0)
1173  check++;
1174 
1175  /* --- Check second edge --- */
1176 
1177  dot = 0;
1178  for(iDim = 0; iDim < nDim; iDim++){
1179  vect1[iDim] = T1[iDim] - T2[iDim];
1180  vect2[iDim] = T3[iDim] - T2[iDim];
1181 
1182  r[iDim] = Point[iDim] - T2[iDim];
1183 
1184  dot += vect2[iDim] * vect2[iDim];
1185  }
1186  dot = sqrt(dot);
1187 
1188  for(iDim = 0; iDim < nDim; iDim++)
1189  vect2[iDim] /= dot;
1190 
1191  dot = 0;
1192  for(iDim = 0; iDim < nDim; iDim++)
1193  dot += vect1[iDim] * vect2[iDim];
1194 
1195  for(iDim = 0; iDim < nDim; iDim++)
1196  vect1[iDim] = T1[iDim] - (T2[iDim] + dot * vect2[iDim]);
1197
