41  public:
42  /*!
43  * \brief Constructor of the class.
44  */
45  CPolynomialViscosity(const su2double* val_coeffs) {
46  for (int i = 0; i < N; ++i) {
47  coeffs_[i] = val_coeffs[i];
48  }
49  }
50 
51  /*!
52  * \brief Set Viscosity.
53  */
54  void SetViscosity(su2double t, su2double rho) override {
55  /* Evaluate the new mu from the coefficients and temperature. */
56  mu_ = coeffs_[0];
57  su2double t_i = 1.0;
58  for (int i = 1; i < N; ++i) {
59  t_i *= t;
60  mu_ += coeffs_[i] * t_i;
61  }
62  }
63 
64  private:
65  array<su2double, N> coeffs_; /*!< \brief Polynomial coefficients for viscosity as a function of temperature. */
66 };
