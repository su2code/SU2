32 CADTPointsOnlyClass::CADTPointsOnlyClass(unsigned short nDim,
33  unsigned long nPoints,
34  const su2double *coor,
35  const unsigned long *pointID,
36  const bool globalTree) {
37 
38  /* Allocate some thread-safe working variables if required. */
39 #ifdef HAVE_OMP
40  FrontLeaves.resize(omp_get_max_threads());
41  FrontLeavesNew.resize(omp_get_max_threads());
42 #endif
43 
44  /*--- Make a distinction between parallel and sequential mode. ---*/
45 
46 #ifdef HAVE_MPI
47 
48  /* Parallel mode. Check whether a global or a local tree must be built. */
49  if( globalTree ) {
50 
51  /*--- A global tree must be built. All points are gathered on all ranks.
52  First determine the number of points per rank and store them in such
53  a way that the info can be used directly in Allgatherv. ---*/
54  int rank, size;
55  SU2_MPI::Comm_rank(SU2_MPI::GetComm(), &rank);
56  SU2_MPI::Comm_size(SU2_MPI::GetComm(), &size);
57 
58  vector<int> recvCounts(size), displs(size);
59  int sizeLocal = (int) nPoints;
60 
61  SU2_MPI::Allgather(&sizeLocal, 1, MPI_INT, recvCounts.data(), 1,
62  MPI_INT, SU2_MPI::GetComm());
63  displs[0] = 0;
64  for(int i=1; i<size; ++i) displs[i] = displs[i-1] + recvCounts[i-1];
65 
66  int sizeGlobal = displs.back() + recvCounts.back();
67 
68  /*--- Gather the local pointID's and the ranks of the nodes on all ranks. ---*/
69  localPointIDs.resize(sizeGlobal);
70  SU2_MPI::Allgatherv(pointID, sizeLocal, MPI_UNSIGNED_LONG, localPointIDs.data(),
71  recvCounts.data(), displs.data(), MPI_UNSIGNED_LONG,
72  SU2_MPI::GetComm());
73 
74  ranksOfPoints.resize(sizeGlobal);
75  vector<int> rankLocal(sizeLocal, rank);
76  SU2_MPI::Allgatherv(rankLocal.data(), sizeLocal, MPI_INT, ranksOfPoints.data(),
77  recvCounts.data(), displs.data(), MPI_INT, SU2_MPI::GetComm());
78 
79  /*--- Gather the coordinates of the points on all ranks. ---*/
80  for(int i=0; i<size; ++i) {recvCounts[i] *= nDim; displs[i] *= nDim;}
81 
82  coorPoints.resize(nDim*sizeGlobal);
83  SU2_MPI::Allgatherv(coor, nDim*sizeLocal, MPI_DOUBLE, coorPoints.data(),
84  recvCounts.data(), displs.data(), MPI_DOUBLE, SU2_MPI::GetComm());
85  }
86  else {
87 
88  /*--- A local tree must be built. Copy the coordinates and point IDs and
89  set the ranks to the rank of this processor. ---*/
90  int rank;
91  SU2_MPI::Comm_rank(SU2_MPI::GetComm(), &rank);
92 
93  coorPoints.assign(coor, coor + nDim*nPoints);
94  localPointIDs.assign(pointID, pointID + nPoints);
95  ranksOfPoints.assign(nPoints, rank);
96  }
97 
98 #else
99 
100  /*--- Sequential mode. Copy the coordinates and point IDs and
101  set the ranks to MASTER_NODE. ---*/
102  coorPoints.assign(coor, coor + nDim*nPoints);
103  localPointIDs.assign(pointID, pointID + nPoints);
104  ranksOfPoints.assign(nPoints, MASTER_NODE);
105 
106 #endif
107 
108  /*--- Build the tree. ---*/
109  BuildADT(nDim, localPointIDs.size(), coorPoints.data());
110 
111  /*--- Reserve the memory for frontLeaves and frontLeavesNew,
112  which are needed during the tree search. ---*/
113  for (auto& vec : FrontLeaves) vec.reserve(200);
114  for (auto& vec : FrontLeavesNew) vec.reserve(200);
115 }

117 void CADTPointsOnlyClass::DetermineNearestNode_impl(vector<unsigned long>& frontLeaves,
118  vector<unsigned long>& frontLeavesNew,
119  const su2double *coor,
120  su2double &dist,
121  unsigned long &pointID,
122  int &rankID) const {
123 
124  const bool wasActive = AD::BeginPassive();
125 
126  /*--------------------------------------------------------------------------*/
127  /*--- Step 1: Initialize the nearest node to the central node of the ---*/
128  /*--- root leaf. Note that the distance is the distance squared ---*/
129  /*--- to avoid a sqrt. ---*/
130  /*--------------------------------------------------------------------------*/
131 
132  unsigned long kk = leaves[0].centralNodeID, minIndex;
133  const su2double *coorTarget = coorPoints.data() + nDimADT*kk;
134 
135  pointID = localPointIDs[kk];
136  rankID = ranksOfPoints[kk];
137  minIndex = kk;
138  dist = 0.0;
139  for(unsigned short l=0; l<nDimADT; ++l) {
140  const su2double ds = coor[l] - coorTarget[l];
141  dist += ds*ds;
142  }
143 
144  /*--------------------------------------------------------------------------*/
145  /*--- Step 2: Traverse the tree and search for the nearest node. ---*/
146  /*--- During the tree traversal the currently stored distance ---*/
147  /*--- squared is modified, because the guaranteed minimum ---*/
148  /*--- distance squared of the children could be smaller. ---*/
149  /*--------------------------------------------------------------------------*/
150 
151  /* Start at the root leaf of the ADT, i.e. initialize frontLeaves such that
152  it only contains the root leaf. Make sure to wipe out any data from a
153  previous search. */
154  frontLeaves.clear();
155  frontLeaves.push_back(0);
156 
157  /* Infinite loop of the tree traversal. */
158  for(;;) {
159 
160  /* Initialize the new front, i.e. the front for the next round, to empty. */
161  frontLeavesNew.clear();
162 
163  /* Loop over the leaves of the current front. */
164  for(unsigned long i=0; i<frontLeaves.size(); ++i) {
165 
166  /* Store the current leaf a bit easier in ll and loop over its children. */
167  const unsigned long ll = frontLeaves[i];
168  for(unsigned short mm=0; mm<2; ++mm) {
169 
170  /* Determine whether this child contains a node or a leaf
171  of the next level of the ADT. */
172  kk = leaves[ll].children[mm];
173  if( leaves[ll].childrenAreTerminal[mm] ) {
174 
175  /*--- Child contains a node. Compute the distance squared to this node
176  and store it if this distance squared is less than the currently
177  stored value. ---*/
178  coorTarget = coorPoints.data() + nDimADT*kk;
179  su2double distTarget = 0;
180  for(unsigned short l=0; l<nDimADT; ++l) {
181  const su2double ds = coor[l] - coorTarget[l];
182  distTarget += ds*ds;
183  }
184 
185  if(distTarget < dist) {
186  dist = distTarget;
187  pointID = localPointIDs[kk];
188  rankID = ranksOfPoints[kk];
189  minIndex = kk;
190  }
191  }
192  else {
193 
194  /*--- Child contains a leaf. Determine the possible minimum distance
195  squared to that leaf. ---*/
196  su2double posDist = 0.0;
197  for(unsigned short l=0; l<nDimADT; ++l) {
198  su2double ds = 0.0;
199  if( coor[l] < leaves[kk].xMin[l]) ds = coor[l] - leaves[kk].xMin[l];
200  else if(coor[l] > leaves[kk].xMax[l]) ds = coor[l] - leaves[kk].xMax[l];
201 
202  posDist += ds*ds;
203  }
204 
205  /*--- Check if the possible minimum distance is less than the currently
206  stored minimum distance. If so this leaf must be stored for the
207  next round. In that case the distance squared to the central node is
208  determined, which is used to update the currently stored value. ---*/
209  if(posDist < dist) {
210  frontLeavesNew.push_back(kk);
211 
212  const unsigned long jj = leaves[kk].centralNodeID;
213 
214  coorTarget = coorPoints.data() + nDimADT*jj;
215  su2double distTarget = 0;
216  for(unsigned short l=0; l<nDimADT; ++l) {
217  const su2double ds = coor[l] - coorTarget[l];
218  distTarget += ds*ds;
219  }
220 
221  if(distTarget < dist) {
222  dist = distTarget;
223  pointID = localPointIDs[jj];
224  rankID = ranksOfPoints[jj];
225  minIndex = jj;
226  }
227  }
228  }
229  }
230  }
231 
232  /*--- End of the loop over the current front. Copy the data from
233  frontLeavesNew to frontLeaves for the next round. If the new front
234  is empty the entire tree has been traversed and a break can be made
235  from the infinite loop. ---*/
236  frontLeaves = frontLeavesNew;
237  if(frontLeaves.size() == 0) break;
238  }
239 
240  AD::EndPassive(wasActive);
241 
242  /* Recompute the distance to get the correct dependency if we use AD */
243  coorTarget = coorPoints.data() + nDimADT*minIndex;
244  dist = 0.0;
245  for(unsigned short l=0; l<nDimADT; ++l) {
246  const su2double ds = coor[l] - coorTarget[l];
247  dist += ds*ds;
248  }
249 
250 
251  /* At the moment the distance squared to the nearest node is stored.
252  Take the sqrt to obtain the correct value. */
253  dist = sqrt(dist);
254 
255 }
