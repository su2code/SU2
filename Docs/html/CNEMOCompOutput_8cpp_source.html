33 CNEMOCompOutput::CNEMOCompOutput(const CConfig *config, unsigned short nDim) : CFlowOutput(config, nDim, false) {
34 
35  turb_model = config->GetKind_Turb_Model();
36  nSpecies = config->GetnSpecies();
37 
38  /*--- Set the default history fields if nothing is set in the config file ---*/
39 
40  if (nRequestedHistoryFields == 0){
41  requestedHistoryFields.emplace_back("ITER");
42  requestedHistoryFields.emplace_back("RMS_RES");
43  nRequestedHistoryFields = requestedHistoryFields.size();
44  }
45  if (nRequestedScreenFields == 0){
46  if (config->GetTime_Domain()) requestedScreenFields.emplace_back("TIME_ITER");
47  if (multiZone) requestedScreenFields.emplace_back("OUTER_ITER");
48  requestedScreenFields.emplace_back("INNER_ITER");
49  for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
50  requestedScreenFields.emplace_back("RMS_DENSITY_" + std::to_string(iSpecies));
51  requestedScreenFields.emplace_back("RMS_MOMENTUM-X");
52  requestedScreenFields.emplace_back("RMS_MOMENTUM-Y");
53  requestedScreenFields.emplace_back("RMS_ENERGY");
54  requestedScreenFields.emplace_back("RMS_ENERGY_VE");
55  nRequestedScreenFields = requestedScreenFields.size();
56  }
57  if (nRequestedVolumeFields == 0){
58  requestedVolumeFields.emplace_back("COORDINATES");
59  requestedVolumeFields.emplace_back("SOLUTION");
60  requestedVolumeFields.emplace_back("PRIMITIVE");
61  requestedVolumeFields.emplace_back("AUXILIARY");
62  nRequestedVolumeFields = requestedVolumeFields.size();
63  }
64 
65  if (gridMovement) {
66  auto notFound = requestedVolumeFields.end();
67  if (find(requestedVolumeFields.begin(), notFound, string("GRID_VELOCITY")) == notFound) {
68  requestedVolumeFields.emplace_back("GRID_VELOCITY");
69  nRequestedVolumeFields ++;
70  }
71  }
72 
73  stringstream ss;
74  ss << "Zone " << config->GetiZone() << " (Comp. Fluid)";
75  multiZoneHeaderString = ss.str();
76 
77  /*--- Set the volume filename --- */
78 
79  volumeFilename = config->GetVolume_FileName();
80 
81  /*--- Set the surface filename --- */
82 
83  surfaceFilename = config->GetSurfCoeff_FileName();
84 
85  /*--- Set the restart filename --- */
86 
87  restartFilename = config->GetRestart_FileName();
88 
89  /*--- Set the default convergence field --- */
90 
91  if (convFields.empty() ) convFields.emplace_back("RMS_DENSITY_0");
92 
93  if (config->GetFixed_CL_Mode()) {
94  bool found = false;
95  for (unsigned short iField = 0; iField < convFields.size(); iField++)
96  if (convFields[iField] == "LIFT") found = true;
97  if (!found) {
98  if (rank == MASTER_NODE)
99  cout<<" Fixed CL: Adding LIFT as Convergence Field to ensure convergence to target CL"<<endl;
100  convFields.emplace_back("LIFT");
101  newFunc.resize(convFields.size());
102  oldFunc.resize(convFields.size());
103  cauchySerie.resize(convFields.size(), vector<su2double>(nCauchy_Elems, 0.0));
104  }
105  }

293 void CNEMOCompOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint){
294 
295  const auto* Node_Flow = solver[FLOW_SOL]->GetNodes();
296  auto* Node_Geo = geometry->nodes;
297  const auto nSpecies = config->GetnSpecies();
298 
299  LoadCoordinates(Node_Geo->GetCoord(iPoint), iPoint);
300 
301  for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
302  SetVolumeOutputValue("DENSITY_" + std::to_string(iSpecies), iPoint, Node_Flow->GetSolution(iPoint, iSpecies));
303 
304  SetVolumeOutputValue("MOMENTUM-X", iPoint, Node_Flow->GetSolution(iPoint, nSpecies));
305  SetVolumeOutputValue("MOMENTUM-Y", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+1));
306  if (nDim == 3){
307  SetVolumeOutputValue("MOMENTUM-Z", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+2));
308  SetVolumeOutputValue("ENERGY", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+3));
309  SetVolumeOutputValue("ENERGY_VE", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+4));
310  } else {
311  SetVolumeOutputValue("ENERGY", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+2));
312  SetVolumeOutputValue("ENERGY_VE", iPoint, Node_Flow->GetSolution(iPoint, nSpecies+3));
313  }
314 
315  for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
316  SetVolumeOutputValue("MASSFRAC_" + std::to_string(iSpecies), iPoint, Node_Flow->GetSolution(iPoint, iSpecies)/Node_Flow->GetDensity(iPoint));
317 
318  if (gridMovement){
319  SetVolumeOutputValue("GRID_VELOCITY-X", iPoint, Node_Geo->GetGridVel(iPoint)[0]);
320  SetVolumeOutputValue("GRID_VELOCITY-Y", iPoint, Node_Geo->GetGridVel(iPoint)[1]);
321  if (nDim == 3)
322  SetVolumeOutputValue("GRID_VELOCITY-Z", iPoint, Node_Geo->GetGridVel(iPoint)[2]);
323  }
324 
325  SetVolumeOutputValue("PRESSURE", iPoint, Node_Flow->GetPressure(iPoint));
326  SetVolumeOutputValue("TEMPERATURE_TR", iPoint, Node_Flow->GetTemperature(iPoint));
327  SetVolumeOutputValue("TEMPERATURE_VE", iPoint, Node_Flow->GetTemperature_ve(iPoint));
328  SetVolumeOutputValue("MACH", iPoint, sqrt(Node_Flow->GetVelocity2(iPoint))/Node_Flow->GetSoundSpeed(iPoint));
329 
330  const su2double factor = solver[FLOW_SOL]->GetReferenceDynamicPressure();
331  SetVolumeOutputValue("PRESSURE_COEFF", iPoint, (Node_Flow->GetPressure(iPoint) - solver[FLOW_SOL]->GetPressure_Inf())/factor);
332 
333  if (config->GetViscous()){
334  SetVolumeOutputValue("LAMINAR_VISCOSITY", iPoint, Node_Flow->GetLaminarViscosity(iPoint));
335  }
336 
337  for(iSpecies = 0; iSpecies < nSpecies; iSpecies++)
338  SetVolumeOutputValue("RES_DENSITY_" + std::to_string(iSpecies), iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, iSpecies));
339 
340  SetVolumeOutputValue("RES_MOMENTUM-X", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies));
341  SetVolumeOutputValue("RES_MOMENTUM-Y", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+1));
342  if (nDim == 3){
343  SetVolumeOutputValue("RES_MOMENTUM-Z", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+2));
344  SetVolumeOutputValue("RES_ENERGY", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+3));
345  SetVolumeOutputValue("RES_ENERGY_VE", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+4));
346  } else {
347  SetVolumeOutputValue("RES_ENERGY", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+2));
348  SetVolumeOutputValue("RES_ENERGY_VE", iPoint, solver[FLOW_SOL]->LinSysRes(iPoint, nSpecies+3));
349  }
350 
351  if (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE && config->GetKind_SlopeLimit_Flow() != LIMITER::VAN_ALBADA_EDGE) {
352  SetVolumeOutputValue("LIMITER_DENSITY", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 0));
353  SetVolumeOutputValue("LIMITER_MOMENTUM-X", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 1));
354  SetVolumeOutputValue("LIMITER_MOMENTUM-Y", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 2));
355  if (nDim == 3){
356  SetVolumeOutputValue("LIMITER_MOMENTUM-Z", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 3));
357  SetVolumeOutputValue("LIMITER_ENERGY", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 4));
358  } else {
359  SetVolumeOutputValue("LIMITER_ENERGY", iPoint, Node_Flow->GetLimiter_Primitive(iPoint, 3));
360  }
361  }
362 
363  LoadVolumeData_Scalar(config, solver, geometry, iPoint);
364 
365  LoadCommonFVMOutputs(config, geometry, iPoint);
366 
367  if (config->GetTime_Domain()) {
368  LoadTimeAveragedData(iPoint, Node_Flow);
369  }
