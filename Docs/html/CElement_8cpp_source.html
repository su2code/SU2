31 CElement::CElement(unsigned short ngauss, unsigned short nnodes, unsigned short ndim) {
32 
33  nGaussPoints = ngauss;
34  nNodes = nnodes;
35  nDim = ndim;
36 
37  /*--- Allocate structures. ---*/
38 
39  /*--- Always allocate MAXNDIM to have enough space in all cases.
40  * e.g. elements embedded on a curved surfaces. ---*/
41  CurrentCoord.resize(nNodes, MAXNDIM) = su2double(0.0);
42  RefCoord.resize(nNodes, MAXNDIM) = su2double(0.0);
43 
44  GaussPoint.reserve(nGaussPoints);
45  for(unsigned short iGauss = 0; iGauss < nGaussPoints; ++iGauss)
46  GaussPoint.emplace_back(iGauss, nDim, nNodes);
47 
48  GaussWeight.resize(nGaussPoints) = su2double(0.0);
49  NodalExtrap.resize(nNodes, nGaussPoints) = su2double(0.0);
50 
51  NodalStress.resize(nNodes, 6) = su2double(0.0);
52 
53  Mab.resize(nNodes, nNodes);
54  Ks_ab.resize(nNodes, nNodes);
55  Kab.resize(nNodes);
56  for(auto& kab_i : Kab) kab_i.resize(nNodes, nDim*nDim);
57 
58  Kt_a.resize(nNodes, nDim);
59  FDL_a.resize(nNodes, nDim);
60 
61  HiHj.resize(nNodes,nNodes);
62  DHiDHj.resize(nNodes);
63  for(auto& DHiDHj_a : DHiDHj) {
64  DHiDHj_a.resize(nNodes);
65  for(auto& DHiDHj_ab : DHiDHj_a) DHiDHj_ab.resize(nDim,nDim);
66  }
67 
68  ClearElement();
69 }
