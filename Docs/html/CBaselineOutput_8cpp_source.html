72 void CBaselineOutput::SetVolumeOutputFields(CConfig *config){
73 
74  unsigned short iField = 0;
75 
76  /*--- The first three fields should be the coordinates, if not, something is wrong ---*/
77 
78  if (fields[0] != "x" || fields[1] != "y"){
79  SU2_MPI::Error("No coordinates found in the restart file!!", CURRENT_FUNCTION);
80  }
81  if (nDim == 3){
82  if (fields[2] != "z"){
83  SU2_MPI::Error("No coordinates found in the restart file!!", CURRENT_FUNCTION);
84  }
85  }
86 
87  // Grid coordinates
88  AddVolumeOutput(fields[0], fields[0], "COORDINATES", "x-component of the coordinate vector");
89  AddVolumeOutput(fields[1], fields[1], "COORDINATES", "y-component of the coordinate vector");
90  if (nDim == 3)
91  AddVolumeOutput(fields[2], fields[2], "COORDINATES", "z-component of the coordinate vector");
92 
93  // Add all the remaining fields
94 
95  for (iField = nDim; iField < fields.size(); iField++){
96  AddVolumeOutput(fields[iField], fields[iField], "SOLUTION","");
97  }
98 
99 }
