195 void CTurbSSTSolver::Postprocessing(CGeometry *geometry, CSolver **solver_container,
196  CConfig *config, unsigned short iMesh) {
197 
198  const su2double a1 = constants[7];
199 
200  /*--- Compute turbulence gradients. ---*/
201 
202  if (config->GetKind_Gradient_Method() == GREEN_GAUSS) {
203  SetSolution_Gradient_GG(geometry, config);
204  }
205  if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) {
206  SetSolution_Gradient_LS(geometry, config);
207  }
208 
209  AD::StartNoSharedReading();
210 
211  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
212 
213  SU2_OMP_FOR_STAT(omp_chunk_size)
214  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++) {
215 
216  /*--- Compute blending functions and cross diffusion ---*/
217 
218  const su2double rho = flowNodes->GetDensity(iPoint);
219  const su2double mu = flowNodes->GetLaminarViscosity(iPoint);
220 
221  const su2double dist = geometry->nodes->GetWall_Distance(iPoint);
222 
223  const su2double VorticityMag = max(GeometryToolbox::Norm(3, flowNodes->GetVorticity(iPoint)), 1e-12);
224  const su2double StrainMag = max(nodes->GetStrainMag(iPoint), 1e-12);
225  nodes->SetBlendingFunc(iPoint, mu, dist, rho, config->GetKind_Trans_Model());
226 
227  const su2double F2 = nodes->GetF2blending(iPoint);
228 
229  /*--- Compute the eddy viscosity ---*/
230 
231  const su2double kine = nodes->GetSolution(iPoint,0);
232  const su2double omega = nodes->GetSolution(iPoint,1);
233 
234  const auto& eddy_visc_var = sstParsedOptions.version == SST_OPTIONS::V1994 ? VorticityMag : StrainMag;
235  const su2double muT = max(0.0, rho * a1 * kine / max(a1 * omega, eddy_visc_var * F2));
236 
237  nodes->SetmuT(iPoint, muT);
238 
239  }
240  END_SU2_OMP_FOR
241 
242 
243  /*--- Compute turbulence index ---*/
244  if (config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE) {
245  for (auto iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++)
246  if (config->GetViscous_Wall(iMarker)) {
247  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
248  for (auto iVertex = 0u; iVertex < geometry->nVertex[iMarker]; iVertex++) {
249  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
250 
251  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
252 
253  if (geometry->nodes->GetDomain(iPoint)) {
254  const auto jPoint = geometry->vertex[iMarker][iVertex]->GetNormal_Neighbor();
255 
256  su2double shearStress = 0.0;
257  for(auto iDim = 0u; iDim < nDim; iDim++) {
258  shearStress += pow(solver_container[FLOW_SOL]->GetCSkinFriction(iMarker, iVertex, iDim), 2.0);
259  }
260  shearStress = sqrt(shearStress);
261 
262  const su2double FrictionVelocity = sqrt(shearStress/flowNodes->GetDensity(iPoint));
263  const su2double wall_dist = geometry->nodes->GetWall_Distance(jPoint);
264  const su2double Derivative = flowNodes->GetLaminarViscosity(jPoint) * pow(nodes->GetSolution(jPoint, 0), 0.673) / wall_dist;
265  const su2double turbulence_index = 6.1 * Derivative / pow(FrictionVelocity, 2.346);
266 
267  nodes->SetTurbIndex(iPoint, turbulence_index);
268 
269  }
270  }
271  END_SU2_OMP_FOR
272  }
273  }

275 
276  AD::EndNoSharedReading();

293 void CTurbSSTSolver::Source_Residual(CGeometry *geometry, CSolver **solver_container,
294  CNumerics **numerics_container, CConfig *config, unsigned short iMesh) {
295 
296  bool axisymmetric = config->GetAxisymmetric();
297 
298  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
299 
300  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
301 
302  /*--- Pick one numerics object per thread. ---*/
303  auto* numerics = numerics_container[SOURCE_FIRST_TERM + omp_get_thread_num()*MAX_TERMS];
304 
305  /*--- Loop over all points. ---*/
306 
307  AD::StartNoSharedReading();
308 
309  SU2_OMP_FOR_DYN(omp_chunk_size)
310  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
311 
312  /*--- Conservative variables w/o reconstruction ---*/
313 
314  numerics->SetPrimitive(flowNodes->GetPrimitive(iPoint), nullptr);
315 
316  /*--- Gradient of the primitive and conservative variables ---*/
317 
318  numerics->SetPrimVarGradient(flowNodes->GetGradient_Primitive(iPoint), nullptr);
319 
320  /*--- Turbulent variables w/o reconstruction, and its gradient ---*/
321 
322  numerics->SetScalarVar(nodes->GetSolution(iPoint), nullptr);
323  numerics->SetScalarVarGradient(nodes->GetGradient(iPoint), nullptr);
324 
325  /*--- Set volume ---*/
326 
327  numerics->SetVolume(geometry->nodes->GetVolume(iPoint));
328 
329  /*--- Set distance to the surface ---*/
330 
331  numerics->SetDistance(geometry->nodes->GetWall_Distance(iPoint), 0.0);
332 
333  /*--- Menter's first blending function ---*/
334 
335  numerics->SetF1blending(nodes->GetF1blending(iPoint),0.0);
336 
337  /*--- Menter's second blending function ---*/
338 
339  numerics->SetF2blending(nodes->GetF2blending(iPoint));
340 
341  /*--- Set vorticity and strain rate magnitude ---*/
342 
343  numerics->SetVorticity(flowNodes->GetVorticity(iPoint), nullptr);
344 
345  numerics->SetStrainMag(flowNodes->GetStrainMag(iPoint), 0.0);
346 
347  /*--- Cross diffusion ---*/
348 
349  numerics->SetCrossDiff(nodes->GetCrossDiff(iPoint));
350 
351  /*--- Effective Intermittency ---*/
352  if (config->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
353  numerics->SetIntermittencyEff(solver_container[TRANS_SOL]->GetNodes()->GetIntermittencyEff(iPoint));
354  }
355 
356  if (axisymmetric){
357  /*--- Set y coordinate ---*/
358  numerics->SetCoord(geometry->nodes->GetCoord(iPoint), geometry->nodes->GetCoord(iPoint));
359  }
360 
361  /*--- Compute the source term ---*/
362 
363  auto residual = numerics->ComputeResidual(config);
364 
365  /*--- Store the intermittency ---*/
366 
367  if (config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE) {
368  nodes->SetIntermittency(iPoint, numerics->GetIntermittencyEff());
369  }
370 
371  /*--- Subtract residual and the Jacobian ---*/
372 
373  LinSysRes.SubtractBlock(iPoint, residual);
374  if (implicit) Jacobian.SubtractBlock2Diag(iPoint, residual.jacobian_i);
375 
376  }
377  END_SU2_OMP_FOR
378 
379  AD::EndNoSharedReading();
380 
381 }

387 void CTurbSSTSolver::BC_HeatFlux_Wall(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
388  CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
389 
390  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
391 
392  bool rough_wall = false;
393  string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
394  WALL_TYPE WallType; su2double Roughness_Height;
395  tie(WallType, Roughness_Height) = config->GetWallRoughnessProperties(Marker_Tag);
396  if (WallType == WALL_TYPE::ROUGH) rough_wall = true;
397 
398  /*--- Evaluate nu tilde at the closest point to the surface using the wall functions. ---*/
399 
400  if (config->GetWall_Functions()) {
401  SU2_OMP_SAFE_GLOBAL_ACCESS(SetTurbVars_WF(geometry, solver_container, config, val_marker);)
402  return;
403  }
404 
405  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
406  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
407 
408  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
409 
410  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
411  if (geometry->nodes->GetDomain(iPoint)) {
412 
413  if (rough_wall) {
414 
415  /*--- Set wall values ---*/
416  su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
417  su2double laminar_viscosity = solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
418  su2double WallShearStress = solver_container[FLOW_SOL]->GetWallShearStress(val_marker, iVertex);
419 
420  /*--- Compute non-dimensional velocity ---*/
421  su2double FrictionVel = sqrt(fabs(WallShearStress)/density);
422 
423  /*--- Compute roughness in wall units. ---*/
424  //su2double Roughness_Height = config->GetWall_RoughnessHeight(Marker_Tag);
425  su2double kPlus = FrictionVel*Roughness_Height*density/laminar_viscosity;
426 
427  su2double S_R= 0.0;
428  /*--- Reference 1 original Wilcox (1998) ---*/
429  /*if (kPlus <= 25)
430  S_R = (50/(kPlus+EPS))*(50/(kPlus+EPS));
431  else
432  S_R = 100/(kPlus+EPS);*/
433 
434  /*--- Reference 2 from D.C. Wilcox Turbulence Modeling for CFD (2006) ---*/
435  if (kPlus <= 5)
436  S_R = (200/(kPlus+EPS))*(200/(kPlus+EPS));
437  else
438  S_R = 100/(kPlus+EPS) + ((200/(kPlus+EPS))*(200/(kPlus+EPS)) - 100/(kPlus+EPS))*exp(5-kPlus);
439 
440  /*--- Modify the omega to account for a rough wall. ---*/
441  su2double solution[2];
442  solution[0] = 0.0;
443  solution[1] = FrictionVel*FrictionVel*S_R/(laminar_viscosity/density);
444 
445  /*--- Set the solution values and zero the residual ---*/
446  nodes->SetSolution_Old(iPoint,solution);
447  nodes->SetSolution(iPoint,solution);
448  LinSysRes.SetBlock_Zero(iPoint);
449 
450  } else { // smooth wall
451 
452  /*--- distance to closest neighbor ---*/
453  const auto jPoint = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
454 
455  su2double distance2 = GeometryToolbox::SquaredDistance(nDim,
456  geometry->nodes->GetCoord(iPoint),
457  geometry->nodes->GetCoord(jPoint));
458  /*--- Set wall values ---*/
459 
460  su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(jPoint);
461  su2double laminar_viscosity = solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(jPoint);
462 
463  su2double beta_1 = constants[4];
464  su2double solution[MAXNVAR];
465  solution[0] = 0.0;
466  solution[1] = 60.0*laminar_viscosity/(density*beta_1*distance2);
467 
468  /*--- Set the solution values and zero the residual ---*/
469  nodes->SetSolution_Old(iPoint,solution);
470  nodes->SetSolution(iPoint,solution);
471  LinSysRes.SetBlock_Zero(iPoint);
472  }
473 
474  if (implicit) {
475  /*--- Change rows of the Jacobian (includes 1 in the diagonal) ---*/
476  Jacobian.DeleteValsRowi(iPoint*nVar);
477  Jacobian.DeleteValsRowi(iPoint*nVar+1);
478  }
479  }
480  }
481  END_SU2_OMP_FOR
482 }

485 void CTurbSSTSolver::SetTurbVars_WF(CGeometry *geometry, CSolver **solver_container,
486  const CConfig *config, unsigned short val_marker) {
487 
488  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
489 
490  /*--- von Karman constant from boundary layer theory ---*/
491  const su2double kappa = config->GetwallModel_Kappa();
492  const su2double minYPlus = config->GetwallModel_MinYPlus();
493  /*--- relaxation factor for k-omega values ---*/
494  const su2double relax = config->GetwallModel_RelFac();
495 
496  /*--- Loop over all of the vertices on this boundary marker ---*/
497 
498  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
499 
500  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
501  const auto iPoint_Neighbor = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
502  if (!geometry->nodes->GetDomain(iPoint_Neighbor)) continue;
503 
504  su2double Y_Plus = solver_container[FLOW_SOL]->GetYPlus(val_marker, iVertex);
505  su2double Lam_Visc_Wall = solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
506 
507  /*--- Do not use wall model at the ipoint when y+ < "limit", use zero flux (Neumann) conditions. ---*/
508 
509  if (Y_Plus < minYPlus) {
510  /* --- Use zero flux (Neumann) conditions, i.e. nothing has to be done. --- */
511  continue;
512  }
513 
514  su2double Eddy_Visc = solver_container[FLOW_SOL]->GetEddyViscWall(val_marker, iVertex);
515  su2double k = nodes->GetSolution(iPoint_Neighbor,0);
516  su2double omega = nodes->GetSolution(iPoint_Neighbor,1);
517  su2double Density_Wall = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
518  su2double U_Tau = solver_container[FLOW_SOL]->GetUTau(val_marker, iVertex);
519  su2double y = Y_Plus*Lam_Visc_Wall/(Density_Wall*U_Tau);
520 
521  su2double omega1 = 6.0*Lam_Visc_Wall/(0.075*Density_Wall*y*y); // eq. 19
522  su2double omega0 = U_Tau/(sqrt(0.09)*kappa*y); // eq. 20
523  su2double omega_new = sqrt(omega0*omega0 + omega1*omega1); // eq. 21 Nichols & Nelson
524  su2double k_new = omega_new * Eddy_Visc/Density_Wall; // eq. 22 Nichols & Nelson
525  // (is this the correct density? paper says rho and not rho_w)
526 
527  /*--- put some relaxation factor on the k-omega values ---*/
528  k += relax*(k_new - k);
529  omega += relax*(omega_new - omega);
530 
531  su2double solution[MAXNVAR] = {k, omega};
532 
533  nodes->SetSolution_Old(iPoint_Neighbor,solution);
534  nodes->SetSolution(iPoint,solution);
535 
536  LinSysRes.SetBlock_Zero(iPoint_Neighbor);
537 
538  if (implicit) {
539  /*--- includes 1 in the diagonal ---*/
540  Jacobian.DeleteValsRowi(iPoint_Neighbor*nVar);
541  Jacobian.DeleteValsRowi(iPoint_Neighbor*nVar+1);
542  }
543  }
