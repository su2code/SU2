34 void CSingleGridIntegration::SingleGrid_Iteration(CGeometry ****geometry, CSolver *****solver_container,
35  CNumerics ******numerics_container, CConfig **config,
36  unsigned short RunTime_EqSystem, unsigned short iZone,
37  unsigned short iInst) {
38 
39  const unsigned short Solver_Position = config[iZone]->GetContainerPosition(RunTime_EqSystem);
40 
41  /*--- Start an OpenMP parallel region covering the entire iteration. ---*/
42 
43  SU2_OMP_PARALLEL_(if(solver_container[iZone][iInst][MESH_0][Solver_Position]->GetHasHybridParallel()))
44  {
45 
46  unsigned short FinestMesh = config[iZone]->GetFinestMesh();
47 
48  CGeometry* geometry_fine = geometry[iZone][iInst][FinestMesh];
49  CSolver** solvers_fine = solver_container[iZone][iInst][FinestMesh];
50 
51  /*--- Preprocessing ---*/
52 
53  solvers_fine[Solver_Position]->Preprocessing(geometry_fine, solvers_fine, config[iZone],
54  FinestMesh, 0, RunTime_EqSystem, false);
55 
56  /*--- Set the old solution ---*/
57 
58  solvers_fine[Solver_Position]->Set_OldSolution();
59 
60  /*--- Time step evaluation ---*/
61 
62  solvers_fine[Solver_Position]->SetTime_Step(geometry_fine, solvers_fine, config[iZone],
63  FinestMesh, config[iZone]->GetTimeIter());
64 
65  /*--- Space integration ---*/
66 
67  Space_Integration(geometry_fine, solvers_fine,
68  numerics_container[iZone][iInst][FinestMesh][Solver_Position],
69  config[iZone], FinestMesh, NO_RK_ITER, RunTime_EqSystem);
70 
71  /*--- Time integration ---*/
72 
73  Time_Integration(geometry_fine, solvers_fine, config[iZone], NO_RK_ITER, RunTime_EqSystem);
74 
75  /*--- Postprocessing ---*/
76 
77  solvers_fine[Solver_Position]->Postprocessing(geometry_fine, solvers_fine, config[iZone], FinestMesh);
78 
79  if (RunTime_EqSystem == RUNTIME_HEAT_SYS) {
80  SU2_OMP_SAFE_GLOBAL_ACCESS(solvers_fine[HEAT_SOL]->Heat_Fluxes(geometry_fine, solvers_fine, config[iZone]);)
81  }
82 
83  /*--- If turbulence model, copy the turbulence variables to the coarse levels ---*/
84 
85  if (RunTime_EqSystem == RUNTIME_TURB_SYS) {
86 
87  for (unsigned short iMesh = FinestMesh; iMesh < config[iZone]->GetnMGLevels(); iMesh++) {
88 
89  SetRestricted_Solution(RunTime_EqSystem,
90  solver_container[iZone][iInst][iMesh][Solver_Position],
91  solver_container[iZone][iInst][iMesh+1][Solver_Position],
92  geometry[iZone][iInst][iMesh],
93  geometry[iZone][iInst][iMesh+1],
94  config[iZone]);
95 
96  SetRestricted_EddyVisc(RunTime_EqSystem,
97  solver_container[iZone][iInst][iMesh][Solver_Position],
98  solver_container[iZone][iInst][iMesh+1][Solver_Position],
99  geometry[iZone][iInst][iMesh],
100  geometry[iZone][iInst][iMesh+1],
101  config[iZone]);
102  }
103 
104  }
105 
106  }
107  END_SU2_OMP_PARALLEL
108 }

118 void CSingleGridIntegration::SetRestricted_EddyVisc(unsigned short RunTime_EqSystem, CSolver *sol_fine, CSolver *sol_coarse,
119  CGeometry *geo_fine, CGeometry *geo_coarse, CConfig *config) {
120 
121  unsigned long iVertex, Point_Fine, Point_Coarse;
122  unsigned short iMarker, iChildren;
123  su2double Area_Parent, Area_Children, EddyVisc_Fine, EddyVisc;
124 
125  /*--- Compute coarse Eddy Viscosity from fine solution ---*/
126 
127  SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPointDomain(), omp_get_num_threads()))
128  for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPointDomain(); Point_Coarse++) {
129 
130  Area_Parent = geo_coarse->nodes->GetVolume(Point_Coarse);
131 
132  EddyVisc = 0.0;
133 
134  for (iChildren = 0; iChildren < geo_coarse->nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
135  Point_Fine = geo_coarse->nodes->GetChildren_CV(Point_Coarse, iChildren);
136  Area_Children = geo_fine->nodes->GetVolume(Point_Fine);
137  EddyVisc_Fine = sol_fine->GetNodes()->GetmuT(Point_Fine);
138  EddyVisc += EddyVisc_Fine*Area_Children/Area_Parent;
139  }
140 
141  sol_coarse->GetNodes()->SetmuT(Point_Coarse,EddyVisc);
142 
143  }
144  END_SU2_OMP_FOR
145 
146  /*--- Update solution at the no slip wall boundary, only the first
147  variable (nu_tilde -in SA and SA_NEG- and k -in SST-), to guarantee that the eddy viscoisty
148  is zero on the surface ---*/
149 
150  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
151  if (config->GetViscous_Wall(iMarker)) {
152  SU2_OMP_FOR_STAT(32)
153  for (iVertex = 0; iVertex < geo_coarse->nVertex[iMarker]; iVertex++) {
154  Point_Coarse = geo_coarse->vertex[iMarker][iVertex]->GetNode();
155  sol_coarse->GetNodes()->SetmuT(Point_Coarse,0.0);
156  }
157  END_SU2_OMP_FOR
158  }
159  }
160 
161  /*--- MPI the new interpolated solution (this also includes the eddy viscosity) ---*/
162 
163  sol_coarse->InitiateComms(geo_coarse, config, SOLUTION_EDDY);
164  sol_coarse->CompleteComms(geo_coarse, config, SOLUTION_EDDY);
165 
166 }
