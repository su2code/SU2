56 void CTurbSSTVariable::SetBlendingFunc(unsigned long iPoint, su2double val_viscosity,
57  su2double val_dist, su2double val_density, TURB_TRANS_MODEL trans_model) {
58  su2double arg2, arg2A, arg2B, arg1;
59 
60  AD::StartPreacc();
61  AD::SetPreaccIn(val_viscosity); AD::SetPreaccIn(val_dist);
62  AD::SetPreaccIn(val_density);
63  AD::SetPreaccIn(Solution[iPoint], nVar);
64  AD::SetPreaccIn(Gradient[iPoint], nVar, nDim);
65 
66  /*--- Cross diffusion ---*/
67 
68  CDkw(iPoint) = 0.0;
69  for (unsigned long iDim = 0; iDim < nDim; iDim++)
70  CDkw(iPoint) += Gradient(iPoint,0,iDim)*Gradient(iPoint,1,iDim);
71  CDkw(iPoint) *= 2.0*val_density*sigma_om2/Solution(iPoint,1);
72  CDkw(iPoint) = max(CDkw(iPoint), pow(10.0, -prod_lim_const));
73 
74  /*--- F1 ---*/
75 
76  arg2A = sqrt(Solution(iPoint,0))/(beta_star*Solution(iPoint,1)*val_dist+EPS*EPS);
77  arg2B = 500.0*val_viscosity / (val_density*val_dist*val_dist*Solution(iPoint,1)+EPS*EPS);
78  arg2 = max(arg2A, arg2B);
79  arg1 = min(arg2, 4.0*val_density*sigma_om2*Solution(iPoint,0) / (CDkw(iPoint)*val_dist*val_dist+EPS*EPS));
80  F1(iPoint) = tanh(pow(arg1, 4.0));
81 
82  /*--- F2 ---*/
83 
84  arg2 = max(2.0*arg2A, arg2B);
85  F2(iPoint) = tanh(pow(arg2, 2.0));
86 
87  /*--- LM model for F1 ---*/
88  if (trans_model == TURB_TRANS_MODEL::LM) {
89  su2double Ry = val_density*val_dist*sqrt(Solution(iPoint,0))/val_viscosity;
90  su2double F3 = exp(-pow(Ry/120.0, 8.0));
91  F1(iPoint) = max(F1(iPoint), F3);
92  }
93 
94  AD::SetPreaccOut(F1(iPoint)); AD::SetPreaccOut(F2(iPoint)); AD::SetPreaccOut(CDkw(iPoint));
95  AD::EndPreacc();
96 
97 }
