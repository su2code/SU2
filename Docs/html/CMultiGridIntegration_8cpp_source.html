34 void CMultiGridIntegration::MultiGrid_Iteration(CGeometry ****geometry,
35  CSolver *****solver_container,
36  CNumerics ******numerics_container,
37  CConfig **config,
38  unsigned short RunTime_EqSystem,
39  unsigned short iZone,
40  unsigned short iInst) {
41 
42  bool direct;
43  switch (config[iZone]->GetKind_Solver()) {
44  case MAIN_SOLVER::EULER:
45  case MAIN_SOLVER::NAVIER_STOKES:
46  case MAIN_SOLVER::NEMO_EULER:
47  case MAIN_SOLVER::NEMO_NAVIER_STOKES:
48  case MAIN_SOLVER::RANS:
49  case MAIN_SOLVER::FEM_EULER:
50  case MAIN_SOLVER::FEM_NAVIER_STOKES:
51  case MAIN_SOLVER::FEM_RANS:
52  case MAIN_SOLVER::FEM_LES:
53  case MAIN_SOLVER::DISC_ADJ_EULER:
54  case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES:
55  case MAIN_SOLVER::DISC_ADJ_FEM_EULER:
56  case MAIN_SOLVER::DISC_ADJ_FEM_NS:
57  case MAIN_SOLVER::DISC_ADJ_RANS:
58  direct = true;
59  break;
60  default:
61  direct = false;
62  break;
63  }
64 
65  const unsigned short Solver_Position = config[iZone]->GetContainerPosition(RunTime_EqSystem);
66 
67  /*--- Start an OpenMP parallel region covering the entire MG iteration, if the solver supports it. ---*/
68 
69  SU2_OMP_PARALLEL_(if(solver_container[iZone][iInst][MESH_0][Solver_Position]->GetHasHybridParallel()))
70  {
71 
72  su2double monitor = 1.0;
73  bool FullMG = false;
74 
75  unsigned short RecursiveParam = config[iZone]->GetMGCycle();
76 
77  if (config[iZone]->GetMGCycle() == FULLMG_CYCLE) {
78  RecursiveParam = V_CYCLE;
79  FullMG = true;
80  }
81 
82  /*--- Full multigrid strategy and start up with fine grid only works with the direct problem ---*/
83 
84  unsigned short FinestMesh = config[iZone]->GetFinestMesh();
85 
86  if (!config[iZone]->GetRestart() && FullMG && direct && ( Convergence_FullMG && (FinestMesh != MESH_0 ))) {
87 
88  SetProlongated_Solution(RunTime_EqSystem,
89  solver_container[iZone][iInst][FinestMesh-1][Solver_Position],
90  solver_container[iZone][iInst][FinestMesh][Solver_Position],
91  geometry[iZone][iInst][FinestMesh-1],
92  geometry[iZone][iInst][FinestMesh],
93  config[iZone]);
94 
95  SU2_OMP_SAFE_GLOBAL_ACCESS(config[iZone]->SubtractFinestMesh();)
96  }
97 
98  /*--- Set the current finest grid (full multigrid strategy) ---*/
99 
100  FinestMesh = config[iZone]->GetFinestMesh();
101 
102  /*--- Perform the Full Approximation Scheme multigrid ---*/
103 
104  MultiGrid_Cycle(geometry, solver_container, numerics_container, config,
105  FinestMesh, RecursiveParam, RunTime_EqSystem, iZone, iInst);
106 
107 
108  /*--- Computes primitive variables and gradients in the finest mesh (useful for the next solver (turbulence) and output ---*/
109 
110  solver_container[iZone][iInst][MESH_0][Solver_Position]->Preprocessing(geometry[iZone][iInst][MESH_0],
111  solver_container[iZone][iInst][MESH_0],
112  config[iZone], MESH_0, NO_RK_ITER,
113  RunTime_EqSystem, true);
114 
115  /*--- Compute non-dimensional parameters and the convergence monitor ---*/
116 
117  NonDimensional_Parameters(geometry[iZone][iInst], solver_container[iZone][iInst],
118  numerics_container[iZone][iInst], config[iZone],
119  FinestMesh, RunTime_EqSystem, &monitor);
120 
121  }
122  END_SU2_OMP_PARALLEL
123 
124 }

126 void CMultiGridIntegration::MultiGrid_Cycle(CGeometry ****geometry,
127  CSolver *****solver_container,
128  CNumerics ******numerics_container,
129  CConfig **config_container,
130  unsigned short iMesh,
131  unsigned short RecursiveParam,
132  unsigned short RunTime_EqSystem,
133  unsigned short iZone,
134  unsigned short iInst) {
135 
136  CConfig* config = config_container[iZone];
137 
138  const unsigned short Solver_Position = config->GetContainerPosition(RunTime_EqSystem);
139  const bool classical_rk4 = (config->GetKind_TimeIntScheme() == CLASSICAL_RK4_EXPLICIT);
140  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
141 
142  /*--- Shorter names to refer to fine grid entities. ---*/
143 
144  CGeometry* geometry_fine = geometry[iZone][iInst][iMesh];
145  CSolver** solver_container_fine = solver_container[iZone][iInst][iMesh];
146  CSolver* solver_fine = solver_container_fine[Solver_Position];
147  CNumerics** numerics_fine = numerics_container[iZone][iInst][iMesh][Solver_Position];
148 
149  /*--- Number of RK steps. ---*/
150 
151  unsigned short iRKLimit = 1;
152 
153  switch (config->GetKind_TimeIntScheme()) {
154  case RUNGE_KUTTA_EXPLICIT:
155  iRKLimit = config->GetnRKStep();
156  break;
157  case CLASSICAL_RK4_EXPLICIT:
158  iRKLimit = 4;
159  break;
160  case EULER_EXPLICIT:
161  case EULER_IMPLICIT:
162  iRKLimit = 1;
163  break;
164  }
165 
166  /*--- Do a presmoothing on the grid iMesh to be restricted to the grid iMesh+1 ---*/
167 
168  for (unsigned short iPreSmooth = 0; iPreSmooth < config->GetMG_PreSmooth(iMesh); iPreSmooth++) {
169 
170  /*--- Time and space integration ---*/
171 
172  for (unsigned short iRKStep = 0; iRKStep < iRKLimit; iRKStep++) {
173 
174  /*--- Send-Receive boundary conditions, and preprocessing ---*/
175 
176  solver_fine->Preprocessing(geometry_fine, solver_container_fine, config, iMesh, iRKStep, RunTime_EqSystem, false);
177 
178 
179  if (iRKStep == 0) {
180 
181  /*--- Set the old solution ---*/
182 
183  solver_fine->Set_OldSolution();
184 
185  if (classical_rk4) solver_fine->Set_NewSolution();
186 
187  /*--- Compute time step, max eigenvalue, and integration scheme (steady and unsteady problems) ---*/
188 
189  solver_fine->SetTime_Step(geometry_fine, solver_container_fine, config, iMesh, config->GetTimeIter());
190 
191  /*--- Restrict the solution and gradient for the adjoint problem ---*/
192 
193  Adjoint_Setup(geometry, solver_container, config_container, RunTime_EqSystem, config->GetTimeIter(), iZone);
194 
195  }
196 
197  /*--- Space integration ---*/
198 
199  Space_Integration(geometry_fine, solver_container_fine, numerics_fine, config, iMesh, iRKStep, RunTime_EqSystem);
200 
201  /*--- Time integration, update solution using the old solution plus the solution increment ---*/
202 
203  Time_Integration(geometry_fine, solver_container_fine, config, iRKStep, RunTime_EqSystem);
204 
205  /*--- Send-Receive boundary conditions, and postprocessing ---*/
206 
207  solver_fine->Postprocessing(geometry_fine, solver_container_fine, config, iMesh);
208 
209  }
210 
211  }
212 
213  /*--- Compute Forcing Term $P_(k+1) = I^(k+1)_k(P_k+F_k(u_k))-F_(k+1)(I^(k+1)_k u_k)$ and update solution for multigrid ---*/
214 
215  if ( iMesh < config->GetnMGLevels() ) {
216 
217  /*--- Shorter names to refer to coarse grid entities. ---*/
218 
219  CGeometry* geometry_coarse = geometry[iZone][iInst][iMesh+1];
220  CSolver** solver_container_coarse = solver_container[iZone][iInst][iMesh+1];
221  CSolver* solver_coarse = solver_container_coarse[Solver_Position];
222  CNumerics** numerics_coarse = numerics_container[iZone][iInst][iMesh+1][Solver_Position];
223 
224  /*--- Temporarily disable implicit integration, for what follows we do not need the Jacobian. ---*/
225 
226  if (implicit) {
227  SU2_OMP_SAFE_GLOBAL_ACCESS(config->SetKind_TimeIntScheme(EULER_EXPLICIT);)
228  }
229 
230  /*--- Compute $r_k = P_k + F_k(u_k)$ ---*/
231 
232  solver_fine->Preprocessing(geometry_fine, solver_container_fine, config, iMesh, NO_RK_ITER, RunTime_EqSystem, false);
233 
234  Space_Integration(geometry_fine, solver_container_fine, numerics_fine, config, iMesh, NO_RK_ITER, RunTime_EqSystem);
235 
236  SetResidual_Term(geometry_fine, solver_fine);
237 
238  /*--- Compute $r_(k+1) = F_(k+1)(I^(k+1)_k u_k)$ ---*/
239 
240  SetRestricted_Solution(RunTime_EqSystem, solver_fine, solver_coarse, geometry_fine, geometry_coarse, config);
241 
242  solver_coarse->Preprocessing(geometry_coarse, solver_container_coarse, config, iMesh+1, NO_RK_ITER, RunTime_EqSystem, false);
243 
244  Space_Integration(geometry_coarse, solver_container_coarse, numerics_coarse, config, iMesh+1, NO_RK_ITER, RunTime_EqSystem);
245 
246  /*--- Compute $P_(k+1) = I^(k+1)_k(r_k) - r_(k+1) ---*/
247 
248  SetForcing_Term(solver_fine, solver_coarse, geometry_fine, geometry_coarse, config, iMesh+1);
249 
250  /*--- Restore the time integration settings. ---*/
251 
252  if (implicit) {
253  SU2_OMP_SAFE_GLOBAL_ACCESS(config->SetKind_TimeIntScheme(EULER_IMPLICIT);)
254  }
255 
256  /*--- Recursive call to MultiGrid_Cycle (this routine). ---*/
257 
258  for (unsigned short imu = 0; imu <= RecursiveParam; imu++) {
259 
260  unsigned short nextRecurseParam = RecursiveParam;
261  if (iMesh == config->GetnMGLevels()-2)
262  nextRecurseParam = 0;
263 
264  MultiGrid_Cycle(geometry, solver_container, numerics_container, config_container,
265  iMesh+1, nextRecurseParam, RunTime_EqSystem, iZone, iInst);
266  }
267 
268  /*--- Compute prolongated solution, and smooth the correction $u^(new)_k = u_k + Smooth(I^k_(k+1)(u_(k+1)-I^(k+1)_k u_k))$ ---*/
269 
270  GetProlongated_Correction(RunTime_EqSystem, solver_fine, solver_coarse, geometry_fine, geometry_coarse, config);
271 
272  SmoothProlongated_Correction(RunTime_EqSystem, solver_fine, geometry_fine, config->GetMG_CorrecSmooth(iMesh), 1.25, config);
273 
274  SetProlongated_Correction(solver_fine, geometry_fine, config, iMesh);
275 
276 
277  /*--- Solution post-smoothing in the prolongated grid. ---*/
278 
279  for (unsigned short iPostSmooth = 0; iPostSmooth < config->GetMG_PostSmooth(iMesh); iPostSmooth++) {
280 
281  for (unsigned short iRKStep = 0; iRKStep < iRKLimit; iRKStep++) {
282 
283  solver_fine->Preprocessing(geometry_fine, solver_container_fine, config, iMesh, iRKStep, RunTime_EqSystem, false);
284 
285  if (iRKStep == 0) {
286  solver_fine->Set_OldSolution();
287 
288  if (classical_rk4) solver_fine->Set_NewSolution();
289 
290  solver_fine->SetTime_Step(geometry_fine, solver_container_fine, config, iMesh, config->GetTimeIter());
291  }
292 
293  Space_Integration(geometry_fine, solver_container_fine, numerics_fine, config, iMesh, iRKStep, RunTime_EqSystem);
294 
295  Time_Integration(geometry_fine, solver_container_fine, config, iRKStep, RunTime_EqSystem);
296 
297  solver_fine->Postprocessing(geometry_fine, solver_container_fine, config, iMesh);
298 
299  }
300  }
301  }
302 
303 }

305 void CMultiGridIntegration::GetProlongated_Correction(unsigned short RunTime_EqSystem, CSolver *sol_fine, CSolver *sol_coarse,
306  CGeometry *geo_fine, CGeometry *geo_coarse, CConfig *config) {
307  unsigned long Point_Fine, Point_Coarse, iVertex;
308  unsigned short iMarker, iChildren, iVar;
309  su2double Area_Parent, Area_Children;
310  const su2double *Solution_Fine = nullptr, *Solution_Coarse = nullptr;
311 
312  const unsigned short nVar = sol_coarse->GetnVar();
313 
314  su2double *Solution = new su2double[nVar];
315 
316  SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPointDomain(), omp_get_num_threads()))
317  for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPointDomain(); Point_Coarse++) {
318 
319  Area_Parent = geo_coarse->nodes->GetVolume(Point_Coarse);
320 
321  for (iVar = 0; iVar < nVar; iVar++) Solution[iVar] = 0.0;
322 
323  for (iChildren = 0; iChildren < geo_coarse->nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
324  Point_Fine = geo_coarse->nodes->GetChildren_CV(Point_Coarse, iChildren);
325  Area_Children = geo_fine->nodes->GetVolume(Point_Fine);
326  Solution_Fine = sol_fine->GetNodes()->GetSolution(Point_Fine);
327  for (iVar = 0; iVar < nVar; iVar++)
328  Solution[iVar] -= Solution_Fine[iVar]*Area_Children/Area_Parent;
329  }
330 
331  Solution_Coarse = sol_coarse->GetNodes()->GetSolution(Point_Coarse);
332 
333  for (iVar = 0; iVar < nVar; iVar++)
334  Solution[iVar] += Solution_Coarse[iVar];
335 
336  for (iVar = 0; iVar < nVar; iVar++)
337  sol_coarse->GetNodes()->SetSolution_Old(Point_Coarse,Solution);
338  }
339  END_SU2_OMP_FOR
340 
341  delete [] Solution;
342 
343  /*--- Remove any contributions from no-slip walls. ---*/
344 
345  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
346  if (config->GetViscous_Wall(iMarker)) {
347 
348  SU2_OMP_FOR_STAT(32)
349  for (iVertex = 0; iVertex < geo_coarse->nVertex[iMarker]; iVertex++) {
350 
351  Point_Coarse = geo_coarse->vertex[iMarker][iVertex]->GetNode();
352 
353  /*--- For dirichlet boundary condtions, set the correction to zero.
354  Note that Solution_Old stores the correction not the actual value ---*/
355 
356  su2double zero[3] = {0.0};
357  sol_coarse->GetNodes()->SetVelocity_Old(Point_Coarse, zero);
358 
359  }
360  END_SU2_OMP_FOR
361  }
362  }
363 
364  /*--- MPI the set solution old ---*/
365 
366  sol_coarse->InitiateComms(geo_coarse, config, SOLUTION_OLD);
367  sol_coarse->CompleteComms(geo_coarse, config, SOLUTION_OLD);
368 
369  SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPointDomain(), omp_get_num_threads()))
370  for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPointDomain(); Point_Coarse++) {
371  for (iChildren = 0; iChildren < geo_coarse->nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
372  Point_Fine = geo_coarse->nodes->GetChildren_CV(Point_Coarse, iChildren);
373  sol_fine->LinSysRes.SetBlock(Point_Fine, sol_coarse->GetNodes()->GetSolution_Old(Point_Coarse));
374  }
375  }
376  END_SU2_OMP_FOR
377 
378 }

380 void CMultiGridIntegration::SmoothProlongated_Correction(unsigned short RunTime_EqSystem, CSolver *solver, CGeometry *geometry,
381  unsigned short val_nSmooth, su2double val_smooth_coeff, CConfig *config) {
382 
383  /*--- Check if there is work to do. ---*/
384  if (val_nSmooth == 0) return;
385 
386  const su2double *Residual_Old, *Residual_Sum, *Residual_j;
387  unsigned short iVar, iSmooth, iMarker, iNeigh;
388  unsigned long iPoint, jPoint, iVertex;
389 
390  const unsigned short nVar = solver->GetnVar();
391 
392  SU2_OMP_FOR_STAT(roundUpDiv(geometry->GetnPoint(), omp_get_num_threads()))
393  for (iPoint = 0; iPoint < geometry->GetnPoint(); iPoint++) {
394  Residual_Old = solver->LinSysRes.GetBlock(iPoint);
395  solver->GetNodes()->SetResidual_Old(iPoint,Residual_Old);
396  }
397  END_SU2_OMP_FOR
398 
399  /*--- Jacobi iterations. ---*/
400 
401  for (iSmooth = 0; iSmooth < val_nSmooth; iSmooth++) {
402 
403  /*--- Loop over all mesh points (sum the residuals of direct neighbors). ---*/
404 
405  SU2_OMP_FOR_STAT(roundUpDiv(geometry->GetnPoint(), omp_get_num_threads()))
406  for (iPoint = 0; iPoint < geometry->GetnPoint(); ++iPoint) {
407 
408  solver->GetNodes()->SetResidualSumZero(iPoint);
409 
410  for (iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); ++iNeigh) {
411  jPoint = geometry->nodes->GetPoint(iPoint, iNeigh);
412  Residual_j = solver->LinSysRes.GetBlock(jPoint);
413  solver->GetNodes()->AddResidual_Sum(iPoint, Residual_j);
414  }
415 
416  }
417  END_SU2_OMP_FOR
418 
419  /*--- Loop over all mesh points (update residuals with the neighbor averages). ---*/
420 
421  SU2_OMP_FOR_STAT(roundUpDiv(geometry->GetnPoint(), omp_get_num_threads()))
422  for (iPoint = 0; iPoint < geometry->GetnPoint(); ++iPoint) {
423 
424  su2double factor = 1.0/(1.0+val_smooth_coeff*su2double(geometry->nodes->GetnPoint(iPoint)));
425 
426  Residual_Sum = solver->GetNodes()->GetResidual_Sum(iPoint);
427  Residual_Old = solver->GetNodes()->GetResidual_Old(iPoint);
428 
429  for (iVar = 0; iVar < nVar; iVar++)
430  solver->LinSysRes(iPoint,iVar) = (Residual_Old[iVar] + val_smooth_coeff*Residual_Sum[iVar])*factor;
431  }
432  END_SU2_OMP_FOR
433 
434  /*--- Restore original residuals (without average) at boundary points. ---*/
435 
436  for (iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
437  if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
438  (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
439  (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
440 
441  SU2_OMP_FOR_STAT(32)
442  for (iVertex = 0; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
443  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
444  Residual_Old = solver->GetNodes()->GetResidual_Old(iPoint);
445  solver->LinSysRes.SetBlock(iPoint, Residual_Old);
446  }
447  END_SU2_OMP_FOR
448  }
449  }
450 
451  }

455 void CMultiGridIntegration::SetProlongated_Correction(CSolver *sol_fine, CGeometry *geo_fine,
456  CConfig *config, unsigned short iMesh) {
457  unsigned long Point_Fine;
458  unsigned short iVar;
459  su2double *Solution_Fine, *Residual_Fine;
460 
461  const unsigned short nVar = sol_fine->GetnVar();
462  const su2double factor = config->GetDamp_Correc_Prolong();
463 
464  SU2_OMP_FOR_STAT(roundUpDiv(geo_fine->GetnPointDomain(), omp_get_num_threads()))
465  for (Point_Fine = 0; Point_Fine < geo_fine->GetnPointDomain(); Point_Fine++) {
466  Residual_Fine = sol_fine->LinSysRes.GetBlock(Point_Fine);
467  Solution_Fine = sol_fine->GetNodes()->GetSolution(Point_Fine);
468  for (iVar = 0; iVar < nVar; iVar++) {
469  /*--- Prevent a fine grid divergence due to a coarse grid divergence ---*/
470  if (Residual_Fine[iVar] != Residual_Fine[iVar])
471  Residual_Fine[iVar] = 0.0;
472  Solution_Fine[iVar] += factor*Residual_Fine[iVar];
473  }
474  }
475  END_SU2_OMP_FOR
476 
477  /*--- MPI the new interpolated solution ---*/
478 
479  sol_fine->InitiateComms(geo_fine, config, SOLUTION);
480  sol_fine->CompleteComms(geo_fine, config, SOLUTION);
481 
482 }

499 void CMultiGridIntegration::SetForcing_Term(CSolver *sol_fine, CSolver *sol_coarse, CGeometry *geo_fine,
500  CGeometry *geo_coarse, CConfig *config, unsigned short iMesh) {
501 
502  unsigned long Point_Fine, Point_Coarse, iVertex;
503  unsigned short iMarker, iVar, iChildren;
504  const su2double *Residual_Fine;
505 
506  const unsigned short nVar = sol_coarse->GetnVar();
507  su2double factor = config->GetDamp_Res_Restric();
508 
509  su2double *Residual = new su2double[nVar];
510 
511  SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPointDomain(), omp_get_num_threads()))
512  for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPointDomain(); Point_Coarse++) {
513 
514  sol_coarse->GetNodes()->SetRes_TruncErrorZero(Point_Coarse);
515 
516  for (iVar = 0; iVar < nVar; iVar++) Residual[iVar] = 0.0;
517 
518  for (iChildren = 0; iChildren < geo_coarse->nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
519  Point_Fine = geo_coarse->nodes->GetChildren_CV(Point_Coarse, iChildren);
520  Residual_Fine = sol_fine->LinSysRes.GetBlock(Point_Fine);
521  for (iVar = 0; iVar < nVar; iVar++)
522  Residual[iVar] += factor*Residual_Fine[iVar];
523  }
524  sol_coarse->GetNodes()->AddRes_TruncError(Point_Coarse, Residual);
525  }
526  END_SU2_OMP_FOR
527 
528  delete [] Residual;
529 
530  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
531  if (config->GetViscous_Wall(iMarker)) {
532  SU2_OMP_FOR_STAT(32)
533  for (iVertex = 0; iVertex < geo_coarse->nVertex[iMarker]; iVertex++) {
534  Point_Coarse = geo_coarse->vertex[iMarker][iVertex]->GetNode();
535  sol_coarse->GetNodes()->SetVel_ResTruncError_Zero(Point_Coarse);
536  }
537  END_SU2_OMP_FOR
538  }
539  }
540 
541  SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPointDomain(), omp_get_num_threads()))
542  for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPointDomain(); Point_Coarse++) {
543  sol_coarse->GetNodes()->SubtractRes_TruncError(Point_Coarse, sol_coarse->LinSysRes.GetBlock(Point_Coarse));
544  }
545  END_SU2_OMP_FOR
546

560 void CMultiGridIntegration::SetRestricted_Solution(unsigned short RunTime_EqSystem, CSolver *sol_fine, CSolver *sol_coarse,
561  CGeometry *geo_fine, CGeometry *geo_coarse, CConfig *config) {
562 
563  const unsigned short Solver_Position = config->GetContainerPosition(RunTime_EqSystem);
564  const bool grid_movement = config->GetGrid_Movement();
565 
566  /*--- Compute coarse solution from fine solution ---*/
567 
568  CSolver::MultigridRestriction(*geo_fine, sol_fine->GetNodes()->GetSolution(),
569  *geo_coarse, sol_coarse->GetNodes()->GetSolution());
570 
571  /*--- Update the solution at the no-slip walls ---*/
572

575

578 
579  const auto Point_Coarse = geo_coarse->vertex[iMarker][iVertex]->GetNode();
580

582 
583  /*--- At moving walls, set the solution based on the new density and wall velocity ---*/
584

587  sol_coarse->GetNodes()->SetVelSolutionVector(Point_Coarse, Grid_Vel);

593  }

595  }

598  sol_coarse->GetNodes()->SetVelSolutionDVector(Point_Coarse);
599  }
600 
601  }
602  END_SU2_OMP_FOR
603  }
604  }
605

613 void CMultiGridIntegration::SetRestricted_Gradient(unsigned short RunTime_EqSystem, CSolver *sol_fine, CSolver *sol_coarse,
614  CGeometry *geo_fine, CGeometry *geo_coarse, CConfig *config) {
615  unsigned long Point_Fine, Point_Coarse;
616  unsigned short iVar, iDim, iChildren;
617  su2double Area_Parent, Area_Children;
618 
619  const unsigned short nDim = geo_coarse->GetnDim();
620  const unsigned short nVar = sol_coarse->GetnVar();
621 
622  su2double **Gradient = new su2double* [nVar];
623  for (iVar = 0; iVar < nVar; iVar++)
624  Gradient[iVar] = new su2double [nDim];
625 
626  SU2_OMP_FOR_STAT(roundUpDiv(geo_coarse->GetnPoint(), omp_get_num_threads()))
627  for (Point_Coarse = 0; Point_Coarse < geo_coarse->GetnPoint(); Point_Coarse++) {
628  Area_Parent = geo_coarse->nodes->GetVolume(Point_Coarse);
629 
630  for (iVar = 0; iVar < nVar; iVar++)
631  for (iDim = 0; iDim < nDim; iDim++)
632  Gradient[iVar][iDim] = 0.0;
633 
634  for (iChildren = 0; iChildren < geo_coarse->nodes->GetnChildren_CV(Point_Coarse); iChildren++) {
635  Point_Fine = geo_coarse->nodes->GetChildren_CV(Point_Coarse, iChildren);
636  Area_Children = geo_fine->nodes->GetVolume(Point_Fine);
637  auto Gradient_fine = sol_fine->GetNodes()->GetGradient(Point_Fine);
638 
639  for (iVar = 0; iVar < nVar; iVar++)
640  for (iDim = 0; iDim < nDim; iDim++)
641  Gradient[iVar][iDim] += Gradient_fine[iVar][iDim]*Area_Children/Area_Parent;
642  }
643  sol_coarse->GetNodes()->SetGradient(Point_Coarse,Gradient);
644  }
645  END_SU2_OMP_FOR
646 
647  for (iVar = 0; iVar < nVar; iVar++)
648  delete [] Gradient[iVar];
649  delete [] Gradient;
