37 CSpeciesSolver::CSpeciesSolver(CGeometry* geometry, CConfig* config, unsigned short iMesh)
38  : CScalarSolver<CSpeciesVariable>(geometry, config, true) {
39  /*--- Store if an implicit scheme is used, for use during periodic boundary conditions. ---*/
40  SetImplicitPeriodic(config->GetKind_TimeIntScheme_Species() == EULER_IMPLICIT);
41 
42  /*--- Dimension of the problem. ---*/
43 
44  nVar = config->GetnSpecies();
45  nPrimVar = nVar;
46 
47  if (nVar > MAXNVAR)
48  SU2_MPI::Error("Increase static array size MAXNVAR for CSpeciesVariable and proceed.", CURRENT_FUNCTION);
49 
50  nPoint = geometry->GetnPoint();
51  nPointDomain = geometry->GetnPointDomain();
52 
53  /*--- Initialize nVarGrad for deallocation ---*/
54 
55  nVarGrad = nVar;
56 
57  /*--- Define geometry constants in the solver structure ---*/
58 
59  nDim = geometry->GetnDim();
60 
61  /*--- Single grid simulation ---*/
62 
63  if (iMesh == MESH_0 || config->GetMGCycle() == FULLMG_CYCLE) {
64 
65  /*--- Define some auxiliary vector related with the residual ---*/
66 
67  Residual_RMS.resize(nVar, 0.0);
68  Residual_Max.resize(nVar, 0.0);
69  Point_Max.resize(nVar, 0);
70  Point_Max_Coord.resize(nVar, nDim) = su2double(0.0);
71 
72  /*--- Initialize the BGS residuals in multizone problems. ---*/
73  if (config->GetMultizone_Problem()) {
74  Residual_BGS.resize(nVar, 0.0);
75  Residual_Max_BGS.resize(nVar, 0.0);
76  Point_Max_BGS.resize(nVar, 0);
77  Point_Max_Coord_BGS.resize(nVar, nDim) = su2double(0.0);
78  }
79 
80  /*--- Initialization of the structure of the whole Jacobian ---*/
81 
82  if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (species transport model)." << endl;
83  Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config, ReducerStrategy);
84  LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
85  LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
86  System.SetxIsZero(true);
87 
88  if (ReducerStrategy) EdgeFluxes.Initialize(geometry->GetnEdge(), geometry->GetnEdge(), nVar, nullptr);
89  }
90 
91  /*--- Initialize lower and upper limits---*/
92 
93  if (config->GetSpecies_Clipping()) {
94  for (auto iVar = 0u; iVar < nVar; iVar++) {
95  lowerlimit[iVar] = config->GetSpecies_Clipping_Min(iVar);
96  upperlimit[iVar] = config->GetSpecies_Clipping_Max(iVar);
97  }
98  } else {
99  for (auto iVar = 0u; iVar < nVar; iVar++) {
100  lowerlimit[iVar] = -1.0e15;
101  upperlimit[iVar] = 1.0e15;
102  }
103  }
104 
105  /*--- Scalar variable state at the far-field. ---*/
106 
107  for (auto iVar = 0u; iVar < nVar; iVar++) {
108  Solution_Inf[iVar] = config->GetSpecies_Init()[iVar];
109  }
110 
111  /*--- Initialize the solution to the far-field state everywhere. ---*/
112 
113  nodes = new CSpeciesVariable(Solution_Inf, nPoint, nDim, nVar, config);
114  SetBaseClassPointerToNodes();
115 
116  /*--- Initialize the mass diffusivity. Nondimensionalization done in the flow solver. ---*/
117  SU2_OMP_FOR_STAT(omp_chunk_size)
118  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++) {
119  for (auto iVar = 0u; iVar < nVar; iVar++) {
120  const auto MassDiffusivity = config->GetDiffusivity_ConstantND();
121  nodes->SetDiffusivity(iPoint, MassDiffusivity, iVar);
122  }
123  }
124  END_SU2_OMP_FOR
125 
126  /*--- MPI solution ---*/
127 
128  InitiateComms(geometry, config, SOLUTION);
129  CompleteComms(geometry, config, SOLUTION);
130 
131  SlidingState.resize(nMarker);
132  SlidingStateNodes.resize(nMarker);
133 
134  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {
135  if (config->GetMarker_All_KindBC(iMarker) == FLUID_INTERFACE) {
136  SlidingState[iMarker].resize(nVertex[iMarker], nPrimVar+1) = nullptr;
137  SlidingStateNodes[iMarker].resize(nVertex[iMarker],0);
138  }
139  }
140 
141  /*--- Set the column number for species in inlet-files.
142  * e.g. Coords(nDim), Temp(1), VelMag(1), Normal(nDim), Turb(1 or 2), Species(arbitrary) ---*/
143  Inlet_Position = nDim + 2 + nDim + config->GetnTurbVar();
144 
145  /*-- Allocation of inlet-values. Will be filled either by an inlet files,
146  * or uniformly by a uniform boundary condition. ---*/
147 
148  Inlet_SpeciesVars.resize(nMarker);
149  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {
150  Inlet_SpeciesVars[iMarker].resize(nVertex[iMarker], nVar);
151  for (unsigned long iVertex = 0; iVertex < nVertex[iMarker]; ++iVertex) {
152  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
153  Inlet_SpeciesVars[iMarker](iVertex, iVar) = Solution_Inf[iVar];
154  }
155  }
156  }
157 
158  /*--- Store the initial CFL number for all grid points. ---*/
159 
160  const su2double CFL = config->GetCFL(MGLevel) * config->GetCFLRedCoeff_Species();
161  SU2_OMP_FOR_STAT(omp_chunk_size)
162  for (auto iPoint = 0u; iPoint < nPoint; iPoint++) {
163  nodes->SetLocalCFL(iPoint, CFL);
164  }
165  END_SU2_OMP_FOR
166  Min_CFL_Local = CFL;
167  Max_CFL_Local = CFL;
168  Avg_CFL_Local = CFL;
169 
170  /*--- Add the solver name (max 8 characters) ---*/
171  SolverName = "SPECIES";
172 }

174 void CSpeciesSolver::LoadRestart(CGeometry** geometry, CSolver*** solver, CConfig* config, int val_iter,
175  bool val_update_geo) {
176  /*--- Restart the solution from file information ---*/
177 
178  const string restart_filename = config->GetFilename(config->GetSolution_FileName(), "", val_iter);
179 
180  /*--- To make this routine safe to call in parallel most of it can only be executed by one thread. ---*/
181  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
182  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
183 
184  if (config->GetRead_Binary_Restart()) {
185  Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
186  } else {
187  Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
188  }
189 
190  /*--- Skip flow variables and turbulence variables. ---*/
191 
192  unsigned short skipVars = nDim + solver[MESH_0][FLOW_SOL]->GetnVar() + config->GetnTurbVar();
193 
194  /*--- Adjust the number of solution variables in the incompressible
195  restart. We always carry a space in nVar for the energy equation in the
196  mean flow solver, but we only write it to the restart if it is active.
197  Therefore, we must reduce skipVars here if energy is inactive so that
198  the turbulent variables are read correctly. ---*/
199 
200  const bool incompressible = (config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE);
201  const bool energy = config->GetEnergy_Equation();
202  const bool weakly_coupled_heat = config->GetWeakly_Coupled_Heat();
203 
204  if (incompressible && ((!energy) && (!weakly_coupled_heat))) skipVars--;
205 
206  /*--- Load data from the restart into correct containers. ---*/
207 
208  unsigned long counter = 0;
209  for (auto iPoint_Global = 0ul; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++) {
210  /*--- Retrieve local index. If this node from the restart file lives
211  on the current processor, we will load and instantiate the vars. ---*/
212 
213  const auto iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
214 
215  if (iPoint_Local > -1) {
216  /*--- We need to store this point's data, so jump to the correct
217  offset in the buffer of data from the restart file and load it. ---*/
218 
219  const auto index = counter * Restart_Vars[1] + skipVars;
220  for (auto iVar = 0u; iVar < nVar; iVar++)
221  nodes->SetSolution(iPoint_Local, iVar, Restart_Data[index + iVar]);
222 
223  /*--- Increment the overall counter for how many points have been loaded. ---*/
224  counter++;
225  }
226  }
227 
228  /*--- Detect a wrong solution file ---*/
229 
230  if (counter != nPointDomain) {
231  SU2_MPI::Error(string("The solution file ") + restart_filename + string(" doesn't match with the mesh file!\n") +
232  string("It could be empty lines at the end of the file."),
233  CURRENT_FUNCTION);
234  }
235 
236  } // end safe global access, pre and postprocessing are thread-safe.
237  END_SU2_OMP_SAFE_GLOBAL_ACCESS
238 
239  /*--- MPI solution and compute the eddy viscosity ---*/
240 
241  solver[MESH_0][SPECIES_SOL]->InitiateComms(geometry[MESH_0], config, SOLUTION);
242  solver[MESH_0][SPECIES_SOL]->CompleteComms(geometry[MESH_0], config, SOLUTION);
243 
244  // Flow-Pre computes/sets mixture properties
245  solver[MESH_0][FLOW_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER,
246  RUNTIME_FLOW_SYS, false);
247  // Update eddy-visc which needs correct mixture density and mixture lam-visc. Note that after this, another Flow-Pre
248  // at the start of the Iteration sets the updated eddy-visc into the Flow-Solvers Primitives.
249  if (config->GetKind_Turb_Model() != TURB_MODEL::NONE)
250  solver[MESH_0][TURB_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
251  // For feature_multicomp this Scalar-Pre only computes the laminar contribution to mass diffusivity
252  solver[MESH_0][SPECIES_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER,
253  RUNTIME_SPECIES_SYS, false);
254 
255  /*--- Interpolate the solution down to the coarse multigrid levels ---*/
256 
257  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); iMesh++) {
258  MultigridRestriction(*geometry[iMesh - 1], solver[iMesh - 1][SPECIES_SOL]->GetNodes()->GetSolution(),
259  *geometry[iMesh], solver[iMesh][SPECIES_SOL]->GetNodes()->GetSolution());
260  solver[iMesh][SPECIES_SOL]->InitiateComms(geometry[iMesh], config, SOLUTION);
261  solver[iMesh][SPECIES_SOL]->CompleteComms(geometry[iMesh], config, SOLUTION);
262 
263  solver[iMesh][FLOW_SOL]->Preprocessing(geometry[iMesh], solver[iMesh], config, iMesh, NO_RK_ITER, RUNTIME_FLOW_SYS,
264  false);
265 
266  if (config->GetKind_Turb_Model() != TURB_MODEL::NONE)
267  solver[iMesh][TURB_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
268 
269  solver[iMesh][SPECIES_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER,
270  RUNTIME_SPECIES_SYS, false);
271  }
272 
273  /*--- Go back to single threaded execution. ---*/
274  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
275  /*--- Delete the class memory that is used to load the restart. ---*/
276 
277  delete[] Restart_Vars;
278  Restart_Vars = nullptr;
279  delete[] Restart_Data;
280  Restart_Data = nullptr;
281  }
282  END_SU2_OMP_SAFE_GLOBAL_ACCESS
283 }

285 void CSpeciesSolver::Preprocessing(CGeometry* geometry, CSolver** solver_container, CConfig* config,
286  unsigned short iMesh, unsigned short iRKStep, unsigned short RunTime_EqSystem,
287  bool Output) {
288  SU2_OMP_SAFE_GLOBAL_ACCESS(config->SetGlobalParam(config->GetKind_Solver(), RunTime_EqSystem);)
289 
290  /*--- Set the laminar mass Diffusivity for the species solver. ---*/
291  SU2_OMP_FOR_STAT(omp_chunk_size)
292  for (auto iPoint = 0u; iPoint < nPoint; iPoint++) {
293  for (auto iVar = 0u; iVar < nVar; iVar++) {
294  solver_container[FLOW_SOL]->GetFluidModel()->SetMassDiffusivityModel(config);
295  su2double mass_diffusivity = solver_container[FLOW_SOL]->GetFluidModel()->GetMassDiffusivity(iVar);
296  nodes->SetDiffusivity(iPoint, mass_diffusivity, iVar);
297  }
298 
299  } // iPoint
300  END_SU2_OMP_FOR
301 
302  /*--- Clear Residual and Jacobian. Upwind second order reconstruction and gradients ---*/
303  CommonPreprocessing(geometry, config, Output);

320 void CSpeciesSolver::BC_Inlet(CGeometry* geometry, CSolver** solver_container, CNumerics* conv_numerics,
321  CNumerics* visc_numerics, CConfig* config, unsigned short val_marker) {
322  string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
323 
324  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
325 
326  /*--- Loop over all the vertices on this boundary marker ---*/
327 
328  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
329  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
330  auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
331 
332  /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
333 
334  if (!geometry->nodes->GetDomain(iPoint)) continue;
335 
336  if (config->GetSpecies_StrongBC()) {
337  nodes->SetSolution_Old(iPoint, Inlet_SpeciesVars[val_marker][iVertex]);
338 
339  LinSysRes.SetBlock_Zero(iPoint);
340 
341  /*--- Includes 1 in the diagonal ---*/
342  for (auto iVar = 0u; iVar < nVar; iVar++) {
343  auto total_index = iPoint * nVar + iVar;
344  Jacobian.DeleteValsRowi(total_index);
345  }
346  } else { // weak BC
347  /*--- Normal vector for this vertex (negate for outward convention) ---*/
348 
349  su2double Normal[MAXNDIM] = {0.0};
350  for (auto iDim = 0u; iDim < nDim; iDim++) Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
351  conv_numerics->SetNormal(Normal);
352 
353  /*--- Allocate the value at the inlet ---*/
354 
355  auto V_inlet = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
356 
357  /*--- Retrieve solution at the farfield boundary node ---*/
358 
359  auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
360 
361  /*--- Set various quantities in the solver class ---*/
362 
363  conv_numerics->SetPrimitive(V_domain, V_inlet);
364 
365  /*--- Set the species variable state at the inlet. ---*/
366 
367  conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), Inlet_SpeciesVars[val_marker][iVertex]);
368 
369  /*--- Set various other quantities in the solver class ---*/
370 
371  if (dynamic_grid)
372  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
373 
374  if (conv_numerics->GetBoundedScalar()) {
375  const su2double* velocity = &V_inlet[prim_idx.Velocity()];
376  const su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
377  conv_numerics->SetMassFlux(BoundedScalarBCFlux(iPoint, implicit, density, velocity, Normal));
378  }
379 
380  /*--- Compute the residual using an upwind scheme ---*/
381 
382  auto residual = conv_numerics->ComputeResidual(config);
383  LinSysRes.AddBlock(iPoint, residual);
384 
385  /*--- Jacobian contribution for implicit integration ---*/
386 
387  if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
388 
389  // Unfinished viscous contribution removed before right after d8a0da9a00. Further testing required.
390 
391  }
392  }
393  END_SU2_OMP_FOR
394 }

405 su2double CSpeciesSolver::GetInletAtVertex(su2double *val_inlet,
406  unsigned long val_inlet_point,
407  unsigned short val_kind_marker,
408  string val_marker,
409  const CGeometry *geometry,
410  const CConfig *config) const {
411  /*--- Local variables ---*/
412 
413  unsigned short iMarker;
414  unsigned long iPoint, iVertex;
415  su2double Area = 0.0;
416  su2double Normal[3] = {0.0,0.0,0.0};
417 
418  /*--- Alias positions within inlet file for readability ---*/
419 
420  if (val_kind_marker == INLET_FLOW) {
421 
422  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
423  if ((config->GetMarker_All_KindBC(iMarker) == INLET_FLOW) &&
424  (config->GetMarker_All_TagBound(iMarker) == val_marker)) {
425 
426  for (iVertex = 0; iVertex < nVertex[iMarker]; iVertex++){
427 
428  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
429 
430  if (iPoint == val_inlet_point) {
431 
432  /*-- Compute boundary face area for this vertex. ---*/
433 
434  geometry->vertex[iMarker][iVertex]->GetNormal(Normal);
435  Area = GeometryToolbox::Norm(nDim, Normal);
436 
437  /*--- Access and store the inlet variables for this vertex. ---*/
438  for (unsigned short iVar = 0; iVar < nVar; iVar++)
439  val_inlet[Inlet_Position + iVar] = Inlet_SpeciesVars[iMarker][iVertex][iVar];
440 
441  /*--- Exit once we find the point. ---*/
442 
443  return Area;
444 
445  }
446  }
447  }
448  }
449 
450  }
451 
452  /*--- If we don't find a match, then the child point is not on the
453  current inlet boundary marker. Return zero area so this point does
454  not contribute to the restriction operator and continue. ---*/
455 
456  return Area;
457 
458 }

472 void CSpeciesSolver::BC_Outlet(CGeometry* geometry, CSolver** solver_container, CNumerics* conv_numerics,
473  CNumerics* visc_numerics, CConfig* config, unsigned short val_marker) {
474 
475  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
476 
477  /*--- Loop over all the vertices on this boundary marker ---*/
478 
479  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
480  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
481  /* strong zero flux Neumann boundary condition at the outlet */
482  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
483 
484  /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
485 
486  if (!geometry->nodes->GetDomain(iPoint)) continue;
487 
488  if (config->GetSpecies_StrongBC()) {
489  /*--- Allocate the value at the outlet ---*/
490  auto Point_Normal = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
491 
492  nodes->SetSolution_Old(iPoint, nodes->GetSolution(Point_Normal));
493 
494  LinSysRes.SetBlock_Zero(iPoint);
495 
496  /*--- Includes 1 on the diagonal ---*/
497  for (auto iVar = 0u; iVar < nVar; iVar++) {
498  auto total_index = iPoint * nVar + iVar;
499  Jacobian.DeleteValsRowi(total_index);
500  }
501  } else { // weak BC
502 
503  /*--- Allocate the value at the outlet ---*/
504  auto V_outlet = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
505 
506  /*--- Retrieve solution at the farfield boundary node ---*/
507 
508  auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
509 
510  /*--- Set various quantities in the solver class ---*/
511 
512  conv_numerics->SetPrimitive(V_domain, V_outlet);
513 
514  /*--- Set the species variables. Here we use a Neumann BC such
515  that the species variable is copied from the interior of the
516  domain to the outlet before computing the residual. ---*/
517 
518  conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), nodes->GetSolution(iPoint));
519 
520  /*--- Set Normal (negate for outward convention) ---*/
521 
522  su2double Normal[MAXNDIM] = {0.0};
523  for (auto iDim = 0u; iDim < nDim; iDim++) Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
524  conv_numerics->SetNormal(Normal);
525 
526  if (dynamic_grid)
527  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
528 
529  if (conv_numerics->GetBoundedScalar()) {
530  const su2double* velocity = &V_outlet[prim_idx.Velocity()];
531  const su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
532  conv_numerics->SetMassFlux(BoundedScalarBCFlux(iPoint, implicit, density, velocity, Normal));
533  }
534 
535  /*--- Compute the residual using an upwind scheme ---*/
536  auto residual = conv_numerics->ComputeResidual(config);
537  LinSysRes.AddBlock(iPoint, residual);
538 
539  /*--- Jacobian contribution for implicit integration ---*/
540  if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
541 
542  // Unfinished viscous contribution removed before right after d8a0da9a00. Further testing required.
543 
544  }
545  }
546  END_SU2_OMP_FOR
547 }

549 void CSpeciesSolver::Source_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
550  CConfig *config, unsigned short iMesh) {
551 
552  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
553  const bool axisymmetric = config->GetAxisymmetric();
554 
555  if (axisymmetric) {
556  CNumerics *numerics = numerics_container[SOURCE_FIRST_TERM + omp_get_thread_num()*MAX_TERMS];
