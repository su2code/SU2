251 void CIncNSSolver::GetStreamwise_Periodic_Properties(const CGeometry *geometry,
252  CConfig *config,
253  const unsigned short iMesh) {
254 
255  /*---------------------------------------------------------------------------------------------*/
256  // 1. Evaluate massflow, area avg density & Temperature and Area at streamwise periodic outlet.
257  // 2. Update delta_p is target massflow is chosen.
258  // 3. Loop Heatflux markers and integrate heat across the boundary. Only if energy equation is on.
259  /*---------------------------------------------------------------------------------------------*/
260 
261  /*-------------------------------------------------------------------------------------------------*/
262  /*--- 1. Evaluate Massflow [kg/s], area-averaged density [kg/m^3] and Area [m^2] at the ---*/
263  /*--- (there can be only one) streamwise periodic outlet/donor marker. Massflow is obviously ---*/
264  /*--- needed for prescribed massflow but also for the additional source and heatflux ---*/
265  /*--- boundary terms of the energy equation. Area and the avg-density are used for the ---*/
266  /*--- Pressure-Drop update in case of a prescribed massflow. ---*/
267  /*-------------------------------------------------------------------------------------------------*/
268 
269  su2double Area_Local = 0.0,
270  MassFlow_Local = 0.0,
271  Average_Density_Local = 0.0,
272  Temperature_Local = 0.0;
273 
274  for (auto iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
275 
276  /*--- Only "outlet"/donor periodic marker. GetnMarker_Periodic counts from 1: First all the inlet marker from
277  the periodic marker pairs and then the all the outlets. I.e. in order to get the outlet of the first pair
278  we need to divide the number of periodic markers by 2 and add 1 (because count starts at 1). ---*/
279  if (config->GetMarker_All_KindBC(iMarker) == PERIODIC_BOUNDARY &&
280  config->GetMarker_All_PerBound(iMarker) == (config->GetnMarker_Periodic()/2 + 1)) {
281 
282  for (auto iVertex = 0ul; iVertex < geometry->nVertex[iMarker]; iVertex++) {
283 
284  auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
285 
286  if (geometry->nodes->GetDomain(iPoint)) {
287 
288  /*--- A = dot_prod(n_A*n_A), with n_A beeing the area-normal. ---*/
289 
290  const auto AreaNormal = geometry->vertex[iMarker][iVertex]->GetNormal();
291 
292  const su2double FaceArea = GeometryToolbox::Norm(nDim, AreaNormal);
293 
294  /*--- m_dot = dot_prod(n*v) * A * rho, with n beeing unit normal. ---*/
295  MassFlow_Local += nodes->GetProjVel(iPoint, AreaNormal) * nodes->GetDensity(iPoint);
296 
297  Area_Local += FaceArea;
298 
299  Average_Density_Local += FaceArea * nodes->GetDensity(iPoint);
300 
301  /*--- Due to periodicity, temperatures are equal one the inlet(1) and outlet(2) ---*/
302  Temperature_Local += FaceArea * nodes->GetTemperature(iPoint);
303 
304  } // if domain
305  } // loop vertices
306  } // loop periodic boundaries
307  } // loop MarkerAll
308 
309  // MPI Communication: Sum Area, Sum rho*A & T*A and divide by AreaGlobal, sum massflow
310  su2double Area_Global(0), Average_Density_Global(0), MassFlow_Global(0), Temperature_Global(0);
311  SU2_MPI::Allreduce(&Area_Local, &Area_Global, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
312  SU2_MPI::Allreduce(&Average_Density_Local, &Average_Density_Global, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
313  SU2_MPI::Allreduce(&MassFlow_Local, &MassFlow_Global, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
314  SU2_MPI::Allreduce(&Temperature_Local, &Temperature_Global, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
315 
316  Average_Density_Global /= Area_Global;
317  Temperature_Global /= Area_Global;
318 
319  /*--- Set solver variables ---*/
320  SPvals.Streamwise_Periodic_MassFlow = MassFlow_Global;
321  SPvals.Streamwise_Periodic_InletTemperature = Temperature_Global;
322  SPvals.Streamwise_Periodic_BoundaryArea = Area_Global;
323  SPvals.Streamwise_Periodic_AvgDensity = Average_Density_Global;
324 
325  if (config->GetEnergy_Equation()) {
326  /*---------------------------------------------------------------------------------------------*/
327  /*--- 3. Compute the integrated Heatflow [W] for the energy equation source term, heatflux ---*/
328  /*--- boundary term and recovered Temperature. The computation is not completely clear. ---*/
329  /*--- Here the Heatflux from all Boundary markers in the config-file is used. ---*/
330  /*---------------------------------------------------------------------------------------------*/
331 
332  su2double HeatFlow_Local = 0.0, HeatFlow_Global = 0.0;
333 
334  /*--- Loop over all heatflux Markers ---*/
335  for (auto iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
336 
337  if (config->GetMarker_All_KindBC(iMarker) == HEAT_FLUX) {
338 
339  /*--- Identify the boundary by string name and retrive heatflux from config ---*/
340  const auto Marker_StringTag = config->GetMarker_All_TagBound(iMarker);
341  const su2double Wall_HeatFlux = config->GetWall_HeatFlux(Marker_StringTag);
342 
343  for (auto iVertex = 0ul; iVertex < geometry->nVertex[iMarker]; iVertex++) {
344 
345  auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
346 
347  if (!geometry->nodes->GetDomain(iPoint)) continue;
348 
349  const auto AreaNormal = geometry->vertex[iMarker][iVertex]->GetNormal();
350 
351  const su2double FaceArea = GeometryToolbox::Norm(nDim, AreaNormal);
352 
353  HeatFlow_Local += FaceArea * (-1.0) * Wall_HeatFlux/config->GetHeat_Flux_Ref();;
354  } // loop Vertices
355  } // loop Heatflux marker
356  } // loop AllMarker
357 
358  /*--- MPI Communication sum up integrated Heatflux from all processes ---*/
359  SU2_MPI::Allreduce(&HeatFlow_Local, &HeatFlow_Global, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
360 
361  /*--- Set the solver variable Integrated Heatflux ---*/
362  SPvals.Streamwise_Periodic_IntegratedHeatFlow = HeatFlow_Global;
363  } // if energy
364 }

367 void CIncNSSolver::Compute_Streamwise_Periodic_Recovered_Values(CConfig *config, const CGeometry *geometry,
368  const unsigned short iMesh) {
369 
370  const bool energy = (config->GetEnergy_Equation() && config->GetStreamwise_Periodic_Temperature());
371  const auto InnerIter = config->GetInnerIter();
372 
373  /*--- Reference node on inlet periodic marker to compute relative distance along periodic translation vector. ---*/
374  const auto ReferenceNode = geometry->GetStreamwise_Periodic_RefNode();
375 
376  /*--- Compute square of the distance between the 2 periodic surfaces. ---*/
377  const su2double norm2_translation = GeometryToolbox::SquaredNorm(nDim, config->GetPeriodic_Translation(0));
378 
379  /*--- Compute recoverd pressure and temperature for all points ---*/
380  SU2_OMP_FOR_STAT(omp_chunk_size)
381  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
382 
383  /*--- First, compute helping terms based on relative distance (0,l) between periodic markers ---*/
384  su2double dot_product = 0.0;
385  for (unsigned short iDim = 0; iDim < nDim; iDim++)
386  dot_product += fabs( (geometry->nodes->GetCoord(iPoint,iDim) - ReferenceNode[iDim]) * config->GetPeriodic_Translation(0)[iDim]);
387 
388  /*--- Second, substract/add correction from reduced pressure/temperature to get recoverd pressure/temperature ---*/
389  const su2double Pressure_Recovered = nodes->GetPressure(iPoint) - SPvals.Streamwise_Periodic_PressureDrop /
390  norm2_translation * dot_product;
391  nodes->SetStreamwise_Periodic_RecoveredPressure(iPoint, Pressure_Recovered);
392 
393  /*--- InnerIter > 0 as otherwise MassFlow in the denominator would be zero ---*/
394  if (energy && InnerIter > 0) {
395  su2double Temperature_Recovered = nodes->GetTemperature(iPoint);
396  Temperature_Recovered += SPvals.Streamwise_Periodic_IntegratedHeatFlow /
397  (SPvals.Streamwise_Periodic_MassFlow * nodes->GetSpecificHeatCp(iPoint) * norm2_translation) * dot_product;
398  nodes->SetStreamwise_Periodic_RecoveredTemperature(iPoint, Temperature_Recovered);
399  }
400  } // for iPoint
401  END_SU2_OMP_FOR
402 
403  /*--- Compute the integrated Heatflux Q into the domain, and massflow over periodic markers ---*/
404  SU2_OMP_SAFE_GLOBAL_ACCESS(GetStreamwise_Periodic_Properties(geometry, config, iMesh);)
405 }

465 void CIncNSSolver::BC_Wall_Generic(const CGeometry *geometry, const CConfig *config,
466  unsigned short val_marker, unsigned short kind_boundary) {
467 
468  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
469  const bool energy = config->GetEnergy_Equation();
470 
471  /*--- Variables for streamwise periodicity ---*/
472  const bool streamwise_periodic = (config->GetKind_Streamwise_Periodic() != ENUM_STREAMWISE_PERIODIC::NONE);
473  const bool streamwise_periodic_temperature = config->GetStreamwise_Periodic_Temperature();
474  su2double Cp, thermal_conductivity, dot_product, scalar_factor;
475 
476 
477  /*--- Identify the boundary by string name ---*/
478 
479  const auto Marker_Tag = config->GetMarker_All_TagBound(val_marker);
480 
481  /*--- Get the specified wall heat flux, temperature or heat transfer coefficient from config ---*/
482 
483  su2double Wall_HeatFlux = 0.0, Twall = 0.0, Tinfinity = 0.0, Transfer_Coefficient = 0.0;
484 
485  switch (kind_boundary) {
486  case HEAT_FLUX:
487  Wall_HeatFlux = config->GetWall_HeatFlux(Marker_Tag) / config->GetHeat_Flux_Ref();
488  if (config->GetIntegrated_HeatFlux()) {
489  Wall_HeatFlux /= geometry->GetSurfaceArea(config, val_marker);
490  }
491  break;
492  case ISOTHERMAL:
493  Twall = config->GetIsothermal_Temperature(Marker_Tag) / config->GetTemperature_Ref();
494  break;
495  case HEAT_TRANSFER:
496  Transfer_Coefficient = config->GetWall_HeatTransfer_Coefficient(Marker_Tag) * config->GetTemperature_Ref() /
497  config->GetHeat_Flux_Ref();
498  Tinfinity = config->GetWall_HeatTransfer_Temperature(Marker_Tag) / config->GetTemperature_Ref();
499  break;
500  default:
501  SU2_MPI::Error("Unknown type of boundary condition.", CURRENT_FUNCTION);
502  break;
503  }
504 
505  /*--- Get wall function treatment from config. ---*/
506 
507  //const auto Wall_Function = config->GetWallFunction_Treatment(Marker_Tag);
508  // nijso: we do not have a special treatment yet for heated walls
509  // the wall function model is written for heat flux, we have to implement isothermal wall conditions
510  //if (Wall_Function != WALL_FUNCTIONS::NONE)
511  // SU2_MPI::Error("Wall function treatment not implemented yet.", CURRENT_FUNCTION);
512 
513  /*--- Loop over all of the vertices on this boundary marker ---*/
514 
515  SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
516  for (auto iVertex = 0ul; iVertex < geometry->nVertex[val_marker]; iVertex++) {
517  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
518 
519  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
520 
521  if (!geometry->nodes->GetDomain(iPoint)) continue;
522 
523  /*--- Compute dual-grid area and boundary normal ---*/
524 
525  const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
526 
527  const su2double Area = GeometryToolbox::Norm(nDim, Normal);
528 
529  /*--- Impose the value of the velocity as a strong boundary
530  condition (Dirichlet). Fix the velocity and remove any
531  contribution to the residual at this node. ---*/
532 
533  if (dynamic_grid) {
534  nodes->SetVelocity_Old(iPoint, geometry->nodes->GetGridVel(iPoint));
535  } else {
536  su2double zero[MAXNDIM] = {0.0};
537  nodes->SetVelocity_Old(iPoint, zero);
538  }
539 
540  for (unsigned short iDim = 0; iDim < nDim; iDim++)
541  LinSysRes(iPoint, iDim+1) = 0.0;
542  nodes->SetVel_ResTruncError_Zero(iPoint);
543 
544  /*--- Enforce the no-slip boundary condition in a strong way by
545  modifying the velocity-rows of the Jacobian (1 on the diagonal). ---*/
546 
547  if (implicit) {
548  for (unsigned short iVar = 1; iVar <= nDim; iVar++)
549  Jacobian.DeleteValsRowi(iPoint*nVar+iVar);
550  }
551 
552  if (!energy) continue;
553 
554  switch(kind_boundary) {
555  case HEAT_FLUX:
556 
557  /*--- Apply a weak boundary condition for the energy equation.
558  Compute the residual due to the prescribed heat flux. ---*/
559 
560  LinSysRes(iPoint, nDim+1) -= Wall_HeatFlux*Area;
561 
562  /*--- With streamwise periodic flow and heatflux walls an additional term is introduced in the boundary formulation ---*/
563  if (streamwise_periodic && streamwise_periodic_temperature) {
564 
565  Cp = nodes->GetSpecificHeatCp(iPoint);
566  thermal_conductivity = nodes->GetThermalConductivity(iPoint);
567 
568  /*--- Scalar factor of the residual contribution ---*/
569  const su2double norm2_translation = GeometryToolbox::SquaredNorm(nDim, config->GetPeriodic_Translation(0));
570  scalar_factor = SPvals.Streamwise_Periodic_IntegratedHeatFlow*thermal_conductivity / (SPvals.Streamwise_Periodic_MassFlow * Cp * norm2_translation);
571 
572  /*--- Dot product ---*/
573  dot_product = GeometryToolbox::DotProduct(nDim, config->GetPeriodic_Translation(0), Normal);
574 
575  LinSysRes(iPoint, nDim+1) += scalar_factor*dot_product;
576  } // if streamwise_periodic
577  break;
578 
579  case HEAT_TRANSFER:
580  Twall = nodes->GetTemperature(iPoint);
581  Wall_HeatFlux = Transfer_Coefficient * (Tinfinity - Twall);
582 
583  /*--- Apply a weak boundary condition for the energy equation.
584  Compute the residual due to the prescribed temperature and transfer coefficient.
585  Note that for the Heat_Transfer wall, basically a heatflux wall that depends on the local Temperature is applied. ---*/
586  LinSysRes(iPoint, nDim+1) -= Wall_HeatFlux*Area;
587 
588  if (implicit) {
589  Jacobian.AddVal2Diag(iPoint, nDim+1, Transfer_Coefficient*Area);
590  }
591  break;
592 
593  case ISOTHERMAL:
594 
595  const auto Point_Normal = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
596 
597  /*--- Get coordinates of i & nearest normal and compute distance ---*/
598 
599  const auto Coord_i = geometry->nodes->GetCoord(iPoint);
600  const auto Coord_j = geometry->nodes->GetCoord(Point_Normal);
601  su2double Edge_Vector[MAXNDIM];
602  GeometryToolbox::Distance(nDim, Coord_j, Coord_i, Edge_Vector);
603  su2double dist_ij_2 = GeometryToolbox::SquaredNorm(nDim, Edge_Vector);
604  su2double dist_ij = sqrt(dist_ij_2);
605 
606  /*--- Compute the normal gradient in temperature using Twall ---*/
607 
608  su2double dTdn = -(nodes->GetTemperature(Point_Normal) - Twall)/dist_ij;
609 
610  /*--- Get thermal conductivity ---*/
611 
612  su2double thermal_conductivity = nodes->GetThermalConductivity(iPoint);
613 
614  /*--- Apply a weak boundary condition for the energy equation.
615  Compute the residual due to the prescribed heat flux. ---*/
616 
617  LinSysRes(iPoint, nDim+1) -= thermal_conductivity*dTdn*Area;
618 
619  /*--- Jacobian contribution for temperature equation. ---*/
620 
621  if (implicit) {
622  su2double proj_vector_ij = 0.0;
623  if (dist_ij_2 > 0.0)
624  proj_vector_ij = GeometryToolbox::DotProduct(nDim, Edge_Vector, Normal) / dist_ij_2;
625  Jacobian.AddVal2Diag(iPoint, nDim+1, thermal_conductivity*proj_vector_ij);
626  }
627  break;
628  } // switch
629  }
630  END_SU2_OMP_FOR

747 void CIncNSSolver::SetTau_Wall_WF(CGeometry *geometry, CSolver **solver_container, const CConfig *config) {
748  /*--- The wall function implemented herein is based on Nichols and Nelson, AIAA J. v32 n6 2004. ---*/
749 
750  unsigned long notConvergedCounter = 0; /*--- Counts the number of wall cells that are not converged ---*/
751  unsigned long smallYPlusCounter = 0; /*--- Counts the number of wall cells where y+ < 5 ---*/
752 
753  const auto max_iter = config->GetwallModel_MaxIter();
754  const su2double relax = config->GetwallModel_RelFac();
755 
756  /*--- Typical constants from boundary layer theory ---*/
757 
758  const su2double kappa = config->GetwallModel_Kappa();
759  const su2double B = config->GetwallModel_B();
760 
761  for (auto iMarker = 0u; iMarker < config->GetnMarker_All(); iMarker++) {
762 
763  if (!config->GetViscous_Wall(iMarker)) continue;
764 
765  /*--- Identify the boundary by string name ---*/
766 
767  const auto Marker_Tag = config->GetMarker_All_TagBound(iMarker);
768 
769  /*--- Jump to another BC if it is not wall function ---*/
770 
771  if (config->GetWallFunction_Treatment(Marker_Tag) != WALL_FUNCTIONS::STANDARD_FUNCTION)
772  continue;
773 
774  /*--- Loop over all of the vertices on this boundary marker ---*/
775 
776  SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
777  for (auto iVertex = 0u; iVertex < geometry->nVertex[iMarker]; iVertex++) {
778 
779  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
780  const auto Point_Normal = geometry->vertex[iMarker][iVertex]->GetNormal_Neighbor();
781  /*--- On the finest mesh compute also on halo nodes to avoid communication of tau wall. ---*/
782  if ((!geometry->nodes->GetDomain(iPoint)) && !(MGLevel==MESH_0)) continue;
783 
784  /*--- Get coordinates of the current vertex and nearest normal point ---*/
785 
786  const auto Coord = geometry->nodes->GetCoord(iPoint);
787  const auto Coord_Normal = geometry->nodes->GetCoord(Point_Normal);
788 
789  /*--- Compute dual-grid area and boundary normal ---*/
790 
791  const auto Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
792 
793  const su2double Area = GeometryToolbox::Norm(nDim, Normal);
794 
795  su2double UnitNormal[MAXNDIM] = {0.0};
796  for (auto iDim = 0u; iDim < nDim; iDim++)
797  UnitNormal[iDim] = -Normal[iDim]/Area;
798 
799  /*--- Get the velocity, pressure, and temperature at the nearest
800  (normal) interior point. ---*/
801 
802  su2double Vel[MAXNDIM] = {0.0};
803  for (auto iDim = 0u; iDim < nDim; iDim++)
804  Vel[iDim] = nodes->GetVelocity(Point_Normal,iDim);
805 
806  /*--- Compute the wall-parallel velocity at first point off the wall ---*/
807 
808  const su2double VelNormal = GeometryToolbox::DotProduct(int(MAXNDIM), Vel, UnitNormal);
809 
810  su2double VelTang[MAXNDIM] = {0.0};
811  for (auto iDim = 0u; iDim < nDim; iDim++)
812  VelTang[iDim] = Vel[iDim] - VelNormal*UnitNormal[iDim];
813 
814  const su2double VelTangMod = GeometryToolbox::Norm(int(MAXNDIM), VelTang);
815 
816  /*--- Compute normal distance of the interior point from the wall ---*/
817 
818  su2double WallDist[MAXNDIM] = {0.0};
819  GeometryToolbox::Distance(nDim, Coord, Coord_Normal, WallDist);
820 
821  su2double WallDistMod = GeometryToolbox::Norm(int(MAXNDIM), WallDist);
822 
823  su2double Density_Wall = nodes->GetDensity(iPoint);
824 
825  /*--- Compute the shear stress at the wall in the regular fashion
826  * by using the stress tensor on the surface ---*/
827 
828  su2double tau[MAXNDIM][MAXNDIM] = {{0.0}};
829  const su2double Lam_Visc_Wall = nodes->GetLaminarViscosity(iPoint);
830  su2double Eddy_Visc_Wall = nodes->GetEddyViscosity(iPoint);
831 
832  CNumerics::ComputeStressTensor(nDim, tau, nodes->GetVelocityGradient(iPoint), Lam_Visc_Wall);
833 
834  su2double TauTangent[MAXNDIM] = {0.0};
835  GeometryToolbox::TangentProjection(nDim, tau, UnitNormal, TauTangent);
836 
837  const su2double WallShearStress = GeometryToolbox::Norm(int(MAXNDIM), TauTangent);
838 
839  /*--- Calculate the quantities from boundary layer theory and
840  * iteratively solve for a new wall shear stress. Use the current wall
841  * shear stress as a starting guess for the wall function. ---*/
842 
843  unsigned long counter = 0;
844  su2double diff = 1.0;
845  su2double U_Tau = max(1.0e-6,sqrt(WallShearStress/Density_Wall));
846  /*--- Use minimum y+ as defined in the config, in case the routine below for computing y+ does not converge ---*/
847  su2double Y_Plus = 0.99*config->GetwallModel_MinYPlus();
848 
849  const su2double Y_Plus_Start = Density_Wall * U_Tau * WallDistMod / Lam_Visc_Wall;
850 
851  /*--- Automatic switch off when y+ < "limit" according to Nichols & Nelson (2004) ---*/
852 
853  if (Y_Plus_Start < config->GetwallModel_MinYPlus()) {
854  smallYPlusCounter++;
855  continue;
856  }
857 
858  /*--- Convergence criterium for the Newton solver, note that 1e-10 is too large ---*/
859  const su2double tol = 1e-12;
860  while (fabs(diff) > tol) {
861 
862  /*--- Friction velocity and u+ ---*/
863 
864  const su2double U_Plus = VelTangMod / U_Tau;
865 
866  /*--- Y+ defined by White & Christoph ---*/
867 
868  const su2double kUp = kappa * U_Plus;
869 
870  // incompressible adiabatic result
871  const su2double Y_Plus_White = exp(kUp) * exp(-kappa * B);
872 
873  /*--- Spalding's universal form for the BL velocity with the
874  * outer velocity form of White & Christoph above. ---*/
875  Y_Plus = U_Plus + Y_Plus_White + (exp(-kappa * B)* (1.0 - kUp - 0.5 * kUp * kUp - kUp * kUp * kUp / 6.0));
876 
877  /*--- incompressible formulation ---*/
878  Eddy_Visc_Wall = Lam_Visc_Wall * kappa*exp(-kappa*B) * (exp(kUp) -1.0 - kUp - kUp * kUp / 2.0);
879 
880  Eddy_Visc_Wall = max(1.0e-6, Eddy_Visc_Wall);
881 
882  /* --- Define function for Newton method to zero --- */
883 
884  diff = (Density_Wall * U_Tau * WallDistMod / Lam_Visc_Wall) - Y_Plus;
885 
886  /* --- Gradient of function defined above wrt U_Tau --- */
887 
888  const su2double dyp_dup = 1.0 + exp(-kappa * B) * (kappa * exp(kUp) - kappa - kUp - 0.5 * kUp * kUp);
889  const su2double dup_dutau = - U_Plus / U_Tau;
890  const su2double grad_diff = Density_Wall * WallDistMod / Lam_Visc_Wall - dyp_dup * dup_dutau;
891 
892  /* --- Newton Step --- */
893 
894  U_Tau = U_Tau - relax*(diff / grad_diff);
895 
896  counter++;
897  if (counter > max_iter) {
898  notConvergedCounter++;
899  // use some safe values for convergence
900  Y_Plus = 30.0;
901  Eddy_Visc_Wall = 1.0;
902  U_Tau = 1.0;
903  break;
904  }
905  }
906 
907  /*--- Calculate an updated value for the wall shear stress
908  * using the y+ value, the definition of y+, and the definition of
909  * the friction velocity. ---*/
910  YPlus[iMarker][iVertex] = Y_Plus;
911  EddyViscWall[iMarker][iVertex] = Eddy_Visc_Wall;
912  UTau[iMarker][iVertex] = U_Tau;
913 
914  const su2double Tau_Wall = (1.0/Density_Wall)*pow(Y_Plus*Lam_Visc_Wall/WallDistMod,2.0);
915 
916  /*--- Store this value for the wall shear stress at the node. ---*/
917  nodes->SetTau_Wall(iPoint, Tau_Wall);
918 
919  }
920  END_SU2_OMP_FOR
921  }
922 
923  if (config->GetComm_Level() == COMM_FULL) {
924  static unsigned long globalCounter1, globalCounter2;
925 
926  ompMasterAssignBarrier(globalCounter1,0, globalCounter2,0);
927 
928  SU2_OMP_ATOMIC
929  globalCounter1 += notConvergedCounter;
930 
931  SU2_OMP_ATOMIC
932  globalCounter2 += smallYPlusCounter;
933 
934  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
935  SU2_MPI::Allreduce(&globalCounter1, &notConvergedCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
936  SU2_MPI::Allreduce(&globalCounter2, &smallYPlusCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
937 
938  if (rank == MASTER_NODE) {
939  if (notConvergedCounter)
940  cout << "Warning: Computation of wall coefficients (y+) did not converge in "
941  << notConvergedCounter << " points." << endl;
942 
943  if (smallYPlusCounter)
944  cout << "Warning: y+ < " << config->GetwallModel_MinYPlus() << " in " << smallYPlusCounter
945  << " points, for which the wall model is not active." << endl;
946  }
947  }
948  END_SU2_OMP_SAFE_GLOBAL_ACCESS
949  }
