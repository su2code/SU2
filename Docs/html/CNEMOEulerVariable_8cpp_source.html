31 CNEMOEulerVariable::CNEMOEulerVariable(su2double val_pressure,
32  const su2double *val_massfrac,
33  const su2double *val_mach,
34  su2double val_temperature,
35  su2double val_temperature_ve,
36  unsigned long npoint,
37  unsigned long ndim,
38  unsigned long nvar,
39  unsigned long nvarprim,
40  unsigned long nvarprimgrad,
41  const CConfig *config,
42  CNEMOGas *fluidmodel)
43  : CFlowVariable(npoint, ndim, nvar, nvarprim, nvarprimgrad, config),
44  indices(ndim, config->GetnSpecies()),
45  implicit(config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT) {
46 
47  unsigned short iDim, iSpecies;
48 
49  const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
50  (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
51  const bool classical_rk4 = (config->GetKind_TimeIntScheme_Flow() == CLASSICAL_RK4_EXPLICIT);
52 
53  /*--- Setting variable amounts ---*/
54 
55  nSpecies = config->GetnSpecies();
56  RHOS_INDEX = 0;
57  T_INDEX = nSpecies;
58  TVE_INDEX = nSpecies+1;
59  VEL_INDEX = nSpecies+2;
60  P_INDEX = nSpecies+nDim+2;
61  RHO_INDEX = nSpecies+nDim+3;
62  H_INDEX = nSpecies+nDim+4;
63  A_INDEX = nSpecies+nDim+5;
64  RHOCVTR_INDEX = nSpecies+nDim+6;
65  RHOCVVE_INDEX = nSpecies+nDim+7;
66  LAM_VISC_INDEX = nSpecies+nDim+8;
67  EDDY_VISC_INDEX = nSpecies+nDim+9;
68 
69  /*--- Set monoatomic flag ---*/
70  if (config->GetMonoatomic()) {
71  monoatomic = true;
72  Tve_Freestream = config->GetTemperature_ve_FreeStream();
73  }
74 
75  /*--- Size Grad_AuxVar for axiysmmetric ---*/
76  if (config->GetAxisymmetric()){
77  nAuxVar = 3;
78  Grad_AuxVar.resize(nPoint,nAuxVar,nDim,0.0);
79  AuxVar.resize(nPoint,nAuxVar) = su2double(0.0);
80  }
81 
82  /*--- Primitive and secondary variables ---*/
83  Primitive_Aux.resize(nPoint,nPrimVar) = su2double(0.0);
84  Secondary.resize(nPoint,nPrimVar) = su2double(0.0);
85 
86  dPdU.resize(nPoint, nVar) = su2double(0.0);
87  dTdU.resize(nPoint, nVar) = su2double(0.0);
88  dTvedU.resize(nPoint, nVar) = su2double(0.0);
89  Cvves.resize(nPoint, nSpecies) = su2double(0.0);
90  eves.resize(nPoint, nSpecies) = su2double(0.0);
91  Gamma.resize(nPoint) = su2double(0.0);
92 
93  /*--- Set mixture state ---*/
94  fluidmodel->SetTDStatePTTv(val_pressure, val_massfrac, val_temperature, val_temperature_ve);
95 
96  /*--- Compute necessary quantities ---*/
97  const su2double rho = fluidmodel->GetDensity();
98  const su2double soundspeed = fluidmodel->ComputeSoundSpeed();
99  const su2double sqvel = GeometryToolbox::SquaredNorm(nDim, val_mach) * pow(soundspeed,2);
100  const auto& energies = fluidmodel->ComputeMixtureEnergies();
101 
102  /*--- Loop over all points --*/
103  for(unsigned long iPoint = 0; iPoint < nPoint; ++iPoint){
104 
105  /*--- Initialize Solution & Solution_Old vectors ---*/
106  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
107  Solution(iPoint,iSpecies) = rho*val_massfrac[iSpecies];
108  for (iDim = 0; iDim < nDim; iDim++)
109  Solution(iPoint,nSpecies+iDim) = rho*val_mach[iDim]*soundspeed;
110 
111  Solution(iPoint,nSpecies+nDim) = rho*(energies[0]+0.5*sqvel);
112  Solution(iPoint,nSpecies+nDim+1) = rho*(energies[1]);
113  }
114 
115  Solution_Old = Solution;
116 
117  if (classical_rk4) Solution_New = Solution;
118 
119  /*--- Allocate and initializate solution for dual time strategy ---*/
120 
121  if (dual_time) {
122  Solution_time_n = Solution;
123  Solution_time_n1 = Solution;
124  }
125 
126 }

128 void CNEMOEulerVariable::SetVelocity2(unsigned long iPoint) {
129 
130  unsigned short iDim;
131 
132  Velocity2(iPoint) = 0.0;
133  for (iDim = 0; iDim < nDim; iDim++) {
134  Primitive(iPoint,VEL_INDEX+iDim) = Solution(iPoint,nSpecies+iDim) / Primitive(iPoint,RHO_INDEX);
135  Velocity2(iPoint) += Solution(iPoint,nSpecies+iDim)*Solution(iPoint,nSpecies+iDim)
136  / (Primitive(iPoint,RHO_INDEX)*Primitive(iPoint,RHO_INDEX));
137  }
138 }

140 bool CNEMOEulerVariable::SetPrimVar(unsigned long iPoint, CFluidModel *FluidModel) {
141 
142  unsigned short iVar;
143 
144  fluidmodel = static_cast<CNEMOGas*>(FluidModel);
145 
146  /*--- Convert conserved to primitive variables ---*/
147  bool nonPhys = Cons2PrimVar(Solution[iPoint], Primitive[iPoint],
148  dPdU[iPoint], dTdU[iPoint], dTvedU[iPoint], eves[iPoint], Cvves[iPoint]);
149 
150  /*--- Reset solution to previous one, if nonphys ---*/
151  if (nonPhys) {
152  for (iVar = 0; iVar < nVar; iVar++)
153  Solution(iPoint,iVar) = Solution_Old(iPoint,iVar);
154 
155  /*--- Recompute Primitive from previous solution ---*/
156  Cons2PrimVar(Solution[iPoint], Primitive[iPoint],
157  dPdU[iPoint], dTdU[iPoint], dTvedU[iPoint], eves[iPoint], Cvves[iPoint]);
158  }
159 
160  /*--- Set additional point quantities ---*/
161  Gamma(iPoint) = fluidmodel->ComputeGamma();
162 
163  SetVelocity2(iPoint);
164 
165  return nonPhys;
166 }

168 bool CNEMOEulerVariable::Cons2PrimVar(su2double *U, su2double *V,
169  su2double *val_dPdU, su2double *val_dTdU,
170  su2double *val_dTvedU, su2double *val_eves,
171  su2double *val_Cvves) {
172 
173  unsigned short iDim, iSpecies;
174  su2double Tmin, Tmax, Tvemin, Tvemax;
175  vector<su2double> rhos;
176 
177  rhos.resize(nSpecies,0.0);
178 
179  /*--- Conserved & primitive vector layout ---*/
180  // U: [rho1, ..., rhoNs, rhou, rhov, rhow, rhoe, rhoeve]^T
181  // V: [rho1, ..., rhoNs, T, Tve, u, v, w, P, rho, h, a, rhoCvtr, rhoCvve]^T
182 
183  /*--- Set booleans ---*/
184  bool nonPhys = false;
185 
186  /*--- Set temperature clipping values ---*/
187  Tmin = 50.0; Tmax = 8E4;
188  Tvemin = 50.0; Tvemax = 8E4;
189 
190  /*--- Rename variables for convenience ---*/
191  su2double rhoE = U[nSpecies+nDim]; // Density * energy [J/m3]
192  su2double rhoEve = U[nSpecies+nDim+1]; // Density * energy_ve [J/m3]
193 
194  /*--- Assign species & mixture density ---*/
195  // Note: if any species densities are < 0, these values are re-assigned
196  // in the primitive AND conserved vectors to ensure positive density
197  V[RHO_INDEX] = 0.0;
198  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
199  if (U[iSpecies] < 0.0) {
200  U[iSpecies] = 1E-20;
201  V[RHOS_INDEX+iSpecies] = 1E-20;
202  rhos[iSpecies] = 1E-20;
203  //nonPhys = true;
204  } else {
205  V[RHOS_INDEX+iSpecies] = U[iSpecies];
206  rhos[iSpecies] = U[iSpecies];
207  }
208  V[RHO_INDEX] += U[iSpecies];
209  }
210 
211  // Rename for convenience
212  su2double rho = V[RHO_INDEX];
213 
214  /*--- Assign velocity^2 ---*/
215  su2double sqvel = 0.0;
216  for (iDim = 0; iDim < nDim; iDim++) {
217  V[VEL_INDEX+iDim] = U[nSpecies+iDim]/V[RHO_INDEX];
218  sqvel += V[VEL_INDEX+iDim]*V[VEL_INDEX+iDim];
219  }
220 
221  /*--- Assign temperatures ---*/
222  const su2double Tve_old = V[TVE_INDEX];
223  const auto& T = fluidmodel->ComputeTemperatures(rhos, rhoE, rhoEve, 0.5*rho*sqvel, Tve_old);
224 
225  /*--- Temperatures ---*/
226  V[T_INDEX] = T[0];
227  V[TVE_INDEX] = T[1];
228 
229  // Determine if the temperature lies within the acceptable range
230  if (V[T_INDEX] <= Tmin) { nonPhys = true; return nonPhys;}
231  else if (V[T_INDEX] >= Tmax) { nonPhys = true; return nonPhys;}
232  else if (V[T_INDEX] != V[T_INDEX]){ nonPhys = true; return nonPhys;}
233 
234  if (!monoatomic){
235  if (V[TVE_INDEX] <= Tvemin) { nonPhys = true; return nonPhys;}
236  else if (V[TVE_INDEX] >= Tvemax) { nonPhys = true; return nonPhys;}
237  else if (V[TVE_INDEX] != V[TVE_INDEX]){ nonPhys = true; return nonPhys;}
238  }
239  else {V[TVE_INDEX] = Tve_Freestream;}
240 
241  // Determine other properties of the mixture at the current state
242  fluidmodel->SetTDStateRhosTTv(rhos, V[T_INDEX], V[TVE_INDEX]);
243 
244  const auto& cvves = fluidmodel->ComputeSpeciesCvVibEle(V[TVE_INDEX]);
245  vector<su2double> eves = fluidmodel->ComputeSpeciesEve(V[TVE_INDEX]);
246 
247  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
248  val_eves[iSpecies] = eves[iSpecies];
249  val_Cvves[iSpecies] = cvves[iSpecies];
250  }
251 
252  V[RHOCVTR_INDEX] = fluidmodel->ComputerhoCvtr();
253  V[RHOCVVE_INDEX] = fluidmodel->ComputerhoCvve();
254 
255  /*--- Pressure ---*/
256  V[P_INDEX] = fluidmodel->ComputePressure();
257 
258  if (V[P_INDEX] < 0.0) {
259  V[P_INDEX] = 1E-20;
260  nonPhys = true;
261  }
262 
263  /*--- Partial derivatives of pressure and temperature ---*/
264  if(implicit){
265  fluidmodel->ComputedPdU (V, eves, val_dPdU );
266  fluidmodel->ComputedTdU (V, val_dTdU );
267  fluidmodel->ComputedTvedU(V, eves, val_dTvedU);
268  }
269 
270  /*--- Sound speed ---*/
271  V[A_INDEX] = fluidmodel->ComputeSoundSpeed();
272 
273  /*--- Enthalpy ---*/
274  V[H_INDEX] = (U[nSpecies+nDim] + V[P_INDEX])/V[RHO_INDEX];
275 
276  return nonPhys;
277 }
