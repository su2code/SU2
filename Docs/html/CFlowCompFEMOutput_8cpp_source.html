159 void CFlowCompFEMOutput::LoadVolumeDataFEM(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iElem, unsigned long index, unsigned short dof){
160 
161  unsigned short iDim;
162 
163  unsigned short nVar = solver[FLOW_SOL]->GetnVar();
164 
165  /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
166  geometrical information for the FEM DG solver. ---*/
167 
168  CMeshFEM_DG *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
169 
170  CVolumeElementFEM *volElem = DGGeometry->GetVolElem();
171 
172  /*--- Get a pointer to the fluid model class from the DG-FEM solver
173  so that we can access the states below. ---*/
174 
175  CFluidModel *DGFluidModel = solver[FLOW_SOL]->GetFluidModel();
176 
177  /* Set the pointers for the solution for this element. */
178 
179  const unsigned long offset = nVar*volElem[iElem].offsetDOFsSolLocal;
180  su2double *solDOFs = solver[FLOW_SOL]->GetVecSolDOFs() + offset;
181 
182  /*--- Get the conservative variables for this particular DOF. ---*/
183 
184  const su2double *U = solDOFs+dof*nVar;
185 
186  /*--- Load the coordinate values of the solution DOFs. ---*/
187 
188  const su2double *coor = volElem[iElem].coorSolDOFs.data() + dof*nDim;
189 
190  /*--- Prepare the primitive states. ---*/
191 
192  const su2double DensityInv = 1.0/U[0];
193  su2double vel[3], Velocity2 = 0.0;
194  for(iDim=0; iDim<nDim; ++iDim) {
195  vel[iDim] = U[iDim+1]*DensityInv;
196  Velocity2 += vel[iDim]*vel[iDim];
197  }
198  su2double StaticEnergy = U[nDim+1]*DensityInv - 0.5*Velocity2;
199  DGFluidModel->SetTDState_rhoe(U[0], StaticEnergy);
200 
201 
202  SetVolumeOutputValue("COORD-X", index, coor[0]);
203  SetVolumeOutputValue("COORD-Y", index, coor[1]);
204  if (nDim == 3)
205  SetVolumeOutputValue("COORD-Z", index, coor[2]);
206  SetVolumeOutputValue("DENSITY", index, U[0]);
207  SetVolumeOutputValue("MOMENTUM-X", index, U[1]);
208  SetVolumeOutputValue("MOMENTUM-Y", index, U[2]);
209  if (nDim == 3){
210  SetVolumeOutputValue("MOMENTUM-Z", index, U[3]);
211  SetVolumeOutputValue("ENERGY", index, U[4]);
212  } else {
213  SetVolumeOutputValue("ENERGY", index, U[3]);
214  }
215 
216  SetVolumeOutputValue("PRESSURE", index, DGFluidModel->GetPressure());
217  SetVolumeOutputValue("TEMPERATURE", index, DGFluidModel->GetTemperature());
218  SetVolumeOutputValue("MACH", index, sqrt(Velocity2)/DGFluidModel->GetSoundSpeed());
219  SetVolumeOutputValue("PRESSURE_COEFF", index, DGFluidModel->GetCp());
220 
221  if (config->GetKind_Solver() == MAIN_SOLVER::FEM_NAVIER_STOKES){
222  SetVolumeOutputValue("LAMINAR_VISCOSITY", index, DGFluidModel->GetLaminarViscosity());
223  }
224  if ((config->GetKind_Solver() == MAIN_SOLVER::FEM_LES) && (config->GetKind_SGS_Model() != TURB_SGS_MODEL::IMPLICIT_LES)){
225  // todo: Export Eddy instead of Laminar viscosity
226  SetVolumeOutputValue("EDDY_VISCOSITY", index, DGFluidModel->GetLaminarViscosity());
227  }
228 }

230 void CFlowCompFEMOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
231 
232  CSolver* flow_solver = solver[FLOW_SOL];
233 
234  SetHistoryOutputValue("RMS_DENSITY", log10(flow_solver->GetRes_RMS(0)));
235  SetHistoryOutputValue("RMS_MOMENTUM-X", log10(flow_solver->GetRes_RMS(1)));
236  SetHistoryOutputValue("RMS_MOMENTUM-Y", log10(flow_solver->GetRes_RMS(2)));
237  if (nDim == 2)
238  SetHistoryOutputValue("RMS_ENERGY", log10(flow_solver->GetRes_RMS(3)));
239  else {
240  SetHistoryOutputValue("RMS_MOMENTUM-Z", log10(flow_solver->GetRes_RMS(3)));
241  SetHistoryOutputValue("RMS_ENERGY", log10(flow_solver->GetRes_RMS(4)));
242  }
243 
244 
245  SetHistoryOutputValue("MAX_DENSITY", log10(flow_solver->GetRes_Max(0)));
246  SetHistoryOutputValue("MAX_MOMENTUM-X", log10(flow_solver->GetRes_Max(1)));
247  SetHistoryOutputValue("MAX_MOMENTUM-Y", log10(flow_solver->GetRes_Max(2)));
248  if (nDim == 2)
249  SetHistoryOutputValue("MAX_ENERGY", log10(flow_solver->GetRes_Max(3)));
250  else {
251  SetHistoryOutputValue("MAX_MOMENTUM-Z", log10(flow_solver->GetRes_Max(3)));
252  SetHistoryOutputValue("MAX_ENERGY", log10(flow_solver->GetRes_Max(4)));
253  }
254 
255  SetHistoryOutputValue("AOA", config->GetAoA());
256  SetHistoryOutputValue("CFL_NUMBER", config->GetCFL(MESH_0));
257 
258  if (config->GetnMarker_Analyze() > 0) {
259  SU2_MPI::Error("SetAnalyzeSurface is not implemented for FEM-DG solver.", CURRENT_FUNCTION);
260  }
261 
262  /*--- Set aeroydnamic coefficients --- */
263 
264  SetAerodynamicCoefficients(config, flow_solver);
265 
266 }
