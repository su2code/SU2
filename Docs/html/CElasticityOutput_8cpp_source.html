34 CElasticityOutput::CElasticityOutput(CConfig *config, unsigned short nDim) : COutput(config, nDim, false) {
35 
36  linear_analysis = (config->GetGeometricConditions() == STRUCT_DEFORMATION::SMALL);
37  nonlinear_analysis = (config->GetGeometricConditions() == STRUCT_DEFORMATION::LARGE);
38  dynamic = (config->GetTime_Domain());
39 
40  /*--- Initialize number of variables ---*/
41  if (linear_analysis) nVar_FEM = nDim;
42  if (nonlinear_analysis) nVar_FEM = 3;
43 
44  /*--- Default fields for screen output ---*/
45  if (nRequestedHistoryFields == 0){
46  requestedHistoryFields.emplace_back("ITER");
47  requestedHistoryFields.emplace_back("RMS_RES");
48  nRequestedHistoryFields = requestedHistoryFields.size();
49  }
50 
51  /*--- Default fields for screen output ---*/
52  if (nRequestedScreenFields == 0){
53  if (dynamic) requestedScreenFields.emplace_back("TIME_ITER");
54  if (multiZone) requestedScreenFields.emplace_back("OUTER_ITER");
55  requestedScreenFields.emplace_back("INNER_ITER");
56  if(linear_analysis){
57  requestedScreenFields.emplace_back("RMS_DISP_X");
58  requestedScreenFields.emplace_back("RMS_DISP_Y");
59  requestedScreenFields.emplace_back("RMS_DISP_Z");
60  }
61  if(nonlinear_analysis){
62  requestedScreenFields.emplace_back("RMS_UTOL");
63  requestedScreenFields.emplace_back("RMS_RTOL");
64  requestedScreenFields.emplace_back("RMS_ETOL");
65  }
66  requestedScreenFields.emplace_back("VMS");
67  nRequestedScreenFields = requestedScreenFields.size();
68  }
69 
70  /*--- Default fields for volume output ---*/
71  if (nRequestedVolumeFields == 0){
72  requestedVolumeFields.emplace_back("COORDINATES");
73  requestedVolumeFields.emplace_back("SOLUTION");
74  requestedVolumeFields.emplace_back("STRESS");
75  if (dynamic) {
76  requestedVolumeFields.emplace_back("VELOCITY");
77  requestedVolumeFields.emplace_back("ACCELERATION");
78  }
79  if (config->GetTopology_Optimization()) requestedVolumeFields.emplace_back("TOPOLOGY");
80  nRequestedVolumeFields = requestedVolumeFields.size();
81  }
82 
83  stringstream ss;
84  ss << "Zone " << config->GetiZone() << " (Structure)";
85  multiZoneHeaderString = ss.str();
86 
87  /*--- Set the volume filename --- */
88 
89  volumeFilename = config->GetVolume_FileName();
90 
91  /*--- Set the surface filename --- */
92 
93  surfaceFilename = config->GetSurfCoeff_FileName();
94 
95  /*--- Set the restart filename --- */
96 
97  restartFilename = config->GetRestart_FileName();
98 
99  /*--- Set the default convergence field --- */
100 
101  if (convFields.empty() ) convFields.emplace_back("RMS_DISP_X");
102 
103 }

105 void CElasticityOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
106 
107  CSolver* fea_solver = solver[FEA_SOL];
108 
109  /*--- Residuals: ---*/
110  /*--- Linear analysis: RMS of the displacements in the nDim coordinates ---*/
111  /*--- Nonlinear analysis: UTOL, RTOL and DTOL (defined in the Postprocessing function) ---*/
112 
113 
114  if (linear_analysis){
115  SetHistoryOutputValue("RMS_DISP_X", log10(fea_solver->GetRes_RMS(0)));
116  SetHistoryOutputValue("RMS_DISP_Y", log10(fea_solver->GetRes_RMS(1)));
117  if (nDim == 3){
118  SetHistoryOutputValue("RMS_DISP_Z", log10(fea_solver->GetRes_RMS(2)));
119  }
120  } else if (nonlinear_analysis){
121  SetHistoryOutputValue("RMS_UTOL", log10(fea_solver->GetRes_FEM(0)));
122  SetHistoryOutputValue("RMS_RTOL", log10(fea_solver->GetRes_FEM(1)));
123  SetHistoryOutputValue("RMS_ETOL", log10(fea_solver->GetRes_FEM(2)));
124  }
125 
126  if (multiZone){
127  SetHistoryOutputValue("BGS_DISP_X", log10(fea_solver->GetRes_BGS(0)));
128  SetHistoryOutputValue("BGS_DISP_Y", log10(fea_solver->GetRes_BGS(1)));
129  if (nDim == 3) SetHistoryOutputValue("BGS_DISP_Z", log10(fea_solver->GetRes_BGS(2)));
130  }
131 
132  SetHistoryOutputValue("VMS", fea_solver->GetTotal_CFEA());
133  SetHistoryOutputValue("LOAD_INCREMENT", fea_solver->GetLoad_Increment()*100);
134  SetHistoryOutputValue("LOAD_RAMP", fea_solver->GetForceCoeff());
135 
136  SetHistoryOutputValue("LINSOL_ITER", fea_solver->GetIterLinSolver());
137  SetHistoryOutputValue("LINSOL_RESIDUAL", log10(fea_solver->GetResLinSolver()));
138 
139  SetHistoryOutputValue("REFERENCE_NODE", fea_solver->GetTotal_OFRefNode());
140  SetHistoryOutputValue("TOPOL_COMPLIANCE", fea_solver->GetTotal_OFCompliance());
141  SetHistoryOutputValue("STRESS_PENALTY", fea_solver->GetTotal_OFStressPenalty());
142  if (config->GetRefGeom()) {
143  SetHistoryOutputValue("REFERENCE_GEOMETRY", fea_solver->GetTotal_OFRefGeom());
144  }
145  if (config->GetTopology_Optimization()) {
146  SetHistoryOutputValue("VOLUME_FRACTION", fea_solver->GetTotal_OFVolFrac());
147  SetHistoryOutputValue("TOPOL_DISCRETENESS", fea_solver->GetTotal_OFDiscreteness());
148  }
149  /*--- Keep this as last, since it uses the history values that were set. ---*/
150  SetCustomAndComboObjectives(FEA_SOL, config, solver);
151 
152 }

191 void CElasticityOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint){
192 
193  CVariable* Node_Struc = solver[FEA_SOL]->GetNodes();
194  CPoint* Node_Geo = geometry->nodes;
195 
196  SetVolumeOutputValue("COORD-X", iPoint, Node_Geo->GetCoord(iPoint, 0));
197  SetVolumeOutputValue("COORD-Y", iPoint, Node_Geo->GetCoord(iPoint, 1));
198  if (nDim == 3)
199  SetVolumeOutputValue("COORD-Z", iPoint, Node_Geo->GetCoord(iPoint, 2));
200 
201  SetVolumeOutputValue("DISPLACEMENT-X", iPoint, Node_Struc->GetSolution(iPoint, 0));
202  SetVolumeOutputValue("DISPLACEMENT-Y", iPoint, Node_Struc->GetSolution(iPoint, 1));
203  if (nDim == 3) SetVolumeOutputValue("DISPLACEMENT-Z", iPoint, Node_Struc->GetSolution(iPoint, 2));
204 
205  if(dynamic){
206  SetVolumeOutputValue("VELOCITY-X", iPoint, Node_Struc->GetSolution_Vel(iPoint, 0));
207  SetVolumeOutputValue("VELOCITY-Y", iPoint, Node_Struc->GetSolution_Vel(iPoint, 1));
208  if (nDim == 3) SetVolumeOutputValue("VELOCITY-Z", iPoint, Node_Struc->GetSolution_Vel(iPoint, 2));
209 
210  SetVolumeOutputValue("ACCELERATION-X", iPoint, Node_Struc->GetSolution_Accel(iPoint, 0));
211  SetVolumeOutputValue("ACCELERATION-Y", iPoint, Node_Struc->GetSolution_Accel(iPoint, 1));
212  if (nDim == 3) SetVolumeOutputValue("ACCELERATION-Z", iPoint, Node_Struc->GetSolution_Accel(iPoint, 2));
213  }
214 
215  SetVolumeOutputValue("STRESS-XX", iPoint, Node_Struc->GetStress_FEM(iPoint)[0]);
216  SetVolumeOutputValue("STRESS-YY", iPoint, Node_Struc->GetStress_FEM(iPoint)[1]);
217  SetVolumeOutputValue("STRESS-XY", iPoint, Node_Struc->GetStress_FEM(iPoint)[2]);
218  if (nDim == 3){
219  SetVolumeOutputValue("STRESS-ZZ", iPoint, Node_Struc->GetStress_FEM(iPoint)[3]);
220  SetVolumeOutputValue("STRESS-XZ", iPoint, Node_Struc->GetStress_FEM(iPoint)[4]);
221  SetVolumeOutputValue("STRESS-YZ", iPoint, Node_Struc->GetStress_FEM(iPoint)[5]);
222  }
223  SetVolumeOutputValue("VON_MISES_STRESS", iPoint, Node_Struc->GetVonMises_Stress(iPoint));
224 
225  if (config->GetTopology_Optimization()) {
226  SetVolumeOutputValue("TOPOL_DENSITY", iPoint, Node_Struc->GetAuxVar(iPoint));
227  }
228 }
