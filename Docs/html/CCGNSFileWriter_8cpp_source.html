35 void CCGNSFileWriter::Write_Data(string val_filename) {
36 
37 #ifdef HAVE_CGNS
38 
39  /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
40  val_filename.append(fileExt);
41 
42  /*--- Open the CGNS file for writing. ---*/
43  InitializeMeshFile(val_filename);
44 
45  /*--- Write point coordinates. ---*/
46  WriteField(0, "CoordinateX");
47  WriteField(1, "CoordinateY");
48  if (nDim == 3) WriteField(2, "CoordinateZ");
49 
50  /*--- Write mesh connectivity. ---*/
51  if (nDim == 2) {
52  WriteConnectivity(LINE, "Lines");
53  WriteConnectivity(TRIANGLE, "Triangles");
54  WriteConnectivity(QUADRILATERAL, "Quadrilaterals");
55  }
56  if (nDim == 3) {
57  WriteConnectivity(TRIANGLE, "Triangles");
58  WriteConnectivity(QUADRILATERAL, "Quadrilaterals");
59  WriteConnectivity(TETRAHEDRON, "Tetrahedra");
60  WriteConnectivity(PYRAMID, "Pyramids");
61  WriteConnectivity(PRISM, "Prisms");
62  WriteConnectivity(HEXAHEDRON, "Hexahedra");
63  }
64 
65  /*--- Initialize and write fields. ---*/
66  InitializeFields();
67 
68  const auto& fieldNames = dataSorter->GetFieldNames();
69  for (unsigned long i = nDim; i < fieldNames.size(); ++i) {
70  WriteField(i, fieldNames[i]);
71  }
72 
73  /*--- Close the CGNS file. ---*/
74  if (rank == MASTER_NODE) CallCGNS(cg_close(cgnsFileID));
75 
76 #endif

80 void CCGNSFileWriter::InitializeMeshFile(string val_filename) {
81  if (!dataSorter->GetConnectivitySorted()) {
82  SU2_MPI::Error("Connectivity must be sorted.", CURRENT_FUNCTION);
83  }
84 
85  nLocalPoints = dataSorter->GetnPoints();
86  nDim = dataSorter->GetnDim();
87  GlobalElem = static_cast<cgsize_t>(dataSorter->GetnElemGlobal());
88  GlobalPoint = static_cast<cgsize_t>(dataSorter->GetnPointsGlobal());
89  cumulative = 0;
90 
91  /*--- If surface file cell dimension is decreased. ---*/
92  const auto nCell = static_cast<int>(nDim - isSurface);
93 
94  if (rank == MASTER_NODE) {
95  /*--- Remove the previous file if present. ---*/
96  remove(val_filename.c_str());
97 
98  /*--- Create CGNS file and open in write mode. ---*/
99  CallCGNS(cg_open(val_filename.c_str(), CG_MODE_WRITE, &cgnsFileID));
100 
101  /*--- Create Base. ---*/
102  CallCGNS(cg_base_write(cgnsFileID, "Base", nCell, nDim, &cgnsBase));
103 
104  /*--- Create Zone. ---*/
105  array<cgsize_t, 3> zoneData;
106 
107  zoneData[0] = GlobalPoint;
108  zoneData[1] = GlobalElem;
109  zoneData[2] = 0;
110 
111  CallCGNS(cg_zone_write(cgnsFileID, cgnsBase, "Zone", zoneData.data(), Unstructured, &cgnsZone));
112  }

115 void CCGNSFileWriter::WriteField(int iField, const string& FieldName) {
116  /*--- Check if field is coordinate. ---*/
117  const bool isCoord = iField < nDim;
118 
119  /*--- Create send buffer. ---*/
120  sendBufferField.resize(nLocalPoints);
121 
122  for (unsigned long iPoint = 0; iPoint < nLocalPoints; iPoint++) {
123  sendBufferField[iPoint] = static_cast<dataPrecision>(dataSorter->GetData(iField, iPoint));
124  }
125 
126  if (rank != MASTER_NODE) {
127  SU2_MPI::Send(sendBufferField.data(), nLocalPoints * sizeof(dataPrecision), MPI_CHAR, MASTER_NODE, 0,
128  SU2_MPI::GetComm());
129  return;
130  }
131 
132  /*--- Coordinate vector is written in blocks, one for each process. ---*/
133  cgsize_t nodeBegin = 1;
134  cgsize_t nodeEnd = static_cast<cgsize_t>(nLocalPoints);
135 
136  if (isCoord) {
137  int CoordinateNumber;
138  CallCGNS(cg_coord_partial_write(cgnsFileID, cgnsBase, cgnsZone, dataType, FieldName.c_str(), &nodeBegin, &nodeEnd,
139  sendBufferField.data(), &CoordinateNumber));
140  } else {
141  int fieldNumber;
142  CallCGNS(cg_field_partial_write(cgnsFileID, cgnsBase, cgnsZone, cgnsFields, dataType, FieldName.c_str(), &nodeBegin,
143  &nodeEnd, sendBufferField.data(), &fieldNumber));
144  }
145 
146  for (int i = 0; i < size; ++i) {
147  if (i == MASTER_NODE) continue;
148  /*--- In CGNS numbering starts form 1 and ranges are inclusive ---*/
149  nodeBegin = static_cast<cgsize_t>(dataSorter->GetnPointCumulative(i) + 1);
150  nodeEnd = static_cast<cgsize_t>(dataSorter->GetnPointCumulative(i + 1));
151 
152  const auto recvSize = static_cast<int>(nodeEnd - nodeBegin + 1);
153  recvBufferField.resize(recvSize);
154 
155  SU2_MPI::Recv(recvBufferField.data(), recvSize * sizeof(dataPrecision), MPI_CHAR, i, 0, SU2_MPI::GetComm(),
156  MPI_STATUS_IGNORE);
157  if (recvSize <= 0) continue;
158  if (isCoord) {
159  int CoordinateNumber;
160  CallCGNS(cg_coord_partial_write(cgnsFileID, cgnsBase, cgnsZone, dataType, FieldName.c_str(), &nodeBegin, &nodeEnd,
161  recvBufferField.data(), &CoordinateNumber));
162  } else {
163  int fieldNumber;
164  CallCGNS(cg_field_partial_write(cgnsFileID, cgnsBase, cgnsZone, cgnsFields, dataType, FieldName.c_str(),
165  &nodeBegin, &nodeEnd, recvBufferField.data(), &fieldNumber));
166  }
167  }

170 void CCGNSFileWriter::WriteConnectivity(GEO_TYPE type, const string& SectionName) {
171  const auto nTotElem = dataSorter->GetnElemGlobal(type);
172  if (nTotElem == 0) return;
173 
174  /*--- Create a new CGNS node to store connectivity. ---*/
175  const auto elementType = GetCGNSType(type);
176 
177  cgsize_t firstElem = cumulative + 1;
178  cgsize_t endElem = cumulative + static_cast<cgsize_t>(nTotElem);
179 
180  int cgnsSection;
181  if (rank == MASTER_NODE)
182  CallCGNS(cg_section_partial_write(cgnsFileID, cgnsBase, cgnsZone, SectionName.c_str(), elementType, firstElem,
183  endElem, 0, &cgnsSection));
184 
185  /*--- Retrieve element distribution among processes. ---*/
186  const auto nLocalElem = dataSorter->GetnElem(type);
187 
188  vector<unsigned long> distElem(size);
189 
190  SU2_MPI::Allgather(&nLocalElem, 1, MPI_UNSIGNED_LONG, distElem.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
191 
192  firstElem = cumulative + 1;
193  endElem = cumulative + static_cast<cgsize_t>(distElem[rank]);
194 
195  /*--- Connectivity is stored in send buffer. ---*/
196  const auto nPointsElem = nPointsOfElementType(type);
197  sendBufferConnectivity.resize(nLocalElem * nPointsElem);
198 
199  for (unsigned long iElem = 0; iElem < nLocalElem; iElem++) {
200  for (unsigned long iPoint = 0; iPoint < nPointsElem; iPoint++) {
201  sendBufferConnectivity[iPoint + nPointsElem * iElem] =
202  static_cast<cgsize_t>(dataSorter->GetElem_Connectivity(type, iElem, iPoint));
203  }
204  }
205 
206  const auto bufferSize = static_cast<int>(nLocalElem * nPointsElem * sizeof(cgsize_t));
207  if (rank != MASTER_NODE) {
208  SU2_MPI::Send(sendBufferConnectivity.data(), bufferSize, MPI_CHAR, MASTER_NODE, 1, SU2_MPI::GetComm());
209  return;
210  }
211 
212  /*--- Connectivity vector is written in blocks, one for each process. ---*/
213  if (nLocalElem > 0)
214  CallCGNS(cg_elements_partial_write(cgnsFileID, cgnsBase, cgnsZone, cgnsSection, firstElem, endElem,
215  sendBufferConnectivity.data()));
216 
217  for (int i = 0; i < size; ++i) {
218  if (i == MASTER_NODE) continue;
219  /*--- In CGNS numbering starts form 1 and ranges are inclusive ---*/
220  firstElem = endElem + 1;
221  endElem += static_cast<cgsize_t>(distElem[i]);
222  const auto recvSize = static_cast<int>((endElem - firstElem + 1) * nPointsElem);
223  recvBufferConnectivity.resize(recvSize);
224 
225  const auto recvByte = static_cast<int>(recvBufferConnectivity.size() * sizeof(cgsize_t));
226  SU2_MPI::Recv(recvBufferConnectivity.data(), recvByte, MPI_CHAR, i, 1, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
227 
228  if (!recvBufferConnectivity.empty())
229  CallCGNS(cg_elements_partial_write(cgnsFileID, cgnsBase, cgnsZone, cgnsSection, firstElem, endElem,
230  recvBufferConnectivity.data()));
231  }
232  cumulative += static_cast<cgsize_t>(nTotElem);
233 }
