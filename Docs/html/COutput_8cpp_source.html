265 void COutput::SetupCustomHistoryOutput(const std::string& expression, CustomHistoryOutput& output) const {
266 
267  std::vector<std::string> symbols;
268  output.expression = mel::Parse<passivedouble>(expression, symbols);
269 
270  output.symbolValues.reserve(symbols.size());
271  for (const auto& symbol : symbols) {
272  const auto* ptr = GetPtrToHistoryOutput(symbol);
273  if (ptr == nullptr) {
274  SU2_MPI::Error(std::string("Invalid history output (") + symbol + std::string(") used in expression:\n") +
275  expression, CURRENT_FUNCTION);
276  }
277  output.symbolValues.push_back(ptr);
278  }
279  output.ready = true;

990 bool COutput::Convergence_Monitoring(CConfig *config, unsigned long Iteration) {
991 
992  convergence = true;
993 
994  for (auto iField_Conv = 0ul; iField_Conv < convFields.size(); iField_Conv++) {
995 
996  const auto& convField = convFields[iField_Conv];
997  const auto it = historyOutput_Map.find(convField);
998 
999  if (it == historyOutput_Map.end()) continue;
1000 
1001  const auto& field = it->second;
1002  const su2double monitor = field.value;
1003 
1004  /*--- Stop the simulation in case a nan appears, do not save the solution. ---*/
1005  if (std::isnan(SU2_TYPE::GetValue(monitor))) {
1006  SU2_MPI::Error("SU2 has diverged (NaN detected).", CURRENT_FUNCTION);
1007  }
1008 
1009  bool fieldConverged = false;
1010 
1011  switch (field.fieldType) {
1012 
1013  /*--- Cauchy based convergence criteria ---*/
1014  case HistoryFieldType::COEFFICIENT: {
1015 
1016  if (Iteration == 0) {
1017  for (auto iCounter = 0ul; iCounter < nCauchy_Elems; iCounter++) {
1018  cauchySerie[iField_Conv][iCounter] = 0.0;
1019  }
1020  newFunc[iField_Conv] = monitor;
1021  }
1022 
1023  oldFunc[iField_Conv] = newFunc[iField_Conv];
1024  newFunc[iField_Conv] = monitor;
1025  /*--- Automatically modify the scaling factor of relative Cauchy convergence for
1026  * coefficients that are close to zero. Example: For the clean aircraft, the rolling
1027  * moment coefficient MOMENT_X is close to zero and thus will never reach a relative
1028  * cauchy convergence ->> dividing tiny numbers is not a good idea. Using absolute
1029  * cauchy convergence is more robust in this case. ---*/
1030  cauchyFunc = fabs(newFunc[iField_Conv] - oldFunc[iField_Conv]) / fmax(fabs(monitor), 0.1);
1031 
1032  cauchySerie[iField_Conv][Iteration % nCauchy_Elems] = cauchyFunc;
1033  cauchyValue = 0.0;
1034  for (auto iCounter = 0ul; iCounter < nCauchy_Elems; iCounter++)
1035  cauchyValue += cauchySerie[iField_Conv][iCounter];
1036 
1037  cauchyValue /= nCauchy_Elems;
1038 
1039  /*--- Start monitoring only if the current iteration
1040  * is larger than the number of cauchy elements. --- */
1041  fieldConverged = (cauchyValue < cauchyEps) && (Iteration >= nCauchy_Elems);
1042 
1043  if (Iteration == 0) cauchyValue = 1.0;
1044  SetHistoryOutputValue("CAUCHY_" + convField, cauchyValue);
1045 
1046  } break;
1047 
1048 
1049  /*--- Residual based convergence criteria ---*/
1050  case HistoryFieldType::RESIDUAL:
1051  case HistoryFieldType::AUTO_RESIDUAL:
1052 
1053  fieldConverged = (Iteration != 0) && (monitor <= minLogResidual);
1054  break;
1055 
1056  default:
1057  break;
1058  }
1059 
1060  convergence = fieldConverged && convergence;
1061  }
1062 
1063  /*--- Do not apply any convergence criteria if the number
1064  * of iterations is less than a particular value. ---*/
1065 
1066  if (convFields.empty() || Iteration < config->GetStartConv_Iter()) convergence = false;
1067 
1068  /*--- Apply the same convergence criteria to all processors. ---*/
1069 
1070  unsigned short local = convergence, global = 0;
1071  SU2_MPI::Allreduce(&local, &global, 1, MPI_UNSIGNED_SHORT, MPI_MAX, SU2_MPI::GetComm());
1072  convergence = global > 0;
1073 
1074  return convergence;
1075 }
