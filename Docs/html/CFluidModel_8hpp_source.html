143  inline virtual su2double GetLaminarViscosity() {
144  LaminarViscosity->SetViscosity(Temperature, Density);
145  Mu = LaminarViscosity->GetViscosity();
146  dmudrho_T = LaminarViscosity->Getdmudrho_T();
147  dmudT_rho = LaminarViscosity->GetdmudT_rho();
148  return Mu;
149  }

151  /*!
152  * \brief Get fluid thermal conductivity.
153  */
154  inline virtual su2double GetThermalConductivity() {
155  ThermalConductivity->SetConductivity(Temperature, Density, Mu, Mu_Turb, Cp, dmudrho_T, dmudT_rho);
156  Kt = ThermalConductivity->GetConductivity();
157  dktdrho_T = ThermalConductivity->Getdktdrho_T();
158  dktdT_rho = ThermalConductivity->GetdktdT_rho();
159  return Kt;
160  }
