90 /*!
91  * \brief Compute the coordinate (c) where the line defined by coordinate l0 and
92  * direction d intersects the plane defined by point p0 and normal n.
93  * \return The intersection distance.
94  */
95 template<class T, int nDim>
96 inline T LinePlaneIntersection(const T* l0, const T* d, const T* p0, const T* n, T* c) {
97  T dist[nDim] = {0.0};
98  Distance(nDim, p0, l0, dist);
99  T alpha = DotProduct(nDim, dist, n) / DotProduct(nDim, d, n);
100  for (int iDim = 0; iDim < nDim; ++iDim)
101  c[iDim] = l0[iDim] + alpha * d[iDim];
102  return fabs(alpha) * Norm(nDim,d);
103 }

122 /*! \brief Normal vector of a triangle, cross product of two sides. */
123 template<class T, class U>
124 inline void TriangleNormal(const T& coords, U* normal) {
125 
126  U a[3], b[3];
127 
128  for (int iDim = 0; iDim < 3; iDim++) {
129  a[iDim] = coords[1][iDim] - coords[0][iDim];
130  b[iDim] = coords[2][iDim] - coords[0][iDim];
131  }
132 
133  CrossProduct(a, b, normal);
134  normal[0] *= 0.5; normal[1] *= 0.5; normal[2] *= 0.5;
135 }

152 /*!
153  * \brief Compute a 3D rotation matrix.
154  * \note The implicit ordering is rotation about the x, y, and then z axis.
155  */
156 template<class Scalar, class Matrix>
157 inline void RotationMatrix(Scalar theta, Scalar phi, Scalar psi, Matrix& mat) {
158 
159  Scalar cosTheta = cos(theta); Scalar cosPhi = cos(phi); Scalar cosPsi = cos(psi);
160  Scalar sinTheta = sin(theta); Scalar sinPhi = sin(phi); Scalar sinPsi = sin(psi);
161 
162  mat[0][0] = cosPhi*cosPsi;
163  mat[1][0] = cosPhi*sinPsi;
164  mat[2][0] = -sinPhi;
165 
166  mat[0][1] = sinTheta*sinPhi*cosPsi - cosTheta*sinPsi;
167  mat[1][1] = sinTheta*sinPhi*sinPsi + cosTheta*cosPsi;
168  mat[2][1] = sinTheta*cosPhi;
169 
170  mat[0][2] = cosTheta*sinPhi*cosPsi + sinTheta*sinPsi;
171  mat[1][2] = cosTheta*sinPhi*sinPsi - sinTheta*cosPsi;
172  mat[2][2] = cosTheta*cosPhi;
173 }

186 /*! \brief Apply a rotation matrix (R) about origin (O) to a point at
187  * distance (d) from it to obtain new coordinate (c). */
188 template<class Scalar, int nDim>
189 inline void Rotate(const Scalar R[][nDim], const Scalar* O, const Scalar* d, Scalar* c) {
190 
191  for (int iDim = 0; iDim < nDim; ++iDim) {
192  c[iDim] = O[iDim];
193  for (int k = 0; k < nDim; ++k) c[iDim] += R[iDim][k] * d[k];
194  }
195 }

197 /*! \brief Tangent projection */
198 template<class Mat, class Scalar, class Int>
199 inline void TangentProjection(Int nDim, const Mat& tensor, const Scalar* vector, Scalar* proj) {
200 
201  for (Int iDim = 0; iDim < nDim; iDim++)
202  proj[iDim] = DotProduct(nDim, tensor[iDim], vector);
203 
204  auto normalProj = DotProduct(nDim, proj, vector);
205 
206  for (Int iDim = 0; iDim < nDim; iDim++)
207  proj[iDim] -= normalProj * vector[iDim];
208 }
