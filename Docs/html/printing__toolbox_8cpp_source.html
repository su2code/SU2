99 void PrintingToolbox::CTablePrinter::PrintHeader(){
100 
101  if (print_header_top_line_) PrintHorizontalLine();
102  *out_stream_ << separator_;
103  int indent = 0;
104  for (int i=0; i<GetNumColumns(); ++i){
105 
106  std::stringstream ss;
107 
108  ss << column_headers_.at(i).substr(0, column_widths_.at(i));
109 
110  indent = 0;
111 
112  if(align_ == LEFT)
113  *out_stream_ << std::left;
114  else if (align_ == RIGHT)
115  *out_stream_ << std::right;
116  else if (align_ == CENTER) {
117  *out_stream_ << std::right;
118  indent = (int)(column_widths_.at(i) - ss.str().size()) / 2;
119  }
120 
121  *out_stream_ << std::setw(column_widths_.at(i) - indent) << column_headers_.at(i).substr(0, column_widths_.at(i));
122  if (i != GetNumColumns()-1){
123  *out_stream_ << std::setw(indent+(int)inner_separator_.size()) << inner_separator_;
124  }
125  }
126 
127  *out_stream_ << std::setw((int)separator_.size()+indent) << separator_;
128  *out_stream_ << "\n";
129  if (print_header_bottom_line_){
130  PrintHorizontalLine();
131  }
132 }
