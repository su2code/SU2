71 void CNSSolver::Preprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh,
72  unsigned short iRKStep, unsigned short RunTime_EqSystem, bool Output) {
73 
74  const auto InnerIter = config->GetInnerIter();
75  const bool muscl = config->GetMUSCL_Flow() && (iMesh == MESH_0);
76  const bool center = (config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED);
77  const bool limiter = (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE) && (InnerIter <= config->GetLimiterIter());
78  const bool van_albada = (config->GetKind_SlopeLimit_Flow() == LIMITER::VAN_ALBADA_EDGE);
79  const bool wall_functions = config->GetWall_Functions();
80 
81  /*--- Common preprocessing steps (implemented by CEulerSolver) ---*/
82 
83  CommonPreprocessing(geometry, solver_container, config, iMesh, iRKStep, RunTime_EqSystem, Output);
84 
85  /*--- Compute gradient for MUSCL reconstruction, for output (i.e. the
86  turbulence solver, and post) only temperature and velocity are needed ---*/
87 
88  const auto nPrimVarGrad_bak = nPrimVarGrad;
89  if (Output) ompMasterAssignBarrier(nPrimVarGrad, 1+nDim);
90 
91  if (config->GetReconstructionGradientRequired() && muscl && !center) {
92  switch (config->GetKind_Gradient_Method_Recon()) {
93  case GREEN_GAUSS:
94  SetPrimitive_Gradient_GG(geometry, config, true); break;
95  case LEAST_SQUARES:
96  case WEIGHTED_LEAST_SQUARES:
97  SetPrimitive_Gradient_LS(geometry, config, true); break;
98  default: break;
99  }
100  }
101 
102  /*--- Compute gradient of the primitive variables ---*/
103 
104  if (config->GetKind_Gradient_Method() == GREEN_GAUSS) {
105  SetPrimitive_Gradient_GG(geometry, config);
106  }
107  else if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) {
108  SetPrimitive_Gradient_LS(geometry, config);
109  }
110 
111  if (Output) ompMasterAssignBarrier(nPrimVarGrad, nPrimVarGrad_bak);
112 
113  /*--- Compute the limiters ---*/
114 
115  if (muscl && !center && limiter && !van_albada && !Output) {
116  SetPrimitive_Limiter(geometry, config);
117  }
118 
119  ComputeVorticityAndStrainMag(*config, geometry, iMesh);
120 
121  /*--- Compute the TauWall from the wall functions ---*/
122 
123  if (wall_functions) {
124  SetTau_Wall_WF(geometry, solver_container, config);
125  }

180 void CNSSolver::Buffet_Monitoring(const CGeometry *geometry, const CConfig *config) {
181 
182  unsigned long iVertex, iMarker;
183  unsigned short iMarker_Monitoring;
184  const su2double* Vel_FS = Velocity_Inf;
185  const su2double k = config->GetBuffet_k(), lam = config->GetBuffet_lambda(), Sref = config->GetRefArea();
186 
187  const su2double VelMag_FS = GeometryToolbox::Norm(nDim, Vel_FS);
188 
189  /*-- Variables initialization ---*/
190 
191  Total_Buffet_Metric = 0.0;
192 
193  for (iMarker_Monitoring = 0; iMarker_Monitoring < config->GetnMarker_Monitoring(); iMarker_Monitoring++) {
194  Surface_Buffet_Metric[iMarker_Monitoring] = 0.0;
195  }
196 
197  /*--- Loop over the Euler and Navier-Stokes markers ---*/
198 
199  for (iMarker = 0; iMarker < nMarker; iMarker++) {
200 
201  Buffet_Metric[iMarker] = 0.0;
202 
203  const auto Monitoring = config->GetMarker_All_Monitoring(iMarker);
204 
205  if (config->GetViscous_Wall(iMarker)) {
206 
207  /*--- Loop over the vertices to compute the buffet sensor ---*/
208 
209  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
210 
211  /*--- Perform dot product of skin friction with freestream velocity ---*/
212 
213  const su2double SkinFrictionMag = GeometryToolbox::Norm(nDim, CSkinFriction[iMarker][iVertex]);
214  su2double SkinFrictionDot = GeometryToolbox::DotProduct(nDim, CSkinFriction[iMarker][iVertex], Vel_FS);
215 
216  /*--- Normalize the dot product ---*/
217 
218  SkinFrictionDot /= SkinFrictionMag*VelMag_FS;
219 
220  /*--- Compute Heaviside function ---*/
221 
222  Buffet_Sensor[iMarker][iVertex] = 1./(1. + exp(2.*k*(SkinFrictionDot + lam)));
223 
224  /*--- Integrate buffet sensor ---*/
225 
226  if (Monitoring == YES){
227 
228  auto Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
229  su2double Area = GeometryToolbox::Norm(nDim, Normal);
230 
231  Buffet_Metric[iMarker] += Buffet_Sensor[iMarker][iVertex]*Area/Sref;
232 
233  }
234 
235  }
236 
237  if (Monitoring == YES){
238 
239  Total_Buffet_Metric += Buffet_Metric[iMarker];
240 
241  /*--- Per surface buffet metric ---*/
242 
243  for (iMarker_Monitoring = 0; iMarker_Monitoring < config->GetnMarker_Monitoring(); iMarker_Monitoring++) {
244  auto Monitoring_Tag = config->GetMarker_Monitoring_TagBound(iMarker_Monitoring);
245  auto Marker_Tag = config->GetMarker_All_TagBound(iMarker);
246  if (Marker_Tag == Monitoring_Tag)
247  Surface_Buffet_Metric[iMarker_Monitoring] = Buffet_Metric[iMarker];
248  }
249 
250  }
251 
252  }
253 
254  }
255 
256  /*--- Add buffet metric information using all the nodes ---*/
257 
258  su2double MyTotal_Buffet_Metric = Total_Buffet_Metric;
259  SU2_MPI::Allreduce(&MyTotal_Buffet_Metric, &Total_Buffet_Metric, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
260 
261  /*--- Add the buffet metric on the surfaces using all the nodes ---*/
262 
263  auto local_copy = Surface_Buffet_Metric;
264  SU2_MPI::Allreduce(local_copy.data(), Surface_Buffet_Metric.data(), local_copy.size(), MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
265 
266 }

319 void CNSSolver::AddDynamicGridResidualContribution(unsigned long iPoint, unsigned long Point_Normal,
320  const CGeometry* geometry, const su2double* UnitNormal,
321  su2double Area, const su2double* GridVel,
322  su2double** Jacobian_i, su2double& Res_Conv,
323  su2double& Res_Visc) const {
324 
325  su2double ProjGridVel = Area * GeometryToolbox::DotProduct(nDim, GridVel, UnitNormal);
326 
327  /*--- Retrieve other primitive quantities and viscosities ---*/
328 
329  su2double Density = nodes->GetDensity(iPoint);
330  su2double Pressure = nodes->GetPressure(iPoint);
331  su2double laminar_viscosity = nodes->GetLaminarViscosity(iPoint);
332  su2double eddy_viscosity = nodes->GetEddyViscosity(iPoint);
333  su2double total_viscosity = laminar_viscosity + eddy_viscosity;
334 
335  /*--- Compute the viscous stress tensor ---*/
336 
337  su2double tau[MAXNDIM][MAXNDIM] = {{0.0}};
338  CNumerics::ComputeStressTensor(nDim, tau, nodes->GetVelocityGradient(iPoint), total_viscosity);
339 
340  /*--- Dot product of the stress tensor with the grid velocity ---*/
341 
342  su2double tau_vel[MAXNDIM] = {0.0};
343  for (auto iDim = 0u; iDim < nDim; iDim++)
344  tau_vel[iDim] = GeometryToolbox::DotProduct(nDim, tau[iDim], GridVel);
345 
346  /*--- Compute the convective and viscous residuals (energy eqn.) ---*/
347 
348  Res_Conv += Pressure*ProjGridVel;
349  Res_Visc += GeometryToolbox::DotProduct(nDim, tau_vel, UnitNormal) * Area;
350 
351  /*--- Implicit Jacobian contributions due to moving walls ---*/
352 
353  if (Jacobian_i != nullptr) {
354 
355  /*--- Jacobian contribution related to the pressure term ---*/
356 
357  su2double GridVel2 = GeometryToolbox::SquaredNorm(nDim, GridVel);
358 
359  Jacobian_i[nDim+1][0] += 0.5*(Gamma-1.0)*GridVel2*ProjGridVel;
360 
361  for (auto jDim = 0u; jDim < nDim; jDim++)
362  Jacobian_i[nDim+1][jDim+1] += -(Gamma-1.0)*GridVel[jDim]*ProjGridVel;
363 
364  Jacobian_i[nDim+1][nDim+1] += (Gamma-1.0)*ProjGridVel;
365 
366  /*--- Now the Jacobian contribution related to the shear stress ---*/
367 
368  /*--- Get coordinates of i & nearest normal and compute distance ---*/
369 
370  const auto Coord_i = geometry->nodes->GetCoord(iPoint);
371  const auto Coord_j = geometry->nodes->GetCoord(Point_Normal);
372 
373  su2double dist_ij = GeometryToolbox::Distance(nDim, Coord_i, Coord_j);
374 
375  const su2double theta2 = 1.0;
376 
377  su2double factor = total_viscosity*Area/(Density*dist_ij);
378 
379  if (nDim == 2) {
380  su2double thetax = theta2 + UnitNormal[0]*UnitNormal[0]/3.0;
381  su2double thetay = theta2 + UnitNormal[1]*UnitNormal[1]/3.0;
382 
383  su2double etaz = UnitNormal[0]*UnitNormal[1]/3.0;
384 
385  su2double pix = GridVel[0]*thetax + GridVel[1]*etaz;
386  su2double piy = GridVel[0]*etaz + GridVel[1]*thetay;
387 
388  Jacobian_i[nDim+1][0] += factor*(-pix*GridVel[0]+piy*GridVel[1]);
389  Jacobian_i[nDim+1][1] += factor*pix;
390  Jacobian_i[nDim+1][2] += factor*piy;
391  }
392  else {
393  su2double thetax = theta2 + UnitNormal[0]*UnitNormal[0]/3.0;
394  su2double thetay = theta2 + UnitNormal[1]*UnitNormal[1]/3.0;
395  su2double thetaz = theta2 + UnitNormal[2]*UnitNormal[2]/3.0;
396 
397  su2double etaz = UnitNormal[0]*UnitNormal[1]/3.0;
398  su2double etax = UnitNormal[1]*UnitNormal[2]/3.0;
399  su2double etay = UnitNormal[0]*UnitNormal[2]/3.0;
400 
401  su2double pix = GridVel[0]*thetax + GridVel[1]*etaz + GridVel[2]*etay;
402  su2double piy = GridVel[0]*etaz + GridVel[1]*thetay + GridVel[2]*etax;
403  su2double piz = GridVel[0]*etay + GridVel[1]*etax + GridVel[2]*thetaz;
404 
405  Jacobian_i[nDim+1][0] += factor*(-pix*GridVel[0]+piy*GridVel[1]+piz*GridVel[2]);
406  Jacobian_i[nDim+1][1] += factor*pix;
407  Jacobian_i[nDim+1][2] += factor*piy;
408  Jacobian_i[nDim+1][3] += factor*piz;
409  }
410  }

424 void CNSSolver::BC_HeatFlux_Wall_Generic(const CGeometry* geometry, const CConfig* config, unsigned short val_marker,
425  unsigned short kind_boundary) {
426  /*--- Identify the boundary by string name and get the specified wall
427  heat flux from config as well as the wall function treatment. ---*/
428 
429  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
430  const auto Marker_Tag = config->GetMarker_All_TagBound(val_marker);
431 
432  /*--- Get the specified wall heat flux, temperature or heat transfer coefficient from config ---*/
433 
434  su2double Wall_HeatFlux = 0.0, Tinfinity = 0.0, Transfer_Coefficient = 0.0;
435 
436  if (kind_boundary == HEAT_FLUX) {
437  Wall_HeatFlux = config->GetWall_HeatFlux(Marker_Tag) / config->GetHeat_Flux_Ref();
438  if (config->GetIntegrated_HeatFlux()) {
439  Wall_HeatFlux /= geometry->GetSurfaceArea(config, val_marker);
440  }
441  } else if (kind_boundary == HEAT_TRANSFER) {
442  /*--- The required heatflux will be computed for each iPoint individually based on local Temperature. ---*/
443  Transfer_Coefficient = config->GetWall_HeatTransfer_Coefficient(Marker_Tag) * config->GetTemperature_Ref() /
444  config->GetHeat_Flux_Ref();
445  Tinfinity = config->GetWall_HeatTransfer_Temperature(Marker_Tag) / config->GetTemperature_Ref();
446  }
447 
448 // Wall_Function = config->GetWallFunction_Treatment(Marker_Tag);
449 // if (Wall_Function != WALL_FUNCTION::NONE) {
450 // SU2_MPI::Error("Wall function treatment not implemented yet", CURRENT_FUNCTION);
451 // }
452 
453  /*--- Jacobian, initialized to zero if needed. ---*/
454  su2double **Jacobian_i = nullptr;
455  if ((dynamic_grid || (kind_boundary == HEAT_TRANSFER)) && implicit) {
456  Jacobian_i = new su2double* [nVar];
457  for (auto iVar = 0u; iVar < nVar; iVar++)
458  Jacobian_i[iVar] = new su2double [nVar] ();
459  }
460 
461  /*--- Loop over all of the vertices on this boundary marker ---*/
462 
463  SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
464  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
465 
466  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
467 
468  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
469 
470  if (!geometry->nodes->GetDomain(iPoint)) continue;
471 
472  /*--- If it is a customizable patch, retrieve the specified wall heat flux. ---*/
473 
474  if (config->GetMarker_All_PyCustom(val_marker))
475  Wall_HeatFlux = geometry->GetCustomBoundaryHeatFlux(val_marker, iVertex);
476  else if (kind_boundary == HEAT_TRANSFER) {
477  const su2double Twall = nodes->GetTemperature(iPoint);
478  Wall_HeatFlux = Transfer_Coefficient * (Tinfinity - Twall);
479  }
480 
481  /*--- Compute dual-grid area and boundary normal ---*/
482 
483  const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
484 
485  su2double Area = GeometryToolbox::Norm(nDim, Normal);
486 
487  su2double UnitNormal[MAXNDIM] = {0.0};
488  for (auto iDim = 0u; iDim < nDim; iDim++)
489  UnitNormal[iDim] = -Normal[iDim]/Area;
490 
491  /*--- Apply a weak boundary condition for the energy equation.
492  Compute the residual due to the prescribed heat flux.
493  The convective part will be zero if the grid is not moving. ---*/
494 
495  su2double Res_Conv = 0.0;
496  su2double Res_Visc = Wall_HeatFlux * Area;
497 
498  /*--- Impose the value of the velocity as a strong boundary
499  condition (Dirichlet). Fix the velocity and remove any
500  contribution to the residual at this node. ---*/
501 
502  if (dynamic_grid) {
503  nodes->SetVelocity_Old(iPoint, geometry->nodes->GetGridVel(iPoint));
504  }
505  else {
506  su2double zero[MAXNDIM] = {0.0};
507  nodes->SetVelocity_Old(iPoint, zero);
508  }
509 
510  for (auto iDim = 0u; iDim < nDim; iDim++)
511  LinSysRes(iPoint, iDim+1) = 0.0;
512  nodes->SetVel_ResTruncError_Zero(iPoint);
513 
514  /*--- If the wall is moving, there are additional residual contributions
515  due to pressure (p v_wall.n) and shear stress (tau.v_wall.n). ---*/
516 
517  if (dynamic_grid) {
518  if (implicit) {
519  for (auto iVar = 0u; iVar < nVar; ++iVar)
520  Jacobian_i[nDim+1][iVar] = 0.0;
521  }
522 
523  const auto Point_Normal = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
524 
525  AddDynamicGridResidualContribution(iPoint, Point_Normal, geometry, UnitNormal,
526  Area, geometry->nodes->GetGridVel(iPoint),
527  Jacobian_i, Res_Conv, Res_Visc);
528  }
529 
530  /*--- Convective and viscous contributions to the residual at the wall ---*/
531 
532  LinSysRes(iPoint, nDim+1) += Res_Conv - Res_Visc;
533 
534  /*--- Enforce the no-slip boundary condition in a strong way by
535  modifying the velocity-rows of the Jacobian (1 on the diagonal).
536  And add the contributions to the Jacobian due to energy. ---*/
537 
538  if (implicit) {
539  if (kind_boundary == HEAT_TRANSFER){
540 
541  /*--- It is necessary to zero the jacobian entries of the energy equation. ---*/
542  if (!dynamic_grid)
543  for (auto iVar = 0u; iVar < nVar; ++iVar)
544  Jacobian_i[nDim+1][iVar] = 0.0;
545 
546  const su2double oneOnRho = 1.0 / nodes->GetDensity(iPoint);
547  const su2double oneOnCv = (Gamma - 1.0) / config->GetGas_ConstantND();
548  const su2double Vel2 = nodes->GetVelocity2(iPoint);
549  const su2double dTdrho = oneOnRho * ( -Tinfinity + oneOnCv * 0.5 * Vel2);
550  const su2double dTdrhoe = oneOnCv * oneOnRho;
551 
552  /*--- Total specific energy: e=c_v*T+1/2*v^2 => T=1/c_v(rho*e/rho - 1/2||rho v||^2/rho^2).
553  Together with cv=R/(gamma-1) the following Jacobian contributions for the energy equation can be derived. ---*/
554  Jacobian_i[nDim+1][0] += Transfer_Coefficient * dTdrho * Area;
555 
556  for (unsigned short iDim = 0; iDim < nDim; iDim++)
557  Jacobian_i[nDim+1][iDim+1] -= Transfer_Coefficient * dTdrhoe * nodes->GetVelocity(iPoint, iDim) * Area;
558 
559  Jacobian_i[nDim+1][nDim+1] += Transfer_Coefficient * dTdrhoe * Area;
560 
561  }
562  if (dynamic_grid || (kind_boundary == HEAT_TRANSFER)) {

616 void CNSSolver::BC_Isothermal_Wall_Generic(CGeometry *geometry, CSolver **solver_container,
617  CNumerics *conv_numerics, CNumerics *visc_numerics,
618  CConfig *config, unsigned short val_marker, bool cht_mode) {
619 
620  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
621  const su2double Temperature_Ref = config->GetTemperature_Ref();
622  const su2double Prandtl_Lam = config->GetPrandtl_Lam();
623  const su2double Prandtl_Turb = config->GetPrandtl_Turb();
624  const su2double Gas_Constant = config->GetGas_ConstantND();
625  const su2double Cp = (Gamma / Gamma_Minus_One) * Gas_Constant;
626 
627  /*--- Identify the boundary and retrieve the specified wall temperature from
628  the config (for non-CHT problems) as well as the wall function treatment. ---*/
629 
630  const auto Marker_Tag = config->GetMarker_All_TagBound(val_marker);
631  su2double Twall = 0.0;
632  if (!cht_mode) {
633  Twall = config->GetIsothermal_Temperature(Marker_Tag) / Temperature_Ref;
634  }
635 
636 // Wall_Function = config->GetWallFunction_Treatment(Marker_Tag);
637 // if (Wall_Function != WALL_FUNCTION::NONE) {
638 // SU2_MPI::Error("Wall function treatment not implemented yet", CURRENT_FUNCTION);
639 // }
640 
641  su2double **Jacobian_i = nullptr;
642  if (implicit) {
643  Jacobian_i = new su2double* [nVar];
644  for (auto iVar = 0u; iVar < nVar; iVar++)
645  Jacobian_i[iVar] = new su2double [nVar] ();
646  }
647 
648  /*--- Loop over boundary points ---*/
649 
650  SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
651  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
652 
653  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
654 
655  if (!geometry->nodes->GetDomain(iPoint)) continue;
656 
657  /*--- Compute dual-grid area and boundary normal ---*/
658 
659  const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
660 
661  su2double Area = GeometryToolbox::Norm(nDim, Normal);
662 
663  su2double UnitNormal[MAXNDIM] = {0.0};
664  for (auto iDim = 0u; iDim < nDim; iDim++)
665  UnitNormal[iDim] = -Normal[iDim]/Area;
666 
667  /*--- Compute closest normal neighbor ---*/
668 
669  const auto Point_Normal = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
670 
671  /*--- Get coordinates of i & nearest normal and compute distance ---*/
672 
673  const auto Coord_i = geometry->nodes->GetCoord(iPoint);
674  const auto Coord_j = geometry->nodes->GetCoord(Point_Normal);
675 
676  su2double dist_ij = GeometryToolbox::Distance(nDim, Coord_i, Coord_j);
677 
678  /*--- Store the corrected velocity at the wall which will
679  be zero (v = 0), unless there is grid motion (v = u_wall)---*/
680 
681  if (dynamic_grid) {
682  nodes->SetVelocity_Old(iPoint, geometry->nodes->GetGridVel(iPoint));
683  }
684  else {
685  su2double zero[MAXNDIM] = {0.0};
686  nodes->SetVelocity_Old(iPoint, zero);
687  }
688 
689  for (auto iDim = 0u; iDim < nDim; iDim++)
690  LinSysRes(iPoint, iDim+1) = 0.0;
691  nodes->SetVel_ResTruncError_Zero(iPoint);
692 
693  /*--- Get transport coefficients ---*/
694 
695  su2double laminar_viscosity = nodes->GetLaminarViscosity(iPoint);
696  su2double eddy_viscosity = nodes->GetEddyViscosity(iPoint);
697  su2double thermal_conductivity = Cp * (laminar_viscosity/Prandtl_Lam + eddy_viscosity/Prandtl_Turb);
698 
699  // work in progress on real-gases...
700  //thermal_conductivity = nodes->GetThermalConductivity(iPoint);
701  //Cp = nodes->GetSpecificHeatCp(iPoint);
702  //thermal_conductivity += Cp*eddy_viscosity/Prandtl_Turb;
703 
704  /*--- If it is a customizable or CHT patch, retrieve the specified wall temperature. ---*/
705 
706  const su2double There = nodes->GetTemperature(Point_Normal);
707 
708  if (cht_mode) {
709  Twall = GetCHTWallTemperature(config, val_marker, iVertex, dist_ij,
710  thermal_conductivity, There, Temperature_Ref);
711  }
712  else if (config->GetMarker_All_PyCustom(val_marker)) {
713  Twall = geometry->GetCustomBoundaryTemperature(val_marker, iVertex);
714  }
715 
716  /*--- Compute the normal gradient in temperature using Twall ---*/
717 
718  su2double dTdn = -(There - Twall)/dist_ij;
719 
720  /*--- Apply a weak boundary condition for the energy equation.
721  Compute the residual due to the prescribed heat flux. ---*/
722 
723  su2double Res_Conv = 0.0;
724  su2double Res_Visc = thermal_conductivity * dTdn * Area;
725 
726  /*--- Calculate Jacobian for implicit time stepping ---*/
727 
728  if (implicit) {
729 
730  /*--- Add contributions to the Jacobian from the weak enforcement of the energy equations. ---*/
731 
732  su2double Density = nodes->GetDensity(iPoint);
733  su2double Vel2 = GeometryToolbox::SquaredNorm(nDim, &nodes->GetPrimitive(iPoint)[prim_idx.Velocity()]);
734  su2double dTdrho = 1.0/Density * ( -Twall + (Gamma-1.0)/Gas_Constant*(Vel2/2.0) );
735 
736  Jacobian_i[nDim+1][0] = thermal_conductivity/dist_ij * dTdrho * Area;
737 
738  for (auto jDim = 0u; jDim < nDim; jDim++)
739  Jacobian_i[nDim+1][jDim+1] = 0.0;
740 
741  Jacobian_i[nDim+1][nDim+1] = thermal_conductivity/dist_ij * (Gamma-1.0)/(Gas_Constant*Density) * Area;
742  }
743 
744  /*--- If the wall is moving, there are additional residual contributions
745  due to pressure (p v_wall.n) and shear stress (tau.v_wall.n). ---*/
746 
747  if (dynamic_grid) {
748  AddDynamicGridResidualContribution(iPoint, Point_Normal, geometry, UnitNormal,
749  Area, geometry->nodes->GetGridVel(iPoint),
750  Jacobian_i, Res_Conv, Res_Visc);
751  }
752 
753  /*--- Convective and viscous contributions to the residual at the wall ---*/
754 
755  LinSysRes(iPoint, nDim+1) += Res_Conv - Res_Visc;
756 
757  /*--- Enforce the no-slip boundary condition in a strong way by
758  modifying the velocity-rows of the Jacobian (1 on the diagonal).
759  And add the contributions to the Jacobian due to energy. ---*/
760 
761  if (implicit) {
762  Jacobian.AddBlock2Diag(iPoint, Jacobian_i);
763 
764  for (auto iVar = 1u; iVar <= nDim; iVar++) {
765  auto total_index = iPoint*nVar+iVar;
766  Jacobian.DeleteValsRowi(total_index);
767  }
768  }
769  }
770  END_SU2_OMP_FOR
771 
772  if (Jacobian_i)
773  for (auto iVar = 0u; iVar < nVar; iVar++)
774  delete [] Jacobian_i[iVar];
775  delete [] Jacobian_i;
776

789 void CNSSolver::SetTau_Wall_WF(CGeometry *geometry, CSolver **solver_container, const CConfig *config) {
790  /*---
791  The wall function implemented herein is based on Nichols and Nelson, AIAA J. v32 n6 2004.
792  ---*/
793 
794  unsigned long notConvergedCounter = 0; /*--- counts the number of wall cells that are not converged ---*/
795  unsigned long smallYPlusCounter = 0; /*--- counts the number of wall cells where y+ < 5 ---*/
796 
797  const su2double Gas_Constant = config->GetGas_ConstantND();
798  const su2double Cp = (Gamma / Gamma_Minus_One) * Gas_Constant;
799  const unsigned short max_iter = config->GetwallModel_MaxIter();
800  const su2double relax = config->GetwallModel_RelFac();
801 
802  /*--- Compute the recovery factor
803  * use Molecular (Laminar) Prandtl number (see Nichols & Nelson, nomenclature ) ---*/
804 
805  const su2double Recovery = pow(config->GetPrandtl_Lam(), (1.0/3.0));
806 
807  /*--- Typical constants from boundary layer theory ---*/
808 
809  const su2double kappa = config->GetwallModel_Kappa();
810  const su2double B = config->GetwallModel_B();
811 
812  for (auto iMarker = 0u; iMarker < config->GetnMarker_All(); iMarker++) {
813 
814  if (!config->GetViscous_Wall(iMarker)) continue;
815 
816  /*--- Identify the boundary by string name ---*/
817 
818  const auto Marker_Tag = config->GetMarker_All_TagBound(iMarker);
819 
820  /*--- Jump to another BC if it is not wall function ---*/
821 
822  if (config->GetWallFunction_Treatment(Marker_Tag) != WALL_FUNCTIONS::STANDARD_FUNCTION)
823  continue;
824 
825  /*--- Loop over all of the vertices on this boundary marker ---*/
826 
827  SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
828  for (auto iVertex = 0u; iVertex < geometry->nVertex[iMarker]; iVertex++) {
829 
830  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
831  const auto Point_Normal = geometry->vertex[iMarker][iVertex]->GetNormal_Neighbor();
832 
833  /*--- Check if the node belongs to the domain (i.e, not a halo node)
834  * and the neighbor is not part of the physical boundary ---*/
835 
836  if (!geometry->nodes->GetDomain(iPoint)) continue;
837 
838  /*--- Get coordinates of the current vertex and nearest normal point ---*/
839 
840  const auto Coord = geometry->nodes->GetCoord(iPoint);
841  const auto Coord_Normal = geometry->nodes->GetCoord(Point_Normal);
842 
843  /*--- Compute dual-grid area and boundary normal ---*/
844 
845  const auto Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
846 
847  const su2double Area = GeometryToolbox::Norm(nDim, Normal);
848 
849  su2double UnitNormal[MAXNDIM] = {0.0};
850  for (auto iDim = 0u; iDim < nDim; iDim++)
851  UnitNormal[iDim] = -Normal[iDim]/Area;
852 
853  /*--- Get the velocity, pressure, and temperature at the nearest
854  (normal) interior point. ---*/
855 
856  su2double Vel[MAXNDIM] = {0.0};
857  for (auto iDim = 0u; iDim < nDim; iDim++)
858  Vel[iDim] = nodes->GetVelocity(Point_Normal,iDim);
859 
860  /*--- Compute the wall-parallel velocity at first point off the wall ---*/
861 
862  const su2double VelNormal = GeometryToolbox::DotProduct(int(MAXNDIM), Vel, UnitNormal);
863 
864  su2double VelTang[MAXNDIM] = {0.0};
865  for (auto iDim = 0u; iDim < nDim; iDim++)
866  VelTang[iDim] = Vel[iDim] - VelNormal*UnitNormal[iDim];
867 
868  const su2double VelTangMod = GeometryToolbox::Norm(int(MAXNDIM), VelTang);
869 
870  /*--- Compute normal distance of the interior point from the wall ---*/
871 
872  su2double WallDist[MAXNDIM] = {0.0};
873  GeometryToolbox::Distance(nDim, Coord, Coord_Normal, WallDist);
874 
875  const su2double WallDistMod = GeometryToolbox::Norm(int(MAXNDIM), WallDist);
876 
877  su2double T_Wall = nodes->GetTemperature(iPoint);
878  const su2double Conductivity_Wall = nodes->GetThermalConductivity(iPoint);
879 
880  /*--- If a wall temperature was given, we compute the local heat flux using k*dT/dn ---*/
881 
882  su2double q_w = 0.0;
883 
884  if (config->GetMarker_All_KindBC(iMarker) == HEAT_FLUX) {
885  q_w = config->GetWall_HeatFlux(Marker_Tag) / config->GetHeat_Flux_Ref();
886  }
887 
888  /*--- Extrapolate the pressure from the interior & compute the
889  wall density using the equation of state ---*/
890 
891  const su2double P_Normal = nodes->GetPressure(Point_Normal);
892  const su2double T_Normal = nodes->GetTemperature(Point_Normal);
893  const su2double P_Wall = P_Normal;
894 
895  /*--- Compressible formulation ---*/
896 
897  su2double Density_Wall = P_Wall / (Gas_Constant * T_Wall);
898  const su2double Lam_Visc_Normal = nodes->GetLaminarViscosity(Point_Normal);
899 
900  /*--- Compute the shear stress at the wall in the regular fashion
901  * by using the stress tensor on the surface ---*/
902 
903  su2double tau[MAXNDIM][MAXNDIM] = {{0.0}};
904  const su2double Lam_Visc_Wall = nodes->GetLaminarViscosity(iPoint);
905  su2double Eddy_Visc_Wall = nodes->GetEddyViscosity(iPoint);
906 
907  CNumerics::ComputeStressTensor(nDim, tau, nodes->GetVelocityGradient(iPoint), Lam_Visc_Wall);
908 
909  su2double TauTangent[MAXNDIM] = {0.0};
910  GeometryToolbox::TangentProjection(nDim, tau, UnitNormal, TauTangent);
911 
912  const su2double WallShearStress = GeometryToolbox::Norm(int(MAXNDIM), TauTangent);
913 
914  /*--- Calculate the quantities from boundary layer theory and
915  * iteratively solve for a new wall shear stress. Use the current wall
916  * shear stress as a starting guess for the wall function. ---*/
917 
918  unsigned long counter = 0;
919  su2double diff = 1.0;
920  su2double U_Tau = max(1.0e-6,sqrt(WallShearStress/Density_Wall));
921  /*--- Use minimum y+ as defined in the config, in case the routine below for computing y+ does not converge ---*/
922  su2double Y_Plus = 0.99*config->GetwallModel_MinYPlus(); // use clipping value as minimum
923 
924  const su2double Y_Plus_Start = Density_Wall * U_Tau * WallDistMod / Lam_Visc_Wall;
925 
926  /*--- Automatic switch off when y+ < "limit" according to Nichols & Nelson (2004) ---*/
927 
928  if (Y_Plus_Start < config->GetwallModel_MinYPlus()) {
929  smallYPlusCounter++;
930  continue;
931  }
932 
933  /*--- Convergence criterium for the Newton solver, note that 1e-10 is too large ---*/
934  const su2double tol = 1e-12;
935  while (fabs(diff) > tol) {
936 
937  /*--- Friction velocity and u+ ---*/
938 
939  const su2double U_Plus = VelTangMod/U_Tau;
940 
941  /*--- Gamma, Beta, Q, and Phi, defined by Nichols & Nelson (2004) page 1108 ---*/
942 
943  const su2double Gam = Recovery*U_Tau*U_Tau/(2.0*Cp*T_Wall);
944  const su2double Beta = q_w*Lam_Visc_Wall/(Density_Wall*T_Wall*Conductivity_Wall*U_Tau);
945  const su2double Q = sqrt(Beta*Beta + 4.0*Gam);
946  const su2double Phi = asin(-1.0*Beta/Q);
947 
948  /*--- Crocco-Busemann equation for wall temperature (eq. 11 of Nichols and Nelson) ---*/
949  /*--- update T_Wall due to aerodynamic heating, unless the wall is isothermal ---*/
950 
951  if (config->GetMarker_All_KindBC(iMarker) != ISOTHERMAL) {
952  const su2double denum = (1.0 + Beta*U_Plus - Gam*U_Plus*U_Plus);
953  if (denum > EPS){
954  T_Wall = T_Normal / denum;
955  nodes->SetTemperature(iPoint,T_Wall);
956  }
957  else {
958  SU2_OMP_CRITICAL
959  {
960  cout << "Warning: T_Wall < 0 " << endl;
961  }
962  END_SU2_OMP_CRITICAL
963  }
964  }
965 
966  /*--- update of wall density using the wall temperature ---*/
967  Density_Wall = P_Wall/(Gas_Constant*T_Wall);
968 
969  /*--- Y+ defined by White & Christoph (compressibility and heat transfer) negative value for (2.0*Gam*U_Plus - Beta)/Q ---*/
970 
971  const su2double Y_Plus_White = exp((kappa/sqrt(Gam))*(asin((2.0*Gam*U_Plus - Beta)/Q) - Phi))*exp(-1.0*kappa*B);
972 
973  /*--- Spalding's universal form for the BL velocity with the
974  * outer velocity form of White & Christoph above. ---*/
975  const su2double kUp = kappa*U_Plus;
976  Y_Plus = U_Plus + Y_Plus_White - (exp(-1.0*kappa*B)* (1.0 + kUp + 0.5*kUp*kUp + kUp*kUp*kUp/6.0));
977 
978  const su2double dypw_dyp = 2.0*Y_Plus_White*(kappa*sqrt(Gam)/Q)*sqrt(1.0 - pow(2.0*Gam*U_Plus - Beta,2.0)/(Q*Q));
979 
980  Eddy_Visc_Wall = Lam_Visc_Wall*(1.0 + dypw_dyp - kappa*exp(-1.0*kappa*B)*
981  (1.0 + kappa*U_Plus + kappa*kappa*U_Plus*U_Plus/2.0)
982  - Lam_Visc_Normal/Lam_Visc_Wall);
983  Eddy_Visc_Wall = max(1.0e-6, Eddy_Visc_Wall);
984 
985  /* --- Define function for Newton method to zero --- */
986 
987  diff = (Density_Wall * U_Tau * WallDistMod / Lam_Visc_Wall) - Y_Plus;
988 
989  /* --- Gradient of function defined above --- */
990 
991  const su2double grad_diff = Density_Wall * WallDistMod / Lam_Visc_Wall + VelTangMod / (U_Tau * U_Tau) +
992  kappa /(U_Tau * sqrt(Gam)) * asin(U_Plus * sqrt(Gam)) * Y_Plus_White -
993  exp(-1.0 * B * kappa) * (0.5 * pow(VelTangMod * kappa / U_Tau, 3) +
994  pow(VelTangMod * kappa / U_Tau, 2) + VelTangMod * kappa / U_Tau) / U_Tau;
995 
996  /* --- Newton Step --- */
997 
998  U_Tau = U_Tau - relax*(diff / grad_diff);
999 
1000  counter++;
1001  if (counter > max_iter) {
1002  notConvergedCounter++;
1003  // use some safe values for convergence
1004  Y_Plus = 30.0;
1005  Eddy_Visc_Wall = 1.0;
1006  U_Tau = 1.0;
1007  break;
1008  }
1009  }
1010 
1011  /*--- Calculate an updated value for the wall shear stress
1012  * using the y+ value, the definition of y+, and the definition of
1013  * the friction velocity. ---*/
1014 
1015  YPlus[iMarker][iVertex] = Y_Plus;
1016  EddyViscWall[iMarker][iVertex] = Eddy_Visc_Wall;
1017  UTau[iMarker][iVertex] = U_Tau;
1018 
1019  const su2double Tau_Wall = (1.0/Density_Wall)*pow(Y_Plus*Lam_Visc_Wall/WallDistMod,2.0);
1020 
1021  /*--- Store this value for the wall shear stress at the node. ---*/
1022 
1023  nodes->SetTau_Wall(iPoint, Tau_Wall);
1024 
1025  }
1026  END_SU2_OMP_FOR
1027  }
1028 
1029  if (config->GetComm_Level() == COMM_FULL) {
1030  static unsigned long globalCounter1, globalCounter2;
1031 
1032  ompMasterAssignBarrier(globalCounter1,0, globalCounter2,0);
1033 
1034  SU2_OMP_ATOMIC
1035  globalCounter1 += notConvergedCounter;
1036 
1037  SU2_OMP_ATOMIC
1038  globalCounter2 += smallYPlusCounter;
1039 
1040  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
1041  SU2_MPI::Allreduce(&globalCounter1, &notConvergedCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
1042  SU2_MPI::Allreduce(&globalCounter2, &smallYPlusCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
1043 
1044  if (rank == MASTER_NODE) {
1045  if (notConvergedCounter)
1046  cout << "Warning: Computation of wall coefficients (y+) did not converge in "
1047  << notConvergedCounter << " points." << endl;
1048 
1049  if (smallYPlusCounter)
1050  cout << "Warning: y+ < " << config->GetwallModel_MinYPlus() << " in " << smallYPlusCounter
1051  << " points, for which the wall model is not active." << endl;
1052  }
1053  }
1054  END_SU2_OMP_SAFE_GLOBAL_ACCESS
1055  }
