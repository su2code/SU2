130

193  restart file (without including the PointID) ---*/

202  if (rank == MASTER_NODE) {

211  MASTER_NODE, SU2_MPI::GetComm());
212 
213  fields.push_back("Point_ID");
214 
215  for (iVar = 0; iVar < nVar; iVar++) {
216  index = iVar*CGNS_STRING_SIZE;
217  field_buf.append("\"");
218  for (iChar = 0; iChar < (unsigned long)CGNS_STRING_SIZE; iChar++) {
219  str_buf[iChar] = mpi_str_buf[index + iChar];

221  field_buf.append(str_buf);
222  field_buf.append("\"");

308  string("SU2 reads/writes binary restart files by default.\n") +

314 
315 #endif

339  restart_file.close();

342  restart file (without including the PointID) ---*/
343 
344  nVar = fields.size() - 1;

350 void CBaselineSolver::LoadRestart(CGeometry **geometry, CSolver ***solver, CConfig *config, int val_iter, bool val_update_geo) {
351 
352  /*--- Restart the solution from file information ---*/
353 
354  string filename;
355  unsigned long index;
356  unsigned short iDim, iVar;
357  bool adjoint = ( config->GetContinuous_Adjoint() || config->GetDiscrete_Adjoint() );
358  unsigned short iInst = config->GetiInst();
359  bool steady_restart = config->GetSteadyRestart();
360  TURB_MODEL turb_model = config->GetKind_Turb_Model();
361 
362  su2double *Coord = new su2double [nDim];
363  for (iDim = 0; iDim < nDim; iDim++)
364  Coord[iDim] = 0.0;
365 
366  /*--- Skip coordinates ---*/
367 
368  unsigned short skipVars = geometry[iInst]->GetnDim();
369 
370  /*--- Retrieve filename from config ---*/
371 
372  if (adjoint) {
373  filename = config->GetSolution_AdjFileName();
374  filename = config->GetObjFunc_Extension(filename);
375  } else {
376  filename = config->GetSolution_FileName();
377  }
378 
379  filename = config->GetFilename(filename, "", val_iter);
380 
381  /*--- Output the file name to the console. ---*/
382 
383  if (rank == MASTER_NODE)
384  cout << "Reading and storing the solution from " << filename
385  << "." << endl;
386 
387  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
388 
389  if (config->GetRead_Binary_Restart()) {
390  Read_SU2_Restart_Binary(geometry[iInst], config, filename);
391  } else {
392  Read_SU2_Restart_ASCII(geometry[iInst], config, filename);
393  }
394 
395  int counter = 0;
396  long iPoint_Local = 0; unsigned long iPoint_Global = 0;
397 
398  /*--- Load data from the restart into correct containers. ---*/
399 
400  for (iPoint_Global = 0; iPoint_Global < geometry[iInst]->GetGlobal_nPointDomain(); iPoint_Global++ ) {
401 
402  /*--- Retrieve local index. If this node from the restart file lives
403  on the current processor, we will load and instantiate the vars. ---*/
404 
405  iPoint_Local = geometry[iInst]->GetGlobal_to_Local_Point(iPoint_Global);
406 
407  if (iPoint_Local > -1) {
408 
409  /*--- We need to store this point's data, so jump to the correct
410  offset in the buffer of data from the restart file and load it. ---*/
411 
412  index = counter*Restart_Vars[1];
413  for (iVar = 0; iVar < nVar; iVar++) Solution[iVar] = Restart_Data[index+iVar];
414  nodes->SetSolution(iPoint_Local,Solution);
415 
416  /*--- For dynamic meshes, read in and store the
417  grid coordinates and grid velocities for each node. ---*/
418 
419  if (dynamic_grid && val_update_geo) {
420 
421  /*--- First, remove any variables for the turbulence model that
422  appear in the restart file before the grid velocities. ---*/
423 
424  if (turb_model == TURB_MODEL::SA) {
425  index++;
426  } else if (turb_model == TURB_MODEL::SST) {
427  index+=2;
428  }
429 
430  /*--- Read in the next 2 or 3 variables which are the grid velocities ---*/
431  /*--- If we are restarting the solution from a previously computed static calculation (no grid movement) ---*/
432  /*--- the grid velocities are set to 0. This is useful for FSI computations ---*/
433 
434  su2double GridVel[3] = {0.0,0.0,0.0};
435  if (!steady_restart) {
436 
437  /*--- Rewind the index to retrieve the Coords. ---*/
438  index = counter*Restart_Vars[1];
439  for (iDim = 0; iDim < nDim; iDim++) { Coord[iDim] = Restart_Data[index+iDim]; }
440 
441  /*--- Move the index forward to get the grid velocities. ---*/
442  index = counter*Restart_Vars[1] + skipVars + nVar;
443  for (iDim = 0; iDim < nDim; iDim++) { GridVel[iDim] = Restart_Data[index+iDim]; }
444  }
445 
446  for (iDim = 0; iDim < nDim; iDim++) {
447  geometry[iInst]->nodes->SetCoord(iPoint_Local, iDim, Coord[iDim]);
448  geometry[iInst]->nodes->SetGridVel(iPoint_Local, iDim, GridVel[iDim]);
449  }
450  }
451 
452  /*--- Increment the overall counter for how many points have been loaded. ---*/
453  counter++;
454  }
455 
456  }
457 
458  /*--- MPI solution ---*/
459 
460  InitiateComms(geometry[iInst], config, SOLUTION);
461  CompleteComms(geometry[iInst], config, SOLUTION);
462 
463  /*--- Update the geometry for flows on dynamic meshes ---*/
464 
465  if (dynamic_grid && val_update_geo) {
466 
467  /*--- Communicate the new coordinates and grid velocities at the halos ---*/
468 
469  geometry[iInst]->InitiateComms(geometry[iInst], config, COORDINATES);
470  geometry[iInst]->CompleteComms(geometry[iInst], config, COORDINATES);
471 
472  geometry[iInst]->InitiateComms(geometry[iInst], config, GRID_VELOCITY);
473  geometry[iInst]->CompleteComms(geometry[iInst], config, GRID_VELOCITY);
474 
475  }
476 
477  delete [] Coord;
478 
479  /*--- Delete the class memory that is used to load the restart. ---*/
480 
481  delete [] Restart_Vars;
482  delete [] Restart_Data;
483  Restart_Vars = nullptr; Restart_Data = nullptr;
484 
485 }
