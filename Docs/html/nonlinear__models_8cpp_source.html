37 void CFEM_NeoHookean_Comp::Compute_Plane_Stress_Term(CElement *element, const CConfig *config) {
38 
39  su2double j_red = 1.0;
40  su2double fx = 0.0, fpx = 1.0;
41  su2double xkm1 = 1.0, xk = 1.0;
42  su2double cte = 0.0;
43 
44  unsigned short iNR, nNR;
45  su2double NRTOL;
46 
47  // Maximum number of iterations and tolerance (relative)
48  nNR = 10;
49  NRTOL = 1E-25;
50 
51  // j_red: reduced jacobian, for the 2x2 submatrix of F
52  j_red = F_Mat[0][0] * F_Mat[1][1] - F_Mat[1][0] * F_Mat[0][1];
53  // cte: constant term in the NR method
54  cte = Lambda*log(j_red) - Mu;
55 
56  // f(f33) = mu*f33^2 + lambda*ln(f33) + (lambda*ln(j_red)-mu) = 0
57  // f'(f33) = 2*mu*f33 + lambda/f33
58 
59  for (iNR = 0; iNR < nNR; iNR++) {
60  fx = Mu*pow(xk,2.0) + Lambda*log(xk) + cte;
61  fpx = 2*Mu*xk + (Lambda / xk);
62  xkm1 = xk - fx / fpx;
63  if (((xkm1 - xk) / xk) < NRTOL) break;
64  xk = xkm1;
65  }
66 
67  f33 = xkm1;
68 
69 }

71 void CFEM_NeoHookean_Comp::Compute_Constitutive_Matrix(CElement *element, const CConfig *config) {
72 
73  su2double Mu_p = 0.0, Lambda_p = 0.0;
74 
75  /*--- This can be done in a better way ---*/

77  Mu_p = (Mu - Lambda*log(J_F))/J_F;
78  Lambda_p = Lambda/J_F;

80

101 void CFEM_NeoHookean_Comp::Compute_Stress_Tensor(CElement *element, const CConfig *config) {
102 
103  unsigned short iVar,jVar;
104  su2double Mu_J = 0.0, Lambda_J = 0.0;
105 
106  /*--- This can be done in a better way ---*/
107  if (J_F != 0.0) {
108  Mu_J = Mu/J_F;
109  Lambda_J = Lambda/J_F;
110  }
111 
112  for (iVar = 0; iVar < 3; iVar++) {
113  for (jVar = 0; jVar < 3; jVar++) {
114  su2double dij = deltaij(iVar,jVar);
115  Stress_Tensor[iVar][jVar] = Mu_J * (b_Mat[iVar][jVar] - dij) + Lambda_J * log(J_F) * dij;
116  }
117  }

147 void CFEM_Knowles_NearInc::Compute_Constitutive_Matrix(CElement *element, const CConfig *config) {
148 
149  /* -- Suchocki (2011) (full reference in class constructor). ---*/
150 
151  /*--- Computation of the tensor cijkl ---*/
152 
153  unsigned short iVar, jVar, kVar, lVar;
154 
155  for (iVar = 0; iVar < 3; iVar++){
156  for (jVar = 0; jVar < 3; jVar++){
157  for (kVar = 0; kVar < 3; kVar++){
158  for (lVar = 0; lVar < 3; lVar++){
159  cijkl[iVar][jVar][kVar][lVar] =
160  term1 * ((1.0/2.0)*( deltaij(iVar,kVar)*b_Mat_Iso[jVar][lVar]
161  +deltaij(jVar,lVar)*b_Mat_Iso[iVar][kVar]
162  +deltaij(iVar,lVar)*b_Mat_Iso[jVar][kVar]
163  +deltaij(jVar,kVar)*b_Mat_Iso[iVar][lVar])
164  +(2.0/3.0)*(trbbar*deltaij(iVar,jVar)*deltaij(kVar,lVar)
165  -b_Mat_Iso[iVar][jVar]*deltaij(kVar,lVar)
166  -deltaij(iVar,jVar)*b_Mat_Iso[kVar][lVar]))
167  +term2 * ( b_Mat_Iso[iVar][jVar]*b_Mat_Iso[kVar][lVar]
168  - trbbar*(b_Mat_Iso[iVar][jVar]*deltaij(kVar,lVar)
169  +deltaij(iVar,jVar)*b_Mat_Iso[kVar][lVar])
170  + pow(trbbar,2.0) * deltaij(iVar,jVar) * deltaij(kVar,lVar))
171  +Kappa * (2.0 * J_F - 1.0) * deltaij(iVar,jVar) * deltaij(kVar,lVar);
172 
173  }
174  }
175  }
176  }
177 
178  /*--- Reorganizing the tensor into the matrix D ---*/
179 
180  Assign_cijkl_D_Mat();

185 void CFEM_Knowles_NearInc::Compute_Stress_Tensor(CElement *element, const CConfig *config) {
186 
187  /* -- Suchocki (2011) (full reference in class constructor). ---*/
188 
189  unsigned short iVar, jVar;
190 
191  /*--- Compute the isochoric deformation gradient Fbar and left Cauchy-Green tensor bbar ---*/
192  Compute_Isochoric_F_b();
193 
194  trbbar = (b_Mat_Iso[0][0] + b_Mat_Iso[1][1] + b_Mat_Iso[2][2]) / 3.0;
195  term1 = (Mu / J_F) * pow((1 + (Bk / Nk) * (3.0 * trbbar - 3.0)), (Nk-1.0));
196  term2 = 2.0 * (Mu / J_F) * (Bk * (Nk - 1.0) / Nk) *
197  pow((1.0 + (Bk / Nk) * (3.0 * trbbar - 3.0)), (Nk-2.0));
198 
199  Ek = Kappa * (2.0 * J_F - 1.0);
200  Pr = Kappa * (J_F - 1.0);
201 
202  for (iVar = 0; iVar < 3; iVar++){
203  for (jVar = 0; jVar < 3; jVar++){
204  Stress_Tensor[iVar][jVar] = term1 * (b_Mat_Iso[iVar][jVar] - deltaij(iVar,jVar)*trbbar) +
205  deltaij(iVar,jVar) * Pr;
206  }
207  }

237 void CFEM_DielectricElastomer::Compute_Stress_Tensor(CElement *element, const CConfig *config) {
238 
239  unsigned short iDim, jDim;
240 
241  su2double E0 = 0.0, E1 = 0.0, E2 = 0.0;
242  su2double E0_2 = 0.0, E1_2 = 0.0, E2_2 = 0.0;
243  su2double E_2 = 0.0;
244 
245  Compute_FmT_Mat();
246 
247  for (iDim = 0; iDim < nDim; iDim++){
248  EField_Curr_Unit[iDim] = 0.0;
249  for (jDim = 0; jDim < nDim; jDim++){
250  EField_Curr_Unit[iDim] += FmT_Mat[iDim][jDim] * EField_Ref_Unit[jDim];
251  }
252  }
253 
254  E0 = EFieldMod_Ref*EField_Curr_Unit[0]; E0_2 = pow(E0,2);
255  E1 = EFieldMod_Ref*EField_Curr_Unit[1]; E1_2 = pow(E1,2);
256  if (nDim == 3) {E2 = EFieldMod_Ref*EField_Curr_Unit[2]; E2_2 = pow(E2,2);}
257 
258  E_2 = E0_2+E1_2+E2_2;
259 
260  Stress_Tensor[0][0] = ke_DE*(E0_2-0.5*E_2); Stress_Tensor[0][1] = ke_DE*E0*E1; Stress_Tensor[0][2] = ke_DE*E0*E2;
261  Stress_Tensor[1][0] = ke_DE*E1*E0; Stress_Tensor[1][1] = ke_DE*(E1_2-0.5*E_2); Stress_Tensor[1][2] = ke_DE*E1*E2;
262  Stress_Tensor[2][0] = ke_DE*E2*E0; Stress_Tensor[2][1] = ke_DE*E2*E1; Stress_Tensor[2][2] = ke_DE*(E2_2-0.5*E_2);
263

318 void CFEM_IdealDE::Compute_Stress_Tensor(CElement *element, const CConfig *config) {
319 
320  /* -- Zhao, X. and Suo, Z. (2008) (full reference in class constructor). ---*/
321 
322  unsigned short iVar, jVar;
323  su2double dij = 0.0;
324 
325  /*--- Compute the isochoric deformation gradient Fbar and left Cauchy-Green tensor bbar ---*/
326  Compute_Isochoric_F_b();
327 
328  // Stress terms
329 
330  trbbar = (b_Mat_Iso[0][0] + b_Mat_Iso[1][1] + b_Mat_Iso[2][2]) / 3.0;
331  Eg = Mu / J_F;
332  Ek = Kappa * (2.0 * J_F - 1.0);
333  Pr = Kappa * (J_F - 1.0);
334  Eg23 = 2.0 * Eg / 3.0;
335 
336  // Stress tensor
337 
338  for (iVar = 0; iVar < 3; iVar++){
339  for (jVar = 0; jVar < 3; jVar++){
340  if (iVar == jVar) dij = 1.0;
341  else if (iVar != jVar) dij = 0.0;
342  Stress_Tensor[iVar][jVar] = Eg * ( b_Mat_Iso[iVar][jVar] - dij * trbbar) + dij * Pr ;
343  }
344  }
