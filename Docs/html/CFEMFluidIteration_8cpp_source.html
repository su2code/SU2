31 void CFEMFluidIteration::Preprocess(COutput* output, CIntegration**** integration, CGeometry**** geometry,
32  CSolver***** solver, CNumerics****** numerics, CConfig** config,
33  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
34  CFreeFormDefBox*** FFDBox, unsigned short val_iZone, unsigned short val_iInst) {
35  unsigned long TimeIter = config[ZONE_0]->GetTimeIter();
36  const bool restart = (config[ZONE_0]->GetRestart() || config[ZONE_0]->GetRestart_Flow());

38  /*--- Set the initial condition if this is not a restart. ---*/
39  if (TimeIter == 0 && !restart)
40  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->SetInitialCondition(

44 void CFEMFluidIteration::Iterate(COutput* output, CIntegration**** integration, CGeometry**** geometry,
45  CSolver***** solver, CNumerics****** numerics, CConfig** config,
46  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
47  CFreeFormDefBox*** FFDBox, unsigned short val_iZone, unsigned short val_iInst) {
48  /*--- Update global parameters ---*/
49 
50  const auto kind_solver = config[val_iZone]->GetKind_Solver();
51  switch (kind_solver) {
52  case MAIN_SOLVER::FEM_EULER:
53  case MAIN_SOLVER::DISC_ADJ_FEM_EULER:
54  case MAIN_SOLVER::FEM_NAVIER_STOKES:
55  case MAIN_SOLVER::DISC_ADJ_FEM_NS:
56  case MAIN_SOLVER::FEM_RANS:
57  case MAIN_SOLVER::DISC_ADJ_FEM_RANS:
58  case MAIN_SOLVER::FEM_LES:
59  config[val_iZone]->SetGlobalParam(kind_solver, RUNTIME_FLOW_SYS);
60  break;
61  default:
62  break;
63  }
64 
65  /*--- Solve the Euler, Navier-Stokes, RANS or LES equations (one iteration) ---*/
66 
67  integration[val_iZone][val_iInst][FLOW_SOL]->SingleGrid_Iteration(geometry, solver, numerics, config,
68  RUNTIME_FLOW_SYS, val_iZone, val_iInst);
69 }
