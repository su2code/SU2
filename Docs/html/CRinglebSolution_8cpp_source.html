88 void CRinglebSolution::GetSolution(const su2double *val_coords,
89  const su2double val_t,
90  su2double *val_solution) const {
91 
92  /* Easier storage of the coordinates and abbreviate y*y. */
93  const su2double x = val_coords[0], y = val_coords[1];
94  const su2double y2 = y*y;
95 
96  /* Initial guess for q (velocity magnitude) and k (streamline parameter). */
97  su2double k = 1.2;
98  su2double q = 1.0;
99 
100  /* Newton algorithm to solve for the variables q and k for the given x and y. */
101  const int iterMax = 500;
102  su2double duMaxPrev = 10.0;
103 
104  int iter;
105  for(iter=0; iter<iterMax; ++iter) {
106 
107  /* Compute the speed of sound, the density, the parameter JJ
108  and its derivatives w.r.t. q. */
109  const su2double a = sqrt(1.0 - 0.5*Gm1*q*q);
110  const su2double rho = pow(a,tovGm1);
111  const su2double JJ = 1.0/a + 1.0/(3.0*a*a*a) + 1.0/(5.0*a*a*a*a*a)
112  - 0.5*log((1.0+a)/(1.0-a));
113 
114  const su2double dadq = -0.5*Gm1*q/a;
115  const su2double drhodq = 2.0*rho*dadq/(Gm1*a);
116  const su2double dJJdq = dadq/(pow(a,6)*(a*a-1.0));
117 
118  /* Determine the values of the nonlinear equations to solve
119  and its corresponding Jacobian matrix. */
120  const su2double y2c = (k*k - q*q)/(k*k*k*k*rho*rho*q*q);
121  const su2double f[] = {(2.0/(k*k) - 1.0/(q*q))/(2.0*rho) - 0.5*JJ - x,
122  y2c - y2};
123  su2double Jac[2][2];
124  Jac[0][0] = -(1.0/(k*k) - 0.50/(q*q))*drhodq/(rho*rho)
125  + 1.0/(rho*q*q*q) - 0.5*dJJdq;
126  Jac[0][1] = -2.0/(rho*k*k*k);
127  Jac[1][0] = -2.0/(k*k*rho*rho*q*q*q) - 2.0*y2c*drhodq/rho;
128  Jac[1][1] = (4.0*q*q - 2.0*k*k)/(k*k*k*k*k*rho*rho*q*q);
129 
130  /* Determine the update dU. */
131  const su2double det = Jac[0][0]*Jac[1][1] - Jac[0][1]*Jac[1][0];
132  const su2double dU[] = {(f[0]*Jac[1][1] - f[1]*Jac[0][1])/det,
133  (f[1]*Jac[0][0] - f[0]*Jac[1][0])/det};
134 
135  /* Determine the underrelaxation coefficient alp. */
136  const su2double dUMax = max(fabs(dU[0]), fabs(dU[1]));
137  su2double alp = 1.0;
138  if( dUMax > 1.0) alp = 0.04;
139  else if(dUMax > 0.1) alp = 0.2;
140 
141  /* Update q and k. */
142  q -= alp*dU[0];
143  k -= alp*dU[1];
144 
145  /* Convergence check, which is independent of the precision used. */
146  if((dUMax < 1.e-3) && (dUMax >= duMaxPrev)) break;
147  duMaxPrev = dUMax;
148  }
149 
150  /* Check if the Newton algorithm actually converged. */
151  if(iter == iterMax)
152  SU2_MPI::Error("Newton algorithm did not converge", CURRENT_FUNCTION);
153 
154  /* Compute the speed of sound, density and pressure. */
155  const su2double a = sqrt(1.0 - 0.5*Gm1*q*q);
156  const su2double rho = pow(a,tovGm1);
157  const su2double p = pow(a,tGamOvGm1)/Gamma;
158 
159  /* Determine the derivative of x w.r.t. q and ydxdq. */
160  const su2double dadq = -0.5*Gm1*q/a;
161  const su2double drhodq = 2.0*rho*dadq/(Gm1*a);
162  const su2double dJJdq = dadq/(pow(a,6)*(a*a-1.0));
163 
164  const su2double dxdq = -(1.0/(k*k) - 0.5/(q*q))*drhodq/(rho*rho)
165  + 1.0/(rho*q*q*q) - 0.5*dJJdq;
166  const su2double ydxdq = y*dxdq;
167 
168  /* Determine the derivative of 1/2 y2 w.r.t. q, which is ydydq. The reason is
169  that ydydq is always well defined, while dydq is singular for y = 0. */
170  const su2double ydydq = drhodq*(q*q-k*k)/(k*k*k*k*rho*rho*rho*q*q)
171  - 1.0/(k*k*rho*rho*q*q*q);
172 
173  /* Determine the direction of the streamline. */
174  const su2double vecLen = sqrt(ydxdq*ydxdq + ydydq*ydydq);
175 
176  su2double velDir[] = {ydxdq/vecLen, ydydq/vecLen};
177  if(velDir[1] > 0.0){velDir[0] = -velDir[0]; velDir[1] = -velDir[1];}
178 
179  /* Compute the conservative variables. Note that both 2D and 3D
180  cases are treated correctly. */
181  val_solution[0] = rho;
182  val_solution[1] = rho*q*velDir[0];
183  val_solution[2] = rho*q*velDir[1];
184  val_solution[3] = 0.0;
185  val_solution[nVar-1] = p/Gm1 + 0.5*rho*q*q;
186 }
