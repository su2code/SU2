59 void CAvgGrad_AdjFlow::ComputeResidual(su2double *val_residual_i, su2double *val_residual_j,
60  su2double **val_Jacobian_ii, su2double **val_Jacobian_ij,
61  su2double **val_Jacobian_ji, su2double **val_Jacobian_jj, CConfig *config) {
62  unsigned short iDim, jDim;
63  su2double sq_vel_i, ViscDens_i, XiDens_i;
64  su2double sq_vel_j, ViscDens_j, XiDens_j;
65  su2double dist_ij_2, dPhiE_dn;
66 
67  su2double Prandtl_Lam = config->GetPrandtl_Lam();
68  su2double Prandtl_Turb = config->GetPrandtl_Turb();
69 
70  /*--- States in point i ---*/
71 
72  sq_vel_i = 0.0;
73  for (iDim = 0; iDim < nDim; iDim++) {
74  Velocity_i[iDim] = V_i[iDim+1];
75  sq_vel_i += 0.5*Velocity_i[iDim]*Velocity_i[iDim];
76  }
77  Pressure_i = V_i[nDim+1];
78  Density_i = V_i[nDim+2];
79  Enthalpy_i = V_i[nDim+3];
80  SoundSpeed_i = sqrt(fabs(Pressure_i*Gamma/Density_i));
81 
82  /*--- Laminar and Eddy viscosity ---*/
83 
84  Laminar_Viscosity_i = V_i[nDim+5];
85  Eddy_Viscosity_i = V_i[nDim+6];
86 
87  ViscDens_i = (Laminar_Viscosity_i + Eddy_Viscosity_i) / Density_i;
88  XiDens_i = Gamma*(Laminar_Viscosity_i/Prandtl_Lam + Eddy_Viscosity_i/Prandtl_Turb) / Density_i;
89 
90  /*--- States in point j ---*/
91 
92  sq_vel_j = 0.0;
93  for (iDim = 0; iDim < nDim; iDim++) {
94  Velocity_j[iDim] = V_j[iDim+1];
95  sq_vel_j += 0.5*Velocity_j[iDim]*Velocity_j[iDim];
96  }
97  Pressure_j = V_j[nDim+1];
98  Density_j = V_j[nDim+2];
99  Enthalpy_j = V_j[nDim+3];
100  SoundSpeed_j = sqrt(fabs(Pressure_j*Gamma/Density_j));
101 
102  /*--- Laminar and Eddy viscosity ---*/
103 
104  Laminar_Viscosity_j = V_j[nDim+5];
105  Eddy_Viscosity_j = V_j[nDim+6];
106 
107  ViscDens_j = (Laminar_Viscosity_j + Eddy_Viscosity_j) / Density_j;
108  XiDens_j = Gamma*(Laminar_Viscosity_j/Prandtl_Lam + Eddy_Viscosity_j/Prandtl_Turb) / Density_j;
109 
110  /*--- Compute vector going from iPoint to jPoint ---*/
111 
112  dist_ij_2 = 0.0;
113  for (iDim = 0; iDim < nDim; iDim++) {
114  Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
115  dist_ij_2 += Edge_Vector[iDim]*Edge_Vector[iDim];
116  }
117 
118  /*--- Average of the derivatives of the adjoint variables ---*/
119 
120  for (iDim = 0; iDim < nDim; iDim++) {
121  Mean_GradPsiE[iDim] = 0.5*(PsiVar_Grad_i[nVar-1][iDim]+PsiVar_Grad_j[nVar-1][iDim]);
122  for (jDim = 0; jDim < nDim; jDim++)
123  Mean_GradPhi[iDim][jDim] = 0.5*(PsiVar_Grad_i[iDim+1][jDim]+PsiVar_Grad_j[iDim+1][jDim]);
124  }
125 
126  dPhiE_dn = 0;
127  for (iDim = 0; iDim < nDim; iDim++)
128  dPhiE_dn += Mean_GradPsiE[iDim]*Normal[iDim];
129 
130  /*--- Compute the viscous residual and jacobian ---*/
131 
132  GetAdjViscousFlux_Jac(Pressure_i, Pressure_j, Density_i, Density_j,
133  ViscDens_i, ViscDens_j, Velocity_i, Velocity_j, sq_vel_i, sq_vel_j,
134  XiDens_i, XiDens_j, Mean_GradPhi, Mean_GradPsiE,
135  dPhiE_dn, Normal, Edge_Vector, dist_ij_2, val_residual_i, val_residual_j,
136  val_Jacobian_ii, val_Jacobian_ij, val_Jacobian_ji, val_Jacobian_jj,
137  implicit);

243 
244  dist_ij_2 = 0.0;

312 void CAvgGrad_AdjTurb::ComputeResidual(su2double *val_residual, su2double **val_Jacobian_i,
313  su2double **val_Jacobian_j, CConfig *config) {
314 
315  bool implicit = (config->GetKind_TimeIntScheme_AdjTurb() == EULER_IMPLICIT);
316 
317  su2double sigma = 2./3.;
318  su2double nu_i, nu_j, nu_e;
319  su2double dist_ij_2 = 0;
320  su2double proj_vector_ij = 0;
321  unsigned short iVar, iDim;
322 
323  /*--- Compute mean effective viscosity ---*/
324  nu_i = Laminar_Viscosity_i/U_i[0];
325  nu_j = Laminar_Viscosity_j/U_j[0];
326  nu_e = 0.5*(nu_i+nu_j+ScalarVar_i[0]+ScalarVar_j[0])/sigma;
327 
328  /*--- Compute vector going from iPoint to jPoint ---*/
329  for (iDim = 0; iDim < nDim; iDim++) {
330  Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
331  dist_ij_2 += Edge_Vector[iDim]*Edge_Vector[iDim];
332  proj_vector_ij += Edge_Vector[iDim]*Normal[iDim];
333  }
334  proj_vector_ij = proj_vector_ij/dist_ij_2;
335 
336  /*--- Mean gradient approximation ---*/
337  for (iVar = 0; iVar < nVar; iVar++) {
338  Proj_Mean_GradTurbPsi_Edge[iVar] = 0.0;
339  for (iDim = 0; iDim < nDim; iDim++) {
340  Mean_GradTurbPsi[iVar][iDim] = 0.5*(TurbPsi_Grad_i[iVar][iDim] + TurbPsi_Grad_j[iVar][iDim]);
341  }
342 
343  /*--- Projection of the corrected gradient ---*/
344  Proj_Mean_GradTurbPsi_Corrected[iVar] = 0.0;
345  for (iDim = 0; iDim < nDim; iDim++)
346  Proj_Mean_GradTurbPsi_Corrected[iVar] += Mean_GradTurbPsi[iVar][iDim]*Normal[iDim];
347  }
348 
349  val_residual[0] = -nu_e*Proj_Mean_GradTurbPsi_Corrected[0];
350 
351  if (implicit) {
352  val_Jacobian_i[0][0] = nu_e*proj_vector_ij;
353  val_Jacobian_j[0][0] = -nu_e*proj_vector_ij;
354  }
355 
356 }

485  su2double **val_Jacobian_ji, su2double **val_Jacobian_jj, CConfig *config) {
486 
487  bool implicit = (config->GetKind_TimeIntScheme_AdjTurb() == EULER_IMPLICIT);
488 
489  su2double sigma = 2./3.;
490  su2double nu_i, nu_j, nu_e_i, nu_e_j;
491  su2double dist_ij_2 = 0;
492  su2double proj_vector_ij = 0;
493  unsigned short iVar, iDim;
494 
495  /*--- Compute mean effective viscosity ---*/
496  nu_i = Laminar_Viscosity_i/U_i[0];
497  nu_j = Laminar_Viscosity_j/U_j[0];
498  nu_e_i = (nu_i+ScalarVar_i[0])/sigma;
499  nu_e_j = (nu_j+ScalarVar_j[0])/sigma;
500 
501  /*--- Compute vector going from iPoint to jPoint ---*/
502  for (iDim = 0; iDim < nDim; iDim++) {
503  Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
504  dist_ij_2 += Edge_Vector[iDim]*Edge_Vector[iDim];
505  proj_vector_ij += Edge_Vector[iDim]*Normal[iDim];
506  }
507  proj_vector_ij = proj_vector_ij/dist_ij_2; // to normalize vectors
508 
509  /*--- Mean gradient approximation ---*/
510  for (iVar = 0; iVar < nVar; iVar++) {
511  Proj_Mean_GradTurbPsi_Edge[iVar] = 0.0;
512  for (iDim = 0; iDim < nDim; iDim++) {
513  Mean_GradTurbPsi[iVar][iDim] = 0.5*(TurbPsi_Grad_i[iVar][iDim] + TurbPsi_Grad_j[iVar][iDim]);
514  Proj_Mean_GradTurbPsi_Edge[iVar] += Mean_GradTurbPsi[iVar][iDim]*Edge_Vector[iDim];
515  }
516 
517  for (iDim = 0; iDim < nDim; iDim++)
518  Mean_GradTurbPsi[iVar][iDim] -= (Proj_Mean_GradTurbPsi_Edge[iVar] -
519  (TurbPsi_j[iVar]-TurbPsi_i[iVar]))*Edge_Vector[iDim]/dist_ij_2;
520 
521  /*--- Projection of the corrected gradient ---*/
522  Proj_Mean_GradTurbPsi_Corrected[iVar] = 0.0;
523  for (iDim = 0; iDim < nDim; iDim++)
524  Proj_Mean_GradTurbPsi_Corrected[iVar] += Mean_GradTurbPsi[iVar][iDim]*Normal[iDim];
525  }
526 
527  val_residual_i[0] = -nu_e_i*Proj_Mean_GradTurbPsi_Corrected[0];
528  val_residual_j[0] = nu_e_j*Proj_Mean_GradTurbPsi_Corrected[0];
529 
530  if (implicit) {
531  val_Jacobian_ii[0][0] = nu_e_i*proj_vector_ij;
532  val_Jacobian_ij[0][0] = -nu_e_i*proj_vector_ij;
533  val_Jacobian_ji[0][0] = -nu_e_j*proj_vector_ij;
534  val_Jacobian_jj[0][0] = nu_e_j*proj_vector_ij;
535  }
536 
537 }
