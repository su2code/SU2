31 std::unique_ptr<CADTElemClass> CMeshFEM_DG::ComputeViscousWallADT(const CConfig *config) const {
32 
33  /*--------------------------------------------------------------------------*/
34  /*--- Step 1: Create the coordinates and connectivity of the linear ---*/
35  /*--- subelements of the local boundaries that must be taken ---*/
36  /*--- into account in the wall distance computation. ---*/
37  /*--------------------------------------------------------------------------*/
38 
39  /* Initialize an array for the mesh points, which eventually contains the
40  mapping from the local nodes to the number used in the connectivity of the
41  local boundary faces. However, in a first pass it is an indicator whether
42  or not a mesh point is on a local wall boundary. */
43  vector<unsigned long> meshToSurface(meshPoints.size(), 0);
44 
45  /* Define the vectors for the connectivity of the local linear subelements,
46  the element ID's, the element type and marker ID's. */
47  vector<unsigned long> surfaceConn;
48  vector<unsigned long> elemIDs;
49  vector<unsigned short> VTK_TypeElem;
50  vector<unsigned short> markerIDs;
51 
52  /* Loop over the boundary markers. */
53  for(unsigned short iMarker=0; iMarker<boundaries.size(); ++iMarker) {
54  if( !boundaries[iMarker].periodicBoundary ) {
55 
56  /* Check for a viscous wall. */
57  if( (config->GetMarker_All_KindBC(iMarker) == HEAT_FLUX) ||
58  (config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL) ) {
59 
60  /* Loop over the surface elements of this marker. */
61  const vector<CSurfaceElementFEM> &surfElem = boundaries[iMarker].surfElem;
62  for(unsigned long i=0; i<surfElem.size(); ++i) {
63 
64  /* Set the flag of the mesh points on this surface to true. */
65  for(unsigned short j=0; j<surfElem[i].nDOFsGrid; ++j)
66  meshToSurface[surfElem[i].DOFsGridFace[j]] = 1;
67 
68  /* Determine the necessary data from the corresponding standard face,
69  such as the number of linear subfaces, the number of DOFs per
70  linear subface and the corresponding local connectivity. */
71  const unsigned short ind = surfElem[i].indStandardElement;
72  const unsigned short VTK_Type = standardBoundaryFacesGrid[ind].GetVTK_Type();
73  const unsigned short nSubFaces = standardBoundaryFacesGrid[ind].GetNSubFaces();
74  const unsigned short nDOFsPerFace = standardBoundaryFacesGrid[ind].GetNDOFsPerSubFace();
75  const unsigned short *connSubFaces = standardBoundaryFacesGrid[ind].GetSubFaceConn();
76 
77  /* Loop over the number of subfaces and store the required data. */
78  unsigned short ii = 0;
79  for(unsigned short j=0; j<nSubFaces; ++j) {
80  markerIDs.push_back(iMarker);
81  VTK_TypeElem.push_back(VTK_Type);
82  elemIDs.push_back(i);
83 
84  for(unsigned short k=0; k<nDOFsPerFace; ++k, ++ii)
85  surfaceConn.push_back(surfElem[i].DOFsGridFace[connSubFaces[ii]]);
86  }
87  }
88  }
89  }
90  }
91 
92  /*--- Create the coordinates of the local points on the viscous surfaces and
93  create the final version of the mapping from all volume points to the
94  points on the viscous surfaces. ---*/
95  vector<su2double> surfaceCoor;
96  unsigned long nVertex_SolidWall = 0;
97 
98  for(unsigned long i=0; i<meshPoints.size(); ++i) {
99  if( meshToSurface[i] ) {
100  meshToSurface[i] = nVertex_SolidWall++;
101 
102  for(unsigned short k=0; k<nDim; ++k)
103  surfaceCoor.push_back(meshPoints[i].coor[k]);
104  }
105  }
106 
107  /*--- Change the surface connectivity, such that it corresponds to
108  the entries in surfaceCoor rather than in meshPoints. ---*/
109  for(unsigned long i=0; i<surfaceConn.size(); ++i)
110  surfaceConn[i] = meshToSurface[surfaceConn[i]];
111 
112  /*--------------------------------------------------------------------------*/
113  /*--- Step 2: Build the ADT, which is an ADT of bounding boxes of the ---*/
114  /*--- surface elements. A nearest point search does not give ---*/
115  /*--- accurate results, especially not for the integration ---*/
116  /*--- points of the elements close to a wall boundary. ---*/
117  /*--------------------------------------------------------------------------*/
118 
119  /* Build the ADT. */
120  std::unique_ptr<CADTElemClass> WallADT(new CADTElemClass(nDim, surfaceCoor, surfaceConn, VTK_TypeElem,
121  markerIDs, elemIDs, true));
122 
123  return WallADT;
124 
125 }

130 void CMeshFEM_DG::SetWallDistance(su2double val){
131 
132  for(unsigned long l=0; l<nVolElemOwned; ++l) {
133 
134  /* Get the required data from the corresponding standard element. */
135  const unsigned short ind = volElem[l].indStandardElement;
136  const unsigned short nInt = standardElementsGrid[ind].GetNIntegration();
137 
138  /* Store the number of solDOFS a bit easier. */
139  const unsigned short nDOFsSol = volElem[l].nDOFsSol;
140 
141  /* Allocate the memory for the wall distance of this element. */
142  volElem[l].wallDistance.resize(nInt, val);
143  volElem[l].wallDistanceSolDOFs.resize(nDOFsSol, val);
144 
145  }
146 
147  for(unsigned long l=0; l<matchingFaces.size(); ++l) {
148  /* Get the required data from the corresponding standard element. */
149  const unsigned short ind = matchingFaces[l].indStandardElement;
150  const unsigned short nInt = standardMatchingFacesGrid[ind].GetNIntegration();
151 
152  /* Allocate the memory for the wall distance for this matching face. */
153  matchingFaces[l].wallDistance.resize(nInt, val);
154  }
155 
156  for(unsigned short iMarker=0; iMarker<boundaries.size(); ++iMarker) {
157  if( !boundaries[iMarker].periodicBoundary ) {
158 
159  /* Loop over the boundary faces and determine the wall distances
160  in the integration points. */
161  vector<CSurfaceElementFEM> &surfElem = boundaries[iMarker].surfElem;
162  for(unsigned long l=0; l<surfElem.size(); ++l) {
163 
164  /* Get the required data from the corresponding standard element. */
165  const unsigned short ind = surfElem[l].indStandardElement;
166  const unsigned short nInt = standardBoundaryFacesGrid[ind].GetNIntegration();
167 
168  /* Allocate the memory for the wall distance for this boundary face. */
169  surfElem[l].wallDistance.resize(nInt, val);
170  }
171  }

175 void CMeshFEM_DG::SetWallDistance(CADTElemClass *WallADT, const CConfig* config, unsigned short iZone){
176 
177  /*--------------------------------------------------------------------------*/
178  /*--- Step 3: Determine the wall distance of the integration points of ---*/
179  /*--- locally owned volume elements. ---*/
180  /*--------------------------------------------------------------------------*/
181 
182  /*--- Loop over the owned elements to compute the wall distance
183  in the integration points. ---*/
184  for(unsigned long l=0; l<nVolElemOwned; ++l) {
185 
186  /* Get the required data from the corresponding standard element. */
187  const unsigned short ind = volElem[l].indStandardElement;
188  const unsigned short nInt = standardElementsGrid[ind].GetNIntegration();
189 
190  /* Allocate the memory for the wall distance of this element. */
191  volElem[l].wallDistance.resize(nInt);
192 
193  if( !WallADT->IsEmpty() ) {
194 
195  /*--- The tree is not empty. Loop over the integration points
196  and determine the wall distance. ---*/
197  for(unsigned short i=0; i<nInt; ++i) {
198 
199  const su2double *coor = volElem[l].coorIntegrationPoints.data() + i*nDim;
200  unsigned short markerID;
201  unsigned long elemID;
202  int rankID;
203  su2double dist;
204  WallADT->DetermineNearestElement(coor, dist, markerID, elemID, rankID);
205 
206  volElem[l].wallDistance[i] = dist;
207  }
208  }
209  }
210 
211  /*--------------------------------------------------------------------------*/
212  /*--- Step 4: Determine the wall distance of the solution DOFs of ---*/
213  /*--- locally owned volume elements. ---*/
214  /*--------------------------------------------------------------------------*/
215 
216  /*--- Loop over the owned elements to compute the wall distance
217  in the integration points. ---*/
218  for(unsigned long l=0; l<nVolElemOwned; ++l) {
219 
220  /* Store the number of solDOFS a bit easier. */
221  const unsigned short nDOFsSol = volElem[l].nDOFsSol;
222 
223  /* Allocate the memory for the wall distance of the solution DOFs. */
224  volElem[l].wallDistanceSolDOFs.resize(nDOFsSol);
225 
226  if( !WallADT->IsEmpty() ) {
227 
228  /*--- The tree is not empty. Loop over the solution DOFs
229  and determine the wall distance. ---*/
230  for(unsigned short i=0; i<nDOFsSol; ++i) {
231 
232  const su2double *coor = volElem[l].coorSolDOFs.data() + i*nDim;
233  unsigned short markerID;
234  unsigned long elemID;
235  int rankID;
236  su2double dist;
237  WallADT->DetermineNearestElement(coor, dist, markerID, elemID, rankID);
238 
239  volElem[l].wallDistanceSolDOFs[i] = dist;
240  }
241  }
242  }
243 
244  /*--------------------------------------------------------------------------*/
245  /*--- Step 5: Determine the wall distance of the integration points of ---*/
246  /*--- the internal matching faces. ---*/
247  /*--------------------------------------------------------------------------*/
248 
249  for(unsigned long l=0; l<matchingFaces.size(); ++l) {
250 
251  /* Get the required data from the corresponding standard element. */
252  const unsigned short ind = matchingFaces[l].indStandardElement;
253  const unsigned short nInt = standardMatchingFacesGrid[ind].GetNIntegration();
254 
255  /* Allocate the memory for the wall distance for this matching face. */
256  matchingFaces[l].wallDistance.resize(nInt);
257 
258  if( !WallADT->IsEmpty() ) {
259 
260  /*--- The tree is not empty. Loop over the integration points
261  and determine the wall distance. */
262  for(unsigned short i=0; i<nInt; ++i) {
263 
264  const su2double *coor = matchingFaces[l].coorIntegrationPoints.data()
265  + i*nDim;
266  unsigned short markerID;
267  unsigned long elemID;
268  int rankID;
269  su2double dist;
270  WallADT->DetermineNearestElement(coor, dist, markerID, elemID, rankID);
271 
272  matchingFaces[l].wallDistance[i] = dist;
273  }
274  }
275  }
276 
277  /*--------------------------------------------------------------------------*/
278  /*--- Step 6: Determine the wall distance of the integration points of ---*/
279  /*--- locally owned boundary faces. ---*/
280  /*--------------------------------------------------------------------------*/
281 
282  /*--- Loop over the boundary markers. Make sure to exclude the periodic
283  boundaries, because these are not physical. ---*/
284  for(unsigned short iMarker=0; iMarker<boundaries.size(); ++iMarker) {
285  if( !boundaries[iMarker].periodicBoundary ) {
286 
287  /* Determine whether or not this is a viscous wall boundary. */
288  const bool viscousWall = config->GetMarker_All_KindBC(iMarker) == HEAT_FLUX ||
289  config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL;
290 
291  /* Loop over the boundary faces and determine the wall distances
292  in the integration points. */
293  vector<CSurfaceElementFEM> &surfElem = boundaries[iMarker].surfElem;
294  for(unsigned long l=0; l<surfElem.size(); ++l) {
295 
296  /* Get the required data from the corresponding standard element. */
297  const unsigned short ind = surfElem[l].indStandardElement;
298  const unsigned short nInt = standardBoundaryFacesGrid[ind].GetNIntegration();
299 
300  /* Allocate the memory for the wall distance for this boundary face. */
301  surfElem[l].wallDistance.resize(nInt);
302 
303  /* Check for an empty tree or a viscous wall.
304  In those case the wall distance is set to zero. */
305  if(viscousWall) {
306 
307  /* Wall distance must be set to zero. */
308  for(unsigned short i=0; i<nInt; ++i)
309  surfElem[l].wallDistance[i] = 0.0;
310  }
311  else if ( !WallADT->IsEmpty() ) {
312  /*--- Not a viscous wall boundary, while viscous walls are present.
313  The distance must be computed. Loop over the integration points
314  and do so. ---*/
315  for(unsigned short i=0; i<nInt; ++i) {
316 
317  const su2double *coor = surfElem[l].coorIntegrationPoints.data() + i*nDim;
318  unsigned short markerID;
319  unsigned long elemID;
320  int rankID;
321  su2double dist;
322  WallADT->DetermineNearestElement(coor, dist, markerID, elemID, rankID);
323 
324  surfElem[l].wallDistance[i] = dist;
325  }
326  }
327  }
328  }
329  }
