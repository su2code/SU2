45  template <class IndexType>
46  struct CIndices {
47  const IndexType nDim;
48  CIndices(IndexType ndim, IndexType) : nDim(ndim) {}
49  inline IndexType Pressure() const { return 0; }
50  inline IndexType Velocity() const { return 1; }
51  inline IndexType Temperature() const { return nDim+1; }
52  inline IndexType Density() const { return nDim+2; }
53  inline IndexType Beta() const { return nDim+3; }
54  inline IndexType SoundSpeed() const { return Beta(); }
55  inline IndexType LaminarViscosity() const { return nDim+4; }
56  inline IndexType EddyViscosity() const { return nDim+5; }
57  inline IndexType ThermalConductivity() const { return nDim+6; }
58  inline IndexType CpTotal() const { return nDim+7; }
59  inline IndexType CvTotal() const { return nDim+8; }

61  /*--- For compatible interface with NEMO. ---*/
62  inline IndexType Temperature_ve() const { return std::numeric_limits<IndexType>::max(); }

100  /*!
101  * \brief Set the value of the density for the incompressible flows.
102  * \param[in] iPoint - Point index.
103  */
104  inline void SetVelocity(unsigned long iPoint) final {
105  Velocity2(iPoint) = 0.0;
106  for (unsigned long iDim = 0; iDim < nDim; iDim++) {
107  Primitive(iPoint, iDim+indices.Velocity()) = Solution(iPoint,iDim+1);
108  Velocity2(iPoint) += pow(Primitive(iPoint, iDim+indices.Velocity()), 2);
109  }
110  }

170  /*!
171  * \brief Get the projected velocity in a unitary vector direction (compressible solver).
172  * \param[in] val_vector - Direction of projection.

174  */
175  inline su2double GetProjVel(unsigned long iPoint, const su2double *val_vector) const final {
176  su2double ProjVel = 0.0;
177  for (unsigned long iDim = 0; iDim < nDim; iDim++)
178  ProjVel += Primitive(iPoint, iDim+indices.Velocity())*val_vector[iDim];
