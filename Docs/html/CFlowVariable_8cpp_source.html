30 CFlowVariable::CFlowVariable(unsigned long npoint, unsigned long ndim, unsigned long nvar, unsigned long nprimvar,
31  unsigned long nprimvargrad, const CConfig* config)
32  : CVariable(npoint, ndim, nvar, config),
33  Gradient_Reconstruction(config->GetReconstructionGradientRequired() ? Gradient_Aux : Gradient_Primitive) {
34  nPrimVar = nprimvar;
35  nPrimVarGrad = nprimvargrad;
36 
37  /*--- Allocate residual structures for multigrid. ---*/
38 
39  Res_TruncError.resize(nPoint, nVar) = su2double(0.0);
40 
41  for (unsigned long iMesh = 0; iMesh <= config->GetnMGLevels(); iMesh++) {
42  if (config->GetMG_CorrecSmooth(iMesh) > 0) {
43  Residual_Sum.resize(nPoint, nVar);
44  Residual_Old.resize(nPoint, nVar);
45  break;
46  }
47  }
48 
49  /*--- Primitive variables and gradients (see derived classes for what is stored each column) ---*/
50 
51  Primitive.resize(nPoint, nPrimVar) = su2double(0.0);
52 
53  if (config->GetMUSCL_Flow() || config->GetViscous()) {
54  Gradient_Primitive.resize(nPoint, nPrimVarGrad, nDim, 0.0);
55  }
56 
57  if (config->GetReconstructionGradientRequired() && config->GetKind_ConvNumScheme_Flow() != SPACE_CENTERED) {
58  Gradient_Aux.resize(nPoint, nPrimVarGrad, nDim, 0.0);
59  }
60 
61  if (config->GetLeastSquaresRequired()) {
62  Rmatrix.resize(nPoint, nDim, nDim, 0.0);
63  }
64 
65  /*--- Allocate undivided laplacian (centered) ---*/
66 
67  if (config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED) {
68  Undivided_Laplacian.resize(nPoint, nVar);
69  }
70 
71  /*--- Allocate the slope limiter (MUSCL upwind) ---*/
72 
73  if (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE && config->GetKind_SlopeLimit_Flow() != LIMITER::VAN_ALBADA_EDGE) {
74  Limiter_Primitive.resize(nPoint, nPrimVarGrad) = su2double(0.0);
75  Solution_Max.resize(nPoint, nPrimVarGrad) = su2double(0.0);
76  Solution_Min.resize(nPoint, nPrimVarGrad) = su2double(0.0);
77  }
78 
79  Velocity2.resize(nPoint) = su2double(0.0);
80  Max_Lambda_Inv.resize(nPoint) = su2double(0.0);
81  Delta_Time.resize(nPoint) = su2double(0.0);
82  Lambda.resize(nPoint) = su2double(0.0);
83  Sensor.resize(nPoint) = su2double(0.0);
84 
85  /* Under-relaxation parameter. */
86  UnderRelaxation.resize(nPoint) = su2double(1.0);
87  LocalCFL.resize(nPoint) = su2double(0.0);
88 
89  /* Non-physical point (first-order) initialization. */
90  Non_Physical.resize(nPoint) = false;
91  Non_Physical_Counter.resize(nPoint) = 0;
92 
93  if (config->GetMultizone_Problem()) {
94  Set_BGSSolution_k();
95  }
96 
97  if (config->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE) {
98  HB_Source.resize(nPoint, nVar) = su2double(0.0);
99  }
100 }
