93  /*!
94  * \brief Invert a square matrix.
95  * \param[in] M - Size.
96  * \param[in,out] mat - Matrix, and inverse on exit.
97  */
98  template<class Mat>
99  static void inverse(const int M, Mat& mat) {
100  using Scalar = typename Mat::Scalar;
101 
102  /*--- Copy the data from A into the augmented matrix and initialize mat with the identity. ---*/
103  Mat aug = mat;
104  mat = Scalar(0);
105  for(int j=0; j<M; ++j) mat(j,j) = 1;
106 
107  /*--- Outer loop of the Gauss-Jordan elimination. ---*/
108  for(int j=0; j<M; ++j) {
109 
110  /*--- Find the pivot in the current column. ---*/
111  int jj = j;
112  Scalar valMax = fabs(aug(j,j));
113  for(int i=j+1; i<M; ++i) {
114  Scalar val = fabs(aug(i,j));
115  if(val > valMax){
116  jj = i;
117  valMax = val;
118  }
119  }
120 
121  /*--- Swap the rows j and jj, if needed. ---*/
122  if(jj > j) {
123  for(int k=j; k<M; ++k) std::swap(aug(j,k), aug(jj,k));
124  for(int k=0; k<M; ++k) std::swap(mat(j,k), mat(jj,k));
125  }
126 
127  /*--- Performing row operations to form required identity
128  matrix out of the input matrix. ---*/
129  for(int i=0; i<M; ++i) {
130  if(i == j) continue;
131  valMax = aug(i,j)/aug(j,j);
132  for(int k=j; k<M; ++k) aug(i,k) -= valMax*aug(j,k);
133  for(int k=0; k<M; ++k) mat(i,k) -= valMax*mat(j,k);
134  }
135 
136  valMax = 1.0/aug(j,j);
137  for(int k=j; k<M; ++k) aug(j,k) *= valMax;
138  for(int k=0; k<M; ++k) mat(j,k) *= valMax;
139  }
140  }

143  * \brief tred2
144  * Author:
145  *
146  * Original FORTRAN77 version by Smith, Boyle, Dongarra, Garbow, Ikebe,
147  * Klema, Moler.
148  * C++ version by Aashwin Mishra and Jayant Mukhopadhaya.
149  *
150  * Reference:
151  *
152  * Martin, Reinsch, Wilkinson,
153  * TRED2,
154  * Numerische Mathematik,
155  * Volume 11, pages 181-195, 1968.
156  *
157  * James Wilkinson, Christian Reinsch,
158  * Handbook for Automatic Computation,
159  * Volume II, Linear Algebra, Part 2,
160  * Springer, 1971,
161  * ISBN: 0387054146,
162  * LC: QA251.W67.
163  *
164  * Brian Smith, James Boyle, Jack Dongarra, Burton Garbow,
165  * Yasuhiko Ikebe, Virginia Klema, Cleve Moler,
166  * Matrix Eigensystem Routines, EISPACK Guide,
167  * Lecture Notes in Computer Science, Volume 6,
168  * Springer Verlag, 1976,
169  * ISBN13: 978-3540075462,
170  * LC: QA193.M37
171  *
172  * \param[in,out] V: matrix that needs to be decomposed
173  * \param[in,out] d: holds eigenvalues
174  * \param[in,out] e: work vector
175  * \param[in] n: order of matrix V
176  */
177  template<class Mat, class Vec, class W>
178  static void tred2(Mat& V, Vec& d, W& e, int n) {
179  using Scalar = typename std::decay<decltype(e[0])>::type;
180 
181  int i,j,k;
182 
183  for (j = 0; j < n; j++) {
184  d[j] = V[n-1][j];
185  }
186 
187  /* Householder reduction to tridiagonal form. */
188 
189  for (i = n-1; i > 0; i--) {
190 
191  /* Scale to avoid under/overflow. */
192 
193  Scalar scale = 0.0;
194  Scalar h = 0.0;
195  for (k = 0; k < i; k++) {
196  scale = scale + fabs(d[k]);
197  }
198  if (scale == 0.0) {
199  e[i] = d[i-1];
200  for (j = 0; j < i; j++) {
201  d[j] = V[i-1][j];
202  V[i][j] = 0.0;
203  V[j][i] = 0.0;
204  }
205  }
206  else {
207 
208  /* Generate Householder vector. */
209 
210  for (k = 0; k < i; k++) {
211  d[k] /= scale;
212  h += d[k] * d[k];
213  }
214  Scalar f = d[i-1];
215  Scalar g = sqrt(h);
216  if (f > 0) {
217  g = -g;
218  }
219  e[i] = scale * g;
220  h = h - f * g;
221  d[i-1] = f - g;
222  for (j = 0; j < i; j++) {
223  e[j] = 0.0;
224  }
225 
226  /* Apply similarity transformation to remaining columns. */
227 
228  for (j = 0; j < i; j++) {
229  f = d[j];
230  V[j][i] = f;
231  g = e[j] + V[j][j] * f;
232  for (k = j+1; k <= i-1; k++) {
233  g += V[k][j] * d[k];
234  e[k] += V[k][j] * f;
235  }
236  e[j] = g;
237  }
238  f = 0.0;
239  for (j = 0; j < i; j++) {
240  e[j] /= h;
241  f += e[j] * d[j];
242  }
243  Scalar hh = f / (h + h);
244  for (j = 0; j < i; j++) {
245  e[j] -= hh * d[j];
246  }
247  for (j = 0; j < i; j++) {
248  f = d[j];
249  g = e[j];
250  for (k = j; k <= i-1; k++) {
251  V[k][j] -= (f * e[k] + g * d[k]);
252  }
253  d[j] = V[i-1][j];
254  V[i][j] = 0.0;
255  }
256  }
257  d[i] = h;
258  }
259 
260  /* Accumulate transformations. */
261 
262  for (i = 0; i < n-1; i++) {
263  V[n-1][i] = V[i][i];
264  V[i][i] = 1.0;
265  Scalar h = d[i+1];
266  if (h != 0.0) {
267  for (k = 0; k <= i; k++) {
268  d[k] = V[k][i+1] / h;
269  }
270  for (j = 0; j <= i; j++) {
271  Scalar g = 0.0;
272  for (k = 0; k <= i; k++) {
273  g += V[k][i+1] * V[k][j];
274  }
275  for (k = 0; k <= i; k++) {
276  V[k][j] -= g * d[k];
277  }
278  }
279  }
280  for (k = 0; k <= i; k++) {
281  V[k][i+1] = 0.0;
282  }
283  }
284  for (j = 0; j < n; j++) {
285  d[j] = V[n-1][j];
286  V[n-1][j] = 0.0;
287  }
288  V[n-1][n-1] = 1.0;
289  e[0] = 0.0;
290  }

292  /*!
293  * \brief tql2
294  * Author:
295  *
296  * Original FORTRAN77 version by Smith, Boyle, Dongarra, Garbow, Ikebe,
297  * Klema, Moler.
298  * C++ version by Aashwin Mishra and Jayant Mukhopadhaya.
299  *
300  * Reference:
301  *
302  * Bowdler, Martin, Reinsch, Wilkinson,
303  * TQL2,
304  * Numerische Mathematik,
305  * Volume 11, pages 293-306, 1968.
306  *
307  * James Wilkinson, Christian Reinsch,
308  * Handbook for Automatic Computation,
309  * Volume II, Linear Algebra, Part 2,
310  * Springer, 1971,
311  * ISBN: 0387054146,
312  * LC: QA251.W67.
313  *
314  * Brian Smith, James Boyle, Jack Dongarra, Burton Garbow,
315  * Yasuhiko Ikebe, Virginia Klema, Cleve Moler,
316  * Matrix Eigensystem Routines, EISPACK Guide,
317  * Lecture Notes in Computer Science, Volume 6,
318  * Springer Verlag, 1976,
319  * ISBN13: 978-3540075462,
320  * LC: QA193.M37
321  *
322  * \param[in,out] V: matrix that will hold the eigenvectors
323  * \param[in,out] d: array that will hold the ordered eigenvalues
324  * \param[in,out] e: work vector
325  * \param[in] n: order of matrix V
326  */
327  template<class Mat, class Vec, class W>
328  static void tql2(Mat& V, Vec& d, W& e, int n) {
329  using Scalar = typename std::decay<decltype(e[0])>::type;
330 
331  int i,j,k,l;
332  for (i = 1; i < n; i++) {
333  e[i-1] = e[i];
334  }
335  e[n-1] = 0.0;
336 
337  Scalar f = 0.0;
338  Scalar tst1 = 0.0;
339  Scalar eps = pow(2.0,-52.0);
340  for (l = 0; l < n; l++) {
341 
342  /* Find small subdiagonal element */
343 
344  tst1 = max(tst1,(fabs(d[l]) + fabs(e[l])));
345  int m = l;
346  while (m < n) {
347  if (fabs(e[m]) <= eps*tst1) {
348  break;
349  }
350  m++;
351  }
352 
353  /* If m == l, d[l] is an eigenvalue, */
354  /* otherwise, iterate. */
355 
356  if (m > l) {
357  int iter = 0;
358  do {
359  iter = iter + 1; /* (Could check iteration count here.) */
360 
361  /* Compute implicit shift */
362 
363  Scalar g = d[l];
364  Scalar p = (d[l+1] - g) / (2.0 * e[l]);
365  Scalar r = sqrt(p*p+1.0);
366  if (p < 0) {
367  r = -r;
368  }
369  d[l] = e[l] / (p + r);
370  d[l+1] = e[l] * (p + r);
371  Scalar dl1 = d[l+1];
372  Scalar h = g - d[l];
373  for (i = l+2; i < n; i++) {
374  d[i] -= h;
375  }
376  f = f + h;
377 
378  /* Implicit QL transformation. */
379 
380  p = d[m];
381  Scalar c = 1.0;
382  Scalar c2 = c;
383  Scalar c3 = c;
384  Scalar el1 = e[l+1];
385  Scalar s = 0.0;
386  Scalar s2 = 0.0;
387  for (i = m-1; i >= l; i--) {
388  c3 = c2;
389  c2 = c;
390  s2 = s;
391  g = c * e[i];
392  h = c * p;
393  r = sqrt(p*p+e[i]*e[i]);
394  e[i+1] = s * r;
395  s = e[i] / r;
396  c = p / r;
397  p = c * d[i] - s * g;
398  d[i+1] = h + s * (c * g + s * d[i]);
399 
400  /* Accumulate transformation. */
401 
402  for (k = 0; k < n; k++) {
403  h = V[k][i+1];
404  V[k][i+1] = s * V[k][i] + c * h;
405  V[k][i] = c * V[k][i] - s * h;
406  }
407  }
408  p = -s * s2 * c3 * el1 * e[l] / dl1;
409  e[l] = s * p;
410  d[l] = c * p;
411 
412  /* Check for convergence. */
413 
414  } while (fabs(e[l]) > eps*tst1);
415  }
416  d[l] = d[l] + f;
417  e[l] = 0.0;
418  }
419 
420  /* Sort eigenvalues and corresponding vectors. */
421 
422  for (i = 0; i < n-1; i++) {
423  k = i;
424  Scalar p = d[i];
425  for (j = i+1; j < n; j++) {
426  if (d[j] < p) {
427  k = j;
428  p = d[j];
429  }
430  }
431  if (k != i) {
432  d[k] = d[i];
433  d[i] = p;
434  for (j = 0; j < n; j++) {
435  p = V[j][i];
436  V[j][i] = V[j][k];
437  V[j][k] = p;
438  }
439  }
440  }
441  }

481  /*!
482  * \brief Algorithm to solve a linear system with a tridiagonal matrix.
483  * \param[in] lower - lower diagonal
484  * \param[in] main - main diagonal
485  * \param[in,out] upper - upper diagonal (modified on exit)
486  * \param[in,out] rhs - right hand side on entry, solution on exit
487  * \note Same size for all vectors. Use row index for lower and upper vector (e.g. lower[0] does not matter).
488  */
489  template<class Vec, class Scalar = su2double>
490  static void tdma(const Vec& lower, const Vec& main, Vec& upper, Vec& rhs) {
491  const int N = main.size();
492 
493  upper[0] /= main[0];
494  rhs[0] /= main[0];
495 
496  for (int i=1; i<N; i++) {
497  const Scalar denom = 1.0 / (main[i]-lower[i]*upper[i-1]);
498  upper[i] *= denom;
499  rhs[i] = (rhs[i]-lower[i]*rhs[i-1])*denom;
500  }
501 
502  for (int i=N-2; i>=0; i--)
503  rhs[i] -= upper[i]*rhs[i+1];
504  }
