67 su2double CRadialBasisFunction::Get_RadialBasisValue(RADIAL_BASIS type, const su2double radius, const su2double dist)
68 {
69  su2double rbf = dist/radius;
70 
71  switch (type) {
72 
73  case RADIAL_BASIS::WENDLAND_C2:
74  if(rbf < 1) rbf = pow(pow((1-rbf),2),2)*(4*rbf+1); // double use of pow(x,2) for optimization
75  else rbf = 0.0;
76  break;
77 
78  case RADIAL_BASIS::GAUSSIAN:
79  rbf = exp(-rbf*rbf);
80  break;
81 
82  case RADIAL_BASIS::THIN_PLATE_SPLINE:
83  if(rbf < numeric_limits<float>::min()) rbf = 0.0;
84  else rbf *= rbf*log(rbf);
85  break;
86 
87  case RADIAL_BASIS::MULTI_QUADRIC:
88  case RADIAL_BASIS::INV_MULTI_QUADRIC:
89  rbf = sqrt(1.0+rbf*rbf);
90  if(type == RADIAL_BASIS::INV_MULTI_QUADRIC) rbf = 1.0/rbf;
91  break;
92  }
93 
94  return rbf;
95 }

97 void CRadialBasisFunction::SetTransferCoeff(const CConfig* const* config) {
98 
99  /*--- RBF options. ---*/
100  const auto kindRBF = config[donorZone]->GetKindRadialBasisFunction();
101  const bool usePolynomial = config[donorZone]->GetRadialBasisFunctionPolynomialOption();
102  const su2double paramRBF = config[donorZone]->GetRadialBasisFunctionParameter();
103  const su2double pruneTol = config[donorZone]->GetRadialBasisFunctionPruneTol();
104 
105  const auto nMarkerInt = config[donorZone]->GetMarker_n_ZoneInterface()/2;
106  const int nDim = donor_geometry->GetnDim();
107 
108  const int nProcessor = size;
109  Buffer_Receive_nVertex_Donor = new unsigned long [nProcessor];
110 
111  targetVertices.resize(config[targetZone]->GetnMarker_All());
112 
113  /*--- Process interface patches in parallel, fetch all donor point coordinates,
114  * then distribute interpolation matrix computation over ranks and threads.
115  * To avoid repeating calls to Collect_VertexInfo we also save the global
116  * indices of the donor points and the mpi rank index that owns them. ---*/
117 
118  vector<su2activematrix> donorCoordinates(nMarkerInt);
119  vector<vector<long> > donorGlobalPoint(nMarkerInt);
120  vector<vector<int> > donorProcessor(nMarkerInt);
121  vector<int> assignedProcessor(nMarkerInt,-1);
122  vector<unsigned long> totalWork(nProcessor,0);
123 
124  for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; ++iMarkerInt) {
125 
126  /*--- On the donor side: find the tag of the boundary sharing the interface. ---*/
127  const auto markDonor = config[donorZone]->FindInterfaceMarker(iMarkerInt);
128 
129  /*--- On the target side: find the tag of the boundary sharing the interface. ---*/
130  const auto markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
131 
132  /*--- If the zone does not contain the interface continue to the next pair of markers. ---*/
133  if (!CheckInterfaceBoundary(markDonor,markTarget)) continue;
134 
135  unsigned long nVertexDonor = 0;
136  if (markDonor != -1) nVertexDonor = donor_geometry->GetnVertex(markDonor);
137 
138  /*--- Sets MaxLocalVertex_Donor, Buffer_Receive_nVertex_Donor. ---*/
139  Determine_ArraySize(markDonor, markTarget, nVertexDonor, nDim);
140 
141  /*--- Compute total number of donor vertices. ---*/
142  const auto nGlobalVertexDonor = accumulate(Buffer_Receive_nVertex_Donor,
143  Buffer_Receive_nVertex_Donor+nProcessor, 0ul);
144 
145  /*--- Gather coordinates and global point indices. ---*/
146  Buffer_Send_Coord.resize(MaxLocalVertex_Donor, nDim);
147  Buffer_Send_GlobalPoint.resize(MaxLocalVertex_Donor);
148  Buffer_Receive_Coord.resize(nProcessor * MaxLocalVertex_Donor, nDim);
149  Buffer_Receive_GlobalPoint.resize(nProcessor * MaxLocalVertex_Donor);
150 
151  Collect_VertexInfo(markDonor, markTarget, nVertexDonor, nDim);
152 
153  /*--- Compresses the gathered donor point information to simplify computations. ---*/
154  auto& donorCoord = donorCoordinates[iMarkerInt];
155  auto& donorPoint = donorGlobalPoint[iMarkerInt];
156  auto& donorProc = donorProcessor[iMarkerInt];
157  donorCoord.resize(nGlobalVertexDonor, nDim);
158  donorPoint.resize(nGlobalVertexDonor);
159  donorProc.resize(nGlobalVertexDonor);
160 
161  auto iCount = 0ul;
162  for (int iProcessor = 0; iProcessor < nProcessor; ++iProcessor) {
163  auto offset = iProcessor * MaxLocalVertex_Donor;
164  for (auto iVertex = 0ul; iVertex < Buffer_Receive_nVertex_Donor[iProcessor]; ++iVertex) {
165  for (int iDim = 0; iDim < nDim; ++iDim)
166  donorCoord(iCount,iDim) = Buffer_Receive_Coord(offset+iVertex, iDim);
167  donorPoint[iCount] = Buffer_Receive_GlobalPoint[offset+iVertex];
168  donorProc[iCount] = iProcessor;
169  ++iCount;
170  }
171  }
172  assert((iCount == nGlobalVertexDonor) && "Global donor point count mismatch.");
173 
174  /*--- Give an MPI-independent order to the points (required due to high condition
175  * number of the RBF matrix, avoids diff results with diff number of ranks. ---*/
176  vector<int> order(nGlobalVertexDonor);
177  iota(order.begin(), order.end(), 0);
178  sort(order.begin(), order.end(), [&donorPoint](int i, int j){return donorPoint[i] < donorPoint[j];});
179 
180  for (int i = 0; i < int(nGlobalVertexDonor); ++i) {
181  int j = order[i];
182  while (j < i) j = order[j];
183  if (i == j) continue;
184  swap(donorProc[i], donorProc[j]);
185  swap(donorPoint[i], donorPoint[j]);
186  for (int iDim = 0; iDim < nDim; ++iDim)
187  swap(donorCoord(i,iDim), donorCoord(j,iDim));
188  }
189 
190  /*--- Static work scheduling over ranks based on which one has less work currently. ---*/
191  int iProcessor = 0;
192  for (int i = 1; i < nProcessor; ++i)
193  if (totalWork[i] < totalWork[iProcessor]) iProcessor = i;
194 
195  totalWork[iProcessor] += pow(nGlobalVertexDonor,3); // based on matrix inversion.
196 
197  assignedProcessor[iMarkerInt] = iProcessor;
198 
199  }
200  delete[] Buffer_Receive_nVertex_Donor;
201 
202  /*--- Compute the interpolation matrices for each patch of coordinates
203  * assigned to the rank. Subdivide work further by threads. ---*/
204  vector<int> nPolynomialVec(nMarkerInt,-1);
205  vector<vector<int> > keepPolynomialRowVec(nMarkerInt, vector<int>(nDim,1));
206  vector<su2passivematrix> CinvTrucVec(nMarkerInt);
207 
208  SU2_OMP_PARALLEL_(for schedule(dynamic,1))
209  for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; ++iMarkerInt) {
210  if (rank == assignedProcessor[iMarkerInt]) {
211  ComputeGeneratorMatrix(kindRBF, usePolynomial, paramRBF,
212  donorCoordinates[iMarkerInt], nPolynomialVec[iMarkerInt],
213  keepPolynomialRowVec[iMarkerInt], CinvTrucVec[iMarkerInt]);
214  }
215  }
216  END_SU2_OMP_PARALLEL
217 
218  /*--- Final loop over interface markers to compute the interpolation coefficients. ---*/
219 
220  /*--- Initialize variables for interpolation statistics. ---*/
221  unsigned long totalTargetPoints = 0, totalDonorPoints = 0, denseSize = 0;
222  MinDonors = 1<<30; MaxDonors = 0; MaxCorrection = 0.0; AvgCorrection = 0.0;
223 
224  for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; iMarkerInt++) {
225 
226  /*--- Identify the rank that computed the interpolation matrix for this marker. ---*/
227  const int iProcessor = assignedProcessor[iMarkerInt];
228  /*--- If no processor was assigned to work, the zone does not contain the interface. ---*/
229  if (iProcessor < 0) continue;
230 
231  /*--- Setup target information. ---*/
232  const auto markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
233  unsigned long nVertexTarget = 0;
234  if (markTarget != -1) nVertexTarget = target_geometry->GetnVertex(markTarget);
235 
236  /*--- Set references to donor information. ---*/
237  auto& donorCoord = donorCoordinates[iMarkerInt];
238  auto& donorPoint = donorGlobalPoint[iMarkerInt];
239  auto& donorProc = donorProcessor[iMarkerInt];
240 
241  auto& C_inv_trunc = CinvTrucVec[iMarkerInt];
242  auto& nPolynomial = nPolynomialVec[iMarkerInt];
243  auto& keepPolynomialRow = keepPolynomialRowVec[iMarkerInt];
244 
245  const auto nGlobalVertexDonor = donorCoord.rows();
246 
247 #ifdef HAVE_MPI
248  /*--- For simplicity, broadcast small information about the interpolation matrix. ---*/
249  SU2_MPI::Bcast(&nPolynomial, 1, MPI_INT, iProcessor, SU2_MPI::GetComm());
250  SU2_MPI::Bcast(keepPolynomialRow.data(), nDim, MPI_INT, iProcessor, SU2_MPI::GetComm());
251 
252  /*--- Send C_inv_trunc only to the ranks that need it (those with target points),
253  * partial broadcast. MPI wrapper not used due to passive double. ---*/
254  vector<unsigned long> allNumVertex(nProcessor);
255  SU2_MPI::Allgather(&nVertexTarget, 1, MPI_UNSIGNED_LONG,
256  allNumVertex.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
257 
258  if (rank == iProcessor) {
259  for (int jProcessor = 0; jProcessor < nProcessor; ++jProcessor)
260  if ((jProcessor != iProcessor) && (allNumVertex[jProcessor] != 0))
261  MPI_Send(C_inv_trunc.data(), C_inv_trunc.size(),
262  MPI_DOUBLE, jProcessor, 0, SU2_MPI::GetComm());
263  }
264  else if (nVertexTarget != 0) {
265  C_inv_trunc.resize(1+nPolynomial+nGlobalVertexDonor, nGlobalVertexDonor);
266  MPI_Recv(C_inv_trunc.data(), C_inv_trunc.size(), MPI_DOUBLE,
267  iProcessor, 0, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
268  }
269 #endif
270 
271  /*--- Compute interpolation matrix (H). This is a large matrix-matrix product with
272  * the generator matrix (C_inv_trunc) on the right. We avoid instantiation
273  * of the entire function matrix (A) and of the result (H), but work
274  * on a slab (set of rows) of A/H to amortize accesses to C_inv_trunc. ---*/
275 
276  /*--- Fetch target vertex coordinates. ---*/
277 
278  if (nVertexTarget) targetVertices[markTarget].resize(nVertexTarget);
279  vector<const su2double*> targetCoord(nVertexTarget);
280 
281  for (auto iVertexTarget = 0ul; iVertexTarget < nVertexTarget; ++iVertexTarget) {
282  const auto pointTarget = target_geometry->vertex[markTarget][iVertexTarget]->GetNode();
283  targetCoord[iVertexTarget] = target_geometry->nodes->GetCoord(pointTarget);
284  }
285  totalTargetPoints += nVertexTarget;
286  denseSize += nVertexTarget*nGlobalVertexDonor;
287 
288  /*--- Distribute target slabs over the threads in the rank for processing. ---*/
289 
290  SU2_OMP_PARALLEL
291  if (nVertexTarget > 0) {
292 
293  constexpr unsigned long targetSlabSize = 32;
294 
295  su2passivematrix funcMat(targetSlabSize, 1+nPolynomial+nGlobalVertexDonor);
296  su2passivematrix interpMat(targetSlabSize, nGlobalVertexDonor);
297 
298  /*--- Thread-local variables for statistics. ---*/
299  unsigned long minDonors = 1<<30, maxDonors = 0, totalDonors = 0;
300  passivedouble sumCorr = 0.0, maxCorr = 0.0;
301 
302  SU2_OMP_FOR_DYN(1)
303  for (auto iVertexTarget = 0ul; iVertexTarget < nVertexTarget; iVertexTarget += targetSlabSize) {
304 
305  const auto iLastVertex = min(nVertexTarget, iVertexTarget+targetSlabSize);
306  const auto slabSize = iLastVertex - iVertexTarget;
307 
308  /*--- Prepare matrix of functions A (the targets to donors matrix). ---*/
309 
310  /*--- Polynominal part: ---*/
311  if (usePolynomial) {
312  /*--- Constant term. ---*/
313  for (auto k = 0ul; k < slabSize; ++k) funcMat(k,0) = 1.0;
314 
315  /*--- Linear terms. ---*/
316  for (int iDim = 0, idx = 1; iDim < nDim; ++iDim) {
317  /*--- Of which one may have been excluded. ---*/
318  if (!keepPolynomialRow[iDim]) continue;
319  for (auto k = 0ul; k < slabSize; ++k)
320  funcMat(k, idx) = SU2_TYPE::GetValue(targetCoord[iVertexTarget+k][iDim]);
321  idx += 1;
322  }
323  }
324  /*--- RBF terms: ---*/
325  for (auto iVertexDonor = 0ul; iVertexDonor < nGlobalVertexDonor; ++iVertexDonor) {
326  for (auto k = 0ul; k < slabSize; ++k) {
327  auto dist = GeometryToolbox::Distance(nDim, targetCoord[iVertexTarget+k], donorCoord[iVertexDonor]);
328  auto rbf = Get_RadialBasisValue(kindRBF, paramRBF, dist);
329  funcMat(k, 1+nPolynomial+iVertexDonor) = SU2_TYPE::GetValue(rbf);
330  }
331  }
332 
333  /*--- Compute slab of the interpolation matrix. ---*/
334 #ifdef HAVE_LAPACK
335  /*--- interpMat = funcMat * C_inv_trunc, but order of gemm arguments
336  * is swapped due to row-major storage of su2passivematrix. ---*/
337  const char op = 'N';
338  const int M = interpMat.cols(), N = slabSize, K = funcMat.cols();
339  // lda = C_inv_trunc.cols() = M; ldb = funcMat.cols() = K; ldc = interpMat.cols() = M;
340  const passivedouble alpha = 1.0, beta = 0.0;
341  DGEMM(&op, &op, &M, &N, &K, &alpha, C_inv_trunc[0], &M, funcMat[0], &K, &beta, interpMat[0], &M);
342 #else
343  /*--- Naive product, loop order considers short-wide
344  * nature of funcMat and interpMat. ---*/
345  interpMat = 0.0;
346  for (auto k = 0ul; k < funcMat.cols(); ++k)
347  for (auto i = 0ul; i < slabSize; ++i)
348  for (auto j = 0ul; j < interpMat.cols(); ++j)
349  interpMat(i,j) += funcMat(i,k) * C_inv_trunc(k,j);
350 #endif
351  /*--- Set interpolation coefficients. ---*/
352 
353  for (auto k = 0ul; k < slabSize; ++k) {
354  auto& targetVertex = targetVertices[markTarget][iVertexTarget+k];
355 
356  /*--- Prune small coefficients. ---*/
357  auto info = PruneSmallCoefficients(SU2_TYPE::GetValue(pruneTol), interpMat.cols(), interpMat[k]);
358  auto nnz = info.first;
359  totalDonors += nnz;
360  minDonors = min(minDonors, nnz);
361  maxDonors = max(maxDonors, nnz);
362  auto corr = fabs(info.second-1.0); // far from 1 either way is bad;
363  sumCorr += corr;
364  maxCorr = max(maxCorr, corr);
365 
366  /*--- Allocate and set donor information for this target point. ---*/
367  targetVertex.resize(nnz);
368 
369  for (unsigned long iVertex = 0, iSet = 0; iVertex < nGlobalVertexDonor; ++iVertex) {
370  auto coeff = interpMat(k,iVertex);
371  if (fabs(coeff) > 0.0) {
372  targetVertex.processor[iSet] = donorProc[iVertex];
373  targetVertex.globalPoint[iSet] = donorPoint[iVertex];
374  targetVertex.coefficient[iSet] = coeff;
375  ++iSet;
376  }
377  }
378  }
379  } // end target vertex loop
380  END_SU2_OMP_FOR
381  SU2_OMP_CRITICAL
382  {
383  totalDonorPoints += totalDonors;
384  MinDonors = min(MinDonors, minDonors);
385  MaxDonors = max(MaxDonors, maxDonors);
386  AvgCorrection += sumCorr;
387  MaxCorrection = max(MaxCorrection, maxCorr);
388  }
389  END_SU2_OMP_CRITICAL
390  }
391  END_SU2_OMP_PARALLEL
392 
393  /*--- Free global data that will no longer be used. ---*/
394  donorCoord.resize(0,0);
395  vector<long>().swap(donorPoint);
396  vector<int>().swap(donorProc);
397  C_inv_trunc.resize(0,0);
398 
399  } // end loop over interface markers
400 
401  /*--- Final reduction of interpolation statistics and basic sanity checks. ---*/
402  auto Reduce = [](SU2_MPI::Op op, unsigned long &val) {
403  auto tmp = val;
404  SU2_MPI::Allreduce(&tmp, &val, 1, MPI_UNSIGNED_LONG, op, SU2_MPI::GetComm());
405  };
406  Reduce(MPI_SUM, totalTargetPoints);
407  Reduce(MPI_SUM, totalDonorPoints);
408  Reduce(MPI_SUM, denseSize);
409  Reduce(MPI_MIN, MinDonors);
410  Reduce(MPI_MAX, MaxDonors);
411 #ifdef HAVE_MPI
412  passivedouble tmp1 = AvgCorrection, tmp2 = MaxCorrection;
413  MPI_Allreduce(&tmp1, &AvgCorrection, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
414  MPI_Allreduce(&tmp2, &MaxCorrection, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
415 #endif
416  if (totalTargetPoints == 0)
417  SU2_MPI::Error("Somehow there are no target interpolation points.", CURRENT_FUNCTION);
418 
419  if (MinDonors == 0)
420  SU2_MPI::Error("One or more target points have no donors, either:\n"
421  " - The interface surfaces are not in contact.\n"
422  " - The RBF radius is too small.\n"
423  " - The pruning tolerance is too aggressive.", CURRENT_FUNCTION);
424 
425  MaxCorrection += 1.0; // put back the reference "1"
426  AvgCorrection = AvgCorrection / totalTargetPoints + 1.0;
427  AvgDonors = totalDonorPoints / totalTargetPoints;
428  Density = totalDonorPoints / (0.01*denseSize);
429 
430 }

432 void CRadialBasisFunction::ComputeGeneratorMatrix(RADIAL_BASIS type, bool usePolynomial,
433  su2double radius, const su2activematrix& coords, int& nPolynomial,
434  vector<int>& keepPolynomialRow, su2passivematrix& C_inv_trunc) {
435 
436  const su2double interfaceCoordTol = 1e6 * numeric_limits<passivedouble>::epsilon();
437 
438  const int nVertexDonor = coords.rows();
439  const int nDim = coords.cols();
440 
441  /*--- Populate interpolation kernel. ---*/
442  CSymmetricMatrix global_M(nVertexDonor);
443 
444  for (int iVertex = 0; iVertex < nVertexDonor; ++iVertex)
445  for (int jVertex = iVertex; jVertex < nVertexDonor; ++jVertex)
446  global_M(iVertex, jVertex) = SU2_TYPE::GetValue(Get_RadialBasisValue(type, radius,
447  GeometryToolbox::Distance(nDim, coords[iVertex], coords[jVertex])));
448 
449  /*--- Invert M matrix (operation is in-place). ---*/
450  const bool kernelIsSPD = (type==RADIAL_BASIS::WENDLAND_C2) || (type==RADIAL_BASIS::GAUSSIAN) ||
451  (type==RADIAL_BASIS::INV_MULTI_QUADRIC);
452  global_M.Invert(kernelIsSPD);
453 
454  /*--- Compute C_inv_trunc. ---*/
455  if (usePolynomial) {
456 
457  /*--- Fill P matrix (P for points, with an extra top row of ones). ---*/
458  su2passivematrix P(1+nDim, nVertexDonor);
459 
460  for (int iVertex = 0; iVertex < nVertexDonor; iVertex++) {
461  P(0, iVertex) = 1.0;
462  for (int iDim = 0; iDim < nDim; ++iDim)
463  P(1+iDim, iVertex) = SU2_TYPE::GetValue(coords(iVertex, iDim));
464  }
465 
466  /*--- Check if points lie on a plane and remove one coordinate from P if so. ---*/
467  nPolynomial = CheckPolynomialTerms(interfaceCoordTol, keepPolynomialRow, P);
468 
469  /*--- Compute Q = P * M^-1 ---*/
470  su2passivematrix Q;
471  global_M.MatMatMult('R', P, Q);
472 
473  /*--- Compute Mp = (Q * P^T)^-1 ---*/
474  CSymmetricMatrix Mp(nPolynomial+1);
475 
476  for (int i = 0; i <= nPolynomial; ++i)
477  for (int j = i; j <= nPolynomial; ++j) {
478  Mp(i,j) = 0.0;
479  for (int k = 0; k < nVertexDonor; ++k)
480  Mp(i,j) += Q(i,k) * P(j,k);
481  }
482  Mp.Invert(false);
483 
484  /*--- Compute M_p * Q, the top part of C_inv_trunc. ---*/
485  su2passivematrix C_inv_top;
486  Mp.MatMatMult('L', Q, C_inv_top);
487 
488  /*--- Compute M^-1 - Q^T * C_inv_top, the bottom (main) part of C_inv_trunc. ---*/
489  su2passivematrix C_inv_bot = global_M.StealData();
490 #ifdef HAVE_LAPACK
491  /*--- Order of gemm arguments swapped due to row-major storage. ---*/
492  const char opa = 'N', opb = 'T';
493  const int M = nVertexDonor, N = nVertexDonor, K = nPolynomial+1;
494  // lda = C_inv_top.cols() = M; ldb = Q.cols() = M; ldc = C_inv_bot.cols() = M;
495  const passivedouble alpha = -1.0, beta = 1.0;
496  DGEMM(&opa, &opb, &M, &N, &K, &alpha, C_inv_top[0], &M, Q[0], &M, &beta, C_inv_bot[0], &M);
497 #else // naive product
498  for (int i = 0; i < nVertexDonor; ++i)
499  for (int j = 0; j < nVertexDonor; ++j)
500  for (int k = 0; k <= nPolynomial; ++k)
501  C_inv_bot(i,j) -= Q(k,i) * C_inv_top(k,j);
502 #endif
503 
504  /*--- Merge top and bottom of C_inv_trunc. More intrusive memory
505  * management, or separate handling of top and bottom, would
506  * avoid these copies (and associated temporary vars). ---*/
507  C_inv_trunc.resize(1+nPolynomial+nVertexDonor, nVertexDonor);
508  memcpy(C_inv_trunc[0], C_inv_top.data(), C_inv_top.size()*sizeof(passivedouble));

520 int CRadialBasisFunction::CheckPolynomialTerms(su2double max_diff_tol, vector<int>& keep_row,
521  su2passivematrix &P) {
522  const int m = P.rows();
523  const int n = P.cols();
524 
525  /*--- The first row of P is all ones and we do not care about it for this analysis. ---*/
526  const int n_rows = m-1;
527  keep_row.resize(n_rows);
528 
529  /*--- By default assume points are not on a plane (all rows kept). ---*/
530  int n_polynomial = n_rows;
531  keep_row.resize(n_rows);
532  for (int i = 0; i < n_rows; ++i) keep_row[i] = 1;
533 
534  /*--- Fit a plane through the points in P. ---*/
535 
536  /*--- Compute P times its transpose and invert. ---*/
537  CSymmetricMatrix PPT(n_rows);
538 
539  for (int i = 0; i < n_rows; ++i)
540  for (int j = i; j < n_rows; ++j) {
541  PPT(i,j) = 0.0;
542  for (int k = 0; k < n; ++k) PPT(i,j) += P(i+1,k) * P(j+1,k);
543  }
544  PPT.Invert(true);
545 
546  /*--- RHS for the least squares fit (vector of ones times P). ---*/
547  vector<passivedouble> rhs(n_rows,0.0), coeff(n_rows);
548 
549  for (int i = 0; i < n_rows; ++i)
550  for (int j = 0; j < n; ++j)
551  rhs[i] += P(i+1,j);
552 
553  /*--- Multiply the RHS by the inverse thus obtaining the coefficients. ---*/
554  PPT.MatVecMult(rhs.begin(), coeff.begin());
555 
556  /*--- Determine the maximum deviation of the points from the fitted plane. ---*/
557  passivedouble max_diff = 0.0;
558 
559  for (int j = 0; j < n; ++j)
560  {
561  passivedouble sum = 0.0;
562  for (int i = 0; i < n_rows; ++i) sum += coeff[i] * P(i+1,j);
563 
564  /*--- 1.0 is the arbitrary constant we are assuming when fitting
565  the plane, i.e. the vector of ones used to generate the RHS. ---*/
566  max_diff = max(abs(1.0-sum), max_diff);
567  }
568 
569  /*--- If points lie on plane remove row associated with the maximum coefficient. ---*/
570  if (max_diff < max_diff_tol)
571  {
572  /*--- Find the max coeff and mark the corresponding row for removal. ---*/
573  int remove_row = 0;
574  for (int i = 1; i < n_rows; ++i)
575  if (abs(coeff[i]) > abs(coeff[remove_row]))
576  remove_row = i;
577 
578  /*--- Mark row as removed and adjust number of polynomial terms. ---*/
579  n_polynomial = n_rows-1;
580  keep_row[remove_row] = 0;
581 
582  /*--- Truncated P by shifting rows "up". ---*/
583  for (auto i = remove_row+1; i < m-1; ++i)
584  for (int j = 0; j < n; ++j)
585  P(i,j) = P(i+1,j);
586  }
587 
588  return n_polynomial;
589 }
