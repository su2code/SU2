44 void CNearestNeighbor::SetTransferCoeff(const CConfig* const* config) {
45 
46  /*--- Desired number of donor points. ---*/
47  const auto nDonor = max<unsigned long>(config[donorZone]->GetNumNearestNeighbors(), 1);
48 
49  /*--- Epsilon used to avoid division by zero. ---*/
50  const su2double eps = numeric_limits<passivedouble>::epsilon();
51 
52  const int nProcessor = size;
53  const auto nMarkerInt = config[donorZone]->GetMarker_n_ZoneInterface()/2;
54  const auto nDim = donor_geometry->GetnDim();
55 
56  Buffer_Receive_nVertex_Donor = new unsigned long [nProcessor];
57 
58  targetVertices.resize(config[targetZone]->GetnMarker_All());
59 
60  vector<vector<DonorInfo> > DonorInfoVec(omp_get_max_threads());
61 
62  /*--- Cycle over nMarkersInt interface to determine communication pattern. ---*/
63 
64  AvgDistance = MaxDistance = 0.0;
65  unsigned long totalTargetPoints = 0;
66 
67  for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; iMarkerInt++) {
68 
69  /*--- On the donor side: find the tag of the boundary sharing the interface. ---*/
70  const auto markDonor = config[donorZone]->FindInterfaceMarker(iMarkerInt);
71 
72  /*--- On the target side: find the tag of the boundary sharing the interface. ---*/
73  const auto markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
74 
75  /*--- Checks if the zone contains the interface, if not continue to the next step. ---*/
76  if (!CheckInterfaceBoundary(markDonor, markTarget)) continue;
77 
78  unsigned long nVertexDonor = 0, nVertexTarget = 0;
79  if (markDonor != -1) nVertexDonor = donor_geometry->GetnVertex(markDonor);
80  if (markTarget != -1) nVertexTarget = target_geometry->GetnVertex(markTarget);
81 
82  /*--- Sets MaxLocalVertex_Donor, Buffer_Receive_nVertex_Donor. ---*/
83  Determine_ArraySize(markDonor, markTarget, nVertexDonor, nDim);
84  if (nVertexTarget) targetVertices[markTarget].resize(nVertexTarget);
85 
86  const auto nPossibleDonor = accumulate(Buffer_Receive_nVertex_Donor,
87  Buffer_Receive_nVertex_Donor+nProcessor, 0ul);
88 
89  Buffer_Send_Coord.resize(MaxLocalVertex_Donor, nDim);
90  Buffer_Send_GlobalPoint.resize(MaxLocalVertex_Donor);
91  Buffer_Receive_Coord.resize(nProcessor * MaxLocalVertex_Donor, nDim);
92  Buffer_Receive_GlobalPoint.resize(nProcessor * MaxLocalVertex_Donor);
93 
94  /*--- Collect coordinates and global point indices. ---*/
95  Collect_VertexInfo(markDonor, markTarget, nVertexDonor, nDim);
96 
97  /*--- Find the closest donor points to each target. ---*/
98  SU2_OMP_PARALLEL
99  {
100  /*--- Working array for this thread. ---*/
101  auto& donorInfo = DonorInfoVec[omp_get_thread_num()];
102  donorInfo.resize(nPossibleDonor);
103 
104  su2double avgDist = 0.0, maxDist = 0.0;
105  unsigned long numTarget = 0;
106 
107  SU2_OMP_FOR_DYN(roundUpDiv(nVertexTarget,2*omp_get_max_threads()))
108  for (auto iVertexTarget = 0ul; iVertexTarget < nVertexTarget; iVertexTarget++) {
109 
110  auto& target_vertex = targetVertices[markTarget][iVertexTarget];
111  const auto Point_Target = target_geometry->vertex[markTarget][iVertexTarget]->GetNode();
112 
113  if (!target_geometry->nodes->GetDomain(Point_Target)) continue;
114 
115  /*--- Coordinates of the target point. ---*/
116  const su2double* Coord_i = target_geometry->nodes->GetCoord(Point_Target);
117 
118  /*--- Compute all distances. ---*/
119  for (int iProcessor = 0, iDonor = 0; iProcessor < nProcessor; ++iProcessor) {
120  for (auto jVertex = 0ul; jVertex < Buffer_Receive_nVertex_Donor[iProcessor]; ++jVertex) {
121 
122  const auto idx = iProcessor*MaxLocalVertex_Donor + jVertex;
123  const auto pGlobalPoint = Buffer_Receive_GlobalPoint[idx];
124  const su2double* Coord_j = Buffer_Receive_Coord[idx];
125  const auto dist2 = GeometryToolbox::SquaredDistance(nDim, Coord_i, Coord_j);
126 
127  donorInfo[iDonor++] = DonorInfo(dist2, pGlobalPoint, iProcessor);
128  }
129  }
130 
131  /*--- Find k closest points. ---*/
132  partial_sort(donorInfo.begin(), donorInfo.begin()+nDonor, donorInfo.end(),
133  [](const DonorInfo& a, const DonorInfo& b) {
134  /*--- Global index is used as tie-breaker to make sorted order independent of initial. ---*/
135  return (a.dist != b.dist)? (a.dist < b.dist) : (a.pidx < b.pidx);
136  }
137  );
138 
139  /*--- Update stats. ---*/
140  numTarget += 1;
141  su2double d = sqrt(donorInfo[0].dist);
142  avgDist += d;
143  maxDist = max(maxDist, d);
144 
145  /*--- Compute interpolation numerators and denominator. ---*/
146  su2double denom = 0.0;
147  for (auto iDonor = 0ul; iDonor < nDonor; ++iDonor) {
148  donorInfo[iDonor].dist = 1.0 / (donorInfo[iDonor].dist + eps);
149  denom += donorInfo[iDonor].dist;
150  }
151 
152  /*--- Set interpolation coefficients. ---*/
153  target_vertex.resize(nDonor);
154 
155  for (auto iDonor = 0ul; iDonor < nDonor; ++iDonor) {
156  target_vertex.globalPoint[iDonor] = donorInfo[iDonor].pidx;
157  target_vertex.processor[iDonor] = donorInfo[iDonor].proc;
158  target_vertex.coefficient[iDonor] = donorInfo[iDonor].dist/denom;
159  }
160  }
161  END_SU2_OMP_FOR
162  SU2_OMP_CRITICAL
163  {
164  totalTargetPoints += numTarget;
165  AvgDistance += avgDist;
166  MaxDistance = max(MaxDistance, maxDist);
167  }
168  END_SU2_OMP_CRITICAL
169  }
170  END_SU2_OMP_PARALLEL
171 
172  }
173 
174  delete[] Buffer_Receive_nVertex_Donor;
175 
176  unsigned long tmp = totalTargetPoints;
177  SU2_MPI::Allreduce(&tmp, &totalTargetPoints, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
178  su2double tmp1 = AvgDistance, tmp2 = MaxDistance;
179  SU2_MPI::Allreduce(&tmp1, &AvgDistance, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
180  SU2_MPI::Allreduce(&tmp2, &MaxDistance, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
181  AvgDistance /= totalTargetPoints;
182 
183 }
