58 CSolver** CSolverFactory::CreateSolverContainer(MAIN_SOLVER kindMainSolver, CConfig *config, CGeometry *geometry, int iMGLevel){
59 
60  CSolver** solver;
61 
62  solver = new CSolver*[MAX_SOLS]();
63 
64  switch (kindMainSolver) {
65  case MAIN_SOLVER::TEMPLATE_SOLVER:
66  solver[FLOW_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::TEMPLATE, solver, geometry, config, iMGLevel);
67  break;
68  case MAIN_SOLVER::INC_EULER:
69  solver[FLOW_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::INC_EULER, solver, geometry, config, iMGLevel);
70  solver[RAD_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::RADIATION, solver, geometry, config, iMGLevel);
71  break;
72  case MAIN_SOLVER::EULER:

74  break;
75  case MAIN_SOLVER::NEMO_EULER:

77  break;
78  case MAIN_SOLVER::INC_NAVIER_STOKES:
79  solver[FLOW_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::INC_NAVIER_STOKES, solver, geometry, config, iMGLevel);
80  solver[HEAT_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::HEAT, solver, geometry, config, iMGLevel);
81  solver[RAD_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::RADIATION, solver, geometry, config, iMGLevel);
82  solver[SPECIES_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::SPECIES, solver, geometry, config, iMGLevel);

88  case MAIN_SOLVER::NEMO_NAVIER_STOKES:
89  solver[FLOW_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::NEMO_NAVIER_STOKES, solver, geometry, config, iMGLevel);
90  solver[SPECIES_SOL] = CreateSubSolver(SUB_SOLVER_TYPE::SPECIES, solver, geometry, config, iMGLevel);

207  return solver;

211 CSolver* CSolverFactory::CreateSubSolver(SUB_SOLVER_TYPE kindSolver, CSolver **solver, CGeometry *geometry, CConfig *config, int iMGLevel){
212 
213  CSolver *genericSolver = nullptr;
214 
215  TURB_MODEL kindTurbModel = config->GetKind_Turb_Model();
216  TURB_TRANS_MODEL kindTransModel = config->GetKind_Trans_Model();
217 
218  SolverMetaData metaData;
219 
220  metaData.solverType = kindSolver;
221 
222  switch (kindSolver) {
223  case SUB_SOLVER_TYPE::CONT_ADJ_EULER:
224  genericSolver = new CAdjEulerSolver(geometry, config, iMGLevel);
225  metaData.integrationType = INTEGRATION_TYPE::MULTIGRID;
226  break;
227  case SUB_SOLVER_TYPE::CONT_ADJ_NAVIER_STOKES:
228  genericSolver = new CAdjNSSolver(geometry, config, iMGLevel);
229  metaData.integrationType = INTEGRATION_TYPE::MULTIGRID;
230  break;
231  case SUB_SOLVER_TYPE::CONT_ADJ_TURB:
232  genericSolver = CreateTurbSolver(kindTurbModel, solver, geometry, config, iMGLevel, true);
233  metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
234  break;
235  case SUB_SOLVER_TYPE::DISC_ADJ_TURB:
236  genericSolver = CreateTurbSolver(kindTurbModel, solver, geometry, config, iMGLevel, true);
237  metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
238  break;
239  case SUB_SOLVER_TYPE::BASELINE:
240  genericSolver = new CBaselineSolver(geometry, config);
241  metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
242  break;
243  case SUB_SOLVER_TYPE::BASELINE_FEM:
244  genericSolver = new CBaselineSolver_FEM(geometry, config);
245  metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
246  break;
247  case SUB_SOLVER_TYPE::DISC_ADJ_FEA:
248  genericSolver = new CDiscAdjFEASolver(geometry, config, solver[FEA_SOL], RUNTIME_FEA_SYS, iMGLevel);
249  metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
250  break;
251  case SUB_SOLVER_TYPE::DISC_ADJ_MESH:
252  genericSolver = CreateMeshSolver(solver, geometry, config, iMGLevel, true);
253  metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
254  break;
255  case SUB_SOLVER_TYPE::DISC_ADJ_FLOW:
256  genericSolver = new CDiscAdjSolver(geometry, config, solver[FLOW_SOL], RUNTIME_FLOW_SYS, iMGLevel);
257  metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
258  break;
259  case SUB_SOLVER_TYPE::EULER:
260  case SUB_SOLVER_TYPE::INC_EULER:
261  case SUB_SOLVER_TYPE::NEMO_EULER:
262  case SUB_SOLVER_TYPE::NAVIER_STOKES:
263  case SUB_SOLVER_TYPE::INC_NAVIER_STOKES:
264  case SUB_SOLVER_TYPE::NEMO_NAVIER_STOKES:
265  genericSolver = CreateFlowSolver(kindSolver, solver, geometry, config, iMGLevel);
266  if (!config->GetNewtonKrylov() || config->GetDiscrete_Adjoint() || config->GetContinuous_Adjoint())
267  metaData.integrationType = INTEGRATION_TYPE::MULTIGRID;
268  else
269  metaData.integrationType = INTEGRATION_TYPE::NEWTON;
270  break;
271  case SUB_SOLVER_TYPE::FEA:
272  genericSolver = new CFEASolver(geometry, config);
273  metaData.integrationType = INTEGRATION_TYPE::STRUCTURAL;
274  break;
275  case SUB_SOLVER_TYPE::MESH:
276  genericSolver = CreateMeshSolver(solver, geometry, config, iMGLevel, false);
277  metaData.integrationType = INTEGRATION_TYPE::NONE;
278  break;
279  case SUB_SOLVER_TYPE::DG_EULER:
280  case SUB_SOLVER_TYPE::DG_NAVIER_STOKES:
281  genericSolver = CreateDGSolver(kindSolver, geometry, config, iMGLevel);
282  metaData.integrationType = INTEGRATION_TYPE::FEM_DG;
283  break;
284  case SUB_SOLVER_TYPE::HEAT:
285  genericSolver = CreateHeatSolver(solver, geometry, config, iMGLevel, false);
286  metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
287  break;
288  case SUB_SOLVER_TYPE::DISC_ADJ_HEAT:
289  genericSolver = CreateHeatSolver(solver, geometry, config, iMGLevel, true);
290  metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
291  break;
292  case SUB_SOLVER_TYPE::TRANSITION:
293  genericSolver = CreateTransSolver(kindTransModel, solver, geometry, config, iMGLevel, false);
294  metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
295  break;
296  case SUB_SOLVER_TYPE::SPECIES:
297  genericSolver = CreateSpeciesSolver(solver, geometry, config, iMGLevel, false);
298  metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
299  break;
300  case SUB_SOLVER_TYPE::DISC_ADJ_SPECIES:
301  genericSolver = CreateSpeciesSolver(solver, geometry, config, iMGLevel, true);
302  metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
303  break;
304  case SUB_SOLVER_TYPE::TURB:
305  case SUB_SOLVER_TYPE::TURB_SA:
306  case SUB_SOLVER_TYPE::TURB_SST:
307  genericSolver = CreateTurbSolver(kindTurbModel, solver, geometry, config, iMGLevel, false);
308  metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
309  break;
310  case SUB_SOLVER_TYPE::TEMPLATE:
311  genericSolver = new CTemplateSolver(geometry, config);
312  metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
313  break;
314  case SUB_SOLVER_TYPE::RADIATION:
315  if (config->AddRadiation()){
316  genericSolver = new CRadP1Solver(geometry, config);
317  }
318  metaData.integrationType = INTEGRATION_TYPE::SINGLEGRID;
319  break;
320  case SUB_SOLVER_TYPE::DISC_ADJ_RADIATION:
321  if (config->AddRadiation()){
322  genericSolver = new CDiscAdjSolver(geometry, config, solver[RAD_SOL], RUNTIME_RADIATION_SYS, iMGLevel);
323  }
324  metaData.integrationType = INTEGRATION_TYPE::DEFAULT;
325  break;
326  default:
327  SU2_MPI::Error("No proper allocation found for requested sub solver", CURRENT_FUNCTION);
328  break;
329  }
330 
331  if (genericSolver != nullptr)
332  allocatedSolvers[genericSolver] = metaData;
333 
334  return genericSolver;
335 
336 }

338 CSolver* CSolverFactory::CreateTurbSolver(TURB_MODEL kindTurbModel, CSolver **solver, CGeometry *geometry, CConfig *config, int iMGLevel, int adjoint){
339 
340  CSolver *turbSolver = nullptr;
341 
342  if (!adjoint){
343  switch (TurbModelFamily(kindTurbModel)) {
344  case TURB_FAMILY::SA:
345  turbSolver = new CTurbSASolver(geometry, config, iMGLevel, solver[FLOW_SOL]->GetFluidModel());
346  solver[FLOW_SOL]->Preprocessing(geometry, solver, config, iMGLevel, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
347  turbSolver->Postprocessing(geometry, solver, config, iMGLevel);
348  break;
349  case TURB_FAMILY::KW:
350  turbSolver = new CTurbSSTSolver(geometry, config, iMGLevel);
351  solver[FLOW_SOL]->Preprocessing(geometry, solver, config, iMGLevel, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
352  turbSolver->Postprocessing(geometry, solver, config, iMGLevel);
353  solver[FLOW_SOL]->Preprocessing(geometry, solver, config, iMGLevel, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
354  break;
355  case TURB_FAMILY::NONE:
356  SU2_MPI::Error("Trying to create TurbSolver container but TURB_MODEL=NONE.", CURRENT_FUNCTION);
357  break;
358  }
359  } else {
360 
361  if (config->GetDiscrete_Adjoint()){
362  if (!config->GetFrozen_Visc_Disc())
363  turbSolver = new CDiscAdjSolver(geometry, config, solver[TURB_SOL], RUNTIME_TURB_SYS, iMGLevel);
364  } else {
365  if (!config->GetFrozen_Visc_Cont())
366  turbSolver = new CAdjTurbSolver(geometry, config, iMGLevel);
367  }
368  }
369 
370  return turbSolver;
371 }

373 CSolver* CSolverFactory::CreateTransSolver(TURB_TRANS_MODEL kindTransModel, CSolver **solver, CGeometry *geometry, CConfig *config, int iMGLevel, int adjoint){
374 
375  CSolver *transSolver = nullptr;
376 
377  if (config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE) {
378  switch (kindTransModel) {
379  case TURB_TRANS_MODEL::LM :
380  transSolver = new CTransLMSolver(geometry, config, iMGLevel);
381  solver[FLOW_SOL]->Preprocessing(geometry, solver, config, iMGLevel, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
382  transSolver->Postprocessing(geometry, solver, config, iMGLevel);
383  solver[FLOW_SOL]->Preprocessing(geometry, solver, config, iMGLevel, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
384  break;
385  case TURB_TRANS_MODEL::NONE:
386  break;
387  }
388  }
389 
390  return transSolver;
