40 void CIntegration::Space_Integration(CGeometry *geometry,
41  CSolver **solver_container,
42  CNumerics **numerics,
43  CConfig *config, unsigned short iMesh,
44  unsigned short iRKStep,
45  unsigned short RunTime_EqSystem) {
46  unsigned short iMarker, KindBC;
47 
48  unsigned short MainSolver = config->GetContainerPosition(RunTime_EqSystem);
49  bool dual_time = ((config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
50  (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND));
51 
52  /*--- Compute inviscid residuals ---*/
53 
54  switch (config->GetKind_ConvNumScheme()) {
55  case SPACE_CENTERED:
56  solver_container[MainSolver]->Centered_Residual(geometry, solver_container, numerics, config, iMesh, iRKStep);
57  break;
58  case SPACE_UPWIND:
59  solver_container[MainSolver]->Upwind_Residual(geometry, solver_container, numerics, config, iMesh);
60  break;
61  }
62 
63  /*--- Compute viscous residuals ---*/
64  solver_container[MainSolver]->Viscous_Residual(geometry, solver_container, numerics, config, iMesh, iRKStep);
65 
66  /*--- Compute source term residuals ---*/
67  solver_container[MainSolver]->Source_Residual(geometry, solver_container, numerics, config, iMesh);
68 
69  /*--- Add viscous and convective residuals, and compute the Dual Time Source term ---*/
70 
71  if (dual_time)
72  solver_container[MainSolver]->SetResidual_DualTime(geometry, solver_container, config, iRKStep, iMesh, RunTime_EqSystem);
73 
74  /*--- Pick convective and viscous numerics objects for the current thread. ---*/
75 
76  CNumerics* conv_bound_numerics = numerics[CONV_BOUND_TERM + omp_get_thread_num()*MAX_TERMS];
77  CNumerics* visc_bound_numerics = numerics[VISC_BOUND_TERM + omp_get_thread_num()*MAX_TERMS];
78 
79  /*--- Pause preaccumulation in boundary conditions for hybrid parallel AD. ---*/
80  /// TODO: Check if this is really needed.
81  //const auto pausePreacc = (omp_get_num_threads() > 1) && AD::PausePreaccumulation();
82 
83  /*--- Boundary conditions that depend on other boundaries (they require MPI sincronization)---*/
84 
85  solver_container[MainSolver]->BC_Fluid_Interface(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config);
86 
87  /*--- Compute Fourier Transformations for markers where NRBC_BOUNDARY is applied---*/
88 
89  if (config->GetBoolGiles() && config->GetSpatialFourier()){
90  solver_container[MainSolver]->PreprocessBC_Giles(geometry, config, conv_bound_numerics, INFLOW);
91 
92  solver_container[MainSolver]->PreprocessBC_Giles(geometry, config, conv_bound_numerics, OUTFLOW);
93  }
94 
95  /*--- Weak boundary conditions ---*/
96 
97  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
98  KindBC = config->GetMarker_All_KindBC(iMarker);
99  switch (KindBC) {
100  case EULER_WALL:
101  solver_container[MainSolver]->BC_Euler_Wall(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
102  break;
103  case ACTDISK_INLET:
104  solver_container[MainSolver]->BC_ActDisk_Inlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
105  break;
106  case ENGINE_INFLOW:
107  solver_container[MainSolver]->BC_Engine_Inflow(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
108  break;
109  case INLET_FLOW:
110  solver_container[MainSolver]->BC_Inlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
111  break;
112  case ACTDISK_OUTLET:
113  solver_container[MainSolver]->BC_ActDisk_Outlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
114  break;
115  case ENGINE_EXHAUST:
116  solver_container[MainSolver]->BC_Engine_Exhaust(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
117  break;
118  case SUPERSONIC_INLET:
119  solver_container[MainSolver]->BC_Supersonic_Inlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
120  break;
121  case OUTLET_FLOW:
122  solver_container[MainSolver]->BC_Outlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
123  break;
124  case SUPERSONIC_OUTLET:
125  solver_container[MainSolver]->BC_Supersonic_Outlet(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
126  break;
127  case GILES_BOUNDARY:
128  solver_container[MainSolver]->BC_Giles(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
129  break;
130  case RIEMANN_BOUNDARY:
131  if (config->GetBoolTurbomachinery()){
132  solver_container[MainSolver]->BC_TurboRiemann(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
133  }
134  else{
135  solver_container[MainSolver]->BC_Riemann(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
136  }
137  break;
138  case FAR_FIELD:
139  solver_container[MainSolver]->BC_Far_Field(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
140  break;
141  case SYMMETRY_PLANE:
142  solver_container[MainSolver]->BC_Sym_Plane(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
143  break;
144  }
145  }
146 
147  /*--- Modification of the system on the whole domain, like for a strong BC. ---*/
148  solver_container[MainSolver]->Impose_Fixed_Values(geometry, config);
149 
150  /*--- Strong boundary conditions (Navier-Stokes and Dirichlet type BCs) ---*/
151 
152  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++)
153  switch (config->GetMarker_All_KindBC(iMarker)) {
154  case ISOTHERMAL:
155  solver_container[MainSolver]->BC_Isothermal_Wall(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
156  break;
157  case HEAT_FLUX:
158  solver_container[MainSolver]->BC_HeatFlux_Wall(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
159  break;
160  case HEAT_TRANSFER:
161  solver_container[MainSolver]->BC_HeatTransfer_Wall(geometry, config, iMarker);
162  break;
163  case CUSTOM_BOUNDARY:
164  solver_container[MainSolver]->BC_Custom(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
165  break;
166  case CHT_WALL_INTERFACE:
167  if ((MainSolver == HEAT_SOL) || ((MainSolver == FLOW_SOL) && ((config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) || config->GetEnergy_Equation()))) {
168  solver_container[MainSolver]->BC_ConjugateHeat_Interface(geometry, solver_container, conv_bound_numerics, config, iMarker);
169  }
170  else {
171  solver_container[MainSolver]->BC_HeatFlux_Wall(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
172  }
173  break;
174  case SMOLUCHOWSKI_MAXWELL:
175  solver_container[MainSolver]->BC_Smoluchowski_Maxwell(geometry, solver_container, conv_bound_numerics, visc_bound_numerics, config, iMarker);
176  break;
177  }
178 
179  /*--- Complete residuals for periodic boundary conditions. We loop over
180  the periodic BCs in matching pairs so that, in the event that there are
181  adjacent periodic markers, the repeated points will have their residuals
182  accumulated correctly during the communications. ---*/
183 
184  if (config->GetnMarker_Periodic() > 0) {
185  solver_container[MainSolver]->BC_Periodic(geometry, solver_container, conv_bound_numerics, config);
186  }
187 
188  //AD::ResumePreaccumulation(pausePreacc);
189 
190 }

192 void CIntegration::Time_Integration(CGeometry *geometry, CSolver **solver_container, CConfig *config,
193  unsigned short iRKStep, unsigned short RunTime_EqSystem) {
194 
195  unsigned short MainSolver = config->GetContainerPosition(RunTime_EqSystem);
196 
197  switch (config->GetKind_TimeIntScheme()) {
198  case (RUNGE_KUTTA_EXPLICIT):
199  solver_container[MainSolver]->ExplicitRK_Iteration(geometry, solver_container, config, iRKStep);
200  break;
201  case (CLASSICAL_RK4_EXPLICIT):
202  solver_container[MainSolver]->ClassicalRK4_Iteration(geometry, solver_container, config, iRKStep);
203  break;
204  case (EULER_EXPLICIT):
205  solver_container[MainSolver]->ExplicitEuler_Iteration(geometry, solver_container, config);
206  break;
207  case (EULER_IMPLICIT):
208  solver_container[MainSolver]->ImplicitEuler_Iteration(geometry, solver_container, config);
209  break;
210  }

233 void CIntegration::SetDualTime_Solver(const CGeometry *geometry, CSolver *solver, const CConfig *config, unsigned short iMesh) {
234 
235  SU2_OMP_PARALLEL
236  {
237  /*--- Store old solution ---*/
238  solver->GetNodes()->Set_Solution_time_n1();
239  solver->GetNodes()->Set_Solution_time_n();
240 
241  SU2_OMP_SAFE_GLOBAL_ACCESS(solver->ResetCFLAdapt();)
242 
243  SU2_OMP_FOR_STAT(roundUpDiv(geometry->GetnPoint(), omp_get_num_threads()))
244  for (auto iPoint = 0ul; iPoint < geometry->GetnPoint(); iPoint++) {
245 
246  /*--- Initialize the underrelaxation ---*/
247  solver->GetNodes()->SetUnderRelaxation(iPoint, 1.0);
248 
249  /*--- Initialize the local CFL number ---*/
250  solver->GetNodes()->SetLocalCFL(iPoint, config->GetCFL(iMesh));
251  }
252  END_SU2_OMP_FOR
253 
254  }
255  END_SU2_OMP_PARALLEL
256 }
