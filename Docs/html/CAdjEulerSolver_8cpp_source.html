462

466

476  nPointTotal_r[iDomain] = nPointTotal_s[iDomain];
477 
478  }
479 
480  /*--- Receive the counts. All processors are sending their counters to
481  iDomain up above, so only iDomain needs to perform the recv here from
482  all other ranks. ---*/
483 
484  if (rank == iDomain) {
485 
486  for (jDomain = 0; jDomain < size; jDomain++) {
487 
488  /*--- A rank does not communicate with itself through MPI ---*/
489 
490  if (rank != jDomain) {
491 
492  /*--- Recv the data by probing for the current sender, jDomain,
493  first and then receiving the values from it. ---*/
494 
495  SU2_MPI::Recv(&nPointTotal_r[jDomain], 1, MPI_UNSIGNED_LONG, jDomain, rank, SU2_MPI::GetComm(), &status);
496 
497  }
498  }
499 
500  }
501  }
502 
503  /*--- Wait for the non-blocking sends to complete. ---*/
504 
505  SU2_MPI::Barrier(SU2_MPI::GetComm());

508

510

512 
513  /*--- Set the value of the interior geometry. Initialize counters. ---*/
514 
515  iPointTotal = 0;
516 
517  /*--- Load up the actual values into the buffers for sending. ---*/
518 
519  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
520 
521  if ((config->GetMarker_All_KindBC(iMarker) == ACTDISK_INLET) ||
522  (config->GetMarker_All_KindBC(iMarker) == ACTDISK_OUTLET)) {
523 
524  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
525  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
526  jDomain = geometry->vertex[iMarker][iVertex]->GetDonorProcessor();
527  if ((iDomain == jDomain) && (geometry->nodes->GetDomain(iPoint))) {
528 
529  iGlobalIndex = geometry->nodes->GetGlobalIndex(iPoint);
530  jVertex = geometry->vertex[iMarker][iVertex]->GetDonorVertex();
531  jMarker = geometry->vertex[iMarker][iVertex]->GetDonorMarker();
532

534  Buffer_Send_AdjVar[(nVar+3)*(PointTotal_Counter+iPointTotal)+iVar] = nodes->GetSolution(iPoint,iVar);
535  }
536  Buffer_Send_AdjVar[(nVar+3)*(PointTotal_Counter+iPointTotal)+(nVar+0)] = su2double(iGlobalIndex);
537  Buffer_Send_AdjVar[(nVar+3)*(PointTotal_Counter+iPointTotal)+(nVar+1)] = su2double(jVertex);
538  Buffer_Send_AdjVar[(nVar+3)*(PointTotal_Counter+iPointTotal)+(nVar+2)] = su2double(jMarker);

542  }

544  }

546  }

561  }

669 void CAdjEulerSolver::SetForceProj_Vector(CGeometry *geometry, CSolver **solver_container, CConfig *config) {
670 
671  su2double *ForceProj_Vector, x = 0.0, y = 0.0, z = 0.0, *Normal, CD, CL, Cp, CpTarget,
672  CT, CQ, x_origin, y_origin, z_origin, WDrag, Area, invCD, CLCD2, invCQ, CTRCQ2;
673  unsigned short iMarker,jMarker,iMarker_Monitoring, iDim;
674  unsigned long iVertex, iPoint;
675  string Marker_Tag, Monitoring_Tag;
676  su2double Weight_ObjFunc=1.0;
677  su2double *ForceProj_Vector2;
678 
679  su2double Alpha = (config->GetAoA()*PI_NUMBER)/180.0;
680  su2double Beta = (config->GetAoS()*PI_NUMBER)/180.0;
681  su2double RefLength = config->GetRefLength();
682  const auto RefOriginMoment = config->GetRefOriginMoment(0);
683  su2double dCD_dCL = config->GetdCD_dCL();
684  su2double dCMx_dCL = config->GetdCMx_dCL();
685  su2double dCMy_dCL = config->GetdCMy_dCL();
686  su2double dCMz_dCL = config->GetdCMz_dCL();
687  bool Fixed_CL = config->GetFixed_CL_Mode();
688 
689  ForceProj_Vector = new su2double[nDim];
690 
691  /*--- Compute coefficients needed for objective function evaluation. ---*/
692 
693  CD = solver_container[FLOW_SOL]->GetTotal_CD();
694  CL = solver_container[FLOW_SOL]->GetTotal_CL();
695  CT = solver_container[FLOW_SOL]->GetTotal_CT();
696  CQ = solver_container[FLOW_SOL]->GetTotal_CQ();
697  invCD = 1.0/CD; CLCD2 = CL/(CD*CD);
698  invCQ = 1.0/CQ; CTRCQ2 = CT/(RefLength*CQ*CQ);
699 
700  x_origin = RefOriginMoment[0]; y_origin = RefOriginMoment[1]; z_origin = RefOriginMoment[2];
701 
702  /*--- Evaluate the boundary condition coefficients,
703  Since there may be more than one objective per marker, first we have to set all Force projection vectors to 0 ---*/
704 
705  for (iMarker = 0; iMarker<nMarker; iMarker++) {
706  if ((config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE) &&
707  (config->GetMarker_All_Monitoring(iMarker) == YES))
708  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
709  for (iDim=0; iDim<nDim; iDim++)
710  ForceProj_Vector[iDim]=0.0;
711  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
712  nodes->SetForceProj_Vector(iPoint,ForceProj_Vector);
713  }
714  }
715 
716  /*--- Find the matching iMarker ---*/
717 
718  for (iMarker_Monitoring = 0; iMarker_Monitoring < config->GetnMarker_Monitoring(); iMarker_Monitoring++) {
719  Weight_ObjFunc = config->GetWeight_ObjFunc(iMarker_Monitoring);
720  Monitoring_Tag = config->GetMarker_Monitoring_TagBound(iMarker_Monitoring);
721  for (jMarker=0; jMarker<nMarker; jMarker++) {
722  Marker_Tag = config->GetMarker_All_TagBound(jMarker);
723  if (Monitoring_Tag==Marker_Tag)
724  iMarker = jMarker;
725  }
726 
727 
728  if ((iMarker<nMarker) && (config->GetMarker_All_KindBC(iMarker) != SEND_RECEIVE) &&
729  (config->GetMarker_All_Monitoring(iMarker) == YES)) {
730  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
731 
732  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
733 
734  x = geometry->nodes->GetCoord(iPoint, 0);
735  y = geometry->nodes->GetCoord(iPoint, 1);
736  if (nDim == 3) z = geometry->nodes->GetCoord(iPoint, 2);
737 
738  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
739  ForceProj_Vector2 = nodes->GetForceProj_Vector(iPoint);
740  for (iDim=0; iDim<nDim;iDim++)
741  ForceProj_Vector[iDim]=ForceProj_Vector2[iDim];
742 
743  switch (config->GetKind_ObjFunc(iMarker_Monitoring)) {
744  case DRAG_COEFFICIENT :
745  if (nDim == 2) {
746 
747  ForceProj_Vector[0] += Weight_ObjFunc*cos(Alpha);
748  ForceProj_Vector[1] += Weight_ObjFunc*sin(Alpha);
749 
750  /*--- Modification to run at a fixed CL and CM value ---*/
751 
752  if (Fixed_CL) { ForceProj_Vector[0] += dCD_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= dCD_dCL*Weight_ObjFunc*cos(Alpha); }
753 
754  }
755  if (nDim == 3) {
756 
757  ForceProj_Vector[0] += Weight_ObjFunc*cos(Alpha)*cos(Beta);
758  ForceProj_Vector[1] += Weight_ObjFunc*sin(Beta);
759  ForceProj_Vector[2] += Weight_ObjFunc*sin(Alpha)*cos(Beta);
760 
761  /*--- Modification to run at a fixed CL value ---*/
762 
763  if (Fixed_CL) { ForceProj_Vector[0] += dCD_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= 0.0; ForceProj_Vector[2] -= dCD_dCL*Weight_ObjFunc*cos(Alpha); }
764 
765  }
766  break;
767  case LIFT_COEFFICIENT :
768  if (nDim == 2) { ForceProj_Vector[0] += -Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] += Weight_ObjFunc*cos(Alpha); }
769  if (nDim == 3) { ForceProj_Vector[0] += -Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] = 0.0; ForceProj_Vector[2] += Weight_ObjFunc*cos(Alpha); }
770  break;
771  case SIDEFORCE_COEFFICIENT :
772  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
773  if (nDim == 3) { ForceProj_Vector[0] += -Weight_ObjFunc*sin(Beta) * cos(Alpha); ForceProj_Vector[1] += Weight_ObjFunc*cos(Beta); ForceProj_Vector[2] += -Weight_ObjFunc*sin(Beta) * sin(Alpha); }
774  break;
775  case INVERSE_DESIGN_PRESSURE :
776  Cp = solver_container[FLOW_SOL]->GetCPressure(iMarker, iVertex);
777  CpTarget = solver_container[FLOW_SOL]->GetCPressureTarget(iMarker, iVertex);
778  Area = sqrt(Normal[0]*Normal[0] + Normal[1]*Normal[1]);
779  if (nDim == 3) Area = sqrt(Normal[0]*Normal[0] + Normal[1]*Normal[1] + Normal[2]*Normal[2]);
780  ForceProj_Vector[0] += -Weight_ObjFunc*2.0*(Cp-CpTarget)*Normal[0]/Area; ForceProj_Vector[1] += -Weight_ObjFunc*2.0*(Cp-CpTarget)*Normal[1]/Area;
781  if (nDim == 3) ForceProj_Vector[2] += -Weight_ObjFunc*2.0*(Cp-CpTarget)*Normal[2]/Area;
782  break;
783  case MOMENT_X_COEFFICIENT :
784  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
785  if (nDim == 3) {
786 
787  ForceProj_Vector[0] += 0.0;
788  ForceProj_Vector[1] += -Weight_ObjFunc*(z - z_origin)/RefLength;
789  ForceProj_Vector[2] += Weight_ObjFunc*(y - y_origin)/RefLength;
790 
791  /*--- Modification to run at a fixed CL value ---*/
792 
793  if (Fixed_CL) { ForceProj_Vector[0] += dCMx_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= 0.0; ForceProj_Vector[2] -= dCMx_dCL*Weight_ObjFunc*cos(Alpha); }
794 
795  }
796  break;
797  case MOMENT_Y_COEFFICIENT :
798  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
799  if (nDim == 3) {
800 
801  ForceProj_Vector[0] += Weight_ObjFunc*(z - z_origin)/RefLength;
802  ForceProj_Vector[1] += 0.0;
803  ForceProj_Vector[2] += -Weight_ObjFunc*(x - x_origin)/RefLength;
804 
805  /*--- Modification to run at a fixed CL value ---*/
806 
807  if (Fixed_CL) { ForceProj_Vector[0] += dCMy_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= 0.0; ForceProj_Vector[2] -= dCMy_dCL*Weight_ObjFunc*cos(Alpha); }
808 
809  }
810  break;
811  case MOMENT_Z_COEFFICIENT :
812  if (nDim == 2) {
813 
814  ForceProj_Vector[0] += -Weight_ObjFunc*(y - y_origin)/RefLength;
815  ForceProj_Vector[1] += Weight_ObjFunc*(x - x_origin)/RefLength;
816 
817  /*--- Modification to run at a fixed CL and CM value ---*/
818 
819  if (Fixed_CL) { ForceProj_Vector[0] += dCMz_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= dCMz_dCL*Weight_ObjFunc*cos(Alpha); }
820 
821  }
822  if (nDim == 3) {
823 
824  ForceProj_Vector[0] += -Weight_ObjFunc*(y - y_origin)/RefLength;
825  ForceProj_Vector[1] += Weight_ObjFunc*(x - x_origin)/RefLength;
826  ForceProj_Vector[2] += 0;
827 
828  /*--- Modification to run at a fixed CL value ---*/
829 
830  if (Fixed_CL) { ForceProj_Vector[0] += dCMz_dCL*Weight_ObjFunc*sin(Alpha); ForceProj_Vector[1] -= 0.0; ForceProj_Vector[2] -= dCMz_dCL*Weight_ObjFunc*cos(Alpha); }
831 
832  }
833  break;
834  case EFFICIENCY :
835  if (nDim == 2) { ForceProj_Vector[0] += -Weight_ObjFunc*(invCD*sin(Alpha)+CLCD2*cos(Alpha)); ForceProj_Vector[1] += Weight_ObjFunc*(invCD*cos(Alpha)-CLCD2*sin(Alpha)); }
836  if (nDim == 3) { ForceProj_Vector[0] += -Weight_ObjFunc*(invCD*sin(Alpha)+CLCD2*cos(Alpha)*cos(Beta)); ForceProj_Vector[1] += -Weight_ObjFunc*CLCD2*sin(Beta); ForceProj_Vector[2] += Weight_ObjFunc*(invCD*cos(Alpha)-CLCD2*sin(Alpha)*cos(Beta)); }
837  break;
838  case EQUIVALENT_AREA :
839  WDrag = Weight_ObjFunc*config->GetWeightCd();
840  if (nDim == 2) { ForceProj_Vector[0] = cos(Alpha)*WDrag; ForceProj_Vector[1] = sin(Alpha)*WDrag; }
841  if (nDim == 3) { ForceProj_Vector[0] = cos(Alpha)*cos(Beta)*WDrag; ForceProj_Vector[1] = sin(Beta)*WDrag; ForceProj_Vector[2] = sin(Alpha)*cos(Beta)*WDrag; }
842  break;
843  case NEARFIELD_PRESSURE :
844  WDrag = Weight_ObjFunc*config->GetWeightCd();
845  if (nDim == 2) { ForceProj_Vector[0] = cos(Alpha)*WDrag; ForceProj_Vector[1] = sin(Alpha)*WDrag; }
846  if (nDim == 3) { ForceProj_Vector[0] = cos(Alpha)*cos(Beta)*WDrag; ForceProj_Vector[1] = sin(Beta)*WDrag; ForceProj_Vector[2] = sin(Alpha)*cos(Beta)*WDrag; }
847  break;
848  case FORCE_X_COEFFICIENT :
849  if (nDim == 2) { ForceProj_Vector[0] += Weight_ObjFunc*1.0; ForceProj_Vector[1] += 0.0; }
850  if (nDim == 3) { ForceProj_Vector[0] += Weight_ObjFunc*1.0; ForceProj_Vector[1] += 0.0; ForceProj_Vector[2] += 0.0; }
851  break;
852  case FORCE_Y_COEFFICIENT :
853  if (nDim == 2) { ForceProj_Vector[0] += 0.0; ForceProj_Vector[1] += Weight_ObjFunc*1.0; }
854  if (nDim == 3) { ForceProj_Vector[0] += 0.0; ForceProj_Vector[1] += Weight_ObjFunc*1.0; ForceProj_Vector[2] += 0.0; }
855  break;
856  case FORCE_Z_COEFFICIENT :
857  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
858  if (nDim == 3) { ForceProj_Vector[0] += 0.0; ForceProj_Vector[1] += 0.0; ForceProj_Vector[2] += Weight_ObjFunc*1.0; }
859  break;
860  case THRUST_COEFFICIENT :
861  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
862  if (nDim == 3) { ForceProj_Vector[0] += 0.0; ForceProj_Vector[1] += 0.0; ForceProj_Vector[2] += Weight_ObjFunc*1.0; }
863  break;
864  case TORQUE_COEFFICIENT :
865  if (nDim == 2) { ForceProj_Vector[0] += Weight_ObjFunc*(y - y_origin)/RefLength; ForceProj_Vector[1] += -Weight_ObjFunc*(x - x_origin)/RefLength; }
866  if (nDim == 3) { ForceProj_Vector[0] += Weight_ObjFunc*(y - y_origin)/RefLength; ForceProj_Vector[1] += -Weight_ObjFunc*(x - x_origin)/RefLength; ForceProj_Vector[2] += 0; }
867  break;
868  case FIGURE_OF_MERIT :
869  if (nDim == 2) { SU2_MPI::Error("This functional is not possible in 2D!!", CURRENT_FUNCTION);}
870  if (nDim == 3) {
871  ForceProj_Vector[0] += -Weight_ObjFunc*invCQ;
872  ForceProj_Vector[1] += -Weight_ObjFunc*CTRCQ2*(z - z_origin);
873  ForceProj_Vector[2] += Weight_ObjFunc*CTRCQ2*(y - y_origin);
874  }
875  break;
876 
877  default :
878  break;
879  }
880 
881  /*--- Store the force projection vector at this node ---*/
882 
883  nodes->SetForceProj_Vector(iPoint,ForceProj_Vector);
884 
885  }
886  }
887  }
888  delete [] ForceProj_Vector;
