116  /*!

125  */
126  inline void DetermineNearestElement(const su2double *coor,
127  su2double &dist,
128  unsigned short &markerID,
129  unsigned long &elemID,
130  int &rankID) {
131  const auto iThread = omp_get_thread_num();
132  DetermineNearestElement_impl(BBoxTargets[iThread], FrontLeaves[iThread],
133  FrontLeavesNew[iThread], coor, dist, markerID, elemID, rankID);
134  }
