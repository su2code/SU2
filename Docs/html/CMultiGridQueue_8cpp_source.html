47 void CMultiGridQueue::AddCV(unsigned long newPoint, short numberNeighbors) {
48 
49  const short maxNeighbors = QueueCV.size()-1;
50 
51  /*--- Basic check ---*/
52  if (newPoint >= nPoint) {
53  SU2_MPI::Error("The index of the CV is greater than the size of the priority list.", CURRENT_FUNCTION);
54  }
55 
56  /*--- Resize the list ---*/
57  if (numberNeighbors > maxNeighbors) {
58  const size_t newSize = numberNeighbors+1;
59  if (QueueCV.capacity() < newSize)
60  QueueCV.reserve(2*newSize);
61  QueueCV.resize(newSize);
62  }
63 
64  /*--- Find the point in the queue ---*/
65  const bool inQueue = (Priority[newPoint] == numberNeighbors);
66 
67  if (!inQueue) {
68  /*--- Add the control volume, and update the priority list ---*/
69  QueueCV[numberNeighbors].push_back(newPoint);
70  Priority[newPoint] = numberNeighbors;
71  }
72 
73 }

75 void CMultiGridQueue::RemoveCV(unsigned long removePoint) {
76 
77  /*--- Basic check ---*/
78  if (removePoint >= nPoint) {
79  SU2_MPI::Error("The index of the CV is greater than the size of the priority list.", CURRENT_FUNCTION);
80  }
81 
82  /*--- Find priority of the Control Volume. ---*/
83  const auto numberNeighbors = Priority[removePoint];
84  if (numberNeighbors == -1) ThrowPointNotInListError(removePoint);
85  Priority[removePoint] = -1;
86 
87  /*--- Find the point in the queue, if the queue is not changed we can exit. ---*/
88  if (!QueueCV[numberNeighbors].findAndErase(removePoint)) return;
89 
90  /*--- Check that the size of the queue is the right one. ---*/
91  /*--- Resize the queue, leaving at least one element in it. ---*/
92  auto sizeQueueCV = QueueCV.size();
93 
94  while (sizeQueueCV > 0) {
95  if (!QueueCV[--sizeQueueCV].empty()) break;
96  }
97  QueueCV.resize(sizeQueueCV+1);
98 
99 }

101 void CMultiGridQueue::MoveCV(unsigned long movePoint, short numberNeighbors) {
102 
103  RightCV[movePoint] = (numberNeighbors >= 0);
104  numberNeighbors = max<short>(numberNeighbors,0);
105 
106  /*--- Remove the control volume ---*/
107  RemoveCV(movePoint);
108 
109  /*--- Add a new control volume ---*/
110  AddCV(movePoint, numberNeighbors);

162 bool CMultiGridQueue::EmptyQueue(void) const {
163 
164  /*--- In case there is only the no agglomerated elements (size 1), check
165  * if they can be agglomerated or if we have already finished. ---*/
166 
167  if (QueueCV.size() == 1) {
168  for (auto iPoint : QueueCV[0])
169  if ((iPoint != QueueType::ErasedValue) && RightCV[iPoint])
170  return false;
171  }
172  else {
173  for (size_t iQ = 1; iQ < QueueCV.size(); ++iQ)
174  if (!QueueCV[iQ].empty()) return false;
175  }
176  return true;
177 }
