31 /*!
32  * \brief Generic limiter computation for methods based on one limiter
33  * value per point (as opposed to one per edge) and per variable.
34  * \ingroup FvmAlgos
35  * \note This implementation can be used to derive most common methods
36  * by specializing the limiter functions (e.g. Venkatakrishnan)
37  * and the geometric modifications (e.g. sharp edges), this is done
38  * via specialization of "CLimiterDetails" (all its methods). Then,
39  * a call to this function with the specialized "LimiterKind" should
40  * be added to the body of "computeLimiters()".
41  * See also the notes in computeGradientsGreenGauss.hpp
42  *
43  * Arguments:
44  * \param[in] solver - Optional, solver associated with the field (used only for MPI).
45  * \param[in] kindMpiComm - Type of MPI communication required.
46  * \param[in] kindPeriodicComm1 - Type of periodic comm. to determine min/max.
47  * \param[in] kindPeriodicComm2 - Type of periodic comm. to adjust limiters.
48  * \param[in] geometry - Geometric grid properties.
49  * \param[in] config - Configuration of the problem.
50  * \param[in] varBegin - First variable index for which to compute limiters.
51  * \param[in] varEnd - End of computation range (nVar = end-begin).
52  * \param[in] field - Variable field.
53  * \param[in] gradient - Gradient of the field.
54  * \param[out] fieldMin - Minimum field values over direct neighbors of each point.
55  * \param[out] fieldMax - As above but maximum values.
56  * \param[out] limiter - Reconstruction limiter for the field.
57  *
58  * Template parameters:
59  * \param nDim - Number of dimensions.
60  * \param LimiterKind - Used to instantiate the right details class.
61  * \param FieldType - Generic object with operator (iPoint,iVar).
62  * \param GradientType - Generic object with operator (iPoint,iVar,iDim).
63  */
64 template<size_t nDim, LIMITER LimiterKind, class FieldType, class GradientType>
65 void computeLimiters_impl(CSolver* solver,
66  MPI_QUANTITIES kindMpiComm,
67  PERIODIC_QUANTITIES kindPeriodicComm1,
68  PERIODIC_QUANTITIES kindPeriodicComm2,
69  CGeometry& geometry,
70  const CConfig& config,
71  size_t varBegin,
72  size_t varEnd,
73  const FieldType& field,
74  const GradientType& gradient,
75  FieldType& fieldMin,
76  FieldType& fieldMax,
77  FieldType& limiter)
78 {
79  constexpr size_t MAXNVAR = 32;
80 
81  if (varEnd > MAXNVAR)
82  SU2_MPI::Error("Number of variables is too large, increase MAXNVAR.", CURRENT_FUNCTION);
83 
84  const size_t nPointDomain = geometry.GetnPointDomain();
85  const size_t nPoint = geometry.GetnPoint();
86 
87  /*--- If we do not have periodicity we can use a
88  * more efficient access pattern to memory. ---*/
89 
90  const bool periodic = (solver != nullptr) &&
91  (kindPeriodicComm1 != PERIODIC_NONE) &&
92  (config.GetnMarker_Periodic() > 0);
93 
94 #ifdef HAVE_OMP
95  constexpr size_t OMP_MAX_CHUNK = 512;
96 
97  const auto chunkSize = computeStaticChunkSize(nPointDomain, omp_get_max_threads(), OMP_MAX_CHUNK);
98 #endif
99 
100  /*--- If limiters are frozen do not record the computation ---*/
101  bool wasActive = false;
102  if (config.GetDiscrete_Adjoint() && config.GetFrozen_Limiter_Disc()) {
103  wasActive = AD::BeginPassive();
104  }
105 
106  CLimiterDetails<LimiterKind> limiterDetails;
107 
108  limiterDetails.preprocess(geometry, config, varBegin, varEnd, field);
109 
110  /*--- Initialize all min/max field values if we have
111  * periodic comms. otherwise do it inside main loop. ---*/
112 
113  if (periodic)
114  {
115  SU2_OMP_FOR_STAT(chunkSize)
116  for (size_t iPoint = 0; iPoint < nPoint; ++iPoint)
117  for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
118  fieldMax(iPoint,iVar) = fieldMin(iPoint,iVar) = field(iPoint,iVar);
119  END_SU2_OMP_FOR
120 
121  for (size_t iPeriodic = 1; iPeriodic <= config.GetnMarker_Periodic()/2; ++iPeriodic)
122  {
123  solver->InitiatePeriodicComms(&geometry, &config, iPeriodic, kindPeriodicComm1);
124  solver->CompletePeriodicComms(&geometry, &config, iPeriodic, kindPeriodicComm1);
125  }
126  }
127 
128  /*--- Compute limiter for each point. ---*/
129 
130  SU2_OMP_FOR_DYN(chunkSize)
131  for (size_t iPoint = 0; iPoint < nPointDomain; ++iPoint)
132  {
133  auto nodes = geometry.nodes;
134  const auto coord_i = nodes->GetCoord(iPoint);
135 
136  /*--- Cannot preaccumulate if hybrid parallel due to shared reading. ---*/
137  if (omp_get_num_threads() == 1) AD::StartPreacc();
138  AD::SetPreaccIn(coord_i, nDim);
139 
140  for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
141  {
142  AD::SetPreaccIn(field(iPoint,iVar));
143 
144  if (periodic) {
145  /*--- Started outside loop, so counts as input. ---*/
146  AD::SetPreaccIn(fieldMax(iPoint,iVar));
147  AD::SetPreaccIn(fieldMin(iPoint,iVar));
148  }
149  else {
150  /*--- Initialize min/max now for iPoint if not periodic. ---*/
151  fieldMax(iPoint,iVar) = field(iPoint,iVar);
152  fieldMin(iPoint,iVar) = field(iPoint,iVar);
153  }
154 
155  for(size_t iDim = 0; iDim < nDim; ++iDim)
156  AD::SetPreaccIn(gradient(iPoint,iVar,iDim));
157  }
158 
159  /*--- Initialize min/max projection out of iPoint. ---*/
160 
161  su2double projMax[MAXNVAR], projMin[MAXNVAR];
162 
163  for (size_t iVar = varBegin; iVar < varEnd; ++iVar)
164  projMax[iVar] = projMin[iVar] = 0.0;
165 
166  /*--- Compute max/min projection and values over direct neighbors. ---*/
167 
168  for (auto jPoint : geometry.nodes->GetPoints(iPoint)) {
169 
170  const auto coord_j = geometry.nodes->GetCoord(jPoint);
171  AD::SetPreaccIn(coord_j, nDim);
172 
173  /*--- Distance vector from iPoint to face (middle of the edge). ---*/
174 
175  su2double dist_ij[nDim] = {0.0};
176 
177  for(size_t iDim = 0; iDim < nDim; ++iDim)
178  dist_ij[iDim] = 0.5 * (coord_j[iDim] - coord_i[iDim]);
179 
180  /*--- Project each variable, update min/max. ---*/
181 
182  for(size_t iVar = varBegin; iVar < varEnd; ++iVar)
183  {
184  su2double proj = 0.0;
185 
186  for(size_t iDim = 0; iDim < nDim; ++iDim)
187  proj += dist_ij[iDim] * gradient(iPoint,iVar,iDim);
188 
189  projMax[iVar] = max(projMax[iVar], proj);
190  projMin[iVar] = min(projMin[iVar], proj);
191 
192  AD::SetPreaccIn(field(jPoint,iVar));
193 
194  fieldMax(iPoint,iVar) = max(fieldMax(iPoint,iVar), field(jPoint,iVar));
195  fieldMin(iPoint,iVar) = min(fieldMin(iPoint,iVar), field(jPoint,iVar));
196  }
197  }
198 
199  /*--- Compute the geometric factor. ---*/
200 
201  su2double geoFactor = limiterDetails.geometricFactor(iPoint, geometry);
202 
203  /*--- Final limiter computation for each variable, get the min limiter
204  * out of the positive/negative projections and deltas. ---*/
205 
206  for(size_t iVar = varBegin; iVar < varEnd; ++iVar)
207  {
208  su2double limMax = limiterDetails.limiterFunction(iVar, projMax[iVar],
209  fieldMax(iPoint,iVar) - field(iPoint,iVar));
210 
211  su2double limMin = limiterDetails.limiterFunction(iVar, projMin[iVar],
212  fieldMin(iPoint,iVar) - field(iPoint,iVar));
213 
214  limiter(iPoint,iVar) = geoFactor * min(limMax, limMin);
215 
216  AD::SetPreaccOut(limiter(iPoint,iVar));
217  }
218 
219  AD::EndPreacc();
220  }
221  END_SU2_OMP_FOR
222 
223  /*--- Account for periodic effects, take the minimum limiter on each periodic pair. ---*/
224  if (periodic)
225  {
226  for (size_t iPeriodic = 1; iPeriodic <= config.GetnMarker_Periodic()/2; ++iPeriodic)
227  {
228  solver->InitiatePeriodicComms(&geometry, &config, iPeriodic, kindPeriodicComm2);
229  solver->CompletePeriodicComms(&geometry, &config, iPeriodic, kindPeriodicComm2);
230  }
231  }
232 
233  /*--- Obtain the limiters at halo points from the MPI ranks that own them.
234  * If no solver was provided we do not communicate. ---*/
235  if (solver != nullptr)
236  {
237  solver->InitiateComms(&geometry, &config, kindMpiComm);
238  solver->CompleteComms(&geometry, &config, kindMpiComm);
239  }
240 
241  AD::EndPassive(wasActive);
242 
243 }
