85 template<class ScalarType>
86 void CSysMatrix<ScalarType>::Initialize(unsigned long npoint, unsigned long npointdomain,
87  unsigned short nvar, unsigned short neqn,
88  bool EdgeConnect, CGeometry *geometry,
89  const CConfig *config, bool needTranspPtr, bool grad_mode) {
90 
91  assert(omp_get_thread_num()==0 && "Only the master thread is allowed to initialize the matrix.");
92 
93  if(npoint == 0) return;
94 
95  if(matrix != nullptr) {
96  SU2_MPI::Error("CSysMatrix can only be initialized once.", CURRENT_FUNCTION);
97  }
98 
99  if(nvar > MAXNVAR) {
100  SU2_MPI::Error("nVar larger than expected, increase MAXNVAR.", CURRENT_FUNCTION);
101  }
102 
103  /*--- Application of this matrix, FVM or FEM. ---*/
104  const auto type = EdgeConnect? ConnectivityType::FiniteVolume : ConnectivityType::FiniteElement;
105 
106  /*--- Type of preconditioner the matrix will be asked to build. ---*/
107  auto prec = config->GetKind_Linear_Solver_Prec();
108 
109  if ((!EdgeConnect && !config->GetStructuralProblem()) ||
110  (config->GetKind_SU2() == SU2_COMPONENT::SU2_DEF) || (config->GetKind_SU2() == SU2_COMPONENT::SU2_DOT)) {
111  /*--- FEM-type connectivity in non-structural context implies mesh deformation. ---*/
112  prec = config->GetKind_Deform_Linear_Solver_Prec();
113  }
114  else if (config->GetDiscrete_Adjoint() && (prec!=ILU)) {
115  /*--- Else "upgrade" primal solver settings. ---*/
116  prec = config->GetKind_DiscAdj_Linear_Prec();
117  }
118 
119  /*--- No else if, but separat if case! ---*/
120  if (config->GetSmoothGradient() && grad_mode) {
121  prec = config->GetKind_Grad_Linear_Solver_Prec();
122  }
123 
124  const bool ilu_needed = (prec==ILU);
125  const bool diag_needed = ilu_needed || (prec==JACOBI) || (prec==LINELET);
126 
127  /*--- Basic dimensions. ---*/
128  nVar = nvar;
129  nEqn = neqn;
130  nPoint = npoint;
131  nPointDomain = npointdomain;
132 
133  /*--- Get sparse structure pointers from geometry,
134  * the data is managed by CGeometry to allow re-use. ---*/
135 
136  const auto& csr = geometry->GetSparsePattern(type,0);
137 
138  nnz = csr.getNumNonZeros();
139  row_ptr = csr.outerPtr();
140  col_ind = csr.innerIdx();
141  dia_ptr = csr.diagPtr();
142 
143  if (needTranspPtr)
144  col_ptr = geometry->GetTransposeSparsePatternMap(type).data();
145 
146  if (type == ConnectivityType::FiniteVolume) {
147  edge_ptr.ptr = geometry->GetEdgeToSparsePatternMap().data();
148  edge_ptr.nEdge = geometry->GetnEdge();
149  }
150 
151  /*--- Get ILU sparse pattern, if fill is 0 no new data is allocated. --*/
152 
153  if(ilu_needed)
154  {
155  ilu_fill_in = config->GetLinear_Solver_ILU_n();
156 
157  const auto& csr_ilu = geometry->GetSparsePattern(type, ilu_fill_in);
158 
159  row_ptr_ilu = csr_ilu.outerPtr();
160  col_ind_ilu = csr_ilu.innerIdx();
161  dia_ptr_ilu = csr_ilu.diagPtr();
162  nnz_ilu = csr_ilu.getNumNonZeros();
163  }
164 
165  /*--- Allocate data. ---*/
166  auto allocAndInit = [](ScalarType*& ptr, unsigned long num) {
167  ptr = MemoryAllocation::aligned_alloc<ScalarType,true>(64, num*sizeof(ScalarType));
168  };
169 
170  allocAndInit(matrix, nnz*nVar*nEqn);
171 
172  /*--- Preconditioners. ---*/
173 
174  if (ilu_needed) allocAndInit(ILU_matrix, nnz_ilu*nVar*nEqn);
175 
176  if (diag_needed) allocAndInit(invM, nPointDomain*nVar*nEqn);
177 
178  /*--- Thread parallel initialization. ---*/
179 
180  int num_threads = omp_get_max_threads();
181 
182  /*--- Set suitable chunk sizes for light static for loops, and heavy
183  dynamic ones, such that threads are approximately evenly loaded. ---*/
184  omp_light_size = computeStaticChunkSize(nnz*nVar*nEqn, num_threads, OMP_MAX_SIZE_L);
185  omp_heavy_size = computeStaticChunkSize(nPointDomain, num_threads, OMP_MAX_SIZE_H);
186 
187  omp_num_parts = config->GetLinear_Solver_Prec_Threads();
188  if (omp_num_parts == 0) omp_num_parts = num_threads;
189 
190  /*--- This is akin to the row_ptr. ---*/
191  omp_partitions = new unsigned long [omp_num_parts+1];
192  for (unsigned long i = 0; i <= omp_num_parts; ++i) omp_partitions[i] = nPointDomain;
193 
194  /*--- Work estimate based on non-zeros to produce balanced partitions. ---*/
195 
196  const auto row_ptr_prec = ilu_needed? row_ptr_ilu : row_ptr;
197  const auto nnz_prec = row_ptr_prec[nPointDomain];
198 
199  const auto nnz_per_part = roundUpDiv(nnz_prec, omp_num_parts);
200 
201  for (auto iPoint = 0ul, part = 0ul; iPoint < nPointDomain; ++iPoint) {
202  if (row_ptr_prec[iPoint] >= part*nnz_per_part)
203  omp_partitions[part++] = iPoint;
204  }
205 
206  for (unsigned long thread = 0; thread < omp_num_parts; ++thread) {
207  const auto begin = omp_partitions[thread];
208  const auto end = omp_partitions[thread + 1];
209  if (begin == end) {
210  cout << "WARNING: Redundant thread has been detected. Performance could be impacted due to low number of nodes per thread." << endl;
211  break;
212  }
213  }
214 
215  /*--- Generate MKL Kernels ---*/
216 
217 #ifdef USE_MKL
218  using mkl = mkl_jit_wrapper<ScalarType>;
219  mkl::create_gemm(&MatrixMatrixProductJitter, MKL_ROW_MAJOR, MKL_NOTRANS,
220  MKL_NOTRANS, nVar, nVar, nVar, 1.0, nVar, nVar, 0.0, nVar);
221  MatrixMatrixProductKernel = mkl::get_gemm(MatrixMatrixProductJitter);
222 
223  mkl::create_gemm(&MatrixVectorProductJitterBetaZero, MKL_COL_MAJOR,
224  MKL_NOTRANS, MKL_NOTRANS, 1, nVar, nEqn, 1.0, 1, nEqn, 0.0, 1);
225  MatrixVectorProductKernelBetaZero = mkl::get_gemm(MatrixVectorProductJitterBetaZero);
226 
227  mkl::create_gemm(&MatrixVectorProductJitterBetaOne, MKL_COL_MAJOR,
228  MKL_NOTRANS, MKL_NOTRANS, 1, nVar, nEqn, 1.0, 1, nEqn, 1.0, 1);
229  MatrixVectorProductKernelBetaOne = mkl::get_gemm(MatrixVectorProductJitterBetaOne);
230 
231  mkl::create_gemm(&MatrixVectorProductJitterAlphaMinusOne, MKL_COL_MAJOR,
232  MKL_NOTRANS, MKL_NOTRANS, 1, nVar, nEqn, -1.0, 1, nEqn, 1.0, 1);
233  MatrixVectorProductKernelAlphaMinusOne = mkl::get_gemm(MatrixVectorProductJitterAlphaMinusOne);
234 #endif
235 
236 }

238 template<class T>
239 void CSysMatrixComms::Initiate(const CSysVector<T>& x, CGeometry *geometry,
240  const CConfig *config, unsigned short commType) {
241 
242  if (geometry->nP2PSend == 0) return;
243 
244  /*--- Local variables ---*/
245 
246  const unsigned short COUNT_PER_POINT = x.GetNVar();
247  const unsigned short MPI_TYPE = COMM_TYPE_DOUBLE;
248 
249  /*--- Create a boolean for reversing the order of comms. ---*/
250 
251  const bool reverse = (commType == SOLUTION_MATRIXTRANS);
252 
253  /*--- Set the size of the data packet and type depending on quantity. ---*/
254 
255  switch (commType) {
256  case SOLUTION_MATRIX:
257  case SOLUTION_MATRIXTRANS:
258  break;
259  default:
260  SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.", CURRENT_FUNCTION);
261  break;
262  }
263 
264  /*--- Check to make sure we have created a large enough buffer
265  for these comms during preprocessing. This is only for the su2double
266  buffer. It will be reallocated whenever we find a larger count
267  per point. After the first cycle of comms, this should be inactive. ---*/
268 
269  geometry->AllocateP2PComms(COUNT_PER_POINT);
270 
271  /*--- Load the specified quantity from the solver into the generic
272  communication buffer in the geometry class. ---*/
273 
274  /*--- Post all non-blocking recvs first before sends. ---*/
275 
276  geometry->PostP2PRecvs(geometry, config, MPI_TYPE, COUNT_PER_POINT, reverse);
277 
278  for (auto iMessage = 0; iMessage < geometry->nP2PSend; iMessage++) {
279 
280  switch (commType) {
281 
282  case SOLUTION_MATRIX: {
283 
284  su2double* bufDSend = geometry->bufD_P2PSend;
285 
286  /*--- Get the offset for the start of this message. ---*/
287 
288  const auto msg_offset = geometry->nPoint_P2PSend[iMessage];
289 
290  /*--- Total count can include multiple pieces of data per point. ---*/
291 
292  const auto nSend = (geometry->nPoint_P2PSend[iMessage+1] - geometry->nPoint_P2PSend[iMessage]);
293 
294  SU2_OMP_FOR_STAT(CSysMatrix<T>::OMP_MIN_SIZE)
295  for (auto iSend = 0; iSend < nSend; iSend++) {
296 
297  /*--- Get the local index for this communicated data. ---*/
298 
299  const auto iPoint = geometry->Local_Point_P2PSend[msg_offset + iSend];
300 
301  /*--- Compute the offset in the recv buffer for this point. ---*/
302 
303  const auto buf_offset = (msg_offset + iSend)*COUNT_PER_POINT;
304 
305  /*--- Load the buffer with the data to be sent. ---*/
306 
307  for (auto iVar = 0ul; iVar < x.GetNVar(); iVar++)
308  bufDSend[buf_offset+iVar] = x(iPoint,iVar);
309  }
310  END_SU2_OMP_FOR
311  break;
312  }
313 
314  case SOLUTION_MATRIXTRANS: {
315 
316  /*--- We are going to communicate in reverse, so we use the
317  recv buffer for the send instead. Also, all of the offsets
318  and counts are derived from the recv data structures. ---*/
319 
320  su2double* bufDSend = geometry->bufD_P2PRecv;
321 
322  /*--- Get the offset for the start of this message. ---*/
323 
324  const auto msg_offset = geometry->nPoint_P2PRecv[iMessage];
325 
326  /*--- Total count can include multiple pieces of data per point. ---*/
327 
328  const auto nSend = (geometry->nPoint_P2PRecv[iMessage+1] - geometry->nPoint_P2PRecv[iMessage]);
329 
330  SU2_OMP_FOR_STAT(CSysMatrix<T>::OMP_MIN_SIZE)
331  for (auto iSend = 0; iSend < nSend; iSend++) {
332 
333  /*--- Get the local index for this communicated data. Here we
334  again use the recv structure to find the send point, since
335  the usual recv points are now the senders in reverse mode. ---*/
336 
337  const auto iPoint = geometry->Local_Point_P2PRecv[msg_offset + iSend];
338 
339  /*--- Compute the offset in the recv buffer for this point. ---*/
340 
341  const auto buf_offset = (msg_offset + iSend)*COUNT_PER_POINT;
342 
343  /*--- Load the buffer with the data to be sent. ---*/
344 
345  for (auto iVar = 0ul; iVar < x.GetNVar(); iVar++)
346  bufDSend[buf_offset+iVar] = x(iPoint,iVar);
347  }
348  END_SU2_OMP_FOR
349  break;
350  }
351 
352  default:
353  SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.",
354  CURRENT_FUNCTION);
355  break;
356 
357  }
358 
359  /*--- Launch the point-to-point MPI send for this message. ---*/
360 
361  geometry->PostP2PSends(geometry, config, MPI_TYPE, COUNT_PER_POINT, iMessage, reverse);
362 
363  }
364

367 template<class T>
368 void CSysMatrixComms::Complete(CSysVector<T>& x, CGeometry *geometry,
369  const CConfig *config, unsigned short commType) {
370 
371  if (geometry->nP2PRecv == 0) return;
372 
373  /*--- Local variables ---*/
374 
375  const unsigned short COUNT_PER_POINT = x.GetNVar();
376 
377  /*--- Global status so all threads can see the result of Waitany. ---*/
378  static SU2_MPI::Status status;
379  int ind;
380 
381  /*--- Store the data that was communicated into the appropriate
382  location within the local class data structures. ---*/
383 
384  for (auto iMessage = 0; iMessage < geometry->nP2PRecv; iMessage++) {
385 
386  /*--- For efficiency, recv the messages dynamically based on
387  the order they arrive. ---*/
388 
389  SU2_OMP_SAFE_GLOBAL_ACCESS(SU2_MPI::Waitany(geometry->nP2PRecv, geometry->req_P2PRecv, &ind, &status);)
390 
391  /*--- Once we have recv'd a message, get the source rank. ---*/
392 
393  const auto source = status.MPI_SOURCE;
394 
395  switch (commType) {
396  case SOLUTION_MATRIX: {
397 
398  const su2double *bufDRecv = geometry->bufD_P2PRecv;
399 
400  /*--- We know the offsets based on the source rank. ---*/
401 
402  const auto jRecv = geometry->P2PRecv2Neighbor[source];
403 
404  /*--- Get the offset for the start of this message. ---*/
405 
406  const auto msg_offset = geometry->nPoint_P2PRecv[jRecv];
407 
408  /*--- Get the number of packets to be received in this message. ---*/
409 
410  const auto nRecv = (geometry->nPoint_P2PRecv[jRecv+1] - geometry->nPoint_P2PRecv[jRecv]);
411 
412  SU2_OMP_FOR_STAT(CSysMatrix<T>::OMP_MIN_SIZE)
413  for (auto iRecv = 0; iRecv < nRecv; iRecv++) {
414 
415  /*--- Get the local index for this communicated data. ---*/
416 
417  const auto iPoint = geometry->Local_Point_P2PRecv[msg_offset + iRecv];
418 
419  /*--- Compute the offset in the recv buffer for this point. ---*/
420 
421  const auto buf_offset = (msg_offset + iRecv)*COUNT_PER_POINT;
422 
423  /*--- Store the data correctly depending on the quantity. ---*/
424 
425  for (auto iVar = 0ul; iVar < x.GetNVar(); iVar++)
426  x(iPoint,iVar) = CSysMatrix<T>::template ActiveAssign<T>(bufDRecv[buf_offset+iVar]);
427  }
428  END_SU2_OMP_FOR
429  break;
430  }
431 
432  case SOLUTION_MATRIXTRANS: {
433 
434  /*--- We are going to communicate in reverse, so we use the
435  send buffer for the recv instead. Also, all of the offsets
436  and counts are derived from the send data structures. ---*/
437 
438  const su2double* bufDRecv = geometry->bufD_P2PSend;
439 
440  /*--- We know the offsets based on the source rank. ---*/
441 
442  const auto jRecv = geometry->P2PSend2Neighbor[source];
443 
444  /*--- Get the offset for the start of this message. ---*/
445 
446  const auto msg_offset = geometry->nPoint_P2PSend[jRecv];
447 
448  /*--- Get the number of packets to be received in this message. ---*/
449 
450  const auto nRecv = (geometry->nPoint_P2PSend[jRecv+1] - geometry->nPoint_P2PSend[jRecv]);
451 
452  SU2_OMP_FOR_STAT(CSysMatrix<T>::OMP_MIN_SIZE)
453  for (auto iRecv = 0; iRecv < nRecv; iRecv++) {
454 
455  /*--- Get the local index for this communicated data. ---*/
456 
457  const auto iPoint = geometry->Local_Point_P2PSend[msg_offset + iRecv];
458 
459  /*--- Compute the offset in the recv buffer for this point. ---*/
460 
461  const auto buf_offset = (msg_offset + iRecv)*COUNT_PER_POINT;
462 
463  /*--- Update receiving point. ---*/
464 
465  for (auto iVar = 0ul; iVar < x.GetNVar(); iVar++)
466  x(iPoint,iVar) += CSysMatrix<T>::template ActiveAssign<T>(bufDRecv[buf_offset+iVar]);
467  }
468  END_SU2_OMP_FOR
469  break;
470  }
471 
472  default:
473  SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.", CURRENT_FUNCTION);
474  break;
475  }
476  }
477 
478  /*--- Verify that all non-blocking point-to-point sends have finished.
479  Note that this should be satisfied, as we have received all of the
480  data in the loop above at this point. ---*/
481 
482 #ifdef HAVE_MPI
483  SU2_OMP_SAFE_GLOBAL_ACCESS(SU2_MPI::Waitall(geometry->nP2PSend, geometry->req_P2PSend, MPI_STATUS_IGNORE);)
484 #endif
485 }

506 template<class ScalarType>
507 void CSysMatrix<ScalarType>::Gauss_Elimination(ScalarType* matrix, ScalarType* vec) const {
508 
509 #ifdef USE_MKL_LAPACK
510  // With MKL_DIRECT_CALL enabled, this is significantly faster than native code on Intel Architectures.
511  lapack_int ipiv[MAXNVAR];
512  LAPACKE_dgetrf( LAPACK_ROW_MAJOR, nVar, nVar, matrix, nVar, ipiv);
513  LAPACKE_dgetrs( LAPACK_ROW_MAJOR, 'N', nVar, 1, matrix, nVar, ipiv, vec, 1 );
514 #else
515 #define A(I,J) matrix[(I)*nVar+(J)]
516 
517  /*--- Transform system in Upper Matrix ---*/
518  for (auto iVar = 1ul; iVar < nVar; iVar++) {
519  for (auto jVar = 0ul; jVar < iVar; jVar++) {
520  ScalarType weight = A(iVar,jVar) / A(jVar,jVar);
521  for (auto kVar = jVar; kVar < nVar; kVar++)
522  A(iVar,kVar) -= weight * A(jVar,kVar);
523  vec[iVar] -= weight * vec[jVar];
524  }
525  }
526 
527  /*--- Backwards substitution ---*/
528  for (auto iVar = nVar; iVar > 0ul;) {
529  iVar--; // unsigned type
530  for (auto jVar = iVar+1; jVar < nVar; jVar++)
531  vec[iVar] -= A(iVar,jVar) * vec[jVar];
532  vec[iVar] /= A(iVar,iVar);
533  }
534 #undef A
535 #endif
536 }

538 template<class ScalarType>
539 void CSysMatrix<ScalarType>::MatrixInverse(ScalarType *matrix, ScalarType *inverse) const {
540 
541  /*--- This is a generalization of Gaussian elimination for multiple rhs' (the basis vectors).
542  We could call "Gauss_Elimination" multiple times or fully generalize it for multiple rhs,
543  the performance of both routines would suffer in both cases without the use of exotic templating.
544  And so it feels reasonable to have some duplication here. ---*/
545 
546  assert((matrix != inverse) && "Output cannot be the same as the input.");
547 
548 #define M(I,J) inverse[(I)*nVar+(J)]
549 
550  /*--- Initialize the inverse with the identity. ---*/
551  for (auto iVar = 0ul; iVar < nVar; iVar++)
552  for (auto jVar = 0ul; jVar < nVar; jVar++)
553  M(iVar,jVar) = ScalarType(iVar==jVar);
554 
555  /*--- Inversion ---*/
556 #ifdef USE_MKL_LAPACK
557  // With MKL_DIRECT_CALL enabled, this is significantly faster than native code on Intel Architectures.
558  lapack_int ipiv[MAXNVAR];
559  LAPACKE_dgetrf( LAPACK_ROW_MAJOR, nVar, nVar, matrix, nVar, ipiv );
560  LAPACKE_dgetrs( LAPACK_ROW_MAJOR, 'N', nVar, nVar, matrix, nVar, ipiv, inverse, nVar );
561 #else
562 #define A(I,J) matrix[(I)*nVar+(J)]
563 
564  /*--- Transform system in Upper Matrix ---*/
565  for (auto iVar = 1ul; iVar < nVar; iVar++) {
566  for (auto jVar = 0ul; jVar < iVar; jVar++)
567  {
568  ScalarType weight = A(iVar,jVar) / A(jVar,jVar);
569 
570  for (auto kVar = jVar; kVar < nVar; kVar++)
571  A(iVar,kVar) -= weight * A(jVar,kVar);
572 
573  /*--- at this stage M is lower triangular so not all cols need updating ---*/
574  for (auto kVar = 0ul; kVar <= jVar; kVar++)
575  M(iVar,kVar) -= weight * M(jVar,kVar);
576  }
577  }
578 
579  /*--- Backwards substitution ---*/
580  for (auto iVar = nVar; iVar > 0ul;) {
581  iVar--; // unsigned type
582  for (auto jVar = iVar+1; jVar < nVar; jVar++)
583  for (auto kVar = 0ul; kVar < nVar; kVar++)
584  M(iVar,kVar) -= A(iVar,jVar) * M(jVar,kVar);
585 
586  for (auto kVar = 0ul; kVar < nVar; kVar++)
587  M(iVar,kVar) /= A(iVar,iVar);
588  }
589 #undef A
590 #endif
591 #undef M
592 }

594 template<class ScalarType>
595 void CSysMatrix<ScalarType>::DeleteValsRowi(unsigned long i) {
596 
597  const auto block_i = i/nVar;
598  const auto row = i%nVar;

669 template<class ScalarType>
670 void CSysMatrix<ScalarType>::BuildILUPreconditioner() {
671 
672  /*--- Copy block matrix to compute factorization in-place. ---*/
673 
674  if (ilu_fill_in == 0) {
675  /*--- ILU0, direct copy. ---*/
676  SU2_OMP_FOR_STAT(omp_light_size)
677  for (auto iVar = 0ul; iVar < nnz*nVar*nVar; ++iVar)
678  ILU_matrix[iVar] = matrix[iVar];
679  END_SU2_OMP_FOR
680  }
681  else {
682  /*--- ILUn clear the ILU matrix first. ---*/
683  SU2_OMP_FOR_STAT(omp_light_size)
684  for (auto iVar = 0ul; iVar < nnz_ilu*nVar*nVar; iVar++)
685  ILU_matrix[iVar] = 0.0;
686  END_SU2_OMP_FOR
687 
688  /*--- ILUn, traverse matrix to access its blocks
689  * sequentially and set them in the ILU matrix. ---*/
690  SU2_OMP_FOR_DYN(omp_heavy_size)
691  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
692  for (auto index = row_ptr[iPoint]; index < row_ptr[iPoint+1]; index++) {
693  auto jPoint = col_ind[index];
694  SetBlock_ILUMatrix(iPoint, jPoint, &matrix[index*nVar*nVar]);
695  }
696  }
697  END_SU2_OMP_FOR
698  }
699 
700  /*--- Transform system in Upper Matrix ---*/
701 
702  /*--- OpenMP Parallelization, a loop construct is used to ensure
703  * the preconditioner is computed correctly even if called
704  * outside of a parallel section. ---*/
705 
706  SU2_OMP_FOR_STAT(1)
707  for(unsigned long thread = 0; thread < omp_num_parts; ++thread)
708  {
709  const auto begin = omp_partitions[thread];
710  const auto end = omp_partitions[thread+1];
711  if (begin == end) continue;
712 
713  /*--- Each thread will work on the submatrix defined from row/col "begin"
714  * to row/col "end-1" (i.e. the range [begin,end[). Which is exactly
715  * what the MPI-only implementation does. ---*/
716 
717  ScalarType weight[MAXNVAR*MAXNVAR], aux_block[MAXNVAR*MAXNVAR];
718 
719  for (auto iPoint = begin+1; iPoint < end; iPoint++) {
720 
721  /*--- Invert and store the previous diagonal block to later compute the weight. ---*/
722 
723  InverseDiagonalBlock_ILUMatrix(iPoint-1, &invM[(iPoint-1)*nVar*nVar]);
724 
725  /*--- For this row (unknown), loop over its lower diagonal entries. ---*/
726 
727  for (auto index = row_ptr_ilu[iPoint]; index < dia_ptr_ilu[iPoint]; index++) {
728 
729  /*--- jPoint is the column index (jPoint < iPoint). ---*/
730 
731  auto jPoint = col_ind_ilu[index];
732 
733  /*--- We only care about the sub matrix within "begin" and "end-1". ---*/
734 
735  if (jPoint < begin) continue;
736 
737  /*--- Multiply the block by the inverse of the corresponding diagonal block. ---*/
738 
739  auto Block_ij = &ILU_matrix[index*nVar*nVar];
740  MatrixMatrixProduct(Block_ij, &invM[jPoint*nVar*nVar], weight);
741 
742  /*--- "weight" holds Aij*inv(Ajj). Jump to the upper part of the jPoint row. ---*/
743 
744  for (auto index_ = dia_ptr_ilu[jPoint]+1; index_ < row_ptr_ilu[jPoint+1]; index_++) {
745 
746  /*--- Get the column index (kPoint > jPoint). ---*/
747 
748  auto kPoint = col_ind_ilu[index_];
749 
750  if (kPoint >= end) break;
751 
752  /*--- If Aik exists, update it: Aik -= Aij*inv(Ajj)*Ajk ---*/
753 
754  auto Block_ik = GetBlock_ILUMatrix(iPoint, kPoint);
755 
756  if (Block_ik != nullptr) {
757  auto Block_jk = &ILU_matrix[index_*nVar*nVar];
758  MatrixMatrixProduct(weight, Block_jk, aux_block);
759  MatrixSubtraction(Block_ik, aux_block, Block_ik);
760  }
761  }
762 
763  /*--- Lastly, store "weight" in the lower triangular part, which
764  will be reused during the forward solve in the precon/smoother. ---*/
765 
766  for (auto iVar = 0ul; iVar < nVar*nVar; ++iVar)
767  Block_ij[iVar] = weight[iVar];
768  }
769  }
770  InverseDiagonalBlock_ILUMatrix(end-1, &invM[(end-1)*nVar*nVar]);
771 
772  }
773  END_SU2_OMP_FOR
774 
775 }

778 void CSysMatrix<ScalarType>::ComputeILUPreconditioner(const CSysVector<ScalarType> & vec, CSysVector<ScalarType> & prod,
779  CGeometry *geometry, const CConfig *config) const {
780  /*--- Coherent view of vectors. ---*/
781  SU2_OMP_BARRIER
782 
783  /*--- OpenMP Parallelization ---*/
784  SU2_OMP_FOR_STAT(1)
785  for(unsigned long thread = 0; thread < omp_num_parts; ++thread)
786  {
787  const auto begin = omp_partitions[thread];
788  const auto end = omp_partitions[thread+1];
789  if (begin == end) continue;
790 
791  ScalarType aux_vec[MAXNVAR];
792 
793  /*--- Copy vector to then work on prod in place ---*/
794 
795  for (auto iVar = begin*nVar; iVar < end*nVar; iVar++)
796  prod[iVar] = vec[iVar];
797 
798  /*--- Forward solve the system using the lower matrix entries that
799  were computed and stored during the ILU preprocessing. Note
800  that we are overwriting the residual vector as we go. ---*/
801 
802  for (auto iPoint = begin+1; iPoint < end; iPoint++) {
803  for (auto index = row_ptr_ilu[iPoint]; index < dia_ptr_ilu[iPoint]; index++) {
804  auto jPoint = col_ind_ilu[index];
805  if (jPoint < begin) continue;
806  auto Block_ij = &ILU_matrix[index*nVar*nVar];
807  MatrixVectorProductSub(Block_ij, &prod[jPoint*nVar], &prod[iPoint*nVar]);
808  }
809  }
810 
811  /*--- Backwards substitution (starts at the last row) ---*/
812 
813  for (auto iPoint = end; iPoint > begin;) {
814  iPoint--; // unsigned type
815  for (auto iVar = 0ul; iVar < nVar; iVar++)
816  aux_vec[iVar] = prod[iPoint*nVar+iVar];
817 
818  for (auto index = dia_ptr_ilu[iPoint]+1; index < row_ptr_ilu[iPoint+1]; index++) {
819  auto jPoint = col_ind_ilu[index];
820  if (jPoint >= end) break;
821  auto Block_ij = &ILU_matrix[index*nVar*nVar];
822  MatrixVectorProductSub(Block_ij, &prod[jPoint*nVar], aux_vec);
823  }
824 
825  MatrixVectorProduct(&invM[iPoint*nVar*nVar], aux_vec, &prod[iPoint*nVar]);
826  }
827  }
828  END_SU2_OMP_FOR
829 
830  /*--- MPI Parallelization ---*/
831 
832  CSysMatrixComms::Initiate(prod, geometry, config);
833  CSysMatrixComms::Complete(prod, geometry, config);

837 template<class ScalarType>
838 void CSysMatrix<ScalarType>::ComputeLU_SGSPreconditioner(const CSysVector<ScalarType> & vec, CSysVector<ScalarType> & prod,
839  CGeometry *geometry, const CConfig *config) const {
840 
841  /*--- First part of the symmetric iteration: (D+L).x* = b ---*/
842 
843  /*--- Coherent view of vectors. ---*/
844  SU2_OMP_BARRIER
845 
846  /*--- OpenMP Parallelization ---*/
847  SU2_OMP_FOR_STAT(1)
848  for(unsigned long thread = 0; thread < omp_num_parts; ++thread)
849  {
850  const auto begin = omp_partitions[thread];
851  const auto end = omp_partitions[thread+1];
852  if (begin == end) continue;
853 
854  /*--- Each thread will work on the submatrix defined from row/col "begin"
855  * to row/col "end-1", except the last thread that also considers halos.
856  * This is NOT exactly equivalent to the MPI implementation on the same
857  * number of domains, for that we would need to define "thread-halos". ---*/
858 
859  ScalarType low_prod[MAXNVAR];
860 
861  for (auto iPoint = begin; iPoint < end; ++iPoint) {
862  auto idx = iPoint*nVar;
863  LowerProduct(prod, iPoint, begin, low_prod); // Compute L.x*
864  VectorSubtraction(&vec[idx], low_prod, &prod[idx]); // Compute y = b - L.x*
865  Gauss_Elimination(iPoint, &prod[idx]); // Solve D.x* = y
866  }
867  }
868  END_SU2_OMP_FOR
869 
870  /*--- MPI Parallelization ---*/
871 
872  CSysMatrixComms::Initiate(prod, geometry, config);
873  CSysMatrixComms::Complete(prod, geometry, config);
874 
875  /*--- Second part of the symmetric iteration: (D+U).x_(1) = D.x* ---*/
876 
877  /*--- OpenMP Parallelization ---*/
878  SU2_OMP_FOR_STAT(1)
879  for(unsigned long thread = 0; thread < omp_num_parts; ++thread)
880  {
881  const auto begin = omp_partitions[thread];
882  const auto row_end = omp_partitions[thread+1];
883  if (begin == row_end) continue;
884 
885  ScalarType up_prod[MAXNVAR], dia_prod[MAXNVAR];
886 
887  for (auto iPoint = row_end; iPoint > begin;) {
888  iPoint--; // because of unsigned type
889  auto idx = iPoint*nVar;
890  DiagonalProduct(prod, iPoint, dia_prod); // Compute D.x*
891  UpperProduct(prod, iPoint, row_end, up_prod); // Compute U.x_(n+1)
892  VectorSubtraction(dia_prod, up_prod, &prod[idx]); // Compute y = D.x*-U.x_(n+1)
893  Gauss_Elimination(iPoint, &prod[idx]); // Solve D.x* = y
894  }
895  }
896  END_SU2_OMP_FOR
897 
898  /*--- MPI Parallelization ---*/
899 
900  CSysMatrixComms::Initiate(prod, geometry, config);
901  CSysMatrixComms::Complete(prod, geometry, config);
902 
903 }

936 void CSysMatrix<ScalarType>::ComputeLineletPreconditioner(const CSysVector<ScalarType> & vec, CSysVector<ScalarType> & prod,
937  CGeometry *geometry, const CConfig *config) const {
938  /*--- Coherent view of vectors. ---*/
939  SU2_OMP_BARRIER
940 
941  const auto& li = geometry->GetLineletInfo(config);
942 
943  /*--- Jacobi preconditioning where there are no linelets. ---*/
944 
945  SU2_OMP_FOR_(schedule(dynamic,omp_heavy_size) SU2_NOWAIT)
946  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++)
947  if (li.lineletIdx[iPoint] == CGeometry::CLineletInfo::NO_LINELET)
948  MatrixVectorProduct(&(invM[iPoint*nVar*nVar]), &vec[iPoint*nVar], &prod[iPoint*nVar]);
949  END_SU2_OMP_FOR
950 
951  /*--- Solve the tridiagonal systems for the linelets. ---*/
952 
953  SU2_OMP_FOR_DYN(1)
954  for (auto iLinelet = 0ul; iLinelet < li.linelets.size(); iLinelet++) {
955 
956  /*--- Get references to the working vectors allocated for this thread. ---*/
957 
958  const int thread = omp_get_thread_num();
959  auto& lineletUpper = LineletUpper[thread];
960  auto& lineletInvDiag = LineletInvDiag[thread];
961  auto& lineletVector = LineletVector[thread];
962 
963  /*--- Initialize the solution vector with the rhs ---*/
964 
965  const auto nElem = li.linelets[iLinelet].size();
966 
967  for (auto iElem = 0ul; iElem < nElem; iElem++) {
968  const auto iPoint = li.linelets[iLinelet][iElem];
969  for (auto iVar = 0ul; iVar < nVar; iVar++)
970  lineletVector[iElem*nVar+iVar] = vec[iPoint*nVar+iVar];
971  }
972 
973  /*--- Forward pass, eliminate lower entries, modify diagonal and rhs. ---*/
974 
975  /*--- Small temporaries. ---*/
976  ScalarType aux_block[MAXNVAR*MAXNVAR], aux_vector[MAXNVAR];
977 
978  /*--- Copy diagonal block for first point in this linelet. ---*/
979  MatrixCopy(&matrix[dia_ptr[li.linelets[iLinelet][0]]*nVar*nVar],
980  lineletInvDiag.data());
981 
982  for (auto iElem = 1ul; iElem < nElem; iElem++) {
983 
984  /*--- Setup pointers to required matrices and vectors ---*/
985  const auto im1Point = li.linelets[iLinelet][iElem-1];
986  const auto iPoint = li.linelets[iLinelet][iElem];
987 
988  const auto* d = &matrix[dia_ptr[iPoint]*nVar*nVar];
989  const auto* l = GetBlock(iPoint, im1Point);
990  const auto* u = GetBlock(im1Point, iPoint);
991 
992  auto* inv_dm1 = &lineletInvDiag[(iElem-1)*nVar*nVar];
993  auto* d_prime = &lineletInvDiag[iElem*nVar*nVar];
994  auto* b_prime = &lineletVector[iElem*nVar];
995 
996  /*--- Invert previous modified diagonal ---*/
997  MatrixCopy(inv_dm1, aux_block);
998  MatrixInverse(aux_block, inv_dm1);
999 
1000  /*--- Left-multiply by lower block to obtain the weight ---*/
1001  MatrixMatrixProduct(l, inv_dm1, aux_block);
1002 
1003  /*--- Multiply weight by upper block to modify current diagonal ---*/
1004  MatrixMatrixProduct(aux_block, u, d_prime);
1005  MatrixSubtraction(d, d_prime, d_prime);
1006 
1007  /*--- Update the rhs ---*/
1008  MatrixVectorProduct(aux_block, &lineletVector[(iElem-1)*nVar], aux_vector);
1009  VectorSubtraction(b_prime, aux_vector, b_prime);
1010 
1011  /*--- Cache upper block pointer for the backward substitution phase ---*/
1012  lineletUpper[iElem-1] = u;
1013  }
1014 
1015  /*--- Backwards substitution, LineletVector becomes the solution ---*/
1016 
1017  /*--- x_n = d_n^{-1} * b_n ---*/
1018  Gauss_Elimination(&lineletInvDiag[(nElem-1)*nVar*nVar], &lineletVector[(nElem-1)*nVar]);
1019 
1020  /*--- x_i = d_i^{-1}*(b_i - u_i*x_{i+1}) ---*/
1021  for (auto iElem = nElem-1; iElem > 0; --iElem) {
1022  const auto* inv_dm1 = &lineletInvDiag[(iElem-1)*nVar*nVar];
1023  MatrixVectorProduct(lineletUpper[iElem-1], &lineletVector[iElem*nVar], aux_vector);
1024  VectorSubtraction(&lineletVector[(iElem-1)*nVar], aux_vector, aux_vector);
1025  MatrixVectorProduct(inv_dm1, aux_vector, &lineletVector[(iElem-1)*nVar]);
1026  }
1027 
1028  /*--- Copy results to product vector ---*/
1029 
1030  for (auto iElem = 0ul; iElem < nElem; iElem++) {
1031  const auto iPoint = li.linelets[iLinelet][iElem];
1032  for (auto iVar = 0ul; iVar < nVar; iVar++)
1033  prod[iPoint*nVar+iVar] = lineletVector[iElem*nVar+iVar];
1034  }
1035 
1036  }
1037  END_SU2_OMP_FOR
1038 
1039  /*--- MPI Parallelization ---*/
1040 
1041  CSysMatrixComms::Initiate(prod, geometry, config);
1042  CSysMatrixComms::Complete(prod, geometry, config);
1043 
1044 }

1059 template<class ScalarType>
1060 template<class OtherType>
1061 void CSysMatrix<ScalarType>::EnforceSolutionAtNode(const unsigned long node_i, const OtherType *x_i, CSysVector<OtherType> & b) {
1062 
1063  /*--- Eliminate the row associated with node i (Block_ii = I and all other Block_ij = 0).
1064  * To preserve eventual symmetry, also attempt to eliminate the column, if the sparse pattern is not
1065  * symmetric the entire column may not be eliminated, the result (matrix and vector) is still correct.
1066  * The vector is updated with the product of column i by the known (enforced) solution at node i. ---*/
1067 
1068  for (auto index = row_ptr[node_i]; index < row_ptr[node_i+1]; ++index) {
1069 
1070  auto node_j = col_ind[index];
1071 
1072  /*--- The diagonal block is handled outside the loop. ---*/
1073  if (node_j == node_i) continue;
1074 
1075  /*--- Delete block j on row i (bij) and ATTEMPT to delete block i on row j (bji). ---*/
1076  auto bij = &matrix[index*nVar*nVar];
1077  auto bji = GetBlock(node_j, node_i);
1078 
1079  /*--- The "attempt" part. ---*/
1080  if (bji == nullptr) {
1081  node_j = node_i;
1082  bji = bij;
1083  }
1084 
1085  for(auto iVar = 0ul; iVar < nVar; ++iVar) {
1086  for(auto jVar = 0ul; jVar < nVar; ++jVar) {
1087  /*--- Column product. ---*/
1088  b[node_j*nVar+iVar] -= bji[iVar*nVar+jVar] * x_i[jVar];
1089  /*--- Delete blocks. ---*/
1090  bij[iVar*nVar+jVar] = bji[iVar*nVar+jVar] = 0.0;
1091  }
1092  }
1093 
1094  }
1095 
1096  /*--- Set the diagonal block to the identity. ---*/
1097  SetVal2Diag(node_i, 1.0);
1098 
1099  /*--- Set known solution in rhs vector. ---*/
1100  b.SetBlock(node_i, x_i);
1101 
1102 }

1145 
1146  SU2_OMP_FOR_DYN(omp_heavy_size)

1200 
1201  swapAndTransp(nVar, bij, bji);
1202  }
1203  }

1205  }
