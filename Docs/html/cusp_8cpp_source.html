65 CNumerics::ResidualType<> CUpwCUSP_Flow::ComputeResidual(const CConfig* config) {
66 
67  implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
68 
69  unsigned short iDim, iVar;
70  su2double Diff_U[5] = {0.0};
71 
72  AD::SetPreaccIn(Normal, nDim);
73  AD::SetPreaccIn(V_i, nDim+4);
74  AD::SetPreaccIn(V_j, nDim+4);
75 
76  /*--- Pressure, density, enthalpy, energy, and velocity at points i and j ---*/
77 
78  Pressure_i = V_i[nDim+1]; Pressure_j = V_j[nDim+1];
79  Density_i = V_i[nDim+2]; Density_j = V_j[nDim+2];
80  Enthalpy_i = V_i[nDim+3]; Enthalpy_j = V_j[nDim+3];
81  su2double Energy_i = Enthalpy_i - Pressure_i/Density_i;
82  su2double Energy_j = Enthalpy_j - Pressure_j/Density_j;
83 
84  su2double sq_vel_i = 0.0, sq_vel_j = 0.0;
85  for (iDim = 0; iDim < nDim; iDim++) {
86  Velocity_i[iDim] = V_i[iDim+1];
87  Velocity_j[iDim] = V_j[iDim+1];
88  sq_vel_i += Velocity_i[iDim]*Velocity_i[iDim];
89  sq_vel_j += Velocity_j[iDim]*Velocity_j[iDim];
90  }
91 
92  /*-- Face area and unit normal ---*/
93 
94  Area = GeometryToolbox::Norm(nDim, Normal);
95 
96  for (iDim = 0; iDim < nDim; iDim++)
97  UnitNormal[iDim] = Normal[iDim]/Area;
98 
99  /*--- Computes differences of conservative variables, with a correction for the enthalpy ---*/
100 
101  Diff_U[0] = Density_i - Density_j;
102  for (iDim = 0; iDim < nDim; iDim++)
103  Diff_U[iDim+1] = Density_i*Velocity_i[iDim] - Density_j*Velocity_j[iDim];
104  Diff_U[nVar-1] = Density_i*Enthalpy_i - Density_j*Enthalpy_j;
105 
106  /*--- Get left and right fluxes ---*/
107 
108  GetInviscidProjFlux(&Density_i, Velocity_i, &Pressure_i, &Enthalpy_i, UnitNormal, ProjFlux_i);
109  GetInviscidProjFlux(&Density_j, Velocity_j, &Pressure_j, &Enthalpy_j, UnitNormal, ProjFlux_j);
110 
111  /*--- Compute dissipation parameters based on Roe-averaged values ---*/
112 
113  su2double Beta, Nu_c;
114 
115  su2double R = sqrt(Density_j/Density_i), ProjVelocity = 0.0, sq_vel = 0.0;
116 
117  for (iDim = 0; iDim < nDim; iDim++) {
118  su2double MeanVel = (R*Velocity_j[iDim]+Velocity_i[iDim])/(R+1.0);
119  ProjVelocity += MeanVel*UnitNormal[iDim];
120  sq_vel += MeanVel*MeanVel;
121  }
122  su2double MeanEnthalpy = (R*Enthalpy_j+Enthalpy_i)/(R+1.0);
123  su2double MeanSoundSpeed = sqrt(Gamma_Minus_One*fabs(MeanEnthalpy-0.5*sq_vel));
124 
125  su2double Mach = ProjVelocity / MeanSoundSpeed;
126 
127  su2double tmp1 = 0.5*(Gamma+1.0)/Gamma*ProjVelocity;
128  su2double tmp2 = sqrt(pow(tmp1-ProjVelocity/Gamma, 2.0) + pow(MeanSoundSpeed,2.0)/Gamma);
129  su2double LamdaNeg = tmp1 - tmp2, LamdaPos = tmp1 + tmp2;
130 
131  if (fabs(Mach) >= 1.0) Beta = Mach/fabs(Mach);
132  else if (Mach >= 0.0) Beta = max(0.0, (ProjVelocity + LamdaNeg)/(ProjVelocity - LamdaNeg));
133  else Beta =-max(0.0, (ProjVelocity + LamdaPos)/(ProjVelocity - LamdaPos));
134 
135  if (fabs(Mach) >= 1.0) Nu_c = 0.0;
136  else {
137  if (Beta > 0.0) Nu_c =-(1.0+Beta)*LamdaNeg;
138  else if (Beta < 0.0) Nu_c = (1.0-Beta)*LamdaPos;
139  /*--- Limit the minimum scalar dissipation ---*/
140  else Nu_c = max(fabs(ProjVelocity), config->GetEntropyFix_Coeff()*MeanSoundSpeed);
141  }
142 
143  /*--- Compute the residual ---*/
144 
145  for (iVar = 0; iVar < nVar; iVar++)
146  Flux[iVar] = 0.5*((1.0+Beta)*ProjFlux_i[iVar] + (1.0-Beta)*ProjFlux_j[iVar] + Nu_c*Diff_U[iVar])*Area;
147 
148  /*--- Jacobian computation ---*/
149 
150  if (implicit) {
151 
152  /*--- Flux average and difference contributions ---*/
153 
154  GetInviscidProjJac(Velocity_i, &Energy_i, Normal, 0.5*(1.0+Beta), Jacobian_i);
155  GetInviscidProjJac(Velocity_j, &Energy_j, Normal, 0.5*(1.0-Beta), Jacobian_j);
156 
157  /*--- Solution difference (scalar dissipation) contribution ---*/
158 
159  su2double cte_0 = 0.5*Nu_c*Area*config->GetCent_Jac_Fix_Factor();
160 
161  /*--- n-1 diagonal entries ---*/
162 
163  for (iVar = 0; iVar < (nVar-1); iVar++) {
164  Jacobian_i[iVar][iVar] += cte_0;
165  Jacobian_j[iVar][iVar] -= cte_0;
166  }
167 
168  /*--- Last rows ---*/
169 
170  Jacobian_i[nVar-1][0] += cte_0*Gamma_Minus_One*0.5*sq_vel_i;
171  for (iDim = 0; iDim < nDim; iDim++)
172  Jacobian_i[nVar-1][iDim+1] -= cte_0*Gamma_Minus_One*Velocity_i[iDim];
173  Jacobian_i[nVar-1][nVar-1] += cte_0*Gamma;
174 
175  Jacobian_j[nVar-1][0] -= cte_0*Gamma_Minus_One*0.5*sq_vel_j;
176  for (iDim = 0; iDim < nDim; iDim++)
177  Jacobian_j[nVar-1][iDim+1] += cte_0*Gamma_Minus_One*Velocity_j[iDim];
178  Jacobian_j[nVar-1][nVar-1] -= cte_0*Gamma;
179 
180  }
181 
182  AD::SetPreaccOut(Flux, nVar);
183  AD::EndPreacc();
184 
185  return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
186
