248 void CFVMFlowSolverBase<V, R>::CommunicateInitialState(CGeometry* geometry, const CConfig* config) {
249  /*--- Define solver parameters needed for execution of destructor ---*/
250 
251  space_centered = (config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED);
252  euler_implicit = (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT);
253  least_squares = (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES);
254 
255  /*--- Communicate and store volume and the number of neighbors for
256  any dual CVs that lie on on periodic markers. ---*/
257 
258  for (unsigned short iPeriodic = 1; iPeriodic <= config->GetnMarker_Periodic() / 2; iPeriodic++) {
259  InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_VOLUME);
260  CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_VOLUME);
261  InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_NEIGHBORS);
262  CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_NEIGHBORS);
263  }
264  SetImplicitPeriodic(euler_implicit);
265  if (MGLevel == MESH_0) SetRotatePeriodic(true);
266 
267  /*--- Perform the MPI communication of the solution ---*/
268 
269  InitiateComms(geometry, config, SOLUTION);
270  CompleteComms(geometry, config, SOLUTION);
271 
272  /*--- Store the initial CFL number for all grid points. ---*/
273 
274  const auto CFL = config->GetCFL(MGLevel);
275  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
276  nodes->SetLocalCFL(iPoint, CFL);
277  }
278  Min_CFL_Local = CFL;
279  Max_CFL_Local = CFL;

283 template <class V, ENUM_REGIME R>
284 void CFVMFlowSolverBase<V, R>::HybridParallelInitialization(const CConfig& config, CGeometry& geometry) {
285 #ifdef HAVE_OMP
286  /*--- Get the edge coloring. If the expected parallel efficiency becomes too low setup the
287  * reducer strategy. Where one loop is performed over edges followed by a point loop to
288  * sum the fluxes for each cell and set the diagonal of the system matrix. ---*/
289 
290  su2double parallelEff = 1.0;
291  const auto& coloring = geometry.GetEdgeColoring(&parallelEff);
292 
293  /*--- The decision to use the strategy is local to each rank. ---*/
294  ReducerStrategy = parallelEff < COLORING_EFF_THRESH;
295 
296  /*--- When using the reducer force a single color to reduce the color loop overhead. ---*/
297  if (ReducerStrategy && (coloring.getOuterSize() > 1)) geometry.SetNaturalEdgeColoring();
298 
299  if (!coloring.empty()) {
300  /*--- If the reducer strategy is used we are not constrained by group
301  * size as we have no other edge loops in the Euler/NS solvers. ---*/
302  auto groupSize = ReducerStrategy ? 1ul : geometry.GetEdgeColorGroupSize();
303  auto nColor = coloring.getOuterSize();
304  EdgeColoring.reserve(nColor);
305 
306  for (auto iColor = 0ul; iColor < nColor; ++iColor)
307  EdgeColoring.emplace_back(coloring.innerIdx(iColor), coloring.getNumNonZeros(iColor), groupSize);
308  }
309 
310  /*--- If the reducer strategy is not being forced (by EDGE_COLORING_GROUP_SIZE=0) print some messages. ---*/
311  if (config.GetEdgeColoringGroupSize() != 1 << 30) {
312  su2double minEff = 1.0;
313  SU2_MPI::Reduce(&parallelEff, &minEff, 1, MPI_DOUBLE, MPI_MIN, MASTER_NODE, SU2_MPI::GetComm());
314 
315  int tmp = ReducerStrategy, numRanksUsingReducer = 0;
316  SU2_MPI::Reduce(&tmp, &numRanksUsingReducer, 1, MPI_INT, MPI_SUM, MASTER_NODE, SU2_MPI::GetComm());
317 
318  if (minEff < COLORING_EFF_THRESH) {
319  cout << "WARNING: On " << numRanksUsingReducer << " MPI ranks the coloring efficiency was less than "
320  << COLORING_EFF_THRESH << " (min value was " << minEff << ").\n"
321  << " Those ranks will now use a fallback strategy, better performance may be possible\n"
322  << " with a different value of config option EDGE_COLORING_GROUP_SIZE (default 512)."
323 #ifdef HAVE_OPDI
324  << "\n The memory usage of the discrete adjoint solver is higher when using the fallback."
325 #endif
326  << endl;
327  }
328  }
329 
330  if (ReducerStrategy) EdgeFluxes.Initialize(geometry.GetnEdge(), geometry.GetnEdge(), nVar, nullptr);
331 
332  omp_chunk_size = computeStaticChunkSize(nPoint, omp_get_max_threads(), OMP_MAX_SIZE);
333 #else
334  EdgeColoring[0] = DummyGridColor<>(geometry.GetnEdge());
335 #endif
336 }

412 
413  numerics->SetCoord(geometry->nodes->GetCoord(iPoint),
414  geometry->nodes->GetCoord(jPoint));
415 
416  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
417 
418  /*--- Primitive and secondary variables. ---*/
419 
420  numerics->SetPrimitive(nodes->GetPrimitive(iPoint),
421  nodes->GetPrimitive(jPoint));
422 
423  numerics->SetSecondary(nodes->GetSecondary(iPoint),
424  nodes->GetSecondary(jPoint));
425 
426  /*--- Gradients. ---*/
427 
428  numerics->SetPrimVarGradient(nodes->GetGradient_Primitive(iPoint),
429  nodes->GetGradient_Primitive(jPoint));
430 
431  /*--- Turbulent kinetic energy. ---*/
432 
433  if (tkeNeeded)
434  numerics->SetTurbKineticEnergy(turbNodes->GetSolution(iPoint,0),
435  turbNodes->GetSolution(jPoint,0));
436 
437  /*--- Wall shear stress values (wall functions) ---*/
438 
439  numerics->SetTau_Wall(nodes->GetTau_Wall(iPoint),
440  nodes->GetTau_Wall(jPoint));
441

443

445 
446  if (ReducerStrategy) {
447  EdgeFluxes.SubtractBlock(iEdge, residual);
448  if (implicit)
449  Jacobian.UpdateBlocksSub(iEdge, residual.jacobian_i, residual.jacobian_j);

451  else {

453  LinSysRes.AddBlock(jPoint, residual);
454

456  Jacobian.UpdateBlocksSub(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
457  }

461 template <class V, ENUM_REGIME R>
462 void CFVMFlowSolverBase<V, R>::ComputeVerificationError(CGeometry* geometry, CConfig* config) {
463 
464  /*--- The errors only need to be computed on the finest grid. ---*/
465  if (MGLevel != MESH_0) return;
466 
467  /*--- If this is a verification case, we can compute the global
468  error metrics by using the difference between the local error
469  and the known solution at each DOF. This is then collected into
470  RMS (L2) and maximum (Linf) global error norms. From these
471  global measures, one can compute the order of accuracy. ---*/
472 
473  bool write_heads =
474  ((((config->GetInnerIter() % (config->GetScreen_Wrt_Freq(2) * 40)) == 0) && (config->GetInnerIter() != 0)) ||
475  (config->GetInnerIter() == 1));
476  if (!write_heads) return;
477 
478  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
479 
480  /*--- Check if there actually is an exact solution for this
481  verification case, if computed at all. ---*/
482  if (VerificationSolution && VerificationSolution->ExactSolutionKnown()) {
483  /*--- Get the physical time if necessary. ---*/
484  su2double time = 0.0;
485  if (config->GetTime_Marching() != TIME_MARCHING::STEADY) time = config->GetPhysicalTime();
486 
487  /*--- Reset the global error measures to zero. ---*/
488  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
489  VerificationSolution->SetError_RMS(iVar, 0.0);
490  VerificationSolution->SetError_Max(iVar, 0.0, 0);
491  }
492 
493  /*--- Loop over all owned points. ---*/
494  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
495  /* Set the pointers to the coordinates and solution of this DOF. */
496  const su2double* coor = geometry->nodes->GetCoord(iPoint);
497  su2double* solDOF = nodes->GetSolution(iPoint);
498 
499  /* Get local error from the verification solution class. */
500  vector<su2double> error(nVar, 0.0);
501  VerificationSolution->GetLocalError(coor, time, solDOF, error.data());
502 
503  /* Increment the global error measures */
504  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
505  VerificationSolution->AddError_RMS(iVar, error[iVar] * error[iVar]);
506  VerificationSolution->AddError_Max(iVar, fabs(error[iVar]), geometry->nodes->GetGlobalIndex(iPoint),
507  geometry->nodes->GetCoord(iPoint));
508  }
509  }
510 
511  /* Finalize the calculation of the global error measures. */
512  VerificationSolution->SetVerificationError(geometry->GetGlobal_nPointDomain(), config);
513 
514  /*--- Screen output of the error metrics. This can be improved
515  once the new output classes are in place. ---*/
516 
517  PrintVerificationError(config);
518  }
519 
520  }
521  END_SU2_OMP_SAFE_GLOBAL_ACCESS
522 }

524 template <class V, ENUM_REGIME R>
525 void CFVMFlowSolverBase<V, R>::ComputeUnderRelaxationFactor(const CConfig* config) {
526  /* Loop over the solution update given by relaxing the linear
527  system for this nonlinear iteration. */
528 
529  const su2double allowableRatio = 0.2;
530 
531  SU2_OMP_FOR_STAT(omp_chunk_size)
532  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
533  su2double localUnderRelaxation = 1.0;
534 
535  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
536  /* We impose a limit on the maximum percentage that the
537  density and energy can change over a nonlinear iteration. */
538 
539  if ((iVar == 0) || (iVar == nVar - 1)) {
540  const unsigned long index = iPoint * nVar + iVar;
541  su2double ratio = fabs(LinSysSol[index]) / (fabs(nodes->GetSolution(iPoint, iVar)) + EPS);
542  if (ratio > allowableRatio) {
543  localUnderRelaxation = min(allowableRatio / ratio, localUnderRelaxation);
544  }
545  }
546  }
547 
548  /* Threshold the relaxation factor in the event that there is
549  a very small value. This helps avoid catastrophic crashes due
550  to non-realizable states by canceling the update. */
551 
552  if (localUnderRelaxation < 1e-10) localUnderRelaxation = 0.0;
553 
554  /* Store the under-relaxation factor for this point. */
555 
556  nodes->SetUnderRelaxation(iPoint, localUnderRelaxation);
557  }
558  END_SU2_OMP_FOR
559 }

586 template <class V, ENUM_REGIME R>
587 void CFVMFlowSolverBase<V, R>::ComputeVorticityAndStrainMag(const CConfig& config, const CGeometry *geometry, unsigned short iMesh) {
588 
589  auto& StrainMag = nodes->GetStrainMag();
590 
591  ompMasterAssignBarrier(StrainMag_Max,0.0, Omega_Max,0.0);
592 
593  su2double strainMax = 0.0, omegaMax = 0.0;
594 
595  SU2_OMP_FOR_STAT(omp_chunk_size)
596  for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint) {
597 
598  const auto VelocityGradient = nodes->GetVelocityGradient(iPoint);
599  auto Vorticity = nodes->GetVorticity(iPoint);
600 
601  /*--- Vorticity ---*/
602 
603  Vorticity[0] = 0.0;
604  Vorticity[1] = 0.0;
605  Vorticity[2] = VelocityGradient(1,0)-VelocityGradient(0,1);
606 
607  if (nDim == 3) {
608  Vorticity[0] = VelocityGradient(2,1)-VelocityGradient(1,2);
609  Vorticity[1] = -(VelocityGradient(2,0)-VelocityGradient(0,2));
610  }
611 
612  /*--- Strain Magnitude ---*/
613 
614  const su2double vy = nodes->GetVelocity(iPoint, 1);
615  const su2double y = geometry->nodes->GetCoord(iPoint, 1);
616  AD::StartPreacc();
617  AD::SetPreaccIn(VelocityGradient, nDim, nDim);
618  AD::SetPreaccIn(vy, y);
619 
620  StrainMag(iPoint) = 0.0;
621 
622  /*--- Add diagonal part ---*/
623 
624  for (unsigned long iDim = 0; iDim < nDim; iDim++) {
625  StrainMag(iPoint) += pow(VelocityGradient(iDim, iDim), 2);
626  }
627  if (config.GetAxisymmetric() && y > EPS) {
628  StrainMag(iPoint) += pow(vy / y, 2);
629  }
630 
631  /*--- Add off diagonals ---*/
632 
633  StrainMag(iPoint) += 2.0*pow(0.5*(VelocityGradient(0,1) + VelocityGradient(1,0)), 2);
634 
635  if (nDim == 3) {
636  StrainMag(iPoint) += 2.0*pow(0.5*(VelocityGradient(0,2) + VelocityGradient(2,0)), 2);
637  StrainMag(iPoint) += 2.0*pow(0.5*(VelocityGradient(1,2) + VelocityGradient(2,1)), 2);
638  }
639 
640  StrainMag(iPoint) = sqrt(2.0*StrainMag(iPoint));
641  AD::SetPreaccOut(StrainMag(iPoint));
642 
643  /*--- Max is not differentiable, so we not register them for preacc. ---*/
644  strainMax = max(strainMax, StrainMag(iPoint));
645  omegaMax = max(omegaMax, GeometryToolbox::Norm(3, Vorticity));
646 
647  AD::EndPreacc();
648  }
649  END_SU2_OMP_FOR
650 
651  if ((iMesh == MESH_0) && (config.GetComm_Level() == COMM_FULL)) {
652  SU2_OMP_CRITICAL {
653  StrainMag_Max = max(StrainMag_Max, strainMax);
654  Omega_Max = max(Omega_Max, omegaMax);
655  }
656  END_SU2_OMP_CRITICAL
657 
658  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
659  {
660  su2double MyOmega_Max = Omega_Max;
661  su2double MyStrainMag_Max = StrainMag_Max;
662 
663  SU2_MPI::Allreduce(&MyStrainMag_Max, &StrainMag_Max, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
664  SU2_MPI::Allreduce(&MyOmega_Max, &Omega_Max, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
665  }
666  END_SU2_OMP_SAFE_GLOBAL_ACCESS
667  }

771 template <class V, ENUM_REGIME R>
772 void CFVMFlowSolverBase<V, R>::LoadRestart_impl(CGeometry **geometry, CSolver ***solver, CConfig *config, int iter,
773  bool update_geo, su2double* SolutionRestart,
774  unsigned short nVar_Restart) {
775  /*--- Restart the solution from file information ---*/
776 
777  const string restart_filename = config->GetFilename(config->GetSolution_FileName(), "", iter);
778  const bool static_fsi = ((config->GetTime_Marching() == TIME_MARCHING::STEADY) && config->GetFSI_Simulation());
779 
780  /*--- To make this routine safe to call in parallel most of it can only be executed by one thread. ---*/
781  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
782 
783  if (nVar_Restart == 0) nVar_Restart = nVar;
784 
785  /*--- Skip coordinates ---*/
786 
787  unsigned short skipVars = nDim;
788 
789  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
790 
791  if (config->GetRead_Binary_Restart()) {
792  Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
793  } else {
794  Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
795  }
796 
797  bool steady_restart = config->GetSteadyRestart();
798  if (update_geo && dynamic_grid) {
799  auto notFound = fields.end();
800  if (find(fields.begin(), notFound, string("\"Grid_Velocity_x\"")) == notFound) {
801  if (rank == MASTER_NODE)
802  cout << "\nWARNING: The restart file does not contain grid velocities, these will be set to zero.\n" << endl;
803  steady_restart = true;
804  }
805  }
806 
807  /*--- Load data from the restart into correct containers. ---*/
808 
809  unsigned long counter = 0;
810  for (auto iPoint_Global = 0ul; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++) {
811 
812  /*--- Retrieve local index. If this node from the restart file lives
813  on the current processor, we will load and instantiate the vars. ---*/
814 
815  const auto iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
816 
817  if (iPoint_Local > -1) {
818 
819  /*--- We need to store this point's data, so jump to the correct
820  offset in the buffer of data from the restart file and load it. ---*/
821 
822  auto index = counter * Restart_Vars[1] + skipVars;
823 
824  if (SolutionRestart == nullptr) {
825  for (auto iVar = 0u; iVar < nVar_Restart; iVar++)
826  nodes->SetSolution(iPoint_Local, iVar, Restart_Data[index+iVar]);
827  }
828  else {
829  /*--- Used as buffer, allows defaults for nVar > nVar_Restart. ---*/
830  for (auto iVar = 0u; iVar < nVar_Restart; iVar++)
831  SolutionRestart[iVar] = Restart_Data[index + iVar];
832  nodes->SetSolution(iPoint_Local, SolutionRestart);
833  }
834 
835  /*--- For dynamic meshes, read in and store the
836  grid coordinates and grid velocities for each node. ---*/
837 
838  if (dynamic_grid && update_geo) {
839 
840  /*--- Read in the next 2 or 3 variables which are the grid velocities ---*/
841  /*--- If we are restarting the solution from a previously computed static calculation (no grid movement) ---*/
842  /*--- the grid velocities are set to 0. This is useful for FSI computations ---*/
843 
844  /*--- Rewind the index to retrieve the Coords. ---*/
845  index = counter * Restart_Vars[1];
846  const auto* Coord = &Restart_Data[index];
847 
848  su2double GridVel[MAXNDIM] = {0.0};
849  if (!steady_restart) {
850  /*--- Move the index forward to get the grid velocities. ---*/
851  index += skipVars + nVar_Restart + config->GetnTurbVar();
852  for (auto iDim = 0u; iDim < nDim; iDim++) { GridVel[iDim] = Restart_Data[index+iDim]; }
853  }
854 
855  for (auto iDim = 0u; iDim < nDim; iDim++) {
856  geometry[MESH_0]->nodes->SetCoord(iPoint_Local, iDim, Coord[iDim]);
857  geometry[MESH_0]->nodes->SetGridVel(iPoint_Local, iDim, GridVel[iDim]);
858  }
859  }
860 
861  /*--- For static FSI problems, grid_movement is 0 but we need to read in and store the
862  grid coordinates for each node (but not the grid velocities, as there are none). ---*/
863 
864  if (static_fsi && update_geo) {
865  /*--- Rewind the index to retrieve the Coords. ---*/
866  index = counter*Restart_Vars[1];
867  const auto* Coord = &Restart_Data[index];
868 
869  for (auto iDim = 0u; iDim < nDim; iDim++) {
870  geometry[MESH_0]->nodes->SetCoord(iPoint_Local, iDim, Coord[iDim]);
871  }
872  }
873 
874  /*--- Increment the overall counter for how many points have been loaded. ---*/
875  counter++;
876  }
877  }
878 
879  /*--- Detect a wrong solution file ---*/
880 
881  if (counter != nPointDomain) {
882  SU2_MPI::Error(string("The solution file ") + restart_filename + string(" does not match with the mesh file.\n") +
883  string("This can be caused by empty lines at the end of the file."), CURRENT_FUNCTION);
884  }
885  }
886  END_SU2_OMP_SAFE_GLOBAL_ACCESS
887 
888  /*--- Update the geometry for flows on deforming meshes. ---*/
889 
890  if ((dynamic_grid || static_fsi) && update_geo) {
891 
892  CGeometry::UpdateGeometry(geometry, config);
893 
894  if (dynamic_grid) {
895  for (auto iMesh = 0u; iMesh <= config->GetnMGLevels(); iMesh++) {
896 
897  /*--- Compute the grid velocities on the coarser levels. ---*/
898  if (iMesh) geometry[iMesh]->SetRestricted_GridVelocity(geometry[iMesh - 1]);
899  else {
900  geometry[MESH_0]->InitiateComms(geometry[MESH_0], config, GRID_VELOCITY);
901  geometry[MESH_0]->CompleteComms(geometry[MESH_0], config, GRID_VELOCITY);
902  }
903  }
904  }
905  }
906 
907  /*--- Communicate the loaded solution on the fine grid before we transfer
908  it down to the coarse levels. We also call the preprocessing routine
909  on the fine level in order to have all necessary quantities updated,
910  especially if this is a turbulent simulation (eddy viscosity). ---*/
911 
912  solver[MESH_0][FLOW_SOL]->InitiateComms(geometry[MESH_0], config, SOLUTION);
913  solver[MESH_0][FLOW_SOL]->CompleteComms(geometry[MESH_0], config, SOLUTION);
914 
915  /*--- For turbulent/species simulations the flow preprocessing is done by the turbulence/species solver
916  * after it loads its variables (they are needed to compute flow primitives). In case turbulence and species, the
917  * species solver does all the Pre-/Postprocessing. ---*/
918  if (config->GetKind_Turb_Model() == TURB_MODEL::NONE &&
919  config->GetKind_Species_Model() == SPECIES_MODEL::NONE) {
920  solver[MESH_0][FLOW_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
921  }
922 
923  /*--- Interpolate the solution down to the coarse multigrid levels ---*/
924 
925  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); iMesh++) {
926  MultigridRestriction(*geometry[iMesh - 1], solver[iMesh - 1][FLOW_SOL]->GetNodes()->GetSolution(),
927  *geometry[iMesh], solver[iMesh][FLOW_SOL]->GetNodes()->GetSolution());
928  solver[iMesh][FLOW_SOL]->InitiateComms(geometry[iMesh], config, SOLUTION);
929  solver[iMesh][FLOW_SOL]->CompleteComms(geometry[iMesh], config, SOLUTION);
930 
931  if (config->GetKind_Turb_Model() == TURB_MODEL::NONE &&
932  config->GetKind_Species_Model() == SPECIES_MODEL::NONE) {
933  solver[iMesh][FLOW_SOL]->Preprocessing(geometry[iMesh], solver[iMesh], config, iMesh, NO_RK_ITER, RUNTIME_FLOW_SYS, false);
934  }
935  }
936 
937  /*--- Update the old geometry (coordinates n and n-1) in dual time-stepping strategy. ---*/
938  const bool dual_time = ((config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
939  (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND));
940  if (dual_time && config->GetGrid_Movement() && !config->GetDeform_Mesh() &&
941  (config->GetKind_GridMovement() != RIGID_MOTION)) {
942  Restart_OldGeometry(geometry[MESH_0], config);
943  }
944 
945  /*--- Go back to single threaded execution. ---*/
946  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
947  {
948  /*--- Delete the class memory that is used to load the restart. ---*/
949 
950  delete [] Restart_Vars;
951  Restart_Vars = nullptr;
952  delete [] Restart_Data;
953  Restart_Data = nullptr;
954  }
955  END_SU2_OMP_SAFE_GLOBAL_ACCESS
956 }

1063 void CFVMFlowSolverBase<V, R>::BC_Sym_Plane(CGeometry* geometry, CSolver** solver_container, CNumerics* conv_numerics,
1064  CNumerics* visc_numerics, CConfig* config, unsigned short val_marker) {
1065  unsigned short iDim, iVar;
1066  unsigned long iVertex, iPoint;
1067 
1068  bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
1069  bool viscous = config->GetViscous();
1070  bool preprocessed = false;
1071 
1072  /*--- Allocation of variables necessary for convective fluxes. ---*/
1073  su2double Area, ProjVelocity_i, *V_reflected, *V_domain, Normal[MAXNDIM] = {0.0}, UnitNormal[MAXNDIM] = {0.0};
1074 
1075  /*--- Allocation of variables necessary for viscous fluxes. ---*/
1076  su2double ProjGradient, ProjNormVelGrad, ProjTangVelGrad, TangentialNorm,
1077  Tangential[MAXNDIM] = {0.0}, GradNormVel[MAXNDIM] = {0.0}, GradTangVel[MAXNDIM] = {0.0};
1078 
1079  /*--- Allocation of primitive gradient arrays for viscous fluxes. ---*/
1080  su2activematrix Grad_Reflected(nPrimVarGrad, nDim);
1081 
1082  /*--- Loop over all the vertices on this boundary marker. ---*/
1083 
1084  SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
1085  for (iVertex = 0; iVertex < geometry->nVertex[val_marker]; iVertex++) {
1086  if (!preprocessed || geometry->bound_is_straight[val_marker] != true) {
1087  /*----------------------------------------------------------------------------------------------*/
1088  /*--- Preprocessing: ---*/
1089  /*--- Compute the unit normal and (in case of viscous flow) a corresponding unit tangential ---*/
1090  /*--- to that normal. On a straight(2D)/plane(3D) boundary these two vectors are constant. ---*/
1091  /*--- This circumstance is checked in geometry->ComputeSurf_Straightness(...) and stored ---*/
1092  /*--- such that the recomputation does not occur for each node. On true symmetry planes, the ---*/
1093  /*--- normal is constant but this routines is used for Symmetry, Euler-Wall in inviscid flow ---*/
1094  /*--- and Euler Wall in viscous flow as well. In the latter curvy boundaries are likely to ---*/
1095  /*--- happen. In doubt, the conditional above which checks straightness can be thrown out ---*/
1096  /*--- such that the recomputation is done for each node (which comes with a tiny performance ---*/
1097  /*--- penalty). ---*/
1098  /*----------------------------------------------------------------------------------------------*/
1099 
1100  preprocessed = true;
1101 
1102  /*--- Normal vector for a random vertex (zero) on this marker (negate for outward convention). ---*/
1103  geometry->vertex[val_marker][iVertex]->GetNormal(Normal);
1104  for (iDim = 0; iDim < nDim; iDim++) Normal[iDim] = -Normal[iDim];
1105 
1106  /*--- Compute unit normal, to be used for unit tangential, projected velocity and velocity
1107  component gradients. ---*/
1108  Area = GeometryToolbox::Norm(nDim, Normal);
1109 
1110  for (iDim = 0; iDim < nDim; iDim++) UnitNormal[iDim] = -Normal[iDim] / Area;
1111 
1112  /*--- Preprocessing: Compute unit tangential, the direction is arbitrary as long as
1113  t*n=0 && |t|_2 = 1 ---*/
1114  if (viscous) {
1115  switch (nDim) {
1116  case 2: {
1117  Tangential[0] = -UnitNormal[1];
1118  Tangential[1] = UnitNormal[0];
1119  break;
1120  }
1121  case 3: {
1122  /*--- n = ai + bj + ck, if |b| > |c| ---*/
1123  if (abs(UnitNormal[1]) > abs(UnitNormal[2])) {
1124  /*--- t = bi + (c-a)j - bk ---*/
1125  Tangential[0] = UnitNormal[1];
1126  Tangential[1] = UnitNormal[2] - UnitNormal[0];
1127  Tangential[2] = -UnitNormal[1];
1128  } else {
1129  /*--- t = ci - cj + (b-a)k ---*/
1130  Tangential[0] = UnitNormal[2];
1131  Tangential[1] = -UnitNormal[2];
1132  Tangential[2] = UnitNormal[1] - UnitNormal[0];
1133  }
1134  /*--- Make it a unit vector. ---*/
1135  TangentialNorm = sqrt(pow(Tangential[0], 2) + pow(Tangential[1], 2) + pow(Tangential[2], 2));
1136  Tangential[0] = Tangential[0] / TangentialNorm;
1137  Tangential[1] = Tangential[1] / TangentialNorm;
1138  Tangential[2] = Tangential[2] / TangentialNorm;
1139  break;
1140  }
1141  } // switch
1142  } // if viscous
1143  } // if bound_is_straight
1144 
1145  iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
1146 
1147  /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
1148  if (geometry->nodes->GetDomain(iPoint)) {
1149  /*-------------------------------------------------------------------------------*/
1150  /*--- Step 1: For the convective fluxes, create a reflected state of the ---*/
1151  /*--- Primitive variables by copying all interior values to the ---*/
1152  /*--- reflected. Only the velocity is mirrored along the symmetry ---*/
1153  /*--- axis. Based on the Upwind_Residual routine. ---*/
1154  /*-------------------------------------------------------------------------------*/
1155 
1156  /*--- Allocate the reflected state at the symmetry boundary. ---*/
1157  V_reflected = GetCharacPrimVar(val_marker, iVertex);
1158 
1159  /*--- Grid movement ---*/
1160  if (dynamic_grid)
1161  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
1162 
1163  /*--- Normal vector for this vertex (negate for outward convention). ---*/
1164  geometry->vertex[val_marker][iVertex]->GetNormal(Normal);
1165  for (iDim = 0; iDim < nDim; iDim++) Normal[iDim] = -Normal[iDim];
1166  conv_numerics->SetNormal(Normal);
1167 
1168  /*--- Get current solution at this boundary node ---*/
1169  V_domain = nodes->GetPrimitive(iPoint);
1170 
1171  /*--- Set the reflected state based on the boundary node. Scalars are copied and
1172  the velocity is mirrored along the symmetry boundary, i.e. the velocity in
1173  normal direction is substracted twice. ---*/
1174  for (iVar = 0; iVar < nPrimVar; iVar++) V_reflected[iVar] = nodes->GetPrimitive(iPoint, iVar);
1175 
1176  /*--- Compute velocity in normal direction (ProjVelcity_i=(v*n)) und substract twice from
1177  velocity in normal direction: v_r = v - 2 (v*n)n ---*/
1178  ProjVelocity_i = nodes->GetProjVel(iPoint, UnitNormal);
1179 
1180  /*--- Adjustment to v.n due to grid movement. ---*/
1181  if (dynamic_grid) {
1182  ProjVelocity_i -= GeometryToolbox::DotProduct(nDim, geometry->nodes->GetGridVel(iPoint), UnitNormal);
1183  }
1184 
1185  for (iDim = 0; iDim < nDim; iDim++)
1186  V_reflected[iDim + 1] = nodes->GetVelocity(iPoint, iDim) - 2.0 * ProjVelocity_i * UnitNormal[iDim];
1187 
1188  /*--- Set Primitive and Secondary for numerics class. ---*/
1189  conv_numerics->SetPrimitive(V_domain, V_reflected);
1190  conv_numerics->SetSecondary(nodes->GetSecondary(iPoint), nodes->GetSecondary(iPoint));
1191 
1192  /*--- Compute the residual using an upwind scheme. ---*/
1193 
1194  auto residual = conv_numerics->ComputeResidual(config);
1195 
1196  /*--- Update residual value ---*/
1197  LinSysRes.AddBlock(iPoint, residual);
1198 
1199  /*--- Jacobian contribution for implicit integration. ---*/
1200  if (implicit) {
1201  Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
1202  }
1203 
1204  if (viscous) {
1205  /*-------------------------------------------------------------------------------*/
1206  /*--- Step 2: The viscous fluxes of the Navier-Stokes equations depend on the ---*/
1207  /*--- Primitive variables and their gradients. The viscous numerics ---*/
1208  /*--- container is filled just as the convective numerics container, ---*/
1209  /*--- but the primitive gradients of the reflected state have to be ---*/
1210  /*--- determined additionally such that symmetry at the boundary is ---*/
1211  /*--- enforced. Based on the Viscous_Residual routine. ---*/
1212  /*-------------------------------------------------------------------------------*/
1213 
1214  /*--- Set the normal vector and the coordinates. ---*/
1215  visc_numerics->SetCoord(geometry->nodes->GetCoord(iPoint), geometry->nodes->GetCoord(iPoint));
1216  visc_numerics->SetNormal(Normal);
1217 
1218  /*--- Set the primitive and Secondary variables. ---*/
1219  visc_numerics->SetPrimitive(V_domain, V_reflected);
1220  visc_numerics->SetSecondary(nodes->GetSecondary(iPoint), nodes->GetSecondary(iPoint));
1221 
1222  /*--- For viscous Fluxes also the gradients of the primitives need to be determined.
1223  1. The gradients of scalars are mirrored along the sym plane just as velocity for the primitives
1224  2. The gradients of the velocity components need more attention, i.e. the gradient of the
1225  normal velocity in tangential direction is mirrored and the gradient of the tangential velocity in
1226  normal direction is mirrored. ---*/
1227 
1228  /*--- Get gradients of primitives of boundary cell ---*/
1229  for (iVar = 0; iVar < nPrimVarGrad; iVar++)
1230  for (iDim = 0; iDim < nDim; iDim++)
1231  Grad_Reflected[iVar][iDim] = nodes->GetGradient_Primitive(iPoint, iVar, iDim);
1232 
1233  /*--- Reflect the gradients for all scalars including the velocity components.
1234  The gradients of the velocity components are set later with the
1235  correct values: grad(V)_r = grad(V) - 2 [grad(V)*n]n, V beeing any primitive ---*/
1236  for (iVar = 0; iVar < nPrimVarGrad; iVar++) {
1237  if (iVar == 0 || iVar > nDim) { // Exclude velocity component gradients
1238 
1239  /*--- Compute projected part of the gradient in a dot product ---*/
1240  ProjGradient = 0.0;
1241  for (iDim = 0; iDim < nDim; iDim++) ProjGradient += Grad_Reflected[iVar][iDim] * UnitNormal[iDim];
1242 
1243  for (iDim = 0; iDim < nDim; iDim++)
1244  Grad_Reflected[iVar][iDim] = Grad_Reflected[iVar][iDim] - 2.0 * ProjGradient * UnitNormal[iDim];
1245  }
1246  }
1247 
1248  /*--- Compute gradients of normal and tangential velocity:
1249  grad(v*n) = grad(v_x) n_x + grad(v_y) n_y (+ grad(v_z) n_z)
1250  grad(v*t) = grad(v_x) t_x + grad(v_y) t_y (+ grad(v_z) t_z) ---*/
1251  for (iVar = 0; iVar < nDim; iVar++) { // counts gradient components
1252  GradNormVel[iVar] = 0.0;
1253  GradTangVel[iVar] = 0.0;
1254  for (iDim = 0; iDim < nDim; iDim++) { // counts sum with unit normal/tangential
1255  GradNormVel[iVar] += Grad_Reflected[iDim + 1][iVar] * UnitNormal[iDim];
1256  GradTangVel[iVar] += Grad_Reflected[iDim + 1][iVar] * Tangential[iDim];
1257  }
1258  }
1259 
1260  /*--- Refelect gradients in tangential and normal direction by substracting the normal/tangential
1261  component twice, just as done with velocity above.
1262  grad(v*n)_r = grad(v*n) - 2 {grad([v*n])*t}t
1263  grad(v*t)_r = grad(v*t) - 2 {grad([v*t])*n}n ---*/
1264  ProjNormVelGrad = 0.0;
1265  ProjTangVelGrad = 0.0;
1266  for (iDim = 0; iDim < nDim; iDim++) {
1267  ProjNormVelGrad += GradNormVel[iDim] * Tangential[iDim]; // grad([v*n])*t
1268  ProjTangVelGrad += GradTangVel[iDim] * UnitNormal[iDim]; // grad([v*t])*n
1269  }
1270 
1271  for (iDim = 0; iDim < nDim; iDim++) {
1272  GradNormVel[iDim] = GradNormVel[iDim] - 2.0 * ProjNormVelGrad * Tangential[iDim];
1273  GradTangVel[iDim] = GradTangVel[iDim] - 2.0 * ProjTangVelGrad * UnitNormal[iDim];
1274  }
1275 
1276  /*--- Transfer reflected gradients back into the Cartesian Coordinate system:
1277  grad(v_x)_r = grad(v*n)_r n_x + grad(v*t)_r t_x
1278  grad(v_y)_r = grad(v*n)_r n_y + grad(v*t)_r t_y
1279  ( grad(v_z)_r = grad(v*n)_r n_z + grad(v*t)_r t_z ) ---*/
1280  for (iVar = 0; iVar < nDim; iVar++) // loops over the velocity component gradients
1281  for (iDim = 0; iDim < nDim; iDim++) // loops over the entries of the above
1282  Grad_Reflected[iVar + 1][iDim] =
1283  GradNormVel[iDim] * UnitNormal[iVar] + GradTangVel[iDim] * Tangential[iVar];
1284 
1285  /*--- Set the primitive gradients of the boundary and reflected state. ---*/
1286  visc_numerics->SetPrimVarGradient(nodes->GetGradient_Primitive(iPoint), CMatrixView<su2double>(Grad_Reflected));
1287 
1288  /*--- Turbulent kinetic energy. ---*/
1289  if (config->GetKind_Turb_Model() == TURB_MODEL::SST)
1290  visc_numerics->SetTurbKineticEnergy(solver_container[TURB_SOL]->GetNodes()->GetSolution(iPoint, 0),
1291  solver_container[TURB_SOL]->GetNodes()->GetSolution(iPoint, 0));
1292 
1293  /*--- Compute and update residual. Note that the viscous shear stress tensor is computed in the
1294  following routine based upon the velocity-component gradients. ---*/
1295  auto residual = visc_numerics->ComputeResidual(config);
1296 
1297  LinSysRes.SubtractBlock(iPoint, residual);
1298 
1299  /*--- Jacobian contribution for implicit integration. ---*/
1300  if (implicit) Jacobian.SubtractBlock2Diag(iPoint, residual.jacobian_i);
1301  } // if viscous
1302  } // if GetDomain
1303  } // for iVertex
1304  END_SU2_OMP_FOR
1305 
1306 }

1622 template <class V, ENUM_REGIME FlowRegime>
1623 void CFVMFlowSolverBase<V, FlowRegime>::SetResidual_DualTime(CGeometry *geometry, CSolver **solver_container,
1624  CConfig *config, unsigned short iRKStep, unsigned short iMesh,
1625  unsigned short RunTime_EqSystem) {
1626  /*--- Local variables ---*/
1627 
1628  unsigned short iVar, iMarker, iDim, iNeigh;
1629  unsigned long iPoint, jPoint, iEdge, iVertex;
1630 
1631  const su2double *U_time_nM1 = nullptr, *U_time_n = nullptr, *U_time_nP1 = nullptr;
1632  su2double Volume_nM1, Volume_nP1, TimeStep;
1633  const su2double *Normal = nullptr, *GridVel_i = nullptr, *GridVel_j = nullptr;
1634  su2double Residual_GCL;
1635 
1636  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
1637  const bool first_order = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST);
1638  const bool second_order = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
1639 
1640  /*--- Store the physical time step ---*/
1641 
1642  TimeStep = config->GetDelta_UnstTimeND();
1643 
1644  /*--- Compute the dual time-stepping source term for static meshes ---*/
1645 
1646  if (!dynamic_grid) {
1647 
1648  /*--- Loop over all nodes (excluding halos) ---*/
1649 
1650  AD::StartNoSharedReading();
1651 
1652  SU2_OMP_FOR_STAT(omp_chunk_size)
1653  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
1654 
1655  /*--- Retrieve the solution at time levels n-1, n, and n+1. Note that
1656  we are currently iterating on U^n+1 and that U^n & U^n-1 are fixed,
1657  previous solutions that are stored in memory. ---*/
1658 
1659  U_time_nM1 = nodes->GetSolution_time_n1(iPoint);
1660  U_time_n = nodes->GetSolution_time_n(iPoint);
1661  U_time_nP1 = nodes->GetSolution(iPoint);
1662 
1663  /*--- CV volume at time n+1. As we are on a static mesh, the volume
1664  of the CV will remained fixed for all time steps. ---*/
1665 
1666  Volume_nP1 = geometry->nodes->GetVolume(iPoint);
1667 
1668  /*--- Compute the dual time-stepping source term based on the chosen
1669  time discretization scheme (1st- or 2nd-order).---*/
1670 
1671  for (iVar = 0; iVar < nVar; iVar++) {
1672  if (first_order)
1673  LinSysRes(iPoint,iVar) += (U_time_nP1[iVar] - U_time_n[iVar])*Volume_nP1 / TimeStep;
1674  if (second_order)
1675  LinSysRes(iPoint,iVar) += ( 3.0*U_time_nP1[iVar] - 4.0*U_time_n[iVar]
1676  +1.0*U_time_nM1[iVar])*Volume_nP1 / (2.0*TimeStep);
1677  }
1678 
1679  /*--- Compute the Jacobian contribution due to the dual time source term. ---*/
1680  if (implicit) {
1681  if (first_order) Jacobian.AddVal2Diag(iPoint, Volume_nP1/TimeStep);
1682  if (second_order) Jacobian.AddVal2Diag(iPoint, (Volume_nP1*3.0)/(2.0*TimeStep));
1683  }
1684  }
1685  END_SU2_OMP_FOR
1686 
1687  AD::EndNoSharedReading();
1688 
1689  }
1690 
1691  else {
1692 
1693  /*--- For unsteady flows on dynamic meshes (rigidly transforming or
1694  dynamically deforming), the Geometric Conservation Law (GCL) should be
1695  satisfied in conjunction with the ALE formulation of the governing
1696  equations. The GCL prevents accuracy issues caused by grid motion, i.e.
1697  a uniform free-stream should be preserved through a moving grid. First,
1698  we will loop over the edges and boundaries to compute the GCL component
1699  of the dual time source term that depends on grid velocities. ---*/
1700 
1701  SU2_OMP_FOR_STAT(omp_chunk_size)
1702  for (iPoint = 0; iPoint < nPointDomain; ++iPoint) {
1703 
1704  GridVel_i = geometry->nodes->GetGridVel(iPoint);
1705  U_time_n = nodes->GetSolution_time_n(iPoint);
1706 
1707  for (iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); iNeigh++) {
1708 
1709  iEdge = geometry->nodes->GetEdge(iPoint, iNeigh);
1710  Normal = geometry->edges->GetNormal(iEdge);
1711 
1712  jPoint = geometry->nodes->GetPoint(iPoint, iNeigh);
1713  GridVel_j = geometry->nodes->GetGridVel(jPoint);
1714 
1715  /*--- Determine whether to consider the normal outward or inward. ---*/
1716  su2double dir = (iPoint < jPoint)? 0.5 : -0.5;
1717 
1718  Residual_GCL = 0.0;
1719  for (iDim = 0; iDim < nDim; iDim++)
1720  Residual_GCL += dir*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
1721 
1722  for (iVar = 0; iVar < nVar; iVar++)
1723  LinSysRes(iPoint,iVar) += U_time_n[iVar]*Residual_GCL;
1724  }
1725  }
1726  END_SU2_OMP_FOR
1727 
1728  /*--- Loop over the boundary edges ---*/
1729 
1730  for (iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
1731  if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
1732  (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
1733  (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
1734 
1735  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
1736  for (iVertex = 0; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
1737 
1738  /*--- Get the index for node i plus the boundary face normal ---*/
1739 
1740  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
1741  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
1742 
1743  /*--- Grid velocities stored at boundary node i ---*/
1744 
1745  GridVel_i = geometry->nodes->GetGridVel(iPoint);
1746 
1747  /*--- Compute the GCL term by dotting the grid velocity with the face
1748  normal. The normal is negated to match the boundary convention. ---*/
1749 
1750  Residual_GCL = 0.0;
1751  for (iDim = 0; iDim < nDim; iDim++)
1752  Residual_GCL -= 0.5*(GridVel_i[iDim]+GridVel_i[iDim])*Normal[iDim];
1753 
1754  /*--- Compute the GCL component of the source term for node i ---*/
1755 
1756  U_time_n = nodes->GetSolution_time_n(iPoint);
1757  for (iVar = 0; iVar < nVar; iVar++)
1758  LinSysRes(iPoint,iVar) += U_time_n[iVar]*Residual_GCL;
1759  }
1760  END_SU2_OMP_FOR
1761  }
1762  }
1763 
1764  /*--- Loop over all nodes (excluding halos) to compute the remainder
1765  of the dual time-stepping source term. ---*/
1766 
1767  AD::StartNoSharedReading();
1768 
1769  SU2_OMP_FOR_STAT(omp_chunk_size)
1770  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
1771 
1772  /*--- Retrieve the solution at time levels n-1, n, and n+1. Note that
1773  we are currently iterating on U^n+1 and that U^n & U^n-1 are fixed,
1774  previous solutions that are stored in memory. ---*/
1775 
1776  U_time_nM1 = nodes->GetSolution_time_n1(iPoint);
1777  U_time_n = nodes->GetSolution_time_n(iPoint);
1778  U_time_nP1 = nodes->GetSolution(iPoint);
1779 
1780  /*--- CV volume at time n-1 and n+1. In the case of dynamically deforming
1781  grids, the volumes will change. On rigidly transforming grids, the
1782  volumes will remain constant. ---*/
1783 
1784  Volume_nM1 = geometry->nodes->GetVolume_nM1(iPoint);
1785  Volume_nP1 = geometry->nodes->GetVolume(iPoint);
1786 
1787  /*--- Compute the dual time-stepping source residual. Due to the
1788  introduction of the GCL term above, the remainder of the source residual
1789  due to the time discretization has a new form.---*/
1790 
1791  for (iVar = 0; iVar < nVar; iVar++) {
1792  if (first_order)
1793  LinSysRes(iPoint,iVar) += (U_time_nP1[iVar] - U_time_n[iVar])*(Volume_nP1/TimeStep);
1794  if (second_order)
1795  LinSysRes(iPoint,iVar) += (U_time_nP1[iVar] - U_time_n[iVar])*(3.0*Volume_nP1/(2.0*TimeStep))
1796  + (U_time_nM1[iVar] - U_time_n[iVar])*(Volume_nM1/(2.0*TimeStep));
1797  }
1798 
1799  /*--- Compute the Jacobian contribution due to the dual time source term. ---*/
1800  if (implicit) {
1801  if (first_order) Jacobian.AddVal2Diag(iPoint, Volume_nP1/TimeStep);
1802  if (second_order) Jacobian.AddVal2Diag(iPoint, (Volume_nP1*3.0)/(2.0*TimeStep));
1803  }
1804  }
1805  END_SU2_OMP_FOR
1806 
1807  AD::EndNoSharedReading();
1808  }
1809 
1810 }
