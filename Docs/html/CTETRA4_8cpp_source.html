31 CTETRA4::CTETRA4() : CElementWithKnownSizes<NGAUSS,NNODE,NDIM>() {
32 
33  /*--- Gauss coordinates and weights ---*/
34 
35  su2double r = ((5.0-sqrt(5.0))/20);
36  su2double s = ((5.0+3*sqrt(5.0))/20);
37  GaussCoord[0][0] = r; GaussCoord[0][1] = r; GaussCoord[0][2] = r; GaussWeight(0) = 1.0/24.0;
38  GaussCoord[0][0] = s; GaussCoord[0][1] = r; GaussCoord[0][2] = r; GaussWeight(1) = 1.0/24.0;
39  GaussCoord[0][0] = r; GaussCoord[0][1] = s; GaussCoord[0][2] = r; GaussWeight(2) = 1.0/24.0;
40  GaussCoord[0][0] = r; GaussCoord[0][1] = r; GaussCoord[0][2] = s; GaussWeight(3) = 1.0/24.0;
41 
42  /*--- Store the values of the shape functions and their derivatives ---*/
43 
44  unsigned short iGauss;
45  su2double Xi, Eta, Zeta, val_Ni;
46 
47  for (iGauss = 0; iGauss < nGaussPoints; iGauss++) {
48 
49  Xi = GaussCoord[iGauss][0];
50  Eta = GaussCoord[iGauss][1];
51  Zeta = GaussCoord[iGauss][2];
52 
53  val_Ni = Xi; GaussPoint[iGauss].SetNi(val_Ni,0);
54  val_Ni = Eta; GaussPoint[iGauss].SetNi(val_Ni,1);
55  val_Ni = 1.0-Xi-Eta-Zeta; GaussPoint[iGauss].SetNi(val_Ni,2);
56  val_Ni = Zeta; GaussPoint[iGauss].SetNi(val_Ni,3);
57 
58  /*--- dN/d xi, dN/d eta, dN/d zeta ---*/
59 
60  dNiXj[iGauss][0][0] = 1.0; dNiXj[iGauss][0][1] = 0.0; dNiXj[iGauss][0][2] = 0.0;
61  dNiXj[iGauss][1][0] = 0.0; dNiXj[iGauss][1][1] = 1.0; dNiXj[iGauss][1][2] = 0.0;
62  dNiXj[iGauss][2][0] = -1.0; dNiXj[iGauss][2][1] = -1.0; dNiXj[iGauss][2][2] = -1.0;
63  dNiXj[iGauss][3][0] = 0.0; dNiXj[iGauss][3][1] = 0.0; dNiXj[iGauss][3][2] = 1.0;

65  }

77 su2double CTETRA4::ComputeVolume(const FrameType mode) const {
78 
79  unsigned short iDim;
80  su2double r1[3] = {0.0,0.0,0.0}, r2[3] = {0.0,0.0,0.0}, r3[3] = {0.0,0.0,0.0}, CrossProduct[3] = {0.0,0.0,0.0};
81  su2double Volume = 0.0;
82 
83  /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
84  for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed)---*/
85  const su2activematrix& Coord = (mode==REFERENCE) ? RefCoord : CurrentCoord;
86 
87  for (iDim = 0; iDim < NDIM; iDim++) {
88  r1[iDim] = Coord[1][iDim] - Coord[0][iDim];
89  r2[iDim] = Coord[2][iDim] - Coord[0][iDim];
90  r3[iDim] = Coord[3][iDim] - Coord[0][iDim];
91  }
92 
93  GeometryToolbox::CrossProduct(r1, r2, CrossProduct);
94  Volume = fabs(GeometryToolbox::DotProduct(3, CrossProduct, r3))/6.0;
95 
96  return Volume;
97 
98 }
