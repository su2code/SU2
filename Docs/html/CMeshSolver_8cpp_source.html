36 CMeshSolver::CMeshSolver(CGeometry *geometry, CConfig *config) : CFEASolver(LINEAR_SOLVER_MODE::MESH_DEFORM) {

170 void CMeshSolver::SetMinMaxVolume(CGeometry *geometry, CConfig *config, bool updated) {
171 
172  /*--- This routine is for post processing, it does not need to be recorded. ---*/
173  const bool wasActive = AD::BeginPassive();
174 
175  /*--- Initialize shared reduction variables. ---*/
176  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
177  MaxVolume = -1E22; MinVolume = 1E22;
178  ElemCounter = 0;
179  }
180  END_SU2_OMP_SAFE_GLOBAL_ACCESS
181 
182  /*--- Local min/max, final reduction outside loop. ---*/
183  su2double maxVol = -1E22, minVol = 1E22;
184  unsigned long elCount = 0;
185 
186  /*--- Loop over the elements in the domain. ---*/
187 
188  SU2_OMP_FOR_DYN(omp_chunk_size)
189  for (unsigned long iElem = 0; iElem < nElement; iElem++) {
190 
191  int thread = omp_get_thread_num();
192 
193  int EL_KIND;
194  unsigned short iNode, nNodes, iDim;
195 
196  GetElemKindAndNumNodes(geometry->elem[iElem]->GetVTK_Type(), EL_KIND, nNodes);
197 
198  CElement* fea_elem = element_container[FEA_TERM][EL_KIND + thread*MAX_FE_KINDS];
199 
200  /*--- For the number of nodes, we get the coordinates from
201  * the connectivity matrix and the geometry structure. ---*/
202 
203  for (iNode = 0; iNode < nNodes; iNode++) {
204 
205  auto indexNode = geometry->elem[iElem]->GetNode(iNode);
206 
207  /*--- Compute the volume with the reference or current coordinates. ---*/
208  for (iDim = 0; iDim < nDim; iDim++) {
209  su2double val_Coord = nodes->GetMesh_Coord(indexNode,iDim);
210  if (updated)
211  val_Coord += nodes->GetSolution(indexNode,iDim);
212 
213  fea_elem->SetRef_Coord(iNode, iDim, val_Coord);
214  }
215  }
216 
217  /*--- Compute the volume of the element (or the area in 2D cases ). ---*/
218 
219  su2double ElemVolume;
220  if (nDim == 2) ElemVolume = fea_elem->ComputeArea();
221  else ElemVolume = fea_elem->ComputeVolume();
222 
223  maxVol = max(maxVol, ElemVolume);
224  minVol = min(minVol, ElemVolume);
225 
226  if (updated) element[iElem].SetCurr_Volume(ElemVolume);
227  else element[iElem].SetRef_Volume(ElemVolume);
228 
229  /*--- Count distorted elements. ---*/
230  if (ElemVolume <= 0.0) elCount++;
231  }
232  END_SU2_OMP_FOR
233  SU2_OMP_CRITICAL
234  {
235  MaxVolume = max(MaxVolume, maxVol);
236  MinVolume = min(MinVolume, minVol);
237  ElemCounter += elCount;
238  }
239  END_SU2_OMP_CRITICAL
240 
241  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
242  {
243  elCount = ElemCounter; maxVol = MaxVolume; minVol = MinVolume;
244  SU2_MPI::Allreduce(&elCount, &ElemCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
245  SU2_MPI::Allreduce(&maxVol, &MaxVolume, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
246  SU2_MPI::Allreduce(&minVol, &MinVolume, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
247  }
248  END_SU2_OMP_SAFE_GLOBAL_ACCESS
249 
250  /*--- Volume from 0 to 1 ---*/
251 
252  SU2_OMP_FOR_STAT(omp_chunk_size)
253  for (unsigned long iElem = 0; iElem < nElement; iElem++) {
254  if (updated) {
255  su2double ElemVolume = element[iElem].GetCurr_Volume()/MaxVolume;
256  element[iElem].SetCurr_Volume(ElemVolume);
257  }
258  else {
259  su2double ElemVolume = element[iElem].GetRef_Volume()/MaxVolume;
260  element[iElem].SetRef_Volume(ElemVolume);
261  }
262  }
263  END_SU2_OMP_FOR
264 
265  /*--- Store the maximum and minimum volume. ---*/
266  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
267  if (updated) {
268  MaxVolume_Curr = MaxVolume;
269  MinVolume_Curr = MinVolume;
270  }
271  else {
272  MaxVolume_Ref = MaxVolume;
273  MinVolume_Ref = MinVolume;
274  }
275 
276  if ((ElemCounter != 0) && (rank == MASTER_NODE))
277  cout <<"There are " << ElemCounter << " elements with negative volume.\n" << endl;
278 
279  }
280  END_SU2_OMP_SAFE_GLOBAL_ACCESS
281 
282  AD::EndPassive(wasActive);
283

286 void CMeshSolver::SetWallDistance(CGeometry *geometry, CConfig *config) {
287 
288  /*--- Initialize min and max distance ---*/
289 
290  MaxDistance = -1E22; MinDistance = 1E22;
291 
292  /*--- Compute the total number of nodes on no-slip boundaries ---*/
293 
294  unsigned long nVertex_SolidWall = 0;
295  for(auto iMarker=0u; iMarker<config->GetnMarker_All(); ++iMarker) {
296  if(config->GetSolid_Wall(iMarker) && !config->GetMarker_All_Deform_Mesh_Sym_Plane(iMarker)) {
297  nVertex_SolidWall += geometry->GetnVertex(iMarker);
298  }
299  }
300 
301  /*--- Allocate the vectors to hold boundary node coordinates
302  and its local ID. ---*/
303 
304  vector<su2double> Coord_bound(nDim*nVertex_SolidWall);
305  vector<unsigned long> PointIDs(nVertex_SolidWall);
306 
307  /*--- Retrieve and store the coordinates of the no-slip boundary nodes
308  and their local point IDs. ---*/
309 
310 
311  for (unsigned long iMarker=0, ii=0, jj=0; iMarker<config->GetnMarker_All(); ++iMarker) {
312 
313  if (!config->GetSolid_Wall(iMarker) || config->GetMarker_All_Deform_Mesh_Sym_Plane(iMarker)) continue;
314 
315  for (auto iVertex=0u; iVertex<geometry->GetnVertex(iMarker); ++iVertex) {
316  auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
317  PointIDs[jj++] = iPoint;
318  for (auto iDim=0u; iDim<nDim; ++iDim){
319  Coord_bound[ii++] = nodes->GetMesh_Coord(iPoint,iDim);
320  }
321  }
322  }
323 
324  /*--- Build the ADT of the boundary nodes. ---*/
325 
326  CADTPointsOnlyClass WallADT(nDim, nVertex_SolidWall, Coord_bound.data(),
327  PointIDs.data(), true);
328 
329  SU2_OMP_PARALLEL
330  {
331  /*--- Loop over all interior mesh nodes and compute the distances to each
332  of the no-slip boundary nodes. Store the minimum distance to the wall
333  for each interior mesh node. ---*/
334 
335  if( WallADT.IsEmpty() ) {
336 
337  /*--- No solid wall boundary nodes in the entire mesh. Set the
338  wall distance to MaxDistance so we get stiffness of 1. ---*/
339 
340  SU2_OMP_FOR_STAT(omp_chunk_size)
341  for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint) {
342  nodes->SetWallDistance(iPoint, MaxDistance);
343  }
344  END_SU2_OMP_FOR
345  }
346  else {
347  su2double MaxDistance_Local = -1E22, MinDistance_Local = 1E22;
348 
349  /*--- Solid wall boundary nodes are present. Compute the wall
350  distance for all nodes. ---*/
351  SU2_OMP_FOR_DYN(omp_chunk_size)
352  for(auto iPoint = 0ul; iPoint < nPoint; ++iPoint) {
353  su2double dist;
354  unsigned long pointID;
355  int rankID;
356  WallADT.DetermineNearestNode(nodes->GetMesh_Coord(iPoint), dist,
357  pointID, rankID);
358  nodes->SetWallDistance(iPoint,dist);
359 
360  MaxDistance_Local = max(MaxDistance_Local, dist);
361 
362  /*--- To discard points on the surface we use > EPS ---*/
363 
364  if (dist > EPS) MinDistance_Local = min(MinDistance_Local, dist);
365 
366  }
367  END_SU2_OMP_FOR
368  SU2_OMP_CRITICAL
369  {
370  MaxDistance = max(MaxDistance, MaxDistance_Local);
371  MinDistance = min(MinDistance, MinDistance_Local);
372  }
373  END_SU2_OMP_CRITICAL
374 
375  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
376  {
377  MaxDistance_Local = MaxDistance;
378  MinDistance_Local = MinDistance;
379  SU2_MPI::Allreduce(&MaxDistance_Local, &MaxDistance, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
380  SU2_MPI::Allreduce(&MinDistance_Local, &MinDistance, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
381  }
382  END_SU2_OMP_SAFE_GLOBAL_ACCESS
383  }
384 
385  /*--- Normalize distance from 0 to 1 ---*/
386  SU2_OMP_FOR_STAT(omp_chunk_size)
387  for (auto iPoint = 0ul; iPoint < nPoint; ++iPoint) {
388  su2double nodeDist = nodes->GetWallDistance(iPoint)/MaxDistance;
389  nodes->SetWallDistance(iPoint,nodeDist);
390  }
391  END_SU2_OMP_FOR
392 
393  /*--- Compute the element distances ---*/
394  SU2_OMP_FOR_STAT(omp_chunk_size)
395  for (auto iElem = 0ul; iElem < nElement; iElem++) {
396 
397  int EL_KIND;
398  unsigned short nNodes = 0;
399  GetElemKindAndNumNodes(geometry->elem[iElem]->GetVTK_Type(), EL_KIND, nNodes);
400 
401  /*--- Average the distance of the nodes in the element ---*/
402 
403  su2double ElemDist = 0.0;
404  for (auto iNode = 0u; iNode < nNodes; iNode++) {
405  auto iPoint = geometry->elem[iElem]->GetNode(iNode);
406  ElemDist += nodes->GetWallDistance(iPoint);
407  }
408  ElemDist = ElemDist/su2double(nNodes);
409 
410  element[iElem].SetWallDistance(ElemDist);
411  }
412  END_SU2_OMP_FOR
413 
414  }
415  END_SU2_OMP_PARALLEL
416 }

418 void CMeshSolver::SetMesh_Stiffness(CGeometry **geometry, CNumerics **numerics, CConfig *config){
419 
420  if (stiffness_set) return;
421 
422  /*--- Use the config option as an upper bound on elasticity modulus.
423  * For RANS meshes the range of element volume or wall distance is
424  * very large and leads to an ill-conditioned stiffness matrix.
425  * Absolute values of elasticity modulus are not important for
426  * mesh deformation, since linear elasticity is used and all
427  * boundary conditions are essential (Dirichlet). ---*/
428  const su2double maxE = config->GetDeform_ElasticityMod();
429 
430  /*--- All threads must execute the entire loop (no worksharing),
431  * each sets the stiffnesses for its numerics instance. ---*/
432  SU2_OMP_PARALLEL
433  {
434  CNumerics* myNumerics = numerics[FEA_TERM + omp_get_thread_num()*MAX_TERMS];
435 
436  switch (config->GetDeform_Stiffness_Type()) {
437 
438  /*--- Stiffness inverse of the volume of the element. ---*/
439  case INVERSE_VOLUME:
440  for (unsigned long iElem = 0; iElem < nElement; iElem++) {
441  su2double E = 1.0 / element[iElem].GetRef_Volume();
442  myNumerics->SetMeshElasticProperties(iElem, min(E,maxE));
443  }
444  break;
445 
446  /*--- Stiffness inverse of the distance of the element to the closest wall. ---*/
447  case SOLID_WALL_DISTANCE: {
448  const su2double offset = config->GetDeform_StiffLayerSize();
449  if (fabs(offset) > 0.0) {
450  /*--- With prescribed layer of maximum stiffness (reaches max and holds). ---*/
451  su2double d0 = offset / MaxDistance;
452  su2double dmin = 1.0 / maxE;
453  su2double scale = 1.0 / (1.0 - d0);
454  for (unsigned long iElem = 0; iElem < nElement; iElem++) {
455  su2double E = 1.0 / max(dmin, (element[iElem].GetWallDistance() - d0)*scale);
456  myNumerics->SetMeshElasticProperties(iElem, E);
457  }
458  } else {
459  /*--- Without prescribed layer of maximum stiffness (may not reach max). ---*/
460  for (unsigned long iElem = 0; iElem < nElement; iElem++) {
461  su2double E = 1.0 / element[iElem].GetWallDistance();
462  myNumerics->SetMeshElasticProperties(iElem, min(E,maxE));
463  }
464  }
465  }
466  break;
467  }
468  }
469  END_SU2_OMP_PARALLEL
470 
471  stiffness_set = true;
472 
473 }

475 void CMeshSolver::DeformMesh(CGeometry **geometry, CNumerics **numerics, CConfig *config){
476 
477  if (multizone) nodes->Set_BGSSolution_k();
478 
479  /*--- Capture a few MPI dependencies for AD. ---*/
480  geometry[MESH_0]->InitiateComms(geometry[MESH_0], config, COORDINATES);
481  geometry[MESH_0]->CompleteComms(geometry[MESH_0], config, COORDINATES);
482 
483  InitiateComms(geometry[MESH_0], config, SOLUTION);
484  CompleteComms(geometry[MESH_0], config, SOLUTION);
485 
486  InitiateComms(geometry[MESH_0], config, MESH_DISPLACEMENTS);
487  CompleteComms(geometry[MESH_0], config, MESH_DISPLACEMENTS);
488 
489  /*--- Compute the stiffness matrix, no point recording because we clear the residual. ---*/
490 
491  const bool wasActive = AD::BeginPassive();
492 
493  Compute_StiffMatrix(geometry[MESH_0], numerics, config);
494 
495  AD::EndPassive(wasActive);
496 
497  /*--- Clear residual (loses AD info), we do not want an incremental solution. ---*/
498  SU2_OMP_PARALLEL {
499  LinSysRes.SetValZero();
500 
501  if (time_domain && config->GetFSI_Simulation()) {
502  SU2_OMP_FOR_STAT(omp_chunk_size)
503  for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint)
504  for (unsigned short iDim = 0; iDim < nDim; ++iDim)
505  LinSysSol(iPoint, iDim) = nodes->GetSolution(iPoint, iDim);
506  END_SU2_OMP_FOR
507  }
508  }
509  END_SU2_OMP_PARALLEL
510 
511  /*--- Impose boundary conditions (all of them are ESSENTIAL BC's - displacements). ---*/
512  SetBoundaryDisplacements(geometry[MESH_0], config, false);
513 
514  /*--- Solve the linear system. ---*/
515  Solve_System(geometry[MESH_0], config);
516 
517  SU2_OMP_PARALLEL {
518 
519  /*--- Update the grid coordinates and cell volumes using the solution
520  of the linear system (usol contains the x, y, z displacements). ---*/
521  UpdateGridCoord(geometry[MESH_0], config);
522 
523  /*--- Update the dual grid. ---*/
524  CGeometry::UpdateGeometry(geometry, config);
525 
526  /*--- Check for failed deformation (negative volumes). ---*/
527  SetMinMaxVolume(geometry[MESH_0], config, true);
528 
529  /*--- The Grid Velocity is only computed if the problem is time domain ---*/
530  if (time_domain && !config->GetFSI_Simulation())
531  ComputeGridVelocity(geometry, config);
532 
533  }
534  END_SU2_OMP_PARALLEL
535 
536  if (time_domain && config->GetFSI_Simulation()) {
537  ComputeGridVelocity_FromBoundary(geometry, numerics, config);
538  }

542 void CMeshSolver::UpdateGridCoord(CGeometry *geometry, const CConfig *config){
543 
544  /*--- Update the grid coordinates using the solution of the linear system ---*/
545 
546  /*--- LinSysSol contains the absolute x, y, z displacements. ---*/
547  SU2_OMP_FOR_STAT(omp_chunk_size)
548  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++){
549  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
550  /*--- Retrieve the displacement from the solution of the linear system ---*/
551  su2double val_disp = LinSysSol(iPoint, iDim);
552  /*--- Store the displacement of the mesh node ---*/
553  nodes->SetSolution(iPoint, iDim, val_disp);
554  /*--- Compute the current coordinate as Mesh_Coord + Displacement ---*/
555  su2double val_coord = nodes->GetMesh_Coord(iPoint,iDim) + val_disp;
556  /*--- Update the geometry container ---*/
557  geometry->nodes->SetCoord(iPoint, iDim, val_coord);
558  }
559  }
560  END_SU2_OMP_FOR
561 
562  /*--- Communicate the updated displacements and mesh coordinates. ---*/
563  geometry->InitiateComms(geometry, config, COORDINATES);
564  geometry->CompleteComms(geometry, config, COORDINATES);
565

568 void CMeshSolver::ComputeGridVelocity_FromBoundary(CGeometry **geometry, CNumerics **numerics, CConfig *config){
569 
570  if (config->GetnZone() == 1)
571  SU2_MPI::Error("It is not possible to compute grid velocity from boundary velocity for single zone problems.\n"
572  "MARKER_FLUID_LOAD should only be used for structural boundaries.", CURRENT_FUNCTION);
573 
574  /*--- Compute the stiffness matrix, no point recording because we clear the residual. ---*/
575 
576  const bool wasActive = AD::BeginPassive();
577 
578  Compute_StiffMatrix(geometry[MESH_0], numerics, config);
579 
580  AD::EndPassive(wasActive);
581 
582  const su2double velRef = config->GetVelocity_Ref();
583  const su2double invVelRef = 1.0 / velRef;
584 
585  /*--- Clear residual (loses AD info), we do not want an incremental solution. ---*/
586  SU2_OMP_PARALLEL {
587  LinSysRes.SetValZero();
588 
589  SU2_OMP_FOR_STAT(omp_chunk_size)
590  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++)
591  for (unsigned short iDim = 0; iDim < nDim; iDim++)
592  LinSysSol(iPoint, iDim) = geometry[MESH_0]->nodes->GetGridVel(iPoint)[iDim] * velRef;
593  END_SU2_OMP_FOR
594  }
595  END_SU2_OMP_PARALLEL
596 
597  /*--- Impose boundary conditions including boundary velocity ---*/
598  SetBoundaryDisplacements(geometry[MESH_0], config, true);
599 
600  /*--- Solve the linear system. ---*/
601  Solve_System(geometry[MESH_0], config);
602 
603  SU2_OMP_PARALLEL {

617 void CMeshSolver::ComputeGridVelocity(CGeometry **geometry, const CConfig *config) const {
618 
619  /*--- Compute the velocity of each node. ---*/
620 
621  const bool firstOrder = config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST;
622  const bool secondOrder = config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND;
623  const su2double invTimeStep = 1.0 / config->GetDelta_UnstTimeND();
624 
625  SU2_OMP_FOR_STAT(omp_chunk_size)
626  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++) {
627 
628  /*--- Coordinates of the current point at n+1, n, & n-1 time levels. ---*/
629 
630  const su2double* Disp_nM1 = nodes->GetSolution_time_n1(iPoint);
631  const su2double* Disp_n = nodes->GetSolution_time_n(iPoint);
632  const su2double* Disp_nP1 = nodes->GetSolution(iPoint);
633 
634  /*--- Compute mesh velocity for this point with 1st or 2nd-order approximation. ---*/
635 
636  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
637 
638  su2double GridVel = 0.0;
639  if (firstOrder)
640  GridVel = (Disp_nP1[iDim] - Disp_n[iDim]) * invTimeStep;
641  else if (secondOrder)
642  GridVel = (1.5*Disp_nP1[iDim] - 2.0*Disp_n[iDim] + 0.5*Disp_nM1[iDim]) * invTimeStep;
643 
644  geometry[MESH_0]->nodes->SetGridVel(iPoint, iDim, GridVel);
645  }
646  }
647  END_SU2_OMP_FOR
648 
649  for (auto iMGlevel = 1u; iMGlevel <= config->GetnMGLevels(); iMGlevel++)
650  geometry[iMGlevel]->SetRestricted_GridVelocity(geometry[iMGlevel-1]);

977  unsigned short iMarker, jMarker, iDim;
978  unsigned long iPoint, iVertex;
979  string Marker_Tag, Moving_Tag;
980 
981  /*--- Retrieve values from the config file ---*/
982 
983  deltaT = config->GetDelta_UnstTimeND();
984  Lref = config->GetLength_Ref();
985 
986  /*--- Compute delta time based on physical time step ---*/
987 
988  time_new = iter*deltaT;
989  if (iter == 0) time_old = time_new;
990  else time_old = (iter-1)*deltaT;
991 
992  /*--- Store displacement of each node on the pitching surface ---*/
993  /*--- Loop over markers and find the particular marker(s) (surface) to pitch ---*/
994 
995  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
996  if (config->GetMarker_All_Moving(iMarker) != YES) continue;
997 
998  Marker_Tag = config->GetMarker_All_TagBound(iMarker);
999 
1000  for (jMarker = 0; jMarker < config->GetnMarker_Moving(); jMarker++) {
1001 
1002  Moving_Tag = config->GetMarker_Moving_TagBound(jMarker);
1003 
1004  if ((Marker_Tag != Moving_Tag) || (config->GetKind_SurfaceMovement(jMarker) != DEFORMING)) {
1005  continue;
1006  }
1007 
1008  /*--- Pitching origin, frequency, and amplitude from config. ---*/
1009 
1010  for (iDim = 0; iDim < 3; iDim++){
1011  Ampl[iDim] = config->GetMarkerPitching_Ampl(jMarker, iDim)*DEG2RAD;
1012  Omega[iDim] = config->GetMarkerPitching_Omega(jMarker, iDim)/config->GetOmega_Ref();
1013  Phase[iDim] = config->GetMarkerPitching_Phase(jMarker, iDim)*DEG2RAD;
1014  Center[iDim] = config->GetMarkerMotion_Origin(jMarker, iDim);
1015  }
1016  /*--- Print some information to the console. Be verbose at the first
1017  iteration only (mostly for debugging purposes). ---*/
1018  // Note that the MASTER_NODE might not contain all the markers being moved.
1019 
1020  if (rank == MASTER_NODE) {
1021  cout << " Storing pitching displacement for marker: ";
1022  cout << Marker_Tag << "." << endl;
1023  if (iter == 0) {
1024  cout << " Pitching frequency: (" << Omega[0] << ", " << Omega[1];
1025  cout << ", " << Omega[2] << ") rad/s about origin: (" << Center[0];
1026  cout << ", " << Center[1] << ", " << Center[2] << ")." << endl;
1027  cout << " Pitching amplitude about origin: (" << Ampl[0]/DEG2RAD;
1028  cout << ", " << Ampl[1]/DEG2RAD << ", " << Ampl[2]/DEG2RAD;
1029  cout << ") degrees."<< endl;
1030  cout << " Pitching phase lag about origin: (" << Phase[0]/DEG2RAD;
1031  cout << ", " << Phase[1]/DEG2RAD <<", "<< Phase[2]/DEG2RAD;
1032  cout << ") degrees."<< endl;
1033  }
1034  }
1035 
1036  /*--- Compute delta change in the angle about the x, y, & z axes. ---*/
1037 
1038  dtheta = -Ampl[0]*(sin(Omega[0]*time_new + Phase[0])
1039  - sin(Omega[0]*time_old + Phase[0]));
1040  dphi = -Ampl[1]*(sin(Omega[1]*time_new + Phase[1])
1041  - sin(Omega[1]*time_old + Phase[1]));
1042  dpsi = -Ampl[2]*(sin(Omega[2]*time_new + Phase[2])
1043  - sin(Omega[2]*time_old + Phase[2]));
1044

1046 
1047  RotationMatrix(dtheta, dphi, dpsi, rotMatrix);
1048

1050 
1051  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
1052 
1053  /*--- Index and coordinates of the current point ---*/
1054 
1055  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
1056  Coord = geometry->nodes->GetCoord(iPoint);
1057 
1058  /*--- Calculate non-dim. position from rotation center ---*/
1059 
1060  for (iDim = 0; iDim < nDim; iDim++)
1061  r[iDim] = (Coord[iDim]-Center[iDim])/Lref;
1062 
1063  /*--- Compute transformed point coordinates ---*/
1064 
1065  Rotate(rotMatrix, Center, r, rotCoord);
1066 
1067  /*--- Calculate delta change in the x, y, & z directions ---*/
1068  for (iDim = 0; iDim < nDim; iDim++)
1069  VarCoord[iDim] = (rotCoord[iDim]-Coord[iDim])/Lref;
1070 
1071  /*--- Set node displacement for volume deformation ---*/
1072 
1073  for (iDim = 0; iDim < nDim; iDim++)
1074  VarCoordAbs[iDim] = nodes->GetBound_Disp(iPoint, iDim) + VarCoord[iDim];
1075 
1076  nodes->SetBound_Disp(iPoint, VarCoordAbs);
1077  }
1078  }
1079  }
