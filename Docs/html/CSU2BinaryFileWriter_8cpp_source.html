40 void CSU2BinaryFileWriter::Write_Data(string val_filename){
41 
42  /*--- Local variables ---*/
43 
44  unsigned short iVar;
45 
46  const vector<string>& fieldNames = dataSorter->GetFieldNames();
47  unsigned short nVar = fieldNames.size();
48  unsigned long nParallel_Poin = dataSorter->GetnPoints();
49  unsigned long nPoint_Global = dataSorter->GetnPointsGlobal();
50 
51  char str_buf[CGNS_STRING_SIZE];
52 
53  /*--- Prepare the first ints containing the counts. The first is a
54  magic number that we can use to check for binary files (it is the hex
55  representation for "SU2"). The second two values are number of variables
56  and number of points (DoFs). ---*/
57 
58  int var_buf_size = 5;
59  int var_buf[5] = {535532, nVar, (int)nPoint_Global, 0, 0};
60 
61  /*--- Open the file using MPI I/O ---*/
62 
63  OpenMPIFile(val_filename);
64 
65  /*--- First, write the number of variables and points (i.e., cols and rows),
66  which we will need in order to read the file later. Also, write the
67  variable string names here. Only the master rank writes the header. ---*/
68 
69  WriteMPIBinaryData(var_buf, var_buf_size*sizeof(int), MASTER_NODE);
70 
71  /*--- Write the variable names to the file. Note that we are adopting a
72  fixed length of 33 for the string length to match with CGNS. This is
73  needed for when we read the strings later. ---*/
74 
75  for (iVar = 0; iVar < nVar; iVar++) {
76  strncpy(str_buf, fieldNames[iVar].c_str(), CGNS_STRING_SIZE);
77  WriteMPIBinaryData(str_buf, CGNS_STRING_SIZE*sizeof(char), MASTER_NODE);
78  }
79 
80  /*--- Compute various data sizes --- */
81 
82  unsigned long sizeInBytesPerPoint = sizeof(passivedouble)*nVar;
83  unsigned long sizeInBytesLocal = sizeInBytesPerPoint*nParallel_Poin;
84  unsigned long sizeInBytesGlobal = sizeInBytesPerPoint*nPoint_Global;
85  unsigned long offsetInBytes = sizeInBytesPerPoint*dataSorter->GetnPointCumulative(rank);
86 
87  /*--- Collectively write the actual data to file ---*/
88 
89  WriteMPIBinaryDataAll(dataSorter->GetData(), sizeInBytesLocal, sizeInBytesGlobal, offsetInBytes);
90 
91  /*--- Close the file ---*/
92 
93  CloseMPIFile();
94 
95 }
