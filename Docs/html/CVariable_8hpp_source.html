121  void RegisterContainer(bool input, su2activematrix& variable, su2matrix<int>* ad_index = nullptr) {
122  const auto nPoint = variable.rows();
123  SU2_OMP_FOR_STAT(roundUpDiv(nPoint,omp_get_num_threads()))
124  for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint) {
125  for(unsigned long iVar=0; iVar<variable.cols(); ++iVar) {
126 
127  if (input) AD::RegisterInput(variable(iPoint,iVar));
128  else AD::RegisterOutput(variable(iPoint,iVar));
129 
130  if (ad_index) AD::SetIndex((*ad_index)(iPoint,iVar), variable(iPoint,iVar));
131  }
132  }
133  END_SU2_OMP_FOR
134  }
