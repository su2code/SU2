32 CSU2ASCIIMeshReaderFVM::CSU2ASCIIMeshReaderFVM(CConfig *val_config,
33  unsigned short val_iZone,
34  unsigned short val_nZone)
35 : CMeshReaderFVM(val_config, val_iZone, val_nZone),
36  myZone(val_iZone),
37  nZones(val_nZone),
38  meshFilename(config->GetMesh_FileName()) {
39 
40  actuator_disk = (((config->GetnMarker_ActDiskInlet() != 0) ||
41  (config->GetnMarker_ActDiskOutlet() != 0)) &&
42  ((config->GetKind_SU2() == SU2_COMPONENT::SU2_CFD) ||
43  ((config->GetKind_SU2() == SU2_COMPONENT::SU2_DEF) &&
44  (config->GetActDisk_SU2_DEF()))));
45  if (config->GetActDisk_DoubleSurface()) actuator_disk = false;
46 
47  /* Read the basic metadata and perform some basic error checks. */
48  const auto try_single_pass = !actuator_disk;
49 
50  if (ReadMetadata(try_single_pass, val_config)) {
51  /* The file contents were read together with the metadata. */
52  return;
53  }
54 
55  /* If the mesh contains an actuator disk as a single surface,
56  we need to first split the surface into repeated points and update
57  the connectivity for each element touching the surface. */
58  if (actuator_disk)
59  SplitActuatorDiskSurface();
60 
61  /* Read and store the points, interior elements, and surface elements.
62  We store only the points and interior elements on our rank's linear
63  partition, but the master stores the entire set of surface connectivity. */
64 
65  mesh_file.open(meshFilename);
66  FastForwardToMyZone();
67 
68  for (auto section : SectionOrder) {
69  switch (section) {
70  case FileSection::ELEMENTS:
71  ReadVolumeElementConnectivity();
72  break;
73  case FileSection::POINTS:
74  ReadPointCoordinates();
75  break;
76  case FileSection::MARKERS:
77  ReadSurfaceElementConnectivity();
78  break;
79  }
80  }
81  mesh_file.close();
82 }

84 bool CSU2ASCIIMeshReaderFVM::ReadMetadata(const bool single_pass, CConfig *config) {
85 
86  const bool harmonic_balance = config->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE;
87  const bool multizone_file = config->GetMultizone_Mesh();
88 
89  /*--- Open grid file ---*/
90 
91  mesh_file.open(meshFilename);
92  if (mesh_file.fail()) {
93  SU2_MPI::Error("Error opening SU2 ASCII grid.\n"
94  "Check that the file exists.", CURRENT_FUNCTION);
95  }
96 
97  /*--- If more than one, find the curent zone in the mesh file. ---*/
98 
99  string text_line;
100  if ((nZones > 1 && multizone_file) || harmonic_balance) {
101  if (harmonic_balance) {
102  if (rank == MASTER_NODE) cout << "Reading time instance " << config->GetiInst()+1 << "." << endl;
103  }
104  else {
105  bool foundZone = false;
106  while (getline (mesh_file,text_line)) {
107  /*--- Search for the current domain ---*/
108  if (text_line.find ("IZONE=",0) != string::npos) {
109  text_line.erase (0,6);
110  unsigned short jZone = atoi(text_line.c_str());
111  if (jZone == myZone+1) {
112  if (rank == MASTER_NODE) cout << "Reading zone " << myZone << " from native SU2 ASCII mesh." << endl;
113  foundZone = true;
114  break;
115  }
116  }
117  }
118  if (!foundZone) {
119  SU2_MPI::Error("Could not find the IZONE= keyword or the zone contents.\n"
120  "Check the SU2 ASCII file format.", CURRENT_FUNCTION);
121  }
122  }
123  }
124 
125  /*--- Read the metadata: problem dimension, offsets for angle
126  of attack and angle of sideslip, global points, global elements,
127  and number of markers. Perform error checks as we go. ---*/
128 
129  bool foundNDIME = false, foundNPOIN = false;
130  bool foundNELEM = false, foundNMARK = false;
131 
132  int current_section_idx = 0;
133  bool single_pass_active = false;
134 
135  while (getline (mesh_file, text_line)) {
136 
137  /*--- Read the dimension of the problem ---*/
138 
139  if (!foundNDIME && text_line.find ("NDIME=",0) != string::npos) {
140  text_line.erase (0,6);
141  dimension = atoi(text_line.c_str());
142  foundNDIME = true;
143  continue;
144  }
145 
146  /*--- The AoA and AoS offset values are optional. ---*/
147 
148  if (text_line.find ("AOA_OFFSET=",0) != string::npos) {
149  text_line.erase (0,11);
150  su2double AoA_Offset = atof(text_line.c_str());
151 
152  /*--- The offset is in deg ---*/
153  const su2double AoA_Current = config->GetAoA() + AoA_Offset;
154  config->SetAoA_Offset(AoA_Offset);
155  config->SetAoA(AoA_Current);
156 
157  if (AoA_Offset != 0.0) {
158  if (!config->GetDiscard_InFiles()) {
159  cout << "WARNING: AoA in the config file (" << config->GetAoA() << " deg.) +\n";
160  cout << " AoA offset in mesh file (" << AoA_Offset << " deg.) = " << AoA_Current << " deg." << endl;
161  }
162  else {
163  cout << "WARNING: Discarding the AoA offset in the mesh file." << endl;
164  }
165  }
166  continue;
167  }
168 
169  if (text_line.find ("AOS_OFFSET=",0) != string::npos) {
170  text_line.erase (0,11);
171  su2double AoS_Offset = atof(text_line.c_str());
172 
173  /*--- The offset is in deg ---*/
174  const su2double AoS_Current = config->GetAoS() + AoS_Offset;
175  config->SetAoS_Offset(AoS_Offset);
176  config->SetAoS(AoS_Current);
177 
178  if (AoS_Offset != 0.0) {
179  if (!config->GetDiscard_InFiles()) {
180  cout << "WARNING: AoS in the config file (" << config->GetAoS() << " deg.) +\n";
181  cout << " AoS offset in mesh file (" << AoS_Offset << " deg.) = " << AoS_Current << " deg." << endl;
182  }
183  else {
184  cout << "WARNING: Discarding the AoS offset in the mesh file." << endl;
185  }
186  }
187  continue;
188  }
189 
190  if (!foundNPOIN && text_line.find ("NPOIN=",0) != string::npos) {
191  text_line.erase (0,6);
192  numberOfGlobalPoints = atoi(text_line.c_str());
193 
194  /* If the points were found first, read them, otherwise just consume the lines. */
195  if (single_pass && foundNDIME && current_section_idx == 0) {
196  single_pass_active = true;
197  ReadPointCoordinates(true);
198  }
199  else {
200  for (auto iPoint = 0ul; iPoint < numberOfGlobalPoints; iPoint++)
201  getline (mesh_file, text_line);
202  }
203  SectionOrder[current_section_idx++] = FileSection::POINTS;
204  foundNPOIN = true;
205  continue;
206  }
207 
208  if (!foundNELEM && text_line.find ("NELEM=",0) != string::npos) {
209  text_line.erase (0,6);
210  numberOfGlobalElements = atoi(text_line.c_str());
211 
212  if (single_pass_active) {
213  ReadVolumeElementConnectivity(true);
214  }
215  else {
216  for (auto iElem = 0ul; iElem < numberOfGlobalElements; iElem++)
217  getline (mesh_file, text_line);
218  }
219  SectionOrder[current_section_idx++] = FileSection::ELEMENTS;
220  foundNELEM = true;
221  continue;
222  }
223 
224  if (!foundNMARK && text_line.find ("NMARK=",0) != string::npos) {
225  text_line.erase (0,6);
226  numberOfMarkers = atoi(text_line.c_str());
227 
228  if (current_section_idx != 2) {
229  SU2_MPI::Error("Markers must be listed after points and elements in the SU2 mesh file.", CURRENT_FUNCTION);
230  }
231 
232  if (single_pass_active) ReadSurfaceElementConnectivity(true);
233 
234  SectionOrder[current_section_idx++] = FileSection::MARKERS;
235  foundNMARK = true;
236  continue;
237  }
238 
239  /* Stop before we reach the next zone then check for errors below. */
240  if (text_line.find ("IZONE=",0) != string::npos) {
241  break;
242  }
243  }
244 
245  mesh_file.close();
246 
247  /* Throw an error if any of the keywords was not found. */
248  if (!foundNDIME) {
249  SU2_MPI::Error("Could not find the keyword \"NDIME=\".\n"
250  "Check the SU2 ASCII file format.", CURRENT_FUNCTION);
251  }
252  if (!foundNPOIN) {
253  SU2_MPI::Error("Could not find the keyword \"NPOIN=\".\n"
254  "Check the SU2 ASCII file format.", CURRENT_FUNCTION);
255  }
256  if (!foundNELEM) {
257  SU2_MPI::Error("Could not find the keyword \"NELEM=\".\n"
258  "Check the SU2 ASCII file format.", CURRENT_FUNCTION);
259  }
260  if (!foundNMARK) {
261  SU2_MPI::Error("Could not find the keyword \"NMARK=\".\n"
262  "Check the SU2 ASCII file format.", CURRENT_FUNCTION);
263  }
264 
265  return single_pass_active;
266 }

268 void CSU2ASCIIMeshReaderFVM::SplitActuatorDiskSurface() {
269 
270  /*--- Actuator disk preprocesing ---*/
271 
272  bool InElem, Perimeter;
273  unsigned long Counter = 0;
274  Xloc = 0.0; Yloc = 0.0; Zloc = 0.0;
275  unsigned long nElem_Bound_;
276 
277  vector<unsigned long long> EdgeBegin, EdgeEnd;
278 
279  unsigned long AuxEdge, iEdge, jEdge, nEdges, nPointVolume;
280  unsigned long long FirstEdgeIndex, SecondEdgeIndex;
281 
282  vector<unsigned long> connectivity(N_POINTS_HEXAHEDRON,0);
283  vector<unsigned long> ActDiskPoint_Front_Inv(numberOfGlobalPoints);
284  vector<unsigned long> ActDiskPoint_Front;
285  vector<unsigned long> VolumePoint;
286  vector<unsigned long> PerimeterPoint;
287 
288  ActDiskNewPoints = 0;
289 
290  /* Note this routine is hard-coded to split for only one actuator disk
291  boundary. Throw an error otherwise. */
292  if (config->GetnMarker_ActDiskInlet() > 1) {
293  SU2_MPI::Error(string("Current implementation can only split a single actuator disk.") +
294  string(" \n Remove disks or re-export your mesh with double surfaces (repeated points)."),
295  CURRENT_FUNCTION);
296  }
297 
298  /*--- Open grid file ---*/
299 
300  mesh_file.open(meshFilename);
301  FastForwardToMyZone();
302 
303  /*--- Read grid file with format SU2 ---*/
304 
305  string text_line;
306  string::size_type position;
307  while (getline (mesh_file, text_line)) {
308 
309  position = text_line.find ("NMARK=",0);
310  if (position != string::npos) {
311 
312  for (unsigned short iMarker = 0 ; iMarker < numberOfMarkers; iMarker++) {
313 
314  getline (mesh_file, text_line);
315  text_line.erase (0,11); string::size_type position;
316  for (unsigned short iChar = 0; iChar < 20; iChar++) {
317  position = text_line.find( " ", 0 );
318  if (position != string::npos) text_line.erase (position,1);
319  position = text_line.find( "\r", 0 );
320  if (position != string::npos) text_line.erase (position,1);
321  position = text_line.find( "\n", 0 );
322  if (position != string::npos) text_line.erase (position,1);
323  }
324  string Marker_Tag = text_line.c_str();
325 
326  getline (mesh_file, text_line);
327  text_line.erase (0,13);
328  nElem_Bound_ = atoi(text_line.c_str());
329 
330  if (Marker_Tag != config->GetMarker_ActDiskInlet_TagBound(0)) {
331  for (unsigned long iElem_Bound = 0; iElem_Bound < nElem_Bound_; iElem_Bound++) { getline (mesh_file, text_line); }
332  }
333  else {
334 
335  if (rank == MASTER_NODE)
336  cout << "Splitting the surface " << Marker_Tag << "( " << nElem_Bound_ << " boundary elements )." << endl;
337 
338  /*--- Create a list of edges ---*/
339 
340  for (unsigned long iElem_Bound = 0; iElem_Bound < nElem_Bound_; iElem_Bound++) {
341 
342  getline(mesh_file, text_line);
343 
344  unsigned short VTK_Type;
345  istringstream bound_line(text_line);
346  bound_line >> VTK_Type;
347 
348  switch(VTK_Type) {
349  case LINE:
350  bound_line >> connectivity[0];
351  bound_line >> connectivity[1];
352 
353  EdgeBegin.push_back(connectivity[0]); EdgeEnd.push_back(connectivity[1]);
354  break;
355  case TRIANGLE:
356  bound_line >> connectivity[0];
357  bound_line >> connectivity[1];
358  bound_line >> connectivity[2];
359  EdgeBegin.push_back(connectivity[0]); EdgeEnd.push_back(connectivity[1]);
360  EdgeBegin.push_back(connectivity[1]); EdgeEnd.push_back(connectivity[2]);
361  EdgeBegin.push_back(connectivity[2]); EdgeEnd.push_back(connectivity[0]);
362  break;
363  case QUADRILATERAL:
364  bound_line >> connectivity[0];
365  bound_line >> connectivity[1];
366  bound_line >> connectivity[2];
367  bound_line >> connectivity[3];
368  EdgeBegin.push_back(connectivity[0]); EdgeEnd.push_back(connectivity[1]);
369  EdgeBegin.push_back(connectivity[1]); EdgeEnd.push_back(connectivity[2]);
370  EdgeBegin.push_back(connectivity[2]); EdgeEnd.push_back(connectivity[3]);
371  EdgeBegin.push_back(connectivity[3]); EdgeEnd.push_back(connectivity[0]);
372  break;
373  }
374 
375  }
376 
377  /*--- Set the total number of edges ---*/
378 
379  nEdges = EdgeBegin.size();
380 
381  /*--- Sort edges based on local point index, first index is always the largest ---*/
382 
383  for (iEdge = 0; iEdge < nEdges; iEdge++) {
384  if (EdgeEnd[iEdge] < EdgeBegin[iEdge]) {
385  AuxEdge = EdgeEnd[iEdge]; EdgeEnd[iEdge] = EdgeBegin[iEdge]; EdgeBegin[iEdge] = AuxEdge;
386  }
387  }
388 
389  /*--- Bubble sort of the points based on the first index ---*/
390 
391  for (iEdge = 0; iEdge < nEdges; iEdge++) {
392  for (jEdge = iEdge+1; jEdge < nEdges; jEdge++) {
393 
394  FirstEdgeIndex = EdgeBegin[jEdge] << 31;
395  FirstEdgeIndex += EdgeEnd[jEdge];
396 
397  SecondEdgeIndex = EdgeBegin[iEdge] << 31;
398  SecondEdgeIndex += EdgeEnd[iEdge];
399 
400  if (FirstEdgeIndex <= SecondEdgeIndex) {
401  AuxEdge = EdgeBegin[iEdge]; EdgeBegin[iEdge] = EdgeBegin[jEdge]; EdgeBegin[jEdge] = AuxEdge;
402  AuxEdge = EdgeEnd[iEdge]; EdgeEnd[iEdge] = EdgeEnd[jEdge]; EdgeEnd[jEdge] = AuxEdge;
403  }
404  }
405  }
406 
407  if (dimension == 3) {
408 
409  /*--- Check the begning of the list ---*/
410 
411  if (!((EdgeBegin[0] == EdgeBegin[1]) && (EdgeEnd[0] == EdgeEnd[1]))) {
412  PerimeterPoint.push_back(EdgeBegin[0]);
413  PerimeterPoint.push_back(EdgeEnd[0]);
414  }
415 
416  for (iEdge = 1; iEdge < nEdges-1; iEdge++) {
417  bool Check_1 = !((EdgeBegin[iEdge] == EdgeBegin[iEdge-1]) && (EdgeEnd[iEdge] == EdgeEnd[iEdge-1]));
418  bool Check_2 = !((EdgeBegin[iEdge] == EdgeBegin[iEdge+1]) && (EdgeEnd[iEdge] == EdgeEnd[iEdge+1]));
419  if ((Check_1 && Check_2)) {
420  PerimeterPoint.push_back(EdgeBegin[iEdge]);
421  PerimeterPoint.push_back(EdgeEnd[iEdge]);
422  }
423  }
424 
425  /*--- Check the end of the list ---*/
426 
427  if (!((EdgeBegin[nEdges-1] == EdgeBegin[nEdges-2]) && (EdgeEnd[nEdges-1] == EdgeEnd[nEdges-2]))) {
428  PerimeterPoint.push_back(EdgeBegin[nEdges-1]);
429  PerimeterPoint.push_back(EdgeEnd[nEdges-1]);
430  }
431 
432  } else {
433 
434 
435  /*--- Create a list with all the points ---*/
436 
437  for (iEdge = 0; iEdge < nEdges; iEdge++) {
438  ActDiskPoint_Front.push_back(EdgeBegin[iEdge]);
439  ActDiskPoint_Front.push_back(EdgeEnd[iEdge]);
440  }
441 
442  sort(ActDiskPoint_Front.begin(), ActDiskPoint_Front.end());
443  vector<unsigned long>::iterator it = unique(ActDiskPoint_Front.begin(), ActDiskPoint_Front.end());
444  ActDiskPoint_Front.resize(it - ActDiskPoint_Front.begin());
445 
446  /*--- Check the begning of the list ---*/
447 
448  if (!(ActDiskPoint_Front[0] == ActDiskPoint_Front[1]) ) { PerimeterPoint.push_back(ActDiskPoint_Front[0]); }
449 
450  for (unsigned long iPoint = 1; iPoint < ActDiskPoint_Front.size()-1; iPoint++) {
451  bool Check_1 = !((ActDiskPoint_Front[iPoint] == ActDiskPoint_Front[iPoint-1]) );
452  bool Check_2 = !((ActDiskPoint_Front[iPoint] == ActDiskPoint_Front[iPoint+1]) );
453  if ((Check_1 && Check_2)) { PerimeterPoint.push_back(ActDiskPoint_Front[iEdge]); }
454  }
455 
456  /*--- Check the end of the list ---*/
457 
458  if (!((EdgeBegin[ActDiskPoint_Front.size()-1] == EdgeBegin[ActDiskPoint_Front.size()-2]) )) {
459  PerimeterPoint.push_back(ActDiskPoint_Front[ActDiskPoint_Front.size()-1]);
460  }
461 
462  ActDiskPoint_Front.clear();
463 
464  }
465 
466  vector<unsigned long>::iterator it;
467  sort(PerimeterPoint.begin(), PerimeterPoint.end());
468  it = unique(PerimeterPoint.begin(), PerimeterPoint.end());
469  PerimeterPoint.resize(it - PerimeterPoint.begin());
470 
471  for (iEdge = 0; iEdge < nEdges; iEdge++) {
472 
473  Perimeter = false;
474  for (unsigned long iPoint = 0; iPoint < PerimeterPoint.size(); iPoint++) {
475  if (EdgeBegin[iEdge] == PerimeterPoint[iPoint]) {
476  Perimeter = true; break;
477  }
478  }
479 
480  if (!Perimeter) ActDiskPoint_Front.push_back(EdgeBegin[iEdge]);
481 
482  Perimeter = false;
483  for (unsigned long iPoint = 0; iPoint < PerimeterPoint.size(); iPoint++) {
484  if (EdgeEnd[iEdge] == PerimeterPoint[iPoint]) {
485  Perimeter = true; break;
486  }
487  }
488 
489  if (!Perimeter) ActDiskPoint_Front.push_back(EdgeEnd[iEdge]);
490 
491  }
492 
493  /*--- Sort, and remove repeated points from the disk list of points ---*/
494 
495  sort(ActDiskPoint_Front.begin(), ActDiskPoint_Front.end());
496  it = unique(ActDiskPoint_Front.begin(), ActDiskPoint_Front.end());
497  ActDiskPoint_Front.resize(it - ActDiskPoint_Front.begin());
498  ActDiskNewPoints = ActDiskPoint_Front.size();
499 
500  if (rank == MASTER_NODE)
501  cout << "Splitting the surface " << Marker_Tag << "( " << ActDiskPoint_Front.size() << " internal points )." << endl;
502 
503  /*--- Create a map from original point to the new ones (back plane) ---*/
504 
505  ActDiskPoint_Back.resize(numberOfGlobalPoints);
506  ActDisk_Bool.resize(numberOfGlobalPoints);
507 
508  for (unsigned long iPoint = 0; iPoint < numberOfGlobalPoints; iPoint++) {
509  ActDisk_Bool[iPoint] = false;
510  ActDiskPoint_Back[iPoint] = 0;
511  }
512 
513  unsigned long kPoint = numberOfGlobalPoints;
514  for (unsigned long iPoint = 0; iPoint < ActDiskPoint_Front.size(); iPoint++) {
515  ActDiskPoint_Front_Inv[ActDiskPoint_Front[iPoint]] = iPoint;
516  ActDisk_Bool[ActDiskPoint_Front[iPoint]] = true;
517  ActDiskPoint_Back[ActDiskPoint_Front[iPoint]] = kPoint;
518  kPoint++;
519  }
520 
521  }
522  }
523  break;
524  }
525  }
526 
527  mesh_file.close();
528 
529  /*--- Store the coordinates of the new points ---*/
530 
531  CoordXActDisk.resize(ActDiskNewPoints);
532  CoordYActDisk.resize(ActDiskNewPoints);
533  CoordZActDisk.resize(ActDiskNewPoints);
534 
535  /* Open the mesh file again to read the coordinates of the new points. */
536 
537  mesh_file.open(meshFilename);
538  FastForwardToMyZone();
539 
540  while (getline (mesh_file, text_line)) {
541 
542  position = text_line.find ("NPOIN=",0);
543  if (position != string::npos) {
544  for (unsigned long iPoint = 0; iPoint < numberOfGlobalPoints; iPoint++) {
545  getline (mesh_file, text_line);
546  istringstream point_line(text_line);
547 
548  su2double Coords[3] = {0.0,0.0,0.0};
549  if (dimension == 2) {
550  point_line >> Coords[0];
551  point_line >> Coords[1];
552  } else {
553  point_line >> Coords[0];
554  point_line >> Coords[1];
555  point_line >> Coords[2];
556  }
557 
558  /*--- Compute the CG of the actuator disk surface ---*/
559 
560  if (ActDisk_Bool[iPoint]) {
561  CoordXActDisk[ActDiskPoint_Front_Inv[iPoint]] = Coords[0];
562  CoordYActDisk[ActDiskPoint_Front_Inv[iPoint]] = Coords[1];
563  Xloc += Coords[0]; Yloc += Coords[1];
564  if (dimension == 3) {
565  CoordZActDisk[ActDiskPoint_Front_Inv[iPoint]] = Coords[2];
566  Zloc += Coords[2];
567  }
568  Counter++;
569  }
570 
571  }
572  }
573 
574  /*--- Locate and tag points that touch the actuator disk surface. ---*/
575 
576  position = text_line.find ("NELEM=",0);
577  if (position != string::npos) {
578  for (unsigned long iElem = 0; iElem < numberOfGlobalElements; iElem++) {
579 
580  getline(mesh_file, text_line);
581  istringstream elem_line(text_line);
582 
583  unsigned short VTK_Type;
584  elem_line >> VTK_Type;
585 
586  switch(VTK_Type) {
587  case TRIANGLE:
588  elem_line >> connectivity[0];
589  elem_line >> connectivity[1];
590  elem_line >> connectivity[2];
591  InElem = false;
592  for (unsigned long i = 0; i < (unsigned long)N_POINTS_TRIANGLE; i++) {
593  if (ActDisk_Bool[connectivity[i]]) { InElem = true; break; } }
594  if (InElem) {
595  for (unsigned long i = 0; i < (unsigned long)N_POINTS_TRIANGLE; i++) {
596  VolumePoint.push_back(connectivity[i]); } }
597  break;
598  case QUADRILATERAL:
599  elem_line >> connectivity[0];
600  elem_line >> connectivity[1];
601  elem_line >> connectivity[2];
602  elem_line >> connectivity[3];
603  InElem = false;
604  for (unsigned long i = 0; i < (unsigned long)N_POINTS_QUADRILATERAL; i++) {
605  if (ActDisk_Bool[connectivity[i]]) { InElem = true; break; } }
606  if (InElem) {
607  for (unsigned long i = 0; i < (unsigned long)N_POINTS_QUADRILATERAL; i++) {
608  VolumePoint.push_back(connectivity[i]); } }
609  break;
610  case TETRAHEDRON:
611  elem_line >> connectivity[0];
612  elem_line >> connectivity[1];
613  elem_line >> connectivity[2];
614  elem_line >> connectivity[3];
615  InElem = false;

679

688  CoordXVolumePoint.resize(nPointVolume);
689  CoordYVolumePoint.resize(nPointVolume);
690  CoordZVolumePoint.resize(nPointVolume);
691  VolumePoint_Inv.resize(numberOfGlobalPoints);

694  for (unsigned long iPoint = 0; iPoint < numberOfGlobalPoints; iPoint++) {
695  MapVolumePointBool[iPoint] = false;
696  }
697 
698  for (unsigned long iPoint = 0; iPoint < nPointVolume; iPoint++) {
699  MapVolumePointBool[VolumePoint[iPoint]] = true;
700  VolumePoint_Inv[VolumePoint[iPoint]] = iPoint;
701  }

704  points that touch the actuator disk ---*/

713  getline (mesh_file, text_line);

717  point_line >> Coords[0];
718  point_line >> Coords[1];

721  point_line >> Coords[1];

748 
749  /* Get a partitioner to help with linear partitioning. */
750  CLinearPartitioner pointPartitioner(numberOfGlobalPoints,0);
751 
752  /* Determine number of local points */
753  numberOfLocalPoints = pointPartitioner.GetSizeOnRank(rank);
754 
755  /* Prepare our data structure for the point coordinates. */
756  localPointCoordinates.resize(dimension);
757  for (int k = 0; k < dimension; k++)
758  localPointCoordinates[k].reserve(numberOfLocalPoints);
759 
760  /*--- Read the point coordinates into our data structure. ---*/
761 
762  while (true) {
763  string text_line;
764  if (!single_pass) {
765  getline(mesh_file, text_line);
766  if (text_line.find("NPOIN=",0) == string::npos) continue;
767  }
768 
769  for (unsigned long GlobalIndex = 0; GlobalIndex < numberOfGlobalPoints; ++GlobalIndex) {
770 
771  if (!actuator_disk) {
772  getline(mesh_file, text_line);
773  }
774  else {
775  if (GlobalIndex < numberOfGlobalPoints-ActDiskNewPoints) {
776  getline(mesh_file, text_line);
777  }
778  else {
779  /* This is a new actuator disk point, so we must construct a
780  string with the new point's coordinates. */
781  ostringstream strsX, strsY, strsZ;
782  unsigned long BackActDisk_Index = GlobalIndex;
783  unsigned long LocalIndex = BackActDisk_Index - (numberOfGlobalPoints-ActDiskNewPoints);
784  strsX.precision(20); strsY.precision(20); strsZ.precision(20);
785  su2double CoordX = CoordXActDisk[LocalIndex]; strsX << scientific << CoordX;
786  su2double CoordY = CoordYActDisk[LocalIndex]; strsY << scientific << CoordY;
787  su2double CoordZ = CoordZActDisk[LocalIndex]; strsZ << scientific << CoordZ;
788  text_line = strsX.str() + "\t" + strsY.str() + "\t" + strsZ.str();
789  }
790  }
791 
792  /*--- We only read information for this node if it is owned by this
793  rank based upon our initial linear partitioning. ---*/
794 
795  passivedouble Coords[3] = {0.0,0.0,0.0};
796  if (pointPartitioner.IndexBelongsToRank(GlobalIndex, rank)) {
797 
798  istringstream point_line(text_line);
799 
800  /* Store the coordinates more clearly. */
801  point_line >> Coords[0];
802  point_line >> Coords[1];
803  if (dimension == 3) {
804  point_line >> Coords[2];
805  }
806 
807  /* Load into the coordinate class data structure. */
808  for (unsigned short iDim = 0; iDim < dimension; iDim++) {
809  localPointCoordinates[iDim].push_back(Coords[iDim]);
810  }
811  }
812  }
813  break;

817 void CSU2ASCIIMeshReaderFVM::ReadVolumeElementConnectivity(const bool single_pass) {
818 
819  /* Get a partitioner to help with linear partitioning. */
820  CLinearPartitioner pointPartitioner(numberOfGlobalPoints,0);
821 
822  /* Loop over our analytically defined of elements and store only those
823  that contain a node within our linear partition of points. */
824  numberOfLocalElements = 0;
825  array<unsigned long, N_POINTS_HEXAHEDRON> connectivity{};
826 
827  while (true) {
828  string text_line;
829  if (!single_pass) {
830  if (!getline(mesh_file, text_line)) break;
831  if (text_line.find("NELEM=",0) == string::npos) continue;
832  }
833 
834  /*--- Loop over all the volumetric elements and store any element that
835  contains at least one of an owned node for this rank (i.e., there will
836  be element redundancy, since multiple ranks will store the same elems
837  on the boundaries of the initial linear partitioning. ---*/
838 
839  numberOfLocalElements = 0;
840 
841  for (unsigned long GlobalIndex = 0; GlobalIndex < numberOfGlobalElements; ++GlobalIndex) {
842  getline(mesh_file, text_line);
843  istringstream elem_line(text_line);
844 
845  /*--- Decide whether this rank needs each element. ---*/
846 
847  unsigned short VTK_Type;
848  elem_line >> VTK_Type;
849 
850  const auto nPointsElem = nPointsOfElementType(VTK_Type);
851 
852  for (unsigned short i = 0; i < nPointsElem; i++) {
853  elem_line >> connectivity[i];
854  }
855 
856  if (actuator_disk) {
857  for (unsigned short i = 0; i<nPointsElem; i++) {
858  if (ActDisk_Bool[connectivity[i]]) {
859 
860  su2double Xcg = 0.0; unsigned long Counter = 0;
861  for (unsigned short j = 0; j<nPointsElem; j++) {
862  if (connectivity[j] < numberOfGlobalPoints-ActDiskNewPoints) {
863  Xcg += CoordXVolumePoint[VolumePoint_Inv[connectivity[j]]];
864  Counter++;
865  }
866  }
867  Xcg = Xcg / su2double(Counter);
868 
869  if (Counter != 0 && Xcg > Xloc) {
870  connectivity[i] = ActDiskPoint_Back[connectivity[i]];
871  }
872  }
873  }
874  }
875 
876  /* Check whether any of the points reside in our linear partition. */
877  bool isOwned = false;
878  for (unsigned short i = 0; i < nPointsElem; i++) {
879  if (pointPartitioner.IndexBelongsToRank(connectivity[i], rank)) {
880  isOwned = true;
881  break;
882  }
883  }
884 
885  /* If element is owned, we need to store it locally. */
886  if (isOwned) {
887  localVolumeElementConnectivity.push_back(GlobalIndex);
888  localVolumeElementConnectivity.push_back(VTK_Type);
889  /// TODO: Use a compressed format.
890  for (unsigned short i = 0; i < N_POINTS_HEXAHEDRON; i++) {
891  localVolumeElementConnectivity.push_back(connectivity[i]);
892  }
893  numberOfLocalElements++;
894  }
895  }
896  break;
897  }

900 void CSU2ASCIIMeshReaderFVM::ReadSurfaceElementConnectivity(const bool single_pass) {
901 
902  /* We already read in the number of markers with the metadata. */
903  surfaceElementConnectivity.resize(numberOfMarkers);
904  markerNames.resize(numberOfMarkers);
905 
906  array<unsigned long, N_POINTS_HEXAHEDRON> connectivity{};
907 
908  /*--- In this routine, the boundary info is read by all ranks,
909  however, the surface connectivity is still handled by the
910  master node (and eventually distributed by the master as well). ---*/
911 
912  while (true) {
913  string text_line;
914  if (!single_pass) {
915  if (!getline(mesh_file, text_line)) break;
916  if (text_line.find("NMARK=",0) == string::npos) continue;
917  }
918 
919  for (unsigned short iMarker = 0; iMarker < numberOfMarkers; ++iMarker) {
920  getline (mesh_file, text_line);
921  text_line.erase (0,11);
922  string::size_type position;
923 
924  for (unsigned short iChar = 0; iChar < 20; iChar++) {
925  position = text_line.find( " ", 0 );
926  if (position != string::npos) text_line.erase (position,1);
927  position = text_line.find( "\r", 0 );
928  if (position != string::npos) text_line.erase (position,1);
929  position = text_line.find( "\n", 0 );
930  if (position != string::npos) text_line.erase (position,1);
931  }
932  markerNames[iMarker] = text_line;
933 
934  bool duplicate = false;
935  if ((actuator_disk) &&
936  (markerNames[iMarker] == config->GetMarker_ActDiskInlet_TagBound(0))) {
937  duplicate = true;
938  markerNames[iMarker+1] = config->GetMarker_ActDiskOutlet_TagBound(0);
939  }
940 
941  /*--- Physical boundaries definition ---*/
942 
943  if (markerNames[iMarker] == "SEND_RECEIVE") {
944  /*--- Throw an error if we find deprecated references to SEND_RECEIVE
945  boundaries in the mesh. ---*/
946  SU2_MPI::Error("Mesh file contains deprecated SEND_RECEIVE marker!\n"
947  "Please remove any SEND_RECEIVE markers from the SU2 ASCII mesh.",
948  CURRENT_FUNCTION);
949  }
950 
951  getline (mesh_file, text_line);
952  text_line.erase (0,13);
953  unsigned long nElem_Bound = atoi(text_line.c_str());
954 
955  /*--- Allocate space for elements ---*/
956 
957  for (unsigned long iElem_Bound = 0; iElem_Bound < nElem_Bound; iElem_Bound++) {
958  getline(mesh_file, text_line);
959  istringstream bound_line(text_line);
960 
961  unsigned short VTK_Type;
962  bound_line >> VTK_Type;
963 
964  const auto nPointsElem = nPointsOfElementType(VTK_Type);
965 
966  if (dimension == 3 && VTK_Type == LINE) {
967  SU2_MPI::Error("Line boundary conditions are not possible for 3D calculations.\n"
968  "Please check the SU2 ASCII mesh file.", CURRENT_FUNCTION);
969  }
970 
971  for (unsigned short i = 0; i < nPointsElem; i++) {
972  bound_line >> connectivity[i];
973  }
974 
975  surfaceElementConnectivity[iMarker].push_back(0);
976  surfaceElementConnectivity[iMarker].push_back(VTK_Type);
977  for (unsigned short i = 0; i < N_POINTS_HEXAHEDRON; i++) {
978  surfaceElementConnectivity[iMarker].push_back(connectivity[i]);
979  }
980 
981  if (duplicate) {
982  for (unsigned short i = 0; i < nPointsElem; i++) {
983  if (ActDisk_Bool[connectivity[i]]) {
984  connectivity[i] = ActDiskPoint_Back[connectivity[i]];
985  }
986  }
987  surfaceElementConnectivity[iMarker + 1].push_back(0);
988  surfaceElementConnectivity[iMarker + 1].push_back(VTK_Type);
989  for (unsigned short i = 0; i < N_POINTS_HEXAHEDRON; i++) {
990  surfaceElementConnectivity[iMarker + 1].push_back(connectivity[i]);
991  }
992  }
993  }
994  /*--- Increment the counter an extra time if we stored a duplicate. ---*/
995  iMarker += duplicate;
996  }
997  break;
998  }
999 
1000  if (rank != MASTER_NODE) return;
1001 
1002  /*--- Final error check for deprecated periodic BC format. ---*/
1003 
1004  string text_line;
1005  while (getline (mesh_file, text_line)) {
1006 
1007  /*--- Find any periodic transformation information. ---*/
1008 
1009  if (text_line.find ("NPERIODIC=",0) != string::npos) {
1010 
1011  /*--- Read and store the number of transformations. ---*/
1012  text_line.erase(0,10);
1013  unsigned short nPeriodic = atoi(text_line.c_str());
1014  if (nPeriodic - 1 != 0) {
1015  SU2_MPI::Error("Mesh file contains deprecated periodic format!\n\n"
1016  "For SU2 v7.0.0 and later, preprocessing of periodic grids by SU2_MSH\n"
1017  "is no longer necessary. Please use the original mesh file (prior to SU2_MSH)\n"
1018  "with the same MARKER_PERIODIC definition in the configuration file.", CURRENT_FUNCTION);
1019  }
1020  }
1021 
1022  /*--- Stop before we reach the next zone. ---*/
1023  if (text_line.find ("IZONE=",0) != string::npos) break;
1024  }
