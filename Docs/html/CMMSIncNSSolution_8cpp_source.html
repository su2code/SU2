32 CMMSIncNSSolution::CMMSIncNSSolution(unsigned short val_nDim,
33  unsigned short val_nVar,
34  unsigned short val_iMesh,
35  CConfig* config)
36 : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
37 
38  /*--- Write a message that the solution is initialized for the manufactured
39  solution for the incompressible Navier-Stokes equations. ---*/
40  if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
41  cout << endl;
42  cout << "Warning: Fluid properties and solution are being " << endl;
43  cout << " initialized for the manufactured solution " << endl;
44  cout << " of the incompressible Navier-Stokes equations!!!" << endl;
45  cout << endl << flush;
46  }
47 
48  /*--- Coefficients, needed to determine the solution. ---*/
49  Viscosity = config->GetViscosity_FreeStreamND();
50  Density = config->GetDensity_FreeStreamND();
51  Temperature = config->GetTemperature_FreeStreamND();
52 
53  /*--- Constants, which describe this manufactured solution. This is a
54  viscous solution where the primitive variables vary as a combination
55  of sine and cosine functions. The solution is from Salari K, and
56  Knupp P, "Code verification by the method of manufactured solutions,"
57  SAND 2000-1444, Sandia National Laboratories, Albuquerque, NM, 2000. ---*/
58 
59  P_0 = 1.0;
60  u_0 = 1.0;
61  v_0 = 1.0;
62  epsilon = 0.001;
63 
64  /*--- Perform some sanity and error checks for this solution here. ---*/
65  if(config->GetTime_Marching() != TIME_MARCHING::STEADY)
66  SU2_MPI::Error("Steady mode must be selected for the MMS incompressible NS case",
67  CURRENT_FUNCTION);
68 
69  if(Kind_Solver != MAIN_SOLVER::INC_EULER && Kind_Solver != MAIN_SOLVER::INC_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::INC_RANS )
70  SU2_MPI::Error("Incompressible flow equations must be selected for the MMS incompressible NS case",
71  CURRENT_FUNCTION);
72 
73  if(Kind_Solver != MAIN_SOLVER::INC_NAVIER_STOKES)
74  SU2_MPI::Error("Navier Stokes equations must be selected for the MMS incompressible NS case",
75  CURRENT_FUNCTION);
76 
77  if(config->GetKind_FluidModel() != CONSTANT_DENSITY)
78  SU2_MPI::Error("Constant density fluid model must be selected for the MMS incompressible NS case",
79  CURRENT_FUNCTION);
80 
81  if(config->GetKind_ViscosityModel() != VISCOSITYMODEL::CONSTANT)
82  SU2_MPI::Error("Constant viscosity must be selected for the MMS incompressible NS case",
83  CURRENT_FUNCTION);
84 
85  if(config->GetEnergy_Equation())
86  SU2_MPI::Error("Energy equation must be disabled (isothermal) for the MMS incompressible NS case",
87  CURRENT_FUNCTION);
88 }

100 void CMMSIncNSSolution::GetSolution(const su2double *val_coords,
101  const su2double val_t,
102  su2double *val_solution) const {
103 
104  /* Easier storage of the x- and y-coordinates. */
105  const su2double x = val_coords[0];
106  const su2double y = val_coords[1];
107 
108  /* Compute the primitives from the defined solution. */
109  const su2double u = u_0*(sin(x*x + y*y) + epsilon);
110  const su2double v = v_0*(cos(x*x + y*y) + epsilon);
111  const su2double p = P_0*(sin(x*x + y*y) + 2.0);
112 
113  /* For the incompressible solver, we return the primitive variables
114  directly, as they are used for the working variables in the solver.
115  Note that the implementation below is valid for both 2D and 3D. */
116  val_solution[0] = p;
117  val_solution[1] = u;
118  val_solution[2] = v;
119  val_solution[3] = 0.0;
120  val_solution[nVar-1] = Temperature;
121 
122 }

124 void CMMSIncNSSolution::GetMMSSourceTerm(const su2double *val_coords,
125  const su2double val_t,
126  su2double *val_source) const {
127 
128  /*--- Easier storage of the x- and y-coordinates. ---*/
129  const su2double x = val_coords[0];
130  const su2double y = val_coords[1];
131 
132  /*--- The expressions for the source terms are generated
133  automatically by the sympy package in python.---*/
134  val_source[0] = 2*Density*(u_0*x*cos(pow(x, 2) + pow(y, 2)) - v_0*y*sin(pow(x, 2) + pow(y, 2)));
135  val_source[1] = 4*Density*pow(u_0, 2)*x*(epsilon + sin(pow(x, 2) + pow(y, 2)))*cos(pow(x, 2) + pow(y, 2)) - 2*Density*u_0*v_0*y*(epsilon + sin(pow(x, 2) + pow(y, 2)))*sin(pow(x, 2) + pow(y, 2)) + 2*Density*u_0*v_0*y*(epsilon + cos(pow(x, 2) + pow(y, 2)))*cos(pow(x, 2) + pow(y, 2)) + 2*P_0*x*cos(pow(x, 2) + pow(y, 2)) - 0.666666666666667*Viscosity*(-8.0*u_0*pow(x, 2)*sin(pow(x, 2) + pow(y, 2)) + 4.0*u_0*cos(pow(x, 2) + pow(y, 2)) + 4*v_0*x*y*cos(pow(x, 2) + pow(y, 2))) + 2*Viscosity*(2*u_0*pow(y, 2)*sin(pow(x, 2) + pow(y, 2)) - u_0*cos(pow(x, 2) + pow(y, 2)) + 2*v_0*x*y*cos(pow(x, 2) + pow(y, 2)));
136  val_source[2] = -2*Density*u_0*v_0*x*(epsilon + sin(pow(x, 2) + pow(y, 2)))*sin(pow(x, 2) + pow(y, 2)) + 2*Density*u_0*v_0*x*(epsilon + cos(pow(x, 2) + pow(y, 2)))*cos(pow(x, 2) + pow(y, 2)) - 4*Density*pow(v_0, 2)*y*(epsilon + cos(pow(x, 2) + pow(y, 2)))*sin(pow(x, 2) + pow(y, 2)) + 2*P_0*y*cos(pow(x, 2) + pow(y, 2)) + 0.666666666666667*Viscosity*(-4*u_0*x*y*sin(pow(x, 2) + pow(y, 2)) + 8.0*v_0*pow(y, 2)*cos(pow(x, 2) + pow(y, 2)) + 4.0*v_0*sin(pow(x, 2) + pow(y, 2))) + 2*Viscosity*(2*u_0*x*y*sin(pow(x, 2) + pow(y, 2)) + 2*v_0*pow(x, 2)*cos(pow(x, 2) + pow(y, 2)) + v_0*sin(pow(x, 2) + pow(y, 2)));
137  val_source[3] = 0.0;
138  val_source[nVar-1] = 0.0;
139 
140 }
