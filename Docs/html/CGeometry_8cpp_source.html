146 void CGeometry::PreprocessP2PComms(CGeometry *geometry,
147  CConfig *config) {
148 
149  /*--- We start with the send and receive lists already available in
150  the form of SEND_RECEIVE boundary markers. We will loop through
151  these markers and establish the neighboring ranks and number of
152  send/recv points per pair. We will store this information and set
153  up persistent data structures so that we can reuse them throughout
154  the calculation for any point-to-point communications. The goal
155  is to break the non-blocking comms into InitiateComms() and
156  CompleteComms() in separate routines so that we can overlap the
157  communication and computation to hide the communication latency. ---*/
158 
159  /*--- Local variables. ---*/
160 
161  unsigned short iMarker;
162  unsigned long nVertexS, nVertexR, iVertex, MarkerS, MarkerR;
163 
164  int iRank, iSend, iRecv, count;
165 
166  /*--- Create some temporary structures for tracking sends/recvs. ---*/
167 
168  int *nPoint_Send_All = new int[size+1]; nPoint_Send_All[0] = 0;
169  int *nPoint_Recv_All = new int[size+1]; nPoint_Recv_All[0] = 0;
170  int *nPoint_Flag = new int[size];
171 
172  for (iRank = 0; iRank < size; iRank++) {
173  nPoint_Send_All[iRank] = 0; nPoint_Recv_All[iRank] = 0; nPoint_Flag[iRank]= -1;
174  }
175  nPoint_Send_All[size] = 0; nPoint_Recv_All[size] = 0;
176 
177  /*--- Loop through all of our SEND_RECEIVE markers and track
178  our sends with each rank. ---*/
179 
180  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
181  if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) &&
182  (config->GetMarker_All_SendRecv(iMarker) > 0)) {
183 
184  /*--- Get the destination rank and number of points to send. ---*/
185 
186  iRank = config->GetMarker_All_SendRecv(iMarker)-1;
187  nVertexS = geometry->nVertex[iMarker];
188 
189  /*--- If we have not visited this element yet, increment our
190  number of elements that must be sent to a particular proc. ---*/
191 
192  if ((nPoint_Flag[iRank] != (int)iMarker)) {
193  nPoint_Flag[iRank] = (int)iMarker;
194  nPoint_Send_All[iRank+1] += nVertexS;
195  }
196 
197  }
198  }
199 
200  delete [] nPoint_Flag;
201 
202  /*--- Communicate the number of points to be sent/recv'd amongst
203  all processors. After this communication, each proc knows how
204  many cells it will receive from each other processor. ---*/
205 
206  SU2_MPI::Alltoall(&(nPoint_Send_All[1]), 1, MPI_INT,
207  &(nPoint_Recv_All[1]), 1, MPI_INT, SU2_MPI::GetComm());
208 
209  /*--- Prepare to send connectivities. First check how many
210  messages we will be sending and receiving. Here we also put
211  the counters into cumulative storage format to make the
212  communications simpler. ---*/
213 
214  nP2PSend = 0; nP2PRecv = 0;
215 
216  for (iRank = 0; iRank < size; iRank++) {
217  if ((iRank != rank) && (nPoint_Send_All[iRank+1] > 0)) nP2PSend++;
218  if ((iRank != rank) && (nPoint_Recv_All[iRank+1] > 0)) nP2PRecv++;
219 
220  nPoint_Send_All[iRank+1] += nPoint_Send_All[iRank];
221  nPoint_Recv_All[iRank+1] += nPoint_Recv_All[iRank];
222  }
223 
224  /*--- Allocate only as much memory as we need for the P2P neighbors. ---*/
225 
226  nPoint_P2PSend = new int[nP2PSend+1]; nPoint_P2PSend[0] = 0;
227  nPoint_P2PRecv = new int[nP2PRecv+1]; nPoint_P2PRecv[0] = 0;
228 
229  Neighbors_P2PSend = new int[nP2PSend];
230  Neighbors_P2PRecv = new int[nP2PRecv];
231 
232  iSend = 0; iRecv = 0;
233  for (iRank = 0; iRank < size; iRank++) {
234 
235  if ((nPoint_Send_All[iRank+1] > nPoint_Send_All[iRank]) && (iRank != rank)) {
236  Neighbors_P2PSend[iSend] = iRank;
237  nPoint_P2PSend[iSend+1] = nPoint_Send_All[iRank+1];
238  iSend++;
239  }
240 
241  if ((nPoint_Recv_All[iRank+1] > nPoint_Recv_All[iRank]) && (iRank != rank)) {
242  Neighbors_P2PRecv[iRecv] = iRank;
243  nPoint_P2PRecv[iRecv+1] = nPoint_Recv_All[iRank+1];
244  iRecv++;
245  }
246 
247  }
248 
249  /*--- Create a reverse mapping of the message to the rank so that we
250  can quickly access the correct data in the buffers when receiving
251  messages dynamically. ---*/
252 
253  P2PSend2Neighbor.clear();
254  for (iSend = 0; iSend < nP2PSend; iSend++)
255  P2PSend2Neighbor[Neighbors_P2PSend[iSend]] = iSend;
256 
257  P2PRecv2Neighbor.clear();
258  for (iRecv = 0; iRecv < nP2PRecv; iRecv++)
259  P2PRecv2Neighbor[Neighbors_P2PRecv[iRecv]] = iRecv;
260 
261  delete [] nPoint_Send_All;
262  delete [] nPoint_Recv_All;
263 
264  /*--- Allocate the memory that we need for receiving the conn
265  values and then cue up the non-blocking receives. Note that
266  we do not include our own rank in the communications. We will
267  directly copy our own data later. ---*/
268 
269  Local_Point_P2PSend = nullptr;
270  Local_Point_P2PSend = new unsigned long[nPoint_P2PSend[nP2PSend]];
271  for (iSend = 0; iSend < nPoint_P2PSend[nP2PSend]; iSend++)
272  Local_Point_P2PSend[iSend] = 0;
273 
274  Local_Point_P2PRecv = nullptr;
275  Local_Point_P2PRecv = new unsigned long[nPoint_P2PRecv[nP2PRecv]];
276  for (iRecv = 0; iRecv < nPoint_P2PRecv[nP2PRecv]; iRecv++)
277  Local_Point_P2PRecv[iRecv] = 0;
278 
279  /*--- We allocate the memory for communicating values in a later step
280  once we know the maximum packet size that we need to communicate. This
281  memory is deallocated and reallocated automatically in the case that
282  the previously allocated memory is not sufficient. ---*/
283 
284  bufD_P2PSend = nullptr;
285  bufD_P2PRecv = nullptr;
286 
287  bufS_P2PSend = nullptr;
288  bufS_P2PRecv = nullptr;
289 
290  /*--- Allocate memory for the MPI requests if we need to communicate. ---*/
291 
292  if (nP2PSend > 0) {
293  req_P2PSend = new SU2_MPI::Request[nP2PSend];
294  }
295  if (nP2PRecv > 0) {
296  req_P2PRecv = new SU2_MPI::Request[nP2PRecv];
297  }
298 
299  /*--- Build lists of local index values for send. ---*/
300 
301  count = 0;
302  for (iSend = 0; iSend < nP2PSend; iSend++) {
303  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
304  if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) &&
305  (config->GetMarker_All_SendRecv(iMarker) > 0)) {
306 
307  MarkerS = iMarker;
308  nVertexS = geometry->nVertex[MarkerS];
309  iRank = config->GetMarker_All_SendRecv(MarkerS)-1;
310 
311  if (iRank == Neighbors_P2PSend[iSend]) {
312  for (iVertex = 0; iVertex < nVertexS; iVertex++) {
313  Local_Point_P2PSend[count] = geometry->vertex[MarkerS][iVertex]->GetNode();
314  count++;
315  }
316  }
317 
318  }
319  }
320  }
321 
322  /*--- Build lists of local index values for receive. ---*/
323 
324  count = 0;
325  for (iRecv = 0; iRecv < nP2PRecv; iRecv++) {
326  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
327  if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) &&
328  (config->GetMarker_All_SendRecv(iMarker) > 0)) {
329 
330  MarkerR = iMarker+1;
331  nVertexR = geometry->nVertex[MarkerR];
332  iRank = abs(config->GetMarker_All_SendRecv(MarkerR))-1;
333 
334  if (iRank == Neighbors_P2PRecv[iRecv]) {
335  for (iVertex = 0; iVertex < nVertexR; iVertex++) {
336  Local_Point_P2PRecv[count] = geometry->vertex[MarkerR][iVertex]->GetNode();
337  count++;
338  }
339  }
340 
341  }
342  }
343  }
344 
345  /*--- In the future, some additional data structures could be created
346  here to separate the interior and boundary nodes in order to help
347  further overlap computation and communication. ---*/
348 
349 }

351 void CGeometry::AllocateP2PComms(unsigned short countPerPoint) {
352 
353  /*--- This routine is activated whenever we attempt to perform
354  a point-to-point MPI communication with our neighbors but the
355  memory buffer allocated is not large enough for the packet size.
356  Therefore, we deallocate the previously allocated space and
357  reallocate a large enough array. Note that after the first set
358  communications, this routine will not need to be called again. ---*/
359 
360  if (countPerPoint <= maxCountPerPoint) return;
361 
362  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
363 
364  /*--- Store the larger packet size to the class data. ---*/
365 
366  maxCountPerPoint = countPerPoint;
367 
368  /*-- Deallocate and reallocate our su2double cummunication memory. ---*/
369 
370  delete [] bufD_P2PSend;
371  bufD_P2PSend = new su2double[maxCountPerPoint*nPoint_P2PSend[nP2PSend]] ();
372 
373  delete [] bufD_P2PRecv;
374  bufD_P2PRecv = new su2double[maxCountPerPoint*nPoint_P2PRecv[nP2PRecv]] ();
375 
376  delete [] bufS_P2PSend;
377  bufS_P2PSend = new unsigned short[maxCountPerPoint*nPoint_P2PSend[nP2PSend]] ();
378 
379  delete [] bufS_P2PRecv;
380  bufS_P2PRecv = new unsigned short[maxCountPerPoint*nPoint_P2PRecv[nP2PRecv]] ();
381 
382  }
383  END_SU2_OMP_SAFE_GLOBAL_ACCESS
384 
385 }

387 void CGeometry::PostP2PRecvs(CGeometry *geometry,
388  const CConfig *config,
389  unsigned short commType,
390  unsigned short countPerPoint,
391  bool val_reverse) const {
392 
393  /*--- Launch the non-blocking recv's first. Note that we have stored
394  the counts and sources, so we can launch these before we even load
395  the data and send from the neighbor ranks. ---*/
396 
397  SU2_OMP_MASTER
398  for (int iRecv = 0; iRecv < nP2PRecv; iRecv++) {
399  const auto iMessage = iRecv;
400 
401  /*--- In some instances related to the adjoint solver, we need
402  to reverse the direction of communications such that the normal
403  send nodes become the recv nodes and vice-versa. ---*/
404 
405  if (val_reverse) {
406 
407  /*--- Compute our location in the buffer using the send data
408  structure since we are reversing the comms. ---*/
409 
410  auto offset = countPerPoint*nPoint_P2PSend[iRecv];
411 
412  /*--- Take advantage of cumulative storage format to get the number
413  of elems that we need to recv. Note again that we select the send
414  points here as the recv points. ---*/
415 
416  auto nPointP2P = nPoint_P2PSend[iRecv+1] - nPoint_P2PSend[iRecv];
417 
418  /*--- Total count can include multiple pieces of data per element. ---*/
419 
420  auto count = countPerPoint*nPointP2P;
421 
422  /*--- Get the rank from which we receive the message. Note again
423  that we use the send rank as the source instead of the recv rank. ---*/
424 
425  auto source = Neighbors_P2PSend[iRecv];
426  auto tag = source + 1;
427 
428  /*--- Post non-blocking recv for this proc. Note that we use the
429  send buffer here too. This is important to make sure the arrays
430  are the correct size. ---*/
431 
432  switch (commType) {
433  case COMM_TYPE_DOUBLE:
434  SU2_MPI::Irecv(&(bufD_P2PSend[offset]), count, MPI_DOUBLE,
435  source, tag, SU2_MPI::GetComm(), &(req_P2PRecv[iRecv]));
436  break;
437  case COMM_TYPE_UNSIGNED_SHORT:
438  SU2_MPI::Irecv(&(bufS_P2PSend[offset]), count, MPI_UNSIGNED_SHORT,
439  source, tag, SU2_MPI::GetComm(), &(req_P2PRecv[iRecv]));
440  break;
441  default:
442  SU2_MPI::Error("Unrecognized data type for point-to-point MPI comms.",
443  CURRENT_FUNCTION);
444  break;
445  }
446 
447  } else {
448 
449  /*--- Compute our location in the recv buffer. ---*/
450 
451  auto offset = countPerPoint*nPoint_P2PRecv[iRecv];
452 
453  /*--- Take advantage of cumulative storage format to get the number
454  of elems that we need to recv. ---*/
455 
456  auto nPointP2P = nPoint_P2PRecv[iRecv+1] - nPoint_P2PRecv[iRecv];
457 
458  /*--- Total count can include multiple pieces of data per element. ---*/
459 
460  auto count = countPerPoint*nPointP2P;
461 
462  /*--- Get the rank from which we receive the message. ---*/
463 
464  auto source = Neighbors_P2PRecv[iRecv];
465  auto tag = source + 1;
466 
467  /*--- Post non-blocking recv for this proc. ---*/
468 
469  switch (commType) {
470  case COMM_TYPE_DOUBLE:
471  SU2_MPI::Irecv(&(bufD_P2PRecv[offset]), count, MPI_DOUBLE,
472  source, tag, SU2_MPI::GetComm(), &(req_P2PRecv[iMessage]));
473  break;
474  case COMM_TYPE_UNSIGNED_SHORT:
475  SU2_MPI::Irecv(&(bufS_P2PRecv[offset]), count, MPI_UNSIGNED_SHORT,
476  source, tag, SU2_MPI::GetComm(), &(req_P2PRecv[iMessage]));
477  break;
478  default:
479  SU2_MPI::Error("Unrecognized data type for point-to-point MPI comms.",
480  CURRENT_FUNCTION);
481  break;
482  }
483 
484  }
485 
486  }
487  END_SU2_OMP_MASTER
488

624 void CGeometry::InitiateComms(CGeometry *geometry,

629

638  int iMessage, iSend, nSend;
639 
640  /*--- Set the size of the data packet and type depending on quantity. ---*/
641 
642  GetCommCountAndType(config, commType, COUNT_PER_POINT, MPI_TYPE);
643 
644  /*--- Check to make sure we have created a large enough buffer
645  for these comms during preprocessing. This is only for the su2double
646  buffer. It will be reallocated whenever we find a larger count
647  per point. After the first cycle of comms, this should be inactive. ---*/
648 
649  geometry->AllocateP2PComms(COUNT_PER_POINT);
650 
651  /*--- Set some local pointers to make access simpler. ---*/
652 
653  su2double *bufDSend = geometry->bufD_P2PSend;
654  unsigned short *bufSSend = geometry->bufS_P2PSend;
655 
656  su2double *vector = nullptr;
657 
658  /*--- Load the specified quantity from the solver into the generic
659  communication buffer in the geometry class. ---*/
660 
661  /*--- Post all non-blocking recvs first before sends. ---*/
662 
663  geometry->PostP2PRecvs(geometry, config, MPI_TYPE, COUNT_PER_POINT, false);
664 
665  for (iMessage = 0; iMessage < nP2PSend; iMessage++) {
666 
667  /*--- Get the offset in the buffer for the start of this message. ---*/
668 
669  msg_offset = nPoint_P2PSend[iMessage];
670 
671  /*--- Total count can include multiple pieces of data per element. ---*/
672 
673  nSend = (nPoint_P2PSend[iMessage+1] - nPoint_P2PSend[iMessage]);
674 
675  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
676  for (iSend = 0; iSend < nSend; iSend++) {
677 
678  /*--- Get the local index for this communicated data. ---*/
679 
680  iPoint = geometry->Local_Point_P2PSend[msg_offset + iSend];
681 
682  /*--- Compute the offset in the recv buffer for this point. ---*/
683 
684  buf_offset = (msg_offset + iSend)*COUNT_PER_POINT;
685 
686  switch (commType) {
687  case COORDINATES:
688  vector = nodes->GetCoord(iPoint);
689  for (iDim = 0; iDim < nDim; iDim++)
690  bufDSend[buf_offset+iDim] = vector[iDim];
691  break;
692  case GRID_VELOCITY:
693  vector = nodes->GetGridVel(iPoint);
694  for (iDim = 0; iDim < nDim; iDim++)
695  bufDSend[buf_offset+iDim] = vector[iDim];
696  break;
697  case COORDINATES_OLD:
698  vector = nodes->GetCoord_n(iPoint);
699  for (iDim = 0; iDim < nDim; iDim++) {
700  bufDSend[buf_offset+iDim] = vector[iDim];
701  }
702  if (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND) {
703  vector = nodes->GetCoord_n1(iPoint);
704  for (iDim = 0; iDim < nDim; iDim++) {
705  bufDSend[buf_offset+nDim+iDim] = vector[iDim];
706  }
707  }
708  break;
709  case MAX_LENGTH:
710  bufDSend[buf_offset] = nodes->GetMaxLength(iPoint);
711  break;
712  case NEIGHBORS:
713  bufSSend[buf_offset] = geometry->nodes->GetnNeighbor(iPoint);
714  break;
715  default:
716  SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.",
717  CURRENT_FUNCTION);
718  break;
719  }
720  }
721  END_SU2_OMP_FOR
722 
723  /*--- Launch the point-to-point MPI send for this message. ---*/
724 
725  geometry->PostP2PSends(geometry, config, MPI_TYPE, COUNT_PER_POINT, iMessage, false);
726 
727  }

731 void CGeometry::CompleteComms(CGeometry *geometry,
732  const CConfig *config,
733  unsigned short commType) {
734 
735  if (nP2PRecv == 0) return;
736 
737  /*--- Local variables ---*/
738 
739  unsigned short iDim, COUNT_PER_POINT = 0, MPI_TYPE = 0;
740  unsigned long iPoint, iRecv, nRecv, msg_offset, buf_offset;
741 
742  int ind, source, iMessage, jRecv;
743 
744  /*--- Global status so all threads can see the result of Waitany. ---*/
745  static SU2_MPI::Status status;
746 
747  /*--- Set the size of the data packet and type depending on quantity. ---*/
748 
749  GetCommCountAndType(config, commType, COUNT_PER_POINT, MPI_TYPE);
750 
751  /*--- Set some local pointers to make access simpler. ---*/
752 
753  const su2double *bufDRecv = geometry->bufD_P2PRecv;
754  const unsigned short *bufSRecv = geometry->bufS_P2PRecv;
755 
756  /*--- Store the data that was communicated into the appropriate
757  location within the local class data structures. Note that we
758  recv and store the data in any order to take advantage of the
759  non-blocking comms. ---*/
760 
761  for (iMessage = 0; iMessage < nP2PRecv; iMessage++) {
762 
763  /*--- For efficiency, recv the messages dynamically based on
764  the order they arrive. ---*/
765 
766  SU2_OMP_SAFE_GLOBAL_ACCESS(SU2_MPI::Waitany(nP2PRecv, req_P2PRecv, &ind, &status);)
767 
768  /*--- Once we have recv'd a message, get the source rank. ---*/
769 
770  source = status.MPI_SOURCE;
771 
772  /*--- We know the offsets based on the source rank. ---*/
773 
774  jRecv = P2PRecv2Neighbor[source];
775 
776  /*--- Get the offset in the buffer for the start of this message. ---*/
777 
778  msg_offset = nPoint_P2PRecv[jRecv];
779 
780  /*--- Get the number of packets to be received in this message. ---*/
781 
782  nRecv = nPoint_P2PRecv[jRecv+1] - nPoint_P2PRecv[jRecv];
783 
784  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
785  for (iRecv = 0; iRecv < nRecv; iRecv++) {
786 
787  /*--- Get the local index for this communicated data. ---*/
788 
789  iPoint = geometry->Local_Point_P2PRecv[msg_offset + iRecv];
790 
791  /*--- Compute the total offset in the recv buffer for this point. ---*/
792 
793  buf_offset = (msg_offset + iRecv)*COUNT_PER_POINT;
794 
795  /*--- Store the data correctly depending on the quantity. ---*/
796 
797  switch (commType) {
798  case COORDINATES:
799  for (iDim = 0; iDim < nDim; iDim++)
800  nodes->SetCoord(iPoint, iDim, bufDRecv[buf_offset+iDim]);
801  break;
802  case GRID_VELOCITY:
803  for (iDim = 0; iDim < nDim; iDim++)
804  nodes->SetGridVel(iPoint, iDim, bufDRecv[buf_offset+iDim]);
805  break;
806  case COORDINATES_OLD:
807  nodes->SetCoord_n(iPoint, &bufDRecv[buf_offset]);
808  if (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND)
809  nodes->SetCoord_n1(iPoint, &bufDRecv[buf_offset+nDim]);
810  break;
811  case MAX_LENGTH:
812  nodes->SetMaxLength(iPoint, bufDRecv[buf_offset]);
813  break;
814  case NEIGHBORS:
815  nodes->SetnNeighbor(iPoint, bufSRecv[buf_offset]);
816  break;
817  default:
818  SU2_MPI::Error("Unrecognized quantity for point-to-point MPI comms.",
819  CURRENT_FUNCTION);
820  break;
821  }
822  }
823  END_SU2_OMP_FOR
824  }
825 
826  /*--- Verify that all non-blocking point-to-point sends have finished.
827  Note that this should be satisfied, as we have received all of the
828  data in the loop above at this point. ---*/
829 
830 #ifdef HAVE_MPI
831  SU2_OMP_SAFE_GLOBAL_ACCESS(SU2_MPI::Waitall(nP2PSend, req_P2PSend, MPI_STATUS_IGNORE);)
832 #endif
833 }

835 void CGeometry::PreprocessPeriodicComms(CGeometry *geometry,
836  CConfig *config) {
837 
838  /*--- We start with the send and receive lists already available in
839  the form of stored periodic point-donor pairs. We will loop through
840  these markers and establish the neighboring ranks and number of
841  send/recv points per pair. We will store this information and set
842  up persistent data structures so that we can reuse them throughout
843  the calculation for any periodic boundary communications. The goal
844  is to break the non-blocking comms into InitiatePeriodicComms() and
845  CompletePeriodicComms() in separate routines so that we can overlap the
846  communication and computation to hide the communication latency. ---*/
847 
848  /*--- Local variables. ---*/
849 
850  unsigned short iMarker;
851  unsigned long iPoint, iVertex, iPeriodic;
852 
853  int iRank, iSend, iRecv, ii, jj;
854 
855  /*--- Create some temporary structures for tracking sends/recvs. ---*/
856 
857  int *nPoint_Send_All = new int[size+1]; nPoint_Send_All[0] = 0;
858  int *nPoint_Recv_All = new int[size+1]; nPoint_Recv_All[0] = 0;
859  int *nPoint_Flag = new int[size];
860 
861  for (iRank = 0; iRank < size; iRank++) {
862  nPoint_Send_All[iRank] = 0;
863  nPoint_Recv_All[iRank] = 0;
864  nPoint_Flag[iRank]= -1;
865  }
866  nPoint_Send_All[size] = 0; nPoint_Recv_All[size] = 0;
867 
868  /*--- Loop through all of our periodic markers and track
869  our sends with each rank. ---*/
870 
871  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
872  if (config->GetMarker_All_KindBC(iMarker) == PERIODIC_BOUNDARY) {
873  iPeriodic = config->GetMarker_All_PerBound(iMarker);
874  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
875 
876  /*--- Get the current periodic point index. We only communicate
877  the owned nodes on a rank, as the MPI comms will take care of
878  the halos after completing the periodic comms. ---*/
879 
880  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
881 
882  if (geometry->nodes->GetDomain(iPoint)) {
883 
884  /*--- Get the rank that holds the matching periodic point
885  on the other marker in the periodic pair. ---*/
886 
887  iRank = (int)geometry->vertex[iMarker][iVertex]->GetDonorProcessor();
888 
889  /*--- If we have not visited this point last, increment our
890  number of points that must be sent to a particular proc. ---*/
891 
892  if ((nPoint_Flag[iRank] != (int)iPoint)) {
893  nPoint_Flag[iRank] = (int)iPoint;
894  nPoint_Send_All[iRank+1] += 1;
895  }
896 
897  }
898  }
899  }
900  }
901 
902  delete [] nPoint_Flag;
903 
904  /*--- Communicate the number of points to be sent/recv'd amongst
905  all processors. After this communication, each proc knows how
906  many periodic points it will receive from each other processor. ---*/
907 
908  SU2_MPI::Alltoall(&(nPoint_Send_All[1]), 1, MPI_INT,
909  &(nPoint_Recv_All[1]), 1, MPI_INT, SU2_MPI::GetComm());
910 
911  /*--- Check how many messages we will be sending and receiving.
912  Here we also put the counters into cumulative storage format to
913  make the communications simpler. Note that we are allowing each
914  rank to communicate to themselves in these counters, although
915  it will not be done through MPI. ---*/
916 
917  nPeriodicSend = 0; nPeriodicRecv = 0;
918 
919  for (iRank = 0; iRank < size; iRank++) {
920  if ((nPoint_Send_All[iRank+1] > 0)) nPeriodicSend++;
921  if ((nPoint_Recv_All[iRank+1] > 0)) nPeriodicRecv++;
922 
923  nPoint_Send_All[iRank+1] += nPoint_Send_All[iRank];
924  nPoint_Recv_All[iRank+1] += nPoint_Recv_All[iRank];
925  }
926 
927  /*--- Allocate only as much memory as needed for the periodic neighbors. ---*/
928 
929  nPoint_PeriodicSend = new int[nPeriodicSend+1]; nPoint_PeriodicSend[0] = 0;
930  nPoint_PeriodicRecv = new int[nPeriodicRecv+1]; nPoint_PeriodicRecv[0] = 0;
931 
932  Neighbors_PeriodicSend = new int[nPeriodicSend];
933  Neighbors_PeriodicRecv = new int[nPeriodicRecv];
934 
935  iSend = 0; iRecv = 0;
936  for (iRank = 0; iRank < size; iRank++) {
937  if ((nPoint_Send_All[iRank+1] > nPoint_Send_All[iRank])) {
938  Neighbors_PeriodicSend[iSend] = iRank;
939  nPoint_PeriodicSend[iSend+1] = nPoint_Send_All[iRank+1];
940  iSend++;
941  }
942  if ((nPoint_Recv_All[iRank+1] > nPoint_Recv_All[iRank])) {
943  Neighbors_PeriodicRecv[iRecv] = iRank;
944  nPoint_PeriodicRecv[iRecv+1] = nPoint_Recv_All[iRank+1];
945  iRecv++;
946  }
947  }
948 
949  /*--- Create a reverse mapping of the message to the rank so that we
950  can quickly access the correct data in the buffers when receiving
951  messages dynamically later during the iterations. ---*/
952 
953  PeriodicSend2Neighbor.clear();
954  for (iSend = 0; iSend < nPeriodicSend; iSend++)
955  PeriodicSend2Neighbor[Neighbors_PeriodicSend[iSend]] = iSend;
956 
957  PeriodicRecv2Neighbor.clear();
958  for (iRecv = 0; iRecv < nPeriodicRecv; iRecv++)
959  PeriodicRecv2Neighbor[Neighbors_PeriodicRecv[iRecv]] = iRecv;
960 
961  delete [] nPoint_Send_All;
962  delete [] nPoint_Recv_All;
963 
964  /*--- Allocate the memory to store the local index values for both
965  the send and receive periodic points and periodic index. ---*/
966 
967  Local_Point_PeriodicSend = nullptr;
968  Local_Point_PeriodicSend = new unsigned long[nPoint_PeriodicSend[nPeriodicSend]];
969  for (iSend = 0; iSend < nPoint_PeriodicSend[nPeriodicSend]; iSend++)
970  Local_Point_PeriodicSend[iSend] = 0;
971 
972  Local_Marker_PeriodicSend = nullptr;
973  Local_Marker_PeriodicSend = new unsigned long[nPoint_PeriodicSend[nPeriodicSend]];
974  for (iSend = 0; iSend < nPoint_PeriodicSend[nPeriodicSend]; iSend++)
975  Local_Marker_PeriodicSend[iSend] = 0;
976 
977  Local_Point_PeriodicRecv = nullptr;
978  Local_Point_PeriodicRecv = new unsigned long[nPoint_PeriodicRecv[nPeriodicRecv]];
979  for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv]; iRecv++)
980  Local_Point_PeriodicRecv[iRecv] = 0;
981 
982  Local_Marker_PeriodicRecv = nullptr;
983  Local_Marker_PeriodicRecv = new unsigned long[nPoint_PeriodicRecv[nPeriodicRecv]];
984  for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv]; iRecv++)
985  Local_Marker_PeriodicRecv[iRecv] = 0;
986 
987  /*--- We allocate the buffers for communicating values in a later step
988  once we know the maximum packet size that we need to communicate. This
989  memory is deallocated and reallocated automatically in the case that
990  the previously allocated memory is not sufficient. ---*/
991 
992  bufD_PeriodicSend = nullptr;
993  bufD_PeriodicRecv = nullptr;
994 
995  bufS_PeriodicSend = nullptr;
996  bufS_PeriodicRecv = nullptr;
997 
998  /*--- Allocate memory for the MPI requests if we need to communicate. ---*/
999 
1000  if (nPeriodicSend > 0) {
1001  req_PeriodicSend = new SU2_MPI::Request[nPeriodicSend];
1002  }
1003  if (nPeriodicRecv > 0) {
1004  req_PeriodicRecv = new SU2_MPI::Request[nPeriodicRecv];
1005  }
1006 
1007  /*--- Allocate arrays for sending the periodic point index and marker
1008  index to the recv rank so that it can store the local values. Therefore,
1009  the recv rank can quickly loop through the buffers to unpack the data. ---*/
1010 
1011  unsigned short nPackets = 2;
1012  unsigned long *idSend = new unsigned long[nPoint_PeriodicSend[nPeriodicSend]*nPackets];
1013  for (iSend = 0; iSend < nPoint_PeriodicSend[nPeriodicSend]*nPackets; iSend++)
1014  idSend[iSend] = 0;
1015 
1016  /*--- Build the lists of local index and periodic marker index values. ---*/
1017 
1018  ii = 0; jj = 0;
1019  for (iSend = 0; iSend < nPeriodicSend; iSend++) {
1020  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
1021  if (config->GetMarker_All_KindBC(iMarker) == PERIODIC_BOUNDARY) {
1022  iPeriodic = config->GetMarker_All_PerBound(iMarker);
1023  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
1024 
1025  /*--- Get the current periodic point index. We only communicate
1026  the owned nodes on a rank, as the MPI comms will take care of
1027  the halos after completing the periodic comms. ---*/
1028 
1029  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
1030 
1031  if (geometry->nodes->GetDomain(iPoint)) {
1032 
1033  /*--- Get the rank that holds the matching periodic point
1034  on the other marker in the periodic pair. ---*/
1035 
1036  iRank = (int)geometry->vertex[iMarker][iVertex]->GetDonorProcessor();
1037 
1038  /*--- If the rank for the current periodic point matches the
1039  rank of the current send message, then store the local point
1040  index on the matching periodic point and the periodic marker
1041  index to be communicated to the recv rank. ---*/
1042 
1043  if (iRank == Neighbors_PeriodicSend[iSend]) {
1044  Local_Point_PeriodicSend[ii] = iPoint;
1045  Local_Marker_PeriodicSend[ii] = (unsigned long)iMarker;
1046  jj = ii*nPackets;
1047  idSend[jj] = geometry->vertex[iMarker][iVertex]->GetDonorPoint();
1048  jj++;
1049  idSend[jj] = (unsigned long)iPeriodic;
1050  ii++;
1051  }
1052 
1053  }
1054  }
1055  }
1056  }
1057  }
1058 
1059  /*--- Allocate arrays for receiving the periodic point index and marker
1060  index to the recv rank so that it can store the local values. ---*/
1061 
1062  unsigned long *idRecv = new unsigned long[nPoint_PeriodicRecv[nPeriodicRecv]*nPackets];
1063  for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv]*nPackets; iRecv++)
1064  idRecv[iRecv] = 0;
1065 
1066 #ifdef HAVE_MPI
1067 
1068  int iMessage, offset, count, source, dest, tag;
1069 
1070  /*--- Launch the non-blocking recv's first. Note that we have stored
1071  the counts and sources, so we can launch these before we even load
1072  the data and send from the periodically matching ranks. ---*/
1073 
1074  iMessage = 0;
1075  for (iRecv = 0; iRecv < nPeriodicRecv; iRecv++) {
1076 
1077  /*--- Compute our location in the recv buffer. ---*/
1078 
1079  offset = nPackets*nPoint_PeriodicRecv[iRecv];
1080 
1081  /*--- Take advantage of cumulative storage format to get the number
1082  of elems that we need to recv. ---*/
1083 
1084  count = nPackets*(nPoint_PeriodicRecv[iRecv+1] - nPoint_PeriodicRecv[iRecv]);
1085 
1086  /*--- Get the rank from which we receive the message. ---*/
1087 
1088  source = Neighbors_PeriodicRecv[iRecv];
1089  tag = source + 1;
1090 
1091  /*--- Post non-blocking recv for this proc. ---*/
1092 
1093  SU2_MPI::Irecv(&(static_cast<unsigned long*>(idRecv)[offset]),
1094  count, MPI_UNSIGNED_LONG, source, tag, SU2_MPI::GetComm(),
1095  &(req_PeriodicRecv[iMessage]));
1096 
1097  /*--- Increment message counter. ---*/
1098 
1099  iMessage++;
1100 
1101  }
1102 
1103  /*--- Post the non-blocking sends. ---*/
1104 
1105  iMessage = 0;
1106  for (iSend = 0; iSend < nPeriodicSend; iSend++) {
1107 
1108  /*--- Compute our location in the send buffer. ---*/
1109 
1110  offset = nPackets*nPoint_PeriodicSend[iSend];
1111 
1112  /*--- Take advantage of cumulative storage format to get the number
1113  of points that we need to send. ---*/
1114 
1115  count = nPackets*(nPoint_PeriodicSend[iSend+1] - nPoint_PeriodicSend[iSend]);
1116 
1117  /*--- Get the rank to which we send the message. ---*/
1118 
1119  dest = Neighbors_PeriodicSend[iSend];
1120  tag = rank + 1;
1121 
1122  /*--- Post non-blocking send for this proc. ---*/
1123 
1124  SU2_MPI::Isend(&(static_cast<unsigned long*>(idSend)[offset]),
1125  count, MPI_UNSIGNED_LONG, dest, tag, SU2_MPI::GetComm(),
1126  &(req_PeriodicSend[iMessage]));
1127 
1128  /*--- Increment message counter. ---*/
1129 
1130  iMessage++;
1131 
1132  }
1133 
1134  /*--- Wait for the non-blocking comms to complete. ---*/
1135 
1136  SU2_MPI::Waitall(nPeriodicSend, req_PeriodicSend, MPI_STATUS_IGNORE);
1137  SU2_MPI::Waitall(nPeriodicRecv, req_PeriodicRecv, MPI_STATUS_IGNORE);
1138 
1139 #else
1140 
1141  /*--- Copy my own rank's data into the recv buffer directly in serial. ---*/
1142 
1143  int myStart, myFinal;
1144  for (int val_iSend = 0; val_iSend < nPeriodicSend; val_iSend++) {
1145  iRank = geometry->PeriodicRecv2Neighbor[rank];
1146  iRecv = geometry->nPoint_PeriodicRecv[iRank]*nPackets;
1147  myStart = nPoint_PeriodicSend[val_iSend]*nPackets;
1148  myFinal = nPoint_PeriodicSend[val_iSend+1]*nPackets;
1149  for (iSend = myStart; iSend < myFinal; iSend++) {
1150  idRecv[iRecv] = idSend[iSend];
1151  iRecv++;
1152  }
1153  }
1154 
1155 #endif
1156 
1157  /*--- Store the local periodic point and marker index values in our
1158  data structures so we can quickly unpack data during the iterations. ---*/
1159 
1160  ii = 0;
1161  for (iRecv = 0; iRecv < nPoint_PeriodicRecv[nPeriodicRecv]; iRecv++) {
1162  Local_Point_PeriodicRecv[iRecv] = idRecv[ii]; ii++;
1163  Local_Marker_PeriodicRecv[iRecv] = idRecv[ii]; ii++;
1164  }
1165 
1166  delete [] idSend;
1167  delete [] idRecv;
1168

1211 void CGeometry::PostPeriodicRecvs(CGeometry *geometry,
1212  const CConfig *config,
1213  unsigned short commType,
1214  unsigned short countPerPeriodicPoint) {
1215 
1216  /*--- In parallel, communicate the data with non-blocking send/recv. ---*/
1217 
1218 #ifdef HAVE_MPI
1219 
1220  /*--- Launch the non-blocking recv's first. Note that we have stored
1221  the counts and sources, so we can launch these before we even load
1222  the data and send from the neighbor ranks. ---*/
1223 
1224  SU2_OMP_MASTER
