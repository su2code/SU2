111  /*!
112  * \brief SST specific steps in the ComputeResidual method
113  * \param[in] config - Definition of the particular problem.
114  */
115  void FinishResidualCalc(const CConfig* config) override {
116  Flux[0] = a0*V_i[idx.Density()]*ScalarVar_i[0] + a1*V_j[idx.Density()]*ScalarVar_j[0];
117  Flux[1] = a0*V_i[idx.Density()]*ScalarVar_i[1] + a1*V_j[idx.Density()]*ScalarVar_j[1];
118  if (implicit) {
119  Jacobian_i[0][0] = a0; Jacobian_i[0][1] = 0.0;
120  Jacobian_i[1][0] = 0.0; Jacobian_i[1][1] = a0;
121 
122  Jacobian_j[0][0] = a1; Jacobian_j[0][1] = 0.0;
123  Jacobian_j[1][0] = 0.0; Jacobian_j[1][1] = a1;
124  }
125  }
