32 CFEMDataSorter::CFEMDataSorter(CConfig *config, CGeometry *geometry, const vector<string> &valFieldNames) :
33  CParallelDataSorter(config, valFieldNames){
34 
35  nDim = geometry->GetnDim();
36 
37  /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
38  geometrical information for the FEM DG solver. ---*/
39 
40  CMeshFEM_DG *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
41 
42  unsigned long nVolElemOwned = DGGeometry->GetNVolElemOwned();
43  CVolumeElementFEM *volElem = DGGeometry->GetVolElem();
44 
45  /*--- Create the map from the global DOF ID to the local index. ---*/
46 
47  vector<unsigned long> globalID;
48 
49  /*--- Update the solution by looping over the owned volume elements. ---*/
50 
51  for(unsigned long l=0; l<nVolElemOwned; ++l) {
52 
53  /* Count up the number of local points we have for allocating storage. */
54 
55  for(unsigned short j=0; j<volElem[l].nDOFsSol; ++j) {
56 
57  const unsigned long globalIndex = volElem[l].offsetDOFsSolGlobal + j;
58  globalID.push_back(globalIndex);
59 
60  nLocalPointsBeforeSort++;
61  }
62  }
63 
64  SU2_MPI::Allreduce(&nLocalPointsBeforeSort, &nGlobalPointBeforeSort, 1,
65  MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
66 
67  /*--- Create a linear partition --- */
68 
69  linearPartitioner.Initialize(nGlobalPointBeforeSort, 0);
70 
71  /*--- Prepare the send buffers ---*/
72 
73  PrepareSendBuffers(globalID);
74 
75 }

100 void CFEMDataSorter::SortVolumetricConnectivity(CConfig *config, CGeometry *geometry, unsigned short Elem_Type) {
101 
102  /* Determine the number of nodes for this element type. */
103  unsigned short NODES_PER_ELEMENT = 0;
104  switch (Elem_Type) {
105  case TRIANGLE:
106  NODES_PER_ELEMENT = N_POINTS_TRIANGLE;
107  break;
108  case QUADRILATERAL:
109  NODES_PER_ELEMENT = N_POINTS_QUADRILATERAL;
110  break;
111  case TETRAHEDRON:
112  NODES_PER_ELEMENT = N_POINTS_TETRAHEDRON;
113  break;
114  case HEXAHEDRON:
115  NODES_PER_ELEMENT = N_POINTS_HEXAHEDRON;
116  break;
117  case PRISM:
118  NODES_PER_ELEMENT = N_POINTS_PRISM;
119  break;
120  case PYRAMID:
121  NODES_PER_ELEMENT = N_POINTS_PYRAMID;
122  break;
123  default:
124  SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
125  }
126 
127  /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
128  geometrical information for the FEM DG solver. ---*/
129  CMeshFEM_DG *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
130 
131  unsigned long nVolElemOwned = DGGeometry->GetNVolElemOwned();
132  CVolumeElementFEM *volElem = DGGeometry->GetVolElem();
133 
134  const CFEMStandardElement *standardElementsSol = DGGeometry->GetStandardElementsSol();
135 
136  /*--- Determine the number of sub-elements on this rank. ---*/
137  unsigned long nSubElem_Local = 0;
138  for(unsigned long i=0; i<nVolElemOwned; ++i) {
139 
140  /* Determine the necessary data from the corresponding standard elem. */
141  const unsigned short ind = volElem[i].indStandardElement;
142  const unsigned short VTK_Type1 = standardElementsSol[ind].GetVTK_Type1();
143  const unsigned short VTK_Type2 = standardElementsSol[ind].GetVTK_Type2();
144 
145  /* Only store the linear sub elements if they are of
146  the current type that we are storing. */
147  if(Elem_Type == VTK_Type1) nSubElem_Local += standardElementsSol[ind].GetNSubElemsType1();
148  if(Elem_Type == VTK_Type2) nSubElem_Local += standardElementsSol[ind].GetNSubElemsType2();
149  }
150 
151  /* Allocate the memory to store the connectivity if the size is
152  larger than zero. */
153  int *Conn_SubElem = nullptr;
154  if(nSubElem_Local > 0) Conn_SubElem = new int[nSubElem_Local*NODES_PER_ELEMENT];
155 
156  /*--- Loop again over the local volume elements and store the global
157  connectivities of the sub-elements. Note one is added to the
158  index value, because visualization softwares typically use
159  1-based indexing. ---*/
160  unsigned long kNode = 0;
161  for(unsigned long i=0; i<nVolElemOwned; ++i) {
162 
163  /* Determine the necessary data from the corresponding standard elem. */
164  const unsigned short ind = volElem[i].indStandardElement;
165  const unsigned short VTK_Type1 = standardElementsSol[ind].GetVTK_Type1();
166  const unsigned short VTK_Type2 = standardElementsSol[ind].GetVTK_Type2();
167 
168  /* Check if the first sub-element is of the required type. */
169  if(Elem_Type == VTK_Type1) {
170 
171  /* Get the number of sub-elements and the local connectivity of
172  the sub-elements. */
173  const unsigned short nSubElems = standardElementsSol[ind].GetNSubElemsType1();
174  const unsigned short *connSubElems = standardElementsSol[ind].GetSubConnType1();
175 
176  /* Store the global connectivities. */
177  const unsigned short kk = NODES_PER_ELEMENT*nSubElems;
178  for(unsigned short k=0; k<kk; ++k, ++kNode)
179  Conn_SubElem[kNode] = connSubElems[k] + volElem[i].offsetDOFsSolGlobal + 1;
180  }
181 
182  /* Check if the second sub-element is of the required type. */
183  if(Elem_Type == VTK_Type2) {
184 
185  /* Get the number of sub-elements and the local connectivity of
186  the sub-elements. */
187  const unsigned short nSubElems = standardElementsSol[ind].GetNSubElemsType2();
188  const unsigned short *connSubElems = standardElementsSol[ind].GetSubConnType2();
189 
190  /* Store the global connectivities. */
191  const unsigned short kk = NODES_PER_ELEMENT*nSubElems;
192  for(unsigned short k=0; k<kk; ++k, ++kNode)
193  Conn_SubElem[kNode] = connSubElems[k] + volElem[i].offsetDOFsSolGlobal + 1;
194  }
195  }
196 
197  nElemPerType[TypeMap.at(Elem_Type)] = nSubElem_Local;
198 
199  /*--- Store the particular global element count in the class data,
200  and set the class data pointer to the connectivity array. ---*/
201  switch (Elem_Type) {
202  case TRIANGLE:
203  delete [] Conn_Tria_Par;
204  Conn_Tria_Par = Conn_SubElem;
205  break;
206  case QUADRILATERAL:
207  delete [] Conn_Quad_Par;
208  Conn_Quad_Par = Conn_SubElem;
209  break;
210  case TETRAHEDRON:
211  delete [] Conn_Tetr_Par;
212  Conn_Tetr_Par = Conn_SubElem;
213  break;
214  case HEXAHEDRON:
215  delete [] Conn_Hexa_Par;
216  Conn_Hexa_Par = Conn_SubElem;
217  break;
218  case PRISM:
219  delete [] Conn_Pris_Par;
220  Conn_Pris_Par = Conn_SubElem;
221  break;
222  case PYRAMID:
223  delete [] Conn_Pyra_Par;
224  Conn_Pyra_Par = Conn_SubElem;
225  break;
226  default:
227  SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
228  break;
229  }
230 }
