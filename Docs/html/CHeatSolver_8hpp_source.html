64  /*!
65  * \brief Applies an isothermal condition to a vertex of a marker.
66  */
67  void IsothermalBoundaryCondition(CGeometry *geometry, CSolver *flow_solver, const CConfig *config,
68  unsigned short iMarker, unsigned long iVertex, const su2double& temperature) {
69 
70  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
71  if (!geometry->nodes->GetDomain(iPoint)) return;
72 
73  const bool implicit = config->GetKind_TimeIntScheme() == EULER_IMPLICIT;
74  const su2double prandtl_lam = config->GetPrandtl_Lam();
75  const su2double const_diffusivity = config->GetThermalDiffusivity();
76 
77  const auto Point_Normal = geometry->vertex[iMarker][iVertex]->GetNormal_Neighbor();
78 
79  const auto* Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
80  const su2double Area = GeometryToolbox::Norm(nDim, Normal);
81 
82  const auto* Coord_i = geometry->nodes->GetCoord(iPoint);
83  const auto* Coord_j = geometry->nodes->GetCoord(Point_Normal);
84  const su2double dist_ij = GeometryToolbox::Distance(nDim, Coord_i, Coord_j);
85 
86  const su2double dTdn = -(nodes->GetTemperature(Point_Normal) - temperature) / dist_ij;
87 
88  su2double thermal_diffusivity = const_diffusivity;
89  if (flow) {
90  thermal_diffusivity = flow_solver->GetNodes()->GetLaminarViscosity(iPoint) / prandtl_lam;
91  }
92  LinSysRes(iPoint, 0) -= thermal_diffusivity * dTdn * Area;
93 
94  if (implicit) {
95  su2double Jacobian_i[] = {-thermal_diffusivity / dist_ij * Area};
96  Jacobian.SubtractBlock2Diag(iPoint, &Jacobian_i);
97  }
98  }

327  /*!
328  * \brief Compute objective output
329  * \param[in] config - Definition of the problem.
330  * \param[in] solver - Container vector with all the solutions.
331  */
332  void Evaluate_ObjFunc(const CConfig *config, CSolver**) override {
333  const auto weight = config->GetWeight_ObjFunc(0);
334 
335  switch (config->GetKind_ObjFunc()) {
336  case TOTAL_HEATFLUX:
337  Total_ComboObj = weight * Total_HeatFlux;
338  break;
339  case AVG_TEMPERATURE:
340  Total_ComboObj = weight * Total_AverageT;
341  break;
342  case CUSTOM_OBJFUNC:
343  Total_ComboObj = weight * Total_Custom_ObjFunc;
344  break;
345  default:
346  Total_ComboObj = 0.0;
347  }
348  }
