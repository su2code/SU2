33 void CIteration::SetGrid_Movement(CGeometry** geometry, CSurfaceMovement* surface_movement,
34  CVolumetricMovement* grid_movement, CSolver*** solver, CConfig* config,
35  unsigned long IntIter, unsigned long TimeIter) {
36  unsigned short Kind_Grid_Movement = config->GetKind_GridMovement();
37  bool adjoint = config->GetContinuous_Adjoint();
38 
39  unsigned short val_iZone = config->GetiZone();
40 
41  /*--- Perform mesh movement depending on specified type ---*/
42  switch (Kind_Grid_Movement) {
43  case RIGID_MOTION:
44 
45  if (rank == MASTER_NODE) cout << endl << " Performing rigid mesh transformation." << endl;
46 
47  /*--- Move each node in the volume mesh using the specified type
48  of rigid mesh motion. These routines also compute analytic grid
49  velocities for the fine mesh. ---*/
50 
51  grid_movement->Rigid_Translation(geometry[MESH_0], config, val_iZone, TimeIter);
52  grid_movement->Rigid_Plunging(geometry[MESH_0], config, val_iZone, TimeIter);
53  grid_movement->Rigid_Pitching(geometry[MESH_0], config, val_iZone, TimeIter);
54  grid_movement->Rigid_Rotation(geometry[MESH_0], config, val_iZone, TimeIter);
55 
56  /*--- Update the multigrid structure after moving the finest grid,
57  including computing the grid velocities on the coarser levels. ---*/
58 
59  grid_movement->UpdateMultiGrid(geometry, config);
60 
61  break;
62 
63  /*--- Already initialized in the static mesh movement routine at driver level. ---*/
64  case STEADY_TRANSLATION:
65  case ROTATING_FRAME:
66  break;
67  }
68 
69  if (config->GetSurface_Movement(AEROELASTIC) || config->GetSurface_Movement(AEROELASTIC_RIGID_MOTION)) {
70  /*--- Apply rigid mesh transformation to entire grid first, if necessary ---*/
71  if (IntIter == 0) {
72  if (Kind_Grid_Movement == AEROELASTIC_RIGID_MOTION) {
73  if (rank == MASTER_NODE) cout << endl << " Performing rigid mesh transformation." << endl;
74 
75  /*--- Move each node in the volume mesh using the specified type
76  of rigid mesh motion. These routines also compute analytic grid
77  velocities for the fine mesh. ---*/
78 
79  grid_movement->Rigid_Translation(geometry[MESH_0], config, val_iZone, TimeIter);
80  grid_movement->Rigid_Plunging(geometry[MESH_0], config, val_iZone, TimeIter);
81  grid_movement->Rigid_Pitching(geometry[MESH_0], config, val_iZone, TimeIter);
82  grid_movement->Rigid_Rotation(geometry[MESH_0], config, val_iZone, TimeIter);
83 
84  /*--- Update the multigrid structure after moving the finest grid,
85  including computing the grid velocities on the coarser levels. ---*/
86 
87  grid_movement->UpdateMultiGrid(geometry, config);
88  }
89 
90  }
91 
92  /*--- Use the if statement to move the grid only at selected dual time step iterations. ---*/
93  else if (IntIter % config->GetAeroelasticIter() == 0) {
94  if (rank == MASTER_NODE) cout << endl << " Solving aeroelastic equations and updating surface positions." << endl;
95 
96  /*--- Solve the aeroelastic equations for the new node locations of the moving markers(surfaces) ---*/
97 
98  solver[MESH_0][FLOW_SOL]->Aeroelastic(surface_movement, geometry[MESH_0], config, TimeIter);
99 
100  /*--- Deform the volume grid around the new boundary locations ---*/
101 
102  if (rank == MASTER_NODE) cout << " Deforming the volume grid due to the aeroelastic movement." << endl;
103  grid_movement->SetVolume_Deformation(geometry[MESH_0], config, true);
104 
105  /*--- Update the grid velocities on the fine mesh using finite
106  differencing based on node coordinates at previous times. ---*/
107 
108  if (rank == MASTER_NODE) cout << " Computing grid velocities by finite differencing." << endl;
109  geometry[MESH_0]->SetGridVelocity(config);
110 
111  /*--- Update the multigrid structure after moving the finest grid,
112  including computing the grid velocities on the coarser levels. ---*/
113 
114  grid_movement->UpdateMultiGrid(geometry, config);
115  }
116  }
117 
118  if (config->GetSurface_Movement(EXTERNAL) || config->GetSurface_Movement(EXTERNAL_ROTATION)) {
119  /*--- Apply rigid rotation to entire grid first, if necessary ---*/
120 
121  if (Kind_Grid_Movement == EXTERNAL_ROTATION) {
122  if (rank == MASTER_NODE) cout << " Updating node locations by rigid rotation." << endl;
123  grid_movement->Rigid_Rotation(geometry[MESH_0], config, val_iZone, TimeIter);
124  }
125 
126  /*--- Load new surface node locations from external files ---*/
127 
128  if (rank == MASTER_NODE) cout << " Updating surface locations from file." << endl;
129  surface_movement->SetExternal_Deformation(geometry[MESH_0], config, val_iZone, TimeIter);
130 
131  /*--- Deform the volume grid around the new boundary locations ---*/
132 
133  if (rank == MASTER_NODE) cout << " Deforming the volume grid." << endl;
134  grid_movement->SetVolume_Deformation(geometry[MESH_0], config, true);
135 
136  /*--- Update the grid velocities on the fine mesh using finite
137  differencing based on node coordinates at previous times. ---*/
138 
139  if (!adjoint) {
140  if (rank == MASTER_NODE) cout << " Computing grid velocities by finite differencing." << endl;
141  geometry[MESH_0]->SetGridVelocity(config);
142  }
143 
144  /*--- Update the multigrid structure after moving the finest grid,
145  including computing the grid velocities on the coarser levels. ---*/
146 
147  grid_movement->UpdateMultiGrid(geometry, config);
148  }
149 }

151 void CIteration::SetMesh_Deformation(CGeometry** geometry, CSolver** solver, CNumerics*** numerics, CConfig* config,
152  RECORDING kind_recording) {
153  if (!config->GetDeform_Mesh()) return;
154 
155  /*--- Perform the elasticity mesh movement ---*/
156 
157  bool wasActive = false;
158  if ((kind_recording != RECORDING::MESH_DEFORM) && !config->GetMultizone_Problem()) {
159  /*--- In a primal run, AD::TapeActive returns a false ---*/
160  /*--- In any other recordings, the tape is passive during the deformation. ---*/
161  wasActive = AD::BeginPassive();
162  }
163 
164  /*--- Set the stiffness of each element mesh into the mesh numerics ---*/
165 
166  solver[MESH_SOL]->SetMesh_Stiffness(geometry, numerics[MESH_SOL], config);
167 
168  /*--- Deform the volume grid around the new boundary locations ---*/
169 
170  solver[MESH_SOL]->DeformMesh(geometry, numerics[MESH_SOL], config);
171 
172  /*--- Continue recording. ---*/
173  AD::EndPassive(wasActive);
174 }
