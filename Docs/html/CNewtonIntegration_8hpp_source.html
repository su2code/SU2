124  /*--- Preconditioner objects for each active solver. ---*/
125  CPreconditioner<MixedScalar>* preconditioner = nullptr;
126 
127  /*--- If mixed precision is used, these temporaries are used to interface with the preconditioner. ---*/
128  mutable CSysVector<MixedScalar> precondIn, precondOut;
129 
130  template<class T, su2enable_if<!std::is_same<T,MixedScalar>::value> = 0>
131  inline unsigned long Preconditioner_impl(const CSysVector<T>& u, CSysVector<T>& v,
132  unsigned long iters, Scalar& eps) const {
133  CNEWTON_PARFOR
134  for (auto i = 0ul; i < u.GetLocSize(); ++i) precondIn[i] = u[i];
135  END_CNEWTON_PARFOR
136 
137  iters = Preconditioner_impl(precondIn, precondOut, iters, eps);
138 
139  CNEWTON_PARFOR
140  for (auto i = 0ul; i < u.GetLocSize(); ++i) v[i] = precondOut[i];
141  END_CNEWTON_PARFOR
142  SU2_OMP_BARRIER
143 
144  return iters;
145  }
146 
147  /*--- Otherwise they are not needed. ---*/
148  template<class T, su2enable_if<std::is_same<T,MixedScalar>::value> = 0>
149  inline unsigned long Preconditioner_impl(const CSysVector<T>& u, CSysVector<T>& v,
150  unsigned long iters, Scalar& eps) const {
151  if (iters == 0) {
152  (*preconditioner)(u, v);
153  return 0;
154  }
155  auto product = CSysMatrixVectorProduct<MixedScalar>(solvers[FLOW_SOL]->Jacobian, geometry, config);
156  v = MixedScalar(0.0);
157  MixedScalar eps_t = eps;
158  iters = solvers[FLOW_SOL]->System.FGMRES_LinSolver(u, v, product, *preconditioner, eps, iters, eps_t, false, config);
159  eps = eps_t;
160  return iters;
161  }
