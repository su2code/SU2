32 CMMSNSTwoHalfSpheresSolution::CMMSNSTwoHalfSpheresSolution(unsigned short val_nDim,
33  unsigned short val_nVar,
34  unsigned short val_iMesh,
35  CConfig* config)
36  : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
37 
38  /*--- Write a message that the solution is initialized for the manufactured
39  solution for the Navier-Stokes equations between two half spheres. ---*/
40  if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
41  cout << endl;
42  cout << "Warning: Fluid properties and solution are being " << endl;
43  cout << " initialized for the manufactured solution " << endl;
44  cout << " of the Navier-Stokes equations between two half spheres!!!" << endl;
45  cout << endl << flush;
46  }
47 
48  /*--- Coefficients, needed to determine the solution. ---*/
49  const su2double Prandtl = config->GetPrandtl_Lam();
50 
51  RGas = config->GetGas_Constant();
52  Gamma = config->GetGamma();
53  Viscosity = config->GetMu_Constant();
54  Conductivity = Viscosity*Gamma*RGas/(Prandtl*(Gamma-1.0));
55 
56  /*--- Initialize TWall to the default value of 300 K (in case the outer wall
57  is not modelled as an isothermal wall) and try to retrieve the wall
58  temperature from the boundary conditions. ---*/
59  TWall = 300.0;
60  for(unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
61  if(config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL) {
62  const string Marker_Tag = config->GetMarker_All_TagBound(iMarker);
63  TWall = config->GetIsothermal_Temperature(Marker_Tag);
64  }
65  }
66 
67  /*--- Get the reference values for pressure, density and velocity. ---*/
68  Pressure_Ref = config->GetPressure_Ref();
69  Density_Ref = config->GetDensity_Ref();
70  Velocity_Ref = config->GetVelocity_Ref();
71 
72  /*--- The constants for the density and velocities. ---*/
73  rho_0 = 1.25;
74  u_0 = 135.78;
75  v_0 = -67.61;
76  w_0 = 82.75;
77 
78  /*--- The constants for the temperature solution. ---*/
79  a_T1 = 1.05;
80  a_T2 = -0.85;
81 
82  /*--- Perform some sanity and error checks for this solution here. ---*/
83  if(nDim != 3)
84  SU2_MPI::Error("Grid must be 3D for the MMS NS Two Half Spheres case",
85  CURRENT_FUNCTION);
86 
87  if(config->GetTime_Marching() != TIME_MARCHING::STEADY)
88  SU2_MPI::Error("Steady mode must be selected for the MMS NS Two Half Spheres case",
89  CURRENT_FUNCTION);
90 
91  if(Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::RANS &&
92  Kind_Solver != MAIN_SOLVER::FEM_EULER && Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
93  Kind_Solver != MAIN_SOLVER::FEM_LES)
94  SU2_MPI::Error("Compressible flow equations must be selected for the MMS NS Two Half Spheres case",
95  CURRENT_FUNCTION);
96 
97  if((Kind_Solver != MAIN_SOLVER::NAVIER_STOKES) &&
98  (Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES))
99  SU2_MPI::Error("Navier Stokes equations must be selected for the MMS NS Two Half Spheres case",
100  CURRENT_FUNCTION);
101 
102  if((config->GetKind_FluidModel() != STANDARD_AIR) &&
103  (config->GetKind_FluidModel() != IDEAL_GAS))
104  SU2_MPI::Error("Standard air or ideal gas must be selected for the MMS NS Two Half Spheres case",
105  CURRENT_FUNCTION);
106 
107  if(config->GetKind_ViscosityModel() != VISCOSITYMODEL::CONSTANT)
108  SU2_MPI::Error("Sutherland must be selected for viscosity for the MMS NS Two Half Spheres case",
109  CURRENT_FUNCTION);
110 
111  if(config->GetKind_ConductivityModel() != CONDUCTIVITYMODEL::CONSTANT_PRANDTL)
112  SU2_MPI::Error("Constant Prandtl number must be selected for the MMS NS Two Half Spheres case",
113  CURRENT_FUNCTION);
114 }

126 void CMMSNSTwoHalfSpheresSolution::GetSolution(const su2double *val_coords,
127  const su2double val_t,
128  su2double *val_solution) const {
129 
130  /* Easier storage of the x-, y- and z-coordinates. */
131  const su2double x = val_coords[0];
132  const su2double y = val_coords[1];
133  const su2double z = val_coords[2];
134 
135  /* Determine the dimensional solution for the temperature. */
136  const su2double Pi = PI_NUMBER;
137  const su2double r = sqrt(x*x + y*y + z*z);
138  const su2double fact = (r-1.0)*(r-1.0)/(a_T1 + a_T2);
139 
140  su2double T = 0.25*TWall*(3.0 + fact*(a_T1*cos(Pi*(r-2.0))
141  + a_T2*cos(Pi*(r-2.0)*2.0)));
142 
143  /* Determine the dimensional solution for the velocities. */
144  su2double u = u_0*(r-1.0)*(2.0-r)*4.0;
145  su2double v = v_0*(r-1.0)*(2.0-r)*4.0;
146  su2double w = w_0*(r-1.0)*(2.0-r)*4.0;
147 
148  /* Compute the pressure from the density and temperature. */
149  su2double rho = rho_0;
150  su2double p = rho*RGas*T;
151 
152  /* Determine the non-dimensional solution. */
153  rho /= Density_Ref;
154  p /= Pressure_Ref;
155  u /= Velocity_Ref;
156  v /= Velocity_Ref;
157  w /= Velocity_Ref;
158 
159  /* Determine the non-dimensional conserved variables. */
160  val_solution[0] = rho;
161  val_solution[1] = rho*u;
162  val_solution[2] = rho*v;
163  val_solution[3] = rho*w;
164  val_solution[4] = p/(Gamma-1.0) + 0.5*rho*(u*u + v*v + w*w);
165 }
