36 void CVanDerWaalsGas::SetTDState_rhoe(su2double rho, su2double e) {
37  Density = rho;
38  StaticEnergy = e;
39 
40  Pressure = Gamma_Minus_One * Density / (1.0 - Density * b) * (StaticEnergy + Density * a) - a * Density * Density;
41  Temperature = (Pressure + Density * Density * a) * ((1 - Density * b) / (Density * Gas_Constant));
42  Entropy = Gas_Constant * (log(Temperature) / Gamma_Minus_One + log(1 / Density - b));
43 
44  dPde_rho = Density * Gamma_Minus_One / (1.0 - Density * b);
45  dPdrho_e = Gamma_Minus_One / (1.0 - Density * b) *
46  ((StaticEnergy + 2 * Density * a) + Density * b * (StaticEnergy + Density * a) / (1.0 - Density * b)) -
47  2 * Density * a;
48  dTdrho_e = Gamma_Minus_One / Gas_Constant * a;
49  dTde_rho = Gamma_Minus_One / Gas_Constant;
50 
51  SoundSpeed2 = dPdrho_e + Pressure / (Density * Density) * dPde_rho;
52 
53  Zed = Pressure / (Gas_Constant * Temperature * Density);
54 }

56 void CVanDerWaalsGas::SetTDState_PT(su2double P, su2double T) {
57  su2double toll = 1e-5;
58  unsigned short nmax = 20, count = 0;
59  su2double A, B, Z, DZ = 1.0, F, F1;
60  A = a * P / (T * Gas_Constant) / (T * Gas_Constant);
61  B = b * P / (T * Gas_Constant);
62 
63  if (Zed > 0.1)
64  Z = min(Zed, 0.99);
65  else
66  Z = 0.99;
67 
68  do {
69  F = Z * Z * Z - Z * Z * (B + 1.0) + Z * A - A * B;
70  F1 = 3 * Z * Z - 2 * Z * (B + 1.0) + A;
71  DZ = F / F1;
72  Z -= 0.7 * DZ;
73  count++;
74  } while (abs(DZ) > toll && count < nmax);
75 
76  if (count == nmax) {
77  cout << "Warning Newton-Raphson exceed number of max iteration in PT" << endl;
78  cout << "Compressibility factor " << Z << " would be substituted with " << Zed << endl;
79  }
80 
81  // check if the solution is physical otherwise uses previous point solution
82  if (Z <= 1.01 && Z >= 0.05 && count < nmax) Zed = Z;
83 
84  Density = P / (Zed * Gas_Constant * T);
85 
86  su2double e = T * Gas_Constant / Gamma_Minus_One - a * Density;
87  SetTDState_rhoe(Density, e);
88 }

95 void CVanDerWaalsGas::SetTDState_hs(su2double h, su2double s) {
96  su2double v, T, rho, f, fmid, rtb;
97  su2double x1, x2, xmid, dx, fx1, fx2;
98  su2double toll = 1e-5, FACTOR = 0.2;
99  unsigned short count = 0, NTRY = 100, ITMAX = 100;
100  su2double cons_s, cons_h;
101 
102  T = 1.0 * h * Gamma_Minus_One / Gas_Constant / Gamma;
103  v = exp(-1 / Gamma_Minus_One * log(T) + s / Gas_Constant);
104  x1 = 0.2 * v;
105  x2 = 0.35 * v;
106  fx1 = log(x1 - b) - s / Gas_Constant +
107  log((h + 2 * a / x1) / Gas_Constant / (1 / Gamma_Minus_One + x1 / (x1 - b))) / Gamma_Minus_One;
108  fx2 = log(x2 - b) - s / Gas_Constant +
109  log((h + 2 * a / x2) / Gas_Constant / (1 / Gamma_Minus_One + x2 / (x2 - b))) / Gamma_Minus_One;
110 
111  // zbrac algorithm NR
112  for (int j = 1; j <= NTRY; j++) {
113  if (fx1 * fx2 > 0.0) {
114  if (fabs(fx1) < fabs(fx2)) {
115  x1 += FACTOR * (x1 - x2);
116  fx1 = log(x1 - b) - s / Gas_Constant +
117  log((h + 2 * a / x1) / Gas_Constant / (1 / Gamma_Minus_One + x1 / (x1 - b))) / Gamma_Minus_One;
118  } else {
119  x2 += FACTOR * (x2 - x1);
120  fx2 = log(x2 - b) - s / Gas_Constant +
121  log((h + 2 * a / x2) / Gas_Constant / (1 / Gamma_Minus_One + x2 / (x2 - b))) / Gamma_Minus_One;
122  }
123  }
124  }
125 
126  // rtbis algorithm NR
127  f = fx1;
128  fmid = fx2;
129  if (f * fmid >= 0.0) {
130  cout << "Root must be bracketed for bisection in rtbis" << endl;
131  SetTDState_rhoT(Density, Temperature);
132  }
133  rtb = f < 0.0 ? (static_cast<void>(dx = x2 - x1), x1) : (static_cast<void>(dx = x1 - x2), x2);
134  do {
135  xmid = rtb + (dx *= 0.5);
136  fmid = log(xmid - b) - s / Gas_Constant +
137  log((h + 2 * a / xmid) / Gas_Constant / (1 / Gamma_Minus_One + xmid / (xmid - b))) / Gamma_Minus_One;
138  if (fmid <= 0.0) rtb = xmid;
139  count++;
140  } while (abs(fmid) > toll && count < ITMAX);
141 
142  v = xmid;
143  if (count == ITMAX) {
144  cout << "Too many bisections in rtbis" << endl;
145  }
146 
147  rho = 1 / v;
148  T = (h + 2 * a / v) / Gas_Constant / (1 / Gamma_Minus_One + v / (v - b));
149  SetTDState_rhoT(rho, T);
150 
151  cons_h = abs(((StaticEnergy + Pressure / Density) - h) / h);
152  cons_s = abs((Entropy - s) / s);
153 
154  if (cons_h > 1e-3 || cons_s > 1e-3) {
155  cout << "TD consistency not verified in hs call" << endl;
156  }
157 }

169 void CVanDerWaalsGas::SetTDState_Ps(su2double P, su2double s) {
170  su2double T, rho, cons_P, cons_s;
171  su2double x1, x2, fx1, fx2, f, fmid, T1, T2, rtb, dx, xmid;
172  su2double toll = 1e-5, FACTOR = 0.2;
173  unsigned short count = 0, NTRY = 100, ITMAX = 100;
174 
175  T = exp(Gamma_Minus_One / Gamma * (s / Gas_Constant + log(P) - log(Gas_Constant)));
176  rho = P / (T * Gas_Constant);
177 
178  if (Zed < 0.9999) {
179  x1 = rho;
180  x2 = rho / Zed;
181 
182  } else {
183  x1 = rho;
184  x2 = rho / 0.5;
185  }
186  T1 = (P + x1 * x1 * a) * ((1 - x1 * b) / (x1 * Gas_Constant));
187  fx1 = Gas_Constant * (log(T1) / Gamma_Minus_One + log(1 / x1 - b)) - s;
188  T2 = (P + x2 * x2 * a) * ((1 - x2 * b) / (x2 * Gas_Constant));
189  fx2 = Gas_Constant * (log(T2) / Gamma_Minus_One + log(1 / x2 - b)) - s;
190 
191  // zbrac algorithm NR
192  for (int j = 1; j <= NTRY; j++) {
193  if (fx1 * fx2 > 0.0) {
194  if (fabs(fx1) < fabs(fx2)) {
195  x1 += FACTOR * (x1 - x2);
196  T1 = (P + x1 * x1 * a) * ((1 - x1 * b) / (x1 * Gas_Constant));
197  fx1 = Gas_Constant * (log(T1) / Gamma_Minus_One + log(1 / x1 - b)) - s;
198  } else {
199  x2 += FACTOR * (x2 - x1);
200  T2 = (P + x2 * x2 * a) * ((1 - x2 * b) / (x2 * Gas_Constant));
201  fx2 = Gas_Constant * (log(T2) / Gamma_Minus_One + log(1 / x2 - b)) - s;
202  }
203  }
204  }
205 
206  // rtbis algorithm NR
207  f = fx1;
208  fmid = fx2;
209  if (f * fmid >= 0.0) {
210  cout << "Root must be bracketed for bisection in rtbis" << endl;
211  SetTDState_rhoT(Density, Temperature);
212  }
213  rtb = f < 0.0 ? (static_cast<void>(dx = x2 - x1), x1) : (static_cast<void>(dx = x1 - x2), x2);
214  do {
215  xmid = rtb + (dx *= 0.5);
216  T = (P + xmid * xmid * a) * ((1 - xmid * b) / (xmid * Gas_Constant));
217  fmid = Gas_Constant * (log(T) / Gamma_Minus_One + log(1 / xmid - b)) - s;
218  if (fmid <= 0.0) rtb = xmid;
219  count++;
220  } while (abs(fmid) > toll && count < ITMAX);
221 
222  if (count == ITMAX) {
223  cout << "Too many bisections in rtbis" << endl;
224  }
225 
226  rho = xmid;
227  T = (P + rho * rho * a) * ((1 - rho * b) / (rho * Gas_Constant));
228  SetTDState_rhoT(rho, T);
229 
230  cons_P = abs((Pressure - P) / P);
231  cons_s = abs((Entropy - s) / s);
232 
233  if (cons_P > 1e-3 || cons_s > 1e-3) {
234  cout << "TD consistency not verified in hs call" << endl;
235  }
236 }

238 void CVanDerWaalsGas::ComputeDerivativeNRBC_Prho(su2double P, su2double rho) {
239  su2double dPdT_rho, dPdrho_T, dPds_rho;
240 
241  SetTDState_Prho(P, rho);
242 
243  dPdT_rho = Gas_Constant * rho / (1.0 - rho * b);
244  dPdrho_T = Gas_Constant * Temperature / (1.0 - rho * b) / (1.0 - rho * b) - 2.0 * rho * a;
245 
246  dhdrho_P = -dPdrho_e / dPde_rho - P / rho / rho;
247  dhdP_rho = 1.0 / dPde_rho + 1.0 / rho;
248  dPds_rho = rho * rho * (SoundSpeed2 - dPdrho_T) / dPdT_rho;
249  dsdP_rho = 1.0 / dPds_rho;
250  dsdrho_P = -SoundSpeed2 / dPds_rho;
251 }
