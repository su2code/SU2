33 void CFEM_DG_Integration::SingleGrid_Iteration(CGeometry ****geometry,
34  CSolver *****solver_container,
35  CNumerics ******numerics_container,
36  CConfig **config,
37  unsigned short RunTime_EqSystem,
38  unsigned short iZone,
39  unsigned short iInst) {
40 
41  unsigned short iMesh, iStep, iLimit = 1;
42  unsigned short SolContainer_Position = config[iZone]->GetContainerPosition(RunTime_EqSystem);
43  unsigned short FinestMesh = config[iZone]->GetFinestMesh();
44 
45  /*--- For now, we assume no geometric multigrid. ---*/
46  iMesh = FinestMesh;
47 
48  /*--- Check if only the Jacobian of the spatial discretization must
49  be computed. If so, call the appropriate function and return. ---*/
50  if (config[iZone]->GetJacobian_Spatial_Discretization_Only()) {
51  solver_container[iZone][iInst][iMesh][SolContainer_Position]->ComputeSpatialJacobian(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
52  numerics_container[iZone][iInst][iMesh][SolContainer_Position],
53  config[iZone], iMesh, RunTime_EqSystem);
54  return;
55  }
56 
57  /*--- Determine the number of stages in the time stepping algorithm.
58  For the Runge-Kutta schemes this is the number of RK stages,
59  while for ADER-DG this information is not used, because a more
60  complicated algorithm must be used to facilitate time accurate
61  local time stepping. Note that we are currently hard-coding
62  the classical RK4 scheme. ---*/
63  bool useADER = false;
64  switch (config[iZone]->GetKind_TimeIntScheme()) {
65  case RUNGE_KUTTA_EXPLICIT: iLimit = config[iZone]->GetnRKStep(); break;
66  case CLASSICAL_RK4_EXPLICIT: iLimit = 4; break;
67  case ADER_DG: iLimit = 1; useADER = true; break;
68  case EULER_EXPLICIT: case EULER_IMPLICIT: iLimit = 1; break; }
69 
70  /*--- In case an unsteady simulation is carried out, it is possible that a
71  synchronization time step is specified. If so, set the boolean
72  TimeSynSpecified to true, which leads to an outer loop in the
73  algorithm below. ---*/
74  bool TimeSyncSpecified = false;
75  const su2double TimeSync = config[iZone]->GetTime_Step()/config[iZone]->GetTime_Ref();
76  if(config[iZone]->GetTime_Marching() == TIME_MARCHING::TIME_STEPPING &&
77  config[iZone]->GetUnst_CFL() != 0.0 &&
78  TimeSync != 0.0) TimeSyncSpecified = true;
79 
80  /*--- Outer loop, which is only active when a synchronization time has been
81  specified for an unsteady simulation. ---*/
82  bool syncTimeReached = false;
83  su2double timeEvolved = 0.0;
84  while( !syncTimeReached ) {
85 
86  /* Compute the time step for stability. */
87  solver_container[iZone][iInst][iMesh][SolContainer_Position]->SetTime_Step(geometry[iZone][iInst][iMesh],
88  solver_container[iZone][iInst][iMesh],
89  config[iZone], iMesh, config[iZone]->GetTimeIter());
90  /* Possibly overrule the specified time step when a synchronization time was
91  specified and determine whether or not the time loop must be continued.
92  When TimeSyncSpecified is false, the loop is always terminated. */
93  if( TimeSyncSpecified )
94  solver_container[iZone][iInst][iMesh][SolContainer_Position]->CheckTimeSynchronization(config[iZone],
95  TimeSync, timeEvolved,
96  syncTimeReached);
97  else
98  syncTimeReached = true;
99 
100  /*--- For ADER in combination with time accurate local time stepping, the
101  space and time integration are tightly coupled and cannot be treated
102  segregatedly. Therefore a different function is called for ADER to
103  carry out the space and time integration. ---*/
104  if( useADER ) {
105  solver_container[iZone][iInst][iMesh][SolContainer_Position]->ADER_SpaceTimeIntegration(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
106  numerics_container[iZone][iInst][iMesh][SolContainer_Position],
107  config[iZone], iMesh, RunTime_EqSystem);
108  }
109  else {
110 
111  /*--- Time and space integration can be decoupled. ---*/
112  for (iStep = 0; iStep < iLimit; iStep++) {
113 
114  /*--- Preprocessing ---*/
115  solver_container[iZone][iInst][iMesh][SolContainer_Position]->Preprocessing(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
116  config[iZone], iMesh, iStep, RunTime_EqSystem, false);
117 
118  /*--- Space integration ---*/
119  Space_Integration(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
120  numerics_container[iZone][iInst][iMesh][SolContainer_Position],
121  config[iZone], iMesh, iStep, RunTime_EqSystem);
122 
123  /*--- Time integration, update solution using the old solution plus the solution increment ---*/
124  Time_Integration(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
125  config[iZone], iStep, RunTime_EqSystem);
126 
127  /*--- Postprocessing ---*/
128  solver_container[iZone][iInst][iMesh][SolContainer_Position]->Postprocessing(geometry[iZone][iInst][iMesh], solver_container[iZone][iInst][iMesh],
129  config[iZone], iMesh);
130  }
131  }
132  }
133 
134  /*--- Calculate the inviscid and viscous forces ---*/
135  solver_container[iZone][iInst][FinestMesh][SolContainer_Position]->Pressure_Forces(geometry[iZone][iInst][iMesh], config[iZone]);
136 
137  solver_container[iZone][iInst][FinestMesh][SolContainer_Position]->Friction_Forces(geometry[iZone][iInst][iMesh], config[iZone]);
138 
139  /*--- Convergence strategy ---*/
140 
141  //Convergence_Monitoring(geometry[iZone][iInst][FinestMesh], config[iZone], Iteration, monitor, FinestMesh);
142 }

144 void CFEM_DG_Integration::Space_Integration(CGeometry *geometry,
145  CSolver **solver_container,
146  CNumerics **numerics,
147  CConfig *config, unsigned short iMesh,
148  unsigned short iStep,
149  unsigned short RunTime_EqSystem) {
150 
151  unsigned short MainSolver = config->GetContainerPosition(RunTime_EqSystem);
152 
153  /*--- Runge-Kutta type of time integration schemes. In the first step, i.e.
154  if iStep == 0, set the old solution (working solution for the DG part),
155  and if needed, the new solution. ---*/
156  if (iStep == 0) {
157  solver_container[MainSolver]->Set_OldSolution();
158 
159  if (config->GetKind_TimeIntScheme() == CLASSICAL_RK4_EXPLICIT) {
160  solver_container[MainSolver]->Set_NewSolution();
161  }
162  }
163 
164  /*--- Compute the spatial residual by processing the task list. ---*/
165  solver_container[MainSolver]->ProcessTaskList_DG(geometry, solver_container, numerics, config, iMesh);
166 }

168 void CFEM_DG_Integration::Time_Integration(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iStep,
169  unsigned short RunTime_EqSystem) {
170 
171  unsigned short MainSolver = config->GetContainerPosition(RunTime_EqSystem);
172 
173  /*--- Perform the time integration ---*/
174  switch (config->GetKind_TimeIntScheme()) {
175  case (RUNGE_KUTTA_EXPLICIT):
176  solver_container[MainSolver]->ExplicitRK_Iteration(geometry, solver_container, config, iStep);
177  break;
178  case (CLASSICAL_RK4_EXPLICIT):
179  solver_container[MainSolver]->ClassicalRK4_Iteration(geometry, solver_container, config, iStep);
180  break;
181  default:
182  SU2_MPI::Error("Time integration scheme not implemented.", CURRENT_FUNCTION);
183  }
