32 CCGNSMeshReaderFVM::CCGNSMeshReaderFVM(CConfig *val_config,
33  unsigned short val_iZone,
34  unsigned short val_nZone)
35 : CMeshReaderFVM(val_config, val_iZone, val_nZone) {
36 
37 #ifdef HAVE_CGNS
38  OpenCGNSFile(config->GetMesh_FileName());
39 
40  /*--- Read the basic information about the database and zone(s). ---*/
41  ReadCGNSDatabaseMetadata();
42  ReadCGNSZoneMetadata();
43 
44  /*--- Read the point coordinates into linear partitions. ---*/
45  ReadCGNSPointCoordinates();
46 
47  /*--- Loop over all sections to access the grid connectivity. We
48  treat the interior and boundary elements with separate routines.
49  If we have found that this is a boundary section (we assume
50  that internal cells and boundary cells do not exist in the same
51  section together), the master node reads the boundary section.
52  Otherwise, all ranks read and communicate the interior sections. ---*/
53  ReadCGNSSectionMetadata();
54  numberOfMarkers = 0;
55  for (int s = 0; s < nSections; s++) {
56  if (isInterior[s]) {
57  ReadCGNSVolumeSection(s);
58  } else {
59  numberOfMarkers++;
60  ReadCGNSSurfaceSection(s);
61  }
62  }
63 
64  /*--- We have extracted all CGNS data. Close the CGNS file. ---*/
65  if (cg_close(cgnsFileID)) cg_error_exit();
66 
67  /*--- Put our CGNS data into the class data for the mesh reader. ---*/
68  ReformatCGNSVolumeConnectivity();
69  ReformatCGNSSurfaceConnectivity();
70 
71 #else
72  SU2_MPI::Error(string(" SU2 built without CGNS support. \n") +
73  string(" To use CGNS, build SU2 accordingly."),
74  CURRENT_FUNCTION);
75 #endif
76 }

81 void CCGNSMeshReaderFVM::OpenCGNSFile(string val_filename) {
82 
83  /*--- Check whether the supplied file is truly a CGNS file. ---*/
84 
85  int file_type;
86  float file_version;
87  if (cg_is_cgns(val_filename.c_str(), &file_type) != CG_OK) {
88  SU2_MPI::Error(val_filename +
89  string(" was not found or is not a properly formatted") +
90  string(" CGNS file.\nNote that SU2 expects unstructured") +
91  string(" CGNS files in ADF data format."),
92  CURRENT_FUNCTION);
93  }
94 
95  /*--- Open the CGNS file for reading. The value of cgnsFileID returned
96  is the specific index number for this file and will be
97  repeatedly used in the function calls. ---*/
98 
99  if (cg_open(val_filename.c_str(), CG_MODE_READ, &cgnsFileID))
100  cg_error_exit();
101  if (rank == MASTER_NODE) {
102  cout << "Reading the CGNS file: ";
103  cout << val_filename.c_str() << "." << endl;
104  }
105  if (cg_version(cgnsFileID, &file_version))
106  cg_error_exit();
107  if (rank == MASTER_NODE) {
108  if (file_version < 4.0) {
109  cout << "WARNING: The CGNS file version (" << file_version << ") is old and may cause high memory usage issues, consider updating the file with the cgnsupdate tool.\n";
110  }
111  }

114 void CCGNSMeshReaderFVM::ReadCGNSDatabaseMetadata() {
115 
116  /*--- Get the number of databases. This is the highest node
117  in the CGNS heirarchy. ---*/
118 
119  int nbases;
120  if (cg_nbases(cgnsFileID, &nbases)) cg_error_exit();
121  if (rank == MASTER_NODE)
122  cout << "CGNS file contains " << nbases << " database(s)." << endl;
123 
124  /*--- Check if there is more than one database. Throw an
125  error if there is because this reader can currently
126  only handle one database. ---*/
127 
128  if ( nbases > 1 ) {
129  SU2_MPI::Error("CGNS reader currently can only handle 1 database.",
130  CURRENT_FUNCTION);
131  }
132 
133  /*--- Read the database. Note that the CGNS indexing starts at 1. ---*/
134 
135  int cell_dim, phys_dim;
136  char basename[CGNS_STRING_SIZE];
137  if (cg_base_read(cgnsFileID, cgnsBase, basename, &cell_dim, &phys_dim))
138  cg_error_exit();
139  if (rank == MASTER_NODE) {
140  cout << "Database " << cgnsBase << ", " << basename << ": ";
141  cout << " cell dimension of " << cell_dim << ", physical ";
142  cout << "dimension of " << phys_dim << "." << endl;
143  }
144 
145  /*--- Set the number of dimensions baed on cell_dim. ---*/
146 
147  dimension = (unsigned short)cell_dim;
148 
149 }

151 void CCGNSMeshReaderFVM::ReadCGNSZoneMetadata() {
152 
153  /*--- First, check all sections to find the element types and to
154  classify them as either surface or volume elements. We will also
155  perform some error checks here to avoid partitioning issues. ---*/
156 
157  /*--- Get the number of zones for this base. ---*/
158 
159  int nzones;
160  if (cg_nzones(cgnsFileID, cgnsBase, &nzones)) cg_error_exit();
161  if (rank == MASTER_NODE) {
162  cout << nzones << " total zone(s)." << endl;
163  }
164 
165  /*--- Check if there is more than one zone. Until we enable it, we
166  will require a single zone CGNS file. Multizone problems can still
167  be run with CGNS by using separate CGNS files for each zone. ---*/
168 
169  if ( nzones > 1 ) {
170  SU2_MPI::Error(string("CGNS reader currently expects only 1 zone per CGNS file.") +
171  string("Multizone problems can be run with separate CGNS files for each zone."), CURRENT_FUNCTION);
172  }
173 
174  /*--- Read the basic information for this zone, including
175  the name and the number of vertices, cells, and
176  boundary cells which are stored in the cgsize variable. ---*/
177 
178  vector<cgsize_t> cgsize(3);
179  ZoneType_t zonetype;
180  char zonename[CGNS_STRING_SIZE];
181  if (cg_zone_read(cgnsFileID, cgnsBase, cgnsZone, zonename, cgsize.data()))
182  cg_error_exit();
183 
184  /*--- Rename the zone size information for clarity.
185  NOTE: The number of cells here may be only the number of
186  interior elements or it may be the total. This needs to
187  be counted explicitly later. ---*/
188 
189  numberOfGlobalPoints = cgsize[0];
190  int nElemCGNS = cgsize[1];
191 
192  /*--- Get some additional information about the current zone. ---*/
193 
194  if (cg_zone_type(cgnsFileID, cgnsBase, cgnsZone, &zonetype)) cg_error_exit();
195 
196  /*--- Check for an unstructured mesh. Throw an error if not found. ---*/
197 
198  if (zonetype != Unstructured)
199  SU2_MPI::Error("Structured CGNS zone found while unstructured expected.",
200  CURRENT_FUNCTION);
201 
202  /*--- Print current zone info to the console. ---*/
203 
204  if (rank == MASTER_NODE) {
205  cout << "Zone " << cgnsZone << ", " << zonename << ": ";
206  cout << numberOfGlobalPoints << " total vertices, ";
207  cout << nElemCGNS << " total elements." << endl;
208  }
209 
210  /*--- Retrieve the number of grids in this zone. For now, we know
211  this is one, but to be more general, this will need to check and
212  allow for a loop over all grids. ---*/
213 
214  int ngrids;
215  if (cg_ngrids(cgnsFileID, cgnsBase, cgnsZone, &ngrids)) cg_error_exit();
216  if (ngrids > 1) {
217  SU2_MPI::Error("CGNS reader currently handles only 1 grid per zone.",
218  CURRENT_FUNCTION);
219  }
220

223 void CCGNSMeshReaderFVM::ReadCGNSPointCoordinates() {
224 
225  /*--- Compute the number of points that will be on each processor.
226  This is a linear partitioning with the addition of a simple load
227  balancing for any remainder points. ---*/
228 
229  CLinearPartitioner pointPartitioner(numberOfGlobalPoints,0);
230 
231  /*--- Store the local number of nodes for this rank. ---*/
232 
233  numberOfLocalPoints = pointPartitioner.GetSizeOnRank(rank);
234 
235  /*--- Create buffer to hold the grid coordinates for our rank. ---*/
236 
237  localPointCoordinates.resize(dimension);
238  for (int k = 0; k < dimension; k++)
239  localPointCoordinates[k].resize(numberOfLocalPoints, 0.0);
240 
241  /*--- Set the value of range_max to the total number of nodes in
242  the unstructured mesh. Also allocate memory for the temporary array
243  that will hold the grid coordinates as they are extracted. Note the
244  +1 for CGNS convention. ---*/
245 
246  cgsize_t range_min = (cgsize_t)pointPartitioner.GetFirstIndexOnRank(rank)+1;
247  cgsize_t range_max = (cgsize_t)pointPartitioner.GetLastIndexOnRank(rank);
248 
249  /*--- Loop over each set of coordinates. ---*/
250 
251  for (int k = 0; k < dimension; k++) {
252 
253  /*--- Read the coordinate info. This will retrieve the
254  data type (either RealSingle or RealDouble) as
255  well as the coordname which will specify the
256  type of data that it is based in the SIDS convention.
257  This might be "CoordinateX," for instance. ---*/
258 
259  char coordname[CGNS_STRING_SIZE];
260  DataType_t datatype;
261  if (cg_coord_info(cgnsFileID, cgnsBase, cgnsZone, k+1,
262  &datatype, coordname)) cg_error_exit();
263  if (rank == MASTER_NODE) {
264  cout << "Loading " << coordname;
265  if (size > SINGLE_NODE) {
266  cout << " values into linear partitions." << endl;
267  } else {
268  cout << " values." << endl;
269  }
270  }
271 
272  /*--- Check the coordinate name to decide the index for storage. ---*/
273 
274  unsigned short indC = 0;
275  if (string(coordname) == "CoordinateX") indC = 0;
276  else if (string(coordname) == "CoordinateY") indC = 1;
277  else if (string(coordname) == "CoordinateZ") indC = 2;
278  else
279  SU2_MPI::Error(string("Unknown coordinate name, ") + coordname +
280  string(", in the CGNS file."), CURRENT_FUNCTION);
281 
282  /*--- Now read our rank's chunk of coordinates from the file.
283  Ask for datatype RealDouble and let CGNS library do the translation
284  when RealSingle is found. ---*/
285 
286  if (cg_coord_read(cgnsFileID, cgnsBase, cgnsZone, coordname, RealDouble,
287  &range_min, &range_max, localPointCoordinates[indC].data()))
288  cg_error_exit();
289  }

293 void CCGNSMeshReaderFVM::ReadCGNSSectionMetadata() {
294 
295  /*--- Begin section for retrieving the connectivity info. ---*/
296 
297  if ((rank == MASTER_NODE) && (size > SINGLE_NODE))
298  cout << "Distributing connectivity across all ranks." << endl;
299 
300  /*--- First check the number of sections. ---*/
301 
302  if (cg_nsections(cgnsFileID, cgnsBase, cgnsZone, &nSections)) cg_error_exit();
303  if (rank == MASTER_NODE) {
304  cout << "Number of connectivity sections is ";
305  cout << nSections << "." << endl;
306  }
307 
308  /*--- Prepare several data structures to hold the various
309  pieces of information describing each section. ---*/
310 
311  isInterior.resize(nSections);
312  nElems.resize(nSections,0);
313  elemOffset.resize(nSections+1, 0); elemOffset[0] = 0;
314  connElems.resize(nSections);
315  sectionNames.resize(nSections, vector<char>(CGNS_STRING_SIZE));
316  numberOfGlobalElements = 0;
317 
318  for (int s = 0; s < nSections; s++) {
319 
320  /*--- Read the connectivity details for this section. ---*/
321 
322  int nbndry, parent_flag, vtk_type;
323  cgsize_t startE, endE, sizeNeeded;
324  ElementType_t elemType;
325  if (cg_section_read(cgnsFileID, cgnsBase, cgnsZone, s+1,
326  sectionNames[s].data(), &elemType, &startE, &endE,
327  &nbndry, &parent_flag)) cg_error_exit();
328 
329  /*--- Compute the total element count in this section (global). ---*/
330 
331  unsigned long element_count = (endE-startE+1);
332 
333  /* Get the details for the CGNS element type in this section. */
334 
335  string elem_name = GetCGNSElementType(elemType, vtk_type);
336 
337  /* We assume that each section contains interior elements by default.
338  If we find 1D elements in a 2D problem or 2D elements in a 3D
339  problem, then we know the section must contain boundary elements.
340  We assume that each section is composed of either entirely interior
341  or entirely boundary elements. */
342 
343  isInterior[s] = true;
344 
345  if (elemType == MIXED) {
346 
347  /* For a mixed section, we check the type of the first element
348  so that we can correctly label this section as an interior or
349  boundary element section. Here, we also assume that a section
350  can not hold both interior and boundary elements. First, get
351  the size required to read a single element from the section. */
352 
353  if (cg_ElementPartialSize(cgnsFileID, cgnsBase, cgnsZone, s+1, startE,
354  startE, &sizeNeeded) != CG_OK)
355  cg_error_exit();
356 
357  /* A couple of auxiliary vectors for mixed element sections. */
358 
359  vector<cgsize_t> connElemCGNS(sizeNeeded);
360  vector<cgsize_t> connOffsetCGNS(2,0);
361 
362  /* Retrieve the connectivity information for the first element. */
363 
364  if (cg_poly_elements_partial_read(cgnsFileID, cgnsBase, cgnsZone, s+1,
365  startE, startE, connElemCGNS.data(),
366  connOffsetCGNS.data(), NULL) != CG_OK)
367  cg_error_exit();
368 
369  /* The element type is in the first position of the connectivity
370  information that we retrieved from the CGNS file. */
371 
372  elemType = ElementType_t(connElemCGNS[0]);
373 
374  }
375 
376  /* Check for 1D elements in 2D problems, or for 2D elements in
377  3D problems. If found, mark the section as a boundary section. */
378 
379  if ((dimension == 2) &&
380  (elemType == BAR_2 || elemType == BAR_3)) isInterior[s] = false;
381  if ((dimension == 3) &&
382  (elemType == TRI_3 || elemType == QUAD_4)) isInterior[s] = false;
383 
384  /*--- Increment the global element offset for each section
385  based on whether or not this is a surface or volume section.
386  We also keep a running count of the total elements globally. ---*/
387 
388  elemOffset[s+1] = elemOffset[s];
389  if (!isInterior[s]) elemOffset[s+1] += element_count;
390  else numberOfGlobalElements += element_count;
391 
392  /*--- Print some information to the console. ---*/
393 
394  if (rank == MASTER_NODE) {
395  cout << "Section " << string(sectionNames[s].data());
396  cout << " contains " << element_count << " elements";
397  cout << " of type " << elem_name << "." <<endl;
398  }
399 
400  }

404 void CCGNSMeshReaderFVM::ReadCGNSVolumeSection(int val_section) {
405 
406  /*--- In this routine, each rank will read a chunk of the element
407  connectivity for a single specified section of the CGNS mesh file.
408  All operations are executed in parallel here and the reading of the
409  section proceeds based on a linear partitioning of the elements
410  across all ranks in the calculation. We will use partial reads of
411  the CGNS section from the CGNS API to accomplish this. Once each
412  rank has a linear chunk of the mesh, we will redistribute the
413  connectivity to match the linear partitioning of the grid points,
414  not the elements, since the points control the overall partitioning. ---*/
415 
416  int nbndry, parent_flag, npe, iProcessor;
417  unsigned long iElem = 0, iPoint = 0, iNode = 0, jNode = 0;
418  cgsize_t startE, endE;
419  ElementType_t elemType;
420  char sectionName[CGNS_STRING_SIZE];
421 
422  /*--- Read the connectivity details for this section.
423  Store the total number of elements in this section
424  to be used later for memory allocation. ---*/
425 
426  if (cg_section_read(cgnsFileID, cgnsBase, cgnsZone, val_section+1, sectionName,
427  &elemType, &startE, &endE, &nbndry, &parent_flag))
428  cg_error_exit();
429 
430  /*--- Compute element linear partitioning ---*/
431 
432  unsigned long element_count = (endE-startE+1);
433  CLinearPartitioner elementPartitioner(element_count,startE,true);
434 
435  /*--- Store the number of elements that this rank is responsible for
436  in the current section. ---*/
437 
438  nElems[val_section] = elementPartitioner.GetSizeOnRank(rank);
439 
440  /*--- Allocate some memory for the handling the connectivity
441  and auxiliary data that we need to communicate. ---*/
442 
443  vector<cgsize_t> elemTypes(nElems[val_section], 0);
444  vector<cgsize_t> nPoinPerElem(nElems[val_section],0);
445  vector<cgsize_t> elemGlobalID(nElems[val_section],0);
446 
447  /*--- Determine the size of the vector needed to read the connectivity
448  data from the CGNS file. Only call the CGNS API if we have a non-zero
449  number of elements on this rank. ---*/
450 
451  cgsize_t sizeNeeded = 0, sizeOffset = 0;
452  if (nElems[val_section] > 0) {
453  if (cg_ElementPartialSize(cgnsFileID, cgnsBase, cgnsZone, val_section+1,
454  (cgsize_t)elementPartitioner.GetFirstIndexOnRank(rank),
455  (cgsize_t)elementPartitioner.GetLastIndexOnRank(rank),
456  &sizeNeeded) != CG_OK)
457  cg_error_exit();
458  }
459 
460  /*--- Allocate the memory for the connectivity, the offset if needed
461  and read the data. ---*/
462 
463  vector<cgsize_t> connElemCGNS(sizeNeeded,0);
464  if (elemType == MIXED || elemType == NFACE_n || elemType == NGON_n) {
465  sizeOffset = nElems[val_section]+1;
466  }
467  vector<cgsize_t> connOffsetCGNS(sizeOffset,0);
468 
469  /*--- Retrieve the connectivity information and store. Note that
470  we are only accessing our rank's piece of the data here in the
471  partial read function in the CGNS API. Only call the CGNS API
472  if we have a non-zero number of elements on this rank. ---*/
473 
474  if (nElems[val_section] > 0) {
475  if (elemType == MIXED || elemType == NFACE_n || elemType == NGON_n) {
476  if (cg_poly_elements_partial_read(cgnsFileID, cgnsBase, cgnsZone, val_section+1,
477  (cgsize_t)elementPartitioner.GetFirstIndexOnRank(rank),
478  (cgsize_t)elementPartitioner.GetLastIndexOnRank(rank),
479  connElemCGNS.data(),
480  connOffsetCGNS.data(), NULL) != CG_OK)
481  cg_error_exit();
482  } else {
483  if (cg_elements_partial_read(cgnsFileID, cgnsBase, cgnsZone, val_section+1,
484  (cgsize_t)elementPartitioner.GetFirstIndexOnRank(rank),
485  (cgsize_t)elementPartitioner.GetLastIndexOnRank(rank),
486  connElemCGNS.data(), NULL) != CG_OK)
487  cg_error_exit();
488  }
489  }
490 
491  /*--- Print some information to the console. ---*/
492 
493  if (rank == MASTER_NODE) {
494  cout << "Loading volume section " << string(sectionName);
495  cout << " from file." << endl;
496  }
497 
498  /*--- Find the number of nodes required to represent
499  this type of element. ---*/
500 
501  if (cg_npe(elemType, &npe)) cg_error_exit();
502 
503  /*--- Check whether the sections contains a mixture of multiple
504  element types, which will require special handling to get the
505  element type one-by-one when reading. ---*/
506 
507  bool isMixed = false;
508  if (elemType == MIXED) {
509  isMixed = true;
510  }
511 
512  /*--- Loop through all of the elements in this section to get more
513  information and to decide whether it has interior elements. ---*/
514 
515  unsigned long counterCGNS = 0;
516  for (iElem = 0; iElem < nElems[val_section]; iElem++) {
517 
518  ElementType_t iElemType = elemType;
519 
520  /*--- If we have a mixed element section, we need to check the elem
521  type one-by-one. We also must manually advance the counter to the
522  next element's position in the buffer. ---*/
523 
524  if (isMixed) {
525  iElemType = ElementType_t(connElemCGNS[counterCGNS]);
526  npe = connOffsetCGNS[iElem+1]-connOffsetCGNS[iElem]-1;
527  counterCGNS++;
528  for (int jj = 0; jj < npe; jj++) counterCGNS++;
529  }
530 
531  /*--- Store the number of points per element for the current elem. ---*/
532 
533  nPoinPerElem[iElem] = npe;
534 
535  /*--- Store the global ID for this element. Note the -1 to move
536  from CGNS convention to SU2 convention. We also subtract off
537  an additional offset in case we have found boundary sections
538  prior to this one, in order to keep the internal element global
539  IDs indexed starting from zero. ---*/
540 
541  elemGlobalID[iElem] = (elementPartitioner.GetFirstIndexOnRank(rank) +
542  iElem - elemOffset[val_section]);
543 
544  /* Get the VTK type for this element. */
545 
546  int vtk_type;
547  string elem_name = GetCGNSElementType(iElemType, vtk_type);
548  elemTypes[iElem] = vtk_type;
549 
550  }
551 
552  /*--- Force free the memory for the conn offset from the CGNS file. ---*/
553 
554  vector<cgsize_t>().swap(connOffsetCGNS);
555 
556  /*--- These are internal elems. Allocate memory on each proc. ---*/
557 
558  vector<cgsize_t> connElemTemp(nElems[val_section]*SU2_CONN_SIZE,0);
559 
560  /*--- Copy the connectivity into the larger array with a standard
561  format per element: [globalID vtkType n0 n1 n2 n3 n4 n5 n6 n7 n8]. ---*/
562 
563  counterCGNS = 0;
564  for (iElem = 0; iElem < nElems[val_section]; iElem++) {
565 
566  /*--- Store the conn in chunks of SU2_CONN_SIZE for simplicity. ---*/
567 
568  unsigned long nn = iElem*SU2_CONN_SIZE;
569 
570  /*--- First, store the global element ID and the VTK type. ---*/
571 
572  connElemTemp[nn] = elemGlobalID[iElem]; nn++;
573  connElemTemp[nn] = elemTypes[iElem]; nn++;
574 
575  /*--- Store the connectivity values. Note we subtract one from
576  the CGNS 1-based convention. We may also need to remove the first
577  entry if this is a mixed element section. ---*/
578 
579  if (isMixed) counterCGNS++;
580  for (iNode = 0; iNode < (unsigned long)nPoinPerElem[iElem]; iNode++) {
581  connElemTemp[nn] = connElemCGNS[counterCGNS + iNode] - 1; nn++;
582  }
583  counterCGNS += nPoinPerElem[iElem];
584 
585  }
586 
587  /*--- Force free the memory for the conn from the CGNS file. ---*/
588 
589  vector<cgsize_t>().swap(connElemCGNS);
590 
591  /*--- We now have the connectivity stored in linearly partitioned
592  chunks. We need to loop through and decide how many elements we
593  must send to each rank in order to have all elements that
594  surround a particular "owned" node on each rank (i.e., elements
595  will appear on multiple ranks). First, initialize a counter
596  and flag. ---*/
597 
598  int *nElem_Send = new int[size+1]; nElem_Send[0] = 0;
599  int *nElem_Recv = new int[size+1]; nElem_Recv[0] = 0;
600  int *nElem_Flag = new int[size];
601 
602  for (iProcessor = 0; iProcessor < size; iProcessor++) {
603  nElem_Send[iProcessor] = 0;
604  nElem_Recv[iProcessor] = 0;
605  nElem_Flag[iProcessor]= -1;
606  }
607  nElem_Send[size] = 0;
608  nElem_Recv[size] = 0;
609 
610  /*--- Create a partitioner object to find the owning rank of points. ---*/
611 
612  CLinearPartitioner pointPartitioner(numberOfGlobalPoints,0);
613 
614  for (iElem = 0; iElem < nElems[val_section]; iElem++) {
615  for (iNode = 0; iNode < (unsigned long)nPoinPerElem[iElem]; iNode++) {
616 
617  /*--- Get the index of the current point. ---*/
618 
619  iPoint = connElemTemp[iElem*SU2_CONN_SIZE + SU2_CONN_SKIP + iNode];
620 
621  /*--- Search for the processor that owns this point. ---*/
622 
623  iProcessor = pointPartitioner.GetRankContainingIndex(iPoint);
624 
625  /*--- If we have not visited this element yet, increment our
626  number of elements that must be sent to a particular proc. ---*/
627 
628  if ((nElem_Flag[iProcessor] != (int)iElem)) {
629  nElem_Flag[iProcessor] = iElem;
630  nElem_Send[iProcessor+1]++;
631  }
632 
633  }
634  }
635 
636  /*--- Communicate the number of cells to be sent/recv'd amongst
637  all processors. After this communication, each proc knows how
638  many cells it will receive from each other processor. ---*/
639 
640  SU2_MPI::Alltoall(&(nElem_Send[1]), 1, MPI_INT,
641  &(nElem_Recv[1]), 1, MPI_INT, SU2_MPI::GetComm());
642 
643  /*--- Prepare to send connectivities. First check how many
644  messages we will be sending and receiving. Here we also put
645  the counters into cumulative storage format to make the
646  communications simpler. ---*/
647 
648  unsigned long nSends = 0, nRecvs = 0;
649  for (iProcessor = 0; iProcessor < size; iProcessor++)
650  nElem_Flag[iProcessor] = -1;
651 
652  for (iProcessor = 0; iProcessor < size; iProcessor++) {
653  if ((iProcessor != rank) && (nElem_Send[iProcessor+1] > 0)) nSends++;
654  if ((iProcessor != rank) && (nElem_Recv[iProcessor+1] > 0)) nRecvs++;
655 
656  nElem_Send[iProcessor+1] += nElem_Send[iProcessor];
657  nElem_Recv[iProcessor+1] += nElem_Recv[iProcessor];
658  }
659 
660  /*--- Allocate memory to hold the connectivity that we are
661  sending. Note that we are also sending the global ID and the
662  VTK element type in the first and second positions, respectively.
663  We have assumed a constant message size of a hex element (8 nodes)
664  + 2 extra values for the ID and VTK. ---*/
665 
666  unsigned long *connSend = NULL, iSend = 0;
667  unsigned long sendSize = (unsigned long)SU2_CONN_SIZE*nElem_Send[size];
668  connSend = new unsigned long[sendSize];
669  for (iSend = 0; iSend < sendSize; iSend++)
670  connSend[iSend] = 0;
671 
672  /*--- Create an index variable to keep track of our index
673  position as we load up the send buffer. ---*/
674 
675  vector<unsigned long> index(size);
676  for (iProcessor = 0; iProcessor < size; iProcessor++)
677  index[iProcessor] = SU2_CONN_SIZE*nElem_Send[iProcessor];
678 
679  /*--- Loop through our elements and load the elems and their
680  additional data that we will send to the other procs. ---*/
681 
682  for (iElem = 0; iElem < (unsigned long)nElems[val_section]; iElem++) {
683  for (iNode = 0; iNode < (unsigned long)nPoinPerElem[iElem]; iNode++) {
684 
685  /*--- Get the index of the current point. ---*/
686 
687  iPoint = connElemTemp[iElem*SU2_CONN_SIZE + SU2_CONN_SKIP + iNode];
688 
689  /*--- Search for the processor that owns this point ---*/
690 
691  iProcessor = pointPartitioner.GetRankContainingIndex(iPoint);
692 
693  /*--- Load connectivity into the buffer for sending ---*/
694 
695  if (nElem_Flag[iProcessor] != (int)iElem) {
696 
697  nElem_Flag[iProcessor] = iElem;
698  unsigned long nn = index[iProcessor];
699 
700  /*--- Load the VTK type first into the conn array,
701  then the connectivity vals, and last, the global ID. ---*/
702 
703  for (jNode = 0; jNode < SU2_CONN_SIZE; jNode++) {
704  connSend[nn] = connElemTemp[iElem*SU2_CONN_SIZE + jNode]; nn++;
705  }
706 
707  /*--- Increment the index by the message length ---*/
708 
709  index[iProcessor] += SU2_CONN_SIZE;
710 
711  }
712 
713  }
714  }
715 
716  /*--- Force free memory after loading up the send buffer. ---*/
717 
718  vector<cgsize_t>().swap(connElemTemp);
719  vector<cgsize_t>().swap(elemTypes);
720  vector<cgsize_t>().swap(nPoinPerElem);
721  vector<cgsize_t>().swap(elemGlobalID);
722 
723  vector<unsigned long>().swap(index);
724 
725  /*--- Allocate the memory that we need for receiving the conn
726  values and then cue up the non-blocking receives. Note that
727  we do not include our own rank in the communications. We will
728  directly copy our own data later. ---*/
729 
730  unsigned long *connRecv = NULL, iRecv = 0;
731  unsigned long recvSize = (unsigned long)SU2_CONN_SIZE*nElem_Recv[size];
732  connRecv = new unsigned long[recvSize];
733  for (iRecv = 0; iRecv < recvSize; iRecv++)
734  connRecv[iRecv] = 0;
735 
736  /*--- Allocate memory for the MPI requests if we will communicate. ---*/
737 
738  SU2_MPI::Request *connSendReq = NULL;
739  SU2_MPI::Request *connRecvReq = NULL;
740 
741  if (nSends > 0) {
742  connSendReq = new SU2_MPI::Request[nSends];
743  }
744  if (nRecvs > 0) {
745  connRecvReq = new SU2_MPI::Request[nRecvs];
746  }
747 
748  /*--- Launch the non-blocking sends and receives. ---*/
749 
750  InitiateCommsAll(connSend, nElem_Send, connSendReq,
751  connRecv, nElem_Recv, connRecvReq,
752  SU2_CONN_SIZE, COMM_TYPE_UNSIGNED_LONG);
753 
754  /*--- Copy the current rank's data into the recv buffer directly. ---*/
755 
756  iRecv = SU2_CONN_SIZE*nElem_Recv[rank];
757  unsigned long myStart = SU2_CONN_SIZE*nElem_Send[rank];
758  unsigned long myFinal = SU2_CONN_SIZE*nElem_Send[rank+1];
759  for (iSend = myStart; iSend < myFinal; iSend++) {
760  connRecv[iRecv] = connSend[iSend];
761  iRecv++;
762  }
763 
764  /*--- Complete the non-blocking communications. ---*/
765 
766  CompleteCommsAll(nSends, connSendReq, nRecvs, connRecvReq);
767 
768  /*--- Store the connectivity for this rank in the proper data
769  structure. First, allocate the appropriate amount of memory
770  for this section, then write the recv'd values. ---*/
771 
772  if (nElem_Recv[size] > 0) {
773  connElems[val_section].resize(nElem_Recv[size]*SU2_CONN_SIZE,0);
774  unsigned long count = 0;
775  for (iElem = 0; iElem < (unsigned long)nElem_Recv[size]; iElem++) {
776  for (iNode = 0; iNode < SU2_CONN_SIZE; iNode++) {
777  unsigned long nn = iElem*SU2_CONN_SIZE+iNode;
778  connElems[val_section][count] = (cgsize_t)connRecv[nn];
779  count++;
780  }
781  }
782 
783  /*--- Store the total number of elements the current rank
784  now has for the current section after completing the comms. ---*/
785 
786  nElems[val_section] = nElem_Recv[size];
787 
788  } else {
789 
790  /*--- The current rank did not recv any elements from this
791  section. Set the count to zero and nullify the data structure. ---*/
792 
793  nElems[val_section] = 0;
794  connElems[val_section].resize(0);
795 
796  }
797 
798  /*--- Free temporary memory from communications ---*/
799 
800  if (connSendReq != NULL) delete [] connSendReq;
801  if (connRecvReq != NULL) delete [] connRecvReq;
802 
803  delete [] connSend;
804  delete [] connRecv;
805  delete [] nElem_Recv;
806  delete [] nElem_Send;
807  delete [] nElem_Flag;
808 
809 }

811 void CCGNSMeshReaderFVM::ReadCGNSSurfaceSection(int val_section) {
812 
813  /*--- In this routine, we access a CGNS surface section and have the
814  master rank load all of the surface conn. This can help avoid issues
815  where there are fewer elements than ranks on a surface. This is later
816  linearly partitioned. A limitation of this approach is that there
817  could be a memory bottleneck for extremely large grids, but this has
818  not been reached yet in practice. In that event, we can also read these
819  surface sections with a linear partitioning. ---*/
820 
821  int nbndry, parent_flag, npe;
822  unsigned long iElem = 0, iNode = 0;
823  cgsize_t startE, endE, ElementDataSize;
824  ElementType_t elemType;
825  char sectionName[CGNS_STRING_SIZE];
826 
827  if (rank == MASTER_NODE) {
828 
829  /*--- Allocate some memory for the handling the connectivity
830  and auxiliary data that we are need to communicate. ---*/
831 
832  vector<cgsize_t> connElemCGNS(nElems[val_section]*SU2_CONN_SIZE,0);
833  vector<unsigned short> elemTypes(nElems[val_section],0);
834  vector<unsigned short> nPoinPerElem(nElems[val_section],0);
835  vector<unsigned long> elemGlobalID(nElems[val_section],0);
836 
837  /*--- Read the section info again ---*/
838 
839  if (cg_section_read(cgnsFileID, cgnsBase, cgnsZone, val_section+1,
840  sectionName, &elemType, &startE, &endE, &nbndry,
841  &parent_flag))
842  cg_error_exit();
843 
844  /*--- Print some information to the console. ---*/
845 
846  cout << "Loading surface section " << string(sectionName);
847  cout << " from file." << endl;
848 
849  /*--- Store the number of elems (all on the master). ---*/
850 
851  nElems[val_section] = (endE-startE+1);
852 
853  /*--- Read and store the total amount of data that will be
854  listed when reading this section. ---*/
855 
856  if (cg_ElementDataSize(cgnsFileID, cgnsBase, cgnsZone, val_section+1,
857  &ElementDataSize))
858  cg_error_exit();
859 
860  /*--- Find the number of nodes required to represent
861  this type of element. ---*/
862 
863  if (cg_npe(elemType, &npe)) cg_error_exit();
864 
865  /*--- Check whether the sections contains a mixture of multiple
866  element types, which will require special handling to get the
867  element type one-by-one when reading. ---*/
868 
869  bool isMixed = false;
870  if (elemType == MIXED) {
871  isMixed = true;
872  }
873 
874  /*--- Allocate memory for accessing the connectivity and to
875  store it in the proper data structure for post-processing. ---*/
876 
877  vector<cgsize_t> connElemTemp(ElementDataSize,0);
878 
879  /*--- Retrieve the connectivity information and store. ---*/
880 
881  if (elemType == MIXED || elemType == NGON_n || elemType == NFACE_n) {
882  vector<cgsize_t> connOffsetTemp(nElems[val_section]+1, 0);
883  if (cg_poly_elements_partial_read(cgnsFileID, cgnsBase, cgnsZone,
884  val_section+1, startE, endE,
885  connElemTemp.data(),
886  connOffsetTemp.data(), NULL) != CG_OK)
887  cg_error_exit();
888  } else {
889  if (cg_elements_read(cgnsFileID, cgnsBase, cgnsZone, val_section+1,
890  connElemTemp.data(), NULL))
891  cg_error_exit();
892  }
893 
894  /*--- Allocate the memory for the data structure used to carry
895  the connectivity for this section. ---*/
896 
897  connElems[val_section].resize(nElems[val_section]*SU2_CONN_SIZE,0);
898 
899  unsigned long counterCGNS = 0;
900  for (iElem = 0; iElem < nElems[val_section]; iElem++) {
901 
902  ElementType_t iElemType = elemType;
903 
904  /*--- If we have a mixed element section, we need to check the elem
905  type one-by-one. We also must manually advance the counter. ---*/
906 
907  if (isMixed) {
908  iElemType = ElementType_t(connElemTemp[counterCGNS]);
909  counterCGNS++;
910  }
911 
912  /*--- Get the VTK type for this element. ---*/
913 
914  int vtk_type;
915  string elem_name = GetCGNSElementType(iElemType, vtk_type);
916 
917  /*--- Get the number of nodes per element. ---*/
918 
919  cg_npe(iElemType, &npe);
920 
921  /*--- Load the surface element connectivity into the SU2 data
922  structure with format: [globalID VTK n1 n2 n3 n4 n5 n6 n7 n8].
923  We do not need a global ID for the surface elements, so we
924  simply set that to zero to maintain the same data structure
925  format as the interior elements. Note that we subtract 1 to
926  move from the CGNS 1-based indexing to SU2's zero-based. ---*/
927 
928  connElems[val_section][iElem*SU2_CONN_SIZE+0] = 0;
929  connElems[val_section][iElem*SU2_CONN_SIZE+1] = vtk_type;
930  for (iNode = 0; iNode < (unsigned long)npe; iNode++) {
931  unsigned long nn = iElem*SU2_CONN_SIZE+SU2_CONN_SKIP+iNode;
932  connElems[val_section][nn] = connElemTemp[counterCGNS] - 1;
933  counterCGNS++;
934  }
935 
936  }
937 
938  } else {
939 
940  /*--- We are not the master, so we resize to zero for safety. ---*/
941 
942  nElems[val_section] = 0;
943  connElems[val_section].resize(0);
944 
945  }
946 
947 }

1087 void CCGNSMeshReaderFVM::InitiateCommsAll(void *bufSend,
1088  const int *nElemSend,
1089  SU2_MPI::Request *sendReq,
1090  void *bufRecv,
1091  const int *nElemRecv,
1092  SU2_MPI::Request *recvReq,
1093  unsigned short countPerElem,
1094  unsigned short commType) {
1095 
1096  /*--- Local variables ---*/
1097 
1098  int iMessage, iProc, offset, nElem, count, source, dest, tag;
1099 
1100  /*--- Launch the non-blocking recv's first. ---*/
1101 
1102  iMessage = 0;
1103  for (iProc = 0; iProc < size; iProc++) {
1104 
1105  /*--- Post recv's only if another proc is sending us data. We do
1106  not communicate with ourselves or post recv's for zero length
1107  messages to keep overhead down. ---*/
1108 
1109  if ((nElemRecv[iProc+1] > nElemRecv[iProc]) && (iProc != rank)) {
1110 
1111  /*--- Compute our location in the recv buffer. ---*/
1112 
1113  offset = countPerElem*nElemRecv[iProc];
1114 
1115  /*--- Take advantage of cumulative storage format to get the number
1116  of elems that we need to recv. ---*/
1117 
1118  nElem = nElemRecv[iProc+1] - nElemRecv[iProc];
1119 
1120  /*--- Total count can include multiple pieces of data per element. ---*/
1121 
1122  count = countPerElem*nElem;
1123 
1124  /*--- Post non-blocking recv for this proc. ---*/
1125 
1126  source = iProc; tag = iProc + 1;
1127 
1128  switch (commType) {
1129  case COMM_TYPE_DOUBLE:
1130  SU2_MPI::Irecv(&(static_cast<su2double*>(bufRecv)[offset]),
1131  count, MPI_DOUBLE, source, tag, SU2_MPI::GetComm(),
1132  &(recvReq[iMessage]));
1133  break;
1134  case COMM_TYPE_UNSIGNED_LONG:
1135  SU2_MPI::Irecv(&(static_cast<unsigned long*>(bufRecv)[offset]),
1136  count, MPI_UNSIGNED_LONG, source, tag, SU2_MPI::GetComm(),
1137  &(recvReq[iMessage]));
1138  break;
1139  case COMM_TYPE_LONG:
1140  SU2_MPI::Irecv(&(static_cast<long*>(bufRecv)[offset]),
1141  count, MPI_LONG, source, tag, SU2_MPI::GetComm(),
1142  &(recvReq[iMessage]));
1143  break;
1144  case COMM_TYPE_UNSIGNED_SHORT:
1145  SU2_MPI::Irecv(&(static_cast<unsigned short*>(bufRecv)[offset]),
1146  count, MPI_UNSIGNED_SHORT, source, tag, SU2_MPI::GetComm(),
1147  &(recvReq[iMessage]));
1148  break;
1149  case COMM_TYPE_CHAR:
1150  SU2_MPI::Irecv(&(static_cast<char*>(bufRecv)[offset]),
1151  count, MPI_CHAR, source, tag, SU2_MPI::GetComm(),
1152  &(recvReq[iMessage]));
1153  break;
1154  case COMM_TYPE_SHORT:
1155  SU2_MPI::Irecv(&(static_cast<short*>(bufRecv)[offset]),
1156  count, MPI_SHORT, source, tag, SU2_MPI::GetComm(),
1157  &(recvReq[iMessage]));
1158  break;
1159  case COMM_TYPE_INT:
1160  SU2_MPI::Irecv(&(static_cast<int*>(bufRecv)[offset]),
1161  count, MPI_INT, source, tag, SU2_MPI::GetComm(),
1162  &(recvReq[iMessage]));
1163  break;
1164  default:
1165  break;
1166  }
1167 
1168  /*--- Increment message counter. ---*/
1169 
1170  iMessage++;
1171 
1172  }
1173  }
1174 
1175  /*--- Launch the non-blocking sends next. ---*/
1176 
1177  iMessage = 0;
1178  for (iProc = 0; iProc < size; iProc++) {
1179 
1180  /*--- Post sends only if we are sending another proc data. We do
1181  not communicate with ourselves or post sends for zero length
1182  messages to keep overhead down. ---*/
1183 
1184  if ((nElemSend[iProc+1] > nElemSend[iProc]) && (iProc != rank)) {
1185 
1186  /*--- Compute our location in the send buffer. ---*/
1187 
1188  offset = countPerElem*nElemSend[iProc];
1189 
1190  /*--- Take advantage of cumulative storage format to get the number
1191  of elems that we need to send. ---*/
1192 
1193  nElem = nElemSend[iProc+1] - nElemSend[iProc];
1194 
1195  /*--- Total count can include multiple pieces of data per element. ---*/
1196 
1197  count = countPerElem*nElem;
1198 
1199  /*--- Post non-blocking send for this proc. ---*/
1200 
1201  dest = iProc; tag = rank + 1;
1202 
1203  switch (commType) {
1204  case COMM_TYPE_DOUBLE:
1205  SU2_MPI::Isend(&(static_cast<su2double*>(bufSend)[offset]),
1206  count, MPI_DOUBLE, dest, tag, SU2_MPI::GetComm(),
1207  &(sendReq[iMessage]));
1208  break;
1209  case COMM_TYPE_UNSIGNED_LONG:
1210  SU2_MPI::Isend(&(static_cast<unsigned long*>(bufSend)[offset]),
1211  count, MPI_UNSIGNED_LONG, dest, tag, SU2_MPI::GetComm(),
1212  &(sendReq[iMessage]));
1213  break;
1214  case COMM_TYPE_LONG:
1215  SU2_MPI::Isend(&(static_cast<long*>(bufSend)[offset]),
1216  count, MPI_LONG, dest, tag, SU2_MPI::GetComm(),
1217  &(sendReq[iMessage]));
1218  break;
1219  case COMM_TYPE_UNSIGNED_SHORT:
1220  SU2_MPI::Isend(&(static_cast<unsigned short*>(bufSend)[offset]),
1221  count, MPI_UNSIGNED_SHORT, dest, tag, SU2_MPI::GetComm(),
1222  &(sendReq[iMessage]));
1223  break;
1224  case COMM_TYPE_CHAR:
1225  SU2_MPI::Isend(&(static_cast<char*>(bufSend)[offset]),
