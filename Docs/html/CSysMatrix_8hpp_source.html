384  /*!
385  * \brief Get a pointer to the start of block "ij"
386  * \param[in] block_i - Row index.
387  * \param[in] block_j - Column index.
388  * \return Pointer to location in memory where the block starts.
389  */
390  FORCEINLINE const ScalarType *GetBlock(unsigned long block_i, unsigned long block_j) const {
391  /*--- The position of the diagonal block is known which allows halving the search space. ---*/
392  const auto end = (block_j<block_i)? dia_ptr[block_i] : row_ptr[block_i+1];
393  for (auto index = (block_j<block_i)? row_ptr[block_i] : dia_ptr[block_i]; index < end; ++index)
394  if (col_ind[index] == block_j)
395  return &matrix[index*nVar*nEqn];
396  return nullptr;
397  }

422  /*!
423  * \brief Set the value of a block (in flat format) in the sparse matrix with scaling.
424  * \note If the template param Overwrite is false we add to the block (bij += alpha*b).
425  * \param[in] block_i - Row index.
426  * \param[in] block_j - Column index.
427  * \param[in] val_block - Block to set to A(i, j).
428  * \param[in] alpha - Scale factor.
429  */
430  template<class OtherType, bool Overwrite = true,
431  su2enable_if<!is_pointer<OtherType>::value> = 0>
432  inline void SetBlock(unsigned long block_i, unsigned long block_j,
433  const OtherType *val_block, OtherType alpha = 1.0) {
434 
435  auto mat_ij = GetBlock(block_i, block_j);
436  if (!mat_ij) return;
437  SU2_OMP_SIMD
438  for (auto iVar = 0ul; iVar < nVar*nEqn; ++iVar) {
439  mat_ij[iVar] = (Overwrite? ScalarType(0) : mat_ij[iVar]) + PassiveAssign(alpha * val_block[iVar]);
440  }
441  }

464  template<class OtherType, bool Overwrite = true>
465  inline void SetBlock(unsigned long block_i, unsigned long block_j,
466  const OtherType* const* val_block, OtherType alpha = 1.0) {
467 
468  auto mat_ij = GetBlock(block_i, block_j);
469  if (!mat_ij) return;
470  for (auto iVar = 0ul; iVar < nVar; ++iVar) {
471  for (auto jVar = 0ul; jVar < nEqn; ++jVar) {
472  *mat_ij = (Overwrite? ScalarType(0) : *mat_ij) + PassiveAssign(alpha * val_block[iVar][jVar]);
473  ++mat_ij;
474  }
475  }
476  }

502  /*!
503  * \brief Set the value of a scaled block in the sparse matrix.
504  * \note This is an templated overload for C2Dcontainer specialization su2matrix.
505  * It assumes that MatrixType supports a member type Scalar and access operator[][].
506  * If the template param Overwrite is false we add to the block (bij += alpha*b).
507  * \param[in] block_i - Row index.
508  * \param[in] block_j - Column index.
509  * \param[in] val_block - Block to set to A(i, j).
510  * \param[in] alpha - Scale factor.
511  */
512  template <class MatrixType, bool Overwrite = true>
513  inline void SetBlock(unsigned long block_i, unsigned long block_j, MatrixType& val_block,
514  typename MatrixType::Scalar alpha = 1.0) {
515  auto mat_ij = GetBlock(block_i, block_j);
516  if (!mat_ij) return;

518  for (auto jVar = 0ul; jVar < nEqn; ++jVar) {

579  /*!
580  * \brief SIMD version, does the update for multiple edges and points.
581  * \note Nothing is updated if the mask is 0.
582  */
583  template<class MatTypeSIMD, size_t N, class I, class F = ScalarType>
584  FORCEINLINE void UpdateBlocks(simd::Array<I,N> iEdge, simd::Array<I,N> iPoint, simd::Array<I,N> jPoint,
585  const MatTypeSIMD& block_i, const MatTypeSIMD& block_j, simd::Array<F,N> mask = 1) {
586 
587  static_assert(MatTypeSIMD::StaticSize, "This method requires static size blocks.");
588  static_assert(MatTypeSIMD::IsRowMajor, "Block storage is not compatible with matrix.");
589  constexpr size_t blkSz = MatTypeSIMD::StaticSize;
590  assert(blkSz == nVar*nEqn);
591 
592  /*--- "Transpose" the blocks, scale, and possibly convert types,
593  * giving the compiler the chance to vectorize all of these. ---*/
594  ScalarType blk_i[N][blkSz], blk_j[N][blkSz];
595 
596  for (size_t i=0; i<blkSz; ++i) {
597  SU2_OMP_SIMD_IF_NOT_AD
598  for (size_t k=0; k<N; ++k) {
599  blk_i[k][i] = PassiveAssign(-mask[k] * block_i.data()[i][k]);
600  blk_j[k][i] = PassiveAssign(mask[k] * block_j.data()[i][k]);
601  }
602  }
603 
604  /*--- Update one by one skipping if mask is 0. ---*/
605  for (size_t k=0; k<N; ++k) {
606  if (mask[k]==0) continue;
607 
608  /*--- Fetch the blocks. ---*/
609  auto bii = &matrix[dia_ptr[iPoint[k]]*blkSz];
610  auto bjj = &matrix[dia_ptr[jPoint[k]]*blkSz];
611  auto bij = &matrix[edge_ptr(iEdge[k],0)*blkSz];
612  auto bji = &matrix[edge_ptr(iEdge[k],1)*blkSz];
613 
614  /*--- Update, block i was negated during transpose in the
615  * hope the assignments below become non-temporal stores. ---*/
616  SU2_OMP_SIMD
617  for (size_t i=0; i<blkSz; ++i) {
618  bii[i] -= blk_i[k][i];
619  bjj[i] -= blk_j[k][i];
620  bij[i] = blk_j[k][i];
621  bji[i] = blk_i[k][i];
622  }
623  }
624  }
