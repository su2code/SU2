40  * \class CTablePrinter
41  * \brief Class for writing output in a table.
42  *
43  * For example the output
44  *
45  * +-------------------------------------------+
46  * | MG Level| Presmooth|PostSmooth|CorrectSmo|
47  * +-------------------------------------------+
48  * | 0| 1| 0| 0|
49  * | 1| 1| 0| 0|
50  * | 2| 1| 0| 0|
51  * | 3| 1| 0| 0|
52  * +-------------------------------------------+
53  *
54  *
55  * can be generated with the code
56  *
57  * CTablePrinter MGTable(&std::cout);
58  * MGTable.AddColumn("MG Level", 10);
59  * MGTable.AddColumn("Presmooth", 10);
60  * MGTable.AddColumn("PostSmooth", 10);
61  * MGTable.AddColumn("CorrectSmooth", 10);
62  * MGTable.PrintHeader();
63  * for (unsigned short iLevel = 0; iLevel < nMGLevels+1; iLevel++) {
64  * MGTable << iLevel << MG_PreSmooth[iLevel] << MG_PostSmooth[iLevel] << MG_CorrecSmooth[iLevel];

146  template<typename T> CTablePrinter& operator<<(T input){
147 
148  int indent = 0;
149 
150  /* --- Set the left separator --- */
151  if (j_ == 0)
152  *out_stream_ << separator_;
153 
154  /* --- Determine and set the current alignment in the stream --- */
155  if(align_ == LEFT)
156  *out_stream_ << std::left;
157  else if (align_ == RIGHT || align_ == CENTER)
158  *out_stream_ << std::right;
159 
160  /*--- Print the current column value to the stream --- */
161  *out_stream_ << std::setw(column_widths_.at(j_) - indent)
162  << std::setprecision(precision_) << input;
163 
164  /*--- Reset the column counter and if it is the last column,
165  * add also a line break ---*/
166  if (j_ == GetNumColumns()-1){
167  *out_stream_ << std::setw(indent+1+(int)separator_.size()) << separator_ + "\n";
168  i_ = i_ + 1;
169  j_ = 0;
170  } else {
171  *out_stream_ << std::setw(indent+(int)inner_separator_.size()) << inner_separator_;
172  j_ = j_ + 1;
173  }
174 
175  return *this;
176  }
