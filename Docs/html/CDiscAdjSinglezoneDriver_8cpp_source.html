152 void CDiscAdjSinglezoneDriver::Run() {
153 
154  CQuasiNewtonInvLeastSquares<passivedouble> fixPtCorrector;
155  if (config->GetnQuasiNewtonSamples() > 1) {
156  fixPtCorrector.resize(config->GetnQuasiNewtonSamples(),
157  geometry_container[ZONE_0][INST_0][MESH_0]->GetnPoint(),
158  GetTotalNumberOfVariables(ZONE_0,true),
159  geometry_container[ZONE_0][INST_0][MESH_0]->GetnPointDomain());
160 
161  if (TimeIter != 0) GetAllSolutions(ZONE_0, true, fixPtCorrector);
162  }
163 
164  for (auto Adjoint_Iter = 0ul; Adjoint_Iter < nAdjoint_Iter; Adjoint_Iter++) {
165 
166  /*--- Initialize the adjoint of the output variables of the iteration with the adjoint solution
167  *--- of the previous iteration. The values are passed to the AD tool.
168  *--- Issues with iteration number should be dealt with once the output structure is in place. ---*/
169 
170  config->SetInnerIter(Adjoint_Iter);
171 
172  iteration->InitializeAdjoint(solver_container, geometry_container, config_container, ZONE_0, INST_0);
173 
174  /*--- Initialize the adjoint of the objective function with 1.0. ---*/
175 
176  SetAdj_ObjFunction();
177 
178  /*--- Interpret the stored information by calling the corresponding routine of the AD tool. ---*/
179 
180  AD::ComputeAdjoint();
181 
182  /*--- Extract the computed adjoint values of the input variables and store them for the next iteration. ---*/
183 
184  iteration->IterateDiscAdj(geometry_container, solver_container,
185  config_container, ZONE_0, INST_0, false);
186 
187  /*--- Monitor the pseudo-time ---*/
188 
189  StopCalc = iteration->Monitor(output_container[ZONE_0], integration_container, geometry_container,
190  solver_container, numerics_container, config_container,
191  surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
192 
193  /*--- Clear the stored adjoint information to be ready for a new evaluation. ---*/
194 
195  AD::ClearAdjoints();
196 
197  /*--- Output files for steady state simulations. ---*/
198 
199  if (!config->GetTime_Domain()) {
200  iteration->Output(output_container[ZONE_0], geometry_container, solver_container,
201  config_container, Adjoint_Iter, false, ZONE_0, INST_0);
202  }
203 
204  if (StopCalc) break;
205 
206  /*--- Correct the solution with the quasi-Newton approach. ---*/
207 
208  if (fixPtCorrector.size()) {
209  GetAllSolutions(ZONE_0, true, fixPtCorrector.FPresult());
210  SetAllSolutions(ZONE_0, true, fixPtCorrector.compute());
211  }
212 
213  }

246 void CDiscAdjSinglezoneDriver::SetRecording(RECORDING kind_recording){
247 
248  AD::Reset();
249 
250  /*--- Prepare for recording by resetting the solution to the initial converged solution. ---*/
251 
252  for (unsigned short iSol=0; iSol < MAX_SOLS; iSol++) {
253  for (unsigned short iMesh = 0; iMesh <= config_container[ZONE_0]->GetnMGLevels(); iMesh++) {
254  auto solver = solver_container[ZONE_0][INST_0][iMesh][iSol];
255  if (solver && solver->GetAdjoint()) {
256  solver->SetRecording(geometry_container[ZONE_0][INST_0][iMesh], config_container[ZONE_0]);
257  }
258  }
259  }
260 
261  if (rank == MASTER_NODE) {
262  cout << "\n-------------------------------------------------------------------------\n";
263  switch(kind_recording) {
264  case RECORDING::CLEAR_INDICES: cout << "Clearing the computational graph." << endl; break;
265  case RECORDING::MESH_COORDS: cout << "Storing computational graph wrt MESH COORDINATES." << endl; break;
266  case RECORDING::SOLUTION_VARIABLES:
267  cout << "Direct iteration to store the primal computational graph." << endl;
268  cout << "Computing residuals to check the convergence of the direct problem." << endl; break;
269  default: break;
270  }
271  }
272 
273  /*---Enable recording and register input of the iteration --- */
274 
275  if (kind_recording != RECORDING::CLEAR_INDICES){
276 
277  AD::StartRecording();
278 
279  iteration->RegisterInput(solver_container, geometry_container, config_container, ZONE_0, INST_0, kind_recording);
280  }
281 
282  /*--- Set the dependencies of the iteration ---*/
283 
284  iteration->SetDependencies(solver_container, geometry_container, numerics_container, config_container, ZONE_0,
285  INST_0, kind_recording);
286 
287  /*--- Do one iteration of the direct solver ---*/
288 
289  DirectRun(kind_recording);
290 
291  /*--- Store the recording state ---*/
292 
293  RecordingState = kind_recording;
294 
295  /*--- Register Output of the iteration ---*/
296 
297  iteration->RegisterOutput(solver_container, geometry_container, config_container, ZONE_0, INST_0);
298  /*--- Extract the objective function and store it --- */
299 
300  SetObjFunction();
301 
302  if (kind_recording != RECORDING::CLEAR_INDICES && config_container[ZONE_0]->GetWrt_AD_Statistics()) {
303  if (rank == MASTER_NODE) AD::PrintStatistics();
304 #ifdef CODI_REVERSE_TYPE
305  if (size > SINGLE_NODE) {
306  su2double myMem = AD::getGlobalTape().getTapeValues().getUsedMemorySize(), totMem = 0.0;
307  SU2_MPI::Allreduce(&myMem, &totMem, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
308  if (rank == MASTER_NODE) {
309  cout << "MPI\n";
310  cout << "-------------------------------------\n";
311  cout << " Total memory used : " << totMem << " MB\n";
312  cout << "-------------------------------------\n" << endl;
313  }
314  }
315 #endif
316  }
317 
318  AD::StopRecording();
319

322 void CDiscAdjSinglezoneDriver::SetAdj_ObjFunction(){
323 
324  const auto IterAvg_Obj = config->GetIter_Avg_Objective();
325  su2double seeding = 1.0;
326 
327  CWindowingTools windowEvaluator = CWindowingTools();
328 
329  if (config->GetTime_Marching() != TIME_MARCHING::STEADY){
330  if (TimeIter < IterAvg_Obj){
331  /*--- Default behavior (in case no specific window is chosen) is to use Square-Windowing, i.e. the numerator equals 1.0 ---*/
332  seeding = windowEvaluator.GetWndWeight(config->GetKindWindow(),TimeIter, IterAvg_Obj-1)/ (static_cast<su2double>(IterAvg_Obj));
333  }
334  else {
335  seeding = 0.0;
336  }
337  }
338 
339  if (rank == MASTER_NODE){
340  SU2_TYPE::SetDerivative(ObjFunc, SU2_TYPE::GetValue(seeding));
341  } else {
342  SU2_TYPE::SetDerivative(ObjFunc, 0.0);
343  }

346 void CDiscAdjSinglezoneDriver::SetObjFunction(){
347 
348  ObjFunc = 0.0;
349 
350  /*--- Specific scalar objective functions ---*/
351 
352  switch (config->GetKind_Solver()) {
353  case MAIN_SOLVER::DISC_ADJ_INC_EULER: case MAIN_SOLVER::DISC_ADJ_INC_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_INC_RANS:
354  case MAIN_SOLVER::DISC_ADJ_EULER: case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_RANS:
355  case MAIN_SOLVER::DISC_ADJ_FEM_EULER: case MAIN_SOLVER::DISC_ADJ_FEM_NS: case MAIN_SOLVER::DISC_ADJ_FEM_RANS:
356 
357  /*--- Surface based obj. function ---*/
358 
359  direct_output->SetHistory_Output(geometry, solver, config, config->GetTimeIter(),
360  config->GetOuterIter(), config->GetInnerIter());
361  ObjFunc += solver[FLOW_SOL]->GetTotal_ComboObj();
362 
363  /*--- These calls to be moved to a generic framework at a next stage ---*/
364  /*--- Some things that are currently hacked into output must be reorganized ---*/
365  if (config->GetBoolTurbomachinery()) {
366  output_legacy->ComputeTurboPerformance(solver[FLOW_SOL], geometry, config);
367 
368  unsigned short nMarkerTurboPerf = config->GetnMarker_TurboPerformance();
369  unsigned short nSpanSections = config->GetnSpanWiseSections();
370 
371  switch (config_container[ZONE_0]->GetKind_ObjFunc()){
372  case ENTROPY_GENERATION:
373  ObjFunc += output_legacy->GetEntropyGen(nMarkerTurboPerf-1, nSpanSections);
374  break;
375  case FLOW_ANGLE_OUT:
376  ObjFunc += output_legacy->GetFlowAngleOut(nMarkerTurboPerf-1, nSpanSections);
377  break;
378  case MASS_FLOW_IN:
379  ObjFunc += output_legacy->GetMassFlowIn(nMarkerTurboPerf-1, nSpanSections);
380  break;
381  default:
382  break;
383  }
384  }
385  break;
386 
387  case MAIN_SOLVER::DISC_ADJ_HEAT:
388  direct_output->SetHistory_Output(geometry, solver, config, config->GetTimeIter(),
389  config->GetOuterIter(), config->GetInnerIter());
390  ObjFunc = solver[HEAT_SOL]->GetTotal_ComboObj();
391  break;
392 
393  case MAIN_SOLVER::DISC_ADJ_FEM:
394  solver[FEA_SOL]->Postprocessing(geometry, config, numerics_container[ZONE_0][INST_0][MESH_0][FEA_SOL], true);
395 
396  direct_output->SetHistory_Output(geometry, solver, config, config->GetTimeIter(),
397  config->GetOuterIter(), config->GetInnerIter());
398  ObjFunc = solver[FEA_SOL]->GetTotal_ComboObj();
399  break;
400 
401  default:
402  break;
403  }
404 
405  if (rank == MASTER_NODE){
406  AD::RegisterOutput(ObjFunc);
407  }
