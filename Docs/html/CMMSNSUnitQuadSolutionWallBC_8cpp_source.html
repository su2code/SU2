32 CMMSNSUnitQuadSolutionWallBC::CMMSNSUnitQuadSolutionWallBC(unsigned short val_nDim,
33  unsigned short val_nVar,
34  unsigned short val_iMesh,
35  CConfig* config)
36  : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
37 
38  /*--- Write a message that the solution is initialized for the manufactured
39  solution for the Navier-Stokes equations on a unit quad with no-slip
40  wall boundary conditions. ---*/
41  if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
42  cout << endl;
43  cout << "Warning: Fluid properties and solution are being " << endl;
44  cout << " initialized for the manufactured solution " << endl;
45  cout << " of the Navier-Stokes equations on a unit quad" << endl;
46  cout << " with no-slip wall boundary conditions!!!" << endl;
47  cout << endl << flush;
48  }
49 
50  /*--- Coefficients, needed to determine the solution. ---*/
51  const su2double Prandtl = config->GetPrandtl_Lam();
52 
53  RGas = config->GetGas_Constant();
54  Gamma = config->GetGamma();
55  Viscosity = config->GetMu_Constant();
56  Conductivity = Viscosity*Gamma*RGas/(Prandtl*(Gamma-1.0));
57 
58  /*--- Initialize TWall to the default value of 300 K (in case the outer wall
59  is not modelled as an isothermal wall) and try to retrieve the wall
60  temperature from the boundary conditions. ---*/
61  TWall = 300.0;
62  for(unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
63  if(config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL) {
64  const string Marker_Tag = config->GetMarker_All_TagBound(iMarker);
65  TWall = config->GetIsothermal_Temperature(Marker_Tag);
66  }
67  }
68 
69  /*--- Get the reference values for pressure, density and velocity. ---*/
70  Pressure_Ref = config->GetPressure_Ref();
71  Density_Ref = config->GetDensity_Ref();
72  Velocity_Ref = config->GetVelocity_Ref();
73 
74  /*--- The constants for the density and velocities. ---*/
75  rho_0 = 1.25;
76  u_0 = 135.78;
77  v_0 = -67.61;
78 
79  /*--- The constants for the temperature solution. ---*/
80  a_T1 = 1.05;
81  a_T2 = -0.85;
82 
83  /*--- Perform some sanity and error checks for this solution here. ---*/
84  if(config->GetTime_Marching() != TIME_MARCHING::STEADY)
85  SU2_MPI::Error("Steady mode must be selected for the MMS NS Unit Quad case with wall BCs.",
86  CURRENT_FUNCTION);
87 
88  if(Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::RANS &&
89  Kind_Solver != MAIN_SOLVER::FEM_EULER && Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
90  Kind_Solver != MAIN_SOLVER::FEM_LES)
91  SU2_MPI::Error("Compressible flow equations must be selected for the MMS NS Unit Quad case with wall BCs.",
92  CURRENT_FUNCTION);
93 
94  if((Kind_Solver != MAIN_SOLVER::NAVIER_STOKES) &&
95  (Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES))
96  SU2_MPI::Error("Navier Stokes equations must be selected for the MMS NS Unit Quad case with wall BCs.",
97  CURRENT_FUNCTION);
98 
99  if((config->GetKind_FluidModel() != STANDARD_AIR) &&
100  (config->GetKind_FluidModel() != IDEAL_GAS))
101  SU2_MPI::Error("Standard air or ideal gas must be selected for the MMS NS Unit Quad case with wall BCs.",
102  CURRENT_FUNCTION);
103 
104  if(config->GetKind_ViscosityModel() != VISCOSITYMODEL::CONSTANT)
105  SU2_MPI::Error("Sutherland must be selected for viscosity for the MMS NS Unit Quad case with wall BCs.",
106  CURRENT_FUNCTION);
107 
108  if(config->GetKind_ConductivityModel() != CONDUCTIVITYMODEL::CONSTANT_PRANDTL)
109  SU2_MPI::Error("Constant Prandtl number must be selected for the MMS NS Unit Quad case with wall BCs.",
110  CURRENT_FUNCTION);
111 }

123 void CMMSNSUnitQuadSolutionWallBC::GetSolution(const su2double *val_coords,
124  const su2double val_t,
125  su2double *val_solution) const {
126  /* Easier storage of the y-coordinate. */
127  const su2double y = val_coords[1];
128 
129  /* Determine the dimensional solution for the temperature. */
130  const su2double Pi = PI_NUMBER;
131  const su2double fact = y*y/(a_T1 + a_T2);
132 
133  su2double T = 0.25*TWall*(3.0 + fact*(a_T1*cos(Pi*(y-1.0))
134  + a_T2*cos(Pi*(y-1.0)*2.0)));
135 
136  /* Determine the dimensional solution for the velocities. */
137  su2double u = u_0*y*(1.0-y)*4.0;
138  su2double v = v_0*y*(1.0-y)*4.0;
139 
140  /* Compute the pressure from the density and temperature. */
141  su2double rho = rho_0;
142  su2double p = rho*RGas*T;
143 
144  /* Determine the non-dimensional solution. */
145  rho /= Density_Ref;
146  p /= Pressure_Ref;
147  u /= Velocity_Ref;
148  v /= Velocity_Ref;
149 
150  /* Compute the conservative variables from the primitive ones.
151  Note that the implementation below is valid for both 2D and 3D. */
152  val_solution[0] = rho;
153  val_solution[1] = rho*u;
154  val_solution[2] = rho*v;
155  val_solution[3] = 0.0;
156  val_solution[nDim+1] = p/(Gamma-1.0) + 0.5*rho*(u*u + v*v);
157 }

159 void CMMSNSUnitQuadSolutionWallBC::GetMMSSourceTerm(const su2double *val_coords,
160  const su2double val_t,
161  su2double *val_source) const {
162 
163  /*--- Abbreviate Pi and the y-coordinate. ---*/
164  const su2double Pi = PI_NUMBER;
165  const su2double y = val_coords[1];
166 
167  /*--- The source code for the source terms is generated in Maple.
168  See the file CMMSNSUnitQuadSolutionWallBC.mw in the directory
169  CreateMMSSourceTerms for the details how to do this. ---*/
170  const su2double t1 = (v_0 * rho_0);
171  const su2double t2 = 1.0 - y;
172  const su2double t6 = rho_0 * u_0;
173  const su2double t7 = t2 * t2;
174  const su2double t8 = t7 * y;
175  const su2double t12 = y * y;
176  const su2double t13 = t2 * t12;
177  const su2double t20 = (v_0 * v_0);
178  const su2double t21 = (t20 * rho_0);
179  const su2double t26 = rho_0 * RGas;
180  const su2double t27 = a_T1 + a_T2;
181  const su2double t28 = 1.0 / t27;
182  const su2double t30 = -t2 * Pi;
183  const su2double t31 = cos(t30);
184  const su2double t32 = t31 * a_T1;
185  const su2double t33 = 2.0 * t30;
186  const su2double t34 = cos(t33);
187  const su2double t36 = t34 * a_T2 + t32;
188  const su2double t39 = t28 * t12;
189  const su2double t41 = sin(t30);
190  const su2double t42 = t41 * Pi * a_T1;
191  const su2double t43 = Pi * a_T2;
192  const su2double t44 = sin(t33);
193  const su2double t50 = (0.2e1 * t36 * t28 * y + (-0.2e1 * t44 * t43 - t42) * t39) * TWall;
194  const su2double t52 = t50 * t26 / 0.4e1;
195  const su2double t57 = 1.0 / (Gamma - 1.0);
196  const su2double t61 = (u_0 * u_0);
197  const su2double t62 = y * t61;
198  const su2double t64 = t12 * t61;
199  const su2double t66 = y * t20;
200  const su2double t68 = t12 * t20;
201  const su2double t75 = t2 * y;
202  const su2double t80 = (t36 * t39 + 0.3e1) * TWall;
203  const su2double t92 = v_0 * (t57 * t80 * t26 / 0.4e1 + (8 * (t7 * t64 + t7 * t68) * rho_0) + t80 * t26 / 0.4e1);
204  const su2double t97 = t41 * t31;
205  const su2double t102 = a_T1 * y;
206  const su2double t104 = t31 * t31;
207  const su2double t111 = 0.1e1 / t27 / 0.4e1;
208  const su2double t115 = Pi * Pi;
209  const su2double t116 = t41 * t41;
210  const su2double t118 = a_T2 * y;
211  const su2double t138 = Viscosity * (-8.0 * y + 4.0);
212  const su2double t149 = Viscosity * (2.0 * y - 1.0);
213  const su2double t155 = 0.4e1 * t75 * v_0 * (t57 * t50 * t26 / 0.4e1 + (16.0 * (-t2 * t64 - t2 * t68 + t7 * t62 + t7 * t66) * rho_0) + t52)
214  + 0.4e1 * t2 * t92 - 0.4e1 * y * t92 - Conductivity * t111 * (-t102 * Pi * t41 - 0.4e1 * y * t43 * t97 + 0.4e1 * t104 * a_T2
215  - 0.2e1 * a_T2 + 0.2e1 * t32) * TWall - Conductivity * t111 * (-t102 * t31 * t115 - 0.4e1 * t118 * t115 * t104
216  + 0.4e1 * t118 * t116 * t115 - 0.12e2 * t43 * t97 - 0.3e1 * t42) * TWall * y + (32.0 * t75 * t61 * Viscosity)
217  - (4.0 * t2 * t61 * t138) + (4.0 * t62 * t138) + 0.128e3 / 0.3e1 * t75 * t20 * Viscosity + 0.64e2 / 0.3e1 * t2 * t20 * t149
218  - 0.64e2 / 0.3e1 * t66 * t149;
219 
220  /*--- Set the source term, which is valid for both 2D and 3D cases.
221  Note the scaling for the correct non-dimensionalization. ---*/
222  val_source[0] = 4.0 * t2 * t1 - 4.0 * y * t1;
223  val_source[1] = (-0.32e2 * v_0 * t13 * t6 + 0.32e2 * v_0 * t8 * t6 + (8.0 * Viscosity * u_0));
224  val_source[2] = ((32.0 * t8 * t21) - (32.0 * t13 * t21) + t52 + 0.32e2 / 0.3e1 * Viscosity * v_0);
225  val_source[3] = 0.0;
226  val_source[nDim+1] = t155;
227 
228  val_source[0] /= Density_Ref*Velocity_Ref;
229  val_source[1] /= Pressure_Ref;
230  val_source[2] /= Pressure_Ref;
231  val_source[nDim+1] /= Velocity_Ref*Pressure_Ref;
232 }
