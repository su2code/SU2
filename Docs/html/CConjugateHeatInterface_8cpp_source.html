39 void CConjugateHeatInterface::GetDonor_Variable(CSolver *donor_solution, CGeometry *donor_geometry,
40  const CConfig *donor_config, unsigned long Marker_Donor,
41  unsigned long Vertex_Donor, unsigned long Point_Donor) {
42 
43  /*--- Compute distance of donor point to PointNormal for T-gradient/heatflux computation ---*/
44  const auto nDim = donor_geometry->GetnDim();
45  const auto Coord = donor_geometry->nodes->GetCoord(Point_Donor);
46  const auto PointNormal = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNormal_Neighbor();
47  const auto Coord_Normal = donor_geometry->nodes->GetCoord(PointNormal);
48 
49  su2double Edge_Vector[MAXNDIM] = {0.0};
50  GeometryToolbox::Distance(nDim, Coord_Normal, Coord, Edge_Vector);
51  const su2double dist = GeometryToolbox::Norm(nDim, Edge_Vector);
52 
53  /*--- Retrieve temperature solution and its gradient ---*/
54 
55  const su2double Twall = donor_solution->GetNodes()->GetTemperature(Point_Donor);
56  const su2double Tnormal = donor_solution->GetNodes()->GetTemperature(PointNormal);
57 
58  const su2double dTdn = (Twall - Tnormal)/dist;
59  // TODO: Check if these improve accuracy, if needed at all
60  // const auto Normal = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNormal();
61  // su2double Area = GeometryToolbox::Norm(nDim, Normal);
62  // for (iDim = 0; iDim < nDim; iDim++) {
63  // dTdn += (Twall - Tnormal)/dist * (Edge_Vector[iDim]/dist) * (Normal[iDim]/Area);
64  // }
65 
66  /*--- Calculate the heat flux density (temperature gradient times thermal conductivity) and
67  thermal conductivity divided by distance. ---*/
68  su2double thermal_conductivity = 0.0;
69  su2double heat_flux_density = 0.0;
70  su2double conductivity_over_dist = 0.0;
71 
72  const bool compressible_flow = (donor_config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE);
73  const bool incompressible_flow = (donor_config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) && donor_config->GetEnergy_Equation();
74 
75  if (compressible_flow) {
76 
77  const su2double Gamma = donor_config->GetGamma();
78  const su2double Gas_Constant = donor_config->GetGas_ConstantND();
79  const su2double Cp = (Gamma / (Gamma - 1.0)) * Gas_Constant;
80 
81  const su2double Prandtl_Lam = donor_config->GetPrandtl_Lam();
82  const su2double laminar_viscosity = donor_solution->GetNodes()->GetLaminarViscosity(Point_Donor); // TDE check for consistency
83 
84  const su2double thermal_conductivityND = Cp*(laminar_viscosity/Prandtl_Lam);
85  heat_flux_density = thermal_conductivityND*dTdn;
86 
87  if ((donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::DIRECT_TEMPERATURE_ROBIN_HEATFLUX) ||
88  (donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::AVERAGED_TEMPERATURE_ROBIN_HEATFLUX)) {
89 
90  thermal_conductivity = thermal_conductivityND*donor_config->GetViscosity_Ref();
91  conductivity_over_dist = thermal_conductivity/dist;
92  }
93  }
94  else if (incompressible_flow) {
95 
96  const auto iPoint = donor_geometry->vertex[Marker_Donor][Vertex_Donor]->GetNode();
97 
98  const su2double thermal_conductivityND = donor_solution->GetNodes()->GetThermalConductivity(iPoint);
99  heat_flux_density = thermal_conductivityND*dTdn;
100 
101  if ((donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::DIRECT_TEMPERATURE_ROBIN_HEATFLUX) ||
102  (donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::AVERAGED_TEMPERATURE_ROBIN_HEATFLUX)) {
103 
104  switch (donor_config->GetKind_ConductivityModel()) {
105 
106  case CONDUCTIVITYMODEL::CONSTANT:
107  case CONDUCTIVITYMODEL::COOLPROP:
108  thermal_conductivity = thermal_conductivityND*donor_config->GetThermal_Conductivity_Ref();
109  break;
110  case CONDUCTIVITYMODEL::CONSTANT_PRANDTL:
111  thermal_conductivity = thermal_conductivityND*donor_config->GetGas_Constant_Ref()
112  *donor_config->GetViscosity_Ref();
113  break;
114  case CONDUCTIVITYMODEL::POLYNOMIAL:
115  SU2_MPI::Error("Polynomial Conductivity model not implemented for CHT interface.", CURRENT_FUNCTION);
116  break;
117  }
118 
119  conductivity_over_dist = thermal_conductivity/dist;
120  }
121  }
122  else if (donor_config->GetHeatProblem()) {
123 
124  /*--- Heat solver stand-alone case ---*/
125 
126  const su2double thermal_diffusivity = donor_config->GetThermalDiffusivity();
127  heat_flux_density = thermal_diffusivity*dTdn;
128 
129 
130  if ((donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::DIRECT_TEMPERATURE_ROBIN_HEATFLUX) ||
131  (donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::AVERAGED_TEMPERATURE_ROBIN_HEATFLUX)) {
132 
133  const su2double rho_cp_solid = donor_config->GetSpecific_Heat_Cp()*donor_config->GetMaterialDensity(0);
134  conductivity_over_dist = thermal_diffusivity*rho_cp_solid/dist;
135  }
136  }
137 
138  /*--- Set the conjugate heat variables that are transferred by default ---*/
139 
140  Donor_Variable[0] = Twall*donor_config->GetTemperature_Ref();
141  Donor_Variable[1] = heat_flux_density*donor_config->GetHeat_Flux_Ref();
142 
143  /*--- We only need these for the Robin BC option ---*/
144 
145  if ((donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::DIRECT_TEMPERATURE_ROBIN_HEATFLUX) ||
146  (donor_config->GetKind_CHT_Coupling() == CHT_COUPLING::AVERAGED_TEMPERATURE_ROBIN_HEATFLUX)) {
147 
148  Donor_Variable[2] = conductivity_over_dist;
149  Donor_Variable[3] = Tnormal*donor_config->GetTemperature_Ref();
150  }
151 
152 }

169  target_config->GetRelaxation_Factor_CHT(), Target_Variable[3]);
