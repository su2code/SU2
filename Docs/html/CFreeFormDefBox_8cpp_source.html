165 void CFreeFormDefBox::SetControlPoints_Parallelepiped (void) {
166  unsigned short iDim, iDegree, jDegree, kDegree;
167 
168  /*--- Set base control points according to the notation of Vtk for hexahedrons ---*/
169  for (iDim = 0; iDim < nDim; iDim++) {
170  Coord_Control_Points [0] [0] [0] [iDim] = Coord_Corner_Points[0][iDim];
171  Coord_Control_Points [lOrder-1] [0] [0] [iDim] = Coord_Corner_Points[1][iDim];
172  Coord_Control_Points [lOrder-1] [mOrder-1] [0] [iDim] = Coord_Corner_Points[2][iDim];
173  Coord_Control_Points [0] [mOrder-1] [0] [iDim] = Coord_Corner_Points[3][iDim];
174  Coord_Control_Points [0] [0] [nOrder-1] [iDim] = Coord_Corner_Points[4][iDim];
175  Coord_Control_Points [lOrder-1] [0] [nOrder-1] [iDim] = Coord_Corner_Points[5][iDim];
176  Coord_Control_Points [lOrder-1] [mOrder-1] [nOrder-1] [iDim] = Coord_Corner_Points[6][iDim];
177  Coord_Control_Points [0] [mOrder-1] [nOrder-1] [iDim] = Coord_Corner_Points[7][iDim];
178  }
179 
180  /*--- Fill the rest of the cubic matrix of control points with uniform spacing (parallelepiped) ---*/
181  for (iDegree = 0; iDegree <= lDegree; iDegree++)
182  for (jDegree = 0; jDegree <= mDegree; jDegree++)
183  for (kDegree = 0; kDegree <= nDegree; kDegree++) {
184  Coord_Control_Points[iDegree][jDegree][kDegree][0] = Coord_Corner_Points[0][0]
185  + su2double(iDegree)/su2double(lDegree)*(Coord_Corner_Points[1][0]-Coord_Corner_Points[0][0]);
186  Coord_Control_Points[iDegree][jDegree][kDegree][1] = Coord_Corner_Points[0][1]
187  + su2double(jDegree)/su2double(mDegree)*(Coord_Corner_Points[3][1]-Coord_Corner_Points[0][1]);
188  Coord_Control_Points[iDegree][jDegree][kDegree][2] = Coord_Corner_Points[0][2]
189  + su2double(kDegree)/su2double(nDegree)*(Coord_Corner_Points[4][2]-Coord_Corner_Points[0][2]);
190  }

816 
817  lmn[0] = lDegree; lmn[1] = mDegree; lmn[2] = nDegree;

852 su2double *CFreeFormDefBox::GetParametricCoord_Iterative(unsigned long iPoint, su2double *xyz, const su2double *ParamCoordGuess, CConfig *config) {
853 
854  su2double *IndepTerm, SOR_Factor = 1.0, MinNormError, NormError, Determinant, AdjHessian[3][3], Temp[3] = {0.0,0.0,0.0};
855  unsigned short iDim, jDim, RandonCounter;
856  unsigned long iter;
857 
858  su2double tol = config->GetFFD_Tol()*1E-3;
859  unsigned short it_max = config->GetnFFD_Iter();
860  unsigned short Random_Trials = 500;
861 
862  /*--- Allocate the Hessian ---*/
863 
864  Hessian = new su2double* [nDim];
865  IndepTerm = new su2double [nDim];
866  for (iDim = 0; iDim < nDim; iDim++) {
867  Hessian[iDim] = new su2double[nDim];
868  ParamCoord[iDim] = ParamCoordGuess[iDim];
869  IndepTerm [iDim] = 0.0;
870  }
871 
872  RandonCounter = 0; MinNormError = 1E6;
873 
874  /*--- External iteration ---*/
875 
876  for (iter = 0; iter < (unsigned long)it_max*Random_Trials; iter++) {
877 
878  /*--- The independent term of the solution of our system is -Gradient(sol_old) ---*/
879 
880  Gradient = GetFFDGradient(ParamCoord, xyz);
881 
882  for (iDim = 0; iDim < nDim; iDim++) IndepTerm[iDim] = - Gradient[iDim];
883 
884  /*--- Hessian = The Matrix of our system, getHessian(sol_old,xyz,...) ---*/
885 
886  GetFFDHessian(ParamCoord, xyz, Hessian);
887 
888  /*--- Adjoint to Hessian ---*/
889 
890  AdjHessian[0][0] = Hessian[1][1]*Hessian[2][2]-Hessian[1][2]*Hessian[2][1];
891  AdjHessian[0][1] = Hessian[0][2]*Hessian[2][1]-Hessian[0][1]*Hessian[2][2];
892  AdjHessian[0][2] = Hessian[0][1]*Hessian[1][2]-Hessian[0][2]*Hessian[1][1];
893  AdjHessian[1][0] = Hessian[1][2]*Hessian[2][0]-Hessian[1][0]*Hessian[2][2];
894  AdjHessian[1][1] = Hessian[0][0]*Hessian[2][2]-Hessian[0][2]*Hessian[2][0];
895  AdjHessian[1][2] = Hessian[0][2]*Hessian[1][0]-Hessian[0][0]*Hessian[1][2];
896  AdjHessian[2][0] = Hessian[1][0]*Hessian[2][1]-Hessian[1][1]*Hessian[2][0];
897  AdjHessian[2][1] = Hessian[0][1]*Hessian[2][0]-Hessian[0][0]*Hessian[2][1];
898  AdjHessian[2][2] = Hessian[0][0]*Hessian[1][1]-Hessian[0][1]*Hessian[1][0];
899 
900  /*--- Determinant of Hessian ---*/
901 
902  Determinant = Hessian[0][0]*AdjHessian[0][0]+Hessian[0][1]*AdjHessian[1][0]+Hessian[0][2]*AdjHessian[2][0];
903 
904  /*--- Hessian inverse ---*/
905 
906  if (Determinant != 0) {
907  for (iDim = 0; iDim < nDim; iDim++) {
908  Temp[iDim] = 0.0;
909  for (jDim = 0; jDim < nDim; jDim++) {
910  Temp[iDim] += AdjHessian[iDim][jDim]*IndepTerm[jDim]/Determinant;
911  }
912  }
913  for (iDim = 0; iDim < nDim; iDim++) {
914  IndepTerm[iDim] = Temp[iDim];
915  }
916  }
917 
918  /*--- Update with Successive over-relaxation ---*/
919 
920  for (iDim = 0; iDim < nDim; iDim++) {
921  ParamCoord[iDim] = (1.0-SOR_Factor)*ParamCoord[iDim] + SOR_Factor*(ParamCoord[iDim] + IndepTerm[iDim]);
922  }
923 
924  /*--- If the gradient is small, we have converged ---*/
925 
926  if ((fabs(IndepTerm[0]) < tol) && (fabs(IndepTerm[1]) < tol) && (fabs(IndepTerm[2]) < tol)) break;
927 
928  /*--- Compute the norm of the error ---*/
929 
930  NormError = 0.0;
931  for (iDim = 0; iDim < nDim; iDim++)
932  NormError += IndepTerm[iDim]*IndepTerm[iDim];
933  NormError = sqrt(NormError);
934 
935  MinNormError = min(NormError, MinNormError);
936 
937  /*--- If we have no convergence with Random_Trials iterations probably we are in a local minima. ---*/
938 
939  if (((iter % it_max) == 0) && (iter != 0)) {
940 
941  RandonCounter++;
942  if (RandonCounter == Random_Trials) {
943  cout << endl << "Unknown point: "<< iPoint <<" (" << xyz[0] <<", "<< xyz[1] <<", "<< xyz[2] <<"). Min Error: "<< MinNormError <<". Iter: "<< iter <<"."<< endl;
944  }
945  else {
946  SOR_Factor = 0.1;
947  for (iDim = 0; iDim < nDim; iDim++)
948  ParamCoord[iDim] = su2double(rand())/su2double(RAND_MAX);
949  }
950 
951  }
952 
953  /* --- Splines are not defined outside of [0,1]. So if the parametric coords are outside of
954  * [0,1] the step was too big and we have to use a smaller relaxation factor. ---*/
955 
956  if ((config->GetFFD_Blending() == BSPLINE_UNIFORM) &&
957  (((ParamCoord[0] < 0.0) || (ParamCoord[0] > 1.0)) ||
958  ((ParamCoord[1] < 0.0) || (ParamCoord[1] > 1.0)) ||
959  ((ParamCoord[2] < 0.0) || (ParamCoord[2] > 1.0)))) {
960 
961  for (iDim = 0; iDim < nDim; iDim++){
962  ParamCoord[iDim] = ParamCoordGuess[iDim];
963  }
964  SOR_Factor = 0.9*SOR_Factor;
965  }
966 
967  }
968 
969  for (iDim = 0; iDim < nDim; iDim++)
970  delete [] Hessian[iDim];
971  delete [] Hessian;
972  delete [] IndepTerm;
973 
974  /*--- The code has hit the max number of iterations ---*/
975 
976  if (iter == (unsigned long)it_max*Random_Trials) {
977  cout << "Unknown point: (" << xyz[0] <<", "<< xyz[1] <<", "<< xyz[2] <<"). Increase the value of FFD_ITERATIONS." << endl;
978  }
979 
980  /*--- Real Solution is now ParamCoord; Return it ---*/
981 
982  return ParamCoord;
983

986 bool CFreeFormDefBox::GetPointFFD(CGeometry *geometry, CConfig *config, unsigned long iPoint) const {
987 
988  bool Inside = true;
989  bool cylindrical = (config->GetFFD_CoordSystem() == CYLINDRICAL);
990  bool spherical = (config->GetFFD_CoordSystem() == SPHERICAL);
991  bool polar = (config->GetFFD_CoordSystem() == POLAR);
992 
993  /*--- indices of the FFD box. Note that the front face is labelled 0,1,2,3 and the back face is 4,5,6,7 ---*/
994 
995  unsigned short Index[6][5] = {
996  {0,1,2,3,0}, // front side
997  {1,5,6,2,1}, // right side
998  {2,6,7,3,2}, // top side
999  {3,7,4,0,3}, // left side
1000  {4,5,1,0,4}, // bottom side
1001  {4,7,6,5,4}}; // back side
1002 
1003  /*--- The current approach is to subdivide each of the 6 faces of the hexahedral FFD box into 4 triangles
1004  by defining a supporting middle point. This allows nonplanar FFD boxes.
1005  Note that the definition of the FFD box is as follows: the FFD box is a 6-sided die and we are looking at the side "1".
1006  The opposite side is side "6".
1007  If we are looking at side "1", we define the nodes counterclockwise.
1008  If we are looking at side "6", we define the face clockwise ---*/
1009 
1010  unsigned short nDim = geometry->GetnDim();
1011 
1012  su2double Coord[3] = {0.0, 0.0, 0.0};
1013  for (unsigned short iDim = 0; iDim < nDim; iDim++)
1014  Coord[iDim] = geometry->nodes->GetCoord(iPoint, iDim);
1015 
1016  su2double X_0, Y_0, Z_0, Xbar, Ybar, Zbar;
1017 
1018  if (cylindrical) {
1019 
1020  X_0 = config->GetFFD_Axis(0); Y_0 = config->GetFFD_Axis(1); Z_0 = config->GetFFD_Axis(2);
1021 
1022  Xbar = Coord[0] - X_0; Ybar = Coord[1] - Y_0; Zbar = Coord[2] - Z_0;
1023 
1024  Coord[0] = sqrt(Ybar*Ybar + Zbar*Zbar);
1025  Coord[1] = atan2(Zbar, Ybar); if (Coord[1] > PI_NUMBER/2.0) Coord[1] -= 2.0*PI_NUMBER;
1026  Coord[2] = Xbar;
1027 
1028  }
1029 
1030  else if (spherical || polar) {
1031 
1032  X_0 = config->GetFFD_Axis(0); Y_0 = config->GetFFD_Axis(1); Z_0 = config->GetFFD_Axis(2);
1033 
1034  Xbar = Coord[0] - X_0; Ybar = Coord[1] - Y_0; Zbar = Coord[2] - Z_0;
1035 
1036  Coord[0] = sqrt(Xbar*Xbar + Ybar*Ybar + Zbar*Zbar);
1037  Coord[1] = atan2(Zbar, Ybar); if (Coord[1] > PI_NUMBER/2.0) Coord[1] -= 2.0*PI_NUMBER;
1038  Coord[2] = acos(Xbar/Coord[0]);
1039 
1040  }
1041 
1042  /*--- loop over the faces of the FFD box ---*/
1043 
1044  for (unsigned short iVar = 0; iVar < 6; iVar++) {
1045 
1046  su2double P[3] = {0.0, 0.0, 0.0};
1047 
1048  /*--- every face needs an interpolated middle point for the triangles ---*/
1049 
1050  for (int p = 0; p < 4; p++){
1051  P[0] += 0.25*Coord_Corner_Points[Index[iVar][p]][0];
1052  P[1] += 0.25*Coord_Corner_Points[Index[iVar][p]][1];
1053  P[2] += 0.25*Coord_Corner_Points[Index[iVar][p]][2];
1054  }
1055 
1056  /*--- loop over the 4 triangles making up the FFD box. The sign is equal for all distances ---*/
1057 
1058  for (unsigned short jVar = 0; jVar < 4; jVar++) {
1059  su2double Distance_Point = geometry->Point2Plane_Distance(Coord,
1060  Coord_Corner_Points[Index[iVar][jVar]],
1061  Coord_Corner_Points[Index[iVar][jVar+1]],
1062  P);
1063  if (Distance_Point < 0) {
1064  Inside = false;
1065  return Inside;
1066  }
1067  }
1068  }
1069 
1070  return Inside;
1071 
1072 }
