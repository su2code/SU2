47 void CBaseMPIWrapper::Error(std::string ErrorMsg, std::string FunctionName){
48 
49  /* Set MinRankError to Rank, as the error message is called on this rank. */
50  MinRankError = Rank;
51  int flag = 0;
52 
53 #if MPI_VERSION >= 3
54  /* Find out whether the error call is collective via MPI_Ibarrier. */
55  Request barrierRequest;
56  MPI_Ibarrier(currentComm, &barrierRequest);
57 
58  /* Try to complete the non-blocking barrier call for a second. */
59  double startTime = SU2_MPI::Wtime();
60  while( true ) {
61 
62  MPI_Test(&barrierRequest, &flag, MPI_STATUS_IGNORE);
63  if( flag ) break;
64 
65  double currentTime = SU2_MPI::Wtime();
66  if(currentTime > startTime + 1.0) break;
67  }
68 #else
69  /* MPI_Ibarrier function is not supported. Simply wait for one
70  second to give other ranks the opportunity to reach this point. */
71 #ifdef _MSC_VER
72  _sleep(1);
73 #else
74  sleep(1);
75 #endif
76 #endif
77 
78  if( flag ) {
79  /* The barrier is completed and hence the error call is collective.
80  Set MinRankError to 0. */
81  MinRankError = 0;
82  }
83  else {
84  /* The error call is not collective and the minimum rank must be
85  determined by one sided communication. Loop over the lower numbered
86  ranks to check if they participate in the error message. */
87  for(int i=0; i<Rank; ++i) {
88 
89  int MinRankErrorOther;
90  MPI_Win_lock(MPI_LOCK_SHARED, i, 0, winMinRankError);
91  MPI_Get(&MinRankErrorOther, 1, MPI_INT, i, 0, 1, MPI_INT, winMinRankError);
92  MPI_Win_unlock(i, winMinRankError);
93 
94  if(MinRankErrorOther < MinRankError) {
95  MinRankError = MinRankErrorOther;
96  break;
97  }
98  }
99  }
100 
101  /* Check if this rank must write the error message and do so. */
102  if (Rank == MinRankError){
103  std::cout << std::endl << std::endl;
104  std::cout << "Error in \"" << FunctionName << "\": " << std::endl;
105  std::cout << "-------------------------------------------------------------------------" << std::endl;
106  std::cout << ErrorMsg << std::endl;
107  std::cout << "------------------------------ Error Exit -------------------------------" << std::endl;
108  std::cout << std::endl << std::endl;
109  }
110  Abort(currentComm, EXIT_FAILURE);
111 }
112 #else // HAVE_MPI
