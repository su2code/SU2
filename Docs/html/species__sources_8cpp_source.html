69 template <class T>
70 CNumerics::ResidualType<> CSourceAxisymmetric_Species<T>::ComputeResidual(const CConfig* config) {
71  /*--- Preaccumulation ---*/
72  AD::StartPreacc();
73  AD::SetPreaccIn(ScalarVar_i, nVar);
74  AD::SetPreaccIn(Volume);
75 
76  if (incompressible) {
77  AD::SetPreaccIn(V_i, nDim+6);
78  }
79  else {
80  AD::SetPreaccIn(V_i, nDim+7);
81  }
82 
83  /*--- Initialization. ---*/
84  for (auto iVar = 0u; iVar < nVar; iVar++) {
85  residual[iVar] = 0.0;
86  for (auto jVar = 0; jVar < nVar; jVar++) {
87  jacobian[iVar][jVar] = 0.0;
88  }
89  }
90 
91  /*--- Contribution due to 2D axisymmetric formulation ---*/
92  if (Coord_i[1] > EPS) {
93 
94  AD::SetPreaccIn(Coord_i[1]);
95  AD::SetPreaccIn(Diffusion_Coeff_i, nVar);
96  AD::SetPreaccIn(ScalarVar_Grad_i, nVar, nDim);
97 
98  const su2double yinv = 1.0 / Coord_i[1];
99 
100  const su2double Density_i = V_i[idx.Density()];
101 
102  su2double Velocity_i[3];
103  for (auto iDim = 0u; iDim < nDim; iDim++)
104  Velocity_i[iDim] = V_i[idx.Velocity() + iDim];
105 
106  /*--- Inviscid component of the source term. When div(v)=0, this term does not occur ---*/
107 
108  if (!bounded_scalar) {
109  for (auto iVar = 0u; iVar < nVar; iVar++)
110  residual[iVar] -= yinv * Volume * Density_i * ScalarVar_i[iVar] * Velocity_i[1];
111 
112  if (implicit) {
113  for (auto iVar = 0u; iVar < nVar; iVar++) {
114  jacobian[iVar][iVar] -= yinv * Volume * Velocity_i[1];
115  }
116  }
117  }
118 
119  /*--- Add the viscous terms if necessary. ---*/
120 
121  if (config->GetViscous()) {
122  su2double Mass_Diffusivity_Tur = 0.0;
123  if (turbulence)
124  Mass_Diffusivity_Tur = V_i[idx.EddyViscosity()] / Sc_t;
125 
126  for (auto iVar=0u; iVar < nVar; iVar++){
127  residual[iVar] += yinv * Volume * (Density_i * Diffusion_Coeff_i[iVar] + Mass_Diffusivity_Tur) * ScalarVar_Grad_i[iVar][1];
128  }
129  }
130 
131  }
132 
133  AD::SetPreaccOut(residual, nVar);
134  AD::EndPreacc();
135 
136  return ResidualType<>(residual, jacobian, nullptr);
137 }
