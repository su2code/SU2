34 CAdjFlowCompOutput::CAdjFlowCompOutput(CConfig *config, unsigned short nDim) : CAdjFlowOutput(config, nDim) {
35 
36  /*--- Set the default history fields if nothing is set in the config file ---*/
37 
38  if (nRequestedHistoryFields == 0){
39  requestedHistoryFields.emplace_back("ITER");
40  requestedHistoryFields.emplace_back("RMS_RES");
41  requestedHistoryFields.emplace_back("SENSITIVITY");
42  nRequestedHistoryFields = requestedHistoryFields.size();
43  }
44 
45  if (nRequestedScreenFields == 0){
46  if (config->GetTime_Domain()) requestedScreenFields.emplace_back("TIME_ITER");
47  if (multiZone) requestedScreenFields.emplace_back("OUTER_ITER");
48  requestedScreenFields.emplace_back("INNER_ITER");
49  requestedScreenFields.emplace_back("RMS_ADJ_DENSITY");
50  requestedScreenFields.emplace_back("RMS_ADJ_MOMENTUM-X");
51  requestedScreenFields.emplace_back("SENS_GEO");
52  requestedScreenFields.emplace_back("SENS_AOA");
53  nRequestedScreenFields = requestedScreenFields.size();
54  }
55 
56  if (nRequestedVolumeFields == 0){
57  requestedVolumeFields.emplace_back("COORDINATES");
58  requestedVolumeFields.emplace_back("SOLUTION");
59  requestedVolumeFields.emplace_back("SENSITIVITY");
60  nRequestedVolumeFields = requestedVolumeFields.size();
61  }
62 
63  if (find(requestedVolumeFields.begin(), requestedVolumeFields.end(), string("SENSITIVITY")) == requestedVolumeFields.end()) {
64  requestedVolumeFields.emplace_back("SENSITIVITY");
65  nRequestedVolumeFields ++;
66  }
67 
68  stringstream ss;
69  ss << "Zone " << config->GetiZone() << " (Adj. Comp. Fluid)";
70  multiZoneHeaderString = ss.str();
71 
72  /*--- Set the volume filename --- */
73 
74  volumeFilename = config->GetAdj_FileName();
75 
76  /*--- Set the surface filename --- */
77 
78  surfaceFilename = config->GetSurfAdjCoeff_FileName();
79 
80  /*--- Set the restart filename --- */
81 
82  restartFilename = config->GetRestart_AdjFileName();
83 
84  /*--- Add the obj. function extension --- */
85 
86  restartFilename = config->GetObjFunc_Extension(restartFilename);
87 
88  /*--- Set the default convergence field --- */
89 
90  if (convFields.empty() ) convFields.emplace_back("RMS_ADJ_DENSITY");
91 
92 }

169 void CAdjFlowCompOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver){
170 
171  CSolver* adjflow_solver = solver[ADJFLOW_SOL];
172  CSolver* mesh_solver = solver[MESH_SOL];
173 
174  SetHistoryOutputValue("RMS_ADJ_DENSITY", log10(adjflow_solver->GetRes_RMS(0)));
175  SetHistoryOutputValue("RMS_ADJ_MOMENTUM-X", log10(adjflow_solver->GetRes_RMS(1)));
176  SetHistoryOutputValue("RMS_ADJ_MOMENTUM-Y", log10(adjflow_solver->GetRes_RMS(2)));
177  if (geometry->GetnDim() == 3) {
178  SetHistoryOutputValue("RMS_ADJ_MOMENTUM-Z", log10(adjflow_solver->GetRes_RMS(3)));
179  SetHistoryOutputValue("RMS_ADJ_ENERGY", log10(adjflow_solver->GetRes_RMS(4)));
180  } else {
181  SetHistoryOutputValue("RMS_ADJ_ENERGY", log10(adjflow_solver->GetRes_RMS(3)));
182  }
183 
184  SetHistoryOutputValue("MAX_ADJ_DENSITY", log10(adjflow_solver->GetRes_Max(0)));
185  SetHistoryOutputValue("MAX_ADJ_MOMENTUM-X", log10(adjflow_solver->GetRes_Max(1)));
186  SetHistoryOutputValue("MAX_ADJ_MOMENTUM-Y", log10(adjflow_solver->GetRes_Max(2)));
187  if (geometry->GetnDim() == 3) {
188  SetHistoryOutputValue("MAX_ADJ_MOMENTUM-Z", log10(adjflow_solver->GetRes_Max(3)));
189  SetHistoryOutputValue("MAX_ADJ_ENERGY", log10(adjflow_solver->GetRes_Max(4)));
190  } else {
191  SetHistoryOutputValue("MAX_ADJ_ENERGY", log10(adjflow_solver->GetRes_Max(3)));
192  }
193 
194  if (multiZone){
195  SetHistoryOutputValue("BGS_ADJ_DENSITY", log10(adjflow_solver->GetRes_BGS(0)));
196  SetHistoryOutputValue("BGS_ADJ_MOMENTUM-X", log10(adjflow_solver->GetRes_BGS(1)));
197  SetHistoryOutputValue("BGS_ADJ_MOMENTUM-Y", log10(adjflow_solver->GetRes_BGS(2)));
198  if (geometry->GetnDim() == 3) {
199  SetHistoryOutputValue("BGS_ADJ_MOMENTUM-Z", log10(adjflow_solver->GetRes_BGS(3)));
200  SetHistoryOutputValue("BGS_ADJ_ENERGY", log10(adjflow_solver->GetRes_BGS(4)));
201  } else {
202  SetHistoryOutputValue("BGS_ADJ_ENERGY", log10(adjflow_solver->GetRes_BGS(3)));
203  }
204  }
205 
206  SetHistoryOutputValue("SENS_GEO", adjflow_solver->GetTotal_Sens_Geo());
207  SetHistoryOutputValue("SENS_AOA", adjflow_solver->GetTotal_Sens_AoA() * PI_NUMBER / 180.0);
208  SetHistoryOutputValue("SENS_MACH", adjflow_solver->GetTotal_Sens_Mach());
209  SetHistoryOutputValue("SENS_PRESS", adjflow_solver->GetTotal_Sens_Press());
210  SetHistoryOutputValue("SENS_TEMP", adjflow_solver->GetTotal_Sens_Temp());
211 
212  SetHistoryOutputValue("LINSOL_ITER", adjflow_solver->GetIterLinSolver());
213  SetHistoryOutputValue("LINSOL_RESIDUAL", log10(adjflow_solver->GetResLinSolver()));
214 
215  if (config->GetDeform_Mesh()) {
216  SetHistoryOutputValue("DEFORM_ITER", mesh_solver->System.GetIterations());
217  SetHistoryOutputValue("DEFORM_RESIDUAL", log10(mesh_solver->System.GetResidual()));
218  }
219 
220  LoadHistoryData_AdjScalar(config, solver);
221 }

277 void CAdjFlowCompOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint){
278 
279  CVariable* Node_AdjFlow = solver[ADJFLOW_SOL]->GetNodes();
280  CPoint* Node_Geo = geometry->nodes;
281 
282  SetVolumeOutputValue("COORD-X", iPoint, Node_Geo->GetCoord(iPoint, 0));
283  SetVolumeOutputValue("COORD-Y", iPoint, Node_Geo->GetCoord(iPoint, 1));
284  if (nDim == 3)
285  SetVolumeOutputValue("COORD-Z", iPoint, Node_Geo->GetCoord(iPoint, 2));
286 
287  SetVolumeOutputValue("ADJ_DENSITY", iPoint, Node_AdjFlow->GetSolution(iPoint, 0));
288  SetVolumeOutputValue("ADJ_MOMENTUM-X", iPoint, Node_AdjFlow->GetSolution(iPoint, 1));
289  SetVolumeOutputValue("ADJ_MOMENTUM-Y", iPoint, Node_AdjFlow->GetSolution(iPoint, 2));
290  if (nDim == 3){
291  SetVolumeOutputValue("ADJ_MOMENTUM-Z", iPoint, Node_AdjFlow->GetSolution(iPoint, 3));
292  SetVolumeOutputValue("ADJ_ENERGY", iPoint, Node_AdjFlow->GetSolution(iPoint, 4));
293  } else {
294  SetVolumeOutputValue("ADJ_ENERGY", iPoint, Node_AdjFlow->GetSolution(iPoint, 3));
295  }
296 
297  // Residuals
298  SetVolumeOutputValue("RES_ADJ_DENSITY", iPoint, Node_AdjFlow->GetSolution(iPoint, 0) - Node_AdjFlow->GetSolution_Old(iPoint, 0));
299  SetVolumeOutputValue("RES_ADJ_MOMENTUM-X", iPoint, Node_AdjFlow->GetSolution(iPoint, 1) - Node_AdjFlow->GetSolution_Old(iPoint, 1));
300  SetVolumeOutputValue("RES_ADJ_MOMENTUM-Y", iPoint, Node_AdjFlow->GetSolution(iPoint, 2) - Node_AdjFlow->GetSolution_Old(iPoint, 2));
301  if (nDim == 3){
302  SetVolumeOutputValue("RES_ADJ_MOMENTUM-Z", iPoint, Node_AdjFlow->GetSolution(iPoint, 3) - Node_AdjFlow->GetSolution_Old(iPoint, 3));
303  SetVolumeOutputValue("RES_ADJ_ENERGY", iPoint, Node_AdjFlow->GetSolution(iPoint, 4) - Node_AdjFlow->GetSolution_Old(iPoint, 4));
304  } else {
305  SetVolumeOutputValue("RES_ADJ_ENERGY", iPoint, Node_AdjFlow->GetSolution(iPoint, 3) - Node_AdjFlow->GetSolution_Old(iPoint, 3));
306  }
307 
308  SetVolumeOutputValue("SENSITIVITY-X", iPoint, Node_AdjFlow->GetSensitivity(iPoint, 0));
309  SetVolumeOutputValue("SENSITIVITY-Y", iPoint, Node_AdjFlow->GetSensitivity(iPoint, 1));
310  if (nDim == 3)
311  SetVolumeOutputValue("SENSITIVITY-Z", iPoint, Node_AdjFlow->GetSensitivity(iPoint, 2));
312 
313  LoadVolumeData_AdjScalar(config, solver, iPoint);
314 }
