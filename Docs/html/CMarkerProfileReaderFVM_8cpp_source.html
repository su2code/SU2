73 void CMarkerProfileReaderFVM::ReadMarkerProfile() {
74 
75  /*--- Open the profile file (we have already error checked) ---*/
76 
77  ifstream profile_file;
78  profile_file.open(filename.data(), ios::in);
79  bool nmarkFound = false;
80  unsigned long skip = 0;
81 
82  /*--- Identify the markers and data set in the profile file ---*/
83 
84  string text_line;
85  /*--- We search the file until we find the keyword NMARK=. Data before NMARK= will be ignored.
86  This allows for some information in a header that will be ignored by the profile reader. ---*/
87  while (getline (profile_file, text_line)) {
88  /*--- read NMARK ---*/
89  string::size_type position = text_line.find ("NMARK=",0);
90  if (position != string::npos) {
91  nmarkFound = true;
92  text_line.erase (0,6); numberOfProfiles = atoi(text_line.c_str());
93 
94  numberOfRowsInProfile.resize(numberOfProfiles);
95  numberOfColumnsInProfile.resize(numberOfProfiles);
96 
97  for (unsigned short iMarker = 0 ; iMarker < numberOfProfiles; iMarker++) {
98 
99  /*--- read MARKER_TAG ---*/
100  getline (profile_file, text_line);
101  text_line.erase (0,11);
102  for (unsigned short iChar = 0; iChar < 20; iChar++) {
103  position = text_line.find( " ", 0 ); if (position != string::npos) text_line.erase (position,1);
104  position = text_line.find( "\r", 0 ); if (position != string::npos) text_line.erase (position,1);
105  position = text_line.find( "\n", 0 ); if (position != string::npos) text_line.erase (position,1);
106  }
107  profileTags.push_back(text_line.c_str());
108 
109  /*--- read NROW ---*/
110  getline (profile_file, text_line);
111  text_line.erase (0,5); numberOfRowsInProfile[iMarker] = atoi(text_line.c_str());
112 
113  /*--- read NCOL ---*/
114  getline (profile_file, text_line);
115  text_line.erase (0,5); numberOfColumnsInProfile[iMarker] = atoi(text_line.c_str());
116 
117  /*--- read the column format description. This line is not required, so if we cannot find it, we just continue ---*/
118  getline (profile_file, text_line);
119  string::size_type dataheader = text_line.find ("# COORD",0);
120  if (dataheader == 0) {
121  skip = 0;
122  } else {
123  /*--- no header, but we have read a line, so we have to read one line less data ---*/
124  skip = 1;
125  }
126 
127  /*--- Skip the data. This is read in the next loop. ---*/
128 
129  for (unsigned long iRow = 0; iRow < (numberOfRowsInProfile[iMarker]-skip); iRow++) getline (profile_file, text_line);
130 
131  }
132  }
133  }
134 
135  profile_file.close();
136 
137  if (nmarkFound==false) {
138  SU2_MPI::Error("While opening profile file, no \"NMARK=\" specification was found", CURRENT_FUNCTION);
139  }
140 
141  /*--- Compute array bounds and offsets. Allocate data structure. ---*/
142 
143  profileData.resize(numberOfProfiles);
144  for (unsigned short iMarker = 0; iMarker < numberOfProfiles; iMarker++) {
145  profileData[iMarker].resize(numberOfRowsInProfile[iMarker]*numberOfColumnsInProfile[iMarker], 0.0);
146  }
147 
148  /*--- Read all lines in the profile file and extract data. ---*/
149 
150  profile_file.open(filename.data(), ios::in);
151 
152  int counter = 0;
153  while (getline (profile_file, text_line)) {
154 
155  string::size_type position = text_line.find ("NMARK=",0);
156  if (position != string::npos) {
157 
158  for (unsigned short iMarker = 0; iMarker < numberOfProfiles; iMarker++) {
159 
160  /*--- Skip the tag, nRow and nCol lines. ---*/
161 
162  getline (profile_file, text_line);
163  getline (profile_file, text_line);
164  getline (profile_file, text_line);
165 
166  /*--- if skip=0 then we can expect column format description ---*/
167  if (skip == 0) getline (profile_file, text_line);
168 
169  /*--- Now read the data for each row and store. ---*/
170 
171  for (unsigned long iRow = 0; iRow < numberOfRowsInProfile[iMarker]; iRow++) {
172 
173  getline (profile_file, text_line);
174 
175  istringstream point_line(text_line);
176 
177  /*--- Store the values (starting with node coordinates) --*/
178 
179  for (unsigned short iVar = 0; iVar < numberOfColumnsInProfile[iMarker]; iVar++)
180  point_line >> profileData[iMarker][iRow*numberOfColumnsInProfile[iMarker] + iVar];
181 
182  /*--- Increment our local row counter. ---*/
183 
184  counter++;
185 
186  }
187  }
188  }
189  }
190 
191  profile_file.close();
192 
193 }

195 void CMarkerProfileReaderFVM::MergeProfileMarkers() {
196 
197  /*--- Local variables needed on all processors ---*/
198 
199  unsigned long iPoint, jPoint, kPoint;
200 
201  int iProcessor, nProcessor = size;
202 
203  unsigned long iVertex, iMarker;
204  unsigned long Buffer_Send_nPoin[1], *Buffer_Recv_nPoin = nullptr;
205  unsigned long nLocalPoint = 0, MaxLocalPoint = 0;
206  unsigned long maxProfiles = 0;
207  unsigned long index, iChar;
208 
209  char str_buf[MAX_STRING_SIZE];
210  vector<string> Marker_Tags;
211 
212  vector<unsigned long> nRowCum_Counter;
213 
214  if (rank == MASTER_NODE) Buffer_Recv_nPoin = new unsigned long[nProcessor];
215 
216  /*--- Search all boundaries on the present rank to count the number
217  of nodes found on profile markers. ---*/
218 
219  nLocalPoint = 0; numberOfProfiles = 0;
220  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
221  if (config->GetMarker_All_KindBC(iMarker) == markerType) {
222  numberOfProfiles++;
223  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
224  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
225 
226  /*--- Only communicate owned nodes to avoid duplicates. ---*/
227 
228  if (geometry->nodes->GetDomain(iPoint))
229  nLocalPoint++;
230 
231  }
232  }
233  }
234  Buffer_Send_nPoin[0] = nLocalPoint;
235 
236  /*--- Communicate the total number of nodes on this domain. ---*/
237 
238  SU2_MPI::Gather(&Buffer_Send_nPoin, 1, MPI_UNSIGNED_LONG,
239  Buffer_Recv_nPoin, 1, MPI_UNSIGNED_LONG, MASTER_NODE, SU2_MPI::GetComm());
240  SU2_MPI::Allreduce(&nLocalPoint, &MaxLocalPoint, 1, MPI_UNSIGNED_LONG, MPI_MAX, SU2_MPI::GetComm());
241  SU2_MPI::Allreduce(&numberOfProfiles, &maxProfiles, 1, MPI_UNSIGNED_LONG, MPI_MAX, SU2_MPI::GetComm());
242 
243  /*--- Send and Recv buffers. ---*/
244 
245  su2double *Buffer_Send_X = new su2double[MaxLocalPoint];
246  su2double *Buffer_Recv_X = nullptr;
247 
248  su2double *Buffer_Send_Y = new su2double[MaxLocalPoint];
249  su2double *Buffer_Recv_Y = nullptr;
250 
251  su2double *Buffer_Send_Z = nullptr, *Buffer_Recv_Z = nullptr;
252  if (dimension == 3) Buffer_Send_Z = new su2double[MaxLocalPoint];
253 
254  char *Buffer_Send_Str = new char[MaxLocalPoint*MAX_STRING_SIZE];
255  char *Buffer_Recv_Str = nullptr;
256 
257  /*--- Prepare the receive buffers in the master node only. ---*/
258 
259  if (rank == MASTER_NODE) {
260 
261  Buffer_Recv_X = new su2double[nProcessor*MaxLocalPoint];
262  Buffer_Recv_Y = new su2double[nProcessor*MaxLocalPoint];
263  if (dimension == 3) Buffer_Recv_Z = new su2double[nProcessor*MaxLocalPoint];
264  Buffer_Recv_Str = new char[nProcessor*MaxLocalPoint*MAX_STRING_SIZE];
265 
266  /*--- Sum total number of nodes to be written and allocate arrays ---*/
267 
268  unsigned long nGlobal_InletPoint = 0;
269  for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
270  nGlobal_InletPoint += Buffer_Recv_nPoin[iProcessor];
271  }
272 
273  profileCoords.resize(maxProfiles);
274  for (iMarker = 0; iMarker < maxProfiles; iMarker++) {
275  profileCoords[iMarker].resize(dimension);
276  }
277 
278  }
279 
280  /*--- Main communication routine. Loop over each coordinate and perform
281  the MPI comm. Temporary 1-D buffers are used to send the coordinates at
282  all nodes on each partition to the master node. These are then unpacked
283  by the master and sorted by marker tag in one large n-dim. array. ---*/
284 
285  su2double *Coords_Local; jPoint = 0;
286  for (iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
287  if (config->GetMarker_All_KindBC(iMarker) == markerType) {
288 
289  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
290  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
291 
292  /*--- Only communicate owned nodes to avoid duplicates. ---*/
293 
294  if (geometry->nodes->GetDomain(iPoint)) {
295 
296  /*--- Retrieve local coordinates at this node. ---*/
297 
298  Coords_Local = geometry->nodes->GetCoord(iPoint);
299 
300  /*--- Load local coords into the temporary send buffer. ---*/
301 
302  Buffer_Send_X[jPoint] = Coords_Local[0];
303  Buffer_Send_Y[jPoint] = Coords_Local[1];
304  if (dimension == 3) Buffer_Send_Z[jPoint] = Coords_Local[2];
305 
306  /*--- If US system, the output should be in inches ---*/
307 
308  if (config->GetSystemMeasurements() == US) {
309  Buffer_Send_X[jPoint] *= 12.0;
310  Buffer_Send_Y[jPoint] *= 12.0;
311  if (dimension == 3) Buffer_Send_Z[jPoint] *= 12.0;
312  }
313 
314  /*--- Store the marker tag for this particular node. ---*/
315 
316  SPRINTF(&Buffer_Send_Str[jPoint*MAX_STRING_SIZE], "%s",
317  config->GetMarker_All_TagBound(iMarker).c_str());
318 
319  /*--- Increment jPoint as the counter. We need this because iPoint
320  may include halo nodes that we skip over during this loop. ---*/
321 
322  jPoint++;
323 
324  }
325  }
326  }
327  }
328 
329  /*--- Gather the coordinate data on the master node using MPI. ---*/
330 
331  SU2_MPI::Gather(Buffer_Send_X, (int)MaxLocalPoint, MPI_DOUBLE,
332  Buffer_Recv_X, (int)MaxLocalPoint, MPI_DOUBLE, MASTER_NODE, SU2_MPI::GetComm());
333  SU2_MPI::Gather(Buffer_Send_Y, (int)MaxLocalPoint, MPI_DOUBLE,
334  Buffer_Recv_Y, (int)MaxLocalPoint, MPI_DOUBLE, MASTER_NODE, SU2_MPI::GetComm());
335  if (dimension == 3) {
336  SU2_MPI::Gather(Buffer_Send_Z, (int)MaxLocalPoint, MPI_DOUBLE,
337  Buffer_Recv_Z, (int)MaxLocalPoint, MPI_DOUBLE, MASTER_NODE, SU2_MPI::GetComm());
338  }
339  SU2_MPI::Gather(Buffer_Send_Str, (int)MaxLocalPoint*MAX_STRING_SIZE, MPI_CHAR,
340  Buffer_Recv_Str, (int)MaxLocalPoint*MAX_STRING_SIZE, MPI_CHAR, MASTER_NODE, SU2_MPI::GetComm());
341 
342  /*--- The master node unpacks and sorts this variable by marker tag. ---*/
343 
344  if (rank == MASTER_NODE) {
345 
346  profileTags.clear();
347 
348  /*--- First, parse the marker tags to count how many total profile markers
349  we have now on the master. ---*/
350 
351  for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
352  for (iPoint = 0; iPoint < Buffer_Recv_nPoin[iProcessor]; iPoint++) {
353  index = (iProcessor*MaxLocalPoint + iPoint)*MAX_STRING_SIZE;
354  for (iChar = 0; iChar < MAX_STRING_SIZE; iChar++) {
355  str_buf[iChar] = Buffer_Recv_Str[index + iChar];
356  }
357  Marker_Tags.push_back(str_buf);
358  profileTags.push_back(str_buf);
359  }
360  }
361 
362  /*--- Remove the duplicate profile marker strings. From 1 per point to 1 per marker. ---*/
363 
364  profileTags.erase(unique(profileTags.begin(),
365  profileTags.end()),
366  profileTags.end());
367 
368  /*--- Store the unique number of markers for writing later. ---*/
369 
370  numberOfProfiles = profileTags.size();
371 
372  /*--- Count the number of rows (nodes) per marker. ---*/
373 
374  numberOfRowsInProfile.resize(numberOfProfiles,0.0);
375  jPoint = 0;
376  for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
377  for (iPoint = 0; iPoint < Buffer_Recv_nPoin[iProcessor]; iPoint++) {
378  for (iMarker = 0; iMarker < numberOfProfiles; iMarker++) {
379  if (profileTags[iMarker] == Marker_Tags[jPoint]) {
380  numberOfRowsInProfile[iMarker]++;
381  }
382  }
383  jPoint++;
384  }
385  }
386 
387  /*--- Load up the coordinates, sorted into chunks per marker. ---*/
388 
389  jPoint = 0; kPoint = 0;
390  for (iProcessor = 0; iProcessor < nProcessor; iProcessor++) {
391  for (iPoint = 0; iPoint < Buffer_Recv_nPoin[iProcessor]; iPoint++) {
392  for (iMarker = 0; iMarker < numberOfProfiles; iMarker++) {
393 
394  if (profileTags[iMarker] == Marker_Tags[kPoint]) {
395 
396  /*--- Find our current index for this marker and store coords. ---*/
397 
398  profileCoords[iMarker][0].push_back(Buffer_Recv_X[jPoint]);
399  profileCoords[iMarker][1].push_back(Buffer_Recv_Y[jPoint]);
400  if (dimension == 3)
401  profileCoords[iMarker][2].push_back(Buffer_Recv_Z[jPoint]);
402 
403  }
404  }
405 
406  /*--- Increment point counter for marker tags and data. ---*/
407 
408  kPoint++;
409  jPoint++;
410 
411  }
412 
413  /*--- Adjust jPoint to index of next proc's data in the buffers. ---*/
414 
415  jPoint = (iProcessor+1)*MaxLocalPoint;
416 
417  }
418  }
419 
420  /*--- Immediately release the temporary data buffers. ---*/
421 
422  delete [] Buffer_Send_X;
423  delete [] Buffer_Send_Y;
424  delete [] Buffer_Send_Z;
425  delete [] Buffer_Send_Str;
426  if (rank == MASTER_NODE) {
427  delete [] Buffer_Recv_X;
428  delete [] Buffer_Recv_Y;
429  delete [] Buffer_Recv_Z;
430  delete [] Buffer_Recv_nPoin;
431  delete [] Buffer_Recv_Str;
432  }
433 
434 }

436 void CMarkerProfileReaderFVM::WriteMarkerProfileTemplate() {
437 
438  /*--- Count the number of columns that we have for this case.
439  Here, we have dimension entries for node coordinates and then the
440  total number of columns of data specified in the constructor. ---*/
441 
442  const unsigned short nColumns = dimension + numberOfVars;
443 
444  /*--- Write the profile file. Note that we have already merged
445  all of the information for the markers and coordinates previously
446  in the MergeProfileMarkers() routine and only the master writes. ---*/
447 
448  if (rank == MASTER_NODE) {
449 
450  ofstream node_file("example_"+filename);
451 
452  node_file << "NMARK= " << numberOfProfiles << endl;
453 
454  unsigned short iMarkerCounter = 0;
455  for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
456  if (config->GetMarker_All_KindBC(iMarker) == markerType) {
457 
458  /*--- Access the default data for this marker. ---*/
459 
460  string Marker_Tag = profileTags[iMarkerCounter];
461 
462  /*--- Header information for this marker. ---*/
463 
464  node_file << "MARKER_TAG= " << Marker_Tag << endl;
465  node_file << "NROW=" << numberOfRowsInProfile[iMarkerCounter] << endl;
466  node_file << "NCOL=" << nColumns << endl;
467 
468  /*--- header line (names of the columns) --- */
469  node_file << columnNames[iMarkerCounter] << endl;
470 
471  node_file << setprecision(15);
472  node_file << std::scientific;
473 
474  /*--- Loop over the data structure and write the coords and vars. ---*/
475 
476  for (unsigned long iPoint = 0; iPoint < numberOfRowsInProfile[iMarkerCounter]; iPoint++) {
477 
478  for (unsigned short iDim = 0; iDim < dimension; iDim++) {
479  node_file << profileCoords[iMarkerCounter][iDim][iPoint] << "\t";
480  }
481 
482  node_file << columnValues[iMarkerCounter] << endl;
483 
484  }
485 
486  iMarkerCounter++;
487  }
488  } // iMarker
489  node_file.close();
490 
491  /*--- Print a message to inform the user about the template file. ---*/
492 
493  stringstream err;
494  err << endl;
495  err << " Could not find the input file for the marker profile." << endl;
496  err << " Looked for: " << filename << "." << endl;
497  err << " Created a template profile file with default values" << endl;
498  err << " named example_" << filename << endl;
499  err << " You can use this file as a guide for making your own profile" << endl;
500  err << " specification." << endl << endl;
501  SU2_MPI::Error(err.str(), CURRENT_FUNCTION);
502 
503  }
