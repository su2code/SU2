167  /*!
168  * \brief TSL-Approximation of Viscous NS Jacobians for arbitrary equations of state.
169  */
170  template <unsigned short NVAR, unsigned short NSPECIES>
171  void ComputeViscousJacs_impl(const su2double *val_Mean_PrimVar,
172  su2double *val_Mean_Eve,
173  const su2double *val_Mean_Cvve,
174  const su2double *val_diffusion_coeff,
175  su2double val_laminar_viscosity,
176  su2double val_eddy_viscosity,
177  su2double val_thermal_conductivity,
178  su2double val_thermal_conductivity_ve,
179  su2double val_dist_ij, const su2double *val_normal,
180  su2double val_dS, const su2double *val_Fv,
181  su2double **val_Jac_i, su2double **val_Jac_j,
182  const CConfig *config){
183 
184  /*--- Initialize matrices ---*/
185  Matrix<NVAR, NVAR> dFdVi, dFdVj, dVdUi, dVdUj;
186 
187  /*--- Need to avoid size of 1 otherwise the type becomes a vector (without [i][j]). ---*/
188  constexpr unsigned short nSpeciesDummy = (NSPECIES == 1) ? 2 : NSPECIES;
189  Matrix<nSpeciesDummy, nSpeciesDummy> dJdr_i, dJdr_j;
190  Matrix<1,NSPECIES> Ys, Ys_i, Ys_j;
191 
192  /*--- Play tricks on the compiler, in static mode use NVAR from the template, in dynamic mode
193  use nVar from the class or from the arguments. ---*/
194  const auto nVar = (NVAR != DynamicSize) ? NVAR : this->nVar;
195  const auto nSpecies = (NSPECIES != DynamicSize) ? NSPECIES : this->nSpecies;
196 
197  /*--- Deduce nDim from the other variables to allow the compiler to also unroll the loops over nDim. ---*/
198  const auto nDim = nVar - nSpecies - 2;
199 
200  // Allocate and initialize, for the static case the compiler optimizes this away.
201  dFdVi.resize(nVar, nVar) = su2double(0.0);
202  dFdVj.resize(nVar, nVar) = su2double(0.0);
203  dVdUi.resize(nVar, nVar) = su2double(0.0);
204  dVdUj.resize(nVar, nVar) = su2double(0.0);
205  if (NSPECIES == DynamicSize) {
206  dJdr_i.resize(nSpecies,nSpecies) = su2double(0.0);
207  dJdr_j.resize(nSpecies,nSpecies) = su2double(0.0);
208  Ys.resize(1,nSpecies) = su2double(0.0);
209  Ys_i.resize(1,nSpecies) = su2double(0.0);
210  Ys_j.resize(1,nSpecies) = su2double(0.0);
211  }
212 
213  /*--- Calculate preliminary geometric quantities ---*/
214  su2double dij = val_dist_ij;
215  su2double theta = GeometryToolbox::SquaredNorm(nDim, val_normal);
216 
217  /*--- Rename for convenience ---*/
218  su2double rho_i = V_i[RHO_INDEX];
219  su2double rho_j = V_j[RHO_INDEX];
220  su2double T = val_Mean_PrimVar[T_INDEX];
221  su2double Tve = val_Mean_PrimVar[TVE_INDEX];
222  su2double mu = val_laminar_viscosity;
223  su2double ktr = val_thermal_conductivity;
224  su2double kve = val_thermal_conductivity_ve;
225  su2double RuSI= UNIVERSAL_GAS_CONSTANT;
226  su2double Ru = 1000.0*RuSI;
227  const auto& Ds = val_diffusion_coeff;
228  const auto& hs = fluidmodel->ComputeSpeciesEnthalpy(T, Tve, val_Mean_Eve);
229  const auto& Cvtr = fluidmodel->GetSpeciesCvTraRot();
230  const auto& Ms = fluidmodel->GetSpeciesMolarMass();
231 
232  /*--- Calculate useful diffusion parameters ---*/
233  // Mass fraction and summation of diffusion fluxes
234  su2double sumY = 0.0;
235  su2double sumY_i = 0.0;
236  su2double sumY_j = 0.0;
237  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
238  Ys[iSpecies] = val_Mean_PrimVar[RHOS_INDEX+iSpecies];
239  Ys_i[iSpecies] = V_i[RHOS_INDEX+iSpecies]/V_i[RHO_INDEX];
240  Ys_j[iSpecies] = V_j[RHOS_INDEX+iSpecies]/V_j[RHO_INDEX];
241 
242  sumY_i += Ds[iSpecies]*theta/dij*Ys_i[iSpecies];
243  sumY_j += Ds[iSpecies]*theta/dij*Ys_j[iSpecies];
244  sumY += Ds[iSpecies]*theta/dij*(Ys_j[iSpecies]-Ys_i[iSpecies]);
245  }
246 
247  su2double vel[MAXNDIM];
248  for (auto iDim = 0; iDim < nDim; iDim++)
249  vel[iDim] = val_Mean_PrimVar[VEL_INDEX+iDim];
250 
251  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
252  for (auto jSpecies = 0; jSpecies < nSpecies; jSpecies++) {
253 
254  // first term
255  dJdr_j[iSpecies][jSpecies] += 0.5*(Ds[iSpecies]*theta/dij *
256  (Ys_j[iSpecies]*rho_i/rho_j +
257  Ys_i[iSpecies]));
258  dJdr_i[iSpecies][jSpecies] += -0.5*(Ds[iSpecies]*theta/dij *
259  (Ys_j[iSpecies] +
260  Ys_i[iSpecies]*rho_j/rho_i));
261 
262  // second term
263  dJdr_j[iSpecies][jSpecies] +=
264  0.25*(Ys_i[iSpecies] - rho_i/rho_j*Ys_j[iSpecies])*sumY
265  + 0.25*(Ys_i[iSpecies]+Ys_j[iSpecies])*(rho_i+rho_j)*Ds[jSpecies]*theta/(dij*rho_j)
266  - 0.25*(Ys_i[iSpecies]+Ys_j[iSpecies])*(rho_i+rho_j)*sumY_j/rho_j;
267 
268  dJdr_i[iSpecies][jSpecies] +=
269  0.25*(-rho_j/rho_i*Ys_i[iSpecies]+Ys_j[iSpecies])*sumY
270  - 0.25*(Ys_i[iSpecies]+Ys_j[iSpecies])*(rho_i+rho_j)*Ds[jSpecies]*theta/(dij*rho_i)
271  + 0.25*(Ys_i[iSpecies]+Ys_j[iSpecies])*(rho_i+rho_j)*sumY_i/rho_i;
272  }
273 
274  // first term
275  dJdr_j[iSpecies][iSpecies] += -0.5*Ds[iSpecies]*theta/dij*(1+rho_i/rho_j);
276  dJdr_i[iSpecies][iSpecies] += 0.5*Ds[iSpecies]*theta/dij*(1+rho_j/rho_i);
277 
278  // second term
279  dJdr_j[iSpecies][iSpecies] += 0.25*(1.0+rho_i/rho_j)*sumY;
280  dJdr_i[iSpecies][iSpecies] += 0.25*(1.0+rho_j/rho_i)*sumY;
281  }
282 
283  /*--- Calculate transformation matrix ---*/
284  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
285  dVdUi[iSpecies][iSpecies] = 1.0;
286  dVdUj[iSpecies][iSpecies] = 1.0;
287  }
288  for (auto iDim = 0; iDim < nDim; iDim++) {
289  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
290  dVdUi[nSpecies+iDim][iSpecies] = -V_i[VEL_INDEX+iDim]/V_i[RHO_INDEX];
291  dVdUj[nSpecies+iDim][iSpecies] = -V_j[VEL_INDEX+iDim]/V_j[RHO_INDEX];
292  }
293  dVdUi[nSpecies+iDim][nSpecies+iDim] = 1.0/V_i[RHO_INDEX];
294  dVdUj[nSpecies+iDim][nSpecies+iDim] = 1.0/V_j[RHO_INDEX];
295  }
296  for (auto iVar = 0; iVar < nVar; iVar++) {
297  dVdUi[nSpecies+nDim][iVar] = dTdU_i[iVar];
298  dVdUj[nSpecies+nDim][iVar] = dTdU_j[iVar];
299  dVdUi[nSpecies+nDim+1][iVar] = dTvedU_i[iVar];
300  dVdUj[nSpecies+nDim+1][iVar] = dTvedU_j[iVar];
301  }
302 
303  if (nDim == 2) {
304 
305  /*--- Geometry parameters ---*/
306  su2double thetax = theta + val_normal[0]*val_normal[0]/3.0;
307  su2double thetay = theta + val_normal[1]*val_normal[1]/3.0;
308  su2double etaz = val_normal[0]*val_normal[1]/3.0;
309  su2double pix = mu/dij * (thetax*vel[0] + etaz*vel[1] );
310  su2double piy = mu/dij * (etaz*vel[0] + thetay*vel[1]);
311 
312  /*--- Populate primitive Jacobian ---*/
313 
314  // X-momentum
315  dFdVj[nSpecies][nSpecies] = mu*thetax/dij*val_dS;
316  dFdVj[nSpecies][nSpecies+1] = mu*etaz/dij*val_dS;
317 
318  // Y-momentum
319  dFdVj[nSpecies+1][nSpecies] = mu*etaz/dij*val_dS;
320  dFdVj[nSpecies+1][nSpecies+1] = mu*thetay/dij*val_dS;
321 
322  // Energy
323  dFdVj[nSpecies+2][nSpecies] = pix*val_dS;
324  dFdVj[nSpecies+2][nSpecies+1] = piy*val_dS;
325  dFdVj[nSpecies+2][nSpecies+2] = ktr*theta/dij*val_dS;
326  dFdVj[nSpecies+2][nSpecies+3] = kve*theta/dij*val_dS;
327 
328  // Vib-el Energy
329  dFdVj[nSpecies+3][nSpecies+3] = kve*theta/dij*val_dS;
330 
331  for (auto iVar = 0; iVar < nVar; iVar++)
332  for (auto jVar = 0; jVar < nVar; jVar++)
333  dFdVi[iVar][jVar] = -dFdVj[iVar][jVar];
334 
335  // Common terms
336  dFdVi[nSpecies+2][nSpecies] += 0.5*val_Fv[nSpecies];
337  dFdVj[nSpecies+2][nSpecies] += 0.5*val_Fv[nSpecies];
338  dFdVi[nSpecies+2][nSpecies+1] += 0.5*val_Fv[nSpecies+1];
339  dFdVj[nSpecies+2][nSpecies+1] += 0.5*val_Fv[nSpecies+1];
340  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
341  dFdVi[nSpecies+2][nSpecies+2] += 0.5*val_Fv[iSpecies]*(Ru/Ms[iSpecies] +
342  Cvtr[iSpecies] );
343  dFdVj[nSpecies+2][nSpecies+2] += 0.5*val_Fv[iSpecies]*(Ru/Ms[iSpecies] +
344  Cvtr[iSpecies] );
345  dFdVi[nSpecies+2][nSpecies+3] += 0.5*val_Fv[iSpecies]*val_Mean_Cvve[iSpecies];
346  dFdVj[nSpecies+2][nSpecies+3] += 0.5*val_Fv[iSpecies]*val_Mean_Cvve[iSpecies];
347  dFdVi[nSpecies+3][nSpecies+3] += 0.5*val_Fv[iSpecies]*val_Mean_Cvve[iSpecies];
348  dFdVj[nSpecies+3][nSpecies+3] += 0.5*val_Fv[iSpecies]*val_Mean_Cvve[iSpecies];
349  }
350 
351  // Unique terms
352  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
353  for (auto jSpecies = 0; jSpecies < nSpecies; jSpecies++) {
354  dFdVj[iSpecies][jSpecies] += -dJdr_j[iSpecies][jSpecies]*val_dS;
355  dFdVi[iSpecies][jSpecies] += -dJdr_i[iSpecies][jSpecies]*val_dS;
356  dFdVj[nSpecies+2][iSpecies] += -dJdr_j[jSpecies][iSpecies]*hs[jSpecies]*val_dS;
357  dFdVi[nSpecies+2][iSpecies] += -dJdr_i[jSpecies][iSpecies]*hs[jSpecies]*val_dS;
358  dFdVj[nSpecies+3][iSpecies] += -dJdr_j[jSpecies][iSpecies]*val_Mean_Eve[jSpecies]*val_dS;
359  dFdVi[nSpecies+3][iSpecies] += -dJdr_i[jSpecies][iSpecies]*val_Mean_Eve[jSpecies]*val_dS;
360  }
361  }
362 
363  } //nDim == 2
364  else {
365 
366  /*--- Geometry parameters ---*/
367  su2double thetax = theta + val_normal[0]*val_normal[0]/3.0;
368  su2double thetay = theta + val_normal[1]*val_normal[1]/3.0;
369  su2double thetaz = theta + val_normal[2]*val_normal[2]/3.0;
370  su2double etax = val_normal[1]*val_normal[2]/3.0;
371  su2double etay = val_normal[0]*val_normal[2]/3.0;
372  su2double etaz = val_normal[0]*val_normal[1]/3.0;
373  su2double pix = mu/dij * (thetax*vel[0] + etaz*vel[1] + etay*vel[2] );
374  su2double piy = mu/dij * (etaz*vel[0] + thetay*vel[1] + etax*vel[2] );
375  su2double piz = mu/dij * (etay*vel[0] + etax*vel[1] + thetaz*vel[2]);
376 
377  /*--- Populate primitive Jacobian ---*/
378 
379  // X-momentum
380  dFdVj[nSpecies][nSpecies] = mu*thetax/dij*val_dS;
381  dFdVj[nSpecies][nSpecies+1] = mu*etaz/dij*val_dS;
382  dFdVj[nSpecies][nSpecies+2] = mu*etay/dij*val_dS;
383 
384  // Y-momentum
385  dFdVj[nSpecies+1][nSpecies] = mu*etaz/dij*val_dS;
386  dFdVj[nSpecies+1][nSpecies+1] = mu*thetay/dij*val_dS;
387  dFdVj[nSpecies+1][nSpecies+2] = mu*etax/dij*val_dS;
388 
389  // Z-momentum
390  dFdVj[nSpecies+2][nSpecies] = mu*etay/dij*val_dS;
391  dFdVj[nSpecies+2][nSpecies+1] = mu*etax/dij*val_dS;
392  dFdVj[nSpecies+2][nSpecies+2] = mu*thetaz/dij*val_dS;
393 
394  // Energy
395  dFdVj[nSpecies+3][nSpecies] = pix*val_dS;
396  dFdVj[nSpecies+3][nSpecies+1] = piy*val_dS;
397  dFdVj[nSpecies+3][nSpecies+2] = piz*val_dS;
398  dFdVj[nSpecies+3][nSpecies+3] = ktr*theta/dij*val_dS;
399  dFdVj[nSpecies+3][nSpecies+4] = kve*theta/dij*val_dS;
400 
401  // Vib.-el energy
402  dFdVj[nSpecies+4][nSpecies+4] = kve*theta/dij*val_dS;
403 
404  for (auto iVar = 0; iVar < nVar; iVar++)
405  for (auto jVar = 0; jVar < nVar; jVar++)
406  dFdVi[iVar][jVar] = -dFdVj[iVar][jVar];
407 
408  // Common terms
409  for (auto iDim = 0; iDim < nDim; iDim++) {
410  dFdVi[nSpecies+3][nSpecies+iDim] += 0.5*val_Fv[nSpecies+iDim];
411  dFdVj[nSpecies+3][nSpecies+iDim] += 0.5*val_Fv[nSpecies+iDim];
412  }
413  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
414  dFdVi[nSpecies+3][nSpecies+3] += 0.5*val_Fv[iSpecies]*(Ru/Ms[iSpecies] +
415  Cvtr[iSpecies] );
416  dFdVj[nSpecies+3][nSpecies+3] += 0.5*val_Fv[iSpecies]*(Ru/Ms[iSpecies] +
417  Cvtr[iSpecies] );
418  dFdVi[nSpecies+3][nSpecies+4] += 0.5*val_Fv[iSpecies]*val_Mean_Cvve[iSpecies];

422  }

425  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
426  for (auto jSpecies = 0; jSpecies < nSpecies; jSpecies++) {
427  dFdVj[iSpecies][jSpecies] += -dJdr_j[iSpecies][jSpecies]*val_dS;
428  dFdVi[iSpecies][jSpecies] += -dJdr_i[iSpecies][jSpecies]*val_dS;
429  dFdVj[nSpecies+3][iSpecies] += -dJdr_j[jSpecies][iSpecies]*hs[jSpecies]*val_dS;
430  dFdVi[nSpecies+3][iSpecies] += -dJdr_i[jSpecies][iSpecies]*hs[jSpecies]*val_dS;
431  dFdVj[nSpecies+4][iSpecies] += -dJdr_j[jSpecies][iSpecies]*val_Mean_Eve[jSpecies]*val_dS;
432  dFdVi[nSpecies+4][iSpecies] += -dJdr_i[jSpecies][iSpecies]*val_Mean_Eve[jSpecies]*val_dS;
433  }
434  }
435 
436  } // nDim == 3

438  /*--- dFv/dUij = dFv/dVij * dVij/dUij ---*/
439  for (auto iVar = 0; iVar < nVar; iVar++)
440  for (auto jVar = 0; jVar < nVar; jVar++)
441  for (auto kVar = 0; kVar < nVar; kVar++) {
442  val_Jac_i[iVar][jVar] += dFdVi[iVar][kVar]*dVdUi[kVar][jVar];
443  val_Jac_j[iVar][jVar] += dFdVj[iVar][kVar]*dVdUj[kVar][jVar];
444  }
445  }
