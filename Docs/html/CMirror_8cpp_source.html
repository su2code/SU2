46 void CMirror::SetTransferCoeff(const CConfig* const* config) {
47 
48  const int nProcessor = size;
49 
50  vector<unsigned long> allNumVertexTarget(nProcessor);
51  vector<unsigned long> allNumVertexDonor(nProcessor);
52  vector<unsigned long> allNumNodeDonor(nProcessor);
53 
54  /*--- The target vertex information of the transpose interpolator. ---*/
55  const auto& donorVertices = transpInterpolator->targetVertices;
56 
57  targetVertices.resize(config[targetZone]->GetnMarker_All());
58 
59  /*--- Number of markers on the interface ---*/
60  const auto nMarkerInt = (config[targetZone]->GetMarker_n_ZoneInterface())/2;
61 
62  /*--- For the number of markers on the interface... ---*/
63  for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; iMarkerInt++) {
64 
65  /* High level procedure:
66  * - Gather the interpolation matrix of the donor geometry;
67  * - Set the interpolation matrix of the target as the transpose.
68  */
69 
70  /*--- On the donor side: find the tag of the boundary sharing the interface ---*/
71  const auto markDonor = config[donorZone]->FindInterfaceMarker(iMarkerInt);
72 
73  /*--- On the target side: find the tag of the boundary sharing the interface ---*/
74  const auto markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
75 
76  /*--- Checks if the zone contains the interface, if not continue to the next step ---*/
77  if (!CheckInterfaceBoundary(markDonor, markTarget)) continue;
78 
79  unsigned long nVertexDonor = 0, nVertexTarget = 0;
80  if (markDonor != -1) nVertexDonor = donor_geometry->GetnVertex( markDonor );
81  if (markTarget != -1) nVertexTarget = target_geometry->GetnVertex( markTarget );
82 
83  /*--- Count the number of donor nodes on the donor geometry. ---*/
84  unsigned long nVertexDonorLocal = 0;
85  unsigned long nNodeDonorLocal = 0;
86  for (auto iVertex = 0ul; iVertex < nVertexDonor; iVertex++) {
87  const auto iPoint = donor_geometry->vertex[markDonor][iVertex]->GetNode();
88  if (donor_geometry->nodes->GetDomain(iPoint)) {
89  nNodeDonorLocal += donorVertices[markDonor][iVertex].nDonor();
90  nVertexDonorLocal++;
91  }
92  }
93 
94  /*--- Communicate vertex and donor node counts. ---*/
95  SU2_MPI::Allgather(&nVertexTarget, 1, MPI_UNSIGNED_LONG,
96  allNumVertexTarget.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
97  SU2_MPI::Allgather(&nVertexDonorLocal, 1, MPI_UNSIGNED_LONG,
98  allNumVertexDonor.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
99  SU2_MPI::Allgather(&nNodeDonorLocal, 1, MPI_UNSIGNED_LONG,
100  allNumNodeDonor.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
101 
102  /*--- Copy donor interpolation matrix (triplet format). ---*/
103  vector<long> sendGlobalIndex(nNodeDonorLocal);
104  vector<long> sendDonorIndex(nNodeDonorLocal);
105  vector<su2double> sendDonorCoeff(nNodeDonorLocal);
106 
107  for (auto iVertex = 0ul, iDonor = 0ul; iVertex < nVertexDonor; ++iVertex) {
108 
109  auto& donor_vertex = donorVertices[markDonor][iVertex];
110  const auto iPoint = donor_geometry->vertex[markDonor][iVertex]->GetNode();
111 
112  if (!donor_geometry->nodes->GetDomain(iPoint)) continue;
113 
114  const auto nDonor = donor_vertex.nDonor();
115  const auto donorGlobalIndex = donor_geometry->nodes->GetGlobalIndex(iPoint);
116 
117  for (auto i = 0u; i < nDonor; ++i) {
118  sendGlobalIndex[iDonor] = donorGlobalIndex;
119  sendDonorIndex[iDonor] = donor_vertex.globalPoint[i];
120  sendDonorCoeff[iDonor] = donor_vertex.coefficient[i];
121  ++iDonor;
122  }
123  }
124 
125  /*--- Sort the matrix by donor index, effectively transposing the triplets. ---*/
126  vector<int> order(nNodeDonorLocal);
127  iota(order.begin(), order.end(), 0);
128  sort(order.begin(), order.end(),
129  [&sendDonorIndex](int i, int j) {
130  return sendDonorIndex[i] < sendDonorIndex[j];
131  }
132  );
133  for (int i = 0; i < int(nNodeDonorLocal); ++i) {
134  int j = order[i];
135  while (j < i) j = order[j];
136  if (i == j) continue;
137  swap(sendGlobalIndex[i], sendGlobalIndex[j]);
138  swap(sendDonorIndex[i], sendDonorIndex[j]);
139  swap(sendDonorCoeff[i], sendDonorCoeff[j]);
140  }
141  vector<int>().swap(order); // no longer needed
142 
143  /*--- Communicate donor interpolation matrix and info. We only gather the
144  * matrix in ranks that need it, i.e. have target vertices, to avoid
145  * potentially huge memory usage when running in parallel. This is an
146  * in-place Gatherv (done manually due to AD issues in the past). ---*/
147 
148  vector<int> iSendProcessor;
149  for (int iProcessor = 0; iProcessor < nProcessor; ++iProcessor)
150  if (allNumVertexDonor[iProcessor] != 0)
151  iSendProcessor.push_back(iProcessor);
152 
153  const int nSend = iSendProcessor.size();
154 
155  vector<long*> GlobalIndex(nSend,nullptr), DonorIndex(nSend,nullptr);
156  vector<su2double*> DonorCoeff(nSend,nullptr);
157 
158  /*--- For each "target processor" that needs the interpolation matrix. ---*/
159  for (int iProcessor = 0; iProcessor < nProcessor; ++iProcessor) {
160  if (allNumVertexTarget[iProcessor] == 0) continue;
161 
162  /*--- For each "donor processor that contain a part of the matrix. ---*/
163  for (int iSend = 0; iSend < nSend; ++iSend) {
164  const auto jProcessor = iSendProcessor[iSend];
165  const auto numCoeff = allNumNodeDonor[jProcessor];
166 
167  if ((rank == iProcessor) && (rank == jProcessor)) {
168  /*--- "Self" communication. ---*/
169  GlobalIndex[iSend] = sendGlobalIndex.data();
170  DonorIndex[iSend] = sendDonorIndex.data();
171  DonorCoeff[iSend] = sendDonorCoeff.data();
172  }
173  else if (rank == iProcessor) {
174  /*--- "I'm" the target, allocate and receive. ---*/
175  GlobalIndex[iSend] = new long [numCoeff];
176  DonorIndex[iSend] = new long [numCoeff];
177  DonorCoeff[iSend] = new su2double [numCoeff];
178  SU2_MPI::Recv(GlobalIndex[iSend], numCoeff, MPI_LONG, jProcessor, 0, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
179  SU2_MPI::Recv(DonorIndex[iSend], numCoeff, MPI_LONG, jProcessor, 0, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
180  SU2_MPI::Recv(DonorCoeff[iSend], numCoeff, MPI_DOUBLE, jProcessor, 0, SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
181  }
182  else if (rank == jProcessor) {
183  /*--- "I'm" the donor, send. ---*/
184  SU2_MPI::Send(sendGlobalIndex.data(), numCoeff, MPI_LONG, iProcessor, 0, SU2_MPI::GetComm());
185  SU2_MPI::Send(sendDonorIndex.data(), numCoeff, MPI_LONG, iProcessor, 0, SU2_MPI::GetComm());
186  SU2_MPI::Send(sendDonorCoeff.data(), numCoeff, MPI_DOUBLE, iProcessor, 0, SU2_MPI::GetComm());
187  }
188  }
189  }
190 
191  if (nVertexTarget) targetVertices[markTarget].resize(nVertexTarget);
192 
193  /*--- Loop over the vertices on the target marker, define one row of the transpose matrix. ---*/
194 
195  SU2_OMP_PARALLEL
196  {
197  SU2_OMP_FOR_DYN(roundUpDiv(nVertexTarget, 2*omp_get_max_threads()))
198  for (auto iVertex = 0ul; iVertex < nVertexTarget; ++iVertex) {
199 
200  auto& target_vertex = targetVertices[markTarget][iVertex];
201  const auto iPoint = target_geometry->vertex[markTarget][iVertex]->GetNode();
202 
203  if (!target_geometry->nodes->GetDomain(iPoint)) continue;
204 
205  /*--- Any point of the donor geometry, that has this target point as a donor, becomes a donor. ---*/
206  const long targetGlobalIndex = target_geometry->nodes->GetGlobalIndex(iPoint);
207 
208  /*--- Count donors and safe the binary search results (this is why we sorted the matrix). ---*/
209  auto nDonor = 0ul;
210  vector<pair<long*,long*> > ranges(nSend);
211  for (int iSend = 0; iSend < nSend; ++iSend) {
212  const auto iProcessor = iSendProcessor[iSend];
213  const auto numCoeff = allNumNodeDonor[iProcessor];
214  auto p = equal_range(DonorIndex[iSend], DonorIndex[iSend]+numCoeff, targetGlobalIndex);
215  nDonor += (p.second - p.first);
216  ranges[iSend] = p;
217  }
218 
219  target_vertex.resize(nDonor);
220 
221  /*--- Use the search results to set the interpolation coefficients. ---*/
222  for (int iSend = 0, iDonor = 0; iSend < nSend; ++iSend) {
223  const auto iProcessor = iSendProcessor[iSend];
224 
225  const auto first = ranges[iSend].first - DonorIndex[iSend];
226  const auto last = ranges[iSend].second - DonorIndex[iSend];
227 
228  for (auto iCoeff = first; iCoeff < last; ++iCoeff) {
229  target_vertex.processor[iDonor] = iProcessor;
230  target_vertex.coefficient[iDonor] = DonorCoeff[iSend][iCoeff];
231  target_vertex.globalPoint[iDonor] = GlobalIndex[iSend][iCoeff];
232  ++iDonor;
233  }
234  }
235  }
236  END_SU2_OMP_FOR
237  }
238  END_SU2_OMP_PARALLEL
239 
240  /*--- Free the heap allocations. ---*/
241  for (auto ptr : GlobalIndex) if (ptr != sendGlobalIndex.data()) delete [] ptr;
242  for (auto ptr : DonorIndex) if (ptr != sendDonorIndex.data()) delete [] ptr;
243  for (auto ptr : DonorCoeff) if (ptr != sendDonorCoeff.data()) delete [] ptr;
244 
245  } // end marker loop
246 
247 }
