39 /*!
40  * \brief A quasi-Newton fixed-point (FP) accelerator based on IQN-ILS.
41  * \note The implementation prioritizes storage, the LS problem is solved
42  * via the normal equations as that is easy to make parallel over MPI, it
43  * may however be unstable with large sample sizes (compared to QR decomp).
44  * Usage: Allocate, store the initial solution (operator (i,j), default is 0),
45  * run the FP, store its result ("FPresult"), compute new solution, use it
46  * as the new input of the FP, run the FP, etc.
47  * \ingroup BLAS
48  */
49 template<class Scalar_t, bool WithMPI = true>
50 class CQuasiNewtonInvLeastSquares {
51 public:
52  using Scalar = Scalar_t;
53  using Index = typename su2matrix<Scalar>::Index;
54  static_assert(std::is_floating_point<Scalar>::value,"");
55 
56 private:
57  using MPI_Wrapper = typename SelectMPIWrapper<Scalar>::W;
58 
59  enum: size_t {BLOCK_SIZE = 1024}; /*!< \brief Loop tiling parameter. */
60  std::vector<su2matrix<Scalar> > X, R; /*!< \brief Input and residual history of the FP. */
61  su2matrix<Scalar> work; /*!< \brief Work matrix (FP result, correction, and approx solution). */
62  su2vector<Scalar> mat, rhs, sol; /*!< \brief Matrix, rhs, and solution of the normal equations. */
63  Index iSample = 0; /*!< \brief Current sample index. */
64  Index nPtDomain = 0; /*!< \brief Local size of the history, considered in dot products. */
65 
66  void shiftHistoryLeft() {
67  for (Index i = 1; i < X.size(); ++i) {
68  /*--- Swap instead of moving to re-use the memory of the first sample.
69  * This is why X and R are not stored as contiguous blocks of mem. ---*/
70  std::swap(X[i-1], X[i]);
71  std::swap(R[i-1], R[i]);
72  }
73  }
74 
75  void computeNormalEquations() {
76  /*--- Size for the dot products. ---*/
77  const auto end = std::min<Index>(nPtDomain,work.rows())*work.cols();
78 
79  mat = Scalar(0); rhs = Scalar(0);
80 
81  /*--- Tiled part of the loop. ---*/
82  Index begin = 0;
83  while (end-begin >= BLOCK_SIZE) {
84  computeNormalEquations<BLOCK_SIZE>(mat, rhs, begin);
85  begin += BLOCK_SIZE;
86  }
87 
88  /*--- Remainder of the loop. ---*/
89  if (begin != end) {
90  computeNormalEquations<0>(mat, rhs, begin, end-begin);
91  }
92 
93  /*--- MPI reduction of the dot products. ---*/
94  if (WithMPI) {
95  const auto type = (sizeof(Scalar) < sizeof(double))? MPI_FLOAT : MPI_DOUBLE;
96 
97  su2vector<Scalar> tmp(mat.size());
98  MPI_Wrapper::Allreduce(mat.data(), tmp.data(), iSample*(iSample+1)/2,
99  type, MPI_SUM, SU2_MPI::GetComm());
100  mat = std::move(tmp);
101 
102  MPI_Wrapper::Allreduce(rhs.data(), sol.data(), iSample,
103  type, MPI_SUM, SU2_MPI::GetComm());
104  std::swap(rhs, sol);
105  }
106  }

108  template<Index StaticSize>
109  void computeNormalEquations(su2vector<Scalar>& mat,
110  su2vector<Scalar>& vec,
111  Index start,
112  Index dynSize = 0) const {
113  /*--- Select either the static or dynamic size, optimizes inner loop. ---*/
114  const auto blkSize = StaticSize? StaticSize : dynSize;
115 
116  for (Index i = 0; i < iSample; ++i) {
117  const auto ri1 = R[i+1].data() + start;
118  const auto ri0 = R[i].data() + start;
119  /*--- Off-diagonal coefficients. ---*/
120  for (Index j = 0; j < i; ++j) {
121  const auto rj1 = R[j+1].data() + start;
122  const auto rj0 = R[j].data() + start;
123  /*--- Sum of partial sums to reduce trunc. error. ---*/
124  Scalar sum = 0;
125  SU2_OMP_SIMD
126  for (Index k = 0; k < blkSize; ++k) {
127  sum += (ri1[k]-ri0[k]) * (rj1[k]-rj0[k]);
128  }
129  /*--- 1D index of (i,j) in lower triangular storage. ---*/
130  const auto iCoeff = i*(i+1)/2 + j;
131  mat(iCoeff) += sum;
132  }
133  /*--- Diagonal coeff and residual fused. ---*/
134  Scalar diag = 0, res = 0;
135  const auto r = R[iSample].data() + start;
136  SU2_OMP_SIMD
137  for (Index k = 0; k < blkSize; ++k) {
138  diag += pow(ri1[k]-ri0[k], 2);
139  res += (ri1[k]-ri0[k]) * r[k];
140  }
141  mat(i*(i+3)/2) += diag;
142  vec(i) -= res;
143  }
144  }

155  /*!
156  * \brief Resize the object.
157  * \param[in] nsample - Number of samples used to build the FP history.
158  * \param[in] npt - Size of the solution including any halos.
159  * \param[in] nvar - Number of solution variables.
160  * \param[in] nptdomain - Local size (<= npt), if 0 it defaults to npt.
161  */
162  void resize(Index nsample, Index npt, Index nvar, Index nptdomain = 0) {
163  if (nptdomain > npt || nsample < 2)
164  SU2_MPI::Error("Invalid quasi-Newton parameters", CURRENT_FUNCTION);
165  iSample = 0;
166  nPtDomain = nptdomain? nptdomain : npt;
167  work.resize(npt,nvar);
168  X.clear();
169  R.clear();
170  for (Index i = 0; i < nsample; ++i) {
171  X.emplace_back(npt,nvar);
172  R.emplace_back(npt,nvar);
173  }
174  X[0] = Scalar(0);
175  /*--- Lower triangular packed storage. ---*/
176  mat.resize(nsample*(nsample-1)/2);
177  rhs.resize(nsample-1);
178  sol.resize(nsample-1);
179  }

184  /*! \brief Discard all history, keeping the current solution. */
185  void reset() { std::swap(X[0], X[iSample]); iSample = 0; }

205  /*!
206  * \brief Compute and return a new approximation.
207  * \note To be used after storing the FP result.
208  */
209  const su2matrix<Scalar>& compute() {
210  /*--- Compute FP residual, clear correction. ---*/
211  SU2_OMP_SIMD
212  for (Index i = 0; i < work.size(); ++i) {
213  R[iSample].data()[i] = work.data()[i] - X[iSample].data()[i];
214  work.data()[i] = Scalar(0);
215  }
216 
217  if (iSample > 0) {
218  /*--- Solve the normal equations. ---*/
219  computeNormalEquations();
220  CSymmetricMatrix pseudoInv(iSample);
221  for (Index i = 0, k = 0; i < iSample; ++i)
222  for (Index j = 0; j <= i; ++j)
223  pseudoInv(i,j) = mat(k++);
224  pseudoInv.Invert(true);
225  pseudoInv.MatVecMult(rhs.data(), sol.data());
226 
227  /*--- Compute correction, cleared before for less trunc. error. ---*/
228  for (Index k = 0; k < iSample; ++k) {
229  const auto x1 = X[k+1].data();
230  const auto r1 = R[k+1].data();
231  const auto x0 = X[k].data();
232  const auto r0 = R[k].data();
233  SU2_OMP_SIMD
234  for (Index i = 0; i < work.size(); ++i) {
235  Scalar dy = r1[i]-r0[i] + x1[i]-x0[i];
236  work.data()[i] += sol(k) * dy;
237  }
238  }
239  }
240 
241  /*--- Check for need to shift left. ---*/
242  if (iSample+1 == X.size()) {
243  shiftHistoryLeft();
244  iSample--;
245  }
246 
247  /*--- Set new solution. ---*/
248  SU2_OMP_SIMD
249  for (Index i = 0; i < work.size(); ++i)
250  work.data()[i] += R[iSample].data()[i] + X[iSample].data()[i];
251  std::swap(X[++iSample], work);
252 
253  return solution();
254  }
