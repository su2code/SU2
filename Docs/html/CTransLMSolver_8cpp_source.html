41 CTransLMSolver::CTransLMSolver(CGeometry *geometry, CConfig *config, unsigned short iMesh)
42  : CTurbSolver(geometry, config, true) {
43  unsigned long iPoint;
44  ifstream restart_file;
45  string text_line;
46 
47  bool multizone = config->GetMultizone_Problem();
48 
49  /*--- Dimension of the problem --> 2 Transport equations (intermittency, Reth) ---*/
50  nVar = 2;
51  nPrimVar = 2;
52  nPoint = geometry->GetnPoint();
53  nPointDomain = geometry->GetnPointDomain();
54 
55  /*--- Initialize nVarGrad for deallocation ---*/
56 
57  nVarGrad = nVar;
58 
59  /*--- Define geometry constants in the solver structure ---*/
60 
61  nDim = geometry->GetnDim();
62 
63  /*--- Define variables needed for transition from config file */
64  options = config->GetLMParsedOptions();
65  TransCorrelations.SetOptions(options);
66  TurbFamily = TurbModelFamily(config->GetKind_Turb_Model());
67 
68  /*--- Single grid simulation ---*/
69 
70  if (iMesh == MESH_0) {
71 
72  /*--- Define some auxiliary vector related with the residual ---*/
73 
74  Residual_RMS.resize(nVar,0.0);
75  Residual_Max.resize(nVar,0.0);
76  Point_Max.resize(nVar,0);
77  Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
78 
79  /*--- Initialization of the structure of the whole Jacobian ---*/
80 
81  if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (LM transition model)." << endl;
82  Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config, ReducerStrategy);
83  LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
84  LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
85  System.SetxIsZero(true);
86 
87  if (ReducerStrategy)
88  EdgeFluxes.Initialize(geometry->GetnEdge(), geometry->GetnEdge(), nVar, nullptr);
89 
90  /*--- Initialize the BGS residuals in multizone problems. ---*/
91  if (multizone){
92  Residual_BGS.resize(nVar,0.0);
93  Residual_Max_BGS.resize(nVar,0.0);
94  Point_Max_BGS.resize(nVar,0);
95  Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
96  }
97 
98  }
99 
100  /*--- Initialize lower and upper limits---*/
101  lowerlimit[0] = 1.0e-4;
102  upperlimit[0] = 5.0;
103 
104  lowerlimit[1] = 1.0e-4;
105  upperlimit[1] = 1.0e15;
106 
107  /*--- Far-field flow state quantities and initialization. ---*/
108  const su2double Intensity = config->GetTurbulenceIntensity_FreeStream()*100.0;
109 
110  const su2double Intermittency_Inf = 1.0;
111  su2double ReThetaT_Inf = 100.0;
112 
113  /*--- Momentum thickness Reynolds number, initialized from freestream turbulent intensity*/
114  if (Intensity <= 1.3) {
115  if(Intensity >=0.027) {
116  ReThetaT_Inf = (1173.51-589.428*Intensity+0.2196/(Intensity*Intensity));
117  }
118  else {
119  ReThetaT_Inf = (1173.51-589.428*Intensity+0.2196/(0.27*0.27));
120  }
121  }
122  else if(Intensity>1.3) {
123  ReThetaT_Inf = 331.5*pow(Intensity-0.5658,-0.671);
124  }
125 
126  Solution_Inf[0] = Intermittency_Inf;
127  Solution_Inf[1] = ReThetaT_Inf;
128 
129  /*--- Initialize the solution to the far-field state everywhere. ---*/
130  nodes = new CTransLMVariable(Intermittency_Inf, ReThetaT_Inf, 1.0, 1.0, nPoint, nDim, nVar, config);
131  SetBaseClassPointerToNodes();
132 
133  /*--- MPI solution ---*/
134 
135  InitiateComms(geometry, config, SOLUTION);
136  CompleteComms(geometry, config, SOLUTION);
137 
138  /*--- Initializate quantities for SlidingMesh Interface ---*/
139 
140  SlidingState.resize(nMarker);
141  SlidingStateNodes.resize(nMarker);
142 
143  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {
144  if (config->GetMarker_All_KindBC(iMarker) == FLUID_INTERFACE) {
145  SlidingState[iMarker].resize(nVertex[iMarker], nPrimVar+1) = nullptr;
146  SlidingStateNodes[iMarker].resize(nVertex[iMarker],0);
147  }
148  }
149 
150  /*-- Allocation of inlets has to happen in derived classes (not CTurbSolver),
151  due to arbitrary number of turbulence variables ---*/
152 
153  Inlet_TurbVars.resize(nMarker);
154  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {
155  Inlet_TurbVars[iMarker].resize(nVertex[iMarker],nVar);
156  for (unsigned long iVertex = 0; iVertex < nVertex[iMarker]; ++iVertex) {
157  Inlet_TurbVars[iMarker](iVertex,0) = Intermittency_Inf;
158  Inlet_TurbVars[iMarker](iVertex,1) = ReThetaT_Inf;
159  }
160  }
161 
162  const su2double CFL = config->GetCFL(MGLevel)*config->GetCFLRedCoeff_Turb();
163  for (iPoint = 0; iPoint < nPoint; iPoint++) {
164  nodes->SetLocalCFL(iPoint, CFL);
165  }
166  Min_CFL_Local = CFL;
167  Max_CFL_Local = CFL;
168  Avg_CFL_Local = CFL;
169 
170  /*--- Add the solver name (max 8 characters) ---*/
171  SolverName = "LM model";
172 
173 }

183 void CTransLMSolver::Postprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh) {
184 
185  /*--- Compute LM model gradients. ---*/
186 
187  if (config->GetKind_Gradient_Method() == GREEN_GAUSS) {
188  SetSolution_Gradient_GG(geometry, config);
189  }
190  if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) {
191  SetSolution_Gradient_LS(geometry, config);
192  }
193 
194  AD::StartNoSharedReading();
195  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
196  auto* turbNodes = su2staticcast_p<CTurbVariable*>(solver_container[TURB_SOL]->GetNodes());
197 
198  SU2_OMP_FOR_STAT(omp_chunk_size)
199  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint ++) {
200 
201  // Here the nodes already have the new solution, thus I have to compute everything from scratch
202 
203  const su2double rho = flowNodes->GetDensity(iPoint);
204  const su2double mu = flowNodes->GetLaminarViscosity(iPoint);
205  const su2double muT = turbNodes->GetmuT(iPoint);
206  const su2double dist = geometry->nodes->GetWall_Distance(iPoint);
207  su2double VorticityMag = GeometryToolbox::Norm(3, flowNodes->GetVorticity(iPoint));
208  su2double StrainMag =flowNodes->GetStrainMag(iPoint);
209  VorticityMag = max(VorticityMag, 1e-12);
210  StrainMag = max(StrainMag, 1e-12); // safety against division by zero
211  const su2double Intermittency = nodes->GetSolution(iPoint,0);
212  const su2double Re_t = nodes->GetSolution(iPoint,1);
213  const su2double Re_v = rho*dist*dist*StrainMag/mu;
214  const su2double vel_u = flowNodes->GetVelocity(iPoint, 0);
215  const su2double vel_v = flowNodes->GetVelocity(iPoint, 1);
216  const su2double vel_w = (nDim ==3) ? flowNodes->GetVelocity(iPoint, 2) : 0.0;
217  const su2double VelocityMag = sqrt(vel_u*vel_u + vel_v*vel_v + vel_w*vel_w);
218  su2double omega = 0.0;
219  su2double k = 0.0;
220  if(TurbFamily == TURB_FAMILY::KW){
221  omega = turbNodes->GetSolution(iPoint,1);
222  k = turbNodes->GetSolution(iPoint,0);
223  }
224  su2double Tu = 1.0;
225  if(TurbFamily == TURB_FAMILY::KW)
226  Tu = max(100.0*sqrt( 2.0 * k / 3.0 ) / VelocityMag,0.027);
227  if(TurbFamily == TURB_FAMILY::SA)
228  Tu = config->GetTurbulenceIntensity_FreeStream()*100;
229 
230  const su2double Corr_Rec = TransCorrelations.ReThetaC_Correlations(Tu, Re_t);
231 
232  su2double R_t = 1.0;
233  if(TurbFamily == TURB_FAMILY::KW)
234  R_t = rho*k/ mu/ omega;
235  if(TurbFamily == TURB_FAMILY::SA)
236  R_t = muT/ mu;
237 
238  const su2double f_reattach = exp(-pow(R_t/20,4));
239  su2double f_wake = 0.0;
240  if(TurbFamily == TURB_FAMILY::KW){
241  const su2double re_omega = rho*omega*dist*dist/mu;
242  f_wake = exp(-pow(re_omega/(1.0e+05),2));
243  }
244  if(TurbFamily == TURB_FAMILY::SA)
245  f_wake = 1.0;
246 
247  const su2double theta_bl = Re_t*mu / rho /VelocityMag;
248  const su2double delta_bl = 7.5*theta_bl;
249  const su2double delta = 50.0*VorticityMag*dist/VelocityMag*delta_bl + 1e-20;
250  const su2double var1 = (Intermittency-1.0/50.0)/(1.0-1.0/50.0);
251  const su2double var2 = 1.0 - pow(var1,2.0);
252  const su2double f_theta = min(max(f_wake*exp(-pow(dist/delta, 4)), var2), 1.0);
253  su2double Intermittency_Sep = 2.0*max(0.0, Re_v/(3.235*Corr_Rec)-1.0)*f_reattach;
254  Intermittency_Sep = min(Intermittency_Sep,2.0)*f_theta;
255  Intermittency_Sep = min(max(0.0, Intermittency_Sep), 2.0);
256  nodes -> SetIntermittencySep(iPoint, Intermittency_Sep);
257  nodes -> SetIntermittencyEff(iPoint, Intermittency_Sep);
258 
259  }
260  END_SU2_OMP_FOR
261 
262  AD::EndNoSharedReading();
263 }

279 void CTransLMSolver::Source_Residual(CGeometry *geometry, CSolver **solver_container,
280  CNumerics **numerics_container, CConfig *config, unsigned short iMesh) {
281 
282  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
283 
284  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
285  //auto* turbNodes = su2staticcast_p<CFlowVariable*>(solver_container[TURB_SOL]->GetNodes());
286  CVariable* turbNodes = solver_container[TURB_SOL]->GetNodes();
287 
288  /*--- Pick one numerics object per thread. ---*/
289  auto* numerics = numerics_container[SOURCE_FIRST_TERM + omp_get_thread_num()*MAX_TERMS];
290 
291  /*--- Loop over all points. ---*/
292 
293  AD::StartNoSharedReading();
294 
295  SU2_OMP_FOR_DYN(omp_chunk_size)
296  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
297 
298 
299 
300  /*--- Conservative variables w/o reconstruction ---*/
301 
302  numerics->SetPrimitive(flowNodes->GetPrimitive(iPoint), nullptr);
303 
304  /*--- Gradient of the primitive and conservative variables ---*/
305 
306  numerics->SetPrimVarGradient(flowNodes->GetGradient_Primitive(iPoint), nullptr);
307 
308  /*--- Turbulent variables w/o reconstruction, and its gradient ---*/
309  /*--- ScalarVar & ScalarVarGradient : Turbulence model solution(k&w) ---*/
310 
311  numerics->SetScalarVar(turbNodes->GetSolution(iPoint), nullptr);
312  numerics->SetScalarVarGradient(turbNodes->GetGradient(iPoint), nullptr);
313 
314  /*--- Transition variables w/o reconstruction, and its gradient ---*/
315 
316  numerics->SetTransVar(nodes->GetSolution(iPoint), nullptr);
317  numerics->SetTransVarGradient(nodes->GetGradient(iPoint), nullptr);
318 
319  /*--- Set volume ---*/
320 
321  numerics->SetVolume(geometry->nodes->GetVolume(iPoint));
322 
323  /*--- Set distance to the surface ---*/
324 
325  numerics->SetDistance(geometry->nodes->GetWall_Distance(iPoint), 0.0);
326 
327  /*--- Set vorticity and strain rate magnitude ---*/
328 
329  numerics->SetVorticity(flowNodes->GetVorticity(iPoint), nullptr);
330 
331  numerics->SetStrainMag(flowNodes->GetStrainMag(iPoint), 0.0);
332 
333  /*--- Set coordinate (for debugging) ---*/
334  numerics->SetCoord(geometry->nodes->GetCoord(iPoint), nullptr);
335 
336  if (options.LM2015) {
337  /*--- Set local grid length (for LM2015)*/
338  numerics->SetLocalGridLength(geometry->nodes->GetMaxLength(iPoint));
339  }
340 
341  /*--- Compute the source term ---*/
342 
343  auto residual = numerics->ComputeResidual(config);
344 
345  /*--- Subtract residual and the Jacobian ---*/
346 
347  LinSysRes.SubtractBlock(iPoint, residual);
348  if (implicit) Jacobian.SubtractBlock2Diag(iPoint, residual.jacobian_i);
349 
350  }
351  END_SU2_OMP_FOR
352 
353  AD::EndNoSharedReading();
354 
355 }

361 void CTransLMSolver::BC_HeatFlux_Wall(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
362  CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
363 
364  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
365 
366  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
367  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
368 
369  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
370 
371  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
372 
373  if (geometry->nodes->GetDomain(iPoint)) {
374 
375  /*--- Allocate the value at the infinity ---*/
376 
377  auto V_infty = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
378 
379  /*--- Retrieve solution at the farfield boundary node ---*/
380 
381  auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
382 
383  conv_numerics->SetPrimitive(V_domain, V_infty);
384 
385  /*--- Set turbulent variable at the wall, and at infinity ---*/
386 
387  conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), Solution_Inf);
388 
389  /*--- Set Normal (it is necessary to change the sign) ---*/
390  /*--- It's mean wall normal zero flux. */
391 
392  su2double Normal[MAXNDIM] = {0.0};
393  for (auto iDim = 0u; iDim < nDim; iDim++)
394  Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
395  conv_numerics->SetNormal(Normal);
396 
397  /*--- Grid Movement ---*/
398 
399  if (dynamic_grid)
400  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint),
401  geometry->nodes->GetGridVel(iPoint));
402 
403  /*--- Compute residuals and Jacobians ---*/
404 
405  auto residual = conv_numerics->ComputeResidual(config);
406 
407  /*--- Add residuals and Jacobians ---*/
408 
409  LinSysRes.AddBlock(iPoint, residual);
410  if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
411  }
412  }
413  END_SU2_OMP_FOR

424 void CTransLMSolver::BC_Inlet(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics, CNumerics *visc_numerics, CConfig *config,
425  unsigned short val_marker) {
426  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
427 
428  /*--- Loop over all the vertices on this boundary marker ---*/
429 
430  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
431  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
432 
433  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
434 
435  /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
436 
437  if (geometry->nodes->GetDomain(iPoint)) {
438 
439  /*--- Normal vector for this vertex (negate for outward convention) ---*/
440 
441  su2double Normal[MAXNDIM] = {0.0};
442  for (auto iDim = 0u; iDim < nDim; iDim++)
443  Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
444  conv_numerics->SetNormal(Normal);
445 
446  /*--- Allocate the value at the inlet ---*/
447 
448  auto V_inlet = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
449 
450  /*--- Retrieve solution at the farfield boundary node ---*/
451 
452  auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
453 
454  /*--- Set various quantities in the solver class ---*/
455 
456  conv_numerics->SetPrimitive(V_domain, V_inlet);
457 
458  /*--- Non-dimensionalize Inlet_TurbVars if Inlet-Files are used. ---*/
459  su2double Inlet_Vars[MAXNVAR];
460  Inlet_Vars[0] = Inlet_TurbVars[val_marker][iVertex][0];
461  Inlet_Vars[1] = Inlet_TurbVars[val_marker][iVertex][1];
462  if (config->GetInlet_Profile_From_File()) {
463  Inlet_Vars[0] /= pow(config->GetVelocity_Ref(), 2);
464  Inlet_Vars[1] *= config->GetViscosity_Ref() / (config->GetDensity_Ref() * pow(config->GetVelocity_Ref(), 2));
465  }
466 
467  /*--- Set the LM variable states. ---*/
468  /*--- Load the inlet transition LM model variables (uniform by default). ---*/
469 
470  conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), Inlet_Vars);
471 
472  /*--- Set various other quantities in the solver class ---*/
473 
474  if (dynamic_grid)
475  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint),
476  geometry->nodes->GetGridVel(iPoint));
477 
478  /*--- Compute the residual using an upwind scheme ---*/
479 
480  auto residual = conv_numerics->ComputeResidual(config);
481  LinSysRes.AddBlock(iPoint, residual);
482 
483  /*--- Jacobian contribution for implicit integration ---*/
484 
485  if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
486  }
487  }
488  END_SU2_OMP_FOR
489 
490 }

497 void CTransLMSolver::LoadRestart(CGeometry** geometry, CSolver*** solver, CConfig* config, int val_iter,
498  bool val_update_geo) {
499 
500  const string restart_filename = config->GetFilename(config->GetSolution_FileName(), "", val_iter);
501 
502  /*--- To make this routine safe to call in parallel most of it can only be executed by one thread. ---*/
503  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
504  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
505 
506  if (config->GetRead_Binary_Restart()) {
507  Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
508  } else {
509  Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
510  }
511 
512  /*--- Skip flow variables ---*/
513 
514  unsigned short skipVars = nDim + solver[MESH_0][FLOW_SOL]->GetnVar() + solver[MESH_0][TURB_SOL] ->GetnVar();
515 
516  /*--- Adjust the number of solution variables in the incompressible
517  restart. We always carry a space in nVar for the energy equation in the
518  mean flow solver, but we only write it to the restart if it is active.
519  Therefore, we must reduce skipVars here if energy is inactive so that
520  the turbulent variables are read correctly. ---*/
521 
522  const bool incompressible = (config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE);
523  const bool energy = config->GetEnergy_Equation();
524  const bool weakly_coupled_heat = config->GetWeakly_Coupled_Heat();
525 
526  if (incompressible && ((!energy) && (!weakly_coupled_heat))) skipVars--;
527 
528  /*--- Load data from the restart into correct containers. ---*/
529 
530  unsigned long counter = 0;
531  for (auto iPoint_Global = 0ul; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++) {
532  /*--- Retrieve local index. If this node from the restart file lives
533  on the current processor, we will load and instantiate the vars. ---*/
534 
535  const auto iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
536 
537  if (iPoint_Local > -1) {
538  /*--- We need to store this point's data, so jump to the correct
539  offset in the buffer of data from the restart file and load it. ---*/
540 
541  const auto index = counter * Restart_Vars[1] + skipVars;
542  for (auto iVar = 0u; iVar < nVar; iVar++) nodes->SetSolution(iPoint_Local, iVar, Restart_Data[index + iVar]);
543  nodes ->SetIntermittencySep(iPoint_Local, Restart_Data[index + 2]);
544  nodes ->SetIntermittencyEff(iPoint_Local, Restart_Data[index + 3]);
545 
546  /*--- Increment the overall counter for how many points have been loaded. ---*/
547  counter++;
548  }
549  }
550 
551  /*--- Detect a wrong solution file ---*/
552 
553  if (counter != nPointDomain) {
554  SU2_MPI::Error(string("The solution file ") + restart_filename + string(" does not match with the mesh file!\n") +
555  string("This can be caused by empty lines at the end of the file."),
556  CURRENT_FUNCTION);
557  }
558 
559  } // end SU2_OMP_MASTER, pre and postprocessing are thread-safe.
560  END_SU2_OMP_SAFE_GLOBAL_ACCESS
561 
562  /*--- MPI solution and compute the eddy viscosity ---*/
563 
564  solver[MESH_0][TRANS_SOL]->InitiateComms(geometry[MESH_0], config, SOLUTION);
565  solver[MESH_0][TRANS_SOL]->CompleteComms(geometry[MESH_0], config, SOLUTION);
566 
567  /*--- For turbulent+species simulations the solver Pre-/Postprocessing is done by the species solver. ---*/
568  if (config->GetKind_Species_Model() == SPECIES_MODEL::NONE) {
569  solver[MESH_0][FLOW_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER,
570  RUNTIME_FLOW_SYS, false);
571  solver[MESH_0][TURB_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
572  solver[MESH_0][TRANS_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
573  }
574 
575  /*--- Interpolate the solution down to the coarse multigrid levels ---*/
576 
577  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); iMesh++) {
578 
579  MultigridRestriction(*geometry[iMesh - 1], solver[iMesh - 1][TRANS_SOL]->GetNodes()->GetSolution(),
580  *geometry[iMesh], solver[iMesh][TRANS_SOL]->GetNodes()->GetSolution());
581  solver[iMesh][TRANS_SOL]->InitiateComms(geometry[iMesh], config, SOLUTION);
582  solver[iMesh][TRANS_SOL]->CompleteComms(geometry[iMesh], config, SOLUTION);
583 
584  if (config->GetKind_Species_Model() == SPECIES_MODEL::NONE) {
585  solver[iMesh][FLOW_SOL]->Preprocessing(geometry[iMesh], solver[iMesh], config, iMesh, NO_RK_ITER, RUNTIME_FLOW_SYS,
586  false);
587  solver[iMesh][TRANS_SOL]->Postprocessing(geometry[iMesh], solver[iMesh], config, iMesh);
588  }
589  }
590 
591  /*--- Go back to single threaded execution. ---*/
592  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
593  /*--- Delete the class memory that is used to load the restart. ---*/
594 
595  delete[] Restart_Vars;
596  Restart_Vars = nullptr;
597  delete[] Restart_Data;
598  Restart_Data = nullptr;
599  }
600  END_SU2_OMP_SAFE_GLOBAL_ACCESS
601 
602 }
