48 void CSinglezoneDriver::StartSolver() {
49 
50  StartTime = SU2_MPI::Wtime();
51 
52  config_container[ZONE_0]->Set_StartTime(StartTime);
53 
54  /*--- Main external loop of the solver. Runs for the number of time steps required. ---*/
55 
56  if (rank == MASTER_NODE)
57  cout << endl <<"------------------------------ Begin Solver -----------------------------" << endl;
58 
59  if (rank == MASTER_NODE){
60  cout << endl <<"Simulation Run using the Single-zone Driver" << endl;
61  if (driver_config->GetTime_Domain())
62  cout << "The simulation will run for "
63  << driver_config->GetnTime_Iter() - config_container[ZONE_0]->GetRestart_Iter() << " time steps." << endl;
64  }
65 
66  /*--- Set the initial time iteration to the restart iteration. ---*/
67  if (config_container[ZONE_0]->GetRestart() && driver_config->GetTime_Domain())
68  TimeIter = config_container[ZONE_0]->GetRestart_Iter();
69 
70  /*--- Run the problem until the number of time iterations required is reached. ---*/
71  while ( TimeIter < config_container[ZONE_0]->GetnTime_Iter() ) {
72 
73  /*--- Perform some preprocessing before starting the time-step simulation. ---*/
74 
75  Preprocess(TimeIter);
76 
77  /*--- Run a time-step iteration of the single-zone problem. ---*/
78 
79  Run();
80 
81  /*--- Perform some postprocessing on the solution before the update ---*/
82 
83  Postprocess();
84 
85  /*--- Update the solution for dual time stepping strategy ---*/
86 
87  Update();
88 
89  /*--- Monitor the computations after each iteration. ---*/
90 
91  Monitor(TimeIter);
92 
93  /*--- Output the solution in files. ---*/
94 
95  Output(TimeIter);
96 
97  /*--- Save iteration solution for libROM ---*/
98  if (config_container[MESH_0]->GetSave_libROM()) {
99  solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->SavelibROM(geometry_container[ZONE_0][INST_0][MESH_0],
100  config_container[ZONE_0], StopCalc);
101  }
102 
103  /*--- If the convergence criteria has been met, terminate the simulation. ---*/
104 
105  if (StopCalc) break;
106 
107  TimeIter++;
108 
109  }
110 
111 }

113 void CSinglezoneDriver::Preprocess(unsigned long TimeIter) {
114 
115  /*--- Set runtime option ---*/
116 
117  Runtime_Options();
118 
119  /*--- Set the current time iteration in the config ---*/
120 
121  config_container[ZONE_0]->SetTimeIter(TimeIter);
122 
123  /*--- Store the current physical time in the config container, as
124  this can be used for verification / MMS. This should also be more
125  general once the drivers are more stable. ---*/
126 
127  if (config_container[ZONE_0]->GetTime_Marching() != TIME_MARCHING::STEADY)
128  config_container[ZONE_0]->SetPhysicalTime(static_cast<su2double>(TimeIter)*config_container[ZONE_0]->GetDelta_UnstTimeND());
129  else
130  config_container[ZONE_0]->SetPhysicalTime(0.0);
131 
132  /*--- Set the initial condition for EULER/N-S/RANS ---------------------------------------------*/
133  if (config_container[ZONE_0]->GetFluidProblem()) {
134  solver_container[ZONE_0][INST_0][MESH_0][FLOW_SOL]->SetInitialCondition(geometry_container[ZONE_0][INST_0],
135  solver_container[ZONE_0][INST_0],
136  config_container[ZONE_0], TimeIter);
137  }
138  else if (config_container[ZONE_0]->GetHeatProblem()) {
139  /*--- Set the initial condition for HEAT equation ---------------------------------------------*/
140  solver_container[ZONE_0][INST_0][MESH_0][HEAT_SOL]->SetInitialCondition(geometry_container[ZONE_0][INST_0],
141  solver_container[ZONE_0][INST_0],
142  config_container[ZONE_0], TimeIter);
143  }
144 
145  SU2_MPI::Barrier(SU2_MPI::GetComm());
146 
147  /*--- Run a predictor step ---*/
148  if (config_container[ZONE_0]->GetPredictor())
149  iteration_container[ZONE_0][INST_0]->Predictor(output_container[ZONE_0], integration_container, geometry_container, solver_container,
150  numerics_container, config_container, surface_movement, grid_movement, FFDBox, ZONE_0, INST_0);
151 
152  /*--- Perform a dynamic mesh update if required. ---*/
153  /*--- For the Disc.Adj. of a case with (rigidly) moving grid, the appropriate
154  mesh cordinates are read from the restart files. ---*/
155  if (!(config_container[ZONE_0]->GetGrid_Movement() && config_container[ZONE_0]->GetDiscrete_Adjoint()))
156  DynamicMeshUpdate(TimeIter);
157 
158 }

221 void CSinglezoneDriver::DynamicMeshUpdate(unsigned long TimeIter) {
222 
223  auto iteration = iteration_container[ZONE_0][INST_0];
224 
225  /*--- Legacy dynamic mesh update - Only if GRID_MOVEMENT = YES ---*/
226  if (config_container[ZONE_0]->GetGrid_Movement()) {
227  iteration->SetGrid_Movement(geometry_container[ZONE_0][INST_0],surface_movement[ZONE_0],
228  grid_movement[ZONE_0][INST_0], solver_container[ZONE_0][INST_0],
229  config_container[ZONE_0], 0, TimeIter);
230  }
231 
232  /*--- New solver - all the other routines in SetGrid_Movement should be adapted to this one ---*/
233  /*--- Works if DEFORM_MESH = YES ---*/
234  iteration->SetMesh_Deformation(geometry_container[ZONE_0][INST_0],
235  solver_container[ZONE_0][INST_0][MESH_0],
236  numerics_container[ZONE_0][INST_0][MESH_0],
237  config_container[ZONE_0], RECORDING::CLEAR_INDICES);
238 
239  /*--- Update the wall distances if the mesh was deformed. ---*/
240  if (config_container[ZONE_0]->GetGrid_Movement() ||
241  config_container[ZONE_0]->GetDeform_Mesh()) {
242  CGeometry::ComputeWallDistance(config_container, geometry_container);
243  }

246 bool CSinglezoneDriver::Monitor(unsigned long TimeIter){
247 
248  unsigned long nInnerIter, InnerIter, nTimeIter;
249  su2double MaxTime, CurTime;
250  bool TimeDomain, InnerConvergence, TimeConvergence, FinalTimeReached, MaxIterationsReached;
251 
252  nInnerIter = config_container[ZONE_0]->GetnInner_Iter();
253  InnerIter = config_container[ZONE_0]->GetInnerIter();
254  nTimeIter = config_container[ZONE_0]->GetnTime_Iter();
255  MaxTime = config_container[ZONE_0]->GetMax_Time();
256  CurTime = output_container[ZONE_0]->GetHistoryFieldValue("CUR_TIME");
257 
258  TimeDomain = config_container[ZONE_0]->GetTime_Domain();
259 
260 
261  /*--- Check whether the inner solver has converged --- */
262 
263  if (TimeDomain == NO){
264 
265  InnerConvergence = output_container[ZONE_0]->GetConvergence();
266  MaxIterationsReached = InnerIter+1 >= nInnerIter;
267 
268  if ((MaxIterationsReached || InnerConvergence) && (rank == MASTER_NODE)) {
269  cout << endl << "----------------------------- Solver Exit -------------------------------" << endl;
270  if (InnerConvergence) cout << "All convergence criteria satisfied." << endl;
271  else cout << endl << "Maximum number of iterations reached (ITER = " << nInnerIter << ") before convergence." << endl;
272  output_container[ZONE_0]->PrintConvergenceSummary();
273  cout << "-------------------------------------------------------------------------" << endl;
274  }
275 
276  StopCalc = MaxIterationsReached || InnerConvergence;
277  }
278 
279 
280 
281  if (TimeDomain == YES) {
282 
283  /*--- Check whether the outer time integration has reached the final time ---*/
284 
285  TimeConvergence = GetTimeConvergence();
286 
287  FinalTimeReached = CurTime >= MaxTime;
288  MaxIterationsReached = TimeIter+1 >= nTimeIter;
289 
290  if ((FinalTimeReached || MaxIterationsReached || TimeConvergence) && (rank == MASTER_NODE)){
291  cout << endl << "----------------------------- Solver Exit -------------------------------";
292  if (TimeConvergence) cout << endl << "All windowed time-averaged convergence criteria are fullfilled." << endl;
293  if (FinalTimeReached) cout << endl << "Maximum time reached (MAX_TIME = " << MaxTime << "s)." << endl;
294  if (MaxIterationsReached) cout << endl << "Maximum number of time iterations reached (TIME_ITER = " << nTimeIter << ")." << endl;
295  cout << "-------------------------------------------------------------------------" << endl;
296  }
297  StopCalc = FinalTimeReached || MaxIterationsReached|| TimeConvergence;
298  }
299 
300  /*--- Reset the inner convergence --- */
301 
302  output_container[ZONE_0]->SetConvergence(false);
303 
304  /*--- Increase the total iteration count --- */
305 
306  IterCount += config_container[ZONE_0]->GetInnerIter()+1;
307 
308  return StopCalc;
309 }
