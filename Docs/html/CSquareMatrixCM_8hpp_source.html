111  /*!

114  template<class ForwardIt>
115  void MatVecMult(ForwardIt vec_in, ForwardIt vec_out) const
116  {
117  for (int i = 0; i < Size(); ++i) {
118  *vec_out = 0.0;
119  auto vec = vec_in;
120  for (int k = 0; k < Size(); ++k)
121  *vec_out += *(vec++) * mat(i,k);
122  ++vec_out;
123  }
124  }
