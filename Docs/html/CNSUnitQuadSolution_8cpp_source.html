37 
38  /*--- Write a message that the solution is initialized for the
39  Navier-Stokes case on a unit quad. Note that heat conduction
40  is neglected for this case. ---*/
41 
42  if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
43  cout << endl;
44  cout << "Warning: Fluid properties and solution are being " << endl;
45  cout << " initialized for the Navier Stokes equations " << endl;
46  cout << " on a unit quad. Note that heat conduction " << endl;
47  cout << " is neglected for this case!!!" << endl;
48  cout << endl << flush;
49  }
50 
51  /*--- Coefficients, needed to determine the solution. ---*/
52  Gm1 = config->GetGamma() - 1.0;
53  flowAngle = config->GetAoA()*PI_NUMBER/180.0;
54  Viscosity = config->GetMu_ConstantND();
55 
56  /*--- Perform some sanity and error checks for this solution here. ---*/
57  if(config->GetTime_Marching() != TIME_MARCHING::STEADY)
58  SU2_MPI::Error("Steady mode must be selected for the NS Unit Quad case",
59  CURRENT_FUNCTION);
60 
61  if(Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::RANS &&
62  Kind_Solver != MAIN_SOLVER::FEM_EULER && Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
63  Kind_Solver != MAIN_SOLVER::FEM_LES)
64  SU2_MPI::Error("Compressible flow equations must be selected for the NS Unit Quad case",
65  CURRENT_FUNCTION);
66 
67  if((Kind_Solver != MAIN_SOLVER::NAVIER_STOKES) &&
68  (Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES))
69  SU2_MPI::Error("Navier Stokes equations must be selected for the NS Unit Quad case",
70  CURRENT_FUNCTION);
71 
72  if(config->GetKind_FluidModel() != IDEAL_GAS)
73  SU2_MPI::Error("Ideal gas must be selected for the NS Unit Quad case",
74  CURRENT_FUNCTION);
75

103 void CNSUnitQuadSolution::GetSolution(const su2double *val_coords,
104  const su2double val_t,
105  su2double *val_solution) const {
106 
107  /*--- Compute the flow direction and the coordinates in
108  the rotated frame. ---*/
109  const su2double cosFlowAngle = cos(flowAngle);
110  const su2double sinFlowAngle = sin(flowAngle);
111 
112  const su2double xTilde = val_coords[0]*cosFlowAngle - val_coords[1]*sinFlowAngle;
113  const su2double yTilde = val_coords[0]*sinFlowAngle + val_coords[1]*cosFlowAngle;
114 
115  /*--- Compute the exact solution for this case. Note that it works
116  both in 2D and 3D. ---*/
117  val_solution[0] = 1.0;
118  val_solution[1] = cosFlowAngle*yTilde*yTilde;
119  val_solution[2] = -sinFlowAngle*yTilde*yTilde;
120  val_solution[3] = 0.0;
121  val_solution[nVar-1] = (2.0*Viscosity*xTilde + 10.0)/Gm1
122  + 0.5*yTilde*yTilde*yTilde*yTilde;
123 }
