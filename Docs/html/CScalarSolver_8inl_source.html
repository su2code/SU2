32 template <class VariableType>
33 CScalarSolver<VariableType>::CScalarSolver(CGeometry* geometry, CConfig* config, bool conservative)
34  : CSolver(), Conservative(conservative),
35  prim_idx(config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE,
36  config->GetNEMOProblem(), geometry->GetnDim(), config->GetnSpecies()) {
37  nMarker = config->GetnMarker_All();
38 
39  /*--- Store the number of vertices on each marker for deallocation later ---*/
40  nVertex.resize(nMarker);
41  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) nVertex[iMarker] = geometry->nVertex[iMarker];
42 
43  /* A grid is defined as dynamic if there's rigid grid movement or grid deformation AND the problem is time domain */
44  dynamic_grid = config->GetDynamic_Grid();
45 
46 #ifdef HAVE_OMP
47  /*--- Get the edge coloring, see notes in CEulerSolver's constructor. ---*/
48  su2double parallelEff = 1.0;
49  const auto& coloring = geometry->GetEdgeColoring(&parallelEff);
50 
51  ReducerStrategy = parallelEff < COLORING_EFF_THRESH;
52 
53  if (ReducerStrategy && (coloring.getOuterSize() > 1)) geometry->SetNaturalEdgeColoring();
54 
55  if (!coloring.empty()) {
56  auto groupSize = ReducerStrategy ? 1ul : geometry->GetEdgeColorGroupSize();
57  auto nColor = coloring.getOuterSize();
58  EdgeColoring.reserve(nColor);
59 
60  for (auto iColor = 0ul; iColor < nColor; ++iColor)
61  EdgeColoring.emplace_back(coloring.innerIdx(iColor), coloring.getNumNonZeros(iColor), groupSize);
62  }
63 
64  nPoint = geometry->GetnPoint();
65  omp_chunk_size = computeStaticChunkSize(nPoint, omp_get_max_threads(), OMP_MAX_SIZE);
66 #else
67  EdgeColoring[0] = DummyGridColor<>(geometry->GetnEdge());
68 #endif
69 
70  /*--- Initialize lower and upper limits for solution clipping. Solvers might overwrite these values. ---*/
71  for (unsigned int iVar = 0; iVar < MAXNVAR; iVar++) {
72  lowerlimit[iVar] = std::numeric_limits<su2double>::lowest();
73  upperlimit[iVar] = std::numeric_limits<su2double>::max();
74  }
75 }

82 template <class VariableType>
83 void CScalarSolver<VariableType>::CommonPreprocessing(CGeometry *geometry, const CConfig *config, const bool Output) {
84  /*--- Define booleans that are solver specific through CConfig's GlobalParams which have to be set in CFluidIteration
85  * before calling these solver functions. ---*/
86  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
87  const bool muscl = config->GetMUSCL();
88  const bool limiter = (config->GetKind_SlopeLimit() != LIMITER::NONE) &&
89  (config->GetInnerIter() <= config->GetLimiterIter());
90 
91  /*--- Clear residual and system matrix, not needed for
92  * reducer strategy as we write over the entire matrix. ---*/
93  if (!ReducerStrategy && !Output) {
94  LinSysRes.SetValZero();
95  if (implicit) {
96  Jacobian.SetValZero();
97  } else {
98  SU2_OMP_BARRIER
99  }
100  }
101 
102  /*--- Upwind second order reconstruction and gradients ---*/
103 
104  if (config->GetReconstructionGradientRequired()) {
105  switch(config->GetKind_Gradient_Method_Recon()) {
106  case GREEN_GAUSS: SetSolution_Gradient_GG(geometry, config, true); break;
107  case LEAST_SQUARES: SetSolution_Gradient_LS(geometry, config, true); break;
108  case WEIGHTED_LEAST_SQUARES: SetSolution_Gradient_LS(geometry, config, true); break;
109  }
110  }
111 
112  switch(config->GetKind_Gradient_Method()) {
113  case GREEN_GAUSS: SetSolution_Gradient_GG(geometry, config); break;
114  case WEIGHTED_LEAST_SQUARES: SetSolution_Gradient_LS(geometry, config); break;
115  }
116 
117  if (limiter && muscl) SetSolution_Limiter(geometry, config);
118 }

120 template <class VariableType>
121 void CScalarSolver<VariableType>::Upwind_Residual(CGeometry* geometry, CSolver** solver_container,
122  CNumerics** numerics_container, CConfig* config,
123  unsigned short iMesh) {
124  /*--- Define booleans that are solver specific through CConfig's GlobalParams which have to be set in CFluidIteration
125  * before calling these solver functions. ---*/
126  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
127  const bool muscl = config->GetMUSCL();
128  const bool limiter = (config->GetKind_SlopeLimit() != LIMITER::NONE) &&
129  (config->GetInnerIter() <= config->GetLimiterIter());
130 
131  /*--- Only reconstruct flow variables if MUSCL is on for flow (requires upwind) and turbulence. ---*/
132  const bool musclFlow = config->GetMUSCL_Flow() && muscl && (config->GetKind_ConvNumScheme_Flow() == SPACE_UPWIND);
133  /*--- Only consider flow limiters for cell-based limiters, edge-based would need to be recomputed. ---*/
134  const bool limiterFlow =
135  (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE) && (config->GetKind_SlopeLimit_Flow() != LIMITER::VAN_ALBADA_EDGE);
136 
137  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
138  const auto& edgeMassFluxes = *(solver_container[FLOW_SOL]->GetEdgeMassFluxes());
139 
140  /*--- Pick one numerics object per thread. ---*/
141  auto* numerics = numerics_container[CONV_TERM + omp_get_thread_num() * MAX_TERMS];
142 
143  /*--- Apply scalar advection correction terms for bounded scalar problems ---*/
144  const bool bounded_scalar = numerics->GetBoundedScalar();
145 
146  /*--- Static arrays of MUSCL-reconstructed flow primitives and turbulence variables (thread safety). ---*/
147  su2double solution_i[MAXNVAR] = {0.0}, flowPrimVar_i[MAXNVARFLOW] = {0.0};
148  su2double solution_j[MAXNVAR] = {0.0}, flowPrimVar_j[MAXNVARFLOW] = {0.0};
149 
150  /*--- For hybrid parallel AD, pause preaccumulation if there is shared reading of
151  * variables, otherwise switch to the faster adjoint evaluation mode. ---*/
152  bool pausePreacc = false;
153  if (ReducerStrategy)
154  pausePreacc = AD::PausePreaccumulation();
155  else
156  AD::StartNoSharedReading();
157 
158  /*--- Loop over edge colors. ---*/
159  for (auto color : EdgeColoring) {
160  /*--- Chunk size is at least OMP_MIN_SIZE and a multiple of the color group size. ---*/
161  SU2_OMP_FOR_DYN(nextMultiple(OMP_MIN_SIZE, color.groupSize))
162  for (auto k = 0ul; k < color.size; ++k) {
163  auto iEdge = color.indices[k];
164 
165  unsigned short iDim, iVar;
166 
167  /*--- Points in edge and normal vectors ---*/
168 
169  auto iPoint = geometry->edges->GetNode(iEdge, 0);
170  auto jPoint = geometry->edges->GetNode(iEdge, 1);
171 
172  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
173 
174  /*--- Primitive variables w/o reconstruction ---*/
175 
176  const auto V_i = flowNodes->GetPrimitive(iPoint);
177  const auto V_j = flowNodes->GetPrimitive(jPoint);
178  numerics->SetPrimitive(V_i, V_j);
179 
180  /*--- Scalar variables w/o reconstruction ---*/
181 
182  const auto Scalar_i = nodes->GetSolution(iPoint);
183  const auto Scalar_j = nodes->GetSolution(jPoint);
184  numerics->SetScalarVar(Scalar_i, Scalar_j);
185 
186  /*--- Grid Movement ---*/
187 
188  if (dynamic_grid) numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(jPoint));
189 
190  if (muscl || musclFlow) {
191  const su2double *Limiter_i = nullptr, *Limiter_j = nullptr;
192 
193  const auto Coord_i = geometry->nodes->GetCoord(iPoint);
194  const auto Coord_j = geometry->nodes->GetCoord(jPoint);
195 
196  su2double Vector_ij[MAXNDIM] = {0.0};
197  for (iDim = 0; iDim < nDim; iDim++) {
198  Vector_ij[iDim] = 0.5 * (Coord_j[iDim] - Coord_i[iDim]);
199  }
200 
201  if (musclFlow && !bounded_scalar) {
202  /*--- Reconstruct mean flow primitive variables, note that in bounded scalar mode this is
203  * not necessary because the edge mass flux is read directly from the flow solver, instead
204  * of being computed from the primitive flow variables. ---*/
205 
206  auto Gradient_i = flowNodes->GetGradient_Reconstruction(iPoint);
207  auto Gradient_j = flowNodes->GetGradient_Reconstruction(jPoint);
208 
209  if (limiterFlow) {
210  Limiter_i = flowNodes->GetLimiter_Primitive(iPoint);
211  Limiter_j = flowNodes->GetLimiter_Primitive(jPoint);
212  }
213 
214  for (iVar = 0; iVar < solver_container[FLOW_SOL]->GetnPrimVarGrad(); iVar++) {
215  su2double Project_Grad_i = 0.0;
216  su2double Project_Grad_j = 0.0;
217  for (iDim = 0; iDim < nDim; iDim++) {
218  Project_Grad_i += Vector_ij[iDim] * Gradient_i[iVar][iDim];
219  Project_Grad_j -= Vector_ij[iDim] * Gradient_j[iVar][iDim];
220  }
221  if (limiterFlow) {
222  Project_Grad_i *= Limiter_i[iVar];
223  Project_Grad_j *= Limiter_j[iVar];
224  }
225  flowPrimVar_i[iVar] = V_i[iVar] + Project_Grad_i;
226  flowPrimVar_j[iVar] = V_j[iVar] + Project_Grad_j;
227  }
228 
229  numerics->SetPrimitive(flowPrimVar_i, flowPrimVar_j);
230  }
231 
232  if (muscl) {
233  /*--- Reconstruct scalar variables. ---*/
234 
235  auto Gradient_i = nodes->GetGradient_Reconstruction(iPoint);
236  auto Gradient_j = nodes->GetGradient_Reconstruction(jPoint);
237 
238  if (limiter) {
239  Limiter_i = nodes->GetLimiter(iPoint);
240  Limiter_j = nodes->GetLimiter(jPoint);
241  }
242 
243  for (iVar = 0; iVar < nVar; iVar++) {
244  su2double Project_Grad_i = 0.0, Project_Grad_j = 0.0;
245  for (iDim = 0; iDim < nDim; iDim++) {
246  Project_Grad_i += Vector_ij[iDim] * Gradient_i[iVar][iDim];
247  Project_Grad_j -= Vector_ij[iDim] * Gradient_j[iVar][iDim];
248  }
249  if (limiter) {
250  Project_Grad_i *= Limiter_i[iVar];
251  Project_Grad_j *= Limiter_j[iVar];
252  }
253  solution_i[iVar] = Scalar_i[iVar] + Project_Grad_i;
254  solution_j[iVar] = Scalar_j[iVar] + Project_Grad_j;
255  }
256 
257  numerics->SetScalarVar(solution_i, solution_j);
258  }
259  }
260 
261  /*--- Convective flux ---*/
262  su2double EdgeMassFlux = 0.0;
263  if (bounded_scalar) {
264  EdgeMassFlux = edgeMassFluxes[iEdge];
265  numerics->SetMassFlux(EdgeMassFlux);
266  }
267 
268  /*--- Update convective residual value ---*/
269 
270  auto residual = numerics->ComputeResidual(config);
271 
272  if (ReducerStrategy) {
273  EdgeFluxes.SetBlock(iEdge, residual);
274  if (implicit) Jacobian.SetBlocks(iEdge, residual.jacobian_i, residual.jacobian_j);
275  } else {
276  LinSysRes.AddBlock(iPoint, residual);
277  LinSysRes.SubtractBlock(jPoint, residual);
278  if (implicit) Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
279  }
280 
281  /*--- Apply convective flux correction to negate the effects of flow divergence in case of incompressible flow.
282  * Note that for the bounded scalar model, we explicitly put div(v)=0.
283  * If the ReducerStrategy is used, the corrections need to be applied in a loop over nodes
284  * to avoid race conditions in accessing nodes shared by edges handled by different threads. ---*/
285 
286  if (bounded_scalar && !ReducerStrategy) {
287  LinSysRes.AddBlock(iPoint, nodes->GetSolution(iPoint), -EdgeMassFlux);
288  LinSysRes.AddBlock(jPoint, nodes->GetSolution(jPoint), EdgeMassFlux);
289 
290  if (implicit) {
291  Jacobian.AddVal2Diag(iPoint, -EdgeMassFlux);
292  Jacobian.AddVal2Diag(jPoint, EdgeMassFlux);
293  }
294  }
295 
296  /*--- Viscous contribution. ---*/
297 
298  Viscous_Residual(iEdge, geometry, solver_container,
299  numerics_container[VISC_TERM + omp_get_thread_num() * MAX_TERMS], config);
300  }
301  END_SU2_OMP_FOR
302  } // end color loop
303 
304  /*--- Restore preaccumulation and adjoint evaluation state. ---*/
305  AD::ResumePreaccumulation(pausePreacc);
306  if (!ReducerStrategy) AD::EndNoSharedReading();
307 
308  if (ReducerStrategy) {
309  SumEdgeFluxes(geometry);
310  if (implicit) Jacobian.SetDiagonalAsColumnSum();
311 
312  /*--- Bounded scalar correction that cannot be applied in the edge loop when using the ReducerStrategy. ---*/
313  if (bounded_scalar) {
314  SU2_OMP_FOR_STAT(omp_chunk_size)
315  for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint) {
316  const auto* solution = nodes->GetSolution(iPoint);
317  su2double divergence = 0;
318 
319  for (auto iEdge : geometry->nodes->GetEdges(iPoint)) {
320  const auto sign = (iPoint == geometry->edges->GetNode(iEdge,0)) ? 1 : -1;
321  const su2double EdgeMassFlux = sign * edgeMassFluxes[iEdge];
322  divergence += EdgeMassFlux;
323  LinSysRes.AddBlock(iPoint, solution, -EdgeMassFlux);
324  }
325  if (implicit) {
326  Jacobian.AddVal2Diag(iPoint, -divergence);
327  }
328  }
329  END_SU2_OMP_FOR
330  }
331  }
332 }

334 template <class VariableType>
335 void CScalarSolver<VariableType>::SumEdgeFluxes(CGeometry* geometry) {
336  SU2_OMP_FOR_STAT(omp_chunk_size)
337  for (unsigned long iPoint = 0; iPoint < nPoint; ++iPoint) {
338  LinSysRes.SetBlock_Zero(iPoint);
339 
340  for (auto iEdge : geometry->nodes->GetEdges(iPoint)) {
341  if (iPoint == geometry->edges->GetNode(iEdge, 0))
342  LinSysRes.AddBlock(iPoint, EdgeFluxes.GetBlock(iEdge));
343  else
344  LinSysRes.SubtractBlock(iPoint, EdgeFluxes.GetBlock(iEdge));
345  }
346  }
347  END_SU2_OMP_FOR
348 }

352  CConfig* config) {
353  /*--- Complete residuals for periodic boundary conditions. We loop over
354  the periodic BCs in matching pairs so that, in the event that there are

356  accumulated corectly during the communications. For implicit calculations
357  the Jacobians and linear system are also correctly adjusted here. ---*/
358 
359  for (unsigned short iPeriodic = 1; iPeriodic <= config->GetnMarker_Periodic() / 2; iPeriodic++) {
360  InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_RESIDUAL);
361  CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_RESIDUAL);
362  }

365 template <class VariableType>
366 void CScalarSolver<VariableType>::BC_Far_Field(CGeometry* geometry, CSolver** solver_container,
367  CNumerics* conv_numerics, CNumerics*, CConfig *config,
368  unsigned short val_marker) {
369 
370  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
371 
372  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
373  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
374 
375  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
376 
377  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
378 
379  if (geometry->nodes->GetDomain(iPoint)) {
380 
381  /*--- Allocate the value at the infinity ---*/
382 
383  auto V_infty = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
384 
385  /*--- Retrieve solution at the farfield boundary node ---*/
386 
387  auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
388 
389  /*--- Grid Movement ---*/
390 
391  if (dynamic_grid)
392  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint), geometry->nodes->GetGridVel(iPoint));
393 
394  conv_numerics->SetPrimitive(V_domain, V_infty);
395 
396  /*--- Set turbulent variable at the wall, and at infinity ---*/
397 
398  conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), Solution_Inf);
399 
400  /*--- Set Normal (it is necessary to change the sign) ---*/
401 
402  su2double Normal[MAXNDIM] = {0.0};
403  for (auto iDim = 0u; iDim < nDim; iDim++)
404  Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
405  conv_numerics->SetNormal(Normal);
406 
407  if (conv_numerics->GetBoundedScalar()) {
408  const su2double* velocity = &V_infty[prim_idx.Velocity()];
409  const su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
410  conv_numerics->SetMassFlux(BoundedScalarBCFlux(iPoint, implicit, density, velocity, Normal));
411  }
412 
413  /*--- Compute residuals and Jacobians ---*/
414 
415  auto residual = conv_numerics->ComputeResidual(config);
416 
417  /*--- Add residuals and Jacobians ---*/
418 
419  LinSysRes.AddBlock(iPoint, residual);
420  if (implicit) Jacobian.AddBlock2Diag(iPoint, residual.jacobian_i);
421  }
422  }
423  END_SU2_OMP_FOR
424 }

439 template <class VariableType>
440 void CScalarSolver<VariableType>::PrepareImplicitIteration(CGeometry* geometry, CSolver** solver_container,
441  CConfig* config) {
442  /*--- Set shared residual variables to 0 and declare
443  * local ones for current thread to work on. ---*/
444 
445  SetResToZero();
446 
447  su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
448  unsigned long idxMax[MAXNVAR] = {0};
449 
450  /*--- Build implicit system ---*/
451 
452  SU2_OMP_FOR_(schedule(static, omp_chunk_size) SU2_NOWAIT)
453  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
454  /*--- Modify matrix diagonal to improve diagonal dominance. ---*/
455  const su2double dt = nodes->GetDelta_Time(iPoint);
456 
457  if (dt != 0.0) {
458  su2double Vol = geometry->nodes->GetVolume(iPoint) + geometry->nodes->GetPeriodicVolume(iPoint);
459  Jacobian.AddVal2Diag(iPoint, Vol / dt);
460  } else {
461  Jacobian.SetVal2Diag(iPoint, 1.0);
462  LinSysRes.SetBlock_Zero(iPoint);
463  }
464 
465  /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
466 
467  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
468  unsigned long total_index = iPoint * nVar + iVar;
469  LinSysRes[total_index] = -LinSysRes[total_index];
470  LinSysSol[total_index] = 0.0;
471 
472  /*--- "Add" residual at (iPoint,iVar) to local residual variables. ---*/
473  ResidualReductions_PerThread(iPoint, iVar, LinSysRes[total_index], resRMS, resMax, idxMax);
474  }
475  }
476  END_SU2_OMP_FOR
477 
478  /*--- "Add" residuals from all threads to global residual variables. ---*/
479  ResidualReductions_FromAllThreads(geometry, config, resRMS, resMax, idxMax);
480 }

482 template <class VariableType>
483 void CScalarSolver<VariableType>::CompleteImplicitIteration(CGeometry* geometry, CSolver** solver_container,
484  CConfig* config) {
485  const bool compressible = (config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE);
486 
487  ComputeUnderRelaxationFactor(config);
488 
489  /*--- Update solution (system written in terms of increments) ---*/
490 
491  if (!adjoint) {
492  /*--- Update the scalar solution. For transport equations, where Solution is not equivalent with the transported
493  * quantity, multiply the respective factor. ---*/
494  if (Conservative) {
495  const auto flowNodes = solver_container[FLOW_SOL]->GetNodes();
496 
497  SU2_OMP_FOR_STAT(omp_chunk_size)
498  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
499  /*--- Multiply the Solution var with density to get the conservative transported quantity, if necessary. ---*/
500  const su2double density = flowNodes->GetDensity(iPoint);
501  const su2double density_old = compressible ? flowNodes->GetSolution_Old(iPoint, 0) : density;
502 
503  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
504  nodes->AddClippedSolution(iPoint, iVar, nodes->GetUnderRelaxation(iPoint) * LinSysSol(iPoint, iVar),
505  lowerlimit[iVar], upperlimit[iVar], density, density_old);
506  }
507  }
508  END_SU2_OMP_FOR
509  } else {
510  SU2_OMP_FOR_STAT(omp_chunk_size)
511  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
512  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
513  nodes->AddClippedSolution(iPoint, iVar, nodes->GetUnderRelaxation(iPoint) * LinSysSol(iPoint, iVar),
514  lowerlimit[iVar], upperlimit[iVar]);
515  }
516  }
517  END_SU2_OMP_FOR
518  }
519  }
520 
521  for (unsigned short iPeriodic = 1; iPeriodic <= config->GetnMarker_Periodic() / 2; iPeriodic++) {
522  InitiatePeriodicComms(geometry, config, iPeriodic, PERIODIC_IMPLICIT);
523  CompletePeriodicComms(geometry, config, iPeriodic, PERIODIC_IMPLICIT);
524  }
525 
526  InitiateComms(geometry, config, SOLUTION);
527  CompleteComms(geometry, config, SOLUTION);
528 }

530 template <class VariableType>
531 void CScalarSolver<VariableType>::ImplicitEuler_Iteration(CGeometry* geometry, CSolver** solver_container,
532  CConfig* config) {
533  PrepareImplicitIteration(geometry, solver_container, config);
534 
535  /*--- Solve or smooth the linear system. ---*/
536 
537  SU2_OMP_FOR_(schedule(static, OMP_MIN_SIZE) SU2_NOWAIT)
538  for (unsigned long iPoint = nPointDomain; iPoint < nPoint; iPoint++) {
539  LinSysRes.SetBlock_Zero(iPoint);
540  LinSysSol.SetBlock_Zero(iPoint);
541  }
542  END_SU2_OMP_FOR
543 
544  auto iter = System.Solve(Jacobian, LinSysRes, LinSysSol, geometry, config);
545 
546  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
547  SetIterLinSolver(iter);
548  SetResLinSolver(System.GetResidual());
549  }

555 template <class VariableType>
556 void CScalarSolver<VariableType>::ExplicitEuler_Iteration(CGeometry* geometry, CSolver** solver_container,
557  CConfig* config) {
558  /*--- Local residual variables for current thread ---*/
559  su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
560  unsigned long idxMax[MAXNVAR] = {0};
561 
562  SU2_OMP_FOR_STAT(omp_chunk_size)
563  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
564  const su2double dt = nodes->GetDelta_Time(iPoint);
565  const su2double Vol = geometry->nodes->GetVolume(iPoint) + geometry->nodes->GetPeriodicVolume(iPoint);
566 
567  for (auto iVar = 0u; iVar < nVar; iVar++) {
568  /*--- "Add" residual at (iPoint,iVar) to local residual variables. ---*/
569  ResidualReductions_PerThread(iPoint, iVar, LinSysRes(iPoint, iVar), resRMS, resMax, idxMax);
570  /*--- Explicit Euler step: ---*/
571  LinSysSol(iPoint, iVar) = -dt / Vol * LinSysRes(iPoint, iVar);
572  }
573  }
574  END_SU2_OMP_FOR
575 
576  /*--- "Add" residuals from all threads to global residual variables. ---*/
577  ResidualReductions_FromAllThreads(geometry, config, resRMS, resMax, idxMax);
578 
579  /*--- Use LinSysSol for solution update. ---*/
580  CompleteImplicitIteration(geometry, solver_container, config);
581 }

583 template <class VariableType>
584 void CScalarSolver<VariableType>::SetResidual_DualTime(CGeometry* geometry, CSolver** solver_container, CConfig* config,
585  unsigned short iRKStep, unsigned short iMesh,
586  unsigned short RunTime_EqSystem) {
587  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
588  const bool first_order = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST);
589  const bool second_order = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
590  const bool incompressible = (config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE);
591 
592  /*--- Flow solution, needed to get density. ---*/
593 
594  CVariable* flowNodes = solver_container[FLOW_SOL]->GetNodes();
595 
596  /*--- Store the physical time step ---*/
597 
598  const su2double TimeStep = config->GetDelta_UnstTimeND();
599 
600  /*--- Local variables ---*/
601 
602  unsigned short iVar, iMarker, iDim, iNeigh;
603  unsigned long iPoint, jPoint, iVertex, iEdge;
604 
605  su2double *U_time_nM1 = nullptr, *U_time_n = nullptr, *U_time_nP1 = nullptr;
606  /*--- For non-Conservative scalars (e.g. SA), multiply the Primitives with 1.0 instead of Density. ---*/
607  su2double Density_nM1 = 1.0, Density_n = 1.0, Density_nP1 = 1.0;
608  su2double Volume_nM1, Volume_nP1;
609  const su2double *Normal = nullptr, *GridVel_i = nullptr, *GridVel_j = nullptr;
610  su2double Residual_GCL;
611 
612  /*--- Compute the dual time-stepping source term for static meshes ---*/
613 
614  if (!dynamic_grid) {
615  /*--- Loop over all nodes (excluding halos) ---*/
616 
617  AD::StartNoSharedReading();
618 
619  SU2_OMP_FOR_STAT(omp_chunk_size)
620  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
621  if (Conservative) {
622  if (incompressible) {
623  /*--- This is temporary and only valid for constant-density problems:
624  density could also be temperature dependent, but as it is not a part
625  of the solution vector it's neither stored for previous time steps
626  nor updated with the solution at the end of each iteration. */
627  Density_nM1 = flowNodes->GetDensity(iPoint);
628  Density_n = flowNodes->GetDensity(iPoint);
629  Density_nP1 = flowNodes->GetDensity(iPoint);
630  } else {
631  Density_nM1 = flowNodes->GetSolution_time_n1(iPoint)[0];
632  Density_n = flowNodes->GetSolution_time_n(iPoint, 0);
633  Density_nP1 = flowNodes->GetSolution(iPoint, 0);
634  }
635  }
636 
637  /*--- Retrieve the solution at time levels n-1, n, and n+1. Note that
638  we are currently iterating on U^n+1 and that U^n & U^n-1 are fixed,
639  previous solutions that are stored in memory. ---*/
640 
641  U_time_nM1 = nodes->GetSolution_time_n1(iPoint);
642  U_time_n = nodes->GetSolution_time_n(iPoint);
643  U_time_nP1 = nodes->GetSolution(iPoint);
644 
645  /*--- CV volume at time n+1. As we are on a static mesh, the volume
646  of the CV will remained fixed for all time steps. ---*/
647 
648  Volume_nP1 = geometry->nodes->GetVolume(iPoint);
649 
650  /*--- Compute the dual time-stepping source term based on the chosen
651  time discretization scheme (1st- or 2nd-order).---*/
652 
653  for (iVar = 0; iVar < nVar; iVar++) {
654  if (first_order)
655  LinSysRes(iPoint, iVar) +=
656  (Density_nP1 * U_time_nP1[iVar] - Density_n * U_time_n[iVar]) * Volume_nP1 / TimeStep;
657  if (second_order)
658  LinSysRes(iPoint, iVar) += (3.0 * Density_nP1 * U_time_nP1[iVar] - 4.0 * Density_n * U_time_n[iVar] +
659  1.0 * Density_nM1 * U_time_nM1[iVar]) *
660  Volume_nP1 / (2.0 * TimeStep);
661  }
662 
663  /*--- Compute the Jacobian contribution due to the dual time source term. ---*/
664  if (implicit) {
665  if (first_order) Jacobian.AddVal2Diag(iPoint, Volume_nP1 / TimeStep);
666  if (second_order) Jacobian.AddVal2Diag(iPoint, (Volume_nP1 * 3.0) / (2.0 * TimeStep));
667  }
668  }
669  END_SU2_OMP_FOR
670 
671  AD::EndNoSharedReading();
672 
673  } else {
674  /*--- For unsteady flows on dynamic meshes (rigidly transforming or
675  dynamically deforming), the Geometric Conservation Law (GCL) should be
676  satisfied in conjunction with the ALE formulation of the governing
677  equations. The GCL prevents accuracy issues caused by grid motion, i.e.
678  a uniform free-stream should be preserved through a moving grid. First,
679  we will loop over the edges and boundaries to compute the GCL component
680  of the dual time source term that depends on grid velocities. ---*/
681 
682  SU2_OMP_FOR_STAT(omp_chunk_size)
683  for (iPoint = 0; iPoint < nPointDomain; ++iPoint) {
684  GridVel_i = geometry->nodes->GetGridVel(iPoint);
685  U_time_n = nodes->GetSolution_time_n(iPoint);
686 
687  if (Conservative) {
688  if (incompressible)
689  Density_n = flowNodes->GetDensity(iPoint); // Temporary fix
690  else
691  Density_n = flowNodes->GetSolution_time_n(iPoint, 0);
692  }
693 
694  for (iNeigh = 0; iNeigh < geometry->nodes->GetnPoint(iPoint); iNeigh++) {
695  iEdge = geometry->nodes->GetEdge(iPoint, iNeigh);
696  Normal = geometry->edges->GetNormal(iEdge);
697 
698  jPoint = geometry->nodes->GetPoint(iPoint, iNeigh);
699  GridVel_j = geometry->nodes->GetGridVel(jPoint);
700 
701  /*--- Determine whether to consider the normal outward or inward. ---*/
702  su2double dir = (iPoint < jPoint) ? 0.5 : -0.5;
703 
704  Residual_GCL = 0.0;
705  for (iDim = 0; iDim < nDim; iDim++) Residual_GCL += dir * (GridVel_i[iDim] + GridVel_j[iDim]) * Normal[iDim];
706 
707  Residual_GCL *= Density_n;
708 
709  for (iVar = 0; iVar < nVar; iVar++) LinSysRes(iPoint, iVar) += U_time_n[iVar] * Residual_GCL;
710  }
711  }
712  END_SU2_OMP_FOR
713 
714  /*--- Loop over the boundary edges ---*/
715 
716  for (iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
717  if ((config->GetMarker_All_KindBC(iMarker) != INTERNAL_BOUNDARY) &&
718  (config->GetMarker_All_KindBC(iMarker) != NEARFIELD_BOUNDARY) &&
719  (config->GetMarker_All_KindBC(iMarker) != PERIODIC_BOUNDARY)) {
720  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
721  for (iVertex = 0; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
722  /*--- Get the index for node i plus the boundary face normal ---*/
723 
724  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
725  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
726 
727  /*--- Grid velocities stored at boundary node i ---*/
728 
729  GridVel_i = geometry->nodes->GetGridVel(iPoint);
730 
731  /*--- Compute the GCL term by dotting the grid velocity with the face
732  normal. The normal is negated to match the boundary convention. ---*/
733 
734  Residual_GCL = 0.0;
735  for (iDim = 0; iDim < nDim; iDim++) Residual_GCL -= 0.5 * (GridVel_i[iDim] + GridVel_i[iDim]) * Normal[iDim];
736 
737  /*--- Compute the GCL component of the source term for node i ---*/
738 
739  U_time_n = nodes->GetSolution_time_n(iPoint);
740 
741  /*--- Multiply by density at node i for the SST model ---*/
742 
743  if (Conservative) {
744  if (incompressible)
745  Density_n = flowNodes->GetDensity(iPoint); // Temporary fix
746  else
747  Density_n = flowNodes->GetSolution_time_n(iPoint, 0);
748  }
749 
750  for (iVar = 0; iVar < nVar; iVar++) LinSysRes(iPoint, iVar) += Density_n * U_time_n[iVar] * Residual_GCL;
751  }
752  END_SU2_OMP_FOR
753  }
754  }
755 
756  /*--- Loop over all nodes (excluding halos) to compute the remainder
757  of the dual time-stepping source term. ---*/
758 
759  AD::EndNoSharedReading();
760 
761  SU2_OMP_FOR_STAT(omp_chunk_size)
762  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
763  /*--- Retrieve the solution at time levels n-1, n, and n+1. Note that
764  we are currently iterating on U^n+1 and that U^n & U^n-1 are fixed,
765  previous solutions that are stored in memory. ---*/
766 
767  U_time_nM1 = nodes->GetSolution_time_n1(iPoint);
768  U_time_n = nodes->GetSolution_time_n(iPoint);
769  U_time_nP1 = nodes->GetSolution(iPoint);
770 
771  /*--- CV volume at time n-1 and n+1. In the case of dynamically deforming
772  grids, the volumes will change. On rigidly transforming grids, the
773  volumes will remain constant. ---*/
774 
775  Volume_nM1 = geometry->nodes->GetVolume_nM1(iPoint);
776  Volume_nP1 = geometry->nodes->GetVolume(iPoint);
777 
778  /*--- Compute the dual time-stepping source residual. Due to the
779  introduction of the GCL term above, the remainder of the source residual
780  due to the time discretization has a new form.---*/
781 
782  if (Conservative) {
783  /*--- If this is the SST model, we need to multiply by the density
784  in order to get the conservative variables ---*/
785  if (incompressible) {
786  /*--- This is temporary and only valid for constant-density problems:
787  density could also be temperature dependent, but as it is not a part
788  of the solution vector it's neither stored for previous time steps
789  nor updated with the solution at the end of each iteration. */
790  Density_nM1 = flowNodes->GetDensity(iPoint);
791  Density_n = flowNodes->GetDensity(iPoint);
792  Density_nP1 = flowNodes->GetDensity(iPoint);
793  } else {
794  Density_nM1 = flowNodes->GetSolution_time_n1(iPoint)[0];
795  Density_n = flowNodes->GetSolution_time_n(iPoint, 0);
796  Density_nP1 = flowNodes->GetSolution(iPoint, 0);
797  }
798  }
799 
800  for (iVar = 0; iVar < nVar; iVar++) {
801  if (first_order)
802  LinSysRes(iPoint, iVar) +=
803  (Density_nP1 * U_time_nP1[iVar] - Density_n * U_time_n[iVar]) * (Volume_nP1 / TimeStep);
804  if (second_order)
805  LinSysRes(iPoint, iVar) +=
806  (Density_nP1 * U_time_nP1[iVar] - Density_n * U_time_n[iVar]) * (3.0 * Volume_nP1 / (2.0 * TimeStep)) +
807  (Density_nM1 * U_time_nM1[iVar] - Density_n * U_time_n[iVar]) * (Volume_nM1 / (2.0 * TimeStep));
808  }
809 
810  /*--- Compute the Jacobian contribution due to the dual time source term. ---*/
811  if (implicit) {
812  if (first_order) Jacobian.AddVal2Diag(iPoint, Volume_nP1 / TimeStep);
813  if (second_order) Jacobian.AddVal2Diag(iPoint, (Volume_nP1 * 3.0) / (2.0 * TimeStep));
814  }
815  }
816  END_SU2_OMP_FOR
817 
818  AD::EndNoSharedReading();
819 
820  } // end dynamic grid
821 }
