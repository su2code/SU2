34 void CADTBaseClass::BuildADT(unsigned short nDim,
35  unsigned long nPoints,
36  const su2double *coor) {
37 
38  /*--- Determine the number of leaves. It can be proved that nLeaves equals
39  nPoints-1 for an optimally balanced tree. Take the exceptional case of
40  nPoints == 1 into account and return if the tree is empty. ---*/
41  nDimADT = nDim;
42  isEmpty = false;
43  nLeaves = nPoints -1;
44  if(nPoints <= 1) ++nLeaves;
45  if(nLeaves == 0) {isEmpty = true; return;}
46 
47  /*--- Allocate the memory for the leaves of the ADT and the minimum and
48  maximum coordinates of the leaves. Note that these coordinates are
49  stored in one vector, rather than that memory is allocated for the
50  individual leaves. ---*/
51  leaves.resize(nLeaves);
52  coorMinLeaves.resize(nDim*nLeaves);
53  coorMaxLeaves.resize(nDim*nLeaves);
54 
55  /*--- Define the vectors, which control the subdivision of the leaves. ---*/
56  unsigned long nn = (nPoints+1)/2;
57  vector<unsigned long> pointIDs(nPoints), pointIDsNew(nPoints);
58  vector<unsigned long> nPointIDs(nn+1), nPointIDsNew(nn+1);
59  vector<unsigned long> curLeaf(nn), curLeafNew(nn);
60 
61  /*--------------------------------------------------------------------------*/
62  /*--- Building of the actual ADT ---*/
63  /*--------------------------------------------------------------------------*/
64 
65  /*--- Initialize the arrays pointIDs, nPointIDs and curLeaf such that all
66  points belong to the root leaf. Also set the counters nLeavesToDivide
67  and nLeavesTot. ---*/
68  nPointIDs[0] = 0; nPointIDs[1] = nPoints;
69  curLeaf[0] = 0;
70 
71  for(unsigned long i=0; i<nPoints; ++i) pointIDs[i] = i;
72 
73  unsigned long nLeavesToDivide = 1, nLeavesTot = 1;
74 
75  /*--- Loop to subdivide the leaves. The division is such that the ADT is
76  optimally balanced. ---*/
77  for(;;) {
78 
79  /* Criterion to exit the loop. */
80  if(nLeavesToDivide == 0) break;
81 
82  /* Initializations for the next round of subdivisions. */
83  unsigned long nLeavesToDivideNew = 0;
84  nPointIDsNew[0] = 0;
85 
86  /*--- Loop over the current number of leaves to be divided. ---*/
87  for(unsigned long i=0; i<nLeavesToDivide; ++i) {
88 
89  /* Store the number of points present in the leaf in nn and the
90  current leaf number in mm. */
91  nn = nPointIDs[i+1] - nPointIDs[i];
92  unsigned long mm = curLeaf[i];
93 
94  /*--- Set the pointers for the coordinates of the leaf to the correct
95  locations in the vectors coorMinLeaves and coorMaxLeaves and
96  determine the bounding box coordinates of this leaf. ---*/
97  leaves[mm].xMin = coorMinLeaves.data() + nDim*mm;
98  leaves[mm].xMax = coorMaxLeaves.data() + nDim*mm;
99 
100  unsigned long ll = nDim*pointIDs[nPointIDs[i]];
101  for(unsigned short l=0; l<nDim; ++l)
102  leaves[mm].xMin[l] = leaves[mm].xMax[l] = coor[ll+l];
103 
104  for(unsigned long j=(nPointIDs[i]+1); j<nPointIDs[i+1]; ++j) {
105  ll = nDim*pointIDs[j];
106  for(unsigned short l=0; l<nDim; ++l) {
107  leaves[mm].xMin[l] = min(leaves[mm].xMin[l], coor[ll+l]);
108  leaves[mm].xMax[l] = max(leaves[mm].xMax[l], coor[ll+l]);
109  }
110  }
111 
112  /*--- Determine the split direction for this leaf. The splitting is done
113  in such a way that isotropy is reached as quickly as possible.
114  Hence the split direction is the direction of the largest dimension
115  of the leaf. ---*/
116  unsigned short splitDir= 0;
117  su2double distMax = -1.0;
118  for(unsigned short l=0; l<nDim; ++l) {
119  const su2double dist = leaves[mm].xMax[l] - leaves[mm].xMin[l];
120  if(dist > distMax) {distMax = dist; splitDir = l;}
121  }
122 
123  /* Sort the points of the current leaf in increasing order. The sorting
124  is based on the coordinate in the split direction, for which the
125  functor CADTComparePointClass is used. */
126  sort(pointIDs.data() + nPointIDs[i], pointIDs.data() + nPointIDs[i+1],
127  CADTComparePointClass(coor, splitDir, nDim));
128 
129  /* Determine the index of the node, which is approximately central
130  in this leave. */
131  leaves[mm].centralNodeID = pointIDs[nPointIDs[i] + nn/2];
132 
133  /*--- Determine the situation of the leaf. It is either a terminal leaf
134  or a leaf that must be subdivided. ---*/
135  if(nn <= 2) {
136 
137  /* Terminal leaf. Store the ID's of the points as children and
138  indicate that the children are terminal. */
139  leaves[mm].children[0] = pointIDs[nPointIDs[i]];
140  leaves[mm].children[1] = pointIDs[nPointIDs[i+1]-1];
141 
142  leaves[mm].childrenAreTerminal[0] = true;
143  leaves[mm].childrenAreTerminal[1] = true;
144  }
145  else {
146 
147  /* The leaf must be divided. Determine the number of points in the
148  left leaf. This number is at least 2. The actual number stored in kk
149  is this number plus an offset. Also initialize the counter nfl, which
150  is used to store the bounding boxes in the arrays for the new round. */
151  unsigned long kk = (nn+1)/2 + nPointIDs[i];
152  unsigned long nfl = nPointIDsNew[nLeavesToDivideNew];
153 
154  /* Copy the ID's of the left points into pointIDsNew. Also update the
155  corresponding entry in nPointIDsNew. */
156  for(unsigned long k=nPointIDs[i]; k<kk; ++k)
157  pointIDsNew[nfl++] = pointIDs[k];
158 
159  nPointIDsNew[nLeavesToDivideNew+1] = nfl;
160 
161  /* Store the leaf info in the tree and in the leafs for next round and
162  update nLeavesToDivideNew and nLeavesTot. */
163  leaves[mm].children[0] = nLeavesTot;
164  leaves[mm].childrenAreTerminal[0] = false;
165 
166  curLeafNew[nLeavesToDivideNew] = nLeavesTot;
167  ++nLeavesToDivideNew;
168  ++nLeavesTot;
169 
170  /*--- The right leaf will only be created if it has more than one point
171  in it, i.e. if the original leaf has more than three points.
172  If the new leaf only has one point in it, it is not created.
173  Instead, the point is stored in the current leaf. ---*/
174  if(nn == 3)
175  {
176  /* Only three points present in the current leaf. The right leaf is
177  not created and the last point is stored as the second child of
178  the current leaf. */
179  leaves[mm].children[1] = pointIDs[nPointIDs[i+1]-1];
180  leaves[mm].childrenAreTerminal[1] = true;
181  }
182  else {
183 
184  /* More than 3 points are present and thus the right leaf is created.
185  Copy the ID's from pointIDs into pointIDsNew and update the
186  counters for the new round. */
187  unsigned long nfr = nPointIDsNew[nLeavesToDivideNew];
188 
189  for(unsigned long k=kk; k<nPointIDs[i+1]; ++k)
190  pointIDsNew[nfr++] = pointIDs[k];
191 
192  nPointIDsNew[nLeavesToDivideNew+1] = nfr;
193 
194  /* Store the leaf info in the tree and in the leaves for next round and
195  update nLeavesToDivideNew and nLeavesTot. */
196  leaves[mm].children[1] = nLeavesTot;
197  leaves[mm].childrenAreTerminal[1] = false;
198 
199  curLeafNew[nLeavesToDivideNew] = nLeavesTot;
200  ++nLeavesToDivideNew;
201  ++nLeavesTot;
202  }
203  }
204  }
205 
206  /* Set the data for the next round. */
207  nLeavesToDivide = nLeavesToDivideNew;
208 
209  for(unsigned long i=0; i<=nLeavesToDivide; ++i) nPointIDs[i] = nPointIDsNew[i];
210  for(unsigned long i=0; i< nLeavesToDivide; ++i) curLeaf[i] = curLeafNew[i];
211  for(unsigned long i=0; i<nPointIDs[nLeavesToDivide]; ++i) pointIDs[i] = pointIDsNew[i];
212  }
213 }
