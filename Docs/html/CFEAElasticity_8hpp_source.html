179  /*!
180  * \brief Compute VonMises stress from components Sxx Syy Sxy Szz Sxz Syz.
181  */
182  template<class T>
183  static su2double VonMisesStress(unsigned short nDim, const T& stress) {
184  if (nDim == 2) {
185  su2double Sxx = stress[0], Syy = stress[1], Sxy = stress[2];
186 
187  su2double S1, S2; S1 = S2 = (Sxx+Syy)/2;
188  su2double tauMax = sqrt(pow((Sxx-Syy)/2, 2) + pow(Sxy,2));
189  S1 += tauMax;
190  S2 -= tauMax;
191 
192  return sqrt(S1*S1+S2*S2-2*S1*S2);
193  }
194  else {
195  su2double Sxx = stress[0], Syy = stress[1], Szz = stress[3];
196  su2double Sxy = stress[2], Sxz = stress[4], Syz = stress[5];
197 
198  return sqrt(0.5*(pow(Sxx - Syy, 2) +
199  pow(Syy - Szz, 2) +
200  pow(Szz - Sxx, 2) +
201  6.0*(Sxy*Sxy+Sxz*Sxz+Syz*Syz)));
202  }
203  }

226  /*!
227  * \brief Update the Lame parameters (required in AD to account for all dependencies).
228  */
229  inline void Compute_Lame_Parameters(void) {
230  Mu = E / (2.0*(1.0 + Nu));
231  Lambda = Nu*E/((1.0+Nu)*(1.0-2.0*Nu));
232  Kappa = Lambda + (2/3)*Mu;
233  }
