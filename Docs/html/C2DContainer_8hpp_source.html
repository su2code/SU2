58 /*!
59  * \class AccessorImpl
60  * \brief Base accessor class and version of template for both sizes known at compile time.
61  *
62  * The actual container inherits from this class, this is to reduce the number of
63  * methods that need to be redefined with each size specialization.
64  */
65 template<typename Index_t, class Scalar_t, StorageType Store, size_t AlignSize, size_t StaticRows, size_t StaticCols>
66 class AccessorImpl
67 {
68  static_assert(!(StaticRows==1 && Store==StorageType::ColumnMajor),
69  "Row vector should have row-major storage.");
70  static_assert(!(StaticCols==1 && Store==StorageType::RowMajor),
71  "Column vector should have column-major storage.");
72 protected:
73  /*!
74  * For static specializations AlignSize will force the alignment
75  * specification of the entire class, not just the data.
76  */
77  alignas(AlignSize) Scalar_t m_data[StaticRows*StaticCols];
78 
79  /*!
80  * Static size specializations use this do-nothing allocation macro.
81  */
82 #define DUMMY_ALLOCATOR \
83  void m_allocate(size_t sz, Index_t rows, Index_t cols) noexcept {}\
84  void m_destroy() noexcept {}
85 
86  /*!
87  * Dynamic size specializations use this one, EXTRA is used to set some
88  * runtime internal value that depend on the number of rows/columns.
89  * What values need setting depends on the specialization as not all have
90  * members for e.g. number of rows and cols (static size optimization).
91  * Because aligned allocation is used, "placement new" is used after to
92  * default construct the elements of non-trivial type. Such types also
93  * need to be destructed explicitly before freeing the memory.
94  */
95 #define REAL_ALLOCATOR(EXTRA) \
96  static_assert(MemoryAllocation::is_power_of_two(AlignSize), \
97  "AlignSize is not a power of two."); \
98  \
99  void m_allocate(size_t sz, Index_t rows, Index_t cols) noexcept { \
100  EXTRA; \
101  m_data = MemoryAllocation::aligned_alloc<Scalar_t>(AlignSize,sz); \
102  if (!std::is_trivial<Scalar_t>::value) \
103  for (size_t i = 0; i < size(); ++i) new (m_data+i) Scalar_t(); \
104  } \
105  \
106  void m_destroy() noexcept { \
107  if (!std::is_trivial<Scalar_t>::value) \
108  for (size_t i = 0; i < size(); ++i) m_data[i].~Scalar_t(); \
109  MemoryAllocation::aligned_free<Scalar_t>(m_data); \
110  }
111 
112  DUMMY_ALLOCATOR
113 
114 public:
115  /*!
116  * Dynamic types need to manage internal data as the derived class would
117  * not compile if it tried to set m_data to null on static specializations.
118  * Move construct/assign are enabled by transferring ownership of data
119  * pointer, the rvalue is left in the empty state.
120  * The default ctor needs to "INIT" some fields. The move ctor/assign need
121  * to "MOVE" those fields, i.e. copy and set "other" appropriately.
122  */
123 #define CUSTOM_CTOR_AND_DTOR_BASE(INIT,MOVE) \
124  AccessorImpl() noexcept : m_data(nullptr) {INIT;} \
125  \
126  AccessorImpl(AccessorImpl&& other) noexcept \
127  { \
128  MOVE; m_data=other.m_data; other.m_data=nullptr; \
129  } \
130  \
131  AccessorImpl& operator= (AccessorImpl&& other) noexcept \
132  { \
133  m_destroy(); \
134  MOVE; m_data=other.m_data; other.m_data=nullptr; \
135  return *this; \
136  } \
137  \
138  ~AccessorImpl() noexcept {m_destroy();}
139 
140  /*!
141  * Shorthand for when specialization has only one more member than m_data.
142  */
143 #define CUSTOM_CTOR_AND_DTOR(X) \
144  CUSTOM_CTOR_AND_DTOR_BASE(X=0, X=other.X; other.X=0)
145 
146  /*!
147  * Universal accessors return a raw pointer to the data.
148  */
149 #define UNIV_ACCESSORS \
150  bool empty() const noexcept {return size()==0;} \
151  Scalar_t* data() noexcept {return m_data;} \
152  const Scalar_t* data() const noexcept {return m_data;} \
153  Scalar_t* begin() noexcept {return data();} \
154  const Scalar_t* begin() const noexcept {return data();} \
155  Scalar_t* end() noexcept {return data()+size();} \
156  const Scalar_t* end() const noexcept {return data()+size();}
157 
158  /*!
159  * Operator (,) gives pointwise access, operator [] returns a pointer to the
160  * first element of the row/column of a row/column-major matrix respectively.
161  */
162 #define MATRIX_ACCESSORS(M,N) \
163  UNIV_ACCESSORS \
164  Index_t rows() const noexcept {return M;} \
165  Index_t cols() const noexcept {return N;} \
166  size_t size() const noexcept {return M*N;} \
167  \
168  const Scalar_t& operator() (const Index_t i, \
169  const Index_t j) const noexcept \
170  { \
171  assert(i>=0 && i<M && j>=0 && j<N); \
172  return m_data[(Store==StorageType::RowMajor)? i*N+j : i+j*M]; \
173  } \
174  \
175  Scalar_t& operator() (const Index_t i, const Index_t j) noexcept \
176  { \
177  const AccessorImpl& const_this = *this; \
178  return const_cast<Scalar_t&>( const_this(i,j) ); \
179  } \
180  \
181  const Scalar_t* operator[] (const Index_t k) const noexcept \
182  { \
183  if(Store == StorageType::RowMajor) { \
184  assert(k>=0 && k<M); \
185  return &m_data[k*N]; \
186  } else { \
187  assert(k>=0 && k<N); \
188  return &m_data[k*M]; \
189  } \
190  } \
191  \
192  Scalar_t* operator[] (const Index_t k) noexcept \
193  { \
194  const AccessorImpl& const_this = *this; \
195  return const_cast<Scalar_t*>( const_this[k] ); \
196  }
197 
198  /*!
199  * Vectors provide both [] and () with the same behavior.
200  */
201 #define VECTOR_ACCESSORS(M,ROWMAJOR) \
202  UNIV_ACCESSORS \
203  Index_t rows() const noexcept {return ROWMAJOR? 1 : M;} \
204  Index_t cols() const noexcept {return ROWMAJOR? M : 1;} \
205  size_t size() const noexcept {return M;} \
206  \
207  Scalar_t& operator() (const Index_t i) noexcept \
208  { \
209  assert(i>=0 && i<M); \
210  return m_data[i]; \
211  } \
212  \
213  const Scalar_t& operator() (const Index_t i) const noexcept \
214  { \
215  assert(i>=0 && i<M); \
216  return m_data[i]; \
217  } \
218  \
219  Scalar_t& operator[] (const Index_t i) noexcept \
220  { \
221  assert(i>=0 && i<M); \
222  return m_data[i]; \
223  } \
224  \
225  const Scalar_t& operator[] (const Index_t i) const noexcept \
226  { \
227  assert(i>=0 && i<M); \
228  return m_data[i]; \
229  }

370  * \class C2DContainer
371  * \brief A templated matrix/vector-like object.
372  *
373  * See notes about MATRIX_ACCESSORS and VECTOR_ACCESSORS for how to access data.

375  *
376  * Template parameters:
377  *
378  * \param Index_t - The data type (built-in) for indices, signed and unsigned allowed.
379  * \param Scalar_t - The stored data type, anything that can be default constructed.
380  * \param Store - Mode to map 1D to 2D, row-major or column-major.

464 private:
465  /*!
466  * \brief Logic to resize data according to arguments, a non DynamicSize cannot be changed.
467  */
468  size_t m_resize(Index_t rows, Index_t cols) noexcept
469  {
470  /*--- fully static, no allocation needed ---*/
471  if(StaticSize!=DynamicSize) return StaticSize;
472 
473  /*--- dynamic row vector, swap size specification ---*/
474  if(StaticRows==1 && IsVector) {cols = rows; rows = 1;}
475 
476  /*--- assert a static size is not being asked to change ---*/
477  if(StaticRows!=DynamicSize) assert(rows==StaticRows && "A static size was asked to change.");
478  if(StaticCols!=DynamicSize) assert(cols==StaticCols && "A static size was asked to change.");
479 
480  /*--- "rectify" sizes before continuing as asserts are usually dissabled ---*/
481  rows = (StaticRows!=DynamicSize)? StaticRows : rows;
482  cols = (StaticCols!=DynamicSize)? StaticCols : cols;
483 
484  /*--- number of requested elements ---*/
485  size_t reqSize = rows*cols;
486 
487  /*--- compare with current dimensions to determine if deallocation
488  is needed, also makes the container safe against self assignment
489  no need to check for 0 size as the allocators handle that ---*/
490  if(rows==this->rows() && cols==this->cols())
491  return reqSize;
492 
493  m_destroy();
494 
495  /*--- request actual allocation to base class as it needs specialization ---*/
496  size_t bytes = reqSize*sizeof(Scalar_t);
497  m_allocate(bytes,rows,cols);
498 
499  return reqSize;
500  }

592  /*!
593  * \brief Return copy of data in a static size container.
594  * \param[in] row - Row of the matrix.
595  * \param[in] start - Starting column to copy the data (amount determined by container size).
596  */
597  template<class StaticContainer>
598  FORCEINLINE StaticContainer get(Index_t row, Index_t start = 0) const noexcept
599  {
600  constexpr size_t Size = StaticContainer::StaticSize;
601  static_assert(Size, "This method requires a static output type.");
602  assert(Size <= cols()-start);
603  StaticContainer ret;
604  SU2_OMP_SIMD
605  for (size_t i=0; i<Size; ++i)
606  ret.data()[i] = m_data[IsRowMajor? row*cols()+i+start : row+(i+start)*rows()];
607  return ret;
608  }

611  * \brief Return copy of data in a static size container, SIMD version.
612  */
613  template<class StaticContainer, class T, size_t N>
614  FORCEINLINE StaticContainer get(simd::Array<T,N> row, Index_t start = 0) const noexcept
615  {
616  constexpr size_t Size = StaticContainer::StaticSize;
617  static_assert(Size, "This method requires a static output type.");
618  assert(Size <= cols()-start);
619  StaticContainer ret;

621  SU2_OMP_SIMD_IF_NOT_AD

623  ret.data()[i][k] = m_data[IsRowMajor? row[k]*cols()+i+start : row[k]+(i+start)*rows()];
