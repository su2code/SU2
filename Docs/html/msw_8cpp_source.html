115 CNumerics::ResidualType<> CUpwMSW_NEMO::ComputeResidual(const CConfig *config) {
116 
117  /*--- Set parameters in the numerical method ---*/
118  su2double alpha = 5.0;
119  su2double epsilon = 0.0;
120 
121  /*--- Calculate supporting geometry parameters ---*/
122  Area = GeometryToolbox::Norm(nDim, Normal);
123 
124  for (unsigned short iDim = 0; iDim < nDim; iDim++)
125  UnitNormal[iDim] = Normal[iDim]/Area;
126 
127  /*--- Initialize flux & Jacobian vectors ---*/
128  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
129  Fc_i[iVar] = 0.0;
130  Fc_j[iVar] = 0.0;
131  }
132  if (implicit) {
133  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
134  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
135  Jacobian_i[iVar][jVar] = 0.0;
136  Jacobian_j[iVar][jVar] = 0.0;
137  }
138  }
139  }
140 
141  /*--- Load variables from nodes i & j ---*/
142  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
143  rhos_i[iSpecies] = V_i[RHOS_INDEX+iSpecies];
144  rhos_j[iSpecies] = V_j[RHOS_INDEX+iSpecies];
145  }
146  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
147  u_i[iDim] = V_i[VEL_INDEX+iDim];
148  u_j[iDim] = V_j[VEL_INDEX+iDim];
149  }
150  P_i = V_i[P_INDEX];
151  P_j = V_j[P_INDEX];
152 
153  /*--- Calculate velocity quantities ---*/
154  ProjVel_i = GeometryToolbox::DotProduct(nDim, u_i, UnitNormal);
155  ProjVel_j = GeometryToolbox::DotProduct(nDim, u_j, UnitNormal);
156 
157  /*--- Calculate the state weighting function ---*/
158  su2double dp = fabs(P_j-P_i) / min(P_j,P_i);
159  su2double w = 0.5 * (1.0/(pow(alpha*dp,2.0) +1.0));
160  su2double onemw = 1.0 - w;
161 
162  /*--- Calculate weighted state vector (*) for i & j ---*/
163  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
164  Ust_i[iVar] = onemw*U_i[iVar] + w*U_j[iVar];
165  Ust_j[iVar] = onemw*U_j[iVar] + w*U_i[iVar];
166  }
167  for (unsigned short iVar = 0; iVar < nPrimVar; iVar++) {
168  Vst_i[iVar] = onemw*V_i[iVar] + w*V_j[iVar];
169  Vst_j[iVar] = onemw*V_j[iVar] + w*V_i[iVar];
170  }
171  su2double ProjVelst_i = onemw*ProjVel_i + w*ProjVel_j;
172  su2double ProjVelst_j = onemw*ProjVel_j + w*ProjVel_i;
173 
174  const auto& eves_st_i = fluidmodel->ComputeSpeciesEve(Vst_i[TVE_INDEX]);
175  const auto& eves_st_j = fluidmodel->ComputeSpeciesEve(Vst_j[TVE_INDEX]);
176 
177  fluidmodel->ComputedPdU(Vst_i, eves_st_i, dPdUst_i);
178  fluidmodel->ComputedPdU(Vst_j, eves_st_j, dPdUst_j);
179 
180  /*--- Flow eigenvalues at i (Lambda+) ---*/
181  for (unsigned short iVar = 0; iVar < nSpecies+nDim-1; iVar++)
182  Lambda_i[iVar] = 0.5*(ProjVelst_i + sqrt(ProjVelst_i*ProjVelst_i +
183  epsilon*epsilon));
184  Lambda_i[nSpecies+nDim-1] = 0.5*(ProjVelst_i + Vst_i[A_INDEX] +
185  sqrt((ProjVelst_i + Vst_i[A_INDEX])*
186  (ProjVelst_i + Vst_i[A_INDEX])+
187  epsilon*epsilon));
188  Lambda_i[nSpecies+nDim] = 0.5*(ProjVelst_i - Vst_i[A_INDEX] +
189  sqrt((ProjVelst_i - Vst_i[A_INDEX])*
190  (ProjVelst_i - Vst_i[A_INDEX])+
191  epsilon*epsilon));
192  Lambda_i[nSpecies+nDim+1] = 0.5*(ProjVelst_i + sqrt(ProjVelst_i*ProjVelst_i +
193  epsilon*epsilon));
194 
195  /*--- Compute projected P, invP, and Lambda ---*/
196  su2double l[MAXNDIM], m[MAXNDIM];
197  CreateBasis(UnitNormal,l,m);
198  GetPMatrix(Ust_i, Vst_i, dPdUst_i, UnitNormal, l, m, P_Tensor);
199  GetPMatrix_inv(Ust_i, Vst_i, dPdUst_i, UnitNormal, l, m, invP_Tensor);
200 
201  /*--- Projected flux (f+) at i ---*/
202  su2double Proj_ModJac_Tensor_i;
203  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
204  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
205  Proj_ModJac_Tensor_i = 0.0;
206 
207  /*--- Compute Proj_ModJac_Tensor = P x Lambda+ x inverse P ---*/
208  for (unsigned short kVar = 0; kVar < nVar; kVar++)
209  Proj_ModJac_Tensor_i += P_Tensor[iVar][kVar]*Lambda_i[kVar]*invP_Tensor[kVar][jVar];
210  Fc_i[iVar] += Proj_ModJac_Tensor_i*U_i[jVar]*Area;
211  if (implicit)
212  Jacobian_i[iVar][jVar] += Proj_ModJac_Tensor_i*Area;
213  }
214  }
215 
216  /*--- Flow eigenvalues at j (Lambda-) ---*/
217  for (unsigned short iVar = 0; iVar < nSpecies+nDim-1; iVar++)
218  Lambda_j[iVar] = 0.5*(ProjVelst_j - sqrt(ProjVelst_j*ProjVelst_j +
219  epsilon*epsilon));
220  Lambda_j[nSpecies+nDim-1] = 0.5*(ProjVelst_j + Vst_j[A_INDEX] -
221  sqrt((ProjVelst_j + Vst_j[A_INDEX])*
222  (ProjVelst_j + Vst_j[A_INDEX])+
223  epsilon*epsilon));
224  Lambda_j[nSpecies+nDim] = 0.5*(ProjVelst_j - Vst_j[A_INDEX] -
225  sqrt((ProjVelst_j - Vst_j[A_INDEX])*
226  (ProjVelst_j - Vst_j[A_INDEX])+
227  epsilon*epsilon) );
228  Lambda_j[nSpecies+nDim+1] = 0.5*(ProjVelst_j - sqrt(ProjVelst_j*ProjVelst_j+
229  epsilon*epsilon));
230 
231  /*--- Compute projected P, invP, and Lambda ---*/
232  CreateBasis(UnitNormal,l,m);
233  GetPMatrix(Ust_j, Vst_j, dPdUst_j, UnitNormal, l, m, P_Tensor);
234  GetPMatrix_inv(Ust_j, Vst_j, dPdUst_j, UnitNormal, l, m, invP_Tensor);
235 
236  /*--- Projected flux (f-) ---*/
237  su2double Proj_ModJac_Tensor_j;
238  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
239  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
240  Proj_ModJac_Tensor_j = 0.0;
241 
242  /*--- Compute Proj_ModJac_Tensor = P x Lambda- x inverse P ---*/
243  for (unsigned short kVar = 0; kVar < nVar; kVar++)
244  Proj_ModJac_Tensor_j += P_Tensor[iVar][kVar]*Lambda_j[kVar]*invP_Tensor[kVar][jVar];
245  Fc_j[iVar] += Proj_ModJac_Tensor_j*U_j[jVar]*Area;
246  if (implicit)
247  Jacobian_j[iVar][jVar] += Proj_ModJac_Tensor_j*Area;
248  }
249  }
250 
251  /*--- Flux splitting ---*/
252  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
253  Flux[iVar] = Fc_i[iVar]+Fc_j[iVar];
254  }
255 
256  return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
257 }
