31 void CFluidIteration::Preprocess(COutput* output, CIntegration**** integration, CGeometry**** geometry,
32  CSolver***** solver, CNumerics****** numerics, CConfig** config,
33  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
34  CFreeFormDefBox*** FFDBox, unsigned short val_iZone, unsigned short val_iInst) {
35  unsigned long TimeIter = config[val_iZone]->GetTimeIter();
36 
37  bool fsi = config[val_iZone]->GetFSI_Simulation();
38  unsigned long OuterIter = config[val_iZone]->GetOuterIter();
39 
40  /*--- Set the initial condition for FSI problems with subiterations ---*/
41  /*--- This is done only in the first block subiteration.---*/
42  /*--- From then on, the solver reuses the partially converged solution obtained in the previous subiteration ---*/
43  if (fsi && !config[val_iZone]->GetDiscrete_Adjoint() && (OuterIter == 0)) {
44  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->SetInitialCondition(
45  geometry[val_iZone][val_iInst], solver[val_iZone][val_iInst], config[val_iZone], TimeIter);
46  }
47 
48  /*--- Apply a Wind Gust ---*/
49 
50  if (config[val_iZone]->GetWind_Gust()) {
51  SetWind_GustField(config[val_iZone], geometry[val_iZone][val_iInst], solver[val_iZone][val_iInst]);
52  }

55 void CFluidIteration::Iterate(COutput* output, CIntegration**** integration, CGeometry**** geometry,
56  CSolver***** solver, CNumerics****** numerics, CConfig** config,
57  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
58  CFreeFormDefBox*** FFDBox, unsigned short val_iZone, unsigned short val_iInst) {
59 
60  const bool unsteady = (config[val_iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
61  (config[val_iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
62  const bool frozen_visc = (config[val_iZone]->GetContinuous_Adjoint() && config[val_iZone]->GetFrozen_Visc_Cont()) ||
63  (config[val_iZone]->GetDiscrete_Adjoint() && config[val_iZone]->GetFrozen_Visc_Disc());
64  const bool disc_adj = (config[val_iZone]->GetDiscrete_Adjoint());
65 
66  /*--- Setting up iteration values depending on if this is a
67  steady or an unsteady simulation */
68 
69  const auto InnerIter = config[val_iZone]->GetInnerIter();
70  const auto TimeIter = config[val_iZone]->GetTimeIter();
71 
72  /*--- Update global parameters ---*/
73 
74  const auto main_solver = config[val_iZone]->GetKind_Solver();
75  config[val_iZone]->SetGlobalParam(main_solver, RUNTIME_FLOW_SYS);
76 
77  /*--- Solve the Euler, Navier-Stokes or Reynolds-averaged Navier-Stokes (RANS) equations (one iteration) ---*/
78 
79  integration[val_iZone][val_iInst][FLOW_SOL]->MultiGrid_Iteration(geometry, solver, numerics, config, RUNTIME_FLOW_SYS,
80  val_iZone, val_iInst);
81 
82  /*--- If the flow integration is not fully coupled, run the various single grid integrations. ---*/
83 
84  if (config[val_iZone]->GetKind_Turb_Model() != TURB_MODEL::NONE && !frozen_visc) {
85 
86  /*--- Solve transition model ---*/
87 
88  if (config[val_iZone]->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
89  config[val_iZone]->SetGlobalParam(main_solver, RUNTIME_TRANS_SYS);
90  integration[val_iZone][val_iInst][TRANS_SOL]->SingleGrid_Iteration(geometry, solver, numerics, config,
91  RUNTIME_TRANS_SYS, val_iZone, val_iInst);
92  }
93 
94  /*--- Solve the turbulence model ---*/
95 
96  config[val_iZone]->SetGlobalParam(main_solver, RUNTIME_TURB_SYS);
97  integration[val_iZone][val_iInst][TURB_SOL]->SingleGrid_Iteration(geometry, solver, numerics, config,
98  RUNTIME_TURB_SYS, val_iZone, val_iInst);
99  }
100 
101  if (config[val_iZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE){
102  config[val_iZone]->SetGlobalParam(main_solver, RUNTIME_SPECIES_SYS);
103  integration[val_iZone][val_iInst][SPECIES_SOL]->SingleGrid_Iteration(geometry, solver, numerics, config,
104  RUNTIME_SPECIES_SYS, val_iZone, val_iInst);
105 
106  // This only applies if mixture properties are used. But this also doesn't hurt if done w/out mixture properties.
107  // In case of turbulence, the Turb-Post computes the correct eddy viscosity based on mixture-density and
108  // mixture lam-visc. In order to get the correct mixture properties, based on the just updated mass-fractions, the
109  // Flow-Pre has to be called upfront. The updated eddy-visc are copied into the flow-solver Primitive in another
110  // Flow-Pre call which is done at the start of the next iteration.
111  if (config[val_iZone]->GetKind_Turb_Model() != TURB_MODEL::NONE) {
112  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->Preprocessing(geometry[val_iZone][val_iInst][MESH_0], solver[val_iZone][val_iInst][MESH_0], config[val_iZone], MESH_0, NO_RK_ITER, RUNTIME_FLOW_SYS, true);
113  solver[val_iZone][val_iInst][MESH_0][TURB_SOL]->Postprocessing(geometry[val_iZone][val_iInst][MESH_0], solver[val_iZone][val_iInst][MESH_0], config[val_iZone], MESH_0);
114  }
115  }
116 
117  if (config[val_iZone]->GetWeakly_Coupled_Heat()) {
118  config[val_iZone]->SetGlobalParam(main_solver, RUNTIME_HEAT_SYS);
119  integration[val_iZone][val_iInst][HEAT_SOL]->SingleGrid_Iteration(geometry, solver, numerics, config,
120  RUNTIME_HEAT_SYS, val_iZone, val_iInst);
121  }
122 
123  /*--- Incorporate a weakly-coupled radiation model to the analysis ---*/
124  if (config[val_iZone]->AddRadiation()) {
125  config[val_iZone]->SetGlobalParam(main_solver, RUNTIME_RADIATION_SYS);
126  integration[val_iZone][val_iInst][RAD_SOL]->SingleGrid_Iteration(geometry, solver, numerics, config,
127  RUNTIME_RADIATION_SYS, val_iZone, val_iInst);
128  }
129 
130  /*--- Adapt the CFL number using an exponential progression with under-relaxation approach. ---*/
131 
132  if ((config[val_iZone]->GetCFL_Adapt() == YES) && (!disc_adj)) {
133  SU2_OMP_PARALLEL
134  solver[val_iZone][val_iInst][MESH_0][FLOW_SOL]->AdaptCFLNumber(geometry[val_iZone][val_iInst],
135  solver[val_iZone][val_iInst], config[val_iZone]);
136  END_SU2_OMP_PARALLEL
137  }
138 
139  /*--- Call Dynamic mesh update if AEROELASTIC motion was specified ---*/
140 
141  if ((config[val_iZone]->GetGrid_Movement()) && (config[val_iZone]->GetAeroelastic_Simulation()) && unsteady) {
142  SetGrid_Movement(geometry[val_iZone][val_iInst], surface_movement[val_iZone], grid_movement[val_iZone][val_iInst],
143  solver[val_iZone][val_iInst], config[val_iZone], InnerIter, TimeIter);
144 
145  /*--- Apply a Wind Gust ---*/
146 
147  if (config[val_iZone]->GetWind_Gust()) {
148  if (InnerIter % config[val_iZone]->GetAeroelasticIter() == 0 && InnerIter != 0)
149  SetWind_GustField(config[val_iZone], geometry[val_iZone][val_iInst], solver[val_iZone][val_iInst]);
150  }
151  }
152 }

154 void CFluidIteration::Update(COutput* output, CIntegration**** integration, CGeometry**** geometry, CSolver***** solver,
155  CNumerics****** numerics, CConfig** config, CSurfaceMovement** surface_movement,
156  CVolumetricMovement*** grid_movement, CFreeFormDefBox*** FFDBox, unsigned short val_iZone,
157  unsigned short val_iInst) {
158  unsigned short iMesh;
159 
160  /*--- Dual time stepping strategy ---*/
161 
162  if ((config[val_iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
163  (config[val_iZone]->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND)) {
164  /*--- Update dual time solver on all mesh levels ---*/
165 
166  for (iMesh = 0; iMesh <= config[val_iZone]->GetnMGLevels(); iMesh++) {
167  integration[val_iZone][val_iInst][FLOW_SOL]->SetDualTime_Solver(geometry[val_iZone][val_iInst][iMesh],
168  solver[val_iZone][val_iInst][iMesh][FLOW_SOL],
169  config[val_iZone], iMesh);
170 
171  integration[val_iZone][val_iInst][FLOW_SOL]->SetDualTime_Geometry(geometry[val_iZone][val_iInst][iMesh],
172  solver[val_iZone][val_iInst][iMesh][MESH_SOL],
173  config[val_iZone], iMesh);
174 
175  integration[val_iZone][val_iInst][FLOW_SOL]->SetConvergence(false);
176  }
177 
178  SetDualTime_Aeroelastic(config[val_iZone]);
179 
180  /*--- Update dual time solver for the turbulence model ---*/
181 
182  if ((config[val_iZone]->GetKind_Solver() == MAIN_SOLVER::RANS) || (config[val_iZone]->GetKind_Solver() == MAIN_SOLVER::DISC_ADJ_RANS) ||
183  (config[val_iZone]->GetKind_Solver() == MAIN_SOLVER::INC_RANS) ||
184  (config[val_iZone]->GetKind_Solver() == MAIN_SOLVER::DISC_ADJ_INC_RANS)) {
185  integration[val_iZone][val_iInst][TURB_SOL]->SetDualTime_Solver(geometry[val_iZone][val_iInst][MESH_0],
186  solver[val_iZone][val_iInst][MESH_0][TURB_SOL],
187  config[val_iZone], MESH_0);
188  integration[val_iZone][val_iInst][TURB_SOL]->SetConvergence(false);
189  }
190 
191  /*--- Update dual time solver for the transition model ---*/
192 
193  if (config[val_iZone]->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
194  integration[val_iZone][val_iInst][TRANS_SOL]->SetDualTime_Solver(geometry[val_iZone][val_iInst][MESH_0],
195  solver[val_iZone][val_iInst][MESH_0][TRANS_SOL],
196  config[val_iZone], MESH_0);
197  integration[val_iZone][val_iInst][TRANS_SOL]->SetConvergence(false);
198  }
199 
200  /*--- Update dual time solver for the weakly coupled energy equation ---*/
201 
202  if (config[val_iZone]->GetWeakly_Coupled_Heat()) {
203  integration[val_iZone][val_iInst][HEAT_SOL]->SetDualTime_Solver(geometry[val_iZone][val_iInst][MESH_0],
204  solver[val_iZone][val_iInst][MESH_0][HEAT_SOL],
205  config[val_iZone], MESH_0);
206  integration[val_iZone][val_iInst][HEAT_SOL]->SetConvergence(false);
207  }
208  }
209 }

211 bool CFluidIteration::Monitor(COutput* output, CIntegration**** integration, CGeometry**** geometry,
212  CSolver***** solver, CNumerics****** numerics, CConfig** config,
213  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
214  CFreeFormDefBox*** FFDBox, unsigned short val_iZone, unsigned short val_iInst) {
215  bool StopCalc = false;
216 
217  StopTime = SU2_MPI::Wtime();
218 
219  UsedTime = StopTime - StartTime;
220 
221  if (config[val_iZone]->GetMultizone_Problem() || config[val_iZone]->GetSinglezone_Driver()) {
222  output->SetHistory_Output(geometry[val_iZone][INST_0][MESH_0], solver[val_iZone][INST_0][MESH_0], config[val_iZone],
223  config[val_iZone]->GetTimeIter(), config[val_iZone]->GetOuterIter(),
224  config[val_iZone]->GetInnerIter());
225  }
226 
227  /*--- If convergence was reached --*/
228  StopCalc = output->GetConvergence();
229 
230  /* --- Checking convergence of Fixed CL mode to target CL, and perform finite differencing if needed --*/
231 
232  if (config[val_iZone]->GetFixed_CL_Mode()) {
233  StopCalc = MonitorFixed_CL(output, geometry[val_iZone][INST_0][MESH_0], solver[val_iZone][INST_0][MESH_0],
234  config[val_iZone]);
235  }
236 
237  return StopCalc;
238 }

254 void CFluidIteration::Solve(COutput* output, CIntegration**** integration, CGeometry**** geometry, CSolver***** solver,
255  CNumerics****** numerics, CConfig** config, CSurfaceMovement** surface_movement,
256  CVolumetricMovement*** grid_movement, CFreeFormDefBox*** FFDBox, unsigned short val_iZone,
257  unsigned short val_iInst) {
258  /*--- Boolean to determine if we are running a static or dynamic case ---*/
259  bool steady = !config[val_iZone]->GetTime_Domain();
260 
261  unsigned long Inner_Iter, nInner_Iter = config[val_iZone]->GetnInner_Iter();
262  bool StopCalc = false;
263 
264  /*--- Synchronization point before a single solver iteration.
265  Compute the wall clock time required. ---*/
266 
267  StartTime = SU2_MPI::Wtime();
268 
269  /*--- Preprocess the solver ---*/
270  Preprocess(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement, FFDBox,
271  val_iZone, INST_0);
272 
273  /*--- For steady-state flow simulations, we need to loop over ExtIter for the number of time steps ---*/
274  /*--- However, ExtIter is the number of FSI iterations, so nIntIter is used in this case ---*/
275 
276  for (Inner_Iter = 0; Inner_Iter < nInner_Iter; Inner_Iter++) {
277  config[val_iZone]->SetInnerIter(Inner_Iter);
278 
279  /*--- Run a single iteration of the solver ---*/
280  Iterate(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement, FFDBox, val_iZone,
281  INST_0);
282 
283  /*--- Monitor the pseudo-time ---*/
284  StopCalc = Monitor(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement, FFDBox,
285  val_iZone, INST_0);
286 
287  /*--- Output files at intermediate iterations if the problem is single zone ---*/
288 
289  if (singlezone && steady) {
290  Output(output, geometry, solver, config, Inner_Iter, StopCalc, val_iZone, val_iInst);
291  }
292 
293  /*--- If the iteration has converged, break the loop ---*/
294  if (StopCalc) break;
295  }
296 
297  if (multizone && steady) {
298  Output(output, geometry, solver, config, config[val_iZone]->GetOuterIter(), StopCalc, val_iZone, val_iInst);
299 
300  /*--- Set the convergence to false (to make sure outer subiterations converge) ---*/
301 
302  if (config[val_iZone]->GetKind_Solver() == MAIN_SOLVER::HEAT_EQUATION) {
303  integration[val_iZone][INST_0][HEAT_SOL]->SetConvergence(false);
304  }
305  else {
306  integration[val_iZone][INST_0][FLOW_SOL]->SetConvergence(false);
307  }
308  }
309 }

311 void CFluidIteration::SetWind_GustField(CConfig* config, CGeometry** geometry, CSolver*** solver) {
312  // The gust is imposed on the flow field via the grid velocities. This method called the Field Velocity Method is
313  // described in the NASA TM2012-217771 - Development, Verification and Use of Gust Modeling in the NASA Computational
314  // Fluid Dynamics Code FUN3D the desired gust is prescribed as the negative of the grid velocity.
315 
316  // If a source term is included to account for the gust field, the method is described by Jones et al. as the Split
317  // Velocity Method in Simulation of Airfoil Gust Responses Using Prescribed Velocities. In this routine the gust
318  // derivatives needed for the source term are calculated when applicable. If the gust derivatives are zero the source
319  // term is also zero. The source term itself is implemented in the class CSourceWindGust
320 
321  if (rank == MASTER_NODE) cout << endl << "Running simulation with a Wind Gust." << endl;
322  unsigned short iDim, nDim = geometry[MESH_0]->GetnDim(); // We assume nDim = 2
323  if (nDim != 2) {
324  if (rank == MASTER_NODE) {
325  cout << endl << "WARNING - Wind Gust capability is only verified for 2 dimensional simulations." << endl;
326  }
327  }
328 
329  /*--- Gust Parameters from config ---*/
330  unsigned short Gust_Type = config->GetGust_Type();
331  su2double xbegin = config->GetGust_Begin_Loc(); // Location at which the gust begins.
332  su2double L = config->GetGust_WaveLength(); // Gust size
333  su2double tbegin = config->GetGust_Begin_Time(); // Physical time at which the gust begins.
334  su2double gust_amp = config->GetGust_Ampl(); // Gust amplitude
335  su2double n = config->GetGust_Periods(); // Number of gust periods
336  unsigned short GustDir = config->GetGust_Dir(); // Gust direction
337 
338  /*--- Variables needed to compute the gust ---*/
339  unsigned short Kind_Grid_Movement = config->GetKind_GridMovement();
340  unsigned long iPoint;
341  unsigned short iMGlevel, nMGlevel = config->GetnMGLevels();
342 
343  su2double x, y, x_gust, dgust_dx, dgust_dy, dgust_dt;
344  su2double *Gust, *GridVel, *NewGridVel, *GustDer;
345 
346  su2double Physical_dt = config->GetDelta_UnstTime();
347  unsigned long TimeIter = config->GetTimeIter();
348  if (config->GetDiscrete_Adjoint()) TimeIter = config->GetUnst_AdjointIter() - TimeIter - 1;
349 
350  su2double Physical_t = TimeIter * Physical_dt;
351 
352  su2double Uinf = solver[MESH_0][FLOW_SOL]->GetVelocity_Inf(0); // Assumption gust moves at infinity velocity
353 
354  Gust = new su2double[nDim];
355  NewGridVel = new su2double[nDim];
356  for (iDim = 0; iDim < nDim; iDim++) {
357  Gust[iDim] = 0.0;
358  NewGridVel[iDim] = 0.0;
359  }
360 
361  GustDer = new su2double[3];
362  for (unsigned short i = 0; i < 3; i++) {
363  GustDer[i] = 0.0;
364  }
365 
366  // Vortex variables
367  unsigned long nVortex = 0;
368  vector<su2double> x0, y0, vort_strenth, r_core; // vortex is positive in clockwise direction.
369  if (Gust_Type == VORTEX) {
370  InitializeVortexDistribution(nVortex, x0, y0, vort_strenth, r_core);
371  }
372 
373  /*--- Check to make sure gust lenght is not zero or negative (vortex gust doesn't use this). ---*/
374  if (L <= 0.0 && Gust_Type != VORTEX) {
375  SU2_MPI::Error("The gust length needs to be positive", CURRENT_FUNCTION);
376  }
377 
378  /*--- Loop over all multigrid levels ---*/
379 
380  for (iMGlevel = 0; iMGlevel <= nMGlevel; iMGlevel++) {
381  /*--- Loop over each node in the volume mesh ---*/
382 
383  for (iPoint = 0; iPoint < geometry[iMGlevel]->GetnPoint(); iPoint++) {
384  /*--- Reset the Grid Velocity to zero if there is no grid movement ---*/
385  if (Kind_Grid_Movement == GUST && !(config->GetFSI_Simulation())) {
386  for (iDim = 0; iDim < nDim; iDim++) geometry[iMGlevel]->nodes->SetGridVel(iPoint, iDim, 0.0);
387  }
388 
389  /*--- initialize the gust and derivatives to zero everywhere ---*/
390 
391  for (iDim = 0; iDim < nDim; iDim++) {
392  Gust[iDim] = 0.0;
393  }
394  dgust_dx = 0.0;
395  dgust_dy = 0.0;
396  dgust_dt = 0.0;
397 
398  /*--- Begin applying the gust ---*/
399 
400  if (Physical_t >= tbegin) {
401  x = geometry[iMGlevel]->nodes->GetCoord(iPoint)[0]; // x-location of the node.
402  y = geometry[iMGlevel]->nodes->GetCoord(iPoint)[1]; // y-location of the node.
403 
404  // Gust coordinate
405  x_gust = (x - xbegin - Uinf * (Physical_t - tbegin)) / L;
406 
407  /*--- Calculate the specified gust ---*/
408  switch (Gust_Type) {
409  case TOP_HAT:
410  // Check if we are in the region where the gust is active
411  if (x_gust > 0 && x_gust < n) {
412  Gust[GustDir] = gust_amp;
413  // Still need to put the gust derivatives. Think about this.
414  }
415  break;
416 
417  case SINE:
418  // Check if we are in the region where the gust is active
419  if (x_gust > 0 && x_gust < n) {
420  Gust[GustDir] = gust_amp * (sin(2 * PI_NUMBER * x_gust));
421 
422  // Gust derivatives
423  // dgust_dx = gust_amp*2*PI_NUMBER*(cos(2*PI_NUMBER*x_gust))/L;
424  // dgust_dy = 0;
425  // dgust_dt = gust_amp*2*PI_NUMBER*(cos(2*PI_NUMBER*x_gust))*(-Uinf)/L;
426  }
427  break;
428 
429  case ONE_M_COSINE:
430  // Check if we are in the region where the gust is active
431  if (x_gust > 0 && x_gust < n) {
432  Gust[GustDir] = gust_amp * (1 - cos(2 * PI_NUMBER * x_gust));
433 
434  // Gust derivatives
435  // dgust_dx = gust_amp*2*PI_NUMBER*(sin(2*PI_NUMBER*x_gust))/L;
436  // dgust_dy = 0;
437  // dgust_dt = gust_amp*2*PI_NUMBER*(sin(2*PI_NUMBER*x_gust))*(-Uinf)/L;
438  }
439  break;
440 
441  case EOG:
442  // Check if we are in the region where the gust is active
443  if (x_gust > 0 && x_gust < n) {
444  Gust[GustDir] = -0.37 * gust_amp * sin(3 * PI_NUMBER * x_gust) * (1 - cos(2 * PI_NUMBER * x_gust));
445  }
446  break;
447 
448  case VORTEX:
449 
450  /*--- Use vortex distribution ---*/
451  // Algebraic vortex equation.
452  for (unsigned long i = 0; i < nVortex; i++) {
453  su2double r2 = pow(x - (x0[i] + Uinf * (Physical_t - tbegin)), 2) + pow(y - y0[i], 2);
454  su2double r = sqrt(r2);
455  su2double v_theta = vort_strenth[i] / (2 * PI_NUMBER) * r / (r2 + pow(r_core[i], 2));
456  Gust[0] = Gust[0] + v_theta * (y - y0[i]) / r;
457  Gust[1] = Gust[1] - v_theta * (x - (x0[i] + Uinf * (Physical_t - tbegin))) / r;
458  }
459  break;
460 
461  case NONE:
462  default:
463 
464  /*--- There is no wind gust specified. ---*/
465  if (rank == MASTER_NODE) {
466  cout << "No wind gust specified." << endl;
467  }
468  break;
469  }
470  }
471 
472  /*--- Set the Wind Gust, Wind Gust Derivatives and the Grid Velocities ---*/
473 
474  GustDer[0] = dgust_dx;
475  GustDer[1] = dgust_dy;
476  GustDer[2] = dgust_dt;
477 
478  solver[iMGlevel][FLOW_SOL]->GetNodes()->SetWindGust(iPoint, Gust);
479  solver[iMGlevel][FLOW_SOL]->GetNodes()->SetWindGustDer(iPoint, GustDer);
480

612  break;
613  }
