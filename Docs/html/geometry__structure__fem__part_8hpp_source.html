35 /*!
36  * \struct CUnsignedLong2T
37  * \brief Helper struct used to store two integral types as one entity.
38  */
39 struct CUnsignedLong2T {
40 
41  unsigned long long0; /*!< \brief First integer to store in this class. */
42  unsigned long long1; /*!< \brief Second integer to store in this class. */
43 
44  CUnsignedLong2T(unsigned long a = 0, unsigned long b = 0) : long0(a), long1(b) {}
45 
46  inline bool operator<(const CUnsignedLong2T &other) const {
47  if(long0 != other.long0)
48  return (long0 < other.long0);
49  return (long1 < other.long1);
50  }
51 
52  inline bool operator==(const CUnsignedLong2T &other) const {
53  return (long0 == other.long0) && (long1 == other.long1);
54  }
55 };

79 /*!
80  * \class CFaceOfElement
81  * \brief Class used in the partitioning of the FEM grid as well as the building of
82  the faces of DG. It stores a face of an element.
83  */
84 class CFaceOfElement {
85 public:
86  unsigned short nCornerPoints; /*!< \brief Number of corner points of the face. */
87  unsigned long cornerPoints[4]; /*!< \brief Global ID's of ther corner points. */
88  unsigned long elemID0, elemID1; /*!< \brief Element ID's to the left and right. */
89  unsigned short nPolyGrid0, nPolyGrid1; /*!< \brief Polynomial degrees of the grid of the elements
90  to the left and right. */
91  unsigned short nPolySol0, nPolySol1; /*!< \brief Polynomial degrees of the solution of the elements
92  to the left and right. */
93  unsigned short nDOFsElem0, nDOFsElem1; /*!< \brief Number of DOFs of the elements to the left and right. */
94  unsigned short elemType0, elemType1; /*!< \brief Type of the elements to the left and right. */
95  unsigned short faceID0, faceID1; /*!< \brief The local face ID in the corresponding elements
96  to the left and right of the face. */
97  unsigned short periodicIndex; /*!< \brief Periodic indicator of the face. A value of 0 means no
98  periodic face. A value larger than 0 gives the index of
99  the periodic boundary + 1. */
100  unsigned short periodicIndexDonor; /*!< \brief Periodic indicator of the donor face. A value of 0 means no
101  periodic donor face. A value larger than 0 gives the index of
102  the periodic donor boundary + 1. */
103  short faceIndicator; /*!< \brief The corresponding boundary marker if this face is on a
104  boundary. For an internal face the value is -1,
105  while an invalidated face has the value -2. */
106  bool JacFaceIsConsideredConstant; /*!< \brief Whether or not the Jacobian of the transformation
107  to the standard element is considered constant. */
108  bool elem0IsOwner; /*!< \brief Whether or not the neighboring element 0 is the owner
109  of the face. If false, element 1 is the owner. */
110 
111  /* Standard constructor and destructor. */
112  CFaceOfElement();
113  ~CFaceOfElement(){}
114 
115  /* Alternative constructor to set the corner points. */
116  CFaceOfElement(const unsigned short VTK_Type,
117  const unsigned short nPoly,
118  const unsigned long *Nodes);
119 
120  /* Copy constructor and assignment operator. */
121  inline CFaceOfElement(const CFaceOfElement &other) { Copy(other); }
122 
123  inline CFaceOfElement& operator=(const CFaceOfElement &other) { Copy(other); return (*this); }
124 
125  /* Less than operator. Needed for the sorting and searching. */
126  bool operator<(const CFaceOfElement &other) const;
127 
128  /* Equal operator. Needed for removing double entities. */
129  bool operator ==(const CFaceOfElement &other) const;
130 
131  /*--- Member function, which creates a unique numbering for the corner points.
132  A sort in increasing order is OK for this purpose. ---*/
133  inline void CreateUniqueNumbering(void) { std::sort(cornerPoints, cornerPoints+nCornerPoints); }
134 
135  /*--- Member function, which creates a unique numbering for the corner points
136  while the orientation is taken into account. ---*/
137  void CreateUniqueNumberingWithOrientation(void);
138 
139 private:
140  /*--- Copy function, which copies the data of the given object into the current object. ---*/
141  void Copy(const CFaceOfElement &other);
142 };
