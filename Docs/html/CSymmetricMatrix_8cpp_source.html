52 void CSymmetricMatrix::CholeskyDecompose()
53 {
54 #ifndef HAVE_LAPACK
55  int j;
56  for (j = 0; j < Size(); ++j) {
57  passivedouble sum = 0.0;
58  for (int k = 0; k < j; ++k) sum -= pow(Get(j,k), 2);
59  sum += Get(j,j);
60  if (sum < 0.0) break; // not SPD
61  Set(j, j, sqrt(sum));
62 
63  for (int i = j+1; i < Size(); ++i) {
64  passivedouble sum = 0.0;
65  for (int k = 0; k < j; ++k) sum -= Get(i,k) * Get(j,k);
66  sum += Get(i,j);
67  Set(i, j, sum / Get(j,j));
68  }
69  }
70  if (j!=Size()) SU2_MPI::Error("LLT factorization failed.", CURRENT_FUNCTION);
71 #endif
72 }

74 void CSymmetricMatrix::CalcInv(bool is_spd)
75 {
76 #ifndef HAVE_LAPACK
77  const int sz = Size();
78 
79  /*--- Compute inverse from decomposed matrices. ---*/
80  if (is_spd)
81  {
82  CholeskyDecompose();
83 
84  /*--- Initialize inverse matrix. ---*/
85  CSymmetricMatrix inv(sz);
86 
87  /*--- Compute L inverse. ---*/
88  /*--- Solve smaller and smaller systems. ---*/
89  for (int j = 0; j < sz; ++j) {
90  /*--- Forward substitution. ---*/
91  inv(j,j) = 1.0 / Get(j,j);
92 
93  for (int i = j+1; i < sz; ++i) {
94  passivedouble sum = 0.0;
95  for (int k = j; k < i; ++k) sum -= Get(i,k) * inv(k,j);
96  inv(i,j) = sum / Get(i,i);
97  }
98  } // L inverse in inv
99 
100  /*--- Multiply inversed matrices overwrite mat. ---*/
101  for (int j = 0; j < sz; ++j)
102  for (int i = j; i < sz; ++i) {
103  passivedouble sum = 0.0;
104  for (int k = i; k < sz; ++k) sum += inv(k,i) * inv(k,j);
105  Set(i, j, sum);
106  }
107  }
108  else {
109  auto inv = StealData();
110  CBlasStructure::inverse(sz, inv);
111  mat = move(inv);
112  }
113 #endif
114 }

116 void CSymmetricMatrix::CalcInv_sytri()
117 {
118 #ifdef HAVE_LAPACK
119  const char uplo = 'L';
120  const int sz = Size();
121  int info;
122  vector<int> ipiv(sz);
123 
124  /*--- Query the optimum work size. ---*/
125  int query = -1; passivedouble tmp;
126  dsytrf_(&uplo, &sz, mat.data(), &sz, ipiv.data(), &tmp, &query, &info);
127  query = static_cast<int>(tmp);
128  vector<passivedouble> work(query);
129 
130  /*--- Factorize and invert. ---*/
131  dsytrf_(&uplo, &sz, mat.data(), &sz, ipiv.data(), work.data(), &query, &info);
132  if (info!=0) SU2_MPI::Error("LDLT factorization failed.", CURRENT_FUNCTION);
133  dsytri_(&uplo, &sz, mat.data(), &sz, ipiv.data(), work.data(), &info);
134  if (info!=0) SU2_MPI::Error("Inversion with LDLT factorization failed.", CURRENT_FUNCTION);
135 #endif
136 }

162 void CSymmetricMatrix::MatMatMult(const char side,
163  const su2passivematrix& mat_in,
164  su2passivematrix& mat_out) const
165 {
166  /*--- Left side: mat_out = this * mat_in. ---*/
167  if (side == 'L' || side == 'l') {
168  const int M = Size(), N = mat_in.cols();
169  assert(M == static_cast<int>(mat_in.rows()));
170 
171  mat_out.resize(M,N);
172 
173 #ifdef HAVE_LAPACK
174  /*--- Right and lower because matrices are in row major order. ---*/
175  const char side = 'R', uplo = 'L';
176  const passivedouble alpha = 1.0, beta = 0.0;
177  DSYMM(&side, &uplo, &N, &M, &alpha, mat.data(), &M,
178  mat_in.data(), &N, &beta, mat_out.data(), &N);
179 #else // Naive product
180  for (int i = 0; i < M; ++i)
181  for (int j = 0; j < N; ++j) {
182  mat_out(i,j) = 0.0;
183  for (int k = 0; k < M; ++k)
184  mat_out(i,j) += Get(i,k) * mat_in(k,j);
185  }
186 #endif
187  }
188  /*--- Right_side: mat_out = mat_in * this. ---*/
189  else {
190  const int M = mat_in.rows(), N = Size();
191  assert(N == static_cast<int>(mat_in.cols()));
192 
193  mat_out.resize(M,N);
194 
195 #ifdef HAVE_LAPACK
196  /*--- Left and lower because matrices are in row major order. ---*/
197  const char side = 'L', uplo = 'L';
198  const passivedouble alpha = 1.0, beta = 0.0;
199  DSYMM(&side, &uplo, &N, &M, &alpha, mat.data(), &N,
200  mat_in.data(), &N, &beta, mat_out.data(), &N);
201 #else // Naive product
202  for (int i = 0; i < M; ++i)
203  for (int j = 0; j < N; ++j) {
204  mat_out(i,j) = 0.0;
205  for (int k = 0; k < N; ++k)
206  mat_out(i,j) += mat_in(i,k) * Get(k,j);
207  }
208 #endif
209  }
210 }
