37 CInterpolator* CreateInterpolator(CGeometry ****geometry_container,
38  const CConfig* const* config,
39  const CInterpolator* transpInterpolator,
40  unsigned iZone, unsigned jZone, bool verbose) {
41 
42  CInterpolator* interpolator = nullptr;
43 
44  /*--- Only print information on master node. ---*/
45  verbose &= (SU2_MPI::GetRank() == MASTER_NODE);
46 
47  /*--- Type of interpolation defined by donor. ---*/
48  const auto type = config[iZone]->GetKindInterpolation();
49 
50  if (verbose) cout << " Setting coupling ";
51 
52  /*--- Conservative interpolation is not applicable to the sliding
53  * mesh approach so that case is handled first. Then we either
54  * return a CMirror if the target requires conservative inter-
55  * polation, or the type of interpolator defined by "type". ---*/
56 
57  if (type == INTERFACE_INTERPOLATOR::WEIGHTED_AVERAGE) {
58  if (verbose) cout << "using a sliding mesh approach." << endl;
59  interpolator = new CSlidingMesh(geometry_container, config, iZone, jZone);
60  }
61  else if (config[jZone]->GetConservativeInterpolation()) {
62  if (verbose) cout << "using the mirror approach, \"transposing\" coefficients from opposite mesh." << endl;
63  interpolator = new CMirror(geometry_container, config, transpInterpolator, iZone, jZone);
64  }
65  else {
66  switch(type) {
67  case INTERFACE_INTERPOLATOR::ISOPARAMETRIC:
68  if (verbose) cout << "using the isoparametric approach." << endl;
69  interpolator = new CIsoparametric(geometry_container, config, iZone, jZone);
70  break;
71 
72  case INTERFACE_INTERPOLATOR::NEAREST_NEIGHBOR:
73  if (verbose) cout << "using a nearest neighbor approach." << endl;
74  interpolator = new CNearestNeighbor(geometry_container, config, iZone, jZone);
75  break;
76 
77  case INTERFACE_INTERPOLATOR::RADIAL_BASIS_FUNCTION:
78  if (verbose) cout << "using a radial basis function approach." << endl;
79  interpolator = new CRadialBasisFunction(geometry_container, config, iZone, jZone);
80  break;
81 
82  default:
83  SU2_MPI::Error("Unknown type of interpolation.", CURRENT_FUNCTION);
84  }
85  }
86 
87  if (verbose) interpolator->PrintStatistics();
88 
89  return interpolator;
90 }
