36 /*!
37  * \brief Class to represent a matrix (without owning the data, this just wraps a pointer).
38  */
39 template<class T>
40 class CMatrixView {
41 public:
42  using Scalar = typename std::remove_const<T>::type;
43  using Index = unsigned long;
44 
45 private:
46  T* m_ptr;
47  Index m_cols;
48 
49 public:
50  CMatrixView(T* ptr = nullptr, Index cols = 0) : m_ptr(ptr), m_cols(cols) {}
51 
52  template<class U> friend class CMatrixView;
53  template<class U>
54  CMatrixView(const CMatrixView<U>& other) : m_ptr(other.m_ptr), m_cols(other.m_cols) {}
55 
56  explicit CMatrixView(su2matrix<Scalar>& mat) : m_ptr(mat.data()), m_cols(mat.cols()) {}
57 
58  template<class U = T, su2enable_if<std::is_const<U>::value> = 0>
59  explicit CMatrixView(const su2matrix<Scalar>& mat) : m_ptr(mat.data()), m_cols(mat.cols()) {}
60 
61  const Scalar* operator[] (Index i) const noexcept { return &m_ptr[i*m_cols]; }
62  const Scalar& operator() (Index i, Index j) const noexcept { return m_ptr[i*m_cols + j]; }
63 
64  template<class U = T, su2enable_if<!std::is_const<U>::value> = 0>
65  Scalar* operator[] (Index i) noexcept { return &m_ptr[i*m_cols]; }
66 
67  template<class U = T, su2enable_if<!std::is_const<U>::value> = 0>
68  Scalar& operator() (Index i, Index j) noexcept { return m_ptr[i*m_cols + j]; }
69 
70  friend CMatrixView operator+ (CMatrixView mv, Index incr) { return CMatrixView(mv[incr], mv.m_cols); }
71 };

73 /*!
74  * \class C3DContainerDecorator
75  * \brief Decorate a matrix type (Storage) with 3 dimensions.
76  */
77 template<class Storage>
78 class C3DContainerDecorator {
79  static_assert(!Storage::IsVector, "Storage type must be a matrix.");
80  static_assert(Storage::IsRowMajor, "Storage type must be row major.");
81 public:
82  using Scalar = typename Storage::Scalar;
83  using Index = typename Storage::Index;
84  static constexpr bool IsRowMajor = true;
85  static constexpr bool IsColumnMajor = false;
86 
87  using Matrix = CMatrixView<Scalar>;
88  using ConstMatrix = CMatrixView<const Scalar>;
89 
90  using CInnerIter = typename Storage::CInnerIter;
91  template<class T, size_t N>
92  using CInnerIterGather = typename Storage::template CInnerIterGather<simd::Array<T,N> >;
93 
94 private:
95  Storage m_storage;
96  Index m_innerSz;
97 
98 public:
99  C3DContainerDecorator() = default;
100 
101  C3DContainerDecorator(Index length, Index rows, Index cols, Scalar value = 0) noexcept {
102  resize(length, rows, cols, value);
103  }
104 
105  void resize(Index length, Index rows, Index cols, Scalar value = 0) noexcept {
106  m_innerSz = cols;
107  m_storage.resize(length, rows*cols) = value;
108  }
109 
110  /*!
111  * \brief Container sizes.
112  */
113  Index size() const noexcept { return m_storage.size(); }
114  Index length() const noexcept { return m_storage.rows(); }
115  Index rows() const noexcept { return m_storage.cols() / m_innerSz; }
116  Index cols() const noexcept { return m_innerSz; }
117 
118  /*!
119  * \brief Element-wise access.
120  */
121  Scalar& operator() (Index i, Index j, Index k) noexcept { return m_storage(i, j*m_innerSz + k); }
122  const Scalar& operator() (Index i, Index j, Index k) const noexcept { return m_storage(i, j*m_innerSz + k); }
123 
124  /*!
125  * \brief Matrix access.
126  */
127  Matrix operator[] (Index i) noexcept { return Matrix(m_storage[i], m_innerSz); }
128  ConstMatrix operator[] (Index i) const noexcept { return ConstMatrix(m_storage[i], m_innerSz); }
129 
130  /*!
131  * \brief Matrix access with an offset.
132  */
133  Matrix operator() (Index i, Index j) noexcept { return Matrix(m_storage[i]+j*m_innerSz, m_innerSz); }
134  ConstMatrix operator() (Index i, Index j) const noexcept { return ConstMatrix(m_storage[i]+j*m_innerSz, m_innerSz); }
135 
136  /*!
137  * \brief Get a scalar iterator to the inner-most dimension of the container.
138  */
139  FORCEINLINE CInnerIter innerIter(Index i, Index j) const noexcept {
140  return CInnerIter(&m_storage(i, j*m_innerSz), 1);
141  }
142 
143  /*!
144  * \brief Get a SIMD gather iterator to the inner-most dimension of the container.
145  */
146  template<class T, size_t N>
147  FORCEINLINE CInnerIterGather<T,N> innerIter(simd::Array<T,N> i, Index j) const noexcept {
148  return CInnerIterGather<T,N>(m_storage.data(), 1, i*m_storage.cols() + j*m_innerSz);
149  }
150 
151  /*!
152  * \brief Return copy of data in a static size container (see C2DContainer::get).
153  * \param[in] i - Outer index.
154  * \param[in] j - Starting middle index for the copy (amount determined by container size).
155  */
156  template<class StaticContainer, class Int>
157  FORCEINLINE StaticContainer get(Int i, Index j = 0) const noexcept {
158  return m_storage.template get<StaticContainer>(i, j*m_innerSz);
159  }
160 };

235 /*!
236  * \brief Allocate a vector of varying-size vectors and initialize with some value.
237  * \param[in] M - the first index is >=0 and < M
238  * \param[in] N - the second index is >=0 and < N[first index]
239  * \param[in,out] X - the vector of vectors
240  * \param[in] val - the value for initialization, default is 0
241  * \tparam IndexVector - type of N
242  * \tparam VectorOfVector - type of X
243  */
244 template<class IndexVector, class VectorOfVector, class Scalar = int>
245 inline void AllocVectorOfVectors(size_t M, const IndexVector& N, VectorOfVector& X, Scalar val = 0) {
246  X.resize(M);
247  for(size_t i = 0; i < M; ++i){
248  X[i].resize(N[i]);
249  for (auto& x : X[i]) x = val;
250  }
251 }

253 /*!

256 template<class IndexVector, class VectorOfVector, class Scalar = int>
257 inline void AllocVectorOfVectors(const IndexVector& N, VectorOfVector& X, Scalar val = 0) {
258  auto M = N.size();
259  AllocVectorOfVectors(M, N, X, val);
260 }
