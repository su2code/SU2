58  Error_RMS = new su2double[nVar];

67  Error_Point_Max_Coord[iVar] = new su2double[nDim];
68  for (unsigned short iDim = 0; iDim < nDim; iDim++)
69  Error_Point_Max_Coord[iVar][iDim] = 0.0;
70  }
71 }

124 void CVerificationSolution::GetLocalError(const su2double *val_coords,
125  const su2double val_t,
126  const su2double *val_solution,
127  su2double *val_error) const {
128 
129  /*--- Get the value of the verification solution first.
130  Use val_error to store this solution. ---*/
131 
132  GetSolution(val_coords, val_t, val_error);
133 
134  /*--- Compute the local error as the difference between the current
135  numerical solution and the verification solution. ---*/
136 
137  for (unsigned short iVar=0; iVar<nVar; ++iVar)
138  val_error[iVar] = val_solution[iVar] - val_error[iVar];
139 
140 }

142 void CVerificationSolution::SetVerificationError(unsigned long nDOFsGlobal,
143  CConfig *config) {
144 
145  /* Disable the reduce for the error to avoid overhead if requested. */
146  if (config->GetComm_Level() == COMM_FULL) {
147 
148  /*--- Get the number of ranks and the MPI communicator. ---*/
149  int size = SU2_MPI::GetSize();
150  SU2_MPI::Comm comm = SU2_MPI::GetComm();
151 
152  /*--- The local L2 norms must be added to obtain the global value. ---*/
153  vector<su2double> rbufError(nVar,0.0);
154  SU2_MPI::Allreduce(Error_RMS, rbufError.data(), nVar,
155  MPI_DOUBLE, MPI_SUM, comm);
156 
157  for(unsigned short iVar=0; iVar<nVar; ++iVar) {
158  SetError_RMS(iVar, max(EPS*EPS, sqrt(rbufError[iVar]/nDOFsGlobal)));
159  }
160 
161  /*--- The global maximum norms must be obtained. ---*/
162  rbufError.resize(nVar*size,0.0);
163  SU2_MPI::Allgather(Error_Max, nVar, MPI_DOUBLE, rbufError.data(),
164  nVar, MPI_DOUBLE, comm);
165 
166  vector<unsigned long> rbufPoint(nVar*size);
167  SU2_MPI::Allgather(Error_Point_Max, nVar, MPI_UNSIGNED_LONG, rbufPoint.data(),
168  nVar, MPI_UNSIGNED_LONG, comm);
169 
170  vector<su2double> sbufCoor(nDim*nVar,0.0);
171  for(unsigned short iVar=0; iVar<nVar; ++iVar) {
172  for(unsigned short iDim=0; iDim<nDim; ++iDim)
173  sbufCoor[iVar*nDim+iDim] = Error_Point_Max_Coord[iVar][iDim];
174  }
175 
176  vector<su2double> rbufCoor(nDim*nVar*size,0.0);
177  SU2_MPI::Allgather(sbufCoor.data(), nVar*nDim, MPI_DOUBLE, rbufCoor.data(),
178  nVar*nDim, MPI_DOUBLE, comm);
179 
180  for(unsigned short iVar=0; iVar<nVar; ++iVar) {
181  for(int proc=0; proc<size; ++proc)
182  AddError_Max(iVar, rbufError[proc*nVar+iVar], rbufPoint[proc*nVar+iVar],
183  &rbufCoor[proc*nVar*nDim+iVar*nDim]);
184  }
185  }
186 }
