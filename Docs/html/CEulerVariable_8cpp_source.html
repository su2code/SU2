31 CEulerVariable::CEulerVariable(su2double density, const su2double *velocity, su2double energy, unsigned long npoint,
32  unsigned long ndim, unsigned long nvar, const CConfig *config)
33  : CFlowVariable(npoint, ndim, nvar, ndim + 9, ndim + 4, config),
34  indices(ndim, 0) {
35 
36  const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) ||
37  (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
38  const bool classical_rk4 = (config->GetKind_TimeIntScheme_Flow() == CLASSICAL_RK4_EXPLICIT);
39 
40  nSecondaryVar = config->GetViscous() ? 8 : 2,
41  nSecondaryVarGrad = 2;
42 
43  /*--- Solution initialization ---*/
44 
45  su2double val_solution[5] = {su2double(1.0), velocity[0], velocity[1], energy, energy};
46  if(nDim==3) val_solution[3] = velocity[2];
47 
48  for(unsigned long iPoint = 0; iPoint < nPoint; ++iPoint)
49  for (unsigned long iVar = 0; iVar < nVar; ++iVar)
50  Solution(iPoint,iVar) = density*val_solution[iVar];
51 
52  Solution_Old = Solution;
53 
54  if (classical_rk4) Solution_New = Solution;
55 
56  /*--- Allocate and initializate solution for dual time strategy ---*/
57 
58  if (dual_time) {
59  Solution_time_n = Solution;
60  Solution_time_n1 = Solution;
61  }
62 
63  Secondary.resize(nPoint,nSecondaryVar) = su2double(0.0);
64 
65  if (config->GetAxisymmetric()){
66  nAuxVar = 3;
67  Grad_AuxVar.resize(nPoint,nAuxVar,nDim,0.0);
68  AuxVar.resize(nPoint,nAuxVar) = su2double(0.0);
69  }
70 
71  if (config->GetWind_Gust()) {
72  WindGust.resize(nPoint,nDim);
73  WindGustDer.resize(nPoint,nDim+1);
74  }
75 
76  if (config->GetVorticityConfinement()) {
77  nAuxVar = 1;
78  Grad_AuxVar.resize(nPoint, nAuxVar, nDim, 0.0);
79  AuxVar.resize(nPoint, nAuxVar) = su2double(0.0);
80  }
81 }

83 bool CEulerVariable::SetPrimVar(unsigned long iPoint, CFluidModel *FluidModel) {
84 
85  bool RightVol = true;
86 
87  SetVelocity(iPoint); // Computes velocity and velocity^2
88  su2double density = GetDensity(iPoint);
89  su2double staticEnergy = GetEnergy(iPoint)-0.5*Velocity2(iPoint);
90 
91  /*--- Check will be moved inside fluid model plus error description strings ---*/
92 
93  FluidModel->SetTDState_rhoe(density, staticEnergy);
94 
95  bool check_dens = SetDensity(iPoint);
96  bool check_press = SetPressure(iPoint, FluidModel->GetPressure());
97  bool check_sos = SetSoundSpeed(iPoint, FluidModel->GetSoundSpeed2());
98  bool check_temp = SetTemperature(iPoint, FluidModel->GetTemperature());
99 
100  /*--- Check that the solution has a physical meaning ---*/
101 
102  if (check_dens || check_press || check_sos || check_temp) {
103 
104  /*--- Copy the old solution ---*/
105 
106  for (unsigned long iVar = 0; iVar < nVar; iVar++)
107  Solution(iPoint, iVar) = Solution_Old(iPoint, iVar);
108 
109  /*--- Recompute the primitive variables ---*/
110 
111  SetVelocity(iPoint); // Computes velocity and velocity^2
112  su2double density = GetDensity(iPoint);
113  su2double staticEnergy = GetEnergy(iPoint)-0.5*Velocity2(iPoint);
114  /* check will be moved inside fluid model plus error description strings*/
115  FluidModel->SetTDState_rhoe(density, staticEnergy);
116 
117  SetDensity(iPoint);
118  SetPressure(iPoint, FluidModel->GetPressure());
119  SetSoundSpeed(iPoint, FluidModel->GetSoundSpeed2());
120  SetTemperature(iPoint, FluidModel->GetTemperature());
121 
122  RightVol = false;
123 
124  }
125 
126  SetEnthalpy(iPoint); // Requires pressure computation.
127 
128  return RightVol;
129 }
