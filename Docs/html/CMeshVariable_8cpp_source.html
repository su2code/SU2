31 CMeshVariable::CMeshVariable(unsigned long npoint, unsigned long ndim, CConfig *config) :
32  CVariable(npoint, ndim, config) {
33 
34  /*--- Store the dimensionality of the problem ---*/
35  nDim = ndim;
36 
37  /*--- Initalize the variables that will always be there in a problem with moving mesh ---*/
38  Mesh_Coord.resize(nPoint,nDim) = su2double(0.0);
39  WallDistance.resize(nPoint) = su2double(1e-9);
40 
41  /*--- Initialize the variables necessary when the problem is time domain ---*/
42  if (config->GetTime_Domain()) {
43  Solution_time_n.resize(nPoint,nDim) = su2double(0.0);
44  Solution_time_n1.resize(nPoint,nDim) = su2double(0.0);
45  }
46 }
