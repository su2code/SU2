34 CAdjHeatOutput::CAdjHeatOutput(CConfig *config, unsigned short nDim) : COutput(config, nDim, false) {
35 
36  /*--- Set the default history fields if nothing is set in the config file ---*/
37 
38  if (nRequestedHistoryFields == 0){
39  requestedHistoryFields.emplace_back("ITER");
40  requestedHistoryFields.emplace_back("RMS_RES");
41  requestedHistoryFields.emplace_back("SENSITIVITY");
42  nRequestedHistoryFields = requestedHistoryFields.size();
43  }
44 
45  if (nRequestedScreenFields == 0){
46  if (multiZone) requestedScreenFields.emplace_back("OUTER_ITER");
47  requestedScreenFields.emplace_back("INNER_ITER");
48  requestedScreenFields.emplace_back("RMS_ADJ_TEMPERATURE");
49  requestedScreenFields.emplace_back("SENS_GEO");
50  nRequestedScreenFields = requestedScreenFields.size();
51  }
52 
53  if (nRequestedVolumeFields == 0){
54  requestedVolumeFields.emplace_back("COORDINATES");
55  requestedVolumeFields.emplace_back("SOLUTION");
56  requestedVolumeFields.emplace_back("SENSITIVITY");
57  nRequestedVolumeFields = requestedVolumeFields.size();
58  }
59 
60  if (find(requestedVolumeFields.begin(), requestedVolumeFields.end(), string("SENSITIVITY")) == requestedVolumeFields.end()) {
61  requestedVolumeFields.emplace_back("SENSITIVITY");
62  nRequestedVolumeFields ++;
63  }
64 
65  stringstream ss;
66  ss << "Zone " << config->GetiZone() << " (Adj. Heat)";
67  multiZoneHeaderString = ss.str();
68 
69  /*--- Set the volume filename --- */
70 
71  volumeFilename = config->GetAdj_FileName();
72 
73  /*--- Set the surface filename --- */
74 
75  surfaceFilename = config->GetSurfAdjCoeff_FileName();
76 
77  /*--- Set the restart filename --- */
78 
79  restartFilename = config->GetRestart_AdjFileName();
80 
81  /*--- Add the obj. function extension --- */
82 
83  restartFilename = config->GetObjFunc_Extension(restartFilename);
84 
85  /*--- Set the default convergence field --- */
86 
87  if (convFields.empty() ) convFields.emplace_back("RMS_ADJ_TEMPERATURE");
88 
89 }

93 void CAdjHeatOutput::SetHistoryOutputFields(CConfig *config){
94 
95  /// BEGIN_GROUP: RMS_RES, DESCRIPTION: The root-mean-square residuals of the conservative variables.
96  /// DESCRIPTION: Root-mean square residual of the adjoint temperature.
97  AddHistoryOutput("RMS_ADJ_TEMPERATURE", "rms[A_T]", ScreenOutputFormat::FIXED, "RMS_RES", "Root-mean square residual of the adjoint temperature.", HistoryFieldType::RESIDUAL);
98  /// END_GROUP
99 
100  /// BEGIN_GROUP: MAX_RES, DESCRIPTION: The maximum residuals of the conservative variables.
101  /// DESCRIPTION: Maximum residual of the adjoint temperature.
102  AddHistoryOutput("MAX_ADJ_TEMPERATURE", "max[A_T]", ScreenOutputFormat::FIXED, "MAX_RES", "Maximum residual of the adjoint temperature.", HistoryFieldType::RESIDUAL);
103 
104  /// BEGIN_GROUP: MAX_RES, DESCRIPTION: The root-mean-square residuals of the conservative variables.
105  /// DESCRIPTION: Root-mean-square residual of the adjoint temperature.
106  AddHistoryOutput("BGS_ADJ_TEMPERATURE", "bgs[A_T]", ScreenOutputFormat::FIXED, "BGS_RES", "BGS residual of the adjoint temperature.", HistoryFieldType::RESIDUAL);
107 
108  /// BEGIN_GROUP: SENSITIVITY, DESCRIPTION: Sensitivities of different geometrical or boundary values.
109  /// DESCRIPTION: Sum of the geometrical sensitivities on all markers set in MARKER_MONITORING.
110  AddHistoryOutput("SENS_GEO", "Sens_Geo", ScreenOutputFormat::SCIENTIFIC, "SENSITIVITY", "Sum of the geometrical sensitivities on all markers set in MARKER_MONITORING.", HistoryFieldType::COEFFICIENT);
111  /// END_GROUP
112 
113  AddHistoryOutput("LINSOL_ITER", "LinSolIter", ScreenOutputFormat::INTEGER, "LINSOL", "Number of iterations of the linear solver.");
114  AddHistoryOutput("LINSOL_RESIDUAL", "LinSolRes", ScreenOutputFormat::FIXED, "LINSOL", "Residual of the linear solver.");
115 
116  if (config->GetDeform_Mesh()){
117  AddHistoryOutput("DEFORM_ITER", "DeformIter", ScreenOutputFormat::INTEGER, "DEFORM", "Linear solver iterations for the mesh deformation");
118  AddHistoryOutput("DEFORM_RESIDUAL", "DeformRes", ScreenOutputFormat::FIXED, "DEFORM", "Residual of the linear solver for the mesh deformation");
119  }
120 
121 }

123 void CAdjHeatOutput::LoadHistoryData(CConfig *config, CGeometry *geometry, CSolver **solver) {
124 
125  CSolver* adjheat_solver = solver[ADJHEAT_SOL];
126 
127  SetHistoryOutputValue("RMS_ADJ_TEMPERATURE", log10(adjheat_solver->GetRes_RMS(0)));
128 
129  SetHistoryOutputValue("MAX_ADJ_TEMPERATURE", log10(adjheat_solver->GetRes_Max(0)));
130 
131  if (multiZone) {
132  SetHistoryOutputValue("BGS_ADJ_TEMPERATURE", log10(adjheat_solver->GetRes_BGS(0)));
133  }
134 
135  SetHistoryOutputValue("SENS_GEO", adjheat_solver->GetTotal_Sens_Geo());
136 
137  SetHistoryOutputValue("LINSOL_ITER", adjheat_solver->GetIterLinSolver());
138  SetHistoryOutputValue("LINSOL_RESIDUAL", log10(adjheat_solver->GetResLinSolver()));
139 
140  if (config->GetDeform_Mesh()) {
141  SetHistoryOutputValue("DEFORM_ITER", solver[MESH_SOL]->System.GetIterations());
142  SetHistoryOutputValue("DEFORM_RESIDUAL", log10(solver[MESH_SOL]->System.GetResidual()));
143  }
144 
145 }

181 void CAdjHeatOutput::LoadVolumeData(CConfig *config, CGeometry *geometry, CSolver **solver, unsigned long iPoint){
182 
183  CVariable* Node_AdjHeat = solver[ADJHEAT_SOL]->GetNodes();
184  CPoint* Node_Geo = geometry->nodes;
185 
186 
187  SetVolumeOutputValue("COORD-X", iPoint, Node_Geo->GetCoord(iPoint, 0));
188  SetVolumeOutputValue("COORD-Y", iPoint, Node_Geo->GetCoord(iPoint, 1));
189  if (nDim == 3)
190  SetVolumeOutputValue("COORD-Z", iPoint, Node_Geo->GetCoord(iPoint, 2));
191 
192  SetVolumeOutputValue("ADJ_TEMPERATURE", iPoint, Node_AdjHeat->GetSolution(iPoint, 0));
193 
194  // Residuals
195  SetVolumeOutputValue("RES_ADJ_TEMPERATURE", iPoint, Node_AdjHeat->GetSolution(iPoint, 0) - Node_AdjHeat->GetSolution_Old(iPoint, 0));
196 
197  SetVolumeOutputValue("SENSITIVITY-X", iPoint, Node_AdjHeat->GetSensitivity(iPoint, 0));
198  SetVolumeOutputValue("SENSITIVITY-Y", iPoint, Node_AdjHeat->GetSensitivity(iPoint, 1));
199  if (nDim == 3)
200  SetVolumeOutputValue("SENSITIVITY-Z", iPoint, Node_AdjHeat->GetSensitivity(iPoint, 2));
201 
202 }
