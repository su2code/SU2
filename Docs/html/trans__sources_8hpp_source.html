49  const su2double c_a2 = 0.06;
50  const su2double sigmaf = 1.0;
51  const su2double s1 = 2.0;
52  const su2double c_theta = 0.03;

89  /*!
90  * \brief Residual for source term integration.
91  * \param[in] config - Definition of the particular problem.
92  * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
93  */
94  ResidualType<> ComputeResidual(const CConfig* config) override {
95  /*--- ScalarVar[0] = k, ScalarVar[0] = w, TransVar[0] = gamma, and TransVar[0] = ReThetaT ---*/
96  /*--- dU/dx = PrimVar_Grad[1][0] ---*/
97  AD::StartPreacc();
98  AD::SetPreaccIn(StrainMag_i);
99  AD::SetPreaccIn(ScalarVar_i, nVar);
100  AD::SetPreaccIn(ScalarVar_Grad_i, nVar, nDim);
101  AD::SetPreaccIn(TransVar_i, nVar);
102  AD::SetPreaccIn(TransVar_Grad_i, nVar, nDim);
103  AD::SetPreaccIn(Volume);
104  AD::SetPreaccIn(dist_i);
105  AD::SetPreaccIn(&V_i[idx.Velocity()], nDim);
106  AD::SetPreaccIn(PrimVar_Grad_i, nDim + idx.Velocity(), nDim);
107  AD::SetPreaccIn(Vorticity_i, 3);
108 
109  su2double VorticityMag =
110  sqrt(Vorticity_i[0] * Vorticity_i[0] + Vorticity_i[1] * Vorticity_i[1] + Vorticity_i[2] * Vorticity_i[2]);
111 
112  const su2double vel_u = V_i[idx.Velocity()];
113  const su2double vel_v = V_i[1 + idx.Velocity()];
114  const su2double vel_w = (nDim == 3) ? V_i[2 + idx.Velocity()] : 0.0;
115 
116  const su2double Velocity_Mag = sqrt(vel_u * vel_u + vel_v * vel_v + vel_w * vel_w);
117 
118  AD::SetPreaccIn(V_i[idx.Density()], V_i[idx.LaminarViscosity()], V_i[idx.EddyViscosity()]);
119 
120  Density_i = V_i[idx.Density()];
121  Laminar_Viscosity_i = V_i[idx.LaminarViscosity()];
122  Eddy_Viscosity_i = V_i[idx.EddyViscosity()];
123 
124  Residual[0] = 0.0;
125  Residual[1] = 0.0;
126  Jacobian_i[0][0] = 0.0;
127  Jacobian_i[0][1] = 0.0;
128  Jacobian_i[1][0] = 0.0;
129  Jacobian_i[1][1] = 0.0;
130 
131  if (dist_i > 1e-10) {
132  su2double Tu = 1.0;
133  if (TurbFamily == TURB_FAMILY::KW) Tu = max(100.0 * sqrt(2.0 * ScalarVar_i[0] / 3.0) / Velocity_Mag, 0.027);
134  if (TurbFamily == TURB_FAMILY::SA) Tu = config->GetTurbulenceIntensity_FreeStream() * 100;
135 
136  /*--- Corr_RetC correlation*/
137  const su2double Corr_Rec = TransCorrelations.ReThetaC_Correlations(Tu, TransVar_i[1]);
138 
139  /*--- F_length correlation*/
140  const su2double Corr_F_length = TransCorrelations.FLength_Correlations(Tu, TransVar_i[1]);
141 
142  /*--- F_length ---*/
143  su2double F_length = 0.0;
144  if (TurbFamily == TURB_FAMILY::KW) {
145  const su2double r_omega = Density_i * dist_i * dist_i * ScalarVar_i[1] / Laminar_Viscosity_i;
146  const su2double f_sub = exp(-pow(r_omega / 200.0, 2));
147  F_length = Corr_F_length * (1. - f_sub) + 40.0 * f_sub;
148  }
149  if (TurbFamily == TURB_FAMILY::SA) F_length = Corr_F_length;
150 
151  /*--- F_onset ---*/
152  su2double R_t = 1.0;
153  if (TurbFamily == TURB_FAMILY::KW) R_t = Density_i * ScalarVar_i[0] / Laminar_Viscosity_i / ScalarVar_i[1];
154  if (TurbFamily == TURB_FAMILY::SA) R_t = Eddy_Viscosity_i / Laminar_Viscosity_i;
155 
156  const su2double Re_v = Density_i * dist_i * dist_i * StrainMag_i / Laminar_Viscosity_i;
157  const su2double F_onset1 = Re_v / (2.193 * Corr_Rec);
158  su2double F_onset2 = 1.0;
159  su2double F_onset3 = 1.0;
160  if (TurbFamily == TURB_FAMILY::KW) {
161  F_onset2 = min(max(F_onset1, pow(F_onset1, 4.0)), 2.0);
162  F_onset3 = max(1.0 - pow(R_t / 2.5, 3.0), 0.0);
163  }
164  if (TurbFamily == TURB_FAMILY::SA) {
165  F_onset2 = min(max(F_onset1, pow(F_onset1, 4.0)), 4.0);
166  F_onset3 = max(2.0 - pow(R_t / 2.5, 3.0), 0.0);
167  }
168  const su2double F_onset = max(F_onset2 - F_onset3, 0.0);
169 
170  /*-- Gradient of velocity magnitude ---*/
171 
172  su2double dU_dx = 0.5 / Velocity_Mag * (2. * vel_u * PrimVar_Grad_i[1][0] + 2. * vel_v * PrimVar_Grad_i[2][0]);
173  if (nDim == 3) dU_dx += 0.5 / Velocity_Mag * (2. * vel_w * PrimVar_Grad_i[3][0]);
174 
175  su2double dU_dy = 0.5 / Velocity_Mag * (2. * vel_u * PrimVar_Grad_i[1][1] + 2. * vel_v * PrimVar_Grad_i[2][1]);
176  if (nDim == 3) dU_dy += 0.5 / Velocity_Mag * (2. * vel_w * PrimVar_Grad_i[3][1]);
177 
178  su2double dU_dz = 0.0;
179  if (nDim == 3)
180  dU_dz =
181  0.5 / Velocity_Mag *
182  (2. * vel_u * PrimVar_Grad_i[1][2] + 2. * vel_v * PrimVar_Grad_i[2][2] + 2. * vel_w * PrimVar_Grad_i[3][2]);
183 
184  su2double du_ds = vel_u / Velocity_Mag * dU_dx + vel_v / Velocity_Mag * dU_dy;
185  if (nDim == 3) du_ds += vel_w / Velocity_Mag * dU_dz;
186 
187  /*-- Calculate blending function f_theta --*/
188  su2double time_scale = 500.0 * Laminar_Viscosity_i / Density_i / Velocity_Mag / Velocity_Mag;
189  if (options.LM2015)
190  time_scale = min(time_scale,
191  Density_i * LocalGridLength_i * LocalGridLength_i / (Laminar_Viscosity_i + Eddy_Viscosity_i));
192  const su2double theta_bl = TransVar_i[1] * Laminar_Viscosity_i / Density_i / Velocity_Mag;
193  const su2double delta_bl = 7.5 * theta_bl;
194  const su2double delta = 50.0 * VorticityMag * dist_i / Velocity_Mag * delta_bl + 1e-20;
195 
196  su2double f_wake = 0.0;
197  if (TurbFamily == TURB_FAMILY::KW) {
198  const su2double re_omega = Density_i * ScalarVar_i[1] * dist_i * dist_i / Laminar_Viscosity_i;
199  f_wake = exp(-pow(re_omega / (1.0e+05), 2));
200  }
201  if (TurbFamily == TURB_FAMILY::SA) f_wake = 1.0;
202 
203  const su2double var1 = (TransVar_i[0] - 1.0 / c_e2) / (1.0 - 1.0 / c_e2);
204  const su2double var2 = 1.0 - pow(var1, 2.0);
205  const su2double f_theta = min(max(f_wake * exp(-pow(dist_i / delta, 4)), var2), 1.0);
206  const su2double f_turb = exp(-pow(R_t / 4, 4));
207 
208  su2double f_theta_2 = 0.0;
209  if (options.LM2015)
210  f_theta_2 = min(f_wake * exp(-pow(dist_i / delta, 4.0)), 1.0);
211 
212  /*--- Corr_Ret correlation*/
213  const su2double Corr_Ret_lim = 20.0;
214  su2double f_lambda = 1.0;
215 
216  su2double Retheta_Error = 200.0, Retheta_old = 0.0;
217  su2double lambda = 0.0;
218  su2double Corr_Ret = 20.0;
219 
220  for (int iter = 0; iter < 100; iter++) {
221  su2double theta = Corr_Ret * Laminar_Viscosity_i / Density_i / Velocity_Mag;
222  lambda = Density_i * theta * theta / Laminar_Viscosity_i * du_ds;
223  lambda = min(max(-0.1, lambda), 0.1);
224 
225  if (lambda <= 0.0) {
226  f_lambda = 1. - (-12.986 * lambda - 123.66 * lambda * lambda - 405.689 * lambda * lambda * lambda) *
227  exp(-pow(Tu / 1.5, 1.5));
228  } else {
229  f_lambda = 1. + 0.275 * (1. - exp(-35. * lambda)) * exp(-Tu / 0.5);
230  }
231 
232  if (Tu <= 1.3) {
233  Corr_Ret = f_lambda * (1173.51 - 589.428 * Tu + 0.2196 / Tu / Tu);
234  } else {
235  Corr_Ret = 331.5 * f_lambda * pow(Tu - 0.5658, -0.671);
236  }
237  Corr_Ret = max(Corr_Ret, Corr_Ret_lim);
238 
239  Retheta_Error = fabs(Retheta_old - Corr_Ret) / Retheta_old;
240 
241  if (Retheta_Error < 0.0000001) {
242  break;
243  }
244 
245  Retheta_old = Corr_Ret;
246  }
247 
248  /*-- Corr_RetT_SCF Correlations--*/
249  su2double ReThetat_SCF = 0.0;
250  if (options.LM2015) {
251  su2double VelocityNormalized[3];
252  VelocityNormalized[0] = vel_u / Velocity_Mag;
253  VelocityNormalized[1] = vel_v / Velocity_Mag;
254  if (nDim == 3) VelocityNormalized[2] = vel_w / Velocity_Mag;
255 
256  su2double StreamwiseVort = 0.0;
257  for (auto iDim = 0u; iDim < nDim; iDim++) {
258  StreamwiseVort += VelocityNormalized[iDim] * Vorticity_i[iDim];
259  }
260  StreamwiseVort = abs(StreamwiseVort);
261 
262  const su2double H_CF = StreamwiseVort * dist_i / Velocity_Mag;
263  const su2double DeltaH_CF = H_CF * (1.0 + min(Eddy_Viscosity_i / Laminar_Viscosity_i, 0.4));
264  const su2double DeltaH_CF_Minus = max(-1.0 * (0.1066 - DeltaH_CF), 0.0);
265  const su2double DeltaH_CF_Plus = max(0.1066 - DeltaH_CF, 0.0);
266  const su2double fDeltaH_CF_Minus = 75.0 * tanh(DeltaH_CF_Minus / 0.0125);
267  const su2double fDeltaH_CF_Plus = 6200 * DeltaH_CF_Plus + 50000 * DeltaH_CF_Plus * DeltaH_CF_Plus;
268 
269  const su2double toll = 1e-5;
270  su2double error = toll + 1.0;
271  su2double thetat_SCF = 0.0;
272  su2double rethetat_SCF_old = 20.0;
273  const int nMax = 100;
274 
275  int iter;
276  for (iter = 0; iter < nMax && error > toll; iter++) {
277  thetat_SCF = rethetat_SCF_old * Laminar_Viscosity_i / (Density_i * (Velocity_Mag / 0.82));
278  thetat_SCF = max(1e-20, thetat_SCF);
279 
280  ReThetat_SCF = -35.088 * log(hRoughness / thetat_SCF) + 319.51 + fDeltaH_CF_Plus - fDeltaH_CF_Minus;
281 
282  error = abs(ReThetat_SCF - rethetat_SCF_old) / rethetat_SCF_old;
283 
284  rethetat_SCF_old = ReThetat_SCF;
285  }
286  }
287 
288  /*-- production term of Intermeittency(Gamma) --*/
289  const su2double Pg =
290  F_length * c_a1 * Density_i * StrainMag_i * sqrt(F_onset * TransVar_i[0]) * (1.0 - c_e1 * TransVar_i[0]);
291 
292  /*-- destruction term of Intermeittency(Gamma) --*/
293  const su2double Dg = c_a2 * Density_i * VorticityMag * TransVar_i[0] * f_turb * (c_e2 * TransVar_i[0] - 1.0);
294 
295  /*-- production term of ReThetaT --*/
296  const su2double PRethetat = c_theta * Density_i / time_scale * (Corr_Ret - TransVar_i[1]) * (1.0 - f_theta);
297 
298  /*-- destruction term of ReThetaT --*/
299  // It should not be with the minus sign but I put for consistency
300  su2double DRethetat = 0.0;
301  if (options.LM2015)
302  DRethetat = -c_theta * (Density_i / time_scale) * c_CF * min(ReThetat_SCF - TransVar_i[1], 0.0) * f_theta_2;
303 
304  /*--- Source ---*/
305  Residual[0] += (Pg - Dg) * Volume;
306  Residual[1] += (PRethetat - DRethetat) * Volume;
307 
308  /*--- Implicit part ---*/
309  Jacobian_i[0][0] = (F_length * c_a1 * StrainMag_i * sqrt(F_onset) *
310  (0.5 * pow(TransVar_i[0], -0.5) - 1.5 * c_e1 * pow(TransVar_i[0], 0.5)) -
311  c_a2 * VorticityMag * f_turb * (2.0 * c_e2 * TransVar_i[0] - 1.0)) *
312  Volume;
313  Jacobian_i[0][1] = 0.0;
314  Jacobian_i[1][0] = 0.0;
315  Jacobian_i[1][1] = -c_theta / time_scale * (1.0 - f_theta) * Volume;
316  if (options.LM2015 && ReThetat_SCF - TransVar_i[1] < 0)
317  Jacobian_i[1][1] += (c_theta / time_scale) * c_CF * f_theta_2 * Volume;
318  }
319 
320  AD::SetPreaccOut(Residual, nVar);
321  AD::EndPreacc();
322 
323  return ResidualType<>(Residual, Jacobian_i, nullptr);
324  }
