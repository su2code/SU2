61 template<class ScalarType>
62 void CSysSolve<ScalarType>::ApplyGivens(ScalarType s, ScalarType c, ScalarType & h1, ScalarType & h2) const {
63 
64  ScalarType temp = c*h1 + s*h2;
65  h2 = c*h2 - s*h1;
66  h1 = temp;
67 }

69 template<class ScalarType>
70 void CSysSolve<ScalarType>::GenerateGivens(ScalarType & dx, ScalarType & dy, ScalarType & s, ScalarType & c) const {
71 
72  if ( (dx == 0.0) && (dy == 0.0) ) {
73  c = 1.0;
74  s = 0.0;
75  }
76  else if ( fabs(dy) > fabs(dx) ) {
77  ScalarType tmp = dx/dy;
78  dx = sqrt(1.0 + tmp*tmp);
79  s = Sign(1.0/dx, dy);
80  c = tmp*s;
81  }
82  else if ( fabs(dy) <= fabs(dx) ) {
83  ScalarType tmp = dy/dx;
84  dy = sqrt(1.0 + tmp*tmp);
85  c = Sign(1.0/dy, dx);
86  s = tmp*c;
87  }
88  else {
89  // dx and/or dy must be invalid
90  dx = 0.0;
91  dy = 0.0;
92  c = 1.0;
93  s = 0.0;
94  }
95  dx = fabs(dx*dy);
96  dy = 0.0;
97 }

99 template<class ScalarType>
100 void CSysSolve<ScalarType>::SolveReduced(int n, const su2matrix<ScalarType>& Hsbg,
101  const su2vector<ScalarType>& rhs, su2vector<ScalarType>& x) const {
102  // initialize...
103  for (int i = 0; i < n; i++)
104  x[i] = rhs[i];
105  // ... and backsolve
106  for (int i = n-1; i >= 0; i--) {
107  x[i] /= Hsbg(i,i);
108  for (int j = i-1; j >= 0; j--) {
109  x[j] -= Hsbg(j,i)*x[i];
110  }
111  }
112 }

114 template<class ScalarType>
115 void CSysSolve<ScalarType>::ModGramSchmidt(int i, su2matrix<ScalarType>& Hsbg,
116  vector<CSysVector<ScalarType> >& w) const {
117 
118  /*--- Parameter for reorthonormalization ---*/
119 
120  const ScalarType reorth = 0.98;
121 
122  /*--- Get the norm of the vector being orthogonalized, and find the
123  threshold for re-orthogonalization ---*/
124 
125  ScalarType nrm = w[i+1].squaredNorm();
126  ScalarType thr = nrm*reorth;
127 
128  /*--- The norm of w[i+1] < 0.0 or w[i+1] = NaN ---*/
129 
130  if ((nrm <= 0.0) || (nrm != nrm)) {
131  /*--- nrm is the result of a dot product, communications are implicitly handled. ---*/
132  SU2_MPI::Error("FGMRES orthogonalization failed, linear solver diverged.", CURRENT_FUNCTION);
133  }
134 
135  /*--- Begin main Gram-Schmidt loop ---*/
136 
137  for (int k = 0; k < i+1; k++) {
138  ScalarType prod = w[i+1].dot(w[k]);
139  Hsbg(k,i) = prod;
140  w[i+1] -= prod * w[k];
141 
142  /*--- Check if reorthogonalization is necessary ---*/
143 
144  if (prod*prod > thr) {
145  prod = w[i+1].dot(w[k]);
146  Hsbg(k,i) += prod;
147  w[i+1] -= prod * w[k];
148  }
149 
150  /*--- Update the norm and check its size ---*/
151 
152  nrm -= pow(Hsbg(k,i),2);
153  nrm = max<ScalarType>(nrm, 0.0);
154  thr = nrm*reorth;
155  }
156 
157  /*--- Test the resulting vector ---*/
158 
159  nrm = w[i+1].norm();
160  Hsbg(i+1,i) = nrm;
161 
162  /*--- Scale the resulting vector ---*/
163 
164  w[i+1] /= nrm;
165 
166 }

198 template<class ScalarType>
199 unsigned long CSysSolve<ScalarType>::CG_LinSolver(const CSysVector<ScalarType> & b, CSysVector<ScalarType> & x,
200  const CMatrixVectorProduct<ScalarType> & mat_vec, const CPreconditioner<ScalarType> & precond,
201  ScalarType tol, unsigned long m, ScalarType & residual, bool monitoring, const CConfig *config) const {
202 
203  const bool master = (SU2_MPI::GetRank() == MASTER_NODE) && (omp_get_thread_num() == 0);
204  ScalarType norm_r = 0.0, norm0 = 0.0;
205  unsigned long i = 0;
206 
207  /*--- Check the subspace size ---*/
208 
209  if (m < 1) {
210  SU2_MPI::Error("Number of linear solver iterations must be greater than 0.", CURRENT_FUNCTION);
211  }
212 
213  /*--- Allocate if not allocated yet, only one thread can
214  * do this since the working vectors are shared. ---*/
215 
216  if (!cg_ready) {
217  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
218  {
219  auto nVar = b.GetNVar();
220  auto nBlk = b.GetNBlk();
221  auto nBlkDomain = b.GetNBlkDomain();
222 
223  A_x.Initialize(nBlk, nBlkDomain, nVar, nullptr);
224  r.Initialize(nBlk, nBlkDomain, nVar, nullptr);
225  z.Initialize(nBlk, nBlkDomain, nVar, nullptr);
226  p.Initialize(nBlk, nBlkDomain, nVar, nullptr);
227 
228  cg_ready = true;
229  }
230  END_SU2_OMP_SAFE_GLOBAL_ACCESS
231  }
232 
233  /*--- Calculate the initial residual, compute norm, and check if system is already solved ---*/
234 
235  if (!xIsZero) {
236  mat_vec(x, A_x);
237  r = b - A_x;
238  } else {
239  r = b;
240  }
241 
242  /*--- Only compute the residuals in full communication mode. ---*/
243 
244  if (config->GetComm_Level() == COMM_FULL) {
245 
246  norm_r = r.norm();
247  norm0 = b.norm();
248 
249  /*--- Set the norm to the initial initial residual value ---*/
250 
251  if (tol_type == LinearToleranceType::RELATIVE) norm0 = norm_r;
252 
253  if ((norm_r < tol*norm0) || (norm_r < eps)) {
254  if (master && (lin_sol_mode!=LINEAR_SOLVER_MODE::MESH_DEFORM)) cout << "CSysSolve::ConjugateGradient(): system solved by initial guess." << endl;
255  return 0;
256  }
257 
258  /*--- Output header information including initial residual ---*/
259 
260  if (monitoring && master) {
261  WriteHeader("CG", tol, norm_r);
262  WriteHistory(i, norm_r/norm0);
263  }
264 
265  }
266 
267  precond(r, z);
268  p = z;
269  ScalarType r_dot_z = r.dot(z);
270 
271  /*--- Loop over all search directions ---*/
272 
273  for (i = 0; i < m; i++) {
274 
275  /*--- Apply matrix to p to build Krylov subspace ---*/
276 
277  mat_vec(p, A_x);
278 
279  /*--- Calculate step-length alpha ---*/
280 
281  ScalarType alpha = r_dot_z / A_x.dot(p);
282 
283  /*--- Update solution and residual: ---*/
284 
285  x += alpha * p;
286  r -= alpha * A_x;
287 
288  /*--- Only compute the residuals in full communication mode. ---*/
289 
290  if (config->GetComm_Level() == COMM_FULL) {
291 
292  /*--- Check if solution has converged, else output the relative residual if necessary ---*/
293 
294  norm_r = r.norm();
295  if (norm_r < tol*norm0) break;
296  if (((monitoring) && (master)) && ((i+1) % monitorFreq == 0))
297  WriteHistory(i+1, norm_r/norm0);
298 
299  }
300 
301  precond(r, z);
302 
303  /*--- Calculate Gram-Schmidt coefficient, beta = (r_{i+1}, z_{i+1}) / (r_{i}, z_{i}) ---*/
304 
305  ScalarType beta = r_dot_z;
306  r_dot_z = r.dot(z);
307  beta = r_dot_z / beta;
308 
309  /*--- Gram-Schmidt orthogonalization. ---*/
310 
311  p = beta*p + z;
312 
313  }
314 
315  /*--- Recalculate final residual (this should be optional) ---*/
316 
317  if ((monitoring) && (config->GetComm_Level() == COMM_FULL)) {
318 
319  if (master) WriteFinalResidual("CG", i, norm_r/norm0);
320 
321  if (recomputeRes) {
322  mat_vec(x, A_x);
323  r = b - A_x;
324  ScalarType true_res = r.norm();
325 
326  if (fabs(true_res - norm_r) > tol*10.0) {
327  if (master) {
328  WriteWarning(norm_r, true_res, tol);
329  }
330  }
331  }
332  }
333 
334  residual = norm_r/norm0;
335  return i;
336 
337 }

339 template<class ScalarType>
340 unsigned long CSysSolve<ScalarType>::FGMRES_LinSolver(const CSysVector<ScalarType> & b, CSysVector<ScalarType> & x,
341  const CMatrixVectorProduct<ScalarType> & mat_vec, const CPreconditioner<ScalarType> & precond,
342  ScalarType tol, unsigned long m, ScalarType & residual, bool monitoring, const CConfig *config) const {
343 
344  const bool master = (SU2_MPI::GetRank() == MASTER_NODE) && (omp_get_thread_num() == 0);
345  const bool flexible = !precond.IsIdentity();
346 
347  /*--- Check the subspace size ---*/
348 
349  if (m < 1) {
350  SU2_MPI::Error("Number of linear solver iterations must be greater than 0.", CURRENT_FUNCTION);
351  }
352 
353  if (m > 5000) {
354  SU2_MPI::Error("FGMRES subspace is too large.", CURRENT_FUNCTION);
355  }
356 
357  /*--- Allocate if not allocated yet ---*/
358 
359  if (W.size() <= m || (flexible && Z.size() <= m)) {
360  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
361  {
362  W.resize(m+1);
363  for (auto& w : W) w.Initialize(x.GetNBlk(), x.GetNBlkDomain(), x.GetNVar(), nullptr);
364  if (flexible) {
365  Z.resize(m+1);
366  for (auto& z : Z) z.Initialize(x.GetNBlk(), x.GetNBlkDomain(), x.GetNVar(), nullptr);
367  }
368  }
369  END_SU2_OMP_SAFE_GLOBAL_ACCESS
370  }
371 
372  /*--- Define various arrays. In parallel, each thread of each rank has and works
373  on its own thread, since calculations on these arrays are based on dot products
374  (reduced across all threads and ranks) all threads do the same computations. ---*/
375 
376  su2vector<ScalarType> g(m+1), sn(m+1), cs(m+1), y(m);
377  g = ScalarType(0);
378  sn = ScalarType(0);
379  cs = ScalarType(0);
380  y = ScalarType(0);
381  su2matrix<ScalarType> H(m+1, m);
382  H = ScalarType(0);
383 
384  /*--- Calculate the norm of the rhs vector. ---*/
385 
386  ScalarType norm0 = b.norm();
387 
388  /*--- Calculate the initial residual (actually the negative residual) and compute its norm. ---*/
389 
390  if (!xIsZero) {
391  mat_vec(x, W[0]);
392  W[0] -= b;
393  }
394  else {
395  W[0] = -b;
396  }
397 
398  ScalarType beta = W[0].norm();
399 
400  /*--- Set the norm to the initial initial residual value ---*/
401 
402  if (tol_type == LinearToleranceType::RELATIVE) norm0 = beta;
403 
404  if ((beta < tol*norm0) || (beta < eps)) {
405 
406  /*--- System is already solved ---*/
407 
408  if (master) cout << "CSysSolve::FGMRES(): system solved by initial guess." << endl;
409  residual = beta;
410  return 0;
411  }
412 
413  /*--- Normalize residual to get w_{0} (the negative sign is because w[0]
414  holds the negative residual, as mentioned above). ---*/
415 
416  W[0] /= -beta;
417 
418  /*--- Initialize the RHS of the reduced system ---*/
419 
420  g[0] = beta;
421 
422  /*--- Output header information including initial residual ---*/
423 
424  unsigned long i = 0;
425  if ((monitoring) && (master)) {
426  WriteHeader("FGMRES", tol, beta);
427  WriteHistory(i, beta/norm0);
428  }
429 
430  /*--- Loop over all search directions ---*/
431 
432  for (i = 0; i < m; i++) {
433 
434  /*--- Check if solution has converged ---*/
435 
436  if (beta < tol*norm0) break;
437 
438  if (flexible) {
439  /*--- Precondition the CSysVector w[i] and store result in z[i] ---*/
440 
441  precond(W[i], Z[i]);
442 
443  /*--- Add to Krylov subspace ---*/
444 
445  mat_vec(Z[i], W[i+1]);
446  }
447  else {
448  mat_vec(W[i], W[i+1]);
449  }
450 
451  /*--- Modified Gram-Schmidt orthogonalization ---*/
452 
453  ModGramSchmidt(i, H, W);
454 
455  /*--- Apply old Givens rotations to new column of the Hessenberg matrix then generate the
456  new Givens rotation matrix and apply it to the last two elements of H[:][i] and g ---*/
457 
458  for (unsigned long k = 0; k < i; k++)
459  ApplyGivens(sn[k], cs[k], H[k][i], H[k+1][i]);
460  GenerateGivens(H[i][i], H[i+1][i], sn[i], cs[i]);
461  ApplyGivens(sn[i], cs[i], g[i], g[i+1]);
462 
463  /*--- Set L2 norm of residual and check if solution has converged ---*/
464 
465  beta = fabs(g[i+1]);
466 
467  /*--- Output the relative residual if necessary ---*/
468 
469  if ((((monitoring) && (master)) && ((i+1) % monitorFreq == 0)) && (master))
470  WriteHistory(i+1, beta/norm0);
471  }
472 
473  /*--- Solve the least-squares system and update solution ---*/
474 
475  SolveReduced(i, H, g, y);
476 
477  const auto& basis = flexible? Z : W;
478 
479  for (unsigned long k = 0; k < i; k++) {
480  x += y[k] * basis[k];
481  }
482 
483  /*--- Recalculate final (neg.) residual (this should be optional) ---*/
484 
485  if ((monitoring) && (config->GetComm_Level() == COMM_FULL)) {
486 
487  if (master) WriteFinalResidual("FGMRES", i, beta/norm0);
488 
489  if (recomputeRes) {
490  mat_vec(x, W[0]);
491  W[0] -= b;
492  ScalarType res = W[0].norm();
493 
494  if (fabs(res - beta) > tol*10) {
495  if (master) {
496  WriteWarning(beta, res, tol);
497  }
498  }
499  }
500  }
501 
502  residual = beta/norm0;
503  return i;
504 
505 }

507 template<class ScalarType>
508 unsigned long CSysSolve<ScalarType>::RFGMRES_LinSolver(const CSysVector<ScalarType> & b, CSysVector<ScalarType> & x,
509  const CMatrixVectorProduct<ScalarType> & mat_vec, const CPreconditioner<ScalarType> & precond,
510  ScalarType tol, unsigned long MaxIter, ScalarType & residual, bool monitoring, const CConfig *config) {
511 
512  const auto restartIter = config->GetLinear_Solver_Restart_Frequency();
513 
514  SU2_OMP_MASTER {
515  xIsZero = false;
516  tol_type = LinearToleranceType::ABSOLUTE;
517  }
518  END_SU2_OMP_MASTER
519 
520  const ScalarType norm0 = b.norm(); // <- Has a barrier
521 
522  for (auto totalIter = 0ul; totalIter < MaxIter;) {
523  /*--- Enforce a hard limit on total number of iterations ---*/
524  auto iterLimit = min(restartIter, MaxIter-totalIter);
525  auto iter = FGMRES_LinSolver(b, x, mat_vec, precond, tol, iterLimit, residual, monitoring, config);
526  totalIter += iter;
527  if (residual <= tol*norm0 || iter < iterLimit) return totalIter;
528  }
529  return 0;
530 }

532 template<class ScalarType>
533 unsigned long CSysSolve<ScalarType>::BCGSTAB_LinSolver(const CSysVector<ScalarType> & b, CSysVector<ScalarType> & x,
534  const CMatrixVectorProduct<ScalarType> & mat_vec, const CPreconditioner<ScalarType> & precond,
535  ScalarType tol, unsigned long m, ScalarType & residual, bool monitoring, const CConfig *config) const {
536 
537  const bool master = (SU2_MPI::GetRank() == MASTER_NODE) && (omp_get_thread_num() == 0);
538  ScalarType norm_r = 0.0, norm0 = 0.0;
539  unsigned long i = 0;
540 
541  /*--- Check the subspace size ---*/
542 
543  if (m < 1) {
544  SU2_MPI::Error("Number of linear solver iterations must be greater than 0.", CURRENT_FUNCTION);
545  }
546 
547  /*--- Allocate if not allocated yet ---*/
548 
549  if (!bcg_ready) {
550  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
551  {
552  auto nVar = b.GetNVar();
553  auto nBlk = b.GetNBlk();
554  auto nBlkDomain = b.GetNBlkDomain();
555 
556  A_x.Initialize(nBlk, nBlkDomain, nVar, nullptr);
557  r_0.Initialize(nBlk, nBlkDomain, nVar, nullptr);
558  r.Initialize(nBlk, nBlkDomain, nVar, nullptr);
559  p.Initialize(nBlk, nBlkDomain, nVar, nullptr);
560  v.Initialize(nBlk, nBlkDomain, nVar, nullptr);
561  z.Initialize(nBlk, nBlkDomain, nVar, nullptr);
562 
563  bcg_ready = true;
564  }
565  END_SU2_OMP_SAFE_GLOBAL_ACCESS
566  }
567 
568  /*--- Calculate the initial residual, compute norm, and check if system is already solved ---*/
569 
570  if (!xIsZero) {
571  mat_vec(x, A_x);
572  r = b - A_x;
573  } else {
574  r = b;
575  }
576 
577  /*--- Only compute the residuals in full communication mode. ---*/
578 
579  if (config->GetComm_Level() == COMM_FULL) {
580 
581  norm_r = r.norm();
582  norm0 = b.norm();
583 
584  /*--- Set the norm to the initial initial residual value ---*/
585 
586  if (tol_type == LinearToleranceType::RELATIVE) norm0 = norm_r;
587 
588  if ((norm_r < tol*norm0) || (norm_r < eps)) {
589  if (master) cout << "CSysSolve::BCGSTAB(): system solved by initial guess." << endl;
590  return 0;
591  }
592 
593  /*--- Output header information including initial residual ---*/
594 
595  if ((monitoring) && (master)) {
596  WriteHeader("BCGSTAB", tol, norm_r);
597  WriteHistory(i, norm_r/norm0);
598  }
599 
600  }
601 
602  /*--- Initialization ---*/
603 
604  ScalarType alpha = 1.0, omega = 1.0, rho = 1.0, rho_prime = 1.0;
605  p = ScalarType(0.0); v = ScalarType(0.0); r_0 = r;
606 
607  /*--- Loop over all search directions ---*/
608 
609  for (i = 0; i < m; i++) {
610 
611  /*--- Compute rho_prime ---*/
612 
613  rho_prime = rho;
614 
615  /*--- Compute rho_i ---*/
616 
617  rho = r.dot(r_0);
618 
619  /*--- Compute beta ---*/
620 
621  ScalarType beta = (rho / rho_prime) * (alpha /omega);
622 
623  /*--- Update p ---*/
624 
625  p = beta * (p - omega*v) + r;
626 
627  /*--- Preconditioning step ---*/
628 
629  precond(p, z);
630  mat_vec(z, v);
631 
632  /*--- Calculate step-length alpha ---*/
633 
634  ScalarType r_0_v = r_0.dot(v);
635  alpha = rho / r_0_v;
636 
637  /*--- Update solution and residual ---*/
638 
639  x += alpha * z;
640  r -= alpha * v;
641 
642  /*--- Preconditioning step ---*/
643 
644  precond(r, z);
645  mat_vec(z, A_x);
646 
647  /*--- Calculate step-length omega, avoid division by 0. ---*/
648 
649  omega = A_x.squaredNorm();
650  if (omega == ScalarType(0)) break;
651  omega = A_x.dot(r) / omega;
652 
653  /*--- Update solution and residual ---*/
654 
655  x += omega * z;
656  r -= omega * A_x;
657 
658  /*--- Only compute the residuals in full communication mode. ---*/
659 
660  if (config->GetComm_Level() == COMM_FULL) {
661 
662  /*--- Check if solution has converged, else output the relative residual if necessary ---*/
663 
664  norm_r = r.norm();
665  if (norm_r < tol*norm0) break;
666  if (((monitoring) && (master)) && ((i+1) % monitorFreq == 0) && (master))
667  WriteHistory(i+1, norm_r/norm0);
668 
669  }
670 
671  }

674 
675  if ((monitoring) && (config->GetComm_Level() == COMM_FULL)) {
676 
677  if (master) WriteFinalResidual("BCGSTAB", i, norm_r/norm0);
678 
679  if (recomputeRes) {
680  mat_vec(x, A_x);
681  r = b - A_x;
682  ScalarType true_res = r.norm();
683 
684  if ((fabs(true_res - norm_r) > tol*10.0) && (master)) {
685  WriteWarning(norm_r, true_res, tol);
686  }
687  }
688  }
689

694 template<class ScalarType>
695 unsigned long CSysSolve<ScalarType>::Smoother_LinSolver(const CSysVector<ScalarType> & b, CSysVector<ScalarType> & x,
696  const CMatrixVectorProduct<ScalarType> & mat_vec, const CPreconditioner<ScalarType> & precond,
697  ScalarType tol, unsigned long m, ScalarType & residual, bool monitoring, const CConfig *config) const {
698 
699  const bool master = (SU2_MPI::GetRank() == MASTER_NODE) && (omp_get_thread_num() == 0);
700  const bool fix_iter_mode = tol < eps;
701  ScalarType norm_r = 0.0, norm0 = 0.0;
702  unsigned long i = 0;
703 
704  /*--- Relaxation factor, see comments inside the loop over the smoothing iterations. ---*/
705  const ScalarType omega = SU2_TYPE::GetValue(config->GetLinear_Solver_Smoother_Relaxation());
706 
707  if (m < 1) {
708  SU2_MPI::Error("Number of linear solver iterations must be greater than 0.", CURRENT_FUNCTION);
709  }
710 
711  /*--- Allocate vectors for residual (r), solution increment (z), and matrix-vector
712  product (A_x), for the latter two this is done only on the first call to the method. ---*/
713 
714  if (!smooth_ready) {
715  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
716  {
717  auto nVar = b.GetNVar();
718  auto nBlk = b.GetNBlk();
719  auto nBlkDomain = b.GetNBlkDomain();
720 
721  A_x.Initialize(nBlk, nBlkDomain, nVar, nullptr);
722  r.Initialize(nBlk, nBlkDomain, nVar, nullptr);
723  z.Initialize(nBlk, nBlkDomain, nVar, nullptr);
724 
725  smooth_ready = true;
726  }
727  END_SU2_OMP_SAFE_GLOBAL_ACCESS
728  }
729 
730  /*--- Compute the initial residual and check if the system is already solved (if in COMM_FULL mode). ---*/
731 
732  if (!xIsZero) {
733  mat_vec(x, A_x);
734  r = b - A_x;
735  } else {
736  r = b;
737  }
738 
739  /*--- Only compute the residuals in full communication mode. ---*/
740 
741  if (config->GetComm_Level() == COMM_FULL) {
742 
743  norm_r = r.norm();
744  norm0 = b.norm();
745 
746  /*--- Set the norm to the initial initial residual value ---*/
747 
748  if (tol_type == LinearToleranceType::RELATIVE) norm0 = norm_r;
749 
750  if ( (norm_r < tol*norm0) || (norm_r < eps) ) {
751  if (master) cout << "CSysSolve::Smoother_LinSolver(): system solved by initial guess." << endl;
752  return 0;
753  }
754 
755  /*--- Output header information including initial residual. ---*/
756 
757  if ((monitoring) && (master)) {
758  WriteHeader("Smoother", tol, norm_r);
759  WriteHistory(i, norm_r/norm0);
760  }
761 
762  }
763 
764  /*--- Smoothing Iterations ---*/
765 
766  for (i=0; i<m; i++) {
767 
768  /*--- Compute the solution increment (z), or "search" direction, by applying
769  the preconditioner to the residual, i.e. z = M^{-1} * r. ---*/
770 
771  precond(r, z);
772 
773  /*--- The increment will be added to the solution (with relaxation omega)
774  to update the residual, needed to compute the next increment, we get the
775  product of the matrix with the direction (into A_x) and subtract from the
776  current residual, the system is linear so this saves some computation
777  compared to re-evaluating r = b-A*x. ---*/
778 
779  mat_vec(z, A_x);
780 
781  /*--- Update solution and residual with relaxation omega. Mathematically this
782  is a modified Richardson iteration for the left-preconditioned system
783  M^{-1}(b-A*x) which converges if ||I-w*M^{-1}*A|| < 1. Combining this method
784  with a Gauss-Seidel preconditioner and w>1 is NOT equivalent to SOR. ---*/
785 
786  x += omega * z;
787  r -= omega * A_x;
788 
789  /*--- Only compute the residuals in full communication mode. ---*/
790  /*--- Check if solution has converged, else output the relative residual if necessary. ---*/
791 
792  if (!fix_iter_mode && config->GetComm_Level() == COMM_FULL) {
793  norm_r = r.norm();
794  if (norm_r < tol*norm0) break;
795  if (((monitoring) && (master)) && ((i+1) % monitorFreq == 0))
796  WriteHistory(i+1, norm_r/norm0);
797  }
798  }
799 
800  if (fix_iter_mode) norm_r = r.norm();
801 
802  if ((monitoring) && (master) && (config->GetComm_Level() == COMM_FULL)) {
803  WriteFinalResidual("Smoother", i, norm_r/norm0);
804  }
805 
806  residual = norm_r/norm0;
807  return i;
808 }

810 template<class ScalarType>
811 unsigned long CSysSolve<ScalarType>::Solve(CSysMatrix<ScalarType> & Jacobian, const CSysVector<su2double> & LinSysRes,
812  CSysVector<su2double> & LinSysSol, CGeometry *geometry, const CConfig *config) {
813  /*---
814  A word about the templated types. It is assumed that the residual and solution vectors are always of su2doubles,
815  meaning that they are active in the discrete adjoint. The same assumption is made in SetExternalSolve.
816  When the Jacobian is passive (and therefore not compatible with the vectors) we go through the "HandleTemporaries"
817  mechanisms. Note that CG, BCGSTAB, and FGMRES, all expect the vector to be compatible with the Product and
818  Preconditioner (and therefore with the Matrix). Likewise for Solve_b (which is used by CSysSolve_b).
819  There are no provisions here for active Matrix and passive Vectors as that makes no sense since we only handle the
820  derivatives of the residual in CSysSolve_b.
821  ---*/
822 
823  unsigned short KindSolver, KindPrecond;
824  unsigned long MaxIter;
825  ScalarType SolverTol;
826  bool ScreenOutput;
827 
828  switch (lin_sol_mode) {
829  /*--- Mesh Deformation mode ---*/
830  case LINEAR_SOLVER_MODE::MESH_DEFORM: {
831  KindSolver = config->GetKind_Deform_Linear_Solver();
832  KindPrecond = config->GetKind_Deform_Linear_Solver_Prec();
833  MaxIter = config->GetDeform_Linear_Solver_Iter();
834  SolverTol = SU2_TYPE::GetValue(config->GetDeform_Linear_Solver_Error());
835  ScreenOutput = config->GetDeform_Output();
836  break;
837  }
838 
839  /*--- Gradient Smoothing mode ---*/
840  case LINEAR_SOLVER_MODE::GRADIENT_MODE: {
841  KindSolver = config->GetKind_Grad_Linear_Solver();
842  KindPrecond = config->GetKind_Grad_Linear_Solver_Prec();
843  MaxIter = config->GetGrad_Linear_Solver_Iter();
844  SolverTol = SU2_TYPE::GetValue(config->GetGrad_Linear_Solver_Error());
845  ScreenOutput = true;
846  break;
847  }
848 
849  /*--- Normal mode
850  * assumes that 'lin_sol_mode==LINEAR_SOLVER_MODE::STANDARD', but does not enforce it to avoid compiler warning. ---*/
851  default: {
852  KindSolver = config->GetKind_Linear_Solver();
853  KindPrecond = config->GetKind_Linear_Solver_Prec();
854  MaxIter = config->GetLinear_Solver_Iter();
855  SolverTol = SU2_TYPE::GetValue(config->GetLinear_Solver_Error());
856  ScreenOutput = false;
857  break;
858  }
859  }
860 
861  /*--- Stop the recording for the linear solver ---*/
862  bool TapeActive = NO;
863 
864  if (config->GetDiscrete_Adjoint()) {
865 #ifdef CODI_REVERSE_TYPE
866 
867  TapeActive = AD::getGlobalTape().isActive();
868 
869  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
870  AD::StartExtFunc(false, false);
871  AD::SetExtFuncIn(&LinSysRes[0], LinSysRes.GetLocSize());
872  }
873  END_SU2_OMP_SAFE_GLOBAL_ACCESS
874 
875  AD::StopRecording();
876 #endif
877  }
878 
879  /*--- Create matrix-vector product, preconditioner, and solve the linear system ---*/
880 
881  HandleTemporariesIn(LinSysRes, LinSysSol);
882 
883  auto mat_vec = CSysMatrixVectorProduct<ScalarType>(Jacobian, geometry, config);
884 
885  const auto kindPrec = static_cast<ENUM_LINEAR_SOLVER_PREC>(KindPrecond);
886 
887  auto precond = CPreconditioner<ScalarType>::Create(kindPrec, Jacobian, geometry, config);
888 
889  /*--- Build preconditioner. ---*/
890 
891  precond->Build();
892 
893  /*--- Solve system. ---*/
894 
895  unsigned long IterLinSol = 0;
896  ScalarType residual = 0.0;
897 
898  switch (KindSolver) {
899  case BCGSTAB:
900  IterLinSol = BCGSTAB_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual, ScreenOutput, config);
901  break;
902  case FGMRES:
903  IterLinSol = FGMRES_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual, ScreenOutput, config);
904  break;
905  case RESTARTED_FGMRES:
906  IterLinSol = RFGMRES_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual, ScreenOutput, config);
907  break;
908  case CONJUGATE_GRADIENT:
909  IterLinSol = CG_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual, ScreenOutput, config);
910  break;
911  case SMOOTHER:
912  IterLinSol = Smoother_LinSolver(*LinSysRes_ptr, *LinSysSol_ptr, mat_vec, *precond, SolverTol, MaxIter, residual, ScreenOutput, config);
913  break;
914  case PASTIX_LDLT : case PASTIX_LU:
915  Jacobian.BuildPastixPreconditioner(geometry, config, KindSolver);
916  Jacobian.ComputePastixPreconditioner(*LinSysRes_ptr, *LinSysSol_ptr, geometry, config);
917  IterLinSol = 1;
918  residual = 1e-20;
919  break;
920  default:
921  SU2_MPI::Error("Unknown type of linear solver.",CURRENT_FUNCTION);
922  }
923 
924  SU2_OMP_MASTER
925  {
926  Residual = residual;
927  Iterations = IterLinSol;
928  }
929  END_SU2_OMP_MASTER
930 
931  HandleTemporariesOut(LinSysSol);
932 
933  delete precond;
934 
935  if(TapeActive) {
936 
937  /*--- To keep the behavior of SU2_DOT, but not strictly required since jacobian is symmetric(?). ---*/
938  const bool RequiresTranspose = ((lin_sol_mode!=LINEAR_SOLVER_MODE::MESH_DEFORM) || (config->GetKind_SU2() == SU2_COMPONENT::SU2_DOT));
939 
940  if (lin_sol_mode==LINEAR_SOLVER_MODE::MESH_DEFORM) KindPrecond = config->GetKind_Deform_Linear_Solver_Prec();
941  else if (lin_sol_mode==LINEAR_SOLVER_MODE::GRADIENT_MODE) KindPrecond = config->GetKind_Grad_Linear_Solver_Prec();
942  else KindPrecond = config->GetKind_DiscAdj_Linear_Prec();
943 
944  /*--- Build preconditioner for the transposed Jacobian ---*/
945 
946  if (RequiresTranspose) Jacobian.TransposeInPlace();
947 
948  switch(KindPrecond) {
949  case ILU:
950  if (RequiresTranspose) Jacobian.BuildILUPreconditioner();
951  break;
952  case JACOBI:
953  case LINELET:
954  if (RequiresTranspose) Jacobian.BuildJacobiPreconditioner();
955  break;
956  case LU_SGS:
957  /*--- Nothing to build. ---*/
958  break;
959  case PASTIX_ILU: case PASTIX_LU_P: case PASTIX_LDLT_P:
960  /*--- It was already built. ---*/
961  break;
962  default:
963  SU2_MPI::Error("The specified preconditioner is not yet implemented for the discrete adjoint method.", CURRENT_FUNCTION);
964  break;
965  }
966 
967  /*--- Start recording if it was stopped for the linear solver ---*/
968 #ifdef CODI_REVERSE_TYPE
969  AD::StartRecording();
970 
971  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
972  {
973  AD::SetExtFuncOut(&LinSysSol[0], LinSysSol.GetLocSize());
974  AD::FuncHelper->addUserData(&LinSysRes);
975  AD::FuncHelper->addUserData(&LinSysSol);
976  AD::FuncHelper->addUserData(&Jacobian);
977  AD::FuncHelper->addUserData(geometry);
978  AD::FuncHelper->addUserData(config);
979  AD::FuncHelper->addUserData(this);
980  }
981  END_SU2_OMP_SAFE_GLOBAL_ACCESS
982 
983  AD::FuncHelper->addToTape(CSysSolve_b<ScalarType>::Solve_b);
984 
985  SU2_OMP_SAFE_GLOBAL_ACCESS(AD::EndExtFunc();)
986 #endif
987  }
988 
989  return IterLinSol;
990 }
