68  CSolver***** solver, CNumerics****** numerics, CConfig** config,
69  CSurfaceMovement** surface_movement, CVolumetricMovement*** grid_movement,
70  CFreeFormDefBox*** FFDBox, unsigned short iZone, unsigned short iInst) {
71  unsigned long iPoint;
72  auto solvers0 = solver[iZone][iInst][MESH_0];
73  auto geometry0 = geometry[iZone][iInst][MESH_0];
74  auto dirNodes = solvers0[FEA_SOL]->GetNodes();
75  auto adjNodes = solvers0[ADJFEA_SOL]->GetNodes();
76 
77  /*--- For the dynamic adjoint, load direct solutions from restart files. ---*/
78 
79  if (config[iZone]->GetTime_Domain()) {
80  const int TimeIter = config[iZone]->GetTimeIter();
81  const int Direct_Iter = SU2_TYPE::Int(config[iZone]->GetUnst_AdjointIter()) - TimeIter - 1;
82 
83  /*--- We want to load the already converged solution at timesteps n and n-1 ---*/
84 
85  /*--- Load solution at timestep n-1 ---*/
86 
87  LoadDynamic_Solution(geometry, solver, config, iZone, iInst, Direct_Iter - 1);
88 
89  /*--- Push solution back to correct array ---*/
90 
91  dirNodes->Set_Solution_time_n();
92 
93  /*--- Load solution timestep n ---*/
94 
95  LoadDynamic_Solution(geometry, solver, config, iZone, iInst, Direct_Iter);
96 
97  /*--- Store FEA solution also in the adjoint solver in order to be able to reset it later ---*/
98 
99  for (iPoint = 0; iPoint < geometry0->GetnPoint(); iPoint++) {
100  adjNodes->SetSolution_Direct(iPoint, dirNodes->GetSolution(iPoint));
101  }
102 
103  } else {
104  /*--- Store FEA solution also in the adjoint solver in order to be able to reset it later ---*/
105 
106  for (iPoint = 0; iPoint < geometry0->GetnPoint(); iPoint++) {
107  adjNodes->SetSolution_Direct(iPoint, dirNodes->GetSolution(iPoint));
108  }
109  }
110 
111  solvers0[ADJFEA_SOL]->Preprocessing(geometry0, solvers0, config[iZone], MESH_0, 0, RUNTIME_ADJFEA_SYS, false);
112

115 void CDiscAdjFEAIteration::LoadDynamic_Solution(CGeometry**** geometry, CSolver***** solver, CConfig** config,
116  unsigned short iZone, unsigned short iInst,
117  int val_DirectIter) {
118  unsigned short iVar;
119  unsigned long iPoint;
120  bool update_geo = false; // TODO: check
121 
122  if (val_DirectIter >= 0) {
123  if (rank == MASTER_NODE && iZone == ZONE_0)
124  cout << " Loading FEA solution from direct iteration " << val_DirectIter << "." << endl;
125  solver[iZone][iInst][MESH_0][FEA_SOL]->LoadRestart(
126  geometry[iZone][iInst], solver[iZone][iInst], config[iZone], val_DirectIter, update_geo);
127  } else {
128  /*--- If there is no solution file we set the freestream condition ---*/
129  if (rank == MASTER_NODE && iZone == ZONE_0)
130  cout << " Setting static conditions at direct iteration " << val_DirectIter << "." << endl;
131  /*--- Push solution back to correct array ---*/
132  for (iPoint = 0; iPoint < geometry[iZone][iInst][MESH_0]->GetnPoint(); iPoint++) {
133  for (iVar = 0; iVar < solver[iZone][iInst][MESH_0][FEA_SOL]->GetnVar(); iVar++) {
134  solver[iZone][iInst][MESH_0][FEA_SOL]->GetNodes()->SetSolution(iPoint, iVar, 0.0);
135  solver[iZone][iInst][MESH_0][FEA_SOL]->GetNodes()->SetSolution_Accel(iPoint, iVar, 0.0);
136  solver[iZone][iInst][MESH_0][FEA_SOL]->GetNodes()->SetSolution_Vel(iPoint, iVar, 0.0);
137  }
138  }
139  }

174 void CDiscAdjFEAIteration::SetDependencies(CSolver***** solver, CGeometry**** geometry, CNumerics****** numerics,
175  CConfig** config, unsigned short iZone, unsigned short iInst,
176  RECORDING kind_recording) {
177  auto dir_solver = solver[iZone][iInst][MESH_0][FEA_SOL];
178  auto adj_solver = solver[iZone][iInst][MESH_0][ADJFEA_SOL];
179  auto structural_geometry = geometry[iZone][iInst][MESH_0];
180  auto structural_numerics = numerics[iZone][iInst][MESH_0][FEA_SOL];
181 
182  /*--- Some numerics are only instanciated under these conditions ---*/
183  const bool fsi = config[iZone]->GetFSI_Simulation() || config[iZone]->GetMultizone_Problem();
184  const bool nonlinear = config[iZone]->GetGeometricConditions() == STRUCT_DEFORMATION::LARGE;
185  const bool de_effects = config[iZone]->GetDE_Effects() && nonlinear;
186  const bool element_based = dir_solver->IsElementBased() && nonlinear;
187 
188  SU2_OMP_PARALLEL
189  {
190 
191  const int thread = omp_get_thread_num();
192  const int offset = thread*MAX_TERMS;
193  const int fea_term = FEA_TERM+offset;
194  const int mat_nhcomp = MAT_NHCOMP+offset;
195  const int mat_idealde = MAT_IDEALDE+offset;
196  const int mat_knowles = MAT_KNOWLES+offset;
197  const int de_term = DE_TERM+offset;
198 
199  for (unsigned short iProp = 0; iProp < config[iZone]->GetnElasticityMod(); iProp++) {
200  su2double E = adj_solver->GetVal_Young(iProp);
201  su2double nu = adj_solver->GetVal_Poisson(iProp);
202  su2double rho = adj_solver->GetVal_Rho(iProp);
203  su2double rhoDL = adj_solver->GetVal_Rho_DL(iProp);
204 
205  /*--- Add dependencies for E and Nu ---*/
206 
207  structural_numerics[fea_term]->SetMaterial_Properties(iProp, E, nu);
208 
209  /*--- Add dependencies for Rho and Rho_DL ---*/
210 
211  structural_numerics[fea_term]->SetMaterial_Density(iProp, rho, rhoDL);
212 
213  /*--- Add dependencies for element-based simulations. ---*/
214 
215  if (element_based) {
216  /*--- Neo Hookean Compressible ---*/
217  structural_numerics[mat_nhcomp]->SetMaterial_Properties(iProp, E, nu);
218  structural_numerics[mat_nhcomp]->SetMaterial_Density(iProp, rho, rhoDL);
219 
220  /*--- Ideal DE ---*/
221  structural_numerics[mat_idealde]->SetMaterial_Properties(iProp, E, nu);
222  structural_numerics[mat_idealde]->SetMaterial_Density(iProp, rho, rhoDL);
223 
224  /*--- Knowles ---*/
225  structural_numerics[mat_knowles]->SetMaterial_Properties(iProp, E, nu);
226  structural_numerics[mat_knowles]->SetMaterial_Density(iProp, rho, rhoDL);
227  }
228  }
229 
230  if (de_effects) {
231  for (unsigned short iEField = 0; iEField < adj_solver->GetnEField(); iEField++) {
232  structural_numerics[fea_term]->Set_ElectricField(iEField, adj_solver->GetVal_EField(iEField));
233  structural_numerics[de_term]->Set_ElectricField(iEField, adj_solver->GetVal_EField(iEField));
234  }
235  }
236 
237  /*--- Add dependencies for element-based simulations. ---*/
238 
239  switch (config[iZone]->GetDV_FEA()) {
240  case YOUNG_MODULUS:
241  case POISSON_RATIO:
242  case DENSITY_VAL:
243  case DEAD_WEIGHT:
244  case ELECTRIC_FIELD:
245 
246  for (unsigned short iDV = 0; iDV < adj_solver->GetnDVFEA(); iDV++) {
247  su2double dvfea = adj_solver->GetVal_DVFEA(iDV);
248 
249  structural_numerics[fea_term]->Set_DV_Val(iDV, dvfea);
250 
251  if (de_effects) structural_numerics[de_term]->Set_DV_Val(iDV, dvfea);
252 
253  if (element_based) {
254  structural_numerics[mat_nhcomp]->Set_DV_Val(iDV, dvfea);
255  structural_numerics[mat_idealde]->Set_DV_Val(iDV, dvfea);
256  structural_numerics[mat_knowles]->Set_DV_Val(iDV, dvfea);
257  }
258  }
259  break;
260  }
261 
262  /*--- MPI dependencies. ---*/
263 
264  dir_solver->InitiateComms(structural_geometry, config[iZone], SOLUTION_FEA);
265  dir_solver->CompleteComms(structural_geometry, config[iZone], SOLUTION_FEA);
266 
267  structural_geometry->InitiateComms(structural_geometry, config[iZone], COORDINATES);
268  structural_geometry->CompleteComms(structural_geometry, config[iZone], COORDINATES);
269 
270  }
271  END_SU2_OMP_PARALLEL
272 
273  /*--- FSI specific dependencies. ---*/
274  if (fsi) {
275  /*--- Set relation between solution and predicted displacements, which are the transferred ones. ---*/
276  dir_solver->PredictStruct_Displacement(structural_geometry, config[iZone]);
277  }
278 
279  /*--- Topology optimization dependencies. ---*/
280 
281  /*--- We only differentiate wrt to this variable in the adjoint secondary recording. ---*/
282  if (config[iZone]->GetTopology_Optimization() && (kind_recording == RECORDING::MESH_COORDS)) {
283  /*--- The filter may require the volumes of the elements. ---*/
284  structural_geometry->SetElemVolume();
285  /// TODO: Ideally there would be a way to capture this dependency without the `static_cast`, but
286  /// making it a virtual method of CSolver does not feel "right" as its purpose could be confused.
287  static_cast<CFEASolver*>(dir_solver)->FilterElementDensities(structural_geometry, config[iZone]);
288  }
289 
290 }
