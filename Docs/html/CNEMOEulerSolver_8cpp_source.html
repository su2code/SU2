83  else Unst_RestartIter = SU2_TYPE::Int(config->GetRestart_Iter())-1;

204  Temperature_Inf, Temperature_ve_Inf,
205  nPoint, nDim, nVar, nPrimVar, nPrimVarGrad,

240 void CNEMOEulerSolver::CommonPreprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh,
241  unsigned short iRKStep, unsigned short RunTime_EqSystem, bool Output) {
242 
243  bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
244  bool center = (config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED);
245  bool center_jst = (config->GetKind_Centered_Flow() == CENTERED::JST) && (iMesh == MESH_0);
246  bool center_jst_ke = (config->GetKind_Centered_Flow() == CENTERED::JST_KE) && (iMesh == MESH_0);
247 
248  /*--- Set the primitive variables ---*/
249  ompMasterAssignBarrier(ErrorCounter,0);
250  SU2_OMP_ATOMIC
251  ErrorCounter += SetPrimitive_Variables(solver_container, config, Output);
252 
253  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
254  { /*--- Ops that are not OpenMP parallel go in this block. ---*/
255 
256  if ((iMesh == MESH_0) && (config->GetComm_Level() == COMM_FULL)) {
257  unsigned long tmp = ErrorCounter;
258  SU2_MPI::Allreduce(&tmp, &ErrorCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
259  config->SetNonphysical_Points(ErrorCounter);
260 
261  if ((rank == MASTER_NODE) && (ErrorCounter != 0))
262  cout << "Warning. The initial solution contains "<< ErrorCounter << " points that are not physical." << endl;
263  }
264  }
265  END_SU2_OMP_SAFE_GLOBAL_ACCESS
266 
267  /*--- Artificial dissipation ---*/
268 
269  if (center && !Output) {
270  SetMax_Eigenvalue(geometry, config);
271  if (center_jst) SetUndivided_Laplacian(geometry, config);
272  if (center_jst || center_jst_ke) SetCentered_Dissipation_Sensor(geometry, config);
273  }
274 
275  /*--- Initialize the Jacobian matrix and residual, not needed for the reducer strategy
276  * as we set blocks (including diagonal ones) and completely overwrite. ---*/
277 
278  if(!ReducerStrategy && !Output) {
279  LinSysRes.SetValZero();
280  if (implicit) Jacobian.SetValZero();
281  }
282 }

284 void CNEMOEulerSolver::Preprocessing(CGeometry *geometry, CSolver **solver_container,
285  CConfig *config, unsigned short iMesh,
286  unsigned short iRKStep,
287  unsigned short RunTime_EqSystem, bool Output) {
288  const unsigned long InnerIter = config->GetInnerIter();
289  const bool muscl = config->GetMUSCL_Flow() && (iMesh == MESH_0);
290  const bool limiter = (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE) && (InnerIter <= config->GetLimiterIter());
291  const bool center = config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED;
292  const bool van_albada = config->GetKind_SlopeLimit_Flow() == LIMITER::VAN_ALBADA_EDGE;
293 
294  /*--- Common preprocessing steps ---*/
295  CommonPreprocessing(geometry, solver_container, config, iMesh, iRKStep, RunTime_EqSystem, Output);
296 
297  /*--- Upwind second order reconstruction ---*/
298  if (muscl && !center && !Output) {
299 
300  /*--- Calculate the gradients ---*/
301  if (config->GetKind_Gradient_Method() == GREEN_GAUSS) {
302  SetPrimitive_Gradient_GG(geometry, config, true);
303  }
304  if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) {
305  SetPrimitive_Gradient_LS(geometry, config, true);
306  }
307 
308  /*--- Limiter computation ---*/
309  if (limiter && !van_albada) {
310  SetPrimitive_Limiter(geometry, config);
311  }
312  }

339 void CNEMOEulerSolver::SetTime_Step(CGeometry *geometry, CSolver **solver_container, CConfig *config,
340  unsigned short iMesh, unsigned long Iteration) {
341 
342  /*--- Define an object to compute the speed of sound. ---*/
343  struct SoundSpeed {
344  FORCEINLINE su2double operator() (const CNEMOEulerVariable& nodes, unsigned long iPoint, unsigned long jPoint) const {
345  return 0.5 * (nodes.GetSoundSpeed(iPoint) + nodes.GetSoundSpeed(jPoint));
346  }
347 
348  FORCEINLINE su2double operator() (const CNEMOEulerVariable& nodes, unsigned long iPoint) const {
349  return nodes.GetSoundSpeed(iPoint);
350  }
351 
352  } soundSpeed;
353 
354  /*--- Define an object to compute the viscous eigenvalue. ---*/
355  struct LambdaVisc {
356  FORCEINLINE su2double lambda(su2double lamVisc, su2double eddyVisc, su2double rho, su2double k, su2double cv) const {
357  /*--- Determine the viscous spectral radius and apply it to the control volume ---*/
358  su2double Lambda_1 = (4.0/3.0)*(lamVisc + eddyVisc);
359  return (Lambda_1 + k/cv)/rho;
360  }
361 
362  FORCEINLINE su2double operator() (const CNEMOEulerVariable& nodes, unsigned long iPoint, unsigned long jPoint) const {
363  su2double lamVisc = 0.5*(nodes.GetLaminarViscosity(iPoint) + nodes.GetLaminarViscosity(jPoint));
364  su2double eddyVisc = 0.5*(nodes.GetEddyViscosity(iPoint) + nodes.GetEddyViscosity(jPoint));
365  su2double thermalCond = 0.5*(nodes.GetThermalConductivity(iPoint) + nodes.GetThermalConductivity(jPoint) +
366  nodes.GetThermalConductivity_ve(iPoint) + nodes.GetThermalConductivity_ve(jPoint));
367  su2double density = 0.5*(nodes.GetDensity(iPoint) + nodes.GetDensity(jPoint));
368  su2double cv = 0.5*(nodes.GetRhoCv_tr(iPoint) + nodes.GetRhoCv_ve(iPoint) +
369  nodes.GetRhoCv_tr(jPoint) + nodes.GetRhoCv_ve(jPoint))/ density;
370  return lambda(lamVisc, eddyVisc, density, thermalCond, cv);
371  }
372 
373  FORCEINLINE su2double operator() (const CNEMOEulerVariable& nodes, unsigned long iPoint) const {
374  su2double lamVisc = nodes.GetLaminarViscosity(iPoint);
375  su2double eddyVisc = nodes.GetEddyViscosity(iPoint);
376  su2double thermalCond = nodes.GetThermalConductivity(iPoint) + nodes.GetThermalConductivity_ve(iPoint);
377  su2double density = nodes.GetDensity(iPoint);
378  su2double cv = (nodes.GetRhoCv_tr(iPoint) + nodes.GetRhoCv_ve(iPoint))/ density;
379  return lambda(lamVisc, eddyVisc, density, thermalCond, cv);
380  }
381 
382  } lambdaVisc;
383 
384  /*--- Now instantiate the generic implementation with the two functors above. ---*/
385 
386  SetTime_Step_impl(soundSpeed, lambdaVisc, geometry, solver_container, config, iMesh, Iteration);
387 
388 }

410 void CNEMOEulerSolver::Centered_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
411  CConfig *config, unsigned short iMesh, unsigned short iRKStep) {
412  unsigned long iEdge, iPoint, jPoint;
413 
414  CNumerics* numerics = numerics_container[CONV_TERM];
415 
416  /*--- Set booleans based on config settings ---*/
417  bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
418 
419  for (iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
420 
421  /*--- Points in edge, set normal vectors, and number of neighbors ---*/
422  iPoint = geometry->edges->GetNode(iEdge, 0);
423  jPoint = geometry->edges->GetNode(iEdge, 1);
424  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
425  numerics->SetNeighbor(geometry->nodes->GetnNeighbor(iPoint),
426  geometry->nodes->GetnNeighbor(jPoint));
427 
428  /*--- Pass conservative & primitive variables w/o reconstruction to CNumerics ---*/
429  numerics->SetConservative(nodes->GetSolution(iPoint), nodes->GetSolution(jPoint));
430  numerics->SetPrimitive (nodes->GetPrimitive(iPoint), nodes->GetPrimitive(jPoint));
431 
432  /*--- Pass supplementary information to CNumerics ---*/
433  numerics->SetdPdU ( nodes->GetdPdU(iPoint), nodes->GetdPdU(jPoint));
434  numerics->SetdTdU ( nodes->GetdTdU(iPoint), nodes->GetdTdU(jPoint));
435  numerics->SetdTvedU( nodes->GetdTvedU(iPoint), nodes->GetdTvedU(jPoint));
436  numerics->SetEve ( nodes->GetEve(iPoint), nodes->GetEve(jPoint));
437  numerics->SetCvve ( nodes->GetCvve(iPoint), nodes->GetCvve(jPoint));
438  numerics->SetGamma ( nodes->GetGamma(iPoint), nodes->GetGamma(jPoint));
439 
440  /*--- Set the largest convective eigenvalue ---*/
441  numerics->SetLambda(nodes->GetLambda(iPoint), nodes->GetLambda(jPoint));
442 
443  /*--- Compute residuals, and Jacobians ---*/
444  auto residual = numerics->ComputeResidual(config);
445 
446  /*--- Check residuals/Jacobians --*/
447  bool err = CNumerics::CheckResidualNaNs(implicit, nVar, residual);
448 
449  /*--- Update the residual and Jacobian ---*/
450  if (!err) {
451  LinSysRes.AddBlock(iPoint, residual);
452  LinSysRes.SubtractBlock(jPoint, residual);
453  if (implicit) {
454  Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
455  }
456  }
457  }

460 void CNEMOEulerSolver::Upwind_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
461  CConfig *config, unsigned short iMesh) {
462 
463  /*--- Set booleans based on config settings ---*/
464  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
465  const bool muscl = (config->GetMUSCL_Flow() && (iMesh == MESH_0));
466  const bool limiter = (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE);
467  const bool van_albada = (config->GetKind_SlopeLimit_Flow() == LIMITER::VAN_ALBADA_EDGE);
468 
469  /*--- Non-physical counter. ---*/
470  unsigned long counter_local = 0;
471  SU2_OMP_MASTER
472  ErrorCounter = 0;
473  END_SU2_OMP_MASTER
474 
475  /*--- Pick one numerics object per thread. ---*/
476  CNumerics* numerics = numerics_container[CONV_TERM];
477 
478  /*--- Static arrays for MUSCL reconstructed variables ---*/
479  su2double Primitive_i[MAXNVAR] = {0.0}, Primitive_j[MAXNVAR] = {0.0};
480  su2double Conserved_i[MAXNVAR] = {0.0}, Conserved_j[MAXNVAR] = {0.0};
481  su2double dPdU_i[MAXNVAR] = {0.0}, dPdU_j[MAXNVAR] = {0.0};
482  su2double dTdU_i[MAXNVAR] = {0.0}, dTdU_j[MAXNVAR] = {0.0};
483  su2double dTvedU_i[MAXNVAR] = {0.0}, dTvedU_j[MAXNVAR] = {0.0};
484  su2double Eve_i[MAXNVAR] = {0.0}, Eve_j[MAXNVAR] = {0.0};
485  su2double Cvve_i[MAXNVAR] = {0.0}, Cvve_j[MAXNVAR] = {0.0};
486  su2double Project_Grad_i[MAXNVAR] = {0.0}, Project_Grad_j[MAXNVAR] = {0.0};
487  su2double Gamma_i = 0.0, Gamma_j = 0.0;
488 
489  /*--- Loop over edges and calculate convective fluxes ---*/
490  for(unsigned long iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
491 
492  unsigned short iDim, iVar;
493 
494  /*--- Retrieve node numbers and pass edge normal to CNumerics ---*/
495  auto iPoint = geometry->edges->GetNode(iEdge, 0);
496  auto jPoint = geometry->edges->GetNode(iEdge, 1);
497 
498  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
499 
500  auto Coord_i = geometry->nodes->GetCoord(iPoint);
501  auto Coord_j = geometry->nodes->GetCoord(jPoint);
502 
503  /*--- Get conserved & primitive variables from CVariable ---*/
504  auto U_i = nodes->GetSolution(iPoint); auto U_j = nodes->GetSolution(jPoint);
505  auto V_i = nodes->GetPrimitive(iPoint); auto V_j = nodes->GetPrimitive(jPoint);
506 
507  /*--- Set them with or without high order reconstruction using MUSCL strategy. ---*/
508  if (!muscl) {
509 
510  numerics->SetPrimitive (V_i, V_j);
511  numerics->SetConservative(U_i, U_j);
512  numerics->SetdPdU (nodes->GetdPdU(iPoint), nodes->GetdPdU(jPoint));
513  numerics->SetdTdU (nodes->GetdTdU(iPoint), nodes->GetdTdU(jPoint));
514  numerics->SetdTvedU(nodes->GetdTvedU(iPoint), nodes->GetdTvedU(jPoint));
515  numerics->SetEve (nodes->GetEve(iPoint), nodes->GetEve(jPoint));
516  numerics->SetCvve (nodes->GetCvve(iPoint), nodes->GetCvve(jPoint));
517  numerics->SetGamma (nodes->GetGamma(iPoint), nodes->GetGamma(jPoint));
518 
519  } else {
520 
521  /*--- High order reconstruction using MUSCL strategy ---*/
522  su2double Vector_ij[MAXNDIM] = {0.0};
523  for (iDim = 0; iDim < nDim; iDim++) {
524  Vector_ij[iDim] = 0.5*(Coord_j[iDim] - Coord_i[iDim]);
525  }
526 
527  /*--- Retrieve gradient information ---*/
528  auto Gradient_i = nodes->GetGradient_Reconstruction(iPoint);
529  auto Gradient_j = nodes->GetGradient_Reconstruction(jPoint);
530 
531  /*--- Set and extract limiters ---*/
532  su2double *Limiter_i = nullptr, *Limiter_j = nullptr;
533 
534  if (limiter && !van_albada){
535  Limiter_i = nodes->GetLimiter_Primitive(iPoint);
536  Limiter_j = nodes->GetLimiter_Primitive(jPoint);
537  }
538 
539  su2double lim_i = 2.0;
540  su2double lim_j = 2.0;
541 
542  for (iVar = 0; iVar < nPrimVarGrad; iVar++) {
543  Project_Grad_i[iVar] = 0.0; Project_Grad_j[iVar] = 0.0;
544 
545  for (iDim = 0; iDim < nDim; iDim++) {
546  Project_Grad_i[iVar] += Vector_ij[iDim]*Gradient_i[iVar][iDim];
547  Project_Grad_j[iVar] -= Vector_ij[iDim]*Gradient_j[iVar][iDim];
548  }
549 
550  if (limiter) {
551  if (van_albada) {
552  su2double V_ij = V_j[iVar] - V_i[iVar];
553  su2double va_lim_i = LimiterHelpers<>::vanAlbadaFunction(Project_Grad_i[iVar], V_ij, EPS);
554  su2double va_lim_j = LimiterHelpers<>::vanAlbadaFunction(-Project_Grad_j[iVar], V_ij, EPS);
555  lim_i = min(lim_i, va_lim_i);
556  lim_j = min(lim_j, va_lim_j);
557  } else {
558  lim_i = min(lim_i, Limiter_i[iVar]);
559  lim_j = min(lim_j, Limiter_j[iVar]);
560  }
561  } else {
562  lim_i = lim_j = 1.0;
563  }
564  }
565  su2double lim_ij = min(lim_i, lim_j);
566 
567  for (iVar = 0; iVar < nPrimVarGrad; iVar++) {
568  Primitive_i[iVar] = V_i[iVar] + lim_ij*Project_Grad_i[iVar];
569  Primitive_j[iVar] = V_j[iVar] + lim_ij*Project_Grad_j[iVar];
570  }
571 
572  /*--- Check for non-physical solutions after reconstruction. If found, use the
573  cell-average value of the solution. This is a locally 1st order approximation,
574  which is typically only active during the start-up of a calculation. ---*/
575  bool chk_err_i = CheckNonPhys(Primitive_i);
576  bool chk_err_j = CheckNonPhys(Primitive_j);
577 
578  nodes->SetNon_Physical(iPoint, chk_err_i);
579  nodes->SetNon_Physical(jPoint, chk_err_j);
580 
581  /*--- Get updated state, in case the point recovered after the set. ---*/
582  chk_err_i = nodes->GetNon_Physical(iPoint);
583  chk_err_j = nodes->GetNon_Physical(jPoint);
584 
585  counter_local += chk_err_i + chk_err_j;
586 
587  /*--- Compute Secondary variables in a thermaodynamically consistent way. ---*/
588  if (!chk_err_i) Gamma_i = ComputeConsistentExtrapolation(GetFluidModel(), nSpecies, Primitive_i, dPdU_i, dTdU_i, dTvedU_i, Eve_i, Cvve_i);
589  if (!chk_err_j) Gamma_j = ComputeConsistentExtrapolation(GetFluidModel(), nSpecies, Primitive_j, dPdU_j, dTdU_j, dTvedU_j, Eve_j, Cvve_j);
590 
591  /*--- Recompute Conserved variables if Roe or MSW scheme ---*/
592  if ((config->GetKind_Upwind_Flow() == UPWIND::ROE) || (config->GetKind_Upwind_Flow() == UPWIND::MSW)){
593  if (!chk_err_i) RecomputeConservativeVector(Conserved_i, Primitive_i);
594  if (!chk_err_j) RecomputeConservativeVector(Conserved_j, Primitive_j);
595  }
596 
597  /*--- If non-physical, revert to first order ---*/
598  numerics->SetConservative(chk_err_i ? U_i : Conserved_i, chk_err_j ? U_j : Conserved_j);
599  numerics->SetPrimitive (chk_err_i ? V_i : Primitive_i, chk_err_j ? V_j : Primitive_j);
600  numerics->SetdPdU (chk_err_i ? nodes->GetdPdU (iPoint) : dPdU_i, chk_err_j ? nodes->GetdPdU (jPoint) : dPdU_j);
601  numerics->SetdTdU (chk_err_i ? nodes->GetdTdU (iPoint) : dTdU_i, chk_err_j ? nodes->GetdTdU (jPoint) : dTdU_j);
602  numerics->SetdTvedU(chk_err_i ? nodes->GetdTvedU(iPoint) : dTvedU_i, chk_err_j ? nodes->GetdTvedU(jPoint) : dTvedU_j);
603  numerics->SetEve (chk_err_i ? nodes->GetEve (iPoint) : Eve_i, chk_err_j ? nodes->GetEve (jPoint) : Eve_j);
604  numerics->SetCvve (chk_err_i ? nodes->GetCvve (iPoint) : Cvve_i, chk_err_j ? nodes->GetCvve (jPoint) : Cvve_j);
605  numerics->SetGamma (chk_err_i ? nodes->GetGamma (iPoint) : Gamma_i, chk_err_j ? nodes->GetGamma (jPoint) : Gamma_j);
606 
607  }
608 
609  /*--- Compute the residual ---*/
610  auto residual = numerics->ComputeResidual(config);
611 
612  /*--- Check for NaNs before applying the residual to the linear system ---*/
613  bool err = CNumerics::CheckResidualNaNs(implicit, nVar, residual);
614 
615  /*--- Update the residual and Jacobian ---*/
616  if (!err) {
617  LinSysRes.AddBlock(iPoint, residual);
618  LinSysRes.SubtractBlock(jPoint, residual);
619  if (implicit) {
620  Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
621  }
622  }
623  }
624 
625  /*--- Warning message about non-physical reconstructions. ---*/
626  if ((iMesh == MESH_0) && (config->GetComm_Level() == COMM_FULL)) {
627  /*--- Add counter results for all threads. ---*/
628  SU2_OMP_ATOMIC
629  ErrorCounter += counter_local;
630 
631  /*--- Add counter results for all ranks. ---*/
632  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS
633  {
634  counter_local = ErrorCounter;
635  SU2_MPI::Reduce(&counter_local, &ErrorCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, MASTER_NODE, SU2_MPI::GetComm());
636  config->SetNonphysical_Reconstr(ErrorCounter);
637  }
638  END_SU2_OMP_SAFE_GLOBAL_ACCESS
639  }

936 void CNEMOEulerSolver::ComputeUnderRelaxationFactor(const CConfig *config) {
937 
938  /* Loop over the solution update given by relaxing the linear
939  system for this nonlinear iteration. */
940 
941  const su2double allowableRatio = 0.2;
942 
943  SU2_OMP_FOR_STAT(omp_chunk_size)
944  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
945  su2double localUnderRelaxation = 1.0;
946 
947  su2double num = 0.0;
948  su2double denom = 0.0;
949 
950  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
951  /* We impose a limit on the maximum percentage that the
952  density (sum of all species) and energy can change over a nonlinear iteration. */
953 
954  const unsigned long index = iPoint * nVar + iVar;
955  if (iVar < config->GetnSpecies()) {
956  num += fabs(LinSysSol[index]);
957  denom += fabs(nodes->GetSolution(iPoint, iVar));
958 
959  /*--- If final density/species, compute Under-relaxation ---*/
960  if (iVar == (config ->GetnSpecies()-1)){
961  su2double ratio = (num/(denom+EPS));
962  if (ratio > allowableRatio) {
963  localUnderRelaxation = min(allowableRatio / ratio, localUnderRelaxation);
964  }
965  }
966 
967  /*--- Energy ---*/
968  if (iVar == (nVar-2)){
969  su2double ratio = fabs(LinSysSol[index]) / (fabs(nodes->GetSolution(iPoint, iVar)) + EPS);
970  if (ratio > allowableRatio) {
971  localUnderRelaxation = min(allowableRatio / ratio, localUnderRelaxation);
972  }
973  }
974  }
975  }
976 
977  /* Threshold the relaxation factor in the event that there is
978  a very small value. This helps avoid catastrophic crashes due
979  to non-realizable states by canceling the update. */
980 
981  if (localUnderRelaxation < 1e-10) localUnderRelaxation = 0.0;
982 
983  /* Store the under-relaxation factor for this point. */
984 
985  nodes->SetUnderRelaxation(iPoint, localUnderRelaxation);
986  }
987  END_SU2_OMP_FOR
988 }
