32 CDiscAdjSolver::CDiscAdjSolver(CGeometry *geometry, CConfig *config, CSolver *direct_sol,
33  unsigned short Kind_Solver, unsigned short iMesh) : CSolver() {
34 
35  /*-- Store some information about direct solver ---*/
36 
37  KindDirect_Solver = Kind_Solver;
38  direct_solver = direct_sol;
39 
40  adjoint = true;
41 
42  nVar = direct_solver->GetnVar();
43  nDim = geometry->GetnDim();
44 
45  nMarker = config->GetnMarker_All();
46  nPoint = geometry->GetnPoint();
47  nPointDomain = geometry->GetnPointDomain();
48 
49  omp_chunk_size = computeStaticChunkSize(nPoint, omp_get_max_threads(), OMP_MAX_SIZE);
50 
51  /*--- Define some auxiliary vectors related to the residual ---*/
52 
53  Residual_RMS.resize(nVar,1.0);
54  Residual_Max.resize(nVar,1.0);
55  Point_Max.resize(nVar,0);
56  Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
57 
58  /*--- Define some auxiliary vectors related to the residual for problems with a BGS strategy---*/
59 
60  if (config->GetMultizone_Residual()){
61 
62  Residual_BGS.resize(nVar,1.0);
63  Residual_Max_BGS.resize(nVar,1.0);
64  Point_Max_BGS.resize(nVar,0);
65  Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
66  }
67 
68  /*--- Sensitivity definition and coefficient in all the markers ---*/
69 
70  CSensitivity.resize(nMarker);
71  for (auto iMarker = 0ul; iMarker < nMarker; iMarker++) {
72  const auto nVertex = geometry->nVertex[iMarker];
73  CSensitivity[iMarker].resize(nVertex, 0.0);
74  }
75 
76  Sens_Geo.resize(config->GetnMarker_Monitoring(), 0.0);
77 
78  /*--- Initialize the discrete adjoint solution to zero everywhere. ---*/
79 
80  if (nVar > MAXNVAR) {
81  SU2_MPI::Error("Oops! The CDiscAdjSolver static array sizes are not large enough.",CURRENT_FUNCTION);
82  }
83 
84  vector<su2double> Solution(nVar,1e-16);
85  nodes = new CDiscAdjVariable(Solution.data(), nPoint, nDim, nVar, config);
86  SetBaseClassPointerToNodes();
87 
88  /*--- Allocate extra solution variables, if any are in use. ---*/
89 
90  const auto nVarExtra = direct_solver->RegisterSolutionExtra(true, config);
91  nodes->AllocateAdjointSolutionExtra(nVarExtra);
92 
93  switch(KindDirect_Solver){
94  case RUNTIME_FLOW_SYS:
95  SolverName = "ADJ.FLOW";
96  break;
97  case RUNTIME_HEAT_SYS:
98  SolverName = "ADJ.HEAT";
99  break;
100  case RUNTIME_TURB_SYS:
101  SolverName = "ADJ.TURB";
102  break;
103  case RUNTIME_SPECIES_SYS:
104  SolverName = "ADJ.SPECIES";
105  break;
106  case RUNTIME_RADIATION_SYS:
107  SolverName = "ADJ.RAD";
108  break;
109  default:
110  SolverName = "ADJ.SOL";
111  break;
112  }
113 }

179 void CDiscAdjSolver::RegisterVariables(CGeometry *geometry, CConfig *config, bool reset) {
180 
181  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
182 
183  /*--- Register farfield values as input ---*/
184 
185  if((config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) && (KindDirect_Solver == RUNTIME_FLOW_SYS && !config->GetBoolTurbomachinery())) {
186 
187  su2double Velocity_Ref = config->GetVelocity_Ref();
188  Alpha = config->GetAoA()*PI_NUMBER/180.0;
189  Beta = config->GetAoS()*PI_NUMBER/180.0;
190  Mach = config->GetMach();
191  Pressure = config->GetPressure_FreeStreamND();
192  Temperature = config->GetTemperature_FreeStreamND();
193 
194  su2double SoundSpeed = 0.0;
195 
196  if (nDim == 2) { SoundSpeed = config->GetVelocity_FreeStreamND()[0]*Velocity_Ref/(cos(Alpha)*Mach); }
197  if (nDim == 3) { SoundSpeed = config->GetVelocity_FreeStreamND()[0]*Velocity_Ref/(cos(Alpha)*cos(Beta)*Mach); }
198 
199  if (!reset) {
200  AD::RegisterInput(Mach);
201  AD::RegisterInput(Alpha);
202  AD::RegisterInput(Temperature);
203  AD::RegisterInput(Pressure);
204  }
205 
206  /*--- Recompute the free stream velocity ---*/
207 
208  if (nDim == 2) {
209  config->GetVelocity_FreeStreamND()[0] = cos(Alpha)*Mach*SoundSpeed/Velocity_Ref;
210  config->GetVelocity_FreeStreamND()[1] = sin(Alpha)*Mach*SoundSpeed/Velocity_Ref;
211  }
212  if (nDim == 3) {
213  config->GetVelocity_FreeStreamND()[0] = cos(Alpha)*cos(Beta)*Mach*SoundSpeed/Velocity_Ref;
214  config->GetVelocity_FreeStreamND()[1] = sin(Beta)*Mach*SoundSpeed/Velocity_Ref;
215  config->GetVelocity_FreeStreamND()[2] = sin(Alpha)*cos(Beta)*Mach*SoundSpeed/Velocity_Ref;
216  }
217 
218  config->SetTemperature_FreeStreamND(Temperature);
219  direct_solver->SetTemperature_Inf(Temperature);
220  config->SetPressure_FreeStreamND(Pressure);
221  direct_solver->SetPressure_Inf(Pressure);
222 
223  }
224 
225  if ((config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) && (KindDirect_Solver == RUNTIME_FLOW_SYS) && config->GetBoolTurbomachinery()){
226 
227  BPressure = config->GetPressureOut_BC();
228  Temperature = config->GetTotalTemperatureIn_BC();
229 
230  if (!reset){
231  AD::RegisterInput(BPressure);
232  AD::RegisterInput(Temperature);
233  }
234 
235  config->SetPressureOut_BC(BPressure);
236  config->SetTotalTemperatureIn_BC(Temperature);
237  }
238 
239  /*--- Register incompressible initialization values as input ---*/
240 
241  if ((config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) &&
242  ((KindDirect_Solver == RUNTIME_FLOW_SYS &&
243  (!config->GetBoolTurbomachinery())))) {
244 
245  /*--- Access the velocity (or pressure) and temperature at the
246  inlet BC and the back pressure at the outlet. Note that we are
247  assuming that have internal flow, which will be true for the
248  majority of cases. External flows with far-field BCs will report
249  zero for these sensitivities. ---*/
250 
251  ModVel = config->GetIncInlet_BC();
252  BPressure = config->GetIncPressureOut_BC();
253  Temperature = config->GetIncTemperature_BC();
254 
255  /*--- Register the variables for AD. ---*/
256 
257  if (!reset) {
258  AD::RegisterInput(ModVel);
259  AD::RegisterInput(BPressure);
260  AD::RegisterInput(Temperature);
261  }
262 
263  /*--- Set the BC values in the config class. ---*/
264 
265  config->SetIncInlet_BC(ModVel);
266  config->SetIncPressureOut_BC(BPressure);
267  config->SetIncTemperature_BC(Temperature);
268 
269  }
270 
271  /*--- Register incompressible radiation values as input ---*/
272 
273  if ((config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) &&
274  ((KindDirect_Solver == RUNTIME_RADIATION_SYS &&
275  (!config->GetBoolTurbomachinery())))) {
276 
277  /*--- Access the nondimensional freestream temperature. ---*/
278 
279  TemperatureRad = config->GetTemperature_FreeStreamND();
280 
281  /*--- Register the variables for AD. ---*/
282 
283  if (!reset) {
284  AD::RegisterInput(TemperatureRad);
285  }
286 
287  /*--- Set the temperature at infinity in the direct solver class. ---*/
288 
289  direct_solver->SetTemperature_Inf(TemperatureRad);
290 
291  }
292 
293  if (config->GetTopology_Optimization()) {
294  if (!reset) direct_solver->RegisterVariables(geometry, config, reset);
295  }
296 
297  /*--- Here it is possible to register other variables as input that influence the flow solution
298  * and thereby also the objective function. The adjoint values (i.e. the derivatives) can be
299  * extracted in the ExtractAdjointVariables routine. ---*/
300 
301  }
302  END_SU2_OMP_SAFE_GLOBAL_ACCESS
303 }

314 void CDiscAdjSolver::ExtractAdjoint_Solution(CGeometry *geometry, CConfig *config, bool CrossTerm) {
315 
316  const bool time_n1_needed = config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND;
317  const bool time_n_needed = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST) || time_n1_needed;
318 
319  const su2double relax = (config->GetInnerIter()==0) ? 1.0 : config->GetRelaxation_Factor_Adjoint();
320 
321  /*--- Thread-local residual variables. ---*/
322  su2double resMax[MAXNVAR] = {0.0}, resRMS[MAXNVAR] = {0.0};
323  unsigned long idxMax[MAXNVAR] = {0};
324 
325  /*--- Set the old solution and compute residuals. ---*/
326 
327  if (!config->GetMultizone_Problem()) nodes->Set_OldSolution();
328 
329  SU2_OMP_FOR_STAT(omp_chunk_size)
330  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
331 
332  /*--- Extract the adjoint solution ---*/
333 
334  su2double Solution[MAXNVAR] = {0.0};
335  direct_solver->GetNodes()->GetAdjointSolution(iPoint,Solution);
336 
337  /*--- Relax and store the adjoint solution, compute the residuals. ---*/
338 
339  for (auto iVar = 0u; iVar < nVar; iVar++) {
340  su2double residual = Solution[iVar]-nodes->GetSolution_Old(iPoint,iVar);
341  nodes->AddSolution(iPoint, iVar, relax*residual);
342 
343  if (iPoint < nPointDomain) {
344  /*--- "Add" residual at (iPoint,iVar) to local residual variables. ---*/
345  ResidualReductions_PerThread(iPoint,iVar,residual,resRMS,resMax,idxMax);
346  }
347  }
348  }
349  END_SU2_OMP_FOR
350 
351  direct_solver->ExtractAdjoint_SolutionExtra(nodes->GetSolutionExtra(), config);
352 
353  /*--- Residuals and time_n terms are not needed when evaluating multizone cross terms. ---*/
354  if (CrossTerm) return;
355 
356  /*--- "Add" residuals from all threads to global residual variables. ---*/
357  ResidualReductions_FromAllThreads(geometry, config, resRMS,resMax,idxMax);
358 
359  SU2_OMP_MASTER {
360  SetIterLinSolver(direct_solver->System.GetIterations());
361  SetResLinSolver(direct_solver->System.GetResidual());
362  }
363  END_SU2_OMP_MASTER
364 
365  /*--- Extract and store the adjoint of the primal solution at time n ---*/
366  if (time_n_needed) {
367  SU2_OMP_FOR_STAT(omp_chunk_size)
368  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
369  su2double Solution[MAXNVAR] = {0.0};
370  direct_solver->GetNodes()->GetAdjointSolution_time_n(iPoint,Solution);
371  nodes->Set_Solution_time_n(iPoint,Solution);
372  }
373  END_SU2_OMP_FOR
374  }
375 
376  /*--- Extract and store the adjoint of the primal solution at time n-1 ---*/
377  if (time_n1_needed) {
378  SU2_OMP_FOR_STAT(omp_chunk_size)
379  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
380  su2double Solution[MAXNVAR] = {0.0};
381  direct_solver->GetNodes()->GetAdjointSolution_time_n1(iPoint,Solution);
382  nodes->Set_Solution_time_n1(iPoint,Solution);
383  }
384  END_SU2_OMP_FOR
385  }
386 
387 }

389 void CDiscAdjSolver::ExtractAdjoint_Variables(CGeometry *geometry, CConfig *config) {
390 
391  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
392 
393  /*--- Extract the adjoint values of the farfield values ---*/
394 
395  if ((config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) && (KindDirect_Solver == RUNTIME_FLOW_SYS) && !config->GetBoolTurbomachinery()) {
396  su2double Local_Sens_Press, Local_Sens_Temp, Local_Sens_AoA, Local_Sens_Mach;
397 
398  Local_Sens_Mach = SU2_TYPE::GetDerivative(Mach);
399  Local_Sens_AoA = SU2_TYPE::GetDerivative(Alpha);
400  Local_Sens_Temp = SU2_TYPE::GetDerivative(Temperature);
401  Local_Sens_Press = SU2_TYPE::GetDerivative(Pressure);
402 
403  SU2_MPI::Allreduce(&Local_Sens_Mach, &Total_Sens_Mach, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
404  SU2_MPI::Allreduce(&Local_Sens_AoA, &Total_Sens_AoA, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
405  SU2_MPI::Allreduce(&Local_Sens_Temp, &Total_Sens_Temp, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
406  SU2_MPI::Allreduce(&Local_Sens_Press, &Total_Sens_Press, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
407  }
408 
409  if ((config->GetKind_Regime() == ENUM_REGIME::COMPRESSIBLE) && (KindDirect_Solver == RUNTIME_FLOW_SYS) && config->GetBoolTurbomachinery()){
410  su2double Local_Sens_BPress, Local_Sens_Temperature;
411 
412  Local_Sens_BPress = SU2_TYPE::GetDerivative(BPressure);
413  Local_Sens_Temperature = SU2_TYPE::GetDerivative(Temperature);
414 
415  SU2_MPI::Allreduce(&Local_Sens_BPress, &Total_Sens_BPress, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
416  SU2_MPI::Allreduce(&Local_Sens_Temperature, &Total_Sens_Temp, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
417  }
418 
419  if ((config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) &&
420  (KindDirect_Solver == RUNTIME_FLOW_SYS &&
421  (!config->GetBoolTurbomachinery()))) {
422 
423  su2double Local_Sens_ModVel, Local_Sens_BPress, Local_Sens_Temp;
424 
425  Local_Sens_ModVel = SU2_TYPE::GetDerivative(ModVel);
426  Local_Sens_BPress = SU2_TYPE::GetDerivative(BPressure);
427  Local_Sens_Temp = SU2_TYPE::GetDerivative(Temperature);
428 
429  SU2_MPI::Allreduce(&Local_Sens_ModVel, &Total_Sens_ModVel, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
430  SU2_MPI::Allreduce(&Local_Sens_BPress, &Total_Sens_BPress, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
431  SU2_MPI::Allreduce(&Local_Sens_Temp, &Total_Sens_Temp, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
432  }
433 
434  if ((config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE) &&
435  (KindDirect_Solver == RUNTIME_RADIATION_SYS &&
436  (!config->GetBoolTurbomachinery()))) {
437 
438  su2double Local_Sens_Temp_Rad;
439  Local_Sens_Temp_Rad = SU2_TYPE::GetDerivative(TemperatureRad);
440 
441  SU2_MPI::Allreduce(&Local_Sens_Temp_Rad, &Total_Sens_Temp_Rad, 1, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
442 
443  /*--- Store it in the Total_Sens_Temp container so it's accessible without the need of a new method ---*/
444  Total_Sens_Temp = Total_Sens_Temp_Rad;
445 
446  }
447 
448  /*--- Extract here the adjoint values of everything else that is registered as input in RegisterInput. ---*/
449 
450  if (config->GetTopology_Optimization()) {
451  direct_solver->ExtractAdjoint_Variables(geometry, config);
452  }
453 
454  }
455  END_SU2_OMP_SAFE_GLOBAL_ACCESS
456 }

458 void CDiscAdjSolver::SetAdjoint_Output(CGeometry *geometry, CConfig *config) {
459 
460  const bool dual_time = (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST ||
461  config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_2ND);
462  const bool multizone = config->GetMultizone_Problem();
463 
464  /*--- Local container to manipulate the adjoint solution. ---*/
465  su2double Solution[MAXNVAR] = {0.0};
466 
467  SU2_OMP_FOR_STAT(omp_chunk_size)
468  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
469 
470  /*--- Get and store the adjoint solution of a point. ---*/
471  for (auto iVar = 0u; iVar < nVar; iVar++) {
472  Solution[iVar] = nodes->GetSolution(iPoint,iVar);
473  }
474 
475  /*--- Add dual time contributions to the adjoint solution. Two terms stored for DT-2nd-order. ---*/
476  if (dual_time && !multizone) {
477  for (auto iVar = 0u; iVar < nVar; iVar++) {
478  Solution[iVar] += nodes->GetDual_Time_Derivative(iPoint,iVar);
479  }
480  }
481 
482  /*--- Set the adjoint values of the primal solution. ---*/
483 
484  direct_solver->GetNodes()->SetAdjointSolution(iPoint,Solution);
485  }
486  END_SU2_OMP_FOR
487 
488  direct_solver->SetAdjoint_SolutionExtra(nodes->GetSolutionExtra(), config);
489 }

491 void CDiscAdjSolver::SetSensitivity(CGeometry *geometry, CConfig *config, CSolver*) {
492 
493  SU2_OMP_PARALLEL {
494 
495  const bool time_stepping = (config->GetTime_Marching() != TIME_MARCHING::STEADY);
496  const su2double eps = config->GetAdjSharp_LimiterCoeff()*config->GetRefElemLength();
497 
498  SU2_OMP_FOR_STAT(omp_chunk_size)
499  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
500 
501  auto Coord = geometry->nodes->GetCoord(iPoint);
502 
503  for (auto iDim = 0u; iDim < nDim; iDim++) {
504 
505  su2double Sensitivity = geometry->nodes->GetAdjointSolution(iPoint, iDim);
506  AD::ResetInput(Coord[iDim]);
507 
508  /*--- If sharp edge, set the sensitivity to 0 on that region ---*/
509 
510  if (config->GetSens_Remove_Sharp() && geometry->nodes->GetSharpEdge_Distance(iPoint) < eps) {
511  Sensitivity = 0.0;
512  }
513  if (!time_stepping) {
514  nodes->SetSensitivity(iPoint,iDim, Sensitivity);
515  } else {
516  nodes->SetSensitivity(iPoint,iDim, nodes->GetSensitivity(iPoint,iDim) + Sensitivity);
517  }
518  }
519  }
520  END_SU2_OMP_FOR
521 
522  SetSurface_Sensitivity(geometry, config);
523 
524  }
525  END_SU2_OMP_PARALLEL
526 }

528 void CDiscAdjSolver::SetSurface_Sensitivity(CGeometry *geometry, CConfig *config) {
529 
530  SU2_OMP_MASTER
531  for (auto& x : Sens_Geo) x = 0.0;
532  END_SU2_OMP_MASTER
533 
534  /*--- Loop over boundary markers to select those for Euler walls and NS walls ---*/
535 
536  for (auto iMarker = 0ul; iMarker < nMarker; iMarker++) {
537 
538  if (!config->GetSolid_Wall(iMarker)) continue;
539 
540  su2double Sens = 0.0;
541 
542  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
543  for (auto iVertex = 0ul; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
544 
545  /*--- Projection of the gradient calculated with AD onto the normal vector of the surface ---*/
546 
547  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
548  const auto Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
549 
550  su2double Sens_Vertex = 0.0;
551  for (auto iDim = 0u; iDim < nDim; iDim++) {
552  Sens_Vertex += Normal[iDim] * nodes->GetSensitivity(iPoint,iDim);
553  }
554  Sens_Vertex /= GeometryToolbox::Norm(nDim, Normal);
555 
556  CSensitivity[iMarker][iVertex] = -Sens_Vertex;
557  Sens += pow(Sens_Vertex,2);
558  }
559  END_SU2_OMP_FOR
560 
561  if (config->GetMarker_All_Monitoring(iMarker) == NO) continue;
562 
563  /*--- Compute sensitivity for each surface point ---*/
564 
565  const auto Marker_Tag = config->GetMarker_All_TagBound(iMarker);
566 
567  for (size_t iMarker_Mon = 0; iMarker_Mon < Sens_Geo.size(); iMarker_Mon++) {
568  if (Marker_Tag == config->GetMarker_Monitoring_TagBound(iMarker_Mon)) {
569  atomicAdd(Sens, Sens_Geo[iMarker_Mon]);
570  break;
