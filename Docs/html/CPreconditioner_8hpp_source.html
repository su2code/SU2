39 /*!
40  * \class CPreconditioner
41  * \brief Abstract base class for defining a preconditioning operation.
42  * \author J. Hicken.
43  *
44  * See the remarks regarding the CMatrixVectorProduct class. The same
45  * idea applies here to the preconditioning operation.
46  */
47 template<class ScalarType>
48 class CPreconditioner {
49 public:
50  /*!
51  * \brief Destructor of the class
52  */
53  virtual ~CPreconditioner() = 0;
54 
55  /*!
56  * \brief Overload of operator (), applies the preconditioner to "u" storing the result in "v".
57  */
58  virtual void operator()(const CSysVector<ScalarType> & u, CSysVector<ScalarType> & v) const = 0;
59 
60  /*!
61  * \brief Generic "preprocessing" hook derived classes may implement to build the preconditioner.
62  */
63  virtual void Build() {}
64 
65  /*!
66  * \brief Return true to identify the identity preconditioner, may allow some solvers to be more efficient.
67  */
68  virtual bool IsIdentity() const { return false; }
69 
70  /*!
71  * \brief Factory method.
72  */
73  static CPreconditioner* Create(ENUM_LINEAR_SOLVER_PREC kind, CSysMatrix<ScalarType>& jacobian,
74  CGeometry* geometry, const CConfig* config);
75 };
76 template<class ScalarType>
77 CPreconditioner<ScalarType>::~CPreconditioner() {}

80 /*!
81  * \class CJacobiPreconditioner
82  * \brief Specialization of preconditioner that uses CSysMatrix class.
83  */
84 template<class ScalarType>
85 class CJacobiPreconditioner final : public CPreconditioner<ScalarType> {
86 private:
87  CSysMatrix<ScalarType>& sparse_matrix; /*!< \brief Pointer to matrix that defines the preconditioner. */
88  CGeometry* geometry; /*!< \brief Pointer to geometry associated with the matrix. */
89  const CConfig *config; /*!< \brief Pointer to problem configuration. */
90 
91 public:
92  /*!
93  * \brief Constructor of the class.
94  * \param[in] matrix_ref - Matrix reference that will be used to define the preconditioner.
95  * \param[in] geometry_ref - Geometry associated with the problem.
96  * \param[in] config_ref - Config of the problem.
97  */
98  inline CJacobiPreconditioner(CSysMatrix<ScalarType> & matrix_ref,
99  CGeometry *geometry_ref, const CConfig *config_ref) :
100  sparse_matrix(matrix_ref)
101  {
102  if((geometry_ref == nullptr) || (config_ref == nullptr))
103  SU2_MPI::Error("Preconditioner needs to be built with valid references.", CURRENT_FUNCTION);
104  geometry = geometry_ref;
105  config = config_ref;
106  }
107 
108  /*!
109  * \note This class cannot be default constructed as that would leave us with invalid Pointers.
110  */
111  CJacobiPreconditioner() = delete;
112 
113  /*!
114  * \brief operator that defines the preconditioner operation
115  * \param[in] u - CSysVector that is being preconditioned
116  * \param[out] v - CSysVector that is the result of the preconditioning
117  */
118  inline void operator()(const CSysVector<ScalarType> & u, CSysVector<ScalarType> & v) const override {
119  sparse_matrix.ComputeJacobiPreconditioner(u, v, geometry, config);
120  }
121 
122  /*!
123  * \note Request the associated matrix to build the preconditioner.
124  */
125  inline void Build() override {
126  sparse_matrix.BuildJacobiPreconditioner();
127  }
128 };

332 template<class ScalarType>
333 CPreconditioner<ScalarType>* CPreconditioner<ScalarType>::Create(ENUM_LINEAR_SOLVER_PREC kind,
334  CSysMatrix<ScalarType>& jacobian,
335  CGeometry* geometry,
336  const CConfig* config) {
337  CPreconditioner<ScalarType>* prec = nullptr;
338 
339  switch (kind) {
340  case JACOBI:
341  prec = new CJacobiPreconditioner<ScalarType>(jacobian, geometry, config);
342  break;
343  case LINELET:
344  prec = new CLineletPreconditioner<ScalarType>(jacobian, geometry, config);
345  break;
346  case LU_SGS:
347  prec = new CLU_SGSPreconditioner<ScalarType>(jacobian, geometry, config);
348  break;
349  case ILU:
350  prec = new CILUPreconditioner<ScalarType>(jacobian, geometry, config);
351  break;
352  case PASTIX_ILU: case PASTIX_LU_P: case PASTIX_LDLT_P:
353  prec = new CPastixPreconditioner<ScalarType>(jacobian, geometry, config, kind);
354  break;
355  }
356 
357  return prec;
358 }
