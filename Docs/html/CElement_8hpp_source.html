35 /*!
36  * \class CElement
37  * \ingroup FemAlgos
38  * \brief Abstract class for defining finite elements.
39  * \note Usage: Element instances are used to compute gradients (in reference or current
40  * coordinates), element matrices (stiffness, mass, etc.), and nodal residuals.
41  * The use-case is to set the coordinates, compute the gradients, and then
42  * incrementally build the matrices using the various "Add_XXX" methods.
43  * \note Implementation: This base class provides a generic interface to an element,
44  * so that client numerics do not need to care about specific details (e.g. shape
45  * functions). However, more efficient code can be generated by caring about them,
46  * we try to strike a balance by tailoring the gradient computation to the specific
47  * sizes (number of integration points, nodes, and dimensions) of derived elements.
48  * This is done via the intermediate template class CElementWithKnownSizes from
49  * which derived elements can inherit the implementation of "ComputeGrad_XXX".
50  * To do so their constructors should consist of:
51  * - Call CElementWithKnownSizes's ctor to allocate parent class members.
52  * - Populate the "GaussWeights", and set the shape function values (Ni) in "GaussPoint".
53  * - Populate the "GaussCoords" and "dNiXj" fields of CElementWithKnownSizes.
54  * - Define the coordinates for nodal extrapolation of quantities at the Gauss points.
55  * Derived elements can also implement cheap methods of computing their area/volume.
56  * \author R. Sanchez
57  */

560  * \brief Templated class to implement the computation of gradients for specific element sizes.
561  * \author P. Gomes, R. Sanchez

563 template<unsigned short NGAUSS, unsigned short NNODE, unsigned short NDIM>
564 class CElementWithKnownSizes : public CElement {
565 private:
566 
567  FORCEINLINE static su2double JacobianAdjoint(const su2double Jacobian[][1], su2double ad[][1]) {
568  /*--- Adjoint to Jacobian, we put 1.0 here so that ad/detJac is the inverse later ---*/
569  ad[0][0] = 1.0;
570  /*--- Determinant of Jacobian ---*/
571  return Jacobian[0][0];
572  }
573 
574  FORCEINLINE static su2double JacobianAdjoint(const su2double Jacobian[][2], su2double ad[][2]) {
575  ad[0][0] = Jacobian[1][1]; ad[0][1] = -Jacobian[0][1];
576  ad[1][0] = -Jacobian[1][0]; ad[1][1] = Jacobian[0][0];
577  /*--- Determinant of Jacobian ---*/
578  return ad[0][0]*ad[1][1]-ad[0][1]*ad[1][0];
579  }
580 
581  FORCEINLINE static su2double JacobianAdjoint(const su2double Jacobian[][3], su2double ad[][3]) {
582  ad[0][0] = Jacobian[1][1]*Jacobian[2][2]-Jacobian[1][2]*Jacobian[2][1];
583  ad[0][1] = Jacobian[0][2]*Jacobian[2][1]-Jacobian[0][1]*Jacobian[2][2];

587  ad[1][2] = Jacobian[0][2]*Jacobian[1][0]-Jacobian[0][0]*Jacobian[1][2];
588  ad[2][0] = Jacobian[1][0]*Jacobian[2][1]-Jacobian[1][1]*Jacobian[2][0];
589  ad[2][1] = Jacobian[0][1]*Jacobian[2][0]-Jacobian[0][0]*Jacobian[2][1];
590  ad[2][2] = Jacobian[0][0]*Jacobian[1][1]-Jacobian[0][1]*Jacobian[1][0];
591  /*--- Determinant of Jacobian ---*/
592  return Jacobian[0][0]*ad[0][0]+Jacobian[0][1]*ad[1][0]+Jacobian[0][2]*ad[2][0];

608  /*!
609  * \brief Implementation of gradient computation leveraging the static sizes.
610  * \param[in] FRAME - template, REFERENCE or CURRENT coordinates.
611  */
612  template<FrameType FRAME>
613  void ComputeGrad_impl(void) {
614 
615  su2double Jacobian[NDIM][NDIM], ad[NDIM][NDIM];
616  unsigned short iNode, iDim, jDim, iGauss;
617 
618  /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
619  for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed) ---*/
620  const su2activematrix& Coord = (FRAME==REFERENCE) ? RefCoord : CurrentCoord;
621 
622  for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
623 
624  /*--- Jacobian transformation ---*/
625  /*--- This does dX/dXi transpose ---*/
626 
627  for (iDim = 0; iDim < NDIM; iDim++)
628  for (jDim = 0; jDim < NDIM; jDim++)
629  Jacobian[iDim][jDim] = 0.0;
630 
631  for (iNode = 0; iNode < NNODE; iNode++)
632  for (iDim = 0; iDim < NDIM; iDim++)
633  for (jDim = 0; jDim < NDIM; jDim++)
634  Jacobian[iDim][jDim] += Coord(iNode,jDim) * dNiXj[iGauss][iNode][iDim];
635 
636  if (NDIM==1) {
637  /*--- Obviously the Jacobian is the slope of the line ---*/
638  Jacobian[0][0] = (Coord[1][0] - Coord[0][0]);
639  }
640 
641  /*--- Adjoint to the Jacobian and determinant ---*/
642 
643  auto detJac = JacobianAdjoint(Jacobian, ad);
644 
645  if (FRAME==REFERENCE)
646  GaussPoint[iGauss].SetJ_X(detJac);
647  else
648  GaussPoint[iGauss].SetJ_x(detJac);
649 
650  /*--- Jacobian inverse (it was already computed as transpose) ---*/
651 
652  for (iDim = 0; iDim < NDIM; iDim++)
653  for (jDim = 0; jDim < NDIM; jDim++)
654  Jacobian[iDim][jDim] = ad[iDim][jDim]/detJac;
655 
656  /*--- Derivatives with respect to global coordinates ---*/
657 
658  for (iNode = 0; iNode < NNODE; iNode++) {
659  for (iDim = 0; iDim < NDIM; iDim++) {
660  su2double GradNi_Xj = 0.0;
661  for (jDim = 0; jDim < NDIM; jDim++)
662  GradNi_Xj += Jacobian[iDim][jDim] * dNiXj[iGauss][iNode][jDim];
663 
664  if (FRAME==REFERENCE)
665  GaussPoint[iGauss].SetGradNi_Xj(GradNi_Xj, iDim, iNode);
666  else
667  GaussPoint[iGauss].SetGradNi_xj(GradNi_Xj, iDim, iNode);
668  }
669  }
670 
671  }
672 
673  }

675  template<FrameType FRAME>
676  void ComputeGrad_impl_surf_embedded() {
677 
678  /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
679  for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed) ---*/
680  const su2activematrix& Coord = (FRAME==REFERENCE) ? RefCoord : CurrentCoord;
681 
682  if (NDIM==1) {
683  unsigned short iNode, iDim, iGauss;
684  su2double Jacobian[2];
685  su2double val_grad;
686 
687  Jacobian[0] = (Coord[1][0] - Coord[0][0]);
688  Jacobian[1] = (Coord[1][1] - Coord[0][1]);
689  const su2double JTJ = Jacobian[0]*Jacobian[0]+Jacobian[1]*Jacobian[1];
690  const su2double volJacobian = sqrt(JTJ);
691 
692  for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
693 
694  GaussPoint[iGauss].SetJ_X(volJacobian);
695 
696  for (iNode = 0; iNode < NNODE; iNode++) {
697  for (iDim=0; iDim<NDIM+1; iDim++) {
698  val_grad = Jacobian[iDim]*dNiXj[iGauss][iNode][0]/JTJ;
699  GaussPoint[iGauss].SetGradNi_Xj( val_grad, iDim, iNode);
700  }
701  }
702  }
703 
704  } else if (NDIM==2) {
705 
706  su2double Jacobian[NDIM+1][NDIM], JTJ[NDIM][NDIM];
707  su2double volJacobian;
708  unsigned short iShape, iDim, jDim, jVertex, kDim, iGauss;
709 
710  for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
711 
712  /*--- Jacobian transformation ---*/
713  /*--- This does dX/dXi transpose ---*/
714  if(NNODE==3) {
715  for (iDim = 0; iDim < NDIM+1; iDim++) {
716  for (jVertex = 0; jVertex < NDIM; jVertex++) {
717  Jacobian[iDim][jVertex] = Coord[jVertex+1][iDim] - Coord[0][iDim];
718  }
719  }
720  }
721  if (NNODE==4) {
722  for (iDim = 0; iDim < NDIM+1; iDim++) {
723  Jacobian[iDim][0] = (Coord[1][iDim] - Coord[0][iDim])/2;
724  Jacobian[iDim][1] = (Coord[3][iDim] - Coord[0][iDim])/2;
725  }
726  }
727 
728  for (iDim = 0; iDim < NDIM; iDim++) {
729  for (jDim = 0; jDim < NDIM; jDim++) {
730  JTJ[iDim][jDim] = 0.0;
731  for (kDim = 0; kDim < NDIM+1; kDim++) {
732  JTJ[iDim][jDim] += Jacobian[kDim][iDim]*Jacobian[kDim][jDim];
733  }
734  }
735  }
736 
737  /*--- matrix volume of Jacobian ---*/
738  const su2double detJacobian=JTJ[0][0]*JTJ[1][1]-JTJ[0][1]*JTJ[1][0];
739 
740  volJacobian = sqrt(detJacobian);
741 
742  GaussPoint[iGauss].SetJ_X(volJacobian);
743 
744  /*--- Derivatives with respect to global coordinates ---*/
745 
746  su2double JTJinv[2][2];
747  JTJinv[0][0]=JTJ[1][1]/detJacobian;
748  JTJinv[1][1]=JTJ[0][0]/detJacobian;
749  JTJinv[0][1]=-JTJ[0][1]/detJacobian;
750  JTJinv[1][0]=-JTJ[1][0]/detJacobian;
751 
752  su2double Jdagger[2][3];
753 
754  for (iDim = 0; iDim < NDIM; iDim++) {
755  for (jDim = 0; jDim < NDIM+1; jDim++) {
756  Jdagger[iDim][jDim] = 0.0;
757  for (kDim = 0; kDim < NDIM; kDim++) {
758  Jdagger[iDim][jDim] += JTJinv[iDim][kDim]*Jacobian[jDim][kDim];
759  }
760  }
761  }
762 
763  su2double GradOut[NDIM+1];
764  for (iShape = 0; iShape < NNODE; iShape++) {
765  for (iDim = 0; iDim < NDIM+1; iDim++) {
766  GradOut[iDim] = 0.0;
767  for (jDim = 0; jDim < NDIM; jDim++) {
768  GradOut[iDim] += Jdagger[jDim][iDim]*dNiXj[iGauss][iShape][jDim];
769  }
770  GaussPoint[iGauss].SetGradNi_Xj(GradOut[iDim], iDim, iShape);
771  }
772  }
773 
774  }
775  }
776  }
