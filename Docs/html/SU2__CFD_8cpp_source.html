40 int main(int argc, char *argv[]) {
41 
42  char config_file_name[MAX_STRING_SIZE];
43  bool dry_run = false;
44  int num_threads = omp_get_max_threads();
45  bool use_thread_mult = false;
46  std::string filename = "default.cfg";
47 
48  /*--- Command line parsing ---*/
49 
50  CLI::App app{"SU2 v7.5.0 \"Blackbird\", The Open-Source CFD Code"};
51  app.add_flag("-d,--dryrun", dry_run, "Enable dry run mode.\n"
52  "Only execute preprocessing steps using a dummy geometry.");
53  app.add_option("-t,--threads", num_threads, "Number of OpenMP threads per MPI rank.");
54  app.add_flag("--thread_multiple", use_thread_mult, "Request MPI_THREAD_MULTIPLE thread support.");
55  app.add_option("configfile", filename, "A config file.")->check(CLI::ExistingFile);
56 
57  CLI11_PARSE(app, argc, argv)
58 
59  /*--- OpenMP initialization ---*/
60 
61  omp_initialize();
62 
63  omp_set_num_threads(num_threads);
64 
65  /*--- MPI initialization, and buffer setting ---*/
66 
67 #if defined(HAVE_OMP) && defined(HAVE_MPI)
68  int required = use_thread_mult? MPI_THREAD_MULTIPLE : MPI_THREAD_FUNNELED;
69  int provided;
70  SU2_MPI::Init_thread(&argc, &argv, required, &provided);
71 #else
72  SU2_MPI::Init(&argc, &argv);
73 #endif
74  SU2_MPI::Comm MPICommunicator = SU2_MPI::GetComm();
75 
76  /*--- Uncomment the following line if runtime NaN catching is desired. ---*/
77  // feenableexcept(FE_INVALID | FE_OVERFLOW | FE_DIVBYZERO );
78 
79  /*--- Initialize libxsmm, if supported. ---*/
80 #ifdef HAVE_LIBXSMM
81  libxsmm_init();
82 #endif
83 
84  /*--- Create a pointer to the main SU2 Driver ---*/
85 
86  CDriver* driver = nullptr;
87 
88  /*--- Load in the number of zones and spatial dimensions in the mesh file (If no config
89  file is specified, default.cfg is used) ---*/
90  strcpy(config_file_name, filename.c_str());
91 
92  /*--- Read the name and format of the input mesh file to get from the mesh
93  file the number of zones and dimensions from the numerical grid (required
94  for variables allocation). ---*/
95 
96  const CConfig config(config_file_name, SU2_COMPONENT::SU2_CFD);
97  const unsigned short nZone = config.GetnZone();
98  const bool turbo = config.GetBoolTurbomachinery();
99 
100  /*--- First, given the basic information about the number of zones and the
101  solver types from the config, instantiate the appropriate driver for the problem
102  and perform all the preprocessing. ---*/
103 
104  const bool disc_adj = config.GetDiscrete_Adjoint();
105  const bool multizone = config.GetMultizone_Problem();
106  const bool harmonic_balance = (config.GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE);
107 
108  if (dry_run) {
109 
110  /*--- Dry Run. ---*/
111  driver = new CDummyDriver(config_file_name, nZone, MPICommunicator);
112 
113  }
114  else if ((!multizone && !harmonic_balance && !turbo) || (turbo && disc_adj)) {
115 
116  /*--- Generic single zone problem: instantiate the single zone driver class. ---*/
117  if (nZone != 1)
118  SU2_MPI::Error("The required solver doesn't support multizone simulations", CURRENT_FUNCTION);
119 
120  if (disc_adj) {
121  driver = new CDiscAdjSinglezoneDriver(config_file_name, nZone, MPICommunicator);
122  }
123  else {
124  driver = new CSinglezoneDriver(config_file_name, nZone, MPICommunicator);
125  }
126 
127  }
128  else if (multizone && !turbo) {
129 
130  /*--- Generic multizone problems. ---*/
131  if (disc_adj) {
132  driver = new CDiscAdjMultizoneDriver(config_file_name, nZone, MPICommunicator);
133  }
134  else {
135  driver = new CMultizoneDriver(config_file_name, nZone, MPICommunicator);
136  }
137 
138  }
139  else if (harmonic_balance) {
140 
141  /*--- Harmonic balance problem: instantiate the Harmonic Balance driver class. ---*/
142  driver = new CHBDriver(config_file_name, nZone, MPICommunicator);
143 
144  }
145  else if (turbo) {
146 
147  /*--- Turbomachinery problem. ---*/
148  driver = new CTurbomachineryDriver(config_file_name, nZone, MPICommunicator);
149 
150  } /*--- These are all the possible cases ---*/
151 
152  /*--- Launch the main external loop of the solver. ---*/
153 
154  driver->StartSolver();
155 
156  /*--- Postprocess all the containers, close history file, exit SU2. ---*/
157 
158  driver->Postprocessing();
159 
160  delete driver;
161 
162  /*---Finalize libxsmm, if supported. ---*/
163 #ifdef HAVE_LIBXSMM
164  libxsmm_finalize();
165 #endif
166 
167  /*--- Finalize AD, if necessary. ---*/
168 #ifdef HAVE_OPDI
169  AD::getGlobalTape().finalize();
170 #endif
171 
172  /*--- Finalize MPI parallelization. ---*/
173  SU2_MPI::Finalize();
174 
175  /*--- Finalize OpenMP. ---*/
176  omp_finalize();
177 
178  return EXIT_SUCCESS;
179 
180 }
