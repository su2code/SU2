61 CWallModel1DEQ::CWallModel1DEQ(CConfig *config,
62  const string &Marker_Tag)
63  : CWallModel(config) {
64 
65  /* Retrieve the integer and floating point information for this
66  boundary marker. */
67  const unsigned short *intInfo = config->GetWallFunction_IntInfo(Marker_Tag);
68  const su2double *doubleInfo = config->GetWallFunction_DoubleInfo(Marker_Tag);
69 
70  /* Copy the data into the member variables. */
71  numPoints = intInfo[0];
72  h_wm = doubleInfo[0];
73  expansionRatio = doubleInfo[1];
74 
75  unsigned short nfa = numPoints + 1;
76  su2double currentHeight = 1.0;
77 
78  /* Allocate the memory for the coordinates of the grid points used
79  in the 1D equilibrium wall model. */
80  y_cv.resize(numPoints);
81  y_fa.resize(numPoints+1);
82 
83  /* Determine the scaled version of the normal coordinates, where the
84  first normal coordinate is simply 1.0. */
85 
86  y_fa[0] = 0.0;
87  for(unsigned short i=1; i<nfa; ++i) {
88  y_fa[i] = y_fa[i-1] + currentHeight;
89  y_cv[i-1] = 0.5 * (y_fa[i] + y_fa[i-1]);
90  currentHeight = currentHeight*expansionRatio;
91  }
92 
93  su2double y_max = y_cv[numPoints-1];
94  for(unsigned short i=0; i<nfa; ++i) {
95  y_fa[i] = y_fa[i]/y_max * h_wm;
96  }
97  for(unsigned short i=0; i<numPoints; ++i) {
98  y_cv[i] = y_cv[i]/y_max * h_wm;
99  }
100 }

102 void CWallModel1DEQ::WallShearStressAndHeatFlux(const su2double tExchange,
103  const su2double velExchange,
104  const su2double muExchange,
105  const su2double pExchange,
106  const su2double Wall_HeatFlux,
107  const bool HeatFlux_Prescribed,
108  const su2double TWall,
109  const bool Temperature_Prescribed,
110  CFluidModel *FluidModel,
111  su2double &tauWall,
112  su2double &qWall,
113  su2double &ViscosityWall,
114  su2double &kOverCvWall) {
115 
116 
117  /* Set tau wall to initial guess
118  */
119  tauWall = 0.5;
120  qWall = 0.0;
121  ViscosityWall = 0.0;
122  kOverCvWall = 0.0;
123 
124  /* Set some constants, assuming air at standard conditions
125  Todo: Read values below from the config file or from solver.
126  */
127  //su2double C_1 = 2.03929e-04;
128  su2double C_1 = 1.716e-5;
129  su2double S = 110.4;
130  su2double T_ref = 273.15;
131  su2double R = 287.058;
132  su2double A = 17;
133  su2double gamma = 1.4;
134  su2double c_p = (gamma*R)/(gamma-1);
135  su2double c_v = R/(gamma-1);
136  su2double h_wall = c_p * TWall;
137  su2double h_bc = c_p * tExchange;
138  unsigned short nfa = numPoints + 1;
139 
140  /* Set up vectors
141  */
142  vector<su2double> T(nfa, tExchange);
143  vector<su2double> mu_fa(nfa, muExchange);
144  vector<su2double> tmp(nfa, 0.0);
145  vector<su2double> u(numPoints, 0.0);
146  vector<su2double> lower(numPoints-1,0.0);
147  vector<su2double> upper(numPoints-1,0.0);
148  vector<su2double> diagonal(numPoints,0.0);
149  vector<su2double> rhs(numPoints,0.0);
150 
151  /* Set parameters for control
152  */
153  bool converged = false;
154  unsigned short iter = 0, max_iter = 25;
155  su2double tauWall_prev = 0.0, tol = 1e-3, aux_rhs=0.0;
156  su2double qWall_prev=0.0;
157  su2double mut, nu, mu_lam, rho, utau, y_plus, D;
158 
159  while (converged == false){
160 
161  iter += 1;
162  if (iter == max_iter) converged = true;
163 
164  tauWall_prev = tauWall;
165  qWall_prev = qWall;
166 
167  /* Calculate total viscosity
168  note: rho and mu_lam will be a function of temperature when solving an energy equation
169  */
170  for(unsigned short i=0; i < nfa; ++i){
171  mu_lam = C_1 * pow(T[i]/T_ref, 1.5) * ((T_ref + S)/ (T[i] + S));
172  mu_fa[i] = mu_lam;
173  }
174 
175  for(unsigned short i=1; i < nfa; ++i){
176  rho = pExchange / (R*T[i]);
177  nu = mu_fa[i] / rho;
178  utau = sqrt(tauWall / rho);
179  y_plus = y_fa[i] * utau / nu;
180  D = pow(1.0 - exp((-y_plus)/A),2.0);
181  mut = rho * karman * y_fa[i] * utau * D;
182  mu_fa[i] += mut;
183  }
184 
185  /* Momentum matrix
186  The solution vector is u at y_cv
187  */
188  lower.assign(numPoints-1,0.0);
189  upper.assign(numPoints-1,0.0);
190  diagonal.assign(numPoints,0.0);
191  rhs.assign(numPoints,0.0);
192 
193  /* Top bc
194  */
195  diagonal[numPoints - 1] = 1.0;
196  rhs[numPoints - 1] = velExchange;
197 
198  /* Internal cvs
199  */
200  for (unsigned short i=1; i < (numPoints - 1); ++i){
201  upper[i] = mu_fa[i + 1] / (y_cv[i + 1] - y_cv[i] );
202  lower[i-1] = mu_fa[i] / (y_cv[i] - y_cv[i - 1] );
203  diagonal[i] = -1.0 * (upper[i] + lower[i - 1]);
204  }
205 
206  /* Wall BC
207  */
208  upper[0] = mu_fa[1]/(y_cv[1] - y_cv[0]);
209  diagonal[0] = -1.0 * (upper[0] + mu_fa[0]/(y_cv[0]-y_fa[0]) );
210  rhs[0] = 0.0;
211 
212  /* Solve the matrix problem to get the velocity field
213  - rhs returns the solution
214  */
215 #if (defined(HAVE_MKL) || defined(HAVE_LAPACK)) && !(defined(CODI_REVERSE_TYPE) || defined(CODI_FORWARD_TYPE))
216  int info, nrhs = 1;
217 
218  dgtsv_(&numPoints,&nrhs,lower.data(),diagonal.data(),upper.data(),rhs.data(),&numPoints, &info);
219  if (info != 0)
220  SU2_MPI::Error("Unsuccessful call to dgtsv_", CURRENT_FUNCTION);
221 #else
222  SU2_MPI::Error("Not compiled with MKL or LAPACK support", CURRENT_FUNCTION);
223 #endif
224 
225  u = rhs;
226 
227  /* Update total viscosity
228  */
229  for(unsigned short i=0; i < nfa; ++i){
230  mu_lam = C_1 * pow(T[i]/T_ref, 1.5) * ((T_ref + S)/ (T[i] + S));
231  mu_fa[i] = mu_lam;
232  tmp[i] = mu_lam/Pr_lam;
233  }
234  /* Update tauWall
235  */
236  tauWall = mu_fa[0] * (u[0] - 0.0)/(y_cv[0]-y_fa[0]);
237  for(unsigned short i=1; i < nfa; ++i){
238  rho = pExchange / (R*T[i]);
239  nu = mu_fa[i] / rho;
240  utau = sqrt(tauWall / rho);
241  y_plus = y_fa[i] * utau / nu;
242  D = pow(1.0 - exp((-y_plus)/A),2.0);
243  mut = rho * karman * y_fa[i] * utau * D;
244  mu_fa[i] += mut;
245  tmp[i] += mut/Pr_turb;
246  }
247 
248  /* Energy matrix
249  The Solution vector is Enthalpy at y_cv
250  */
251  lower.assign(numPoints-1,0.0);
252  upper.assign(numPoints-1,0.0);
253  diagonal.assign(numPoints,0.0);
254  rhs.assign(numPoints,0.0);
255 
256  /* Internal cvs
257  */
258  for (unsigned short i=1; i < (numPoints - 1); ++i){
259  upper[i] = tmp[i + 1] / (y_cv[i + 1] - y_cv[i] );
260  lower[i-1] = tmp[i] / (y_cv[i] - y_cv[i - 1] );
261  diagonal[i] = -1.0 * (upper[i] + lower[i - 1]);
262  }
263 
264  /* Top BC
265  */
266  diagonal[numPoints - 1] = 1.0;
267 
268  /* Wall BC
269  */
270  upper[0] = tmp[1]/(y_cv[1] - y_cv[0]);
271  diagonal[0] = -1.0 * (upper[0] + tmp[0]/(y_cv[0]-y_fa[0]) );
272  aux_rhs = tmp[0]/(y_cv[0]-y_fa[0]);
273 
274  /* RHS of the Energy equation
275  - Compute flux -- (mu + mu_t) * u * du/dy --
276  */
277 
278  /* Zero flux at the wall
279  */
280  tmp[0] = 0. ;
281  for (unsigned short i=1; i < numPoints; ++i){
282  tmp[i] = 0.5* (mu_fa[i]) * (u[i] + u[i-1]) * (u[i] -u[i-1])/(y_cv[i] - y_cv[i - 1] ) ;
283  }
284  for (unsigned short i=0; i < (numPoints - 1); ++i){
285  rhs[i] = -tmp[i+1] + tmp[i];
286  }
287 
288  if (HeatFlux_Prescribed == true){
289  /* dT/dy = 0 -> Twall = T[1] */
290  h_wall = c_p * T[1];
291  }
292 
293  rhs[0] -= aux_rhs * h_wall;
294  rhs[numPoints-1] = h_bc;
295 
296  /* Solve the matrix problem to get the Enthalpy field
297  */
298 #if (defined(HAVE_MKL) || defined(HAVE_LAPACK)) && !(defined(CODI_REVERSE_TYPE) || defined(CODI_FORWARD_TYPE))
299  dgtsv_(&numPoints,&nrhs,lower.data(),diagonal.data(),upper.data(),rhs.data(),&numPoints, &info);
300  if (info != 0)
301  SU2_MPI::Error("Unsuccessful call to dgtsv_", CURRENT_FUNCTION);
302 #else
303  SU2_MPI::Error("Not compiled with MKL or LAPACK support", CURRENT_FUNCTION);
304 #endif
305 
306  /* Get Temperature from enthalpy
307  - Temperature will be at face
308  */
309  T[0] = h_wall/c_p;
310  T[numPoints] = h_bc/c_p;
311  for (unsigned short i=0; i < numPoints-1; i++){
312  T[i+1] = 0.5 * (rhs[i] + rhs[i+1])/c_p;
313  }
314 
315  /* Final update tauWall
316  */
317  mu_lam = C_1 * pow(T[0]/T_ref, 1.5) * ((T_ref + S)/ (T[0] + S));
318 
319  /* These quantities will be returned.
320  */
321  tauWall = mu_lam * (u[0] - 0.0)/(y_cv[0]-y_fa[0]);
322  qWall = mu_lam * (c_p / Pr_lam) * -(T[1] - T[0]) / (y_fa[1]-y_fa[0]);
323  ViscosityWall = mu_lam;
324  //kOverCvWall = c_p / c_v * (mu[0]/Pr_lam + muTurb[0]/Pr_turb);
325  kOverCvWall = c_p / c_v * (mu_lam/Pr_lam);
326 
327  /* Final check of the Y+
328  */
329  rho = pExchange / (R * T[0]);
330  if (y_cv[0] * sqrt(tauWall/rho) / (mu_lam/rho) > 1.0)
331  SU2_MPI::Error("Y+ greater than one: Increase the number of points or growth ratio.", CURRENT_FUNCTION);
332 
333  /* Define a norm
334  */
335  if (abs(1.0 - tauWall/tauWall_prev) < tol && abs(1.0 - qWall/qWall_prev) < tol){
336  converged = true;
337  }
338  }
339 }

353 void CWallModelLogLaw::WallShearStressAndHeatFlux(const su2double tExchange,
354  const su2double velExchange,
355  const su2double muExchange,
356  const su2double pExchange,
357  const su2double Wall_HeatFlux,
358  const bool HeatFlux_Prescribed,
359  const su2double Wall_Temperature,
360  const bool Temperature_Prescribed,
361  CFluidModel *FluidModel,
362  su2double &tauWall,
363  su2double &qWall,
364  su2double &ViscosityWall,
365  su2double &kOverCvWall) {
366 
367  /* Set the wall temperature, depending whether or not the temperature
368  was prescribed and initialize the fluid model. */
369  const su2double TWall = Temperature_Prescribed ? Wall_Temperature : tExchange;
370 
371  FluidModel->SetTDState_PT(pExchange, TWall);
372 
373  /* Get the required data from the fluid model. */
374  const su2double rho_wall = FluidModel->GetDensity();
375  const su2double mu_wall = FluidModel->GetLaminarViscosity();
376  const su2double c_p = FluidModel->GetCp();
377  const su2double c_v = FluidModel->GetCv();
378  const su2double nu_wall = mu_wall / rho_wall;
379 
380  /* Initial guess of the friction velocity. */
381  su2double u_tau = max(0.01*velExchange, 1.e-5);
382 
383  /* Set parameters for control of the Newton iteration. */
384  bool converged = false;
385  unsigned short iter = 0, max_iter = 50;
386  const su2double tol=1e-3;
387 
388  while (converged == false){
389 
390  iter += 1;
391  if (iter == max_iter) converged = true;
392 
393  const su2double u_tau0 = u_tau;
394  const su2double y_plus = u_tau0*h_wm/nu_wall;
395 
396  /* Reichardt boundary layer analytical law
397  fprime is the differentiation of the Reichardt law with repect to u_tau.
398  */
399  const su2double fval = velExchange/u_tau0 - ((C - log(karman)/karman)*(1.0 - exp(-y_plus/11.0)
400  - (y_plus/11.0)*exp(-0.33*y_plus))) - log(karman*y_plus + 1.0)/karman;
401  const su2double fprime = -velExchange/pow(u_tau0,2.0)
402  + (- C + log(karman)/karman)*(-(1.0/11.0)*h_wm*exp(-0.33*y_plus)/nu_wall
403  + (1.0/11.0)*h_wm*exp(-(1.0/11.0)*y_plus)/nu_wall
404  + (1.0/33.0)*u_tau0*pow(h_wm,2.0)*exp(-0.33*y_plus)/pow(nu_wall, 2.0))
405  - 1.0*h_wm/(nu_wall*(karman*y_plus + 1.0));
406 
407  /* Newton method
408  */
409  const su2double newton_step = fval/fprime;
410  u_tau = u_tau0 - newton_step;
411 
412  /* Define a norm
413  */
414  if (abs(1.0 - u_tau/u_tau0) < tol) converged = true;
415  }
416 
417  tauWall = rho_wall * pow(u_tau,2.0);
418 
419  if (Temperature_Prescribed){
420  /* The Kader's law will be used to approximate the variations of the temperature inside the boundary layer.
421  */
422  const su2double y_plus = u_tau*h_wm/nu_wall;
423  const su2double lhs = - ((tExchange - TWall) * rho_wall * c_p * u_tau);
424  const su2double Gamma = - (0.01 * (Pr_lam * pow(y_plus,4.0))/(1.0 + 5.0*y_plus*pow(Pr_lam,3.0)));
425  const su2double rhs_1 = Pr_lam * y_plus * exp(Gamma);
426  const su2double rhs_2 = (2.12*log(1.0+y_plus) + pow((3.85*pow(Pr_lam,(1.0/3.0)) - 1.3),2.0) + 2.12*log(Pr_lam)) * exp(1./Gamma);
427  qWall = lhs/(rhs_1 + rhs_2);
428  }
429  else{
430  qWall = Wall_HeatFlux;
431  }
432 
433  ViscosityWall = mu_wall;
434  kOverCvWall = FluidModel->GetThermalConductivity()/c_v;
435 }
