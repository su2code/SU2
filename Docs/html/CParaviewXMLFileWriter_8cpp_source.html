53 
54  if (!dataSorter->GetConnectivitySorted()){
55  SU2_MPI::Error("Connectivity must be sorted.", CURRENT_FUNCTION);
56  }
57 
58  /*--- We always have 3 coords, independent of the actual value of nDim ---*/
59

61  const unsigned short nDim = dataSorter->GetnDim();
62  unsigned short iDim = 0;
63

69 
70  char str_buf[255];
71 
72  OpenMPIFile(val_filename);
73 
74  dataOffset = 0;
75 
76  /*--- Communicate the number of total points that will be
77  written by each rank. After this communication, each proc knows how
78  many poinnts will be written before its location in the file and the
79  offsets can be correctly set. ---*/
80 
81  unsigned long myPoint, GlobalPoint;
82 
83  GlobalPoint = dataSorter->GetnPointsGlobal();
84  myPoint = dataSorter->GetnPoints();
85 
86  /*--- Compute our local number of elements, the required storage,
87  and reduce the total number of elements and storage globally. ---*/
88 
89  unsigned long myElem, myElemStorage, GlobalElem, GlobalElemStorage;
90 
91  unsigned long nParallel_Line = dataSorter->GetnElem(LINE),
92  nParallel_Tria = dataSorter->GetnElem(TRIANGLE),
93  nParallel_Quad = dataSorter->GetnElem(QUADRILATERAL),
94  nParallel_Tetr = dataSorter->GetnElem(TETRAHEDRON),
95  nParallel_Hexa = dataSorter->GetnElem(HEXAHEDRON),
96  nParallel_Pris = dataSorter->GetnElem(PRISM),
97  nParallel_Pyra = dataSorter->GetnElem(PYRAMID);
98 
99  myElem = dataSorter->GetnElem();
100  myElemStorage = dataSorter->GetnConn();
101  GlobalElem = dataSorter->GetnElemGlobal();
102  GlobalElemStorage = dataSorter->GetnConnGlobal();
103 
104  /* Write the ASCII XML header. Note that we use the appended format for the data,
105  * which means that all data is appended at the end of the file in one binary blob.
106  */
107 
108  if (!bigEndian){
109  WriteMPIString("<VTKFile type=\"UnstructuredGrid\" version=\"1.0\" byte_order=\"LittleEndian\" header_type=\"UInt64\">\n", MASTER_NODE);
110  } else {
111  WriteMPIString("<VTKFile type=\"UnstructuredGrid\" version=\"1.0\" byte_order=\"BigEndian\" header_type=\"UInt64\">\n", MASTER_NODE);
112  }
113 
114  WriteMPIString("<UnstructuredGrid>\n", MASTER_NODE);
115 
116  SPRINTF(str_buf, "<Piece NumberOfPoints=\"%i\" NumberOfCells=\"%i\">\n",
117  SU2_TYPE::Int(GlobalPoint), SU2_TYPE::Int(GlobalElem));
118 
119  WriteMPIString(std::string(str_buf), MASTER_NODE);
120  WriteMPIString("<Points>\n", MASTER_NODE);
121  AddDataArray(VTKDatatype::FLOAT32, "", NCOORDS, myPoint*NCOORDS, GlobalPoint*NCOORDS);
122  WriteMPIString("</Points>\n", MASTER_NODE);
123  WriteMPIString("<Cells>\n", MASTER_NODE);
124  AddDataArray(VTKDatatype::INT32, "connectivity", 1, myElemStorage, GlobalElemStorage);
125  AddDataArray(VTKDatatype::INT32, "offsets", 1, myElem, GlobalElem);
126  AddDataArray(VTKDatatype::UINT8, "types", 1, myElem, GlobalElem);
127  WriteMPIString("</Cells>\n", MASTER_NODE);
128 
129  WriteMPIString("<PointData>\n", MASTER_NODE);
130 
131  /*--- Adjust container start location to avoid point coords. ---*/
132 
133  unsigned short varStart = 2;
134  if (nDim == 3) varStart++;
135 
136  /*--- Loop over all variables that have been registered in the output. ---*/
137 
138  unsigned short iField, VarCounter = varStart;
139  for (iField = varStart; iField < fieldNames.size(); iField++) {
140 
141  string fieldname = fieldNames[iField];
142  fieldname.erase(remove(fieldname.begin(), fieldname.end(), '"'),
143  fieldname.end());
144 
145  /*--- Check whether this field is a vector or scalar. ---*/
146 
147  bool output_variable = true, isVector = false;
148  size_t found = fieldNames[iField].find("_x");
149  if (found!=string::npos) {
150  output_variable = true;
151  isVector = true;
152  }
153  found = fieldNames[iField].find("_y");
154  if (found!=string::npos) {
155  /*--- We have found a vector, so skip the Y component. ---*/
156  output_variable = false;
157  VarCounter++;
158  }
159  found = fieldNames[iField].find("_z");
160  if (found!=string::npos) {
161  /*--- We have found a vector, so skip the Z component. ---*/
162  output_variable = false;
163  VarCounter++;
164  }
165 
166  /*--- Write the point data as an <X,Y,Z> vector or a scalar. ---*/
167 
168  if (output_variable && isVector) {
169 
170  /*--- Adjust the string name to remove the leading "X-" ---*/
171 
172  fieldname.erase(fieldname.end()-2,fieldname.end());
173 
174  AddDataArray(VTKDatatype::FLOAT32, fieldname, NCOORDS, myPoint*NCOORDS, GlobalPoint*NCOORDS);
175 
176  } else if (output_variable) {
177 
178  AddDataArray(VTKDatatype::FLOAT32, fieldname, 1, myPoint, GlobalPoint);
179 
180  }
181 
182  }
183  WriteMPIString("</PointData>\n", MASTER_NODE);
184  WriteMPIString("</Piece>\n", MASTER_NODE);
185  WriteMPIString("</UnstructuredGrid>\n", MASTER_NODE);
186 
187  /*--- Now write all the data we have previously defined into the binary section of the file ---*/
188 
189  WriteMPIString("<AppendedData encoding=\"raw\">\n_", MASTER_NODE);
190 
191  /*--- Load/write the 1D buffer of point coordinates. Note that we
192  always have 3 coordinate dimensions, even for 2D problems. ---*/
193 
194  vector<float> dataBufferFloat(myPoint*NCOORDS);
195  for (iPoint = 0; iPoint < myPoint; iPoint++) {
196  for (iDim = 0; iDim < NCOORDS; iDim++) {
197  if (nDim == 2 && iDim == 2) {
198  dataBufferFloat[iPoint*NCOORDS + iDim] = 0.0;
199  } else {
200  float val = (float)dataSorter->GetData(iDim, iPoint);
201  dataBufferFloat[iPoint*NCOORDS + iDim] = val;
202  }
203  }
204  }
205 
206  WriteDataArray(dataBufferFloat.data(), VTKDatatype::FLOAT32, NCOORDS*myPoint, GlobalPoint*NCOORDS,
207  dataSorter->GetnPointCumulative(rank)*NCOORDS);
208 
209  /*--- Load/write 1D buffers for the connectivity of each element type. ---*/
210 
211  vector<int> connBuf(myElemStorage);
212  vector<int> offsetBuf(myElem);
213  unsigned long iStorage = 0, iElemID = 0;
214  unsigned short iNode = 0;
215 
216  auto copyToBuffer = [&](GEO_TYPE type, unsigned long nElem, unsigned short nPoints){
217  for (iElem = 0; iElem < nElem; iElem++) {
218  for (iNode = 0; iNode < nPoints; iNode++){
219  connBuf[iStorage+iNode] = int(dataSorter->GetElem_Connectivity(type, iElem, iNode)-1);
220  }
221  iStorage += nPoints;
222  offsetBuf[iElemID++] = int(iStorage + dataSorter->GetnElemConnCumulative(rank));
223  }
224  };
225 
226  copyToBuffer(LINE, nParallel_Line, N_POINTS_LINE);
227  copyToBuffer(TRIANGLE, nParallel_Tria, N_POINTS_TRIANGLE);
228  copyToBuffer(QUADRILATERAL, nParallel_Quad, N_POINTS_QUADRILATERAL);
229  copyToBuffer(TETRAHEDRON, nParallel_Tetr, N_POINTS_TETRAHEDRON);
230  copyToBuffer(HEXAHEDRON, nParallel_Hexa, N_POINTS_HEXAHEDRON);
231  copyToBuffer(PRISM, nParallel_Pris, N_POINTS_PRISM);
232  copyToBuffer(PYRAMID, nParallel_Pyra, N_POINTS_PYRAMID);
233 
234  WriteDataArray(connBuf.data(), VTKDatatype::INT32, myElemStorage, GlobalElemStorage,
235  dataSorter->GetnElemConnCumulative(rank));
236  WriteDataArray(offsetBuf.data(), VTKDatatype::INT32, myElem, GlobalElem, dataSorter->GetnElemCumulative(rank));
237 
238  /*--- Load/write the cell type for all elements in the file. ---*/
239 
240  vector<uint8_t> typeBuf(myElem);
241  vector<uint8_t>::iterator typeIter = typeBuf.begin();
242 
243  std::fill(typeIter, typeIter+nParallel_Line, LINE); typeIter += nParallel_Line;
244  std::fill(typeIter, typeIter+nParallel_Tria, TRIANGLE); typeIter += nParallel_Tria;
245  std::fill(typeIter, typeIter+nParallel_Quad, QUADRILATERAL); typeIter += nParallel_Quad;
246  std::fill(typeIter, typeIter+nParallel_Tetr, TETRAHEDRON); typeIter += nParallel_Tetr;
247  std::fill(typeIter, typeIter+nParallel_Hexa, HEXAHEDRON); typeIter += nParallel_Hexa;
248  std::fill(typeIter, typeIter+nParallel_Pris, PRISM); typeIter += nParallel_Pris;
249  std::fill(typeIter, typeIter+nParallel_Pyra, PYRAMID); typeIter += nParallel_Pyra;
250 
251  WriteDataArray(typeBuf.data(), VTKDatatype::UINT8, myElem, GlobalElem, dataSorter->GetnElemCumulative(rank));
252 
253  /*--- Loop over all variables that have been registered in the output. ---*/
254 
255  VarCounter = varStart;
256  for (iField = varStart; iField < fieldNames.size(); iField++) {
257 
258  /*--- Check whether this field is a vector or scalar. ---*/
259 
260  bool output_variable = true, isVector = false;
261  size_t found = fieldNames[iField].find("_x");
262  if (found!=string::npos) {
263  output_variable = true;
264  isVector = true;
265  }
266  found = fieldNames[iField].find("_y");
267  if (found!=string::npos) {
268  /*--- We have found a vector, so skip the Y component. ---*/
269  output_variable = false;
270  VarCounter++;
271  }
272  found = fieldNames[iField].find("_z");
273  if (found!=string::npos) {
274  /*--- We have found a vector, so skip the Z component. ---*/
275  output_variable = false;
276  VarCounter++;
277  }
278 
279  /*--- Write the point data as an <X,Y,Z> vector or a scalar. ---*/
280 
281  if (output_variable && isVector) {
282 
283  /*--- Load up the buffer for writing this rank's vector data. ---*/
284 
285  float val = 0.0;
286  for (iPoint = 0; iPoint < myPoint; iPoint++) {
287  for (iDim = 0; iDim < NCOORDS; iDim++) {
288  if (nDim == 2 && iDim == 2) {
289  dataBufferFloat[iPoint*NCOORDS + iDim] = 0.0;
290  } else {
291  val = (float)dataSorter->GetData(VarCounter+iDim,iPoint);
292  dataBufferFloat[iPoint*NCOORDS + iDim] = val;
293  }
294  }
295  }
296 
297  WriteDataArray(dataBufferFloat.data(), VTKDatatype::FLOAT32, myPoint*NCOORDS, GlobalPoint*NCOORDS,
298  dataSorter->GetnPointCumulative(rank)*NCOORDS);
299 
300  VarCounter++;
301 
302  } else if (output_variable) {
303 
304 
305  /*--- For now, create a temp 1D buffer to load up the data for writing.
306  This will be replaced with a derived data type most likely. ---*/
307 
308  for (iPoint = 0; iPoint < myPoint; iPoint++) {
309  float val = (float)dataSorter->GetData(VarCounter,iPoint);
310  dataBufferFloat[iPoint] = val;
311  }
312 
313  WriteDataArray(dataBufferFloat.data(), VTKDatatype::FLOAT32, myPoint, GlobalPoint,
314  dataSorter->GetnPointCumulative(rank));
315 
316  VarCounter++;
317  }
318 
319  }
320 
321  WriteMPIString("</AppendedData>\n", MASTER_NODE);
322  WriteMPIString("</VTKFile>\n", MASTER_NODE);

328 void CParaviewXMLFileWriter::WriteDataArray(void* data, VTKDatatype type, unsigned long arraySize,
329  unsigned long globalSize, unsigned long offset){
330 
331  std::string typeStr;
332  unsigned long typeSize = 0;
333 
334  GetTypeInfo(type, typeStr, typeSize);
335 
336  /*--- Compute the size of the data to write in bytes ---*/
337 
338  size_t byteSize = arraySize*typeSize;
339 
340  /*--- The total data size ---*/
341  size_t totalByteSize = globalSize*typeSize;
342 
343  /*--- Only the master node writes the total size in bytes as unsigned long in front of the array data ---*/
344 
345  if (!WriteMPIBinaryData(&totalByteSize, sizeof(size_t), MASTER_NODE)){
346  SU2_MPI::Error("Writing array size failed", CURRENT_FUNCTION);
347  }
348 
349  /*--- Collectively write all the data ---*/
350 
351  if (!WriteMPIBinaryDataAll(data, byteSize, totalByteSize, offset*typeSize)){
352  SU2_MPI::Error("Writing data array failed", CURRENT_FUNCTION);
353  }
354 }

356 void CParaviewXMLFileWriter::AddDataArray(VTKDatatype type, string name,
357  unsigned short nComponents, unsigned long arraySize, unsigned long globalSize){
358 
359  /*--- Add quotation marks around the arguments ---*/
360 
361  name = "\"" + name + "\"";
362 
363  string nComp ="\"" + PrintingToolbox::to_string(nComponents) + "\"";
364 
365  /*--- Full precision ASCII output of offset for 32 bit integer ---*/
366 
367  stringstream ss;
368  ss.precision(10); ss.setf(std::ios::fixed, std::ios::floatfield);
369  ss << "\"" << dataOffset << "\"";
370  string offsetStr = ss.str();
371 
372  std::string typeStr;
373  unsigned long typeSize = 0;
374 
375  GetTypeInfo(type, typeStr, typeSize);
376 
377  /*--- Total data size ---*/
378 
379  size_t totalByteSize = globalSize*typeSize;
380 
381  /*--- Write the ASCII XML header information for this array ---*/
382 
383  WriteMPIString(string("<DataArray type=") + typeStr +
384  string(" Name=") + name +
385  string(" NumberOfComponents= ") + nComp +
386  string(" offset=") + offsetStr +
387  string(" format=\"appended\"/>\n"), MASTER_NODE);
388 
389  dataOffset += totalByteSize + sizeof(size_t);
390 
391 }
