64 su2double CEdge::GetVolume(const su2double *coord_Edge_CG,
65  const su2double *coord_Elem_CG,
66  const su2double *coord_Point) {

71 
72  Distance(nDim, coord_Elem_CG, coord_Point, vec_a);
