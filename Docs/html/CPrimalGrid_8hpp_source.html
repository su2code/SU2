187  /*!
188  * \brief Set the center of gravity of an element (including edges).
189  * \param[in] nDim - Number of dimensions (2 or 3).
190  * \param[in] val_coord - Coordinates of the element.
191  */
192  template<class T>
193  inline su2double* SetCoord_CG(unsigned short nDim, const T& val_coord) {
194  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
195  Coord_CG[iDim] = 0.0;
196  for (unsigned short iNode = 0; iNode < GetnNodes(); iNode++)
197  Coord_CG[iDim] += val_coord[iNode][iDim]/su2double(GetnNodes());
198  }
199  return Coord_CG;
200  }

443 /*! \class CPrimalGridWithConnectivity
444  * \brief Override the connectivity getters of CPrimalGrid.
445  *
446  * \details Non-FEM primal grid classes like CLine, CTriangle etc
447  * are derived from CPrimalGridWithConnectivity<CLineConnectivity> etc.
448  * The Connectivity class must have the following static members of
449  * type unsigned short / unsigned short array, typically constexpr:
450  * - nNodes: Number of nodes of the element.
451  * - nFaces: Number of faces of the element.
452  * - nNodesFace[]: Number of nodes of each face of the element.
453  * - maxNodesFace: Maximum number of nodes for a face.
454  * - Faces[][]: Matrix to store the local nodes of all the faces.
455  * - nNeighbor_Nodes[]: Number of neighbor nodes of each node of the element.
456  * - Neighbor_Nodes[][]: Matrix to store the neighbors of all the nodes.
457  * - VTK_Type: Type of element using VTK nomenclature.
458  *
459  * The getter functions for connectivity in CPrimalGrid have final overrides
460  * in this class, accessing the static members of Connectivity.
461  *
462  * \tparam Connectivity - class defining the connectivity structure
463  */
464 template<typename Connectivity>
465 class CPrimalGridWithConnectivity : public CPrimalGrid {
466 public:
467 
468  CPrimalGridWithConnectivity(bool FEM) : CPrimalGrid(FEM, Connectivity::nNodes, Connectivity::nFaces) {}
469 
470  inline unsigned short GetnNodes() const final {
471  return Connectivity::nNodes;
472  }
473 
474  inline unsigned short GetnFaces() const final {
475  return Connectivity::nFaces;
476  }
477 
478  inline unsigned short GetnNodesFace(unsigned short val_face) const final {
479  assert(val_face < Connectivity::nFaces);
480  return Connectivity::nNodesFace[val_face];
481  }
482 
483  inline unsigned short GetMaxNodesFace() const final {
484  return Connectivity::maxNodesFace;
485  }
486 
487  inline unsigned short GetFaces(unsigned short val_face, unsigned short val_index) const final {
488  assert(val_face < GetnFaces() && val_index < GetnNodesFace(val_face));
489  return Connectivity::Faces[val_face][val_index];
490  }
491 
492  inline unsigned short GetnNeighbor_Nodes(unsigned short val_node) const final {
493  assert(val_node < Connectivity::nNodes);
494  return Connectivity::nNeighbor_Nodes[val_node];
495  }
496 
497  inline unsigned short GetNeighbor_Nodes(unsigned short val_node, unsigned short val_index) const final {
498  assert(val_node < Connectivity::nNodes && val_index < GetnNeighbor_Nodes(val_node));
499  return Connectivity::Neighbor_Nodes[val_node][val_index];
500  }
501 
502  inline unsigned short GetVTK_Type() const final {
503  return Connectivity::VTK_Type;
504  }
505 
506 };
