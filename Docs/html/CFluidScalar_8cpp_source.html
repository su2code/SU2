91 void CFluidScalar::MassToMoleFractions(const su2double* val_scalars) {
92 
93  su2double val_scalars_sum{0.0};
94  for (int i_scalar = 0; i_scalar < n_species_mixture - 1; i_scalar++) {
95  massFractions[i_scalar] = val_scalars[i_scalar];
96  val_scalars_sum += val_scalars[i_scalar];
97  }
98  massFractions[n_species_mixture - 1] = 1 - val_scalars_sum;
99 
100  su2double mixtureMolarMass{0.0};
101  for (int iVar = 0; iVar < n_species_mixture; iVar++) {
102  mixtureMolarMass += massFractions[iVar] / molarMasses[iVar];
103  }
104 
105  for (int iVar = 0; iVar < n_species_mixture; iVar++) {
106  moleFractions[iVar] = (massFractions[iVar] / molarMasses[iVar]) / mixtureMolarMass;
107  }
108 }

110 su2double CFluidScalar::WilkeViscosity(const su2double* val_scalars) {
111 
112  /* Fill laminarViscosity with n_species_mixture viscosity values. */
113  for (int iVar = 0; iVar < n_species_mixture; iVar++) {
114  LaminarViscosityPointers[iVar]->SetViscosity(Temperature, Density);
115  laminarViscosity[iVar] = LaminarViscosityPointers[iVar]->GetViscosity();
116  }
117 
118  su2double viscosityMixture = 0.0;
119 
120  for (int i = 0; i < n_species_mixture; i++) {
121  su2double wilkeDenumerator = 0.0;
122  for (int j = 0; j < n_species_mixture; j++) {
123  if (j != i) {
124  const su2double phi =
125  pow(1 + sqrt(laminarViscosity[i] / laminarViscosity[j]) * pow(molarMasses[j] / molarMasses[i], 0.25), 2) /
126  sqrt(8 * (1 + molarMasses[i] / molarMasses[j]));
127  wilkeDenumerator += moleFractions[j] * phi;
128  } else {
129  wilkeDenumerator += moleFractions[j];
130  }
131  }
132  const su2double wilkeNumerator = moleFractions[i] * laminarViscosity[i];
133  viscosityMixture += wilkeNumerator / wilkeDenumerator;
134  }
135  return viscosityMixture;
136 }

138 su2double CFluidScalar::DavidsonViscosity(const su2double* val_scalars) {
139  const su2double A = 0.375;
140 
141  for (int iVar = 0; iVar < n_species_mixture; iVar++) {
142  LaminarViscosityPointers[iVar]->SetViscosity(Temperature, Density);
143  laminarViscosity[iVar] = LaminarViscosityPointers[iVar]->GetViscosity();
144  }
145 
146  su2double mixtureFractionDenumerator = 0.0;
147  for (int i = 0; i < n_species_mixture; i++) {
148  mixtureFractionDenumerator += moleFractions[i] * sqrt(molarMasses[i]);
149  }
150 
151  std::array<su2double, ARRAYSIZE> mixtureFractions;
152  for (int j = 0; j < n_species_mixture; j++) {
153  mixtureFractions[j] = (moleFractions[j] * sqrt(molarMasses[j])) / mixtureFractionDenumerator;
154  }
155 
156  su2double fluidity = 0.0;
157  for (int i = 0; i < n_species_mixture; i++) {
158  for (int j = 0; j < n_species_mixture; j++) {
159  const su2double E = (2 * sqrt(molarMasses[i]) * sqrt(molarMasses[j])) / (molarMasses[i] + molarMasses[j]);
160  fluidity +=
161  ((mixtureFractions[i] * mixtureFractions[j]) / (sqrt(laminarViscosity[i]) * sqrt(laminarViscosity[j]))) *
162  pow(E, A);
163  }
164  }
165  return 1.0 / fluidity;
166 }

195 su2double CFluidScalar::ComputeGasConstant() {
196  su2double MeanMolecularWeight = 0.0;
197 
198  for (int i = 0; i < n_species_mixture; i++) {
199  MeanMolecularWeight += moleFractions[i] * molarMasses[i] / 1000;
200  }
201  Gas_Constant = UNIVERSAL_GAS_CONSTANT / (GasConstant_Ref * MeanMolecularWeight);
202 
203  return Gas_Constant;
204 }

215 void CFluidScalar::SetTDState_T(const su2double val_temperature, const su2double* val_scalars) {
216  MassToMoleFractions(val_scalars);
217  ComputeGasConstant();
218  Temperature = val_temperature;
219  Density = Pressure_Thermodynamic / (Temperature * Gas_Constant);
220  Cp = ComputeMeanSpecificHeatCp(val_scalars);
221  Cv = Cp - Gas_Constant;
222 
223  if (wilke) {
224  Mu = WilkeViscosity(val_scalars);
225  } else if (davidson) {
226  Mu = DavidsonViscosity(val_scalars);
227  }
228 
229  Kt = WilkeConductivity(val_scalars);
