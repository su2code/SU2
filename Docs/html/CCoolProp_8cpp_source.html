34 CCoolProp::CCoolProp(string fluidname) : CFluidModel() {
35  fluid_entity = std::unique_ptr<CoolProp::AbstractState>(CoolProp::AbstractState::factory("HEOS",fluidname));
36  Gas_Constant = fluid_entity->gas_constant()/fluid_entity->molar_mass();
37  Pressure_Critical = fluid_entity->p_critical();
38  Temperature_Critical = fluid_entity->T_critical();

40 }

44 void CCoolProp::SetTDState_rhoe(su2double rho, su2double e) {
45  Density = rho;
46  StaticEnergy = e;
47  fluid_entity->update(CoolProp::DmassUmass_INPUTS, Density, StaticEnergy);
48  Cp = fluid_entity->cpmass();
49  Cv = fluid_entity->cvmass();
50  Gamma = Cp / Cv;
51  Pressure = fluid_entity->p();
52  Temperature = fluid_entity->T();
53  Entropy = fluid_entity->smass();
54  dPdrho_e = fluid_entity->first_partial_deriv(CoolProp::iP, CoolProp::iDmass, CoolProp::iUmass);
55  dPde_rho = fluid_entity->first_partial_deriv(CoolProp::iP, CoolProp::iUmass, CoolProp::iDmass);
56  dTdrho_e = fluid_entity->first_partial_deriv(CoolProp::iT, CoolProp::iDmass, CoolProp::iUmass);
57  dTde_rho = fluid_entity->first_partial_deriv(CoolProp::iT, CoolProp::iUmass, CoolProp::iDmass);
58  if (fluid_entity->phase() == 6) {
59  fluid_entity->specify_phase(CoolProp::iphase_gas);
60  SetTDState_PT(Pressure,Temperature);
61  }
62  else{
63  SoundSpeed2 = pow(fluid_entity->speed_sound(), 2);
64  }
65 }
