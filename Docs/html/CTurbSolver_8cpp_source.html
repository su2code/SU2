103 void CTurbSolver::LoadRestart(CGeometry** geometry, CSolver*** solver, CConfig* config, int val_iter,
104  bool val_update_geo) {
105  /*--- Restart the solution from file information ---*/
106 
107  const string restart_filename = config->GetFilename(config->GetSolution_FileName(), "", val_iter);
108 
109  /*--- To make this routine safe to call in parallel most of it can only be executed by one thread. ---*/
110  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
111  /*--- Read the restart data from either an ASCII or binary SU2 file. ---*/
112 
113  if (config->GetRead_Binary_Restart()) {
114  Read_SU2_Restart_Binary(geometry[MESH_0], config, restart_filename);
115  } else {
116  Read_SU2_Restart_ASCII(geometry[MESH_0], config, restart_filename);
117  }
118 
119  /*--- Skip flow variables ---*/
120 
121  unsigned short skipVars = nDim + solver[MESH_0][FLOW_SOL]->GetnVar();
122 
123  /*--- Adjust the number of solution variables in the incompressible
124  restart. We always carry a space in nVar for the energy equation in the
125  mean flow solver, but we only write it to the restart if it is active.
126  Therefore, we must reduce skipVars here if energy is inactive so that
127  the turbulent variables are read correctly. ---*/
128 
129  const bool incompressible = (config->GetKind_Regime() == ENUM_REGIME::INCOMPRESSIBLE);
130  const bool energy = config->GetEnergy_Equation();
131  const bool weakly_coupled_heat = config->GetWeakly_Coupled_Heat();
132 
133  if (incompressible && ((!energy) && (!weakly_coupled_heat))) skipVars--;
134 
135  /*--- Load data from the restart into correct containers. ---*/
136 
137  unsigned long counter = 0;
138  for (auto iPoint_Global = 0ul; iPoint_Global < geometry[MESH_0]->GetGlobal_nPointDomain(); iPoint_Global++) {
139  /*--- Retrieve local index. If this node from the restart file lives
140  on the current processor, we will load and instantiate the vars. ---*/
141 
142  const auto iPoint_Local = geometry[MESH_0]->GetGlobal_to_Local_Point(iPoint_Global);
143 
144  if (iPoint_Local > -1) {
145  /*--- We need to store this point's data, so jump to the correct
146  offset in the buffer of data from the restart file and load it. ---*/
147 
148  const auto index = counter * Restart_Vars[1] + skipVars;
149  for (auto iVar = 0u; iVar < nVar; iVar++) nodes->SetSolution(iPoint_Local, iVar, Restart_Data[index + iVar]);
150 
151  /*--- Increment the overall counter for how many points have been loaded. ---*/
152  counter++;
153  }
154  }
155 
156  /*--- Detect a wrong solution file ---*/
157 
158  if (counter != nPointDomain) {
159  SU2_MPI::Error(string("The solution file ") + restart_filename + string(" does not match with the mesh file!\n") +
160  string("This can be caused by empty lines at the end of the file."),
161  CURRENT_FUNCTION);
162  }
163 
164  } // end safe global access, pre and postprocessing are thread-safe.
165  END_SU2_OMP_SAFE_GLOBAL_ACCESS
166 
167  /*--- MPI solution and compute the eddy viscosity ---*/
168 
169  solver[MESH_0][TURB_SOL]->InitiateComms(geometry[MESH_0], config, SOLUTION);
170  solver[MESH_0][TURB_SOL]->CompleteComms(geometry[MESH_0], config, SOLUTION);
171 
172  /*--- For turbulent+species simulations the solver Pre-/Postprocessing is done by the species solver. ---*/
173  if (config->GetKind_Species_Model() == SPECIES_MODEL::NONE && config->GetKind_Trans_Model() == TURB_TRANS_MODEL::NONE) {
174  solver[MESH_0][FLOW_SOL]->Preprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0, NO_RK_ITER,
175  RUNTIME_FLOW_SYS, false);
176  solver[MESH_0][TURB_SOL]->Postprocessing(geometry[MESH_0], solver[MESH_0], config, MESH_0);
177  }
178 
179  /*--- Interpolate the solution down to the coarse multigrid levels ---*/
180 
181  for (auto iMesh = 1u; iMesh <= config->GetnMGLevels(); iMesh++) {
182  MultigridRestriction(*geometry[iMesh - 1], solver[iMesh - 1][TURB_SOL]->GetNodes()->GetSolution(),
183  *geometry[iMesh], solver[iMesh][TURB_SOL]->GetNodes()->GetSolution());
184  solver[iMesh][TURB_SOL]->InitiateComms(geometry[iMesh], config, SOLUTION);
185  solver[iMesh][TURB_SOL]->CompleteComms(geometry[iMesh], config, SOLUTION);
186 
187  if (config->GetKind_Species_Model() == SPECIES_MODEL::NONE) {
188  solver[iMesh][FLOW_SOL]->Preprocessing(geometry[iMesh], solver[iMesh], config, iMesh, NO_RK_ITER, RUNTIME_FLOW_SYS,
189  false);
190  solver[iMesh][TURB_SOL]->Postprocessing(geometry[iMesh], solver[iMesh], config, iMesh);
191  }
192  }
193 
194  /*--- Go back to single threaded execution. ---*/
195  BEGIN_SU2_OMP_SAFE_GLOBAL_ACCESS {
196  /*--- Delete the class memory that is used to load the restart. ---*/
197 
198  delete[] Restart_Vars;
199  Restart_Vars = nullptr;
200  delete[] Restart_Data;
201  Restart_Data = nullptr;
202  }
203  END_SU2_OMP_SAFE_GLOBAL_ACCESS
204 }

206 void CTurbSolver::Impose_Fixed_Values(const CGeometry *geometry, const CConfig *config){
207  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
208 
209  /*--- Check whether turbulence quantities are fixed to far-field values on a half-plane. ---*/
210  if(config->GetTurb_Fixed_Values()){
211 
212  /*--- Form normalized far-field velocity ---*/
213  const su2double* velocity_inf = config->GetVelocity_FreeStreamND();
214  su2double velmag_inf = GeometryToolbox::Norm(nDim, velocity_inf);
215  if(velmag_inf==0)
216  SU2_MPI::Error("Far-field velocity is zero, cannot fix turbulence quantities to inflow values.", CURRENT_FUNCTION);
217  su2double unit_velocity_inf[MAXNDIM];
218  for(unsigned short iDim=0; iDim<nDim; iDim++)
219  unit_velocity_inf[iDim] = velocity_inf[iDim] / velmag_inf;
220 
221  SU2_OMP_FOR_DYN(omp_chunk_size)
222  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
223  if( GeometryToolbox::DotProduct(nDim, geometry->nodes->GetCoord(iPoint), unit_velocity_inf)
224  < config->GetTurb_Fixed_Values_MaxScalarProd() ) {
225  /*--- Set the solution values and zero the residual ---*/
226  nodes->SetSolution_Old(iPoint, Solution_Inf);
227  nodes->SetSolution(iPoint, Solution_Inf);
228  LinSysRes.SetBlock_Zero(iPoint);
229  if (implicit) {
230  /*--- Change rows of the Jacobian (includes 1 in the diagonal) ---*/
231  for(unsigned long iVar=0; iVar<nVar; iVar++)
232  Jacobian.DeleteValsRowi(iPoint*nVar+iVar);
233  }
234  }
235  }
236  END_SU2_OMP_FOR
237  }
