32 CInviscidVortexSolution::CInviscidVortexSolution(unsigned short val_nDim,
33  unsigned short val_nVar,
34  unsigned short val_iMesh,
35  CConfig* config)
36  : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
37 
38  /*--- Write a message that the solution is initialized for the
39  inviscid vortex test case. ---*/
40  if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
41  cout << endl;
42  cout << "Warning: Fluid properties and solution are being " << endl;
43  cout << " initialized for the inviscid vortex case!!!" << endl;
44  cout << endl << flush;
45  }
46 
47  /*--- Store the inviscid vortex specific parameters here. ---*/
48  x0Vortex = -0.5; // Initial x-coordinate of the vortex center.
49  y0Vortex = 0.0; // Initial y-coordinate of the vortex center.
50  RVortex = 0.1; // Radius of the vortex.
51  epsVortex = 1.0; // Strength of the vortex.
52 
53  /* Get the Mach number and advection angle (in degrees). */
54  MachVortex = config->GetMach();
55  thetaVortex = config->GetAoA();
56 
57  /*--- Useful coefficients in which Gamma is present. ---*/
58  Gamma = config->GetGamma();
59  Gm1 = Gamma - 1.0;
60  ovGm1 = 1.0/Gm1;
61  gamOvGm1 = ovGm1*Gamma;
62 
63  /*--- Perform some sanity and error checks for this solution here. ---*/
64  if((config->GetTime_Marching() != TIME_MARCHING::TIME_STEPPING) &&
65  (config->GetTime_Marching() != TIME_MARCHING::DT_STEPPING_1ST) &&
66  (config->GetTime_Marching() != TIME_MARCHING::DT_STEPPING_2ND))
67  SU2_MPI::Error("Unsteady mode must be selected for the inviscid vortex",
68  CURRENT_FUNCTION);
69 
70  if(Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::RANS &&
71  Kind_Solver != MAIN_SOLVER::FEM_EULER && Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
72  Kind_Solver != MAIN_SOLVER::FEM_LES)
73  SU2_MPI::Error("Compressible flow equations must be selected for the inviscid vortex",
74  CURRENT_FUNCTION);
75 
76  if((Kind_Solver != MAIN_SOLVER::EULER) &&
77  (Kind_Solver != MAIN_SOLVER::FEM_EULER))
78  SU2_MPI::Error("Euler equations must be selected for the inviscid vortex",
79  CURRENT_FUNCTION);
80 
81  if((config->GetKind_FluidModel() != STANDARD_AIR) &&
82  (config->GetKind_FluidModel() != IDEAL_GAS))
83  SU2_MPI::Error("Standard air or ideal gas must be selected for the inviscid vortex",
84  CURRENT_FUNCTION);
85 
86  if(fabs(config->GetPressure_FreeStreamND() - 1.0) > 1.e-8)
87  SU2_MPI::Error("Free-stream pressure must be 1.0 for the inviscid vortex",
88  CURRENT_FUNCTION);
89 
90  if(fabs(config->GetDensity_FreeStreamND() - 1.0) > 1.e-8)
91  SU2_MPI::Error("Free-stream density must be 1.0 for the inviscid vortex",
92  CURRENT_FUNCTION);
93 }

107 void CInviscidVortexSolution::GetSolution(const su2double *val_coords,
108  const su2double val_t,
109  su2double *val_solution) const {
110 
111  /* Compute the free stream velocities in x- and y-direction. */
112  const su2double VelInf = MachVortex*sqrt(Gamma);
113  const su2double uInf = VelInf*cos(thetaVortex*PI_NUMBER/180.0);
114  const su2double vInf = VelInf*sin(thetaVortex*PI_NUMBER/180.0);
115 
116  /* Compute the coordinates relative to the center of the vortex. */
117  const su2double dx = val_coords[0] - (x0Vortex + val_t*uInf);
118  const su2double dy = val_coords[1] - (y0Vortex + val_t*vInf);
119 
120  /* Compute the components of the velocity. */
121  su2double f = 1.0 - (dx*dx + dy*dy)/(RVortex*RVortex);
122  su2double t1 = epsVortex*dy*exp(0.5*f)/(2.0*PI_NUMBER*RVortex);
123  su2double u = uInf - VelInf*t1;
124 
125  t1 = epsVortex*dx*exp(0.5*f)/(2.0*PI_NUMBER*RVortex);
126  su2double v = vInf + VelInf*t1;
127 
128  /* Compute the density and the pressure. */
129  t1 = 1.0 - epsVortex*epsVortex*Gm1
130  * MachVortex*MachVortex*exp(f)/(8.0*PI_NUMBER*PI_NUMBER);
131 
132  su2double rho = pow(t1,ovGm1);
133  su2double p = pow(t1,gamOvGm1);
134 
135  /* Compute the conservative variables. Note that both 2D and 3D
136  cases are treated correctly. */
137  val_solution[0] = rho;
138  val_solution[1] = rho*u;
139  val_solution[2] = rho*v;
140  val_solution[3] = 0.0;
141  val_solution[nVar-1] = p*ovGm1 + 0.5*rho*(u*u + v*v);
142 }
