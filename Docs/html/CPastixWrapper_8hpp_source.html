93  /*!
94  * \brief Run the external solver for the task it is currently setup to execute.
95  */
96  void Run() {
97  dpastix(&state, SU2_MPI::GetComm(), nCols, colptr.data(), rowidx.data(), values.data(),
98  loc2glb.data(), perm.data(), NULL, workvec.data(), 1, iparm, dparm);
99  }

101  /*!
102  * \brief Run the "clean" task, releases all memory, leaves object in unusable state.
103  */
104  void Clean() {
105  using namespace PaStiX;
106  if(isfactorized) {
107  iparm[IPARM_VERBOSE] = (verb > 0) ? API_VERBOSE_NO : API_VERBOSE_NOT;
108  iparm[IPARM_START_TASK] = API_TASK_CLEAN;
109  iparm[IPARM_END_TASK] = API_TASK_CLEAN;
110  Run();
111  isfactorized = false;
112  }
113  }

184  /*!
185  * \brief Runs the "solve" task for any rhs/sol with operator []
186  * \param[in] rhs - Right hand side of the linear system.
187  * \param[out] sol - Solution of the system.
188  */
189  template<class T>
190  void Solve(const T& rhs, T& sol) {
191  using namespace PaStiX;
192 
193  if (!isfactorized)
194  SU2_MPI::Error("The factorization has not been computed yet.", CURRENT_FUNCTION);
195 
196  unsigned long i;
197 
198  for (i=0; i < matrix.size_rhs(); ++i) workvec[i] = rhs[i];
199 
200  iparm[IPARM_VERBOSE] = API_VERBOSE_NOT;
201  iparm[IPARM_START_TASK] = API_TASK_SOLVE;
202  iparm[IPARM_END_TASK] = API_TASK_SOLVE;
203  Run();
204 
205  for (i=0; i < matrix.size_rhs(); ++i) sol[i] = workvec[i];
206  }
