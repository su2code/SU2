43 COutput* COutputFactory::CreateOutput(MAIN_SOLVER kindSolver, CConfig* config, int nDim){
44 
45  COutput* output = nullptr;
46 
47  switch(kindSolver){
48  case MAIN_SOLVER::EULER: case MAIN_SOLVER::NAVIER_STOKES: case MAIN_SOLVER::RANS:
49  output = new CFlowCompOutput(config, nDim);
50  break;
51  case MAIN_SOLVER::INC_EULER: case MAIN_SOLVER::INC_NAVIER_STOKES: case MAIN_SOLVER::INC_RANS:
52  output = new CFlowIncOutput(config, nDim);
53  break;
54  case MAIN_SOLVER::NEMO_EULER: case MAIN_SOLVER::NEMO_NAVIER_STOKES:
55  output = new CNEMOCompOutput(config, nDim);
56  break;
57  case MAIN_SOLVER::HEAT_EQUATION:
58  output = new CHeatOutput(config, nDim);
59  break;
60  case MAIN_SOLVER::FEM_ELASTICITY:
61  output = new CElasticityOutput(config, nDim);
62  break;
63  case MAIN_SOLVER::DISC_ADJ_EULER: case MAIN_SOLVER::DISC_ADJ_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_RANS:
64  case MAIN_SOLVER::ADJ_EULER: case MAIN_SOLVER::ADJ_NAVIER_STOKES: case MAIN_SOLVER::ADJ_RANS:
65  output = new CAdjFlowCompOutput(config, nDim);
66  break;
67  case MAIN_SOLVER::DISC_ADJ_INC_EULER: case MAIN_SOLVER::DISC_ADJ_INC_NAVIER_STOKES: case MAIN_SOLVER::DISC_ADJ_INC_RANS:
68  output = new CAdjFlowIncOutput(config, nDim);
69  break;
70  case MAIN_SOLVER::DISC_ADJ_FEM:
71  output = new CAdjElasticityOutput(config, nDim);
72  break;
73  case MAIN_SOLVER::DISC_ADJ_HEAT:
74  output = new CAdjHeatOutput(config, nDim);
75  break;
76  case MAIN_SOLVER::FEM_EULER: case MAIN_SOLVER::FEM_LES: case MAIN_SOLVER::FEM_RANS: case MAIN_SOLVER::FEM_NAVIER_STOKES:
77  output = new CFlowCompFEMOutput(config, nDim);
78  break;
79  default:
80  output = new COutput(config, nDim, false);
81  break;
82  }
83 
84  return output;
85 }
