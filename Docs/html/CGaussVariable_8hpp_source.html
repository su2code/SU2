38 class CGaussVariable {
39 protected:
40 
41  su2activematrix GradNi_Xj; /*!< \brief Gradient of the shape functions N[i] wrt the reference configuration. */
42  su2activematrix GradNi_xj; /*!< \brief Gradient of the shape functions N[i] wrt the current configuration. */
43  su2activevector Ni; /*!< \brief Shape functions N[i] at the gaussian point. */
44  su2double J_X = 0.0; /*!< \brief Element Jacobian evaluated at this Gauss Point wrt the reference configuration. */
45  su2double J_x = 0.0; /*!< \brief Element Jacobian evaluated at this Gauss Point wrt the current configuration. */
46  unsigned short iGaussPoint = 0; /*!< \brief Identifier of the Gauss point considered. */
47 
48 public:
49  /*!
50  * \brief Deleted default constructor as this class does not allow resizing once created.
51  */
52  CGaussVariable() = delete;
53 
54  /*!
55  * \brief Class constructor
56  * \param[in] val_iGauss - ID of the Gaussian Point
57  * \param[in] val_nDim - Number of dimensions of the problem.
58  * \param[in] config - Definition of the particular problem.
59  */
60  CGaussVariable(unsigned short val_iGauss, unsigned short val_nDim, unsigned short val_nNodes)
61  : J_X(0.0), J_x(0.0), iGaussPoint(val_iGauss)
62  {
63 
64  /* --- For the structural mechanics solver the dimensions (nNodes x nDim) are sufficient.
65  * For the Sobolev smoothing solver dimensions (nNodes x (nDim+1)) are necessary
66  * when working on a curved design surface embedded in 3D. ---*/
67  GradNi_Xj.resize(val_nNodes,val_nDim+1) = su2double(0.0);
68  GradNi_xj = GradNi_Xj;
69 
70  Ni.resize(val_nNodes) = su2double(0.0);
71  }
72 
73  /*!
74  * \brief Destructor of the class.
75  */
76  ~CGaussVariable(void) = default;
77

80  }
81 
82  inline void SetGradNi_xj(su2double val_GradNi_xj, unsigned short val_iDim, unsigned short val_Ni) {

84  }
85

87 
88  inline void SetJ_X(su2double valJ_X) { J_X = valJ_X; }
89 
90  inline void SetJ_x(su2double valJ_x) { J_x = valJ_x; }
91 
92

96

99  }
100

102 
103  inline su2double GetJ_X(void) const { return J_X; }
104

106
