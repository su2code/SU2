36 /*!
37  * \class CFastFindAndEraseQueue
38  * \ingroup Containers
39  * \brief A queue-type container (push back, pop front), but with
40  * fast deletion of arbitrary items (possibly in the middle).
41  * \param[in] ItemType_ - Type of the stored items.
42  * \param[in] ErasedValue_ - Value used to mark items for erasure.
43  * \param[in] CleanupThreshold - Number of marked items that triggers full cleanup.
44  * \note It would not be a good idea to use non-trivial item types.
45  */
46 template<typename ItemType_ = unsigned long,
47  ItemType_ ErasedValue_ = std::numeric_limits<ItemType_>::max(),
48  size_t CleanupThreshold = 8192>
49 class CFastFindAndEraseQueue {
50 public:
51  enum {ErasedValue = ErasedValue_};
52  using ItemType = ItemType_;
53  using Iterator = typename std::vector<ItemType>::const_iterator;
54 
55 private:
56  size_t erasedCounter = 0; /*!< \brief How many items have been marked since last cleanup. */
57  std::vector<ItemType> items; /*!< \brief The stored items. */
58  std::unordered_map<ItemType,size_t> indexes; /*!< \brief Map items to their location. */
59 
60  /*!
61  * \brief Cleanup, shifts non-erased items forward, re-mapping them.
62  * \note This is the performance hotspot (due to the re-mapping).
63  */
64  void cleanup() {
65  size_t idx = 0;
66  for (auto value : items) {
67  if (value != ErasedValue) {
68  items[idx] = value;
69  indexes[value] = idx;
70  ++idx;
71  }
72  }
73  items.resize(idx);
74  erasedCounter = 0;
75  }
76 
77 public:
78  /*!
79  * \brief Default construct.
80  */
81  CFastFindAndEraseQueue() = default;
82 
83  /*!
84  * \brief Construct and initialize with [0,N) range.
85  */
86  CFastFindAndEraseQueue(size_t N) {
87  items.resize(N);
88  for (size_t i=0; i<N; ++i) {
89  items[i] = i;
90  indexes[i] = i;
91  }
92  }
93 
94  /*--- Wrapper methods for stl-type operations. ---*/
95  size_t size() const { return indexes.size(); }
96  bool empty() const { return indexes.empty(); }
97  Iterator begin() const { return items.cbegin(); }
98  Iterator end() const { return items.cend(); }
99 
100  /*!
101  * \brief Push back an item, that should be unique.
102  * \param[in] value - The new item.
103  */
104  void push_back(ItemType value) {
105  assert(value != ErasedValue && "#bad_idea");
106  assert(!indexes.count(value) && "Items must be unique.");
107  indexes[value] = items.size();
108  items.push_back(value);
109  }
110 
111  /*!
112  * \brief Get the item in front of the queue.
113  */
114  ItemType front() const {
115  for (auto value : items) if (value != ErasedValue) return value;
116  assert(false && "Queue is empty.");
117  return ErasedValue;
118  }
119 
120  /*!
121  * \brief Main event, look for an item and erase it if present.
122  * \param[in] value - Item to look for.
123  * \return True, if something was erased, false otherwise.
124  */
125  bool findAndErase(const ItemType value) {
126  if (empty()) return false;
127 
128  /*--- Get the item position. ---*/
129  auto it = indexes.find(value);
130  if (it == indexes.end()) return false; // value is not here
131 
132  const auto idx = it->second;
133  assert(items[idx] != ErasedValue && "The item was already erased?!");
134 
135  /*--- Erase and cleanup items. ---*/
136  indexes.erase(it);
137  items[idx] = ErasedValue;
138  if (++erasedCounter == CleanupThreshold) cleanup();
139 
140  return true;
141  }
142 
143 };
