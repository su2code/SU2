40 CIncEulerSolver::CIncEulerSolver(CGeometry *geometry, CConfig *config, unsigned short iMesh,
41  const bool navier_stokes) :

44  /*--- Based on the navier_stokes boolean, determine if this constructor is
45  * being called by itself, or by its derived class CIncNSSolver. ---*/
46  const string description = navier_stokes? "Navier-Stokes" : "Euler";

48  unsigned short iMarker;
49  ifstream restart_file;
50  unsigned short nZone = geometry->GetnZone();
51  bool restart = (config->GetRestart() || config->GetRestart_Flow());
52  int Unst_RestartIter = 0;

69

71 
72  auto filename_ = config->GetSolution_FileName();
73

77

80  else if (config->GetTime_Marching() == TIME_MARCHING::DT_STEPPING_1ST)

82  else Unst_RestartIter = SU2_TYPE::Int(config->GetRestart_Iter())-2;
83  filename_ = config->GetUnsteady_FileName(filename_, Unst_RestartIter, ".dat");
84  }
85 
86  /*--- Modify file name for a time stepping unsteady restart ---*/
87 
88  if (time_stepping) {
89  if (adjoint) Unst_RestartIter = SU2_TYPE::Int(config->GetUnst_AdjointIter())-1;
90  else Unst_RestartIter = SU2_TYPE::Int(config->GetRestart_Iter())-1;
91  filename_ = config->GetUnsteady_FileName(filename_, Unst_RestartIter, ".dat");
92  }

94  /*--- Read and store the restart metadata. ---*/
95 
96  filename_ = "flow";
97  filename_ = config->GetFilename(filename_, ".meta", Unst_RestartIter);
98  Read_SU2_Restart_Metadata(geometry, config, adjoint, filename_);
99 
100  }

102  string filename_ = "flow";
103  filename_ = config->GetFilename(filename_, ".meta", Unst_RestartIter);

116  nDim = geometry->GetnDim();

138  SetNondimensionalization(config, iMesh);

140  /*--- Check if we are executing a verification case. If so, the
141  VerificationSolution object will be instantiated for a particular

143  that this is done after SetNondim(), as problem-specific initial
144  parameters are needed by the solution constructors. ---*/

146  SetVerificationSolution(nDim, nVar, config);

148  /*--- Allocate base class members. ---*/
149 
150  Allocate(*config);
151 
152  /*--- MPI + OpenMP initialization. ---*/
153 
154  HybridParallelInitialization(*config, *geometry);
155

157 
158  if (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT) {
159 
160  if (rank == MASTER_NODE)
161  cout << "Initialize Jacobian structure (" << description << "). MG level: " << iMesh <<"." << endl;

163  Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config, ReducerStrategy);

165  else {
166  if (rank == MASTER_NODE)
167  cout << "Explicit scheme. No Jacobian structure (" << description << "). MG level: " << iMesh <<"." << endl;
168  }
169

171 
172  Density_Inf = config->GetDensity_FreeStreamND();
173  Pressure_Inf = config->GetPressure_FreeStreamND();
174  Velocity_Inf = config->GetVelocity_FreeStreamND();
175  Temperature_Inf = config->GetTemperature_FreeStreamND();
176 
177  /*--- Initialize the secondary values for direct derivative approxiations ---*/
178 
179  switch (config->GetDirectDiff()) {
180  case NO_DERIVATIVE:
181  /*--- Default ---*/
182  break;
183  case D_DENSITY:
184  SU2_TYPE::SetDerivative(Density_Inf, 1.0);
185  break;
186  case D_PRESSURE:
187  SU2_TYPE::SetDerivative(Pressure_Inf, 1.0);
188  break;
189  case D_TEMPERATURE:
190  SU2_TYPE::SetDerivative(Temperature_Inf, 1.0);
191  break;
192  case D_MACH: case D_AOA:
193  case D_SIDESLIP: case D_REYNOLDS:
194  case D_TURB2LAM: case D_DESIGN:
195  /*--- Already done in postprocessing of config ---*/
196  break;
197  default:
198  break;
199  }
200 
201  SetReferenceValues(*config);
202 
203  /*--- Initialize the solution to the far-field state everywhere. ---*/
204 
205  if (navier_stokes) {
206  nodes = new CIncNSVariable(Pressure_Inf, Velocity_Inf, Temperature_Inf, nPoint, nDim, nVar, config);
207  } else {
208  nodes = new CIncEulerVariable(Pressure_Inf, Velocity_Inf, Temperature_Inf, nPoint, nDim, nVar, config);
209  }
210  SetBaseClassPointerToNodes();
211 
212  if (iMesh == MESH_0) {
213  nodes->NonPhysicalEdgeCounter.resize(geometry->GetnEdge()) = 0;
214  }
215 
216  /*--- Initial comms. ---*/
217 
218  CommunicateInitialState(geometry, config);
219 
220  /*--- Sizing edge mass flux array ---*/

224  /*--- Add the solver name (max 8 characters) ---*/

229  if((nDim > MAXNDIM) || (nPrimVar > MAXNVAR))
230  SU2_MPI::Error("Oops! The CIncEulerSolver static array sizes are not large enough.", CURRENT_FUNCTION);
231

239 void CIncEulerSolver::SetNondimensionalization(CConfig *config, unsigned short iMesh) {
240 
241  su2double Temperature_FreeStream = 0.0, ModVel_FreeStream = 0.0,Energy_FreeStream = 0.0,
242  ModVel_FreeStreamND = 0.0, Omega_FreeStream = 0.0, Omega_FreeStreamND = 0.0, Viscosity_FreeStream = 0.0,
243  Density_FreeStream = 0.0, Pressure_FreeStream = 0.0, Pressure_Thermodynamic = 0.0, Tke_FreeStream = 0.0,
244  Length_Ref = 0.0, Density_Ref = 0.0, Pressure_Ref = 0.0, Temperature_Ref = 0.0, Velocity_Ref = 0.0, Time_Ref = 0.0,
245  Gas_Constant_Ref = 0.0, Omega_Ref = 0.0, Force_Ref = 0.0, Viscosity_Ref = 0.0, Conductivity_Ref = 0.0, Heat_Flux_Ref = 0.0, Energy_Ref= 0.0, Pressure_FreeStreamND = 0.0, Pressure_ThermodynamicND = 0.0, Density_FreeStreamND = 0.0,
246  Temperature_FreeStreamND = 0.0, Gas_ConstantND = 0.0, Specific_Heat_CpND = 0.0, Thermal_Expansion_CoeffND = 0.0,
247  Velocity_FreeStreamND[3] = {0.0, 0.0, 0.0}, Viscosity_FreeStreamND = 0.0,
248  Tke_FreeStreamND = 0.0, Energy_FreeStreamND = 0.0,
249  Total_UnstTimeND = 0.0, Delta_UnstTimeND = 0.0;
250 
251  unsigned short iDim, iVar;
252 
253  /*--- Local variables ---*/
254 
255  su2double Mach = config->GetMach();
256  su2double Reynolds = config->GetReynolds();
257 
258  bool unsteady = (config->GetTime_Marching() != TIME_MARCHING::STEADY);
259  bool viscous = config->GetViscous();
260  bool turbulent = ((config->GetKind_Solver() == MAIN_SOLVER::INC_RANS) ||
261  (config->GetKind_Solver() == MAIN_SOLVER::DISC_ADJ_INC_RANS));
262  bool tkeNeeded = ((turbulent) && ((config->GetKind_Turb_Model() == TURB_MODEL::SST)));
263  bool energy = config->GetEnergy_Equation();
264  bool boussinesq = (config->GetKind_DensityModel() == INC_DENSITYMODEL::BOUSSINESQ);
265 
266  /*--- Compute dimensional free-stream values. ---*/
267 
268  Density_FreeStream = config->GetInc_Density_Init(); config->SetDensity_FreeStream(Density_FreeStream);
269  Temperature_FreeStream = config->GetInc_Temperature_Init(); config->SetTemperature_FreeStream(Temperature_FreeStream);
270  Pressure_FreeStream = 0.0; config->SetPressure_FreeStream(Pressure_FreeStream);
271 
272  /*--- The dimensional viscosity is needed to determine the free-stream conditions.
273  To accomplish this, simply set the non-dimensional coefficients to the
274  dimensional ones. This will be overruled later.---*/
275 
276  config->SetTemperature_Ref(1.0);
277  config->SetViscosity_Ref(1.0);
278  config->SetConductivity_Ref(1.0);
279  config->SetGas_Constant_Ref(1.0);
280 
281  ModVel_FreeStream = 0.0;
282  for (iDim = 0; iDim < nDim; iDim++) {
283  ModVel_FreeStream += config->GetInc_Velocity_Init()[iDim]*config->GetInc_Velocity_Init()[iDim];
284  config->SetVelocity_FreeStream(config->GetInc_Velocity_Init()[iDim],iDim);
285  }
286  ModVel_FreeStream = sqrt(ModVel_FreeStream); config->SetModVel_FreeStream(ModVel_FreeStream);
287 
288  CFluidModel* auxFluidModel = nullptr;
289 
290  switch (config->GetKind_FluidModel()) {
291 
292  case CONSTANT_DENSITY:
293 
294  auxFluidModel = new CConstantDensity(Density_FreeStream, config->GetSpecific_Heat_Cp());
295  auxFluidModel->SetTDState_T(Temperature_FreeStream);
296  break;
297 
298  case INC_IDEAL_GAS:
299 
300  config->SetGas_Constant(UNIVERSAL_GAS_CONSTANT/(config->GetMolecular_Weight()/1000.0));
301  Pressure_Thermodynamic = Density_FreeStream*Temperature_FreeStream*config->GetGas_Constant();
302  auxFluidModel = new CIncIdealGas(config->GetSpecific_Heat_Cp(), config->GetGas_Constant(), Pressure_Thermodynamic);
303  auxFluidModel->SetTDState_T(Temperature_FreeStream);
304  Pressure_Thermodynamic = auxFluidModel->GetPressure();
305  config->SetPressure_Thermodynamic(Pressure_Thermodynamic);
306  break;
307 
308  case INC_IDEAL_GAS_POLY:
309 
310  config->SetGas_Constant(UNIVERSAL_GAS_CONSTANT/(config->GetMolecular_Weight()/1000.0));
311  Pressure_Thermodynamic = Density_FreeStream*Temperature_FreeStream*config->GetGas_Constant();
312  auxFluidModel = new CIncIdealGasPolynomial<N_POLY_COEFFS>(config->GetGas_Constant(), Pressure_Thermodynamic);
313  if (viscous) {
314  /*--- Variable Cp model via polynomial. ---*/
315  for (iVar = 0; iVar < config->GetnPolyCoeffs(); iVar++)
316  config->SetCp_PolyCoeffND(config->GetCp_PolyCoeff(iVar), iVar);
317  auxFluidModel->SetCpModel(config);
318  }
319  auxFluidModel->SetTDState_T(Temperature_FreeStream);
320  Pressure_Thermodynamic = auxFluidModel->GetPressure();
321  config->SetPressure_Thermodynamic(Pressure_Thermodynamic);
322  break;
323 
324  case FLUID_MIXTURE:
325 
326  config->SetGas_Constant(UNIVERSAL_GAS_CONSTANT / (config->GetMolecular_Weight() / 1000.0));
327  Pressure_Thermodynamic = config->GetPressure_Thermodynamic();
328  auxFluidModel = new CFluidScalar(config->GetSpecific_Heat_Cp(), config->GetGas_Constant(), Pressure_Thermodynamic, config);
329  auxFluidModel->SetTDState_T(Temperature_FreeStream, config->GetSpecies_Init());
330  break;
331 
332  default:
333 
334  SU2_MPI::Error("Fluid model not implemented for incompressible solver.", CURRENT_FUNCTION);
335  break;
336  }
337 
338  if (viscous) {
339 
340  for (iVar = 0; iVar < config->GetnPolyCoeffs(); iVar++)
341  config->SetMu_PolyCoeffND(config->GetMu_PolyCoeff(iVar), iVar);
342 
343  /*--- Use the fluid model to compute the dimensional viscosity/conductivity. ---*/
344 
345  auxFluidModel->SetLaminarViscosityModel(config);
346  Viscosity_FreeStream = auxFluidModel->GetLaminarViscosity();
347  config->SetViscosity_FreeStream(Viscosity_FreeStream);
348 
349  Reynolds = Density_FreeStream*ModVel_FreeStream/Viscosity_FreeStream; config->SetReynolds(Reynolds);
350 
351  /*--- Turbulence kinetic energy ---*/
352 
353  Tke_FreeStream = 3.0/2.0*(ModVel_FreeStream*ModVel_FreeStream*config->GetTurbulenceIntensity_FreeStream()*config->GetTurbulenceIntensity_FreeStream());
354 
355  }
356 
357  /*--- The non-dim. scheme for incompressible flows uses the following ref. values:
358  Reference length = 1 m (fixed by default, grid in meters)
359  Reference density = liquid density or freestream (input)
360  Reference velocity = liquid velocity or freestream (input)
361  Reference temperature = liquid temperature or freestream (input)
362  Reference pressure = Reference density * Reference velocity * Reference velocity
363  Reference viscosity = Reference Density * Reference velocity * Reference length
364  This is the same non-dim. scheme as in the compressible solver.
365  Note that the Re and Re Length are not used as part of initialization. ---*/
366 
367  if (config->GetRef_Inc_NonDim() == DIMENSIONAL) {
368  Density_Ref = 1.0;
369  Velocity_Ref = 1.0;
370  Temperature_Ref = 1.0;
371  Pressure_Ref = 1.0;
372  }
373  else if (config->GetRef_Inc_NonDim() == INITIAL_VALUES) {
374  Density_Ref = Density_FreeStream;
375  Velocity_Ref = ModVel_FreeStream;
376  Temperature_Ref = Temperature_FreeStream;
377  Pressure_Ref = Density_Ref*Velocity_Ref*Velocity_Ref;
378  }
379  else if (config->GetRef_Inc_NonDim() == REFERENCE_VALUES) {
380  Density_Ref = config->GetInc_Density_Ref();
381  Velocity_Ref = config->GetInc_Velocity_Ref();
382  Temperature_Ref = config->GetInc_Temperature_Ref();
383  Pressure_Ref = Density_Ref*Velocity_Ref*Velocity_Ref;
384  }
385  config->SetDensity_Ref(Density_Ref);
386  config->SetVelocity_Ref(Velocity_Ref);
387  config->SetTemperature_Ref(Temperature_Ref);
388  config->SetPressure_Ref(Pressure_Ref);
389 
390  /*--- More derived reference values ---*/
391 
392  Length_Ref = 1.0; config->SetLength_Ref(Length_Ref);
393  Time_Ref = Length_Ref/Velocity_Ref; config->SetTime_Ref(Time_Ref);
394  Omega_Ref = Velocity_Ref/Length_Ref; config->SetOmega_Ref(Omega_Ref);
395  Force_Ref = Velocity_Ref*Velocity_Ref/Length_Ref; config->SetForce_Ref(Force_Ref);
396  Heat_Flux_Ref = Density_Ref*Velocity_Ref*Velocity_Ref*Velocity_Ref; config->SetHeat_Flux_Ref(Heat_Flux_Ref);
397  Gas_Constant_Ref = Velocity_Ref*Velocity_Ref/Temperature_Ref; config->SetGas_Constant_Ref(Gas_Constant_Ref);
398  Viscosity_Ref = Density_Ref*Velocity_Ref*Length_Ref; config->SetViscosity_Ref(Viscosity_Ref);
399  Conductivity_Ref = Viscosity_Ref*Gas_Constant_Ref; config->SetConductivity_Ref(Conductivity_Ref);
400 
401  /*--- Get the freestream energy. Only useful if energy equation is active. ---*/
402 
403  Energy_FreeStream = auxFluidModel->GetStaticEnergy() + 0.5*ModVel_FreeStream*ModVel_FreeStream;
404  if (tkeNeeded) { Energy_FreeStream += Tke_FreeStream; };
405  config->SetEnergy_FreeStream(Energy_FreeStream);
406 
407  /*--- Compute Mach number ---*/
408 
409  if (config->GetKind_FluidModel() == CONSTANT_DENSITY) {
410  Mach = ModVel_FreeStream / sqrt(config->GetBulk_Modulus()/Density_FreeStream);
411  } else {
412  Mach = 0.0;
413  }
414  config->SetMach(Mach);
415 
416  /*--- Divide by reference values, to compute the non-dimensional free-stream values ---*/
417 
418  Pressure_FreeStreamND = Pressure_FreeStream/config->GetPressure_Ref(); config->SetPressure_FreeStreamND(Pressure_FreeStreamND);
419  Pressure_ThermodynamicND = Pressure_Thermodynamic/config->GetPressure_Ref(); config->SetPressure_ThermodynamicND(Pressure_ThermodynamicND);
420  Density_FreeStreamND = Density_FreeStream/config->GetDensity_Ref(); config->SetDensity_FreeStreamND(Density_FreeStreamND);
421 
422  for (iDim = 0; iDim < nDim; iDim++) {
423  Velocity_FreeStreamND[iDim] = config->GetVelocity_FreeStream()[iDim]/Velocity_Ref; config->SetVelocity_FreeStreamND(Velocity_FreeStreamND[iDim], iDim);
424  }
425 
426  Temperature_FreeStreamND = Temperature_FreeStream/config->GetTemperature_Ref(); config->SetTemperature_FreeStreamND(Temperature_FreeStreamND);
427  Gas_ConstantND = config->GetGas_Constant()/Gas_Constant_Ref; config->SetGas_ConstantND(Gas_ConstantND);
428  Specific_Heat_CpND = config->GetSpecific_Heat_CpND();
429 
430  Thermal_Expansion_CoeffND = config->GetThermal_Expansion_Coeff()*config->GetTemperature_Ref(); config->SetThermal_Expansion_CoeffND(Thermal_Expansion_CoeffND);
431 
432  ModVel_FreeStreamND = 0.0;
433  for (iDim = 0; iDim < nDim; iDim++) ModVel_FreeStreamND += Velocity_FreeStreamND[iDim]*Velocity_FreeStreamND[iDim];
434  ModVel_FreeStreamND = sqrt(ModVel_FreeStreamND); config->SetModVel_FreeStreamND(ModVel_FreeStreamND);
435 
436  Viscosity_FreeStreamND = Viscosity_FreeStream / Viscosity_Ref; config->SetViscosity_FreeStreamND(Viscosity_FreeStreamND);
437 
438  Tke_FreeStream = 3.0/2.0*(ModVel_FreeStream*ModVel_FreeStream*config->GetTurbulenceIntensity_FreeStream()*config->GetTurbulenceIntensity_FreeStream());
439  config->SetTke_FreeStream(Tke_FreeStream);
440 
441  Tke_FreeStreamND = 3.0/2.0*(ModVel_FreeStreamND*ModVel_FreeStreamND*config->GetTurbulenceIntensity_FreeStream()*config->GetTurbulenceIntensity_FreeStream());
442  config->SetTke_FreeStreamND(Tke_FreeStreamND);
443 
444  Omega_FreeStream = Density_FreeStream*Tke_FreeStream/(Viscosity_FreeStream*config->GetTurb2LamViscRatio_FreeStream());
445  config->SetOmega_FreeStream(Omega_FreeStream);
446 
447  Omega_FreeStreamND = Density_FreeStreamND*Tke_FreeStreamND/(Viscosity_FreeStreamND*config->GetTurb2LamViscRatio_FreeStream());
448  config->SetOmega_FreeStreamND(Omega_FreeStreamND);
449 
450  const su2double MassDiffusivityND = config->GetDiffusivity_Constant() / (Velocity_Ref * Length_Ref);
451  config->SetDiffusivity_ConstantND(MassDiffusivityND);
452 
453 
454  /*--- Delete the original (dimensional) FluidModel object. No fluid is used for inscompressible cases. ---*/
455 
456  delete auxFluidModel;
457 
458  /*--- Create one final fluid model object per OpenMP thread to be able to use them in parallel.
459  * GetFluidModel() should be used to automatically access the "right" object of each thread. ---*/
460 
461  assert(FluidModel.empty() && "Potential memory leak!");
462  FluidModel.resize(omp_get_max_threads());
463 
464  for (auto& fluidModel : FluidModel) {
465 
466  switch (config->GetKind_FluidModel()) {
467 
468  case CONSTANT_DENSITY:
469  fluidModel = new CConstantDensity(Density_FreeStreamND, Specific_Heat_CpND);
470  break;
471 
472  case INC_IDEAL_GAS:
473  fluidModel = new CIncIdealGas(Specific_Heat_CpND, Gas_ConstantND, Pressure_ThermodynamicND);
474  fluidModel->SetTDState_T(Temperature_FreeStreamND);
475  break;
476 
477  case FLUID_MIXTURE:
478  fluidModel = new CFluidScalar(Specific_Heat_CpND, Gas_ConstantND, Pressure_ThermodynamicND, config);
479  fluidModel->SetTDState_T(Temperature_FreeStreamND, config->GetSpecies_Init());
480  break;
481 
482  case INC_IDEAL_GAS_POLY:
483  fluidModel = new CIncIdealGasPolynomial<N_POLY_COEFFS>(Gas_ConstantND, Pressure_ThermodynamicND);
484  if (viscous) {
485  /*--- Variable Cp model via polynomial. ---*/
486  config->SetCp_PolyCoeffND(config->GetCp_PolyCoeff(0)/Gas_Constant_Ref, 0);
487  for (iVar = 1; iVar < config->GetnPolyCoeffs(); iVar++)
488  config->SetCp_PolyCoeffND(config->GetCp_PolyCoeff(iVar)*pow(Temperature_Ref,iVar)/Gas_Constant_Ref, iVar);
489  fluidModel->SetCpModel(config);
490  }
491  fluidModel->SetTDState_T(Temperature_FreeStreamND);
492  break;
493 
494  }
495 
496  if (viscous) {
497 
498  /*--- Viscosity model via polynomial. ---*/

1276  Primitive_i[iVar] = V_i[iVar];
1277  Primitive_j[iVar] = V_j[iVar];
1278  }
1279  }
1280  }
1281

1283 
1284  } else {
1285

1287 
1288  numerics->SetPrimitive(V_i, V_j);
1289 
1290  }
1291 
1292  /*--- Compute the residual ---*/
1293 
1294  auto residual = numerics->ComputeResidual(config);
1295 
1296  if (bounded_scalar) EdgeMassFluxes[iEdge] = residual[0];
1297 
1298  /*--- Update residual value ---*/
1299 
1300  if (ReducerStrategy) {
1301  EdgeFluxes.SetBlock(iEdge, residual);
1302  if (implicit)
1303  Jacobian.SetBlocks(iEdge, residual.jacobian_i, residual.jacobian_j);
1304  }
1305  else {
1306  LinSysRes.AddBlock(iPoint, residual);
1307  LinSysRes.SubtractBlock(jPoint, residual);
1308 
1309  /*--- Set implicit computation ---*/
1310  if (implicit)
1311  Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
1312  }
1313

1315 
1316  Viscous_Residual(iEdge, geometry, solver_container,
1317  numerics_container[VISC_TERM + omp_get_thread_num()*MAX_TERMS], config);
1318

1966 void CIncEulerSolver::SetPreconditioner(const CConfig *config, unsigned long iPoint,
1967  su2double delta, su2activematrix& Preconditioner) const {
1968 
1969  unsigned short iDim, jDim, iVar, jVar;
1970 
1971  su2double BetaInc2, Density, dRhodT, Temperature, oneOverCp, Cp;
1972  su2double Velocity[MAXNDIM] = {0.0};
1973 
1974  bool variable_density = (config->GetKind_DensityModel() == INC_DENSITYMODEL::VARIABLE);
1975  bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
1976  bool energy = config->GetEnergy_Equation();
1977 
1978  /*--- Access the primitive variables at this node. ---*/
1979 
1980  Density = nodes->GetDensity(iPoint);
1981  BetaInc2 = nodes->GetBetaInc2(iPoint);
1982  Cp = nodes->GetSpecificHeatCp(iPoint);
1983  oneOverCp = 1.0/Cp;
1984  Temperature = nodes->GetTemperature(iPoint);
1985 
1986  for (iDim = 0; iDim < nDim; iDim++)
1987  Velocity[iDim] = nodes->GetVelocity(iPoint,iDim);
1988 
1989  /*--- We need the derivative of the equation of state to build the
1990  preconditioning matrix. For now, the only option is the ideal gas
1991  law, but in the future, dRhodT should be in the fluid model. ---*/
1992 
1993  if (variable_density) {
1994  dRhodT = -Density/Temperature;
1995  } else {
1996  dRhodT = 0.0;
1997  }
1998 
1999  /*--- Calculating the inverse of the preconditioning matrix
2000  that multiplies the time derivative during time integration. ---*/
2001 
2002  if (implicit) {
2003 
2004  /*--- For implicit calculations, we multiply the preconditioner
2005  by the cell volume over the time step and add to the Jac diagonal. ---*/
2006 
2007  Preconditioner[0][0] = 1.0/BetaInc2;
2008  for (iDim = 0; iDim < nDim; iDim++)
2009  Preconditioner[iDim+1][0] = Velocity[iDim]/BetaInc2;
2010 
2011  if (energy) Preconditioner[nDim+1][0] = Cp*Temperature/BetaInc2;
2012  else Preconditioner[nDim+1][0] = 0.0;
2013 
2014  for (jDim = 0; jDim < nDim; jDim++) {
2015  Preconditioner[0][jDim+1] = 0.0;
2016  for (iDim = 0; iDim < nDim; iDim++) {
2017  if (iDim == jDim) Preconditioner[iDim+1][jDim+1] = Density;
2018  else Preconditioner[iDim+1][jDim+1] = 0.0;
2019  }
2020  Preconditioner[nDim+1][jDim+1] = 0.0;
2021  }
2022 
2023  Preconditioner[0][nDim+1] = dRhodT;
2024  for (iDim = 0; iDim < nDim; iDim++)
2025  Preconditioner[iDim+1][nDim+1] = Velocity[iDim]*dRhodT;
2026 
2027  if (energy) Preconditioner[nDim+1][nDim+1] = Cp*(dRhodT*Temperature + Density);
2028  else Preconditioner[nDim+1][nDim+1] = 1.0;
2029 
2030  for (iVar = 0; iVar < nVar; iVar ++ )
2031  for (jVar = 0; jVar < nVar; jVar ++ )
2032  Preconditioner[iVar][jVar] = delta*Preconditioner[iVar][jVar];
2033 
2034  } else {
2035 
2036  /*--- For explicit calculations, we move the residual to the
2037  right-hand side and pre-multiply by the preconditioner inverse.
2038  Therefore, we build inv(Precon) here and multiply by the residual
2039  later in the R-K and Euler Explicit time integration schemes. ---*/
2040 
2041  Preconditioner[0][0] = Temperature*BetaInc2*dRhodT/Density + BetaInc2;
2042  for (iDim = 0; iDim < nDim; iDim ++)
2043  Preconditioner[iDim+1][0] = -1.0*Velocity[iDim]/Density;
2044 
2045  if (energy) Preconditioner[nDim+1][0] = -1.0*Temperature/Density;
2046  else Preconditioner[nDim+1][0] = 0.0;
2047 
2048 
2049  for (jDim = 0; jDim < nDim; jDim++) {
2050  Preconditioner[0][jDim+1] = 0.0;
2051  for (iDim = 0; iDim < nDim; iDim++) {
2052  if (iDim == jDim) Preconditioner[iDim+1][jDim+1] = 1.0/Density;
2053  else Preconditioner[iDim+1][jDim+1] = 0.0;
2054  }
2055  Preconditioner[nDim+1][jDim+1] = 0.0;
2056  }
2057 
2058  Preconditioner[0][nDim+1] = -1.0*BetaInc2*dRhodT*oneOverCp/Density;
2059  for (iDim = 0; iDim < nDim; iDim ++)
2060  Preconditioner[iDim+1][nDim+1] = 0.0;
2061 
2062  if (energy) Preconditioner[nDim+1][nDim+1] = oneOverCp/Density;
2063  else Preconditioner[nDim+1][nDim+1] = 0.0;
2064 
2065  }
2066 
2067 }
