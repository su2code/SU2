32 CFEAElasticity::CFEAElasticity(unsigned short val_nDim, unsigned short val_nVar,
33  const CConfig *config) : CNumerics() {
34 
35  nDim = val_nDim;
36  nVar = val_nVar;
37 
38  bool body_forces = config->GetDeadLoad(); // Body forces (dead loads).
39  bool pseudo_static = config->GetPseudoStatic();
40 
41  unsigned short iVar, nProp;
42 
43  /*--- Initialize vector structures for multiple material definition ---*/
44  nProp = config->GetnElasticityMod();
45 
46  E_i = new su2double[nProp];
47  for (iVar = 0; iVar < nProp; iVar++)
48  E_i[iVar] = config->GetElasticyMod(iVar);
49 
50  nProp = config->GetnPoissonRatio();
51 
52  Nu_i = new su2double[nProp];
53  for (iVar = 0; iVar < nProp; iVar++)
54  Nu_i[iVar] = config->GetPoissonRatio(iVar);
55 
56  nProp = config->GetnMaterialDensity();
57 
58  Rho_s_i = new su2double[nProp]; // For inertial effects
59  Rho_s_DL_i = new su2double[nProp]; // For dead loads
60 
61  for (iVar = 0; iVar < nProp; iVar++) {
62  Rho_s_DL_i[iVar] = config->GetMaterialDensity(iVar);
63  Rho_s_i[iVar] = pseudo_static ? 0.0 : config->GetMaterialDensity(iVar);
64  }
65 
66  // Initialization
67  E = E_i[0];
68  Nu = Nu_i[0];
69  Rho_s = Rho_s_i[0];
70  Rho_s_DL = Rho_s_DL_i[0];
71 
72  Compute_Lame_Parameters();
73 
74  // Auxiliary vector for body forces (dead load)
75  FAux_Dead_Load = nullptr;
76  if (body_forces) FAux_Dead_Load = new su2double [nDim];
77 
78  plane_stress = (config->GetElas2D_Formulation() == STRUCT_2DFORM::PLANE_STRESS);
79 
80  KAux_ab = new su2double* [nDim];
81  for (iVar = 0; iVar < nDim; iVar++) {
82  KAux_ab[iVar] = new su2double[nDim];
83  }
84 
85  unsigned short nStrain = (nDim==2) ? DIM_STRAIN_2D : DIM_STRAIN_3D;
86  unsigned short nNodes = (nDim==2) ? NNODES_2D : NNODES_3D;
87 
88  Ba_Mat = new su2double* [nStrain];
89  Bb_Mat = new su2double* [nStrain];
90  D_Mat = new su2double* [nStrain];
91  Ni_Vec = new su2double [nNodes];
92  GradNi_Ref_Mat = new su2double* [nNodes];
93  GradNi_Curr_Mat = new su2double* [nNodes];
94  for (iVar = 0; iVar < nStrain; iVar++) {
95  Ba_Mat[iVar] = new su2double[nDim];
96  Bb_Mat[iVar] = new su2double[nDim];
97  D_Mat[iVar] = new su2double[nStrain];
98  }
99  for (iVar = 0; iVar < nNodes; iVar++) {
100  GradNi_Ref_Mat[iVar] = new su2double[nDim];
101  GradNi_Curr_Mat[iVar] = new su2double[nDim];
102  }
103 
104  DV_Val = nullptr;
105  n_DV = 0;
106  switch (config->GetDV_FEA()) {
107  case YOUNG_MODULUS:
108  case POISSON_RATIO:
109  case DENSITY_VAL:
110  case DEAD_WEIGHT:
111  case ELECTRIC_FIELD:
112  ReadDV(config);
113  for (unsigned short iDV = 0; iDV < n_DV; iDV++)
114  if ((config->GetDirectDiff() == D_YOUNG) ||
115  (config->GetDirectDiff() == D_POISSON) ||
116  (config->GetDirectDiff() == D_RHO) ||
117  (config->GetDirectDiff() == D_RHO_DL) ||
118  (config->GetDirectDiff() == D_EFIELD)){
119  SU2_TYPE::SetDerivative(DV_Val[config->GetnID_DV()],1.0);
120  }
121  break;
122 
123  default:
124  switch (config->GetDirectDiff()){
125  case D_YOUNG:
126  SU2_TYPE::SetDerivative(E_i[config->GetnID_DV()],1.0);
127  break;
128  case D_POISSON:
129  SU2_TYPE::SetDerivative(Nu_i[config->GetnID_DV()],1.0);
130  break;
131  case D_RHO:
132  SU2_TYPE::SetDerivative(Rho_s_i[config->GetnID_DV()],1.0);
133  break;
134  case D_RHO_DL:
135  SU2_TYPE::SetDerivative(Rho_s_DL_i[config->GetnID_DV()],1.0);
136  break;
137  }
138 
139  }
140 }

181 void CFEAElasticity::Compute_Mass_Matrix(CElement *element, const CConfig *config) {
182 
183  /*--- Initialize values for the material model considered ---*/
184  SetElement_Properties(element, config);
185  /*-----------------------------------------------------------*/
186 
187  unsigned short iGauss, nGauss;
188  unsigned short iNode, jNode, nNode;
189 
190  su2double Weight, Jac_X, val_Mab;
191 
192  /*--- Register pre-accumulation inputs, density and reference coords. ---*/
193  AD::StartPreacc();
194  AD::SetPreaccIn(Rho_s);
195  element->SetPreaccIn_Coords(false);
196 
197  element->ClearElement(); /*--- Restarts the element: avoids adding over previous results in other elements --*/
198  element->ComputeGrad_Linear(); /*--- Need to compute the gradients to obtain the Jacobian ---*/
199 
200  nNode = element->GetnNodes();
201  nGauss = element->GetnGaussPoints();
202 
203  for (iGauss = 0; iGauss < nGauss; iGauss++) {
204 
205  Weight = element->GetWeight(iGauss);
206  Jac_X = element->GetJ_X(iGauss); /*--- The mass matrix is computed in the reference configuration ---*/
207 
208  /*--- Retrieve the values of the shape functions for each node ---*/
209  /*--- This avoids repeated operations ---*/
210  for (iNode = 0; iNode < nNode; iNode++) {
211  Ni_Vec[iNode] = element->GetNi(iNode,iGauss);
212  }
213 
214  for (iNode = 0; iNode < nNode; iNode++) {
215 
216  /*--- Assumming symmetry ---*/
217  for (jNode = iNode; jNode < nNode; jNode++) {
218 
219  val_Mab = Weight * Ni_Vec[iNode] * Ni_Vec[jNode] * Jac_X * Rho_s;
220 
221  element->Add_Mab(iNode, jNode, val_Mab);
222  /*--- Symmetric terms --*/
223  if (iNode != jNode) {
224  element->Add_Mab(jNode, iNode, val_Mab);
225  }
226 
227  }
228 
229  }
230 
231  }
232 
233  /*--- Register the mass matrix as preaccumulation output. ---*/
234  element->SetPreaccOut_Mab();
235  AD::EndPreacc();
236 
237 }

240 void CFEAElasticity::Compute_Dead_Load(CElement *element, const CConfig *config) {
241 
242  /*--- Initialize values for the material model considered ---*/
243  SetElement_Properties(element, config);
244  /*-----------------------------------------------------------*/
245 
246  /*--- Register pre-accumulation inputs, density and reference coords. ---*/
247  AD::StartPreacc();
248  AD::SetPreaccIn(Rho_s_DL);
249  element->SetPreaccIn_Coords(false);
250 
251  unsigned short iGauss, nGauss;
252  unsigned short iNode, iDim, nNode;
253 
254  su2double Weight, Jac_X;
255 
256  /* -- Gravity directionality:
257  * -- For 2D problems, we assume the direction for gravity is -y
258  * -- For 3D problems, we assume the direction for gravity is -z
259  */
260  su2double g_force[3] = {0.0,0.0,0.0};
261 
262  if (nDim == 2) g_force[1] = -1*STANDARD_GRAVITY;
263  else if (nDim == 3) g_force[2] = -1*STANDARD_GRAVITY;
264 
265  element->ClearElement(); /*--- Restart the element to avoid adding over previous results. --*/
266  element->ComputeGrad_Linear(); /*--- Need to compute the gradients to obtain the Jacobian. ---*/
267 
268  nNode = element->GetnNodes();
269  nGauss = element->GetnGaussPoints();
270 
271  for (iGauss = 0; iGauss < nGauss; iGauss++) {
272 
273  Weight = element->GetWeight(iGauss);
274  Jac_X = element->GetJ_X(iGauss); /*--- The dead load is computed in the reference configuration ---*/
275 
276  /*--- Retrieve the values of the shape functions for each node ---*/
277  /*--- This avoids repeated operations ---*/
278  for (iNode = 0; iNode < nNode; iNode++) {
279  Ni_Vec[iNode] = element->GetNi(iNode,iGauss);
280  }
281 
282  for (iNode = 0; iNode < nNode; iNode++) {
283 
284  for (iDim = 0; iDim < nDim; iDim++) {
285  FAux_Dead_Load[iDim] = Weight * Ni_Vec[iNode] * Jac_X * Rho_s_DL * g_force[iDim];
286  }
287 
288  element->Add_FDL_a(iNode, FAux_Dead_Load);
289 
290  }
291 
292  }
293 
294  /*--- Register the dead load as preaccumulation output. ---*/
295  element->SetPreaccOut_FDL_a();
296  AD::EndPreacc();
297 
298 }

301 void CFEAElasticity::SetElement_Properties(const CElement *element, const CConfig *config) {
302 
303  /*--- These variables are set as preaccumulation inputs in Compute_Tangent_Matrix and
304  Compute_NodalStress_Term, if you add variables here be sure to register them in those routines too. ---*/
305  E = E_i[element->Get_iProp()];
306  Nu = Nu_i[element->Get_iProp()];
307  Rho_s = Rho_s_i[element->Get_iProp()];
308  Rho_s_DL = Rho_s_DL_i[element->Get_iProp()];
309 
310  switch (config->GetDV_FEA()) {
311  case YOUNG_MODULUS:
312  E = DV_Val[element->Get_iDV()] * E;
313  break;
314  case POISSON_RATIO:
315  Nu = DV_Val[element->Get_iDV()] * Nu;
316  break;
317  case DENSITY_VAL:
318  Rho_s = DV_Val[element->Get_iDV()] * Rho_s;
319  break;
320  case DEAD_WEIGHT:
321  Rho_s_DL = DV_Val[element->Get_iDV()] * Rho_s_DL;
322  break;
323  }
324 
325  Compute_Lame_Parameters();
326 
327 }

330 void CFEAElasticity::ReadDV(const CConfig *config) {
331 
332  int rank = SU2_MPI::GetRank();
333  bool master_node = false;
334  SU2_OMP_MASTER
335  master_node = (rank == MASTER_NODE);
336  END_SU2_OMP_MASTER
337 
338  unsigned long index;
339 
340  string filename;
341  ifstream properties_file;
342 
343  /*--- Choose the filename of the design variable ---*/
344 
345  string input_name;
346 
347  switch (config->GetDV_FEA()) {
348  case YOUNG_MODULUS:
349  input_name = "dv_young.opt";
350  break;
351  case POISSON_RATIO:
352  input_name = "dv_poisson.opt";
353  break;
354  case DENSITY_VAL:
355  case DEAD_WEIGHT:
356  input_name = "dv_density.opt";
357  break;
358  case ELECTRIC_FIELD:
359  input_name = "dv_efield.opt";
360  break;
361  default:
362  input_name = "dv.opt";
363  break;
364  }
365 
366  filename = input_name;
367 
368  if (master_node) cout << "Filename: " << filename << "." << endl;
369 
370  properties_file.open(filename.data(), ios::in);
371 
372  /*--- In case there is no file, all elements get the same property (0) ---*/
373 
374  if (properties_file.fail()) {
375 
376  if (master_node)
377  cout << "There is no design variable file." << endl;
378 
379  n_DV = 1;
380  DV_Val = new su2double[n_DV];
381  for (unsigned short iDV = 0; iDV < n_DV; iDV++)
382  DV_Val[iDV] = 1.0;
383 
384  }
385  else{
386 
387  string text_line;
388 
389  /*--- First pass: determine number of design variables ---*/
390 
391  unsigned short iDV = 0;
392 
393  /*--- Skip the first line: it is the header ---*/
394 
395  getline (properties_file, text_line);
396 
397  while (getline (properties_file, text_line)) iDV++;
398 
399  /*--- Close the restart file ---*/
400 
401  properties_file.close();
402 
403  n_DV = iDV;
404  DV_Val = new su2double[n_DV];
405 
406  /*--- Reopen the file (TODO: improve this) ---*/
407 
408  properties_file.open(filename.data(), ios::in);
409 
410  /*--- Skip the first line: it is the header ---*/
411 
412  getline (properties_file, text_line);
413 
414  iDV = 0;
415  while (getline (properties_file, text_line)) {
416 
417  istringstream point_line(text_line);
418 
419  point_line >> index >> DV_Val[iDV];
420 
421  iDV++;
422 
423  }
424 
425  /*--- Close the restart file ---*/
426 
427  properties_file.close();
428 
429  }
