31 CPYRAM5::CPYRAM5() : CElementWithKnownSizes<NGAUSS,NNODE,NDIM>() {
32 
33  /*--- Gauss coordinates and weights ---*/
34 
35  GaussCoord[0][0] = 0.5; GaussCoord[0][1] = 0.0; GaussCoord[0][2] = 0.1531754163448146; GaussWeight(0) = 2.0/15.0;
36  GaussCoord[1][0] = 0.0; GaussCoord[1][1] = 0.5; GaussCoord[1][2] = 0.1531754163448146; GaussWeight(1) = 2.0/15.0;
37  GaussCoord[2][0] =-0.5; GaussCoord[2][1] = 0.0; GaussCoord[2][2] = 0.1531754163448146; GaussWeight(2) = 2.0/15.0;
38  GaussCoord[3][0] = 0.0; GaussCoord[3][1] =-0.5; GaussCoord[3][2] = 0.1531754163448146; GaussWeight(3) = 2.0/15.0;
39  GaussCoord[4][0] = 0.0; GaussCoord[4][1] = 0.0; GaussCoord[4][2] = 0.6372983346207416; GaussWeight(4) = 2.0/15.0;
40 
41  /*--- Store the values of the shape functions and their derivatives ---*/
42 
43  unsigned short iNode, iGauss;
44  su2double Xi, Eta, Zeta, val_Ni;
45 
46  for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
47 
48  Xi = GaussCoord[iGauss][0];
49  Eta = GaussCoord[iGauss][1];
50  Zeta = GaussCoord[iGauss][2];
51 
52  val_Ni = 0.25*(-Xi+Eta+Zeta-1.0)*(-Xi-Eta+Zeta-1.0)/(1.0-Zeta); GaussPoint[iGauss].SetNi(val_Ni,0);
53  val_Ni = 0.25*(-Xi-Eta+Zeta-1.0)*( Xi-Eta+Zeta-1.0)/(1.0-Zeta); GaussPoint[iGauss].SetNi(val_Ni,1);
54  val_Ni = 0.25*( Xi+Eta+Zeta-1.0)*( Xi-Eta+Zeta-1.0)/(1.0-Zeta); GaussPoint[iGauss].SetNi(val_Ni,2);
55  val_Ni = 0.25*( Xi+Eta+Zeta-1.0)*(-Xi+Eta+Zeta-1.0)/(1.0-Zeta); GaussPoint[iGauss].SetNi(val_Ni,3);
56  val_Ni = Zeta; GaussPoint[iGauss].SetNi(val_Ni,4);
57 
58  /*--- dN/d xi ---*/
59 
60  dNiXj[iGauss][0][0] = 0.5*(Zeta-Xi-1.0)/(Zeta-1.0);
61  dNiXj[iGauss][1][0] = 0.5*Xi/(Zeta-1.0);
62  dNiXj[iGauss][2][0] = 0.5*(1.0-Zeta-Xi)/(Zeta-1.0);
63  dNiXj[iGauss][3][0] = dNiXj[iGauss][1][0];
64  dNiXj[iGauss][4][0] = 0.0;
65 
66  /*--- dN/d eta ---*/
67 
68  dNiXj[iGauss][0][1] = 0.5*Eta/(Zeta-1.0);
69  dNiXj[iGauss][1][1] = 0.5*(Zeta-Eta-1.0)/(Zeta-1.0);
70  dNiXj[iGauss][2][1] = dNiXj[iGauss][0][1];
71  dNiXj[iGauss][3][1] = 0.5*(1.0-Zeta-Eta)/(Zeta-1.0);
72  dNiXj[iGauss][4][1] = 0.0;
73 
74  /*--- dN/d zeta ---*/
75 
76  dNiXj[iGauss][0][2] = 0.25*(-1.0 + 2.0*Zeta - Zeta*Zeta - Eta*Eta + Xi*Xi)/((1.0-Zeta)*(1.0-Zeta));
77  dNiXj[iGauss][1][2] = 0.25*(-1.0 + 2.0*Zeta - Zeta*Zeta + Eta*Eta - Xi*Xi)/((1.0-Zeta)*(1.0-Zeta));
78  dNiXj[iGauss][2][2] = dNiXj[iGauss][0][2];
79  dNiXj[iGauss][3][2] = dNiXj[iGauss][1][2];
80  dNiXj[iGauss][4][2] = 1.0;

82  }

102  NodalExtrap[iNode][2] = 0.25*( Xi+Eta+Zeta-1.0)*( Xi-Eta+Zeta-1.0)/(1.0-Zeta);
103  NodalExtrap[iNode][3] = 0.25*( Xi+Eta+Zeta-1.0)*(-Xi+Eta+Zeta-1.0)/(1.0-Zeta);
104  NodalExtrap[iNode][4] = Zeta;

110 su2double CPYRAM5::ComputeVolume(const FrameType mode) const {
111 
112  unsigned short iDim;
113  su2double r1[3] = {0.0,0.0,0.0}, r2[3] = {0.0,0.0,0.0}, r3[3] = {0.0,0.0,0.0}, CrossProduct[3] = {0.0,0.0,0.0};
114  su2double Volume = 0.0;
115 
116  /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
117  for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed)---*/
118  const su2activematrix& Coord = (mode==REFERENCE) ? RefCoord : CurrentCoord;
119 
120  for (iDim = 0; iDim < NDIM; iDim++) {
121  r1[iDim] = Coord[1][iDim] - Coord[0][iDim];
122  r2[iDim] = Coord[2][iDim] - Coord[0][iDim];
123  r3[iDim] = Coord[4][iDim] - Coord[0][iDim];
124  }
125 
126  CrossProduct[0] = (r1[1]*r2[2] - r1[2]*r2[1])*r3[0];
127  CrossProduct[1] = (r1[2]*r2[0] - r1[0]*r2[2])*r3[1];
128  CrossProduct[2] = (r1[0]*r2[1] - r1[1]*r2[0])*r3[2];
129 
130  Volume = fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2])/6.0;
131 
132  for (iDim = 0; iDim < NDIM; iDim++) {
133  r1[iDim] = Coord[2][iDim] - Coord[0][iDim];
134  r2[iDim] = Coord[3][iDim] - Coord[0][iDim];
135  r3[iDim] = Coord[4][iDim] - Coord[0][iDim];
136  }
137 
138  CrossProduct[0] = (r1[1]*r2[2] - r1[2]*r2[1])*r3[0];
139  CrossProduct[1] = (r1[2]*r2[0] - r1[0]*r2[2])*r3[1];
140  CrossProduct[2] = (r1[0]*r2[1] - r1[1]*r2[0])*r3[2];
141 
142  Volume += fabs(CrossProduct[0] + CrossProduct[1] + CrossProduct[2])/6.0;
143 
144  return Volume;
145 
146 }
