51 /*!
52  * \class CCompressedSparsePattern
53  * \brief A simple class to store adjacency information in a
54  * compressed format suitable for sparse matrix operations.
55  * If built for row-major storage the inner indices are column indices
56  * and the pattern should be used as (row,icol), otherwise as (col,irow).

89  /*!
90  * \brief Construct with default inner indices.
91  * \param[in] outerPtrBegin - Start of outer pointers.
92  * \param[in] outerPtrEnd - End of outer pointers.
93  * \param[in] defaultInnerIdx - Default value for inner indices.
94  */
95  template<class Iterator>
96  CCompressedSparsePattern(Iterator outerPtrBegin, Iterator outerPtrEnd, Index_t defaultInnerIdx)
97  {
98  const auto size = outerPtrEnd - outerPtrBegin;
99  m_outerPtr.resize(size);
100  Index_t k = 0;
101  for(auto it = outerPtrBegin; it != outerPtrEnd; ++it)
102  m_outerPtr(k++) = *it;
103 
104  m_innerIdx.resize(m_outerPtr(size-1)) = defaultInnerIdx;
105  }

107  /*!
108  * \brief Construct from rvalue refs.
109  * \note This is the most efficient constructor as no data copy occurs.
110  * \param[in] outerPtr - Outer index pointers.
111  * \param[in] innerIdx - Inner indices.
112  */
113  CCompressedSparsePattern(su2vector<Index_t>&& outerPtr,
114  su2vector<Index_t>&& innerIdx) :

121  /*!
122  * \brief Construct from vector-like objects of any type with
123  * methods "size()" and "data()" (returning a pointer).
124  * \param[in] outerPtr - Outer index pointers.
125  * \param[in] innerIdx - Inner indices.
126  */
127  template<class T>
128  CCompressedSparsePattern(const T& outerPtr, const T& innerIdx)
129  {
130  m_outerPtr.resize(outerPtr.size());
131  for(Index_t i=0; i<outerPtr.size(); ++i)
132  m_outerPtr(i) = outerPtr.data()[i];
133 
134  m_innerIdx.resize(innerIdx.size());
135  for(Index_t i=0; i<innerIdx.size(); ++i)
136  m_innerIdx(i) = innerIdx.data()[i];
137 
138  /*--- perform a basic sanity check ---*/
139  assert(m_innerIdx.size() == m_outerPtr(m_outerPtr.size()-1));
140  }

142  /*!
143  * \brief Build from a "list of lists" type object
144  * \param[in] lil - An object with operator [][] and method "size" e.g. vector<vector<?> >.
145  */
146  template<class T>
147  CCompressedSparsePattern(const T& lil)
148  {
149  m_outerPtr.resize(lil.size()+1);
150  m_outerPtr(0) = 0;
151  for(Index_t i=1; i < Index_t(m_outerPtr.size()); ++i)
152  m_outerPtr(i) = m_outerPtr(i-1) + lil[i-1].size();
153 
154  m_innerIdx.resize(m_outerPtr(lil.size()));
155  Index_t k = 0;
156  for(Index_t i=0; i < Index_t(lil.size()); ++i)
157  for(Index_t j=0; j < Index_t(lil[i].size()); ++j)
158  m_innerIdx(k++) = lil[i][j];
159  }

161  /*!
162  * \brief Build a list of pointers to the diagonal entries of the pattern.
163  */
164  void buildDiagPtr() {
165  if(!m_diagPtr.empty()) return;
166 
167  m_diagPtr.resize(getOuterSize());
168 
169  SU2_OMP_PARALLEL_(for schedule(static,roundUpDiv(getOuterSize(),omp_get_max_threads())))
170  for(Index_t k = 0; k < getOuterSize(); ++k)
171  m_diagPtr(k) = findInnerIdx(k,k);
172  END_SU2_OMP_PARALLEL
173  }
174 
175  /*!
176  * \brief Build a list of pointers to the transpose entries of the pattern, requires symmetry.
177  */
178  void buildTransposePtr() {
179  if(!m_innerIdxTransp.empty()) return;
180 
181  m_innerIdxTransp.resize(getNumNonZeros());
182 
183  SU2_OMP_PARALLEL_(for schedule(static,roundUpDiv(getOuterSize(),omp_get_max_threads())))
184  for(Index_t i = 0; i < getOuterSize(); ++i) {
185  for(Index_t k = m_outerPtr(i); k < m_outerPtr(i+1); ++k) {
186  auto j = m_innerIdx(k);
187  m_innerIdxTransp(k) = findInnerIdx(j,i);
188  assert(m_innerIdxTransp(k) != m_innerIdx.size() && "The pattern is not symmetric.");
189  }
190  }
191  END_SU2_OMP_PARALLEL
192  }

253  * \param[in] iOuterIdx - Outer index (row/col).
254  * \param[in] iInnerIdx - Inner index (col/row).
255  * \return Absolute position of non zero entry (iOuterIdx,iInnerIdx),
256  * or NNZ if position does not belong to the pattern.

259  for(Index_t k = m_outerPtr(iOuterIdx); k < m_outerPtr(iOuterIdx+1); ++k)
260  if(m_innerIdx(k) == iInnerIdx) return k;

273  /*!
274  * \param[in] iOuterIdx - Outer index (row/col).
275  * \param[in] iInnerIdx - Inner index (col/row).
276  * \return Absolute position of non zero entry (iOuterIdx,iInnerIdx).
277  * \note This method is only safe if the entry exists.
278  */
279  inline Index_t quickFindInnerIdx(Index_t iOuterIdx, Index_t iInnerIdx) const {
280  assert(isNonZero(iOuterIdx, iInnerIdx) && "Error, j does not belong to NZ(i).");
281  Index_t k = m_outerPtr(iOuterIdx);
282  while(m_innerIdx(k) != iInnerIdx) ++k;
283  return k;
284  }

369 /*!
370  * \brief Build a sparse pattern from geometry information, of type FVM or FEM,
371  * for a given fill-level. At fill-level N, the immediate neighbors of the
372  * points in level N-1 are also considered neighbors of the base point.
373  * The resulting pattern is that of A^{N+1} where A is the sparse matrix
374  * of immediate neighbors.
375  * \note Algorithm is equivalent to the implementation by F. Palacios,
376  * A. Bueno, and T. Economon from CSysMatrix.
377  * \param[in] geometry - Definition of the grid.
378  * \param[in] type - Of connectivity.
379  * \param[in] fillLvl - Target degree of neighborhood (immediate neighbors always added).
380  * \return Compressed-Storage-Row sparse pattern.
381  */
382 template<class Geometry_t, typename Index_t>
383 CCompressedSparsePattern<Index_t> buildCSRPattern(Geometry_t& geometry,
384  ConnectivityType type,
385  Index_t fillLvl)
386 {
387  Index_t nPoint = geometry.GetnPoint();
388 
389  std::vector<Index_t> outerPtr(nPoint+1);
390  std::vector<Index_t> innerIdx;
391  innerIdx.reserve(nPoint); // at least this much space is needed
392 
393  for(Index_t iPoint = 0; iPoint < nPoint; ++iPoint)
394  {
395  /*--- Inner indices for iPoint start here. ---*/
396  outerPtr[iPoint] = innerIdx.size();
397 
398  /*--- Use a set to avoid duplication and keep ascending order. ---*/
399  std::set<Index_t> neighbors;
400 
401  /*--- Insert base point. ---*/
402  neighbors.insert(iPoint);
403 
404  /*--- Neighbors added in previous level. ---*/
405  std::set<Index_t> addedNeighbors(neighbors);
406 
407  for(Index_t iLevel = 0; ; ++iLevel)
408  {
409  /*--- New points added in this level. ---*/
410  std::set<Index_t> newNeighbors;
411 
412  /*--- For each point previously added, add its level 0
413  * neighbors, not duplicating any existing neighbor. ---*/
414  for(auto jPoint : addedNeighbors)
415  {
416  if(type == ConnectivityType::FiniteVolume)
417  {
418  /*--- For FVM we know the neighbors of point j directly. ---*/
419  for(Index_t kPoint : geometry.nodes->GetPoints(jPoint))
420  if(neighbors.count(kPoint) == 0) // no duplication
421  newNeighbors.insert(kPoint);
422  }
423  else // FiniteElement
424  {
425  /*--- For FEM we need the nodes of all elements that contain point j. ---*/
426  for(auto iElem : geometry.nodes->GetElems(jPoint))
427  {
428  auto elem = geometry.elem[iElem];
429 
430  for(unsigned short iNode = 0; iNode < elem->GetnNodes(); ++iNode)
431  {
432  Index_t kPoint = elem->GetNode(iNode);
433 
434  if(neighbors.count(kPoint) == 0) // no duplication
435  newNeighbors.insert(kPoint);
436  }
437  }
438  }
439  }
440 
441  neighbors.insert(newNeighbors.begin(), newNeighbors.end());
442 
443  if(iLevel >= fillLvl) break;
444 
445  /*--- For the next level we get the neighbours of the new points. ---*/
446  addedNeighbors = newNeighbors;
447  }
448 
449  /*--- Store final sparse pattern for iPoint. ---*/
450  innerIdx.insert(innerIdx.end(), neighbors.begin(), neighbors.end());
451  }
452  outerPtr.back() = innerIdx.size();
453 
454  /*--- Return pattern as CCompressedSparsePattern object. ---*/
455  return CCompressedSparsePattern<Index_t>(outerPtr, innerIdx);
456 }

459 /*!
460  * \brief Build a lookup table of the absolute positions of the non zero entries
461  * of a compressed sparse pattern, accessed when visiting the FVM edges
462  * of a grid. The table can then be used for fast access (avoids searches)
463  * to the non zero entries of a sparse matrix associated with the pattern.
464  * \param[in] geometry - Definition of the grid.
465  * \param[in] pattern - Sparse pattern.
466  * \return nEdge by 2 matrix.
467  */
468 template<class Geometry_t, typename Index_t>
469 CEdgeToNonZeroMap<Index_t> mapEdgesToSparsePattern(Geometry_t& geometry,
470  const CCompressedSparsePattern<Index_t>& pattern)
471 {
472  assert(!pattern.empty());
473 
474  CEdgeToNonZeroMap<Index_t> edgeMap(geometry.GetnEdge(),2);
475 
476  for(Index_t iEdge = 0; iEdge < geometry.GetnEdge(); ++iEdge)
477  {
478  Index_t iPoint = geometry.edges->GetNode(iEdge,0);
479  Index_t jPoint = geometry.edges->GetNode(iEdge,1);
480 
481  edgeMap(iEdge,0) = pattern.quickFindInnerIdx(iPoint,jPoint);
482  edgeMap(iEdge,1) = pattern.quickFindInnerIdx(jPoint,iPoint);
483  }
484 
485  return edgeMap;
486 }

489 /*!
490  * \brief Create the natural coloring (equivalent to the normal sequential loop
491  * order) for a given number of inner indexes.
492  * \note This is to reduce overhead in "OpenMP-ready" code when only 1 thread is used.
493  * \param[in] numInnerIndexes - Number of indexes that are to be colored.
494  * \return Natural (sequential) coloring of the inner indices.
495  */
496 template<class T = CCompressedSparsePatternUL,
497  class Index_t = typename T::IndexType>
498 T createNaturalColoring(Index_t numInnerIndexes)
499 {
500  /*--- One color. ---*/
501  su2vector<Index_t> outerPtr(2);
502  outerPtr(0) = 0;
503  outerPtr(1) = numInnerIndexes;
504 
505  /*--- Containing all indexes in ascending order. ---*/
506  su2vector<Index_t> innerIdx(numInnerIndexes);
507  std::iota(innerIdx.data(), innerIdx.data()+numInnerIndexes, 0);
508

513 /*!
514  * \brief Color contiguous groups of outer indices of a sparse pattern such that
515  * within each color, any two groups do not have inner indices in common.
516  * \note Within a group, two outer indices will generally have common inner indices.
517  * The coloring is returned as a compressed sparse pattern where the colors
518  * are outer indices, and the outer indices of the input pattern are the
519  * inner indices of the coloring. A simple greedy algorithm is used.
520  * Using a sparse pattern as input allows "anything" to be colored e.g.
521  * FVM edges, FEM elements, the rows/columns of a sparse matrix, etc.
522  * \note The worst that can happen in this method is needing an unreasonable number
523  * of colors, or too much memory due to a large range of the inner indices.
524  * The last two template parameters limit both, in case of failure an empty
525  * pattern is returned.
526  * \param[in] pattern - Sparse pattern to be colored.
527  * \param[in] groupSize - Size of the outer index groups, default 1.
528  * \param[in] balanceColors - Try to balance number of indexes per color,
529  * tends to result in worse locality (thus false by default).
530  * \param[out] indexColor - Optional, vector with colors given to the outer indices.
531  * \return Coloring in the same type of the input pattern.
532  */
533 template<typename Color_t = char, size_t MaxColors = 64, size_t MaxMB = 128, class T>
534 T colorSparsePattern(const T& pattern, size_t groupSize = 1, bool balanceColors = false,
535  std::vector<Color_t>* indexColor = nullptr)
536 {
537  static_assert(std::is_integral<Color_t>::value,"");
538  static_assert(std::numeric_limits<Color_t>::max() >= MaxColors,"");
539 
540  using Index_t = typename T::IndexType;
541 
542  const Index_t grpSz = groupSize;
543  const Index_t nOuter = pattern.getOuterSize();
544 
545  /*--- Trivial case. ---*/
546  if(groupSize >= nOuter) return createNaturalColoring(nOuter);
547 
548  const Index_t minIdx = pattern.getMinInnerIdx();
549  const Index_t nInner = pattern.getMaxInnerIdx()+1-minIdx;
550 
551  /*--- Check the max memory condition (<< 23 is to count bits). ---*/
552  if(size_t(nInner) > (MaxMB << 23)) return T();
553 
554  /*--- Vector with the color given to each outer index. ---*/
555  std::vector<Color_t> idxColor(nOuter);
556 
557  /*--- Start with one color, with no indices assigned. ---*/
558  std::vector<Index_t> colorSize(1,0);
559  Color_t nColor = 1;
560 
561  {
562  /*--- For each color keep track of the inner indices that are in it. ---*/
563  std::vector<std::vector<bool> > innerInColor;
564  innerInColor.emplace_back(nInner, false);
565 
566  /*--- Order in which we look for space in the colors to insert a new group. ---*/
567  std::vector<Color_t> searchOrder(MaxColors);
568 
569  auto outerPtr = pattern.outerPtr();
570  auto innerIdx = pattern.innerIdx();
571 
572  for(Index_t iOuter = 0; iOuter < nOuter; iOuter += grpSz)
573  {
574  Index_t grpEnd = std::min(iOuter+grpSz, nOuter);
575 
576  searchOrder.resize(nColor);
577  std::iota(searchOrder.begin(), searchOrder.end(), 0);
578 
579  /*--- Balance sizes by looking for space in smaller colors first. ---*/
580  if(balanceColors) {
581  std::sort(searchOrder.begin(), searchOrder.end(),
582  [&colorSize](Color_t a, Color_t b){return colorSize[a] < colorSize[b];});
583  }
584 
585  auto it = searchOrder.begin();
586 
587  for(; it != searchOrder.end(); ++it)
588  {
589  bool free = true;
590  /*--- Traverse entire group as a large outer index. ---*/
591  for(Index_t k = outerPtr[iOuter]; k < outerPtr[grpEnd] && free; ++k)
592  {
593  free = !innerInColor[*it][innerIdx[k]-minIdx];
594  }
595  /*--- If none of the inner indices in the group appears in
596  * this color yet, it is assigned to the group. ---*/
597  if(free) break;
598  }
599 
600  Color_t color;
601 
602  if(it != searchOrder.end())
603  {
604  /*--- Found a free color. ---*/
605  color = *it;
606  }
607  else {
608  /*--- No color was free, make space for a new one. ---*/
609  color = nColor++;
610  if(nColor == MaxColors) return T();
611  colorSize.push_back(0);
612  innerInColor.emplace_back(nInner, false);
613  }
614 
615  /*--- Assign color to group. ---*/

622  }

626  }

631 
632  su2vector<Index_t> colorPtr(nColor+1); colorPtr(0) = 0;
633  su2vector<Index_t> outerIdx(nOuter);
634 
635  Index_t k = 0;

637  {
638  colorPtr(color+1) = colorPtr(color)+colorSize[color];
639

643  }
644

646  if(indexColor) *indexColor = std::move(idxColor);
647 
648  /*--- Move compressed coloring into result pattern instance. ---*/
649  return T(std::move(colorPtr), std::move(outerIdx));

706  * \brief Computes the efficiency of a grid coloring for given number of threads and chunk size.
707  */
708 template<class SparsePattern>
709 su2double coloringEfficiency(const SparsePattern& coloring, int numThreads, int chunkSize)
710 {
711  using Index_t = typename SparsePattern::IndexType;
712 
713  /*--- Ideally compute time is proportional to total work over number of threads. ---*/
714  su2double ideal = coloring.getNumNonZeros() / su2double(numThreads);
715 
716  /*--- In practice the total work is quantized first by colors and then by chunks. ---*/
717  Index_t real = 0;
718  for(Index_t color = 0; color < coloring.getOuterSize(); ++color)
719  real += chunkSize * roundUpDiv(roundUpDiv(coloring.getNumNonZeros(color), chunkSize), numThreads);
720 
721  return ideal / real;
722 }
