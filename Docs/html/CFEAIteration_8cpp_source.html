31 void CFEAIteration::Iterate(COutput* output, CIntegration**** integration, CGeometry**** geometry, CSolver***** solver,
32  CNumerics****** numerics, CConfig** config, CSurfaceMovement** surface_movement,
33  CVolumetricMovement*** grid_movement, CFreeFormDefBox*** FFDBox, unsigned short val_iZone,
34  unsigned short val_iInst) {
35  bool StopCalc = false;
36  unsigned long IntIter = 0;
37 
38  const unsigned long TimeIter = config[val_iZone]->GetTimeIter();
39  const unsigned long nIncrements = config[val_iZone]->GetNumberIncrements();
40 
41  const bool nonlinear = (config[val_iZone]->GetGeometricConditions() == STRUCT_DEFORMATION::LARGE);
42  const bool linear = (config[val_iZone]->GetGeometricConditions() == STRUCT_DEFORMATION::SMALL);
43  const bool disc_adj_fem = config[val_iZone]->GetDiscrete_Adjoint();
44 
45  /*--- Loads applied in steps (not used for discrete adjoint). ---*/
46  const bool incremental_load = config[val_iZone]->GetIncrementalLoad() && !disc_adj_fem;
47 
48  CIntegration* feaIntegration = integration[val_iZone][val_iInst][FEA_SOL];
49  CSolver* feaSolver = solver[val_iZone][val_iInst][MESH_0][FEA_SOL];
50 
51  /*--- Set the convergence monitor to false, to prevent the solver to stop in intermediate FSI subiterations ---*/
52  feaIntegration->SetConvergence(false);
53 
54  /*--- FEA equations ---*/
55  config[val_iZone]->SetGlobalParam(MAIN_SOLVER::FEM_ELASTICITY, RUNTIME_FEA_SYS);
56 
57  if (linear) {
58  /*--- Run the (one) iteration ---*/
59 
60  config[val_iZone]->SetInnerIter(0);
61 
62  feaIntegration->Structural_Iteration(geometry, solver, numerics, config, RUNTIME_FEA_SYS, val_iZone, val_iInst);
63 
64  if (!disc_adj_fem) {
65  Monitor(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement, FFDBox,
66  val_iZone, INST_0);
67 
68  /*--- Set the convergence monitor to true, to prevent the solver to stop in intermediate FSI subiterations ---*/
69  output->SetConvergence(true);
70  }
71 
72  } else if (nonlinear && !incremental_load) {
73  /*--- THIS IS THE DIRECT APPROACH (NO INCREMENTAL LOAD APPLIED) ---*/
74 
75  /*--- Keep the current inner iter, we need to restore it in discrete adjoint cases as file output depends on it
76  * ---*/
77  const auto CurIter = config[val_iZone]->GetInnerIter();
78 
79  /*--- Newton-Raphson subiterations ---*/
80 
81  for (IntIter = 0; IntIter < config[val_iZone]->GetnInner_Iter(); IntIter++) {
82  config[val_iZone]->SetInnerIter(IntIter);
83 
84  feaIntegration->Structural_Iteration(geometry, solver, numerics, config, RUNTIME_FEA_SYS, val_iZone, val_iInst);
85 
86  /*--- Limit to only one iteration for the discrete adjoint recording, restore inner iter (see above) ---*/
87  if (disc_adj_fem) {
88  config[val_iZone]->SetInnerIter(CurIter);
89  break;
90  } else {
91  StopCalc = Monitor(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement,
92  FFDBox, val_iZone, INST_0);
93 
94  if (StopCalc && (IntIter > 0)) break;
95  }
96  }
97 
98  } else {
99  /*--- THIS IS THE INCREMENTAL LOAD APPROACH (only makes sense for nonlinear) ---*/
100 
101  /*--- Assume the initial load increment as 1.0 ---*/
102 
103  feaSolver->SetLoad_Increment(0, 1.0);
104  feaSolver->SetForceCoeff(1.0);
105 
106  /*--- Run two nonlinear iterations to check if incremental loading can be skipped ---*/
107 
108  for (IntIter = 0; IntIter < 2; ++IntIter) {
109  config[val_iZone]->SetInnerIter(IntIter);
110 
111  feaIntegration->Structural_Iteration(geometry, solver, numerics, config, RUNTIME_FEA_SYS, val_iZone, val_iInst);
112 
113  StopCalc = Monitor(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement,
114  FFDBox, val_iZone, INST_0);
115  }
116 
117  /*--- Early return if we already meet the convergence criteria. ---*/
118  if (StopCalc) return;
119 
120  /*--- Check user-defined criteria to either increment loads or continue with NR iterations. ---*/
121 
122  bool meetCriteria = true;
123  for (int i = 0; i < 3; ++i)
124  meetCriteria &= (log10(feaSolver->GetRes_FEM(i)) < config[val_iZone]->GetIncLoad_Criteria(i));
125 
126  /*--- If the criteria is met, i.e. the load is not too large, continue the regular calculation. ---*/
127 
128  if (meetCriteria) {
129  /*--- Newton-Raphson subiterations ---*/
130 
131  for (IntIter = 2; IntIter < config[val_iZone]->GetnInner_Iter(); IntIter++) {
132  config[val_iZone]->SetInnerIter(IntIter);
133 
134  feaIntegration->Structural_Iteration(geometry, solver, numerics, config, RUNTIME_FEA_SYS, val_iZone, val_iInst);
135 
136  StopCalc = Monitor(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement,
137  FFDBox, val_iZone, INST_0);
138 
139  if (StopCalc) break;
140  }
141 
142  }
143 
144  /*--- If the criteria is not met, a whole set of subiterations for the different loads must be done. ---*/
145 
146  else {
147  /*--- Restore solution to initial. Because we ramp the load from zero, in multizone cases it is not
148  * adequate to take "old values" as those will be for maximum loading on the previous outer iteration. ---*/
149 
150  feaSolver->SetInitialCondition(geometry[val_iZone][val_iInst], solver[val_iZone][val_iInst], config[val_iZone],
151  TimeIter);
152 
153  /*--- For the number of increments ---*/
154  for (auto iIncrement = 1ul; iIncrement <= nIncrements; iIncrement++) {
155  /*--- Set the load increment and the initial condition, and output the
156  * parameters of UTOL, RTOL, ETOL for the previous iteration ---*/
157 
158  su2double loadIncrement = su2double(iIncrement) / nIncrements;
159  feaSolver->SetLoad_Increment(iIncrement, loadIncrement);
160 
161  /*--- Set the convergence monitor to false, to force the solver to converge every subiteration ---*/
162  output->SetConvergence(false);
163 
164  if (rank == MASTER_NODE) cout << "\nIncremental load: increment " << iIncrement << endl;
165 
166  /*--- Newton-Raphson subiterations ---*/
167 
168  for (IntIter = 0; IntIter < config[val_iZone]->GetnInner_Iter(); IntIter++) {
169  config[val_iZone]->SetInnerIter(IntIter);
170 
171  feaIntegration->Structural_Iteration(geometry, solver, numerics, config, RUNTIME_FEA_SYS, val_iZone,
172  val_iInst);
173 
174  StopCalc = Monitor(output, integration, geometry, solver, numerics, config, surface_movement, grid_movement,
175  FFDBox, val_iZone, INST_0);
176 
177  if (StopCalc && (IntIter > 0)) break;
178  }
179  }
180  /*--- Just to be sure, set default increment settings. ---*/
181  feaSolver->SetLoad_Increment(0, 1.0);
182  }
183  }
184 }

186 void CFEAIteration::Update(COutput* output, CIntegration**** integration, CGeometry**** geometry, CSolver***** solver,
187  CNumerics****** numerics, CConfig** config, CSurfaceMovement** surface_movement,
188  CVolumetricMovement*** grid_movement, CFreeFormDefBox*** FFDBox, unsigned short val_iZone,
189  unsigned short val_iInst) {
190  const auto TimeIter = config[val_iZone]->GetTimeIter();
191  const bool dynamic = (config[val_iZone]->GetTime_Domain());
192  const bool fsi = config[val_iZone]->GetFSI_Simulation();
193 
194  CSolver* feaSolver = solver[val_iZone][val_iInst][MESH_0][FEA_SOL];
195 
196  /*----------------- Update structural solver ----------------------*/
197 
198  if (dynamic) {
199  integration[val_iZone][val_iInst][FEA_SOL]->SetDualTime_Solver(
200  geometry[val_iZone][val_iInst][MESH_0], solver[val_iZone][val_iInst][MESH_0][FEA_SOL], config[val_iZone],
201  MESH_0);
202  integration[val_iZone][val_iInst][FEA_SOL]->SetConvergence(false);
203 
204  /*--- Verify convergence criteria (based on total time) ---*/
205 
206  const su2double Physical_dt = config[val_iZone]->GetDelta_DynTime();
207  const su2double Physical_t = (TimeIter + 1) * Physical_dt;
208  if (Physical_t >= config[val_iZone]->GetTotal_DynTime())
209  integration[val_iZone][val_iInst][FEA_SOL]->SetConvergence(true);
210 
211  } else if (fsi) {
212  /*--- For FSI problems, output the relaxed result, which is the one transferred into the fluid domain (for restart
213  * purposes) ---*/
214  if (config[val_iZone]->GetKind_TimeIntScheme_FEA() == STRUCT_TIME_INT::NEWMARK_IMPLICIT) {
215  feaSolver->ImplicitNewmark_Relaxation(geometry[val_iZone][val_iInst][MESH_0], config[val_iZone]);
216  }
217  }

247 bool CFEAIteration::Monitor(COutput* output, CIntegration**** integration, CGeometry**** geometry, CSolver***** solver,
248  CNumerics****** numerics, CConfig** config, CSurfaceMovement** surface_movement,
249  CVolumetricMovement*** grid_movement, CFreeFormDefBox*** FFDBox, unsigned short val_iZone,
250  unsigned short val_iInst) {
251  StopTime = SU2_MPI::Wtime();
252 
253  UsedTime = StopTime - StartTime;
254 
255  if (config[val_iZone]->GetMultizone_Problem() || config[val_iZone]->GetSinglezone_Driver()) {
256  output->SetHistory_Output(geometry[val_iZone][val_iInst][MESH_0], solver[val_iZone][val_iInst][MESH_0],
257  config[val_iZone], config[val_iZone]->GetTimeIter(), config[val_iZone]->GetOuterIter(),
258  config[val_iZone]->GetInnerIter());
259  }
260 
261  return output->GetConvergence();
