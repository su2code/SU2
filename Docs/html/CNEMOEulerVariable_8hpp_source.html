262  /*!

266  inline void SetVelocity(unsigned long iPoint) final {
267  Velocity2(iPoint) = 0.0;
268  for (unsigned long iDim = 0; iDim < nDim; iDim++) {
269  Primitive(iPoint,VEL_INDEX+iDim) = Solution(iPoint,nSpecies+iDim) / Primitive(iPoint,RHO_INDEX);
270  Velocity2(iPoint) += pow(Primitive(iPoint,VEL_INDEX+iDim),2);
271  }
272  }
