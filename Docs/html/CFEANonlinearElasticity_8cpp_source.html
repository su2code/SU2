32 CFEANonlinearElasticity::CFEANonlinearElasticity(unsigned short val_nDim, unsigned short val_nVar,
33  const CConfig *config) : CFEAElasticity(val_nDim, val_nVar, config) {
34 
35  nearly_incompressible = (config->GetMaterialCompressibility() == STRUCT_COMPRESS::NEARLY_INCOMP);
36 
37  unsigned short iVar;
38 
39  F_Mat = new su2double *[3];
40  b_Mat = new su2double *[3];
41  FmT_Mat = new su2double *[3];
42  Stress_Tensor = new su2double *[3];
43  for (iVar = 0; iVar < 3; iVar++) {
44  F_Mat[iVar] = new su2double [3];
45  b_Mat[iVar] = new su2double [3];
46  FmT_Mat[iVar] = new su2double [3];
47  Stress_Tensor[iVar] = new su2double [3];
48  }
49 
50  KAux_t_a = new su2double [nDim];
51 
52  KAux_P_ab = new su2double* [nDim];
53  for (iVar = 0; iVar < nDim; iVar++) {
54  KAux_P_ab[iVar] = new su2double[nDim];
55  }
56 
57  unsigned short nNodes = (nDim==2) ? NNODES_2D : NNODES_3D;
58 
59  currentCoord = new su2double* [nNodes];
60  for (iVar = 0; iVar < nNodes; iVar++)
61  currentCoord[iVar] = new su2double[nDim];
62 
63  J_F = 1.0; J_F_Iso = 1.0;
64  f33 = 1.0;
65 
66  C10 = Mu/2.0;
67  D1 = 2.0/Kappa;
68 
69  F_Mat_Iso = nullptr;
70  b_Mat_Iso = nullptr;
71 
72  F_Mat_Iso = new su2double *[3];
73  b_Mat_Iso = new su2double *[3];
74  for (iVar = 0; iVar < 3; iVar++){
75  F_Mat_Iso[iVar] = new su2double [3];
76  b_Mat_Iso[iVar] = new su2double [3];
77  }
78 
79  maxwell_stress = config->GetDE_Effects();
80 
81  ke_DE = 0.0;
82  nElectric_Field = 0;
83  nDim_Electric_Field = 0;
84 
85  EField_Ref_Unit = nullptr;
86  EField_Ref_Mod = nullptr;
87  EField_Curr_Unit = nullptr;
88 
89  if (maxwell_stress == true) {
90 
91  const su2double *Electric_Field_Dir = config->Get_Electric_Field_Dir();
92  unsigned short iVar, iDim;
93  su2double ref_Efield_mod;
94 
95  // Initialize value of the electric field in the reference configuration
96  EFieldMod_Ref = 0.0;
97 
98  ke_DE = config->GetElectric_Constant(0);
99 
100  nDim_Electric_Field = config->GetnDim_Electric_Field();
101 
102  if (nDim != nDim_Electric_Field) {
103  SU2_MPI::Error("The electric field dimensions do not agree with the geometry.", CURRENT_FUNCTION);
104  }
105 
106  /*--- DV_Val: Vector to store the value of the design variable. ---*/
107 
108  nElectric_Field = config->GetnElectric_Field();
109 
110  /*--- We initialize the modulus ---*/
111  ref_Efield_mod = 0.0;
112  /*--- Normalize the electric field vector ---*/
113  for (iDim = 0; iDim < nDim_Electric_Field; iDim++) {
114  ref_Efield_mod += Electric_Field_Dir[iDim]*Electric_Field_Dir[iDim];
115  }
116  ref_Efield_mod = sqrt(ref_Efield_mod);
117 
118  if (ref_Efield_mod == 0) {
119  SU2_MPI::Error("The electric field has not been defined.", CURRENT_FUNCTION);
120  }
121 
122  /*--- Initialize pointer for the electric field ---*/
123  EField_Ref_Unit = new su2double[nDim_Electric_Field];
124  /*--- Assign values to the auxiliary Electric_Field structure ---*/
125  for (iDim = 0; iDim < nDim_Electric_Field; iDim++) {
126  EField_Ref_Unit[iDim] = Electric_Field_Dir[iDim]/ref_Efield_mod;
127  }
128 
129  /*--- Auxiliary vector for hosting the electric field modulus in the reference configuration ---*/
130  EField_Ref_Mod = new su2double[nElectric_Field];
131  for (iVar = 0; iVar < nElectric_Field; iVar++)
132  EField_Ref_Mod[iVar] = config->Get_Electric_Field_Mod(iVar);
133 
134  /*--- Auxiliary vector for computing the electric field in the current configuration ---*/
135  EField_Curr_Unit = new su2double[nDim_Electric_Field];
136  for (iDim = 0; iDim < nDim_Electric_Field; iDim++) {
137  EField_Curr_Unit[iDim] = 0.0;
138  }
139 
140  /*--- Auxiliary vector for storing the electric field constant ---*/
141  unsigned short nElectric_Constant = config->GetnElectric_Constant();
142  ke_DE_i = new su2double[nElectric_Field];
143 
144  if (nElectric_Constant == nElectric_Field)
145  for (iVar = 0; iVar < nElectric_Field; iVar++) ke_DE_i[iVar] = config->GetElectric_Constant(iVar);
146  else
147  for (iVar = 0; iVar < nElectric_Field; iVar++) ke_DE_i[iVar] = config->GetElectric_Constant(0);
148 
149  switch (config->GetDV_FEA()) {
150  case YOUNG_MODULUS:
151  case POISSON_RATIO:
152  case DENSITY_VAL:
153  case DEAD_WEIGHT:
154  case ELECTRIC_FIELD:
155  break;
156  default:
157  switch (config->GetDirectDiff()){
158  case D_EFIELD:
159  SU2_TYPE::SetDerivative(EField_Ref_Mod[config->GetnID_DV()],1.0);
160  break;
161  default:
162  break;
163  }
164  break;
165  }
166 
167  }

220 void CFEANonlinearElasticity::Compute_Tangent_Matrix(CElement *element, const CConfig *config) {
221 
222  unsigned short iVar, jVar, kVar;
223  unsigned short iGauss, nGauss;
224  unsigned short iNode, jNode, nNode;
225  unsigned short iDim, bDim;
226 
227  su2double Ks_Aux_ab;
228 
229  su2double Weight, Jac_x;
230 
231  su2double AuxMatrixKc[3][6];
232  su2double AuxMatrixKs[3];
233 
234  /*--- TODO: Initialize values for the material model considered ---*/
235 // cout << "PROPERTY: " << element->Get_iProp() << " and DV " << element->Get_iDV() << endl;
236  SetElement_Properties(element, config);
237  if (maxwell_stress) SetElectric_Properties(element, config);
238 
239  /*--- Register pre-accumulation inputs, material props and nodal coords ---*/
240  AD::StartPreacc();
241  AD::SetPreaccIn(E);
242  AD::SetPreaccIn(Nu);
243  if (maxwell_stress) {
244  AD::SetPreaccIn(EFieldMod_Ref);
245  AD::SetPreaccIn(ke_DE);
246  }
247  element->SetPreaccIn_Coords();
248  /*--- Recompute Lame parameters as they depend on the material properties ---*/
249  Compute_Lame_Parameters();
250 
251  /*-----------------------------------------------------------*/
252 
253  /*--- Initialize auxiliary matrices ---*/
254 
255  bDim = (nDim == 2) ? DIM_STRAIN_2D : DIM_STRAIN_3D;
256 
257  for (iVar = 0; iVar < bDim; iVar++) {
258  for (jVar = 0; jVar < nDim; jVar++) {
259  Ba_Mat[iVar][jVar] = 0.0;
260  Bb_Mat[iVar][jVar] = 0.0;
261  }
262  }
263 
264  for (iVar = 0; iVar < 3; iVar++) {
265  for (jVar = 0; jVar < 6; jVar++) {
266  AuxMatrixKc[iVar][jVar] = 0.0;
267  }
268  }
269 
270  for (iVar = 0; iVar < 3; iVar++) {
271  AuxMatrixKs[iVar] = 0.0;
272  }
273 
274  element->ClearElement(); /*--- Restarts the element to avoid adding over previous results. --*/
275  element->ComputeGrad_Linear();
276  element->ComputeGrad_NonLinear();
277 
278  nNode = element->GetnNodes();
279  nGauss = element->GetnGaussPoints();
280 
281  /*--- Full integration of the constitutive and stress term ---*/
282 
283  for (iGauss = 0; iGauss < nGauss; iGauss++) {
284 
285  Weight = element->GetWeight(iGauss);
286  Jac_x = element->GetJ_x(iGauss);
287 
288  /*--- Initialize the deformation gradient for each Gauss Point ---*/
289 
290  for (iVar = 0; iVar < 3; iVar++) {
291  for (jVar = 0; jVar < 3; jVar++) {
292  F_Mat[iVar][jVar] = 0.0;
293  b_Mat[iVar][jVar] = 0.0;
294  }
295  }
296 
297  /*--- Retrieve the values of the gradients of the shape functions for each node ---*/
298  /*--- This avoids repeated operations ---*/
299 
300  for (iNode = 0; iNode < nNode; iNode++) {
301 
302  for (iDim = 0; iDim < nDim; iDim++) {
303  GradNi_Ref_Mat[iNode][iDim] = element->GetGradNi_X(iNode,iGauss,iDim);
304  GradNi_Curr_Mat[iNode][iDim] = element->GetGradNi_x(iNode,iGauss,iDim);
305  currentCoord[iNode][iDim] = element->GetCurr_Coord(iNode, iDim);
306  }
307 
308  /*--- Compute the deformation gradient ---*/
309 
310  for (iVar = 0; iVar < nDim; iVar++) {
311  for (jVar = 0; jVar < nDim; jVar++) {
312  F_Mat[iVar][jVar] += currentCoord[iNode][iVar]*GradNi_Ref_Mat[iNode][jVar];
313  }
314  }
315  }
316 
317  if (nDim == 2) {
318  if (plane_stress) {
319  // Compute the value of the term 33 for the deformation gradient
320  Compute_Plane_Stress_Term(element, config);
321  F_Mat[2][2] = f33;
322  }
323  else { // plane strain
324  F_Mat[2][2] = 1.0;
325  }
326  }
327 
328  /*--- Determinant of F --> Jacobian of the transformation ---*/
329 
330  J_F = F_Mat[0][0]*F_Mat[1][1]*F_Mat[2][2]+
331  F_Mat[0][1]*F_Mat[1][2]*F_Mat[2][0]+
332  F_Mat[0][2]*F_Mat[1][0]*F_Mat[2][1]-
333  F_Mat[0][2]*F_Mat[1][1]*F_Mat[2][0]-
334  F_Mat[1][2]*F_Mat[2][1]*F_Mat[0][0]-
335  F_Mat[2][2]*F_Mat[0][1]*F_Mat[1][0];
336 
337  /*--- Compute the left Cauchy deformation tensor ---*/
338 
339  for (iVar = 0; iVar < 3; iVar++) {
340  for (jVar = 0; jVar < 3; jVar++) {
341  for (kVar = 0; kVar < 3; kVar++) {
342  b_Mat[iVar][jVar] += F_Mat[iVar][kVar]*F_Mat[jVar][kVar];
343  }
344  }
345  }
346 
347  /*--- Compute the constitutive matrix ---*/
348 
349  Compute_Stress_Tensor(element, config);
350 // if (maxwell_stress) Add_MaxwellStress(element, config);
351  Compute_Constitutive_Matrix(element, config);
352 
353 
354  for (iNode = 0; iNode < nNode; iNode++) {
355 
356  /*--------------------------------------------------------------------------------*/
357  /*---------------------------- NODAL STRESS TERM ---------------------------------*/
358  /*--------------------------------------------------------------------------------*/
359  /*--- Compute the nodal stress term for each gaussian point and for each node, ---*/
360  /*--- and add it to the element structure to be retrieved from the solver ---*/
361 
362  for (iVar = 0; iVar < nDim; iVar++) {
363  KAux_t_a[iVar] = 0.0;
364  for (jVar = 0; jVar < nDim; jVar++) {
365  KAux_t_a[iVar] += Weight * Stress_Tensor[iVar][jVar] * GradNi_Curr_Mat[iNode][jVar] * Jac_x;
366  }
367  }
368 
369  element->Add_Kt_a(iNode, KAux_t_a);
370 
371  /*--------------------------------------------------------------------------------*/
372  /*----------------------- CONSTITUTIVE AND STRESS TERM ---------------------------*/
373  /*--------------------------------------------------------------------------------*/
374 
375  if (nDim == 2) {
376  Ba_Mat[0][0] = GradNi_Curr_Mat[iNode][0];
377  Ba_Mat[1][1] = GradNi_Curr_Mat[iNode][1];
378  Ba_Mat[2][0] = GradNi_Curr_Mat[iNode][1];
379  Ba_Mat[2][1] = GradNi_Curr_Mat[iNode][0];
380  }
381  else if (nDim ==3) {
382  Ba_Mat[0][0] = GradNi_Curr_Mat[iNode][0];
383  Ba_Mat[1][1] = GradNi_Curr_Mat[iNode][1];
384  Ba_Mat[2][2] = GradNi_Curr_Mat[iNode][2];
385  Ba_Mat[3][0] = GradNi_Curr_Mat[iNode][1];
386  Ba_Mat[3][1] = GradNi_Curr_Mat[iNode][0];
387  Ba_Mat[4][0] = GradNi_Curr_Mat[iNode][2];
388  Ba_Mat[4][2] = GradNi_Curr_Mat[iNode][0];
389  Ba_Mat[5][1] = GradNi_Curr_Mat[iNode][2];
390  Ba_Mat[5][2] = GradNi_Curr_Mat[iNode][1];
391  }
392 
393  /*--- Compute the BT.D Matrix ---*/
394 
395  for (iVar = 0; iVar < nDim; iVar++) {
396  for (jVar = 0; jVar < bDim; jVar++) {
397  AuxMatrixKc[iVar][jVar] = 0.0;
398  for (kVar = 0; kVar < bDim; kVar++) {
399  AuxMatrixKc[iVar][jVar] += Ba_Mat[kVar][iVar]*D_Mat[kVar][jVar];
400  }
401  }
402  }
403 
404  /*--- Compute the BT.D Matrix ---*/
405 
406  for (iVar = 0; iVar < nDim; iVar++) {
407  AuxMatrixKs[iVar] = 0.0;
408  for (jVar = 0; jVar < nDim; jVar++) {
409  AuxMatrixKs[iVar] += GradNi_Curr_Mat[iNode][jVar]*Stress_Tensor[jVar][iVar]; // DOUBLE CHECK
410  }
411  }
412 
413  /*--- Assumming symmetry ---*/
414  for (jNode = iNode; jNode < nNode; jNode++) {
415  if (nDim == 2) {
416  Bb_Mat[0][0] = GradNi_Curr_Mat[jNode][0];
417  Bb_Mat[1][1] = GradNi_Curr_Mat[jNode][1];
418  Bb_Mat[2][0] = GradNi_Curr_Mat[jNode][1];
419  Bb_Mat[2][1] = GradNi_Curr_Mat[jNode][0];
420  }
421  else if (nDim ==3) {
422  Bb_Mat[0][0] = GradNi_Curr_Mat[jNode][0];
423  Bb_Mat[1][1] = GradNi_Curr_Mat[jNode][1];
424  Bb_Mat[2][2] = GradNi_Curr_Mat[jNode][2];
425  Bb_Mat[3][0] = GradNi_Curr_Mat[jNode][1];
426  Bb_Mat[3][1] = GradNi_Curr_Mat[jNode][0];
427  Bb_Mat[4][0] = GradNi_Curr_Mat[jNode][2];
428  Bb_Mat[4][2] = GradNi_Curr_Mat[jNode][0];
429  Bb_Mat[5][1] = GradNi_Curr_Mat[jNode][2];
430  Bb_Mat[5][2] = GradNi_Curr_Mat[jNode][1];
431  }
432 
433  /*--- KAux_ab is the term for the constitutive part of the tangent matrix ---*/
434  for (iVar = 0; iVar < nDim; iVar++) {
435  for (jVar = 0; jVar < nDim; jVar++) {
436  KAux_ab[iVar][jVar] = 0.0;
437  for (kVar = 0; kVar < bDim; kVar++) {
438  KAux_ab[iVar][jVar] += Weight * AuxMatrixKc[iVar][kVar] * Bb_Mat[kVar][jVar] * Jac_x;
439  }
440  }
441  }
442 
443  /*--- Ks_Aux_ab is the term for the constitutive part of the tangent matrix ---*/
444  Ks_Aux_ab = 0.0;
445  for (iVar = 0; iVar < nDim; iVar++) {
446  Ks_Aux_ab += Weight * AuxMatrixKs[iVar] * GradNi_Curr_Mat[jNode][iVar] * Jac_x;
447  }
448 
449  element->Add_Kab(iNode, jNode, KAux_ab);
450  element->Add_Ks_ab(iNode, jNode, Ks_Aux_ab);
451  /*--- Symmetric terms --*/
452  if (iNode != jNode) {
453  element->Add_Kab_T(jNode, iNode, KAux_ab);
454  element->Add_Ks_ab(jNode, iNode, Ks_Aux_ab);
455  }
456 
457  }
458 
459  }
460 
461  }
462 
463  /*--- Register the stress residual as preaccumulation output ---*/
464  element->SetPreaccOut_Kt_a();
465  AD::EndPreacc();
466 
467 }

553  F_Mat[0][1]*F_Mat[1][2]*F_Mat[2][0]+
554  F_Mat[0][2]*F_Mat[1][0]*F_Mat[2][1]-
555  F_Mat[0][2]*F_Mat[1][1]*F_Mat[2][0]-

565  }
566  }

577  /*--- and add it to the element structure to be retrieved from the solver ---*/

581  for (jVar = 0; jVar < nDim; jVar++) {
582  KAux_t_a[iVar] += Weight * Stress_Tensor[iVar][jVar] * GradNi_Curr_Mat[iNode][jVar] * Jac_x;
583  }

598 void CFEANonlinearElasticity::Add_MaxwellStress(CElement *element, const CConfig *config) {
599 
600 // Adds the Maxwell stress to the output of the stress Sxx, Syy, Szz, SVM...
601 
602  unsigned short iDim, jDim;
603 
604  su2double E0 = 0.0, E1 = 0.0, E2 = 0.0;
605  su2double E0_2 = 0.0, E1_2 = 0.0, E2_2 = 0.0;
606  su2double E_2 = 0.0;
607 
608  Compute_FmT_Mat();
609 
610  for (iDim = 0; iDim < nDim; iDim++){
611  EField_Curr_Unit[iDim] = 0.0;
612  for (jDim = 0; jDim < nDim; jDim++){
613  EField_Curr_Unit[iDim] += FmT_Mat[iDim][jDim] * EField_Ref_Unit[jDim];
614  }
615  }
616 
617  E0 = EFieldMod_Ref*EField_Curr_Unit[0]; E0_2 = pow(E0,2);
618  E1 = EFieldMod_Ref*EField_Curr_Unit[1]; E1_2 = pow(E1,2);
619  if (nDim == 3) {E2 = EFieldMod_Ref*EField_Curr_Unit[2]; E2_2 = pow(E2,2);}
620 
621  E_2 = E0_2+E1_2+E2_2;
622 
623  Stress_Tensor[0][0] += ke_DE*(E0_2-0.5*E_2); Stress_Tensor[0][1] += ke_DE*E0*E1; Stress_Tensor[0][2] += ke_DE*E0*E2;
624  Stress_Tensor[1][0] += ke_DE*E1*E0; Stress_Tensor[1][1] += ke_DE*(E1_2-0.5*E_2); Stress_Tensor[1][2] += ke_DE*E1*E2;
625  Stress_Tensor[2][0] += ke_DE*E2*E0; Stress_Tensor[2][1] += ke_DE*E2*E1; Stress_Tensor[2][2] += ke_DE*(E2_2-0.5*E_2);

744 su2double CFEANonlinearElasticity::Compute_Averaged_NodalStress(CElement *element, const CConfig *config) {
745 
746  unsigned short iVar, jVar, kVar;
747  unsigned short iGauss, nGauss;
748  unsigned short iDim, iNode, nNode;
749 
750  su2double avgStress[DIM_STRAIN_3D] = {0.0};
751 
752  /*--- TODO: Initialize values for the material model considered ---*/
753  SetElement_Properties(element, config);
754  if (maxwell_stress) SetElectric_Properties(element, config);
755  /*-----------------------------------------------------------*/
756 
757  /*--- Register pre-accumulation inputs ---*/
758  /*--- WARNING: Outputs must be registered outside of this method, this allows more
759  * flexibility in selecting what is captured by AD, capturing the entire stress
760  * tensor would use more memory than that used by the stress residuals. ---*/
761  AD::StartPreacc();
762  AD::SetPreaccIn(E);
763  AD::SetPreaccIn(Nu);
764  if (maxwell_stress) {
765  AD::SetPreaccIn(EFieldMod_Ref);
766  AD::SetPreaccIn(ke_DE);
767  }
768  element->SetPreaccIn_Coords();
769  /*--- Recompute Lame parameters as they depend on the material properties ---*/
770  Compute_Lame_Parameters();
771 
772  su2double Weight, Jac_x;
773 
774  element->ClearStress();
775  element->ClearElement(); /*--- Restarts the element to avoid adding over previous results. ---*/
776  element->ComputeGrad_Linear();
777  element->ComputeGrad_NonLinear();
778 
779  nNode = element->GetnNodes();
780  nGauss = element->GetnGaussPoints();
781 
782  /*--- Computation of the deformation gradient ---*/
783 
784  for (iGauss = 0; iGauss < nGauss; iGauss++) {
785 
786  Weight = element->GetWeight(iGauss);
787  Jac_x = element->GetJ_x(iGauss);
788 
789  /*--- Initialize the deformation gradient for each Gauss Point ---*/
790 
791  for (iVar = 0; iVar < 3; iVar++) {
792  for (jVar = 0; jVar < 3; jVar++) {
793  F_Mat[iVar][jVar] = 0.0;
794  b_Mat[iVar][jVar] = 0.0;
795  }
796  }
797 
798  /*--- Retrieve the values of the gradients of the shape functions for each node ---*/
799  /*--- This avoids repeated operations ---*/
800 
801  for (iNode = 0; iNode < nNode; iNode++) {
802 
803  for (iDim = 0; iDim < nDim; iDim++) {
804  currentCoord[iNode][iDim] = element->GetCurr_Coord(iNode,iDim);
805  GradNi_Ref_Mat[iNode][iDim] = element->GetGradNi_X(iNode,iGauss,iDim);
806  GradNi_Curr_Mat[iNode][iDim] = element->GetGradNi_x(iNode,iGauss,iDim);
807  }
808 
809  /*--- Compute the deformation gradient ---*/
810 
811  for (iVar = 0; iVar < nDim; iVar++) {
812  for (jVar = 0; jVar < nDim; jVar++) {
813  F_Mat[iVar][jVar] += currentCoord[iNode][iVar]*GradNi_Ref_Mat[iNode][jVar];
814  }
815  }
816  }
817 
818  if (nDim == 2) {
819  if (plane_stress) {
820  // Compute the value of the term 33 for the deformation gradient
821  Compute_Plane_Stress_Term(element, config);
822  F_Mat[2][2] = f33;
823  }
824  else { // plane strain
825  F_Mat[2][2] = 1.0;
826  }
827  }
828 
829  /*--- Determinant of F --> Jacobian of the transformation ---*/
830 
831  J_F = F_Mat[0][0]*F_Mat[1][1]*F_Mat[2][2]+
832  F_Mat[0][1]*F_Mat[1][2]*F_Mat[2][0]+
833  F_Mat[0][2]*F_Mat[1][0]*F_Mat[2][1]-
834  F_Mat[0][2]*F_Mat[1][1]*F_Mat[2][0]-
835  F_Mat[1][2]*F_Mat[2][1]*F_Mat[0][0]-
836  F_Mat[2][2]*F_Mat[0][1]*F_Mat[1][0];
837 
838  /*--- Compute the left Cauchy deformation tensor ---*/
839 
840  for (iVar = 0; iVar < 3; iVar++) {
841  for (jVar = 0; jVar < 3; jVar++) {
842  for (kVar = 0; kVar < 3; kVar++) {
843  b_Mat[iVar][jVar] += F_Mat[iVar][kVar]*F_Mat[jVar][kVar];
844  }
845  }
846  }
847 
848  /*--- Compute the stress tensor ---*/
849 
850  Compute_Stress_Tensor(element, config);
851  if (maxwell_stress) Add_MaxwellStress(element, config);
852 
853  avgStress[0] += Stress_Tensor[0][0] / nGauss;
854  avgStress[1] += Stress_Tensor[1][1] / nGauss;
855  avgStress[2] += Stress_Tensor[0][1] / nGauss;
856  if (nDim == 3) {
857  avgStress[3] += Stress_Tensor[2][2] / nGauss;
858  avgStress[4] += Stress_Tensor[0][2] / nGauss;
859  avgStress[5] += Stress_Tensor[1][2] / nGauss;
860  }
861 
862  for (iNode = 0; iNode < nNode; iNode++) {
863 
864  /*--- Compute the nodal stress term for each gaussian point and for each node, ---*/
865  /*--- and add it to the element structure to be retrieved from the solver ---*/
866 
867  for (iVar = 0; iVar < nDim; iVar++) {
868  KAux_t_a[iVar] = 0.0;
869  for (jVar = 0; jVar < nDim; jVar++) {
870  KAux_t_a[iVar] += Weight * Stress_Tensor[iVar][jVar] * GradNi_Curr_Mat[iNode][jVar] * Jac_x;
871  }
872  }
873 
874  element->Add_Kt_a(iNode, KAux_t_a);
875 
876  /*--- Compute the average nodal stresses for each node ---*/
877 
878  su2double Ni_Extrap = element->GetNi_Extrap(iNode, iGauss);
879 
880  element->Add_NodalStress(iNode, 0, Stress_Tensor[0][0] * Ni_Extrap);
881  element->Add_NodalStress(iNode, 1, Stress_Tensor[1][1] * Ni_Extrap);
882  element->Add_NodalStress(iNode, 2, Stress_Tensor[0][1] * Ni_Extrap);
883  if (nDim == 3) {
884  element->Add_NodalStress(iNode, 3, Stress_Tensor[2][2] * Ni_Extrap);
885  element->Add_NodalStress(iNode, 4, Stress_Tensor[0][2] * Ni_Extrap);
886  element->Add_NodalStress(iNode, 5, Stress_Tensor[1][2] * Ni_Extrap);
887  }
888 
889  }
890 
891  }
892 
893  auto elStress = VonMisesStress(nDim, avgStress);
894 
895  /*--- We only differentiate w.r.t. an avg VM stress for the element as
896  * considering all nodal stresses would use too much memory. ---*/
897  AD::SetPreaccOut(elStress);
898  AD::EndPreacc();
899 
900  return elStress;
901 }
