57 void CPrimalGridBoundFEM::GetLocalCornerPointsFace(unsigned short elementType,
58  unsigned short nPoly,
59  unsigned short nDOFs,
60  unsigned short &nPointsPerFace,
61  unsigned long faceConn[]) {
62 
63  /*--- Determine the element type and set the face data accordingly.
64  The faceConn values are local to the element. ---*/
65 
66  switch( elementType ) {
67  case LINE:
68  nPointsPerFace = 2;
69  faceConn[0] = 0; faceConn[1] = nPoly;
70  break;
71 
72  case TRIANGLE:
73  nPointsPerFace = 3;
74  faceConn[0] = 0; faceConn[1] = nPoly; faceConn[2] = nDOFs -1;
75  break;
76 
77  case QUADRILATERAL:
78  unsigned short nn2 = nPoly*(nPoly+1);
79  nPointsPerFace = 4;
80  faceConn[0] = 0; faceConn[1] = nPoly; faceConn[2] = nDOFs -1; faceConn[3] = nn2;
81  break;
82  }
83 }

85 void CPrimalGridBoundFEM::GetCornerPointsAllFaces(unsigned short &nFaces,
86  unsigned short nPointsPerFace[],
87  unsigned long faceConn[6][4]) const {
88 
89  /*--- For compatibility reasons with the base class, nFaces is also an
90  argument to this function. However, it is always 1, because the
91  boundary element is the face. ---*/
92  nFaces = 1;
93 
94  /*--- Get the corner points of the face local to the element. ---*/
95 
96  unsigned long thisFaceConn[4] = {0, 0, 0, 0};
97  GetLocalCornerPointsFace(VTK_Type, nPolyGrid, nDOFsGrid,
98  nPointsPerFace[0], thisFaceConn);
99 
100 
101  /*--- Convert the local values of thisFaceConn to global values. ---*/
102 
103  for(unsigned short j=0; j<nPointsPerFace[0]; ++j)
104  faceConn[0][j] = Nodes[thisFaceConn[j]];
105 }
