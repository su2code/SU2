39 template<class ScalarType>
40 void CPastixWrapper<ScalarType>::Initialize(CGeometry *geometry, const CConfig *config) {
41 
42  using namespace PaStiX;
43 
44  if (isinitialized) return; // only need to do this once
45 
46  unsigned long nVar = matrix.nVar,
47  nPoint = matrix.nPoint,
48  nPointDomain = matrix.nPointDomain;
49  const unsigned long *row_ptr = matrix.rowptr,
50  *col_ind = matrix.colidx;
51 
52  unsigned long iPoint, offset = 0, nNonZero = row_ptr[nPointDomain];
53 
54  /*--- Allocate ---*/
55 
56  nCols = pastix_int_t(nPointDomain);
57  colptr.resize(nPointDomain+1);
58  rowidx.clear();
59  rowidx.reserve(nNonZero);
60  values.resize(nNonZero*nVar*nVar);
61  loc2glb.resize(nPointDomain);
62  perm.resize(nPointDomain);
63  workvec.resize(nPointDomain*nVar);
64 
65  /*--- Set default parameter values ---*/
66 
67  pastix_int_t incomplete = iparm[IPARM_INCOMPLETE];
68 
69  iparm[IPARM_MODIFY_PARAMETER] = API_NO;
70  Run();
71 
72  /*--- Customize important parameters ---*/
73 
74  switch (verb) {
75  case 1:
76  iparm[IPARM_VERBOSE] = API_VERBOSE_NO;
77  break;
78  case 2:
79  iparm[IPARM_VERBOSE] = API_VERBOSE_YES;
80  break;
81  default:
82  iparm[IPARM_VERBOSE] = API_VERBOSE_NOT;
83  break;
84  }
85  iparm[IPARM_DOF_NBR] = pastix_int_t(nVar);
86  iparm[IPARM_MATRIX_VERIFICATION] = API_NO;
87  iparm[IPARM_FREE_CSCPASTIX] = API_CSC_FREE;
88  iparm[IPARM_CSCD_CORRECT] = API_NO;
89  iparm[IPARM_RHSD_CHECK] = API_NO;
90  iparm[IPARM_ORDERING] = API_ORDER_PTSCOTCH;
91  iparm[IPARM_INCOMPLETE] = incomplete;
92  iparm[IPARM_LEVEL_OF_FILL] = pastix_int_t(config->GetPastixFillLvl());
93  iparm[IPARM_THREAD_NBR] = omp_get_max_threads();
94 #if defined(HAVE_MPI) && defined(HAVE_OMP)
95  int comm_mode = MPI_THREAD_SINGLE;
96  MPI_Query_thread(&comm_mode);
97  if (comm_mode == MPI_THREAD_MULTIPLE)
98  iparm[IPARM_THREAD_COMM_MODE] = API_THREAD_MULTIPLE;
99  else
100  iparm[IPARM_THREAD_COMM_MODE] = API_THREAD_FUNNELED;
101 #endif
102 
103  /*--- Prepare sparsity structure ---*/
104 
105  /*--- We need it in global coordinates, i.e. shifted according to the position
106  of the current rank in the linear partitioning space, and "unpacked" halo part.
107  The latter forces us to re-sort the column indices of rows with halo points, which
108  in turn requires blocks to be swapped accordingly. Moreover we need "pointer" and
109  indices in Fortran-style numbering (start at 1), effectively the matrix is copied.
110  Here we prepare the pointer and index part, and map the required swaps. ---*/
111 
112  /*--- 1 - Determine position in the linear partitioning ---*/
113 
114 #ifdef HAVE_MPI
115  vector<unsigned long> domain_sizes(mpi_size);
116  MPI_Allgather(&nPointDomain, 1, MPI_UNSIGNED_LONG, domain_sizes.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
117  for (int i=0; i<mpi_rank; ++i) offset += domain_sizes[i];
118 #endif
119 
120  iota(loc2glb.begin(), loc2glb.end(), offset+1);
121 
122  /*--- 2 - Communicate global indices of halo points to then renumber
123  column indices from local to global when unpacking halos. ---*/
124 
125  vector<pastix_int_t> map(nPoint-nPointDomain,0);
126 
127 #ifdef HAVE_MPI
128  for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++)
129  {
130  if ((config->GetMarker_All_KindBC(iMarker) == SEND_RECEIVE) &&
131  (config->GetMarker_All_SendRecv(iMarker) > 0))
132  {
133  unsigned short MarkerS = iMarker, MarkerR = iMarker+1;
134 
135  int sender = config->GetMarker_All_SendRecv(MarkerS)-1;
136  int recver = abs(config->GetMarker_All_SendRecv(MarkerR))-1;
137 
138  unsigned long nVertexS = geometry->nVertex[MarkerS];
139  unsigned long nVertexR = geometry->nVertex[MarkerR];
140 
141  /*--- Allocate Send/Receive buffers ---*/
142  vector<unsigned long> Buffer_Recv(nVertexR), Buffer_Send(nVertexS);
143 
144  /*--- Prepare data to send ---*/
145  for (unsigned long iVertex = 0; iVertex < nVertexS; iVertex++)
146  Buffer_Send[iVertex] = geometry->vertex[MarkerS][iVertex]->GetNode()+offset;
147 
148  /*--- Send and Receive data ---*/
149  MPI_Sendrecv(Buffer_Send.data(), nVertexS, MPI_UNSIGNED_LONG, sender, 0,
150  Buffer_Recv.data(), nVertexR, MPI_UNSIGNED_LONG, recver, 0,
151  SU2_MPI::GetComm(), MPI_STATUS_IGNORE);
152 
153  /*--- Store received data---*/
154  for (unsigned long iVertex = 0; iVertex < nVertexR; iVertex++)
155  map[ geometry->vertex[MarkerR][iVertex]->GetNode()-nPointDomain ] = Buffer_Recv[iVertex];
156  }
157  }
158 #endif
159 
160  /*--- 3 - Copy, map the sparsity, and put it in Fortran numbering ---*/
161 
162  for (iPoint = 0; iPoint < nPointDomain; ++iPoint)
163  {
164  colptr[iPoint] = pastix_int_t(row_ptr[iPoint]+1);
165 
166  unsigned long begin = row_ptr[iPoint], end = row_ptr[iPoint+1], j;
167 
168  /*--- If last point of row is halo ---*/
169  bool sort_required = (col_ind[end-1] >= nPointDomain);
170 
171  if (sort_required)
172  {
173  unsigned long nnz_row = end-begin;
174 
175  sort_rows.push_back(iPoint);
176  sort_order.push_back(vector<unsigned long>(nnz_row));
177 
178  /*--- Sort mapped indices ("first") and keep track of source ("second")
179  for when we later need to swap blocks for these rows. ---*/
180 
181  vector<pair<pastix_int_t,unsigned long> > aux(nnz_row);
182 
183  for (j = begin; j < end; ++j)
184  {
185  if (col_ind[j] < nPointDomain)
186  aux[j-begin].first = pastix_int_t(offset+col_ind[j]+1);
187  else
188  aux[j-begin].first = pastix_int_t(map[col_ind[j]-nPointDomain]+1);
189  aux[j-begin].second = j;
190  }
191  sort(aux.begin(), aux.end());
192 
193  for (j = 0; j < nnz_row; ++j)
194  {
195  rowidx.push_back(aux[j].first);
196  sort_order.back()[j] = aux[j].second;
197  }
198  }
199  else
200  {
201  /*--- These are all internal, no need to go through map. ---*/
202  for (j = begin; j < end; ++j)
203  rowidx.push_back(pastix_int_t(offset+col_ind[j]+1));
204  }
205  }
206  colptr[nPointDomain] = pastix_int_t(nNonZero+1);
207 
208  if (rowidx.size() != nNonZero)
209  SU2_MPI::Error("Error during preparation of PaStiX data", CURRENT_FUNCTION);
210 
211  /*--- 4 - Perform ordering, symbolic factorization, and analysis steps ---*/
212 
213  if (mpi_rank == MASTER_NODE && verb > 0) cout << endl;
214 
215  iparm[IPARM_START_TASK] = API_TASK_ORDERING;
216  iparm[IPARM_END_TASK] = API_TASK_ANALYSE;
217  Run();
218 
219  if (mpi_rank == MASTER_NODE && verb > 0)
220  cout << " +--------------------------------------------------------------------+" << endl;
221 
222  isinitialized = true;
223 }

225 template<class ScalarType>
226 void CPastixWrapper<ScalarType>::Factorize(CGeometry *geometry, const CConfig *config,
227  unsigned short kind_fact) {
228  using namespace PaStiX;
229 
230  /*--- Detect a possible change of settings between direct and adjoint that requires a reset ---*/
231  if (isinitialized)
232  if ((kind_fact == PASTIX_ILU) != (iparm[IPARM_INCOMPLETE] == API_YES)) {
233  Clean();
234  isinitialized = false;
235  iter = 0;
236  }
237 
238  verb = config->GetPastixVerbLvl();
239  iparm[IPARM_INCOMPLETE] = (kind_fact == PASTIX_ILU);
240 
241  Initialize(geometry, config);
242 
243  /*--- Set some options that affect "compute" and could (one day) change during run ---*/
244 
245  switch (verb) {
246  case 1:
247  iparm[IPARM_VERBOSE] = API_VERBOSE_NO;
248  break;
249  case 2:
250  iparm[IPARM_VERBOSE] = API_VERBOSE_YES;
251  break;
252  default:
253  iparm[IPARM_VERBOSE] = API_VERBOSE_NOT;
254  break;
255  }
256 
257  if (kind_fact == PASTIX_LDLT || kind_fact == PASTIX_LDLT_P)
258  iparm[IPARM_TRANSPOSE_SOLVE] = API_NO; // symmetric so no need for slower transp. solve
259  else
260  iparm[IPARM_TRANSPOSE_SOLVE] = API_YES; // negated due to CSR to CSC copy
261 
262  /*--- Is factorizing needed on this iteration? ---*/
263 
264  bool factorize = false;
265  if (config->GetPastixFactFreq() != 0)
266  factorize = (iter % config->GetPastixFactFreq() == 0);
267 
268  iter++;
269 
270  if (isfactorized && !factorize) return; // No
271 
272  /*--- Yes ---*/
273 
274  if (mpi_rank == MASTER_NODE && verb > 0) {
275  cout << endl;
276  cout << " +--------------------------------------------------------------------+" << endl;
277  cout << " + PaStiX : Parallel Sparse matriX package +" << endl;
278  cout << " +--------------------------------------------------------------------+" << endl;
279  }
280 
281  unsigned long i, j, k, iRow, begin, target, source,
282  szBlk = matrix.nVar*matrix.nVar, nNonZero = values.size();
283 
284  /*--- Copy matrix values and swap blocks as required ---*/
285 
286  for (i = 0; i < nNonZero; ++i)
287  values[i] = SU2_TYPE::GetValue(matrix.values[i]);
288 
289  for (i = 0; i < sort_rows.size(); ++i)
290  {
291  iRow = sort_rows[i];
292  begin = matrix.rowptr[iRow];
293 
294  for (j = 0; j < sort_order[i].size(); ++j)
295  {
296  target = (begin+j)*szBlk;
297  source = sort_order[i][j]*szBlk;
298 
299  for (k = 0; k < szBlk; ++k)
300  values[target+k] = SU2_TYPE::GetValue(matrix.values[source+k]);
301  }
302  }
303 
304  /*--- Set factorization options ---*/
305 
306  switch (kind_fact) {
307  case PASTIX_LDLT: case PASTIX_LDLT_P:
308  iparm[IPARM_SYM] = API_SYM_YES;
309  iparm[IPARM_FACTORIZATION] = API_FACT_LDLT;
310  break;
311  case PASTIX_LU: case PASTIX_LU_P: case PASTIX_ILU:
312  iparm[IPARM_SYM] = API_SYM_NO;
313  iparm[IPARM_FACTORIZATION] = API_FACT_LU;
314  break;
315  default:
316  SU2_MPI::Error("Unknown type of PaStiX factorization.", CURRENT_FUNCTION);
317  break;
318  }
319 
320  /*--- Compute factorization ---*/
321 
322  iparm[IPARM_START_TASK] = API_TASK_NUMFACT;
323  iparm[IPARM_END_TASK] = API_TASK_NUMFACT;
324  Run();
325 
326  if (mpi_rank == MASTER_NODE && verb > 0)
327  cout << " +--------------------------------------------------------------------+" << endl << endl;
328 
329  isfactorized = true;
330 }
