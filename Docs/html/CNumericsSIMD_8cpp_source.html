61 /*!
62  * \brief Centered factory implementation.
63  */
64 template<class ViscousDecorator>
65 CNumericsSIMD* createCenteredNumerics(const CConfig& config, int iMesh, const CVariable* turbVars) {
66  CNumericsSIMD* obj = nullptr;
67  switch ((iMesh==MESH_0)? config.GetKind_Centered_Flow() : CENTERED::LAX) {
68  case CENTERED::NONE:
69  break;
70  case CENTERED::LAX:
71  obj = new CLaxScheme<ViscousDecorator>(config, iMesh, turbVars);
72  break;
73  case CENTERED::JST:
74  obj = new CJSTScheme<ViscousDecorator>(config, iMesh, turbVars);
75  break;
76  case CENTERED::JST_KE:
77  obj = new CJSTkeScheme<ViscousDecorator>(config, iMesh, turbVars);
78  break;
79  case CENTERED::JST_MAT:
80  obj = new CJSTmatScheme<ViscousDecorator>(config, iMesh, turbVars);
81  break;
82  }
83  return obj;
84 }

86 /*!
87  * \brief Generic factory implementation.
88  */
89 template<int nDim>
90 CNumericsSIMD* createNumerics(const CConfig& config, int iMesh, const CVariable* turbVars) {
91  CNumericsSIMD* obj = nullptr;
92  const bool ideal_gas = (config.GetKind_FluidModel() == STANDARD_AIR) ||
93  (config.GetKind_FluidModel() == IDEAL_GAS);
94 
95  switch (config.GetKind_ConvNumScheme_Flow()) {
96  case SPACE_UPWIND:
97  if (config.GetViscous()) {
98  if (ideal_gas)
99  obj = createUpwindIdealNumerics<CCompressibleViscousFlux<nDim> >(config, iMesh, turbVars);
100  else
101  obj = createUpwindGeneralNumerics<CGeneralCompressibleViscousFlux<nDim> >(config, iMesh, turbVars);
102  }
103  else {
104  if (ideal_gas)
105  obj = createUpwindIdealNumerics<CNoViscousFlux<nDim> >(config, iMesh, turbVars);
106  else
107  obj = createUpwindGeneralNumerics<CNoViscousFlux<nDim> >(config, iMesh, turbVars);
108  }
109  break;
110 
111  case SPACE_CENTERED:
112  if (config.GetViscous()) {
113  if (ideal_gas)
114  obj = createCenteredNumerics<CCompressibleViscousFlux<nDim> >(config, iMesh, turbVars);
115  else
116  obj = createCenteredNumerics<CGeneralCompressibleViscousFlux<nDim> >(config, iMesh, turbVars);
117  }
118  else {
119  obj = createCenteredNumerics<CNoViscousFlux<nDim> >(config, iMesh, turbVars);
120  }
121  break;
122  default:
123  break;
124  }
125 
126  return obj;
127 }
