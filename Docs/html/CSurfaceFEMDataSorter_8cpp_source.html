32 CSurfaceFEMDataSorter::CSurfaceFEMDataSorter(CConfig *config, CGeometry *geometry, const CFEMDataSorter* valVolumeSorter) :
33  CParallelDataSorter(config, valVolumeSorter->GetFieldNames()){
34 
35  nDim = geometry->GetnDim();
36 
37  volumeSorter = valVolumeSorter;
38 
39  connectivitySorted = false;
40 
41  /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
42  geometrical information for the FEM DG solver. ---*/
43 
44  CMeshFEM_DG *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
45 
46  unsigned long nVolElemOwned = DGGeometry->GetNVolElemOwned();
47  CVolumeElementFEM *volElem = DGGeometry->GetVolElem();
48 
49  /*--- Update the solution by looping over the owned volume elements. ---*/
50 
51  for(unsigned long l=0; l<nVolElemOwned; ++l) {
52 
53  /* Count up the number of local points we have for allocating storage. */
54 
55  for(unsigned short j=0; j<volElem[l].nDOFsSol; ++j) {
56  nLocalPointsBeforeSort++;
57  }
58  }
59 
60  SU2_MPI::Allreduce(&nLocalPointsBeforeSort, &nGlobalPointBeforeSort, 1,
61  MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
62 
63  /*--- Create the linear partitioner --- */
64 
65  linearPartitioner.Initialize(nGlobalPointBeforeSort, 0);

69 void CSurfaceFEMDataSorter::SortOutputData() {
70 
71  if (!connectivitySorted){
72  SU2_MPI::Error("Connectivity must be sorted before sorting output data", CURRENT_FUNCTION);
73  }
74 
75  const int VARS_PER_POINT = GlobalField_Counter;
76 
77  const unsigned long nElemLine = GetnElem(LINE);
78  const unsigned long nElemTria = GetnElem(TRIANGLE);
79  const unsigned long nElemQuad = GetnElem(QUADRILATERAL);
80 
81  /*---------------------------------------------------*/
82  /*--- Step 1: Determine the global DOF ID's of the */
83  /*--- locally stored surface connectivities. */
84  /*---------------------------------------------------*/
85 
86  /* Loop over the surface connectivities and store global
87  DOF ID's in a vector. Subtract 1, because the stored
88  connectivities are 1 based. */
89  globalSurfaceDOFIDs.clear();
90  globalSurfaceDOFIDs.reserve(nElemLine*N_POINTS_LINE +
91  nElemTria*N_POINTS_TRIANGLE +
92  nElemQuad*N_POINTS_QUADRILATERAL);
93 
94  for(unsigned long i=0; i<nElemLine*N_POINTS_LINE; ++i) {
95  const unsigned long globalID = Conn_Line_Par[i]-1;
96  globalSurfaceDOFIDs.push_back(globalID);
97  }
98 
99  for(unsigned long i=0; i<nElemTria*N_POINTS_TRIANGLE; ++i) {
100  const unsigned long globalID = Conn_Tria_Par[i]-1;
101  globalSurfaceDOFIDs.push_back(globalID);
102  }
103 
104  for(unsigned long i=0; i<nElemQuad*N_POINTS_QUADRILATERAL; ++i) {
105  const unsigned long globalID = Conn_Quad_Par[i]-1;
106  globalSurfaceDOFIDs.push_back(globalID);
107  }
108 
109  /* Sort globalSurfaceDOFIDs in increasing order and remove the
110  multiple entries. */
111  sort(globalSurfaceDOFIDs.begin(), globalSurfaceDOFIDs.end());
112  vector<unsigned long>::iterator lastEntry;
113  lastEntry = unique(globalSurfaceDOFIDs.begin(), globalSurfaceDOFIDs.end());
114  globalSurfaceDOFIDs.erase(lastEntry, globalSurfaceDOFIDs.end());
115 
116  /*-------------------------------------------------------------------*/
117  /*--- Step 2: Communicate the information of globalSurfaceDOFIDs ---*/
118  /*--- to the ranks, which actually store this information ---*/
119  /*--- in Parallel_Data. ---*/
120  /*-------------------------------------------------------------------*/
121 
122  /* Allocate the memory for the first index of the communication buffers. */
123  vector<vector<unsigned long> > sendBuf(size, vector<unsigned long>(0));
124 
125  /* Loop over the entries of globalSurfaceDOFIDs and fill the
126  communication buffers accordingly. */
127  for(unsigned long i=0; i<globalSurfaceDOFIDs.size(); ++i) {
128 
129  /* Search for the processor that owns this point. */
130  unsigned long iProcessor = linearPartitioner.GetRankContainingIndex(globalSurfaceDOFIDs[i]);
131 
132  /* Store the global ID in the send buffer for iProcessor. */
133  sendBuf[iProcessor].push_back(globalSurfaceDOFIDs[i]);
134  }
135 
136  /* Determine the number of DOFs to be sent to each processor. */
137  int nRankSend = 0;
138  vector<unsigned long> nDOFSend(size);
139  for(int i=0; i<size; ++i) {
140  nDOFSend[i] = sendBuf[i].size();
141  if(nDOFSend[i] && (i != rank)) ++nRankSend;
142  }
143 
144  /* Communicate nDOFSend using Alltoall. */
145  vector<unsigned long> nDOFRecv(size);
146 
147  SU2_MPI::Alltoall(nDOFSend.data(), 1, MPI_UNSIGNED_LONG,
148  nDOFRecv.data(), 1, MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
149 
150  /* Determine the number of messages this rank will receive. */
151  int nRankRecv = 0;
152  for(int i=0; i<size; ++i) {
153  if(nDOFRecv[i] && (i != rank)) ++nRankRecv;
154  }
155 
156  /* Define the receive buffers. */
157  vector<vector<unsigned long> > recvBuf(size, vector<unsigned long>(0));
158 
159 #ifdef HAVE_MPI
160  /* Launch the non-blocking sends. Do not send anything to myself. */
161  vector<SU2_MPI::Request> sendReq(nRankSend);
162  nRankSend = 0;
163  for(int i=0; i<size; ++i) {
164  if(nDOFSend[i] && (i != rank)) {
165  SU2_MPI::Isend(sendBuf[i].data(), nDOFSend[i], MPI_UNSIGNED_LONG,
166  i, rank, SU2_MPI::GetComm(), &sendReq[nRankSend]);
167  ++nRankSend;
168  }
169  }
170 
171  /* Launch the non-blocking receives. Do not receive anything from myself. */
172  vector<SU2_MPI::Request> recvReq(nRankRecv);
173  nRankRecv = 0;
174  for(int i=0; i<size; ++i) {
175  if(nDOFRecv[i] && (i != rank)) {
176  recvBuf[i].resize(nDOFRecv[i]);
177  SU2_MPI::Irecv(recvBuf[i].data(), nDOFRecv[i], MPI_UNSIGNED_LONG,
178  i, i, SU2_MPI::GetComm(), &recvReq[nRankRecv]);
179  ++nRankRecv;
180  }
181  }
182 
183 #endif
184 
185  /* Copy my own data. */
186  recvBuf[rank] = sendBuf[rank];
187 
188 #ifdef HAVE_MPI
189  /* Complete the non-blocking communication. */
190  SU2_MPI::Waitall(nRankSend, sendReq.data(), MPI_STATUSES_IGNORE);
191  SU2_MPI::Waitall(nRankRecv, recvReq.data(), MPI_STATUSES_IGNORE);
192 #endif
193 
194  /*-------------------------------------------------------------------*/
195  /*--- Step 2: Determine the number of locally stored surface DOFs ---*/
196  /*--- and store the data in Parallel_Surf_Data. ---*/
197  /*-------------------------------------------------------------------*/
198 
199  /* Store all received surface ID in one vector and sort it. */
200  globalSurfaceDOFIDs.resize(0);
201  for(int i=0; i<size; ++i) {
202  if( nDOFRecv[i] )
203  globalSurfaceDOFIDs.insert(globalSurfaceDOFIDs.end(),
204  recvBuf[i].begin(), recvBuf[i].end());
205  }
206 
207  sort(globalSurfaceDOFIDs.begin(), globalSurfaceDOFIDs.end());
208  lastEntry = unique(globalSurfaceDOFIDs.begin(), globalSurfaceDOFIDs.end());
209  globalSurfaceDOFIDs.erase(lastEntry, globalSurfaceDOFIDs.end());
210 
211  /* Allocate the memory for Parallel_Surf_Data. */
212  nPoints = globalSurfaceDOFIDs.size();
213 
214  delete [] dataBuffer;
215  dataBuffer = new passivedouble[nPoints*VARS_PER_POINT];
216 
217  /* Determine the local index of the global surface DOFs and
218  copy the data into Parallel_Surf_Data. */
219  for(unsigned long i=0; i<nPoints; ++i) {
220  const unsigned long ii = globalSurfaceDOFIDs[i] - linearPartitioner.GetCumulativeSizeBeforeRank(rank);
221 
222  for(int jj=0; jj<VARS_PER_POINT; jj++)
223  dataBuffer[i*VARS_PER_POINT+jj] = volumeSorter->GetData(jj,ii);
224  }
225 
226  /*--- Reduce the total number of surf points we have. This will be
227  needed for writing the surface solution files later. ---*/
228  SU2_MPI::Allreduce(&nPoints, &nPointsGlobal, 1,
229  MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
230 
231  /*-------------------------------------------------------------------*/
232  /*--- Step 3: Modify the surface connectivities, such that only ---*/
233  /*--- the data of surface DOFs needs to be written. ---*/
234  /*-------------------------------------------------------------------*/
235 
236  /* Determine the offset for my surface DOFs. */
237  unsigned long offsetSurfaceDOFs = 0;
238 
239  nSurfaceDOFsRanks.resize(size, 0);
240 #ifdef HAVE_MPI
241 
242  SU2_MPI::Allgather(&nPoints, 1, MPI_UNSIGNED_LONG,
243  nSurfaceDOFsRanks.data(), 1, MPI_UNSIGNED_LONG,
244  SU2_MPI::GetComm());
245 
246  for(int i=0; i<rank; ++i) offsetSurfaceDOFs += nSurfaceDOFsRanks[i];
247 #endif
248 
249  /* Create the map from the global volume numbering to the new global
250  surface numbering. */
251  map<unsigned long, unsigned long> mapGlobalVol2Surf;
252  for(unsigned long i=0; i<nPoints; ++i)
253  mapGlobalVol2Surf[globalSurfaceDOFIDs[i]] = offsetSurfaceDOFs + i;
254 
255 
256  /* Fill the receive buffers with the modified global surface DOF numbers,
257  such that this information can be returned to the original ranks. */
258  for(int i=0; i<size; ++i) {
259  for(unsigned long j=0; j<nDOFRecv[i]; ++j)
260  recvBuf[i][j] = mapGlobalVol2Surf.find(recvBuf[i][j])->second;
261  }
262 
263  /* Copy the original send buffers, because that information is
264  still needed. */
265  vector<vector<unsigned long> > originalSendBuf = sendBuf;
266 
267 #ifdef HAVE_MPI
268  /* Launch the non-blocking sends for the reverse communication, where the
269  receive buffers must be used for sending. Do not send anything to myself. */
270  nRankRecv = 0;
271  for(int i=0; i<size; ++i) {
272  if(nDOFRecv[i] && (i != rank)) {
273  SU2_MPI::Isend(recvBuf[i].data(), nDOFRecv[i], MPI_UNSIGNED_LONG,
274  i, rank+1, SU2_MPI::GetComm(), &recvReq[nRankRecv]);
275  ++nRankRecv;
276  }
277  }
278 
279  /* Launch the non-blocking receives for the reverse communication, where the
280  send buffers must be used for receiving. Do not receive anything from myself. */
281  nRankSend = 0;
282  for(int i=0; i<size; ++i) {
283  if(nDOFSend[i] && (i != rank)) {
284  SU2_MPI::Irecv(sendBuf[i].data(), nDOFSend[i], MPI_UNSIGNED_LONG,
285  i, i+1, SU2_MPI::GetComm(), &sendReq[nRankSend]);
286  ++nRankSend;
287  }
288  }
289 
290 #endif
291 
292  /* Copy my own data. */
293  sendBuf[rank] = recvBuf[rank];
294 
295 #ifdef HAVE_MPI
296  /* Complete the non-blocking communication. */
297  SU2_MPI::Waitall(nRankRecv, recvReq.data(), MPI_STATUSES_IGNORE);
298  SU2_MPI::Waitall(nRankSend, sendReq.data(), MPI_STATUSES_IGNORE);
299 #endif
300 
301  /* Clear the map mapGlobalVol2Surf and fill it with the data
302  needed to modify the surface connectivity. Note that 1 is added,
303  because the visualization softwares typically use 1 based indexing. */
304  mapGlobalVol2Surf.clear();
305 
306  for(int i=0; i<size; ++i) {
307  for(unsigned long j=0; j<nDOFSend[i]; ++j)
308  mapGlobalVol2Surf[originalSendBuf[i][j]+1] = sendBuf[i][j]+1;
309  }
310 
311  /* Modify the locally stored surface connectivities. */
312  for(unsigned long i=0; i<nElemLine*N_POINTS_LINE; ++i)
313  Conn_Line_Par[i] = mapGlobalVol2Surf.find(Conn_Line_Par[i])->second;
314 
315  for(unsigned long i=0; i<nElemTria*N_POINTS_TRIANGLE; ++i)
316  Conn_Tria_Par[i] = mapGlobalVol2Surf.find(Conn_Tria_Par[i])->second;
317 
318  for(unsigned long i=0; i<nElemQuad*N_POINTS_QUADRILATERAL; ++i)
319  Conn_Quad_Par[i] = mapGlobalVol2Surf.find(Conn_Quad_Par[i])->second;
320 }

353 void CSurfaceFEMDataSorter::SortSurfaceConnectivity(CConfig *config, CGeometry *geometry, unsigned short Elem_Type,
354  const vector<string> &markerList) {
355 
356  /* Determine the number of nodes for this element type. */
357  unsigned short NODES_PER_ELEMENT = 0;
358  switch (Elem_Type) {
359  case LINE:
360  NODES_PER_ELEMENT = N_POINTS_LINE;
361  break;
362  case TRIANGLE:
363  NODES_PER_ELEMENT = N_POINTS_TRIANGLE;
364  break;
365  case QUADRILATERAL:
366  NODES_PER_ELEMENT = N_POINTS_QUADRILATERAL;
367  break;
368  default:
369  SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
370  }
371 
372  /*--- Create an object of the class CMeshFEM_DG and retrieve the necessary
373  geometrical information for the FEM DG solver. ---*/
374  CMeshFEM_DG *DGGeometry = dynamic_cast<CMeshFEM_DG *>(geometry);
375 
376  unsigned long nVolElemOwned = DGGeometry->GetNVolElemOwned();
377  CVolumeElementFEM *volElem = DGGeometry->GetVolElem();
378 
379  const CBoundaryFEM *boundaries = DGGeometry->GetBoundaries();
380  const CFEMStandardBoundaryFace *standardBoundaryFacesSol = DGGeometry->GetStandardBoundaryFacesSol();
381 
382  /*--- Create the map from the global DOF ID to the local index.
383  Note one is added to the index value, because visualization
384  softwares typically use 1-based indexing. ---*/
385  vector<unsigned long> globalID;
386  for(unsigned long l=0; l<nVolElemOwned; ++l) {
387  for(unsigned short j=0; j<volElem[l].nDOFsSol; ++j) {
388  const unsigned long globalIndex = volElem[l].offsetDOFsSolGlobal + j + 1;
389  globalID.push_back(globalIndex);
390  }
391  }
392 
393  /*--- Determine the number of sub-elements on this rank by looping
394  over the surface elements of the boundary markers that must
395  be plotted. ---*/
396  unsigned long nSubElem_Local = 0;
397  for(unsigned short iMarker=0; iMarker < config->GetnMarker_All(); ++iMarker) {
398  if( !boundaries[iMarker].periodicBoundary ) {
399  string markerTag = boundaries[iMarker].markerTag;
400  auto it = std::find(markerList.begin(), markerList.end(), markerTag);
401  if (it != markerList.end()) {
402  const vector<CSurfaceElementFEM> &surfElem = boundaries[iMarker].surfElem;
403  for(unsigned long i=0; i<surfElem.size(); ++i) {
404  const unsigned short ind = surfElem[i].indStandardElement;
405  const unsigned short VTK_Type = standardBoundaryFacesSol[ind].GetVTK_Type();
406  if(Elem_Type == VTK_Type) nSubElem_Local += standardBoundaryFacesSol[ind].GetNSubFaces();
407  }
408  }
409  }
410  }
411 
412  /* Allocate the memory to store the connectivity if the size is
413  larger than zero. */
414  int *Conn_SubElem = nullptr;
415  if(nSubElem_Local > 0) Conn_SubElem = new int[nSubElem_Local*NODES_PER_ELEMENT]();
416 
417  /*--- Repeat the loop over the surface elements of the boundary markers
418  that must be plotted, but now store the connectivity. ---*/
419  unsigned long kNode = 0;
420  for(unsigned short iMarker=0; iMarker < config->GetnMarker_All(); ++iMarker) {
421  if( !boundaries[iMarker].periodicBoundary ) {
422  string markerTag = boundaries[iMarker].markerTag;
423  auto it = std::find(markerList.begin(), markerList.end(), markerTag);
424  if (it != markerList.end()) {
425  const vector<CSurfaceElementFEM> &surfElem = boundaries[iMarker].surfElem;
426 
427  /* Loop over the surface elements of this boundary marker. */
428  for(unsigned long i=0; i<surfElem.size(); ++i) {
429 
430  /* Check if this is the element type to be stored. */
431  const unsigned short ind = surfElem[i].indStandardElement;
432  const unsigned short VTK_Type = standardBoundaryFacesSol[ind].GetVTK_Type();
433  if(Elem_Type == VTK_Type) {
434 
435  /* Get the number of sub-elements and the local connectivity of
436  the sub-elements. */
437  const unsigned short nSubFaces = standardBoundaryFacesSol[ind].GetNSubFaces();
438  const unsigned short *connSubFaces = standardBoundaryFacesSol[ind].GetSubFaceConn();
439 
440  /* Store the global connectivities. */
441  const unsigned short kk = NODES_PER_ELEMENT*nSubFaces;
442  for(unsigned short k=0; k<kk; ++k, ++kNode)
443  Conn_SubElem[kNode] = globalID[surfElem[i].DOFsSolFace[connSubFaces[k]]];
444  }
445  }
446  }
447  }
448  }
449 
450  nElemPerType[TypeMap.at(Elem_Type)] = nSubElem_Local;
451 
452  /*--- Store the particular global element count in the class data,
453  and set the class data pointer to the connectivity array. ---*/
454  switch (Elem_Type) {
455  case LINE:
456  delete [] Conn_Line_Par;
457  Conn_Line_Par = Conn_SubElem;
458  break;
459  case TRIANGLE:
460  delete [] Conn_Tria_Par;
461  Conn_Tria_Par = Conn_SubElem;
462  break;
463  case QUADRILATERAL:
464  delete [] Conn_Quad_Par;
465  Conn_Quad_Par = Conn_SubElem;
466  break;
467  default:
468  SU2_MPI::Error("Unrecognized element type", CURRENT_FUNCTION);
469  break;
470  }
471 
472 }
