48 unique_ptr<CViscosityModel> CFluidModel::MakeLaminarViscosityModel(const CConfig* config, unsigned short iSpecies) {
49  switch (config->GetKind_ViscosityModel()) {
50  case VISCOSITYMODEL::CONSTANT:
51  return unique_ptr<CConstantViscosity>(new CConstantViscosity(config->GetMu_ConstantND(iSpecies)));
52  case VISCOSITYMODEL::COOLPROP:
53  return unique_ptr<CCoolPropViscosity>(new CCoolPropViscosity(config->GetFluid_Name()));
54  case VISCOSITYMODEL::SUTHERLAND:
55  return unique_ptr<CSutherland>(new CSutherland(config->GetMu_RefND(iSpecies),
56  config->GetMu_Temperature_RefND(iSpecies),
57  config->GetMu_SND(iSpecies)));
58  case VISCOSITYMODEL::POLYNOMIAL:
59  return unique_ptr<CPolynomialViscosity<N_POLY_COEFFS>>(
60  new CPolynomialViscosity<N_POLY_COEFFS>(config->GetMu_PolyCoeffND()));
61  default:
62  SU2_MPI::Error("Viscosity model not available.", CURRENT_FUNCTION);
63  return nullptr;
64  }
65 }

71 unique_ptr<CConductivityModel> CFluidModel::MakeThermalConductivityModel(const CConfig* config,
72  unsigned short iSpecies) {
73  switch (config->GetKind_ConductivityModel()) {
74  case CONDUCTIVITYMODEL::CONSTANT:
75  if (config->GetKind_ConductivityModel_Turb() == CONDUCTIVITYMODEL_TURB::CONSTANT_PRANDTL) {
76  return unique_ptr<CConstantConductivityRANS>(
77  new CConstantConductivityRANS(config->GetThermal_Conductivity_ConstantND(iSpecies),
78  config->GetPrandtl_Turb(iSpecies)));
79  } else {
80  return unique_ptr<CConstantConductivity>(
81  new CConstantConductivity(config->GetThermal_Conductivity_ConstantND(iSpecies)));
82  }
83  break;
84  case CONDUCTIVITYMODEL::COOLPROP:
85  if (config->GetKind_ConductivityModel_Turb() == CONDUCTIVITYMODEL_TURB::CONSTANT_PRANDTL) {
86  return unique_ptr<CConstantConductivityRANS>(
87  new CConstantConductivityRANS(config->GetThermal_Conductivity_ConstantND(iSpecies),
88  config->GetPrandtl_Turb(iSpecies)));
89  } else {
90  return unique_ptr<CCoolPropConductivity>(
91  new CCoolPropConductivity(config->GetFluid_Name()));
92  }
93  break;
94  case CONDUCTIVITYMODEL::CONSTANT_PRANDTL:
95  if (config->GetKind_ConductivityModel_Turb() == CONDUCTIVITYMODEL_TURB::CONSTANT_PRANDTL) {
96  return unique_ptr<CConstantPrandtlRANS>(
97  new CConstantPrandtlRANS(config->GetPrandtl_Lam(iSpecies), config->GetPrandtl_Turb(iSpecies)));
98  } else {
99  return unique_ptr<CConstantPrandtl>(new CConstantPrandtl(config->GetPrandtl_Lam(iSpecies)));
100  }
101  break;
102  case CONDUCTIVITYMODEL::POLYNOMIAL:
103  if (config->GetKind_ConductivityModel_Turb() == CONDUCTIVITYMODEL_TURB::CONSTANT_PRANDTL) {
104  return unique_ptr<CPolynomialConductivityRANS<N_POLY_COEFFS>>(
105  new CPolynomialConductivityRANS<N_POLY_COEFFS>(config->GetKt_PolyCoeffND(), config->GetPrandtl_Turb()));
106  } else {
107  return unique_ptr<CPolynomialConductivity<N_POLY_COEFFS>>(
108  new CPolynomialConductivity<N_POLY_COEFFS>(config->GetKt_PolyCoeffND()));
109  }
110  break;
111  default:
112  SU2_MPI::Error("Conductivity model not available.", CURRENT_FUNCTION);
113  return nullptr;
114  break;
115  }
116 }

122 unique_ptr<CDiffusivityModel> CFluidModel::MakeMassDiffusivityModel(const CConfig* config, unsigned short iSpecies) {
123  switch (config->GetKind_Diffusivity_Model()) {
124  case DIFFUSIVITYMODEL::CONSTANT_DIFFUSIVITY:
125  return unique_ptr<CConstantDiffusivity>(new CConstantDiffusivity(config->GetDiffusivity_ConstantND()));
126  break;
127  case DIFFUSIVITYMODEL::CONSTANT_SCHMIDT:
128  return unique_ptr<CConstantSchmidt>(new CConstantSchmidt(config->GetSchmidt_Number_Laminar()));
129  break;
130  case DIFFUSIVITYMODEL::UNITY_LEWIS:
131  return unique_ptr<CConstantLewisDiffusivity>(new CConstantLewisDiffusivity(1.0));
132  break;
133  case DIFFUSIVITYMODEL::CONSTANT_LEWIS:
134  return unique_ptr<CConstantLewisDiffusivity>(
135  new CConstantLewisDiffusivity(config->GetConstant_Lewis_Number(iSpecies)));
136  break;
137  default:
138  SU2_MPI::Error("Diffusivity model not available.", CURRENT_FUNCTION);
139  return nullptr;
140  }
141 }
