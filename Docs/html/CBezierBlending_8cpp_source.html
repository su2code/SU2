48 su2double CBezierBlending::GetBernstein(short val_n, short val_i, su2double val_t){
49 
50  su2double value = 0.0;
51 
52  if (val_i > val_n) { value = 0.0; return value; }
53 
54  if (val_i == 0) {
55  if (val_t == 0) value = 1.0;
56  else if (val_t == 1) value = 0.0;
57  else value = Binomial(val_n, val_i) * pow(val_t, val_i) * pow(1.0 - val_t, val_n - val_i);
58  }
59  else if (val_i == val_n) {
60  if (val_t == 0) value = 0.0;
61  else if (val_t == 1) value = 1.0;
62  else value = pow(val_t, val_n);
63  }
64  else {
65  if ((val_t == 0) || (val_t == 1)) value = 0.0;
66  value = Binomial(val_n, val_i) * pow(val_t, val_i) * pow(1.0-val_t, val_n - val_i);
67  }
68 
69  return value;
70 }

103 su2double CBezierBlending::Binomial(unsigned short n, unsigned short m){
104 
105  unsigned short i, j;
106  su2double result;
107 
108  binomial[0] = 1.0;
109  for (i = 1; i <= n; ++i) {
110  binomial[i] = 1.0;
111  for (j = i-1U; j > 0; --j) {
112  binomial[j] += binomial[j-1U];
113  }
114  }
115 
116  result = binomial[m];
117  if (fabs(result) < EPS*EPS) { result = 0.0; }
