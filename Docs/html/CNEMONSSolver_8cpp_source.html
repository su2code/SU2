57 void CNEMONSSolver::Preprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh,
58  unsigned short iRKStep, unsigned short RunTime_EqSystem, bool Output) {
59 
60  const auto InnerIter = config->GetInnerIter();
61  const bool limiter = (config->GetKind_SlopeLimit_Flow() != LIMITER::NONE) && (InnerIter <= config->GetLimiterIter());
62  const bool van_albada = config->GetKind_SlopeLimit_Flow() == LIMITER::VAN_ALBADA_EDGE;
63  const bool muscl = config->GetMUSCL_Flow() && (iMesh == MESH_0);
64  const bool center = config->GetKind_ConvNumScheme_Flow() == SPACE_CENTERED;
65  const bool wall_functions = config->GetWall_Functions();
66 
67  /*--- Common preprocessing steps (implemented by CNEMOEulerSolver) ---*/
68 
69  CommonPreprocessing(geometry, solver_container, config, iMesh, iRKStep, RunTime_EqSystem, Output);
70 
71  /*--- Compute gradient for MUSCL reconstruction. ---*/
72 
73  if (config->GetReconstructionGradientRequired() && muscl && !center) {
74  switch (config->GetKind_Gradient_Method_Recon()) {
75  case GREEN_GAUSS:
76  SetPrimitive_Gradient_GG(geometry, config, true); break;
77  case LEAST_SQUARES:
78  case WEIGHTED_LEAST_SQUARES:
79  SetPrimitive_Gradient_LS(geometry, config, true); break;
80  default: break;
81  }
82  }
83 
84  /*--- Compute gradient of the primitive variables ---*/
85 
86  if (config->GetKind_Gradient_Method() == GREEN_GAUSS) {
87  SetPrimitive_Gradient_GG(geometry, config);
88  }
89  else if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) {
90  SetPrimitive_Gradient_LS(geometry, config);
91  }
92 
93  /*--- Compute the limiters ---*/
94 
95  if (muscl && !center && limiter && !van_albada && !Output) {
96  SetPrimitive_Limiter(geometry, config);
97  }
98 
99  /*--- Compute vorticity and strain mag. ---*/
100 
101  ComputeVorticityAndStrainMag(*config, geometry, iMesh);
102 
103  /*--- Compute the TauWall from the wall functions ---*/
104 
105  if (wall_functions) {
106  SetTau_Wall_WF(geometry, solver_container, config);
107  }
108 }

174 void CNEMONSSolver::Viscous_Residual(CGeometry *geometry,
175  CSolver **solver_container,
176  CNumerics **numerics_container,
177  CConfig *config, unsigned short iMesh,
178  unsigned short iRKStep) {
179 
180  bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
181  CNumerics* numerics = numerics_container[VISC_TERM];
182 
183  for (unsigned long iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
184 
185  /*--- Points, coordinates and normal vector in edge ---*/
186  auto iPoint = geometry->edges->GetNode(iEdge, 0);
187  auto jPoint = geometry->edges->GetNode(iEdge, 1);
188  numerics->SetCoord(geometry->nodes->GetCoord(iPoint),
189  geometry->nodes->GetCoord(jPoint) );
190  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
191 
192  /*--- Primitive variables, and gradient ---*/
193  numerics->SetConservative (nodes->GetSolution(iPoint),
194  nodes->GetSolution(jPoint) );
195  numerics->SetPrimitive (nodes->GetPrimitive(iPoint),
196  nodes->GetPrimitive(jPoint) );
197  numerics->SetPrimVarGradient(nodes->GetGradient_Primitive(iPoint),
198  nodes->GetGradient_Primitive(jPoint) );
199 
200  /*--- Pass supplementary information to CNumerics ---*/
201  numerics->SetdPdU (nodes->GetdPdU(iPoint), nodes->GetdPdU(jPoint));
202  numerics->SetdTdU (nodes->GetdTdU(iPoint), nodes->GetdTdU(jPoint));
203  numerics->SetdTvedU(nodes->GetdTvedU(iPoint), nodes->GetdTvedU(jPoint));
204  numerics->SetEve (nodes->GetEve(iPoint), nodes->GetEve(jPoint));
205  numerics->SetCvve (nodes->GetCvve(iPoint), nodes->GetCvve(jPoint));
206 
207  /*--- Species diffusion coefficients ---*/
208  numerics->SetDiffusionCoeff(nodes->GetDiffusionCoeff(iPoint),
209  nodes->GetDiffusionCoeff(jPoint) );
210 
211  /*--- Laminar viscosity ---*/
212  numerics->SetLaminarViscosity(nodes->GetLaminarViscosity(iPoint),
213  nodes->GetLaminarViscosity(jPoint) );
214 
215  /*--- Eddy viscosity ---*/
216  numerics->SetEddyViscosity(nodes->GetEddyViscosity(iPoint),
217  nodes->GetEddyViscosity(jPoint) );
218 
219  /*--- Thermal conductivity ---*/
220  numerics->SetThermalConductivity(nodes->GetThermalConductivity(iPoint),
221  nodes->GetThermalConductivity(jPoint));
222 
223  /*--- Vib-el. thermal conductivity ---*/
224  numerics->SetThermalConductivity_ve(nodes->GetThermalConductivity_ve(iPoint),
225  nodes->GetThermalConductivity_ve(jPoint) );
226 
227  /*--- Compute and update residual ---*/
228  auto residual = numerics->ComputeResidual(config);
229 
230  /*--- Check for NaNs before applying the residual to the linear system ---*/
231  bool err = CNumerics::CheckResidualNaNs(implicit, nVar, residual);
232 
233  /*--- Update the residual and Jacobian ---*/
234  if (!err) {
235  LinSysRes.SubtractBlock(iPoint, residual);
236  LinSysRes.AddBlock(jPoint, residual);
237  if (implicit) {
238  Jacobian.UpdateBlocks(iEdge, iPoint, jPoint, residual.jacobian_i, residual.jacobian_j);
239  }
240  }
241  } //iEdge
242 }

244 void CNEMONSSolver::BC_HeatFluxNonCatalytic_Wall(CGeometry *geometry,
245  CSolver **solver_container,
246  CNumerics *conv_numerics,
247  CNumerics *sour_numerics,
248  CConfig *config,
249  unsigned short val_marker) {
250 
251  /*--- Local variables ---*/
252  const auto Marker_Tag = config->GetMarker_All_TagBound(val_marker);
253  su2double Wall_HeatFlux = config->GetWall_HeatFlux(Marker_Tag) / config->GetHeat_Flux_Ref();
254  if (config->GetIntegrated_HeatFlux()) {
255  Wall_HeatFlux /= geometry->GetSurfaceArea(config, val_marker);
256  }
257  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
258 
259  /*--- Set "Proportional control" coefficient ---*/
260  const su2double pcontrol = 1.0;
261 
262  /*--- Get the locations of the primitive variables ---*/
263  const unsigned short T_INDEX = nodes->GetTIndex();
264  const unsigned short TVE_INDEX = nodes->GetTveIndex();
265 
266  /*--- Loop over all of the vertices on this boundary marker ---*/
267  SU2_OMP_FOR_DYN(OMP_MIN_SIZE)
268  for(auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
269 
270  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
271 
272  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
273  if (!geometry->nodes->GetDomain(iPoint)) continue;
274 
275  /*--- Compute dual-grid area and boundary normal ---*/
276  const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
277  su2double Area = GeometryToolbox::Norm(nDim, Normal);
278 
279  /*--- Initialize the convective & viscous residuals to zero ---*/
280  for (auto iVar = 0u; iVar < nVar; iVar++) {Res_Visc[iVar] = 0.0;}
281 
282  /*--- Set the residual on the boundary with the specified heat flux ---*/
283  // TODO: Look into this!
284  // Note: Contributions from qtr and qve are used for proportional control
285  // to drive the solution toward the specified heatflux more quickly.
286  const auto GradV = nodes->GetGradient_Primitive(iPoint);
287  su2double dTdn = 0.0;
288  su2double dTvedn = 0.0;
289  for (auto iDim = 0u; iDim < nDim; iDim++) {
290  dTdn += GradV[T_INDEX][iDim]*Normal[iDim];
291  dTvedn += GradV[TVE_INDEX][iDim]*Normal[iDim];
292  }
293  su2double ktr = nodes->GetThermalConductivity(iPoint);
294  su2double kve = nodes->GetThermalConductivity_ve(iPoint);
295 
296  /*--- Compute residual ---*/
297  Res_Visc[nSpecies+nDim] += pcontrol*(ktr*dTdn+kve*dTvedn) +
298  Wall_HeatFlux*Area;
299  Res_Visc[nSpecies+nDim+1] += pcontrol*(kve*dTvedn) +
300  Wall_HeatFlux*Area;
301 
302  /*--- Impose the value of the velocity as a strong boundary
303  condition (Dirichlet). Fix the velocity and remove any
304  contribution to the residual at this node. ---*/
305  su2double zero[MAXNDIM] = {0.0};
306  nodes->SetVelocity_Old(iPoint, zero);
307 
308  for (auto iDim = 0u; iDim < nDim; iDim++)
309  LinSysRes(iPoint, nSpecies+iDim) = 0.0;
310  nodes->SetVel_ResTruncError_Zero(iPoint);
311 
312  /*--- Apply viscous residual to the linear system ---*/
313  LinSysRes.SubtractBlock(iPoint, Res_Visc);
314 
315  if (implicit) {
316  /*--- Enforce the no-slip boundary condition in a strong way ---*/
317  for (int iVar = nSpecies; iVar < nSpecies+nDim; iVar++) {
318  auto total_index = iPoint*nVar+iVar;
319  Jacobian.DeleteValsRowi(total_index);
320  }
321  }
322 
323  }
324  END_SU2_OMP_FOR
325 }

549  su2double Twall = config->GetIsothermal_Temperature(Marker_Tag);

553  Jacobian_i = new su2double* [nVar];
554  for (auto iVar = 0u; iVar < nVar; iVar++)
555  Jacobian_i[iVar] = new su2double [nVar] ();
556  }

559  SU2_OMP_FOR_DYN(OMP_MIN_SIZE)

576  const auto Coord_i = geometry->nodes->GetCoord(iPoint);
577  const auto Coord_j = geometry->nodes->GetCoord(Point_Normal);

583  su2double zero[MAXNDIM] = {0.0};
584  nodes->SetVelocity_Old(iPoint, zero);

587  for (auto iVar = 0u; iVar < nVar; iVar ++) {Res_Visc[iVar] = 0.0;}

613  for (auto jVar = 0u; jVar < nVar; jVar++)
614  Jacobian_i[iVar][jVar] = 0.0;

617  const auto dTdU = nodes->GetdTdU(iPoint);
618  const auto dTvedU = nodes->GetdTvedU(iPoint);
619  const su2double theta = GeometryToolbox::SquaredNorm(nDim, UnitNormal);

623  kve*theta/dist_ij*dTvedU[iVar])*Area;
624  Jacobian_i[nSpecies+nDim+1][iVar] = - kve*theta/dist_ij*dTvedU[iVar]*Area;
625  }
626  } // implicit

629  LinSysRes.SubtractBlock(iPoint, Res_Visc);

632  modifying the velocity-rows of the Jacobian (1 on the diagonal).
633  And add the contributions to the Jacobian due to energy. ---*/
634  if (implicit) {
635  Jacobian.SubtractBlock2Diag(iPoint, Jacobian_i);
636 
637  for (auto iVar = 1u; iVar <= nDim; iVar++) {
638  auto total_index = iPoint*nVar+iVar;
639  Jacobian.DeleteValsRowi(total_index);
640  }
641  }
642  }
643  END_SU2_OMP_FOR
644 
645  if (Jacobian_i)
646  for (auto iVar = 0u; iVar < nVar; iVar++)
647  delete [] Jacobian_i[iVar];
648  delete [] Jacobian_i;

652 void CNEMONSSolver::BC_IsothermalCatalytic_Wall(CGeometry *geometry,
653  CSolver **solver_container,
654  CNumerics *conv_numerics,
655  CNumerics *sour_numerics,
656  CConfig *config,
657  unsigned short val_marker) {
658 
659  /*--- Call standard isothermal BC to apply no-slip and energy b.c.'s ---*/
660  BC_IsothermalNonCatalytic_Wall(geometry, solver_container, conv_numerics,
661  sour_numerics, config, val_marker);
662 
663  /*--- Local variables ---*/
664  unsigned short iSpecies, jSpecies, iVar, jVar, kVar;
665  su2double **GradY, **dVdU;
666 
667  /*--- Assign booleans ---*/
668  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
669  const bool Supercatalytic_Wall = config->GetSupercatalytic_Wall();
670 
671  /*--- Get universal information ---*/
672  const su2double RuSI = UNIVERSAL_GAS_CONSTANT;
673  const su2double Ru = 1000.0*RuSI;
674  const auto& Ms = FluidModel->GetSpeciesMolarMass();
675 
676  /*--- Get the locations of the primitive variables ---*/
677  const unsigned short RHOS_INDEX = nodes->GetRhosIndex();
678  const unsigned short RHO_INDEX = nodes->GetRhoIndex();
679  const unsigned short T_INDEX = nodes->GetTIndex();
680  const unsigned short TVE_INDEX = nodes->GetTveIndex();
681 
682  /*--- Allocate arrays ---*/
683  GradY = new su2double*[nSpecies];
684  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
685  GradY[iSpecies] = new su2double[nDim];
686  dVdU = new su2double*[nVar];
687  for (iVar = 0; iVar < nVar; iVar++)
688  dVdU[iVar] = new su2double[nVar];
689 
690  /*--- Loop over all of the vertices on this boundary marker ---*/
691  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
692 
693  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
694 
695  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
696  if (geometry->nodes->GetDomain(iPoint)) {
697 
698  /*--- Compute closest normal neighbor ---*/
699  const auto jPoint = geometry->vertex[val_marker][iVertex]->GetNormal_Neighbor();
700 
701  /*--- Compute distance between wall & normal neighbor ---*/
702  const auto Coord_i = geometry->nodes->GetCoord(iPoint);
703  const auto Coord_j = geometry->nodes->GetCoord(jPoint);
704  const su2double dij = GeometryToolbox::Distance(nDim, Coord_i, Coord_j);
705 
706  /*--- Compute dual-grid area and boundary normal ---*/
707  const auto Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
708  const su2double Area = GeometryToolbox::Norm(nDim, Normal);
709 
710  /*--- Initialize the viscous residual to zero ---*/
711  for (iVar = 0; iVar < nVar; iVar++) Res_Visc[iVar] = 0.0;
712 
713  /*--- Get primitive information ---*/
714  const auto& Vi = nodes->GetPrimitive(iPoint);
715  const auto& Vj = nodes->GetPrimitive(jPoint);
716  const auto& Di = nodes->GetDiffusionCoeff(iPoint);
717  const auto& eves = nodes->GetEve(iPoint);
718  const auto& hs = FluidModel->ComputeSpeciesEnthalpy(Vi[T_INDEX], Vi[TVE_INDEX], eves);
719  const su2double rho = Vi[RHO_INDEX];
720  const auto& dTdU = nodes->GetdTdU(iPoint);
721  const auto& dTvedU = nodes->GetdTvedU(iPoint);
722 
723  if (Supercatalytic_Wall) {
724 
725  const auto& Yst = config->GetSupercatalytic_Wall_Composition();
726 
727  /*--- Calculate supplementary quantities ---*/
728  su2double dYdn = 0.0, SdYdn = 0.0;
729  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
730  dYdn = (Yst[iSpecies]-Vj[RHOS_INDEX+iSpecies]/Vj[RHO_INDEX])/dij;
731  SdYdn += rho*Di[iSpecies]*dYdn;
732  }
733 
734  /*--- Calculate species residual at wall ---*/
735  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
736  dYdn = (Yst[iSpecies]-Vj[RHOS_INDEX+iSpecies]/Vj[RHO_INDEX])/dij;
737  Res_Visc[iSpecies] = -(-rho*Di[iSpecies]*dYdn+Yst[iSpecies]*SdYdn)*Area;
738  }
739 
740  if (implicit) {
741  /*--- Initialize the transformation matrix ---*/
742  for (iVar = 0; iVar < nVar; iVar++)
743  for (jVar = 0; jVar < nVar; jVar++) {
744  dVdU[iVar][jVar] = 0.0;
745  Jacobian_j[iVar][jVar] = 0.0;
746  Jacobian_i[iVar][jVar] = 0.0;
747  }
748 
749  /*--- Populate transformation matrix ---*/
750  // dYsdrk
751  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
752  for (jSpecies = 0; jSpecies < nSpecies; jSpecies++)
753  dVdU[iSpecies][jSpecies] += -1.0/rho*Yst[iSpecies];
754  dVdU[iSpecies][iSpecies] += 1.0/rho;
755  }
756  for (iVar = 0; iVar < nVar; iVar++) {
757  dVdU[nSpecies+nDim][iVar] = dTdU[iVar];
758  dVdU[nSpecies+nDim+1][iVar] = dTvedU[iVar];
759  }
760 
761  /*--- Calculate supplementary quantities ---*/
762  const auto& Cvtrs = FluidModel->GetSpeciesCvTraRot();
763  const auto& Cvve = nodes->GetCvve(iPoint);
764 
765  /*--- Take the primitive var. Jacobian & store in Jac. jj ---*/
766  // Species mass fraction
767  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
768  for (jSpecies = 0; jSpecies < nSpecies; jSpecies++)
769  Jacobian_j[iSpecies][jSpecies] += -Yst[iSpecies]*rho*Di[jSpecies]/dij;
770  Jacobian_j[iSpecies][iSpecies] += rho*Di[iSpecies]/dij - SdYdn;
771  }
772 
773  // Temperature
774  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
775  for (jSpecies = 0; jSpecies < nSpecies; jSpecies++) {
776  Jacobian_j[nSpecies+nDim][iSpecies] += Jacobian_j[jSpecies][iSpecies]*hs[iSpecies];
777  }
778  Jacobian_j[nSpecies+nDim][nSpecies+nDim] += Res_Visc[iSpecies]/Area*(Ru/Ms[iSpecies] +
779  Cvtrs[iSpecies] );
780  Jacobian_j[nSpecies+nDim][nSpecies+nDim+1] += Res_Visc[iSpecies]/Area*Cvve[iSpecies];
781  }
782 
783  // Vib.-El. Temperature
784  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
785  for (jSpecies = 0; jSpecies < nSpecies; jSpecies++)
786  Jacobian_j[nSpecies+nDim+1][iSpecies] += Jacobian_j[jSpecies][iSpecies]*eves[iSpecies];
787  Jacobian_j[nSpecies+nDim+1][nSpecies+nDim+1] += Res_Visc[iSpecies]/Area*Cvve[iSpecies];
788  }
789 
790  /*--- Multiply by the transformation matrix and store in Jac. ii ---*/
791  for (iVar = 0; iVar < nVar; iVar++)
792  for (jVar = 0; jVar < nVar; jVar++)
793  for (kVar = 0; kVar < nVar; kVar++)
794  Jacobian_i[iVar][jVar] += Jacobian_j[iVar][kVar]*dVdU[kVar][jVar]*Area;
795 
796  /*--- Apply to the linear system ---*/
797  Jacobian.SubtractBlock(iPoint, iPoint, Jacobian_i);
798  }
799 
800  } else {
801 
802  if (implicit) {
803  SU2_MPI::Error("Implicit not currently available for partially catalytic wall.",CURRENT_FUNCTION);
804  }
805 
806  /*--- Identify the boundary ---*/
807  string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
808 
809  /*--- Get isothermal wall temp ----*/
810  const su2double Tw = config->GetIsothermal_Temperature(Marker_Tag);
811 
812  /*--- Get wall catalytic efficiency ----*/
813  const su2double gam = config->GetCatalytic_Efficiency();
814 
815  /*--- Get cataltyic reaction map ---*/
816  const auto& RxnTable = FluidModel->GetCatalyticRecombination();
817 
818  /*--- Common catalytic flux factor ---*/
819  const su2double factor = gam*rho*sqrt(RuSI*Tw/2/PI_NUMBER)*Area;
820 
821  /*--- Compute catalytic recombination flux ---*/
822  // Ref: 10.2514/6.2022-1636
823  // ws = gam_s*Ys*rho_wall*sqrt(Ru*Tw/(2*Pi*M_combine)*Area
824  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
825  int Index = SU2_TYPE::Int(RxnTable(iSpecies,1));
826  Res_Visc[iSpecies] = RxnTable(iSpecies,0)*factor*Vi[Index]/Vi[RHO_INDEX]*sqrt(1/Ms[Index]);
827  }
828  }
829 
830  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
831  Res_Visc[nSpecies+nDim] += (Res_Visc[iSpecies]*hs[iSpecies])*Area;
832  Res_Visc[nSpecies+nDim+1] += (Res_Visc[iSpecies]*eves[iSpecies])*Area;
833  }
834 
835  /*--- Viscous contribution to the residual at the wall ---*/
836  LinSysRes.SubtractBlock(iPoint, Res_Visc);
837  }
838  }
839 
840  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
841  delete [] GradY[iSpecies];
842  delete [] GradY;
843  for (iVar = 0; iVar < nVar; iVar++)
844  delete [] dVdU[iVar];
845  delete [] dVdU;

991  (ktr*(Tslip-Ti) + kve*(Tslip_ve-Tvei))*C)*Area/dij;
