117  /*!
118  * \brief Get the passive value of any variable. For most types return directly,

120  * \note This is a struct instead of a function because the return type of the

154  /*!
155  * \brief Wrapper to sprintf to be able to print active types and AD expressions.
156  * \note This is for compatibility with old code, stringstreams should be the preferred way to build strings.
157  * \param[in] str - Target char buffer.
158  * \param[in] format - Format string.
159  * \param[in] args - Values to be printed to the string.
160  */
161  template<class... Ts>
162  FORCEINLINE void sprintf(char* str, const char* format, Ts&&... args) {
163  ::sprintf(str, format, SU2_TYPE::_printGetValue(args)...);
164  }
