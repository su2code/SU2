35 : CMeshReaderFVM(val_config, val_iZone, val_nZone) {
36 
37  /* The rectangular mesh is always 2D. */

40  /* Set the VTK type for the interior elements and the boundary elements. */

67 void CRectangularMeshReaderFVM::ComputeRectangularPointCoordinates() {
68 
69  /* Set the global count of points based on the grid dimensions. */
70  numberOfGlobalPoints = (nNode)*(mNode);
71 
72  /* Get a partitioner to help with linear partitioning. */
73  CLinearPartitioner pointPartitioner(numberOfGlobalPoints,0);
74 
75  /* Determine number of local points */
76  for(unsigned long globalIndex=0; globalIndex < numberOfGlobalPoints; globalIndex++) {
77  if ((int)pointPartitioner.GetRankContainingIndex(globalIndex) == rank) {
78  numberOfLocalPoints++;
79  }
80  }
81 
82  /* Loop over our analytically defined of coordinates and store only
83  those that contain a node within our linear partition of points. */
84  localPointCoordinates.resize(dimension);
85  for (int k = 0; k < dimension; k++)
86  localPointCoordinates[k].reserve(numberOfLocalPoints);
87  unsigned long globalIndex = 0;
88  for (unsigned long jNode = 0; jNode < mNode; jNode++) {
89  for (unsigned long iNode = 0; iNode < nNode; iNode++) {
90  if ((int)pointPartitioner.GetRankContainingIndex(globalIndex) == rank) {
91 
92  /* Store the coordinates more clearly. */
93  const passivedouble x = SU2_TYPE::GetValue(Lx*((su2double)iNode)/((su2double)(nNode-1))+Ox);
94  const passivedouble y = SU2_TYPE::GetValue(Ly*((su2double)jNode)/((su2double)(mNode-1))+Oy);
95 
96  /* Load into the coordinate class data structure. */
97  localPointCoordinates[0].push_back(x);
98  localPointCoordinates[1].push_back(y);
99  }
100  globalIndex++;
101  }
102  }

106 void CRectangularMeshReaderFVM::ComputeRectangularVolumeConnectivity() {
107 
108  /* Set the global count of elements based on the grid dimensions. */
109  numberOfGlobalElements = (nNode-1)*(mNode-1);
110 
111  /* Get a partitioner to help with linear partitioning. */
112  CLinearPartitioner pointPartitioner(numberOfGlobalPoints,0);
113 
114  /* Loop over our analytically defined of elements and store only those
115  that contain a node within our linear partition of points. */
116  numberOfLocalElements = 0;
117  vector<unsigned long> connectivity(N_POINTS_HEXAHEDRON,0);
118  unsigned long globalIndex = 0;
119  for (unsigned long jNode = 0; jNode < mNode-1; jNode++) {
120  for (unsigned long iNode = 0; iNode < nNode-1; iNode++) {
121 
122  /* Compute connectivity based on the i,j index. */
123  connectivity[0] = jNode*nNode + iNode;
124  connectivity[1] = jNode*nNode + iNode + 1;
125  connectivity[2] = (jNode + 1)*nNode + (iNode + 1);
126  connectivity[3] = (jNode + 1)*nNode + iNode;
127 
128  /* Check whether any of the points is in our linear partition. */
129  bool isOwned = false;
130  for (unsigned short i = 0; i < N_POINTS_QUADRILATERAL; i++) {
131  if ((int)pointPartitioner.GetRankContainingIndex(connectivity[i]) == rank) {
132  isOwned = true;
133  }
134  }
135 
136  /* If so, we need to store the element locally. */
137  if (isOwned) {
138  localVolumeElementConnectivity.push_back(globalIndex);
139  localVolumeElementConnectivity.push_back(KindElem);
140  for (unsigned short i = 0; i < N_POINTS_HEXAHEDRON; i++) {
141  localVolumeElementConnectivity.push_back(connectivity[i]);
142  }
143  numberOfLocalElements++;
144  }
145  globalIndex++;
146  }
147  }

151 void CRectangularMeshReaderFVM::ComputeRectangularSurfaceConnectivity() {
152 
153  /* The rectangle alays has 4 markers. */
154  numberOfMarkers = 4;
155  surfaceElementConnectivity.resize(numberOfMarkers);
156  markerNames.resize(numberOfMarkers);
157 
158  /* Compute and store the 4 sets of connectivity. */
159  markerNames[0] = "y_minus";
160  if (rank == MASTER_NODE) {
161  for (unsigned long iNode = 0; iNode < nNode-1; iNode++) {
162  surfaceElementConnectivity[0].push_back(0);
163  surfaceElementConnectivity[0].push_back(KindBound);
164  surfaceElementConnectivity[0].push_back(iNode);
165  surfaceElementConnectivity[0].push_back(iNode + 1);
166  for (unsigned short i = 0; i < 6; i++)
167  surfaceElementConnectivity[0].push_back(0);
168  }
169  }
170 
171  markerNames[1] = "x_plus";
172  if (rank == MASTER_NODE) {
173  for (unsigned long jNode = 0; jNode < mNode-1; jNode++) {
174  surfaceElementConnectivity[1].push_back(0);
175  surfaceElementConnectivity[1].push_back(KindBound);
176  surfaceElementConnectivity[1].push_back(jNode*nNode + (nNode - 1));
177  surfaceElementConnectivity[1].push_back((jNode + 1)*nNode + (nNode - 1));
178  for (unsigned short i = 0; i < 6; i++)
179  surfaceElementConnectivity[1].push_back(0);
180  }
181  }
182 
183  markerNames[2] = "y_plus";
184  if (rank == MASTER_NODE) {
185  for (unsigned long iNode = 0; iNode < nNode-1; iNode++) {
186  surfaceElementConnectivity[2].push_back(0);
187  surfaceElementConnectivity[2].push_back(KindBound);
188  surfaceElementConnectivity[2].push_back((nNode*mNode - 1) - iNode);
189  surfaceElementConnectivity[2].push_back((nNode*mNode - 1) - (iNode + 1));
190  for (unsigned short i = 0; i < 6; i++)
191  surfaceElementConnectivity[2].push_back(0);
192  }
193  }
194 
195  markerNames[3] = "x_minus";
196  if (rank == MASTER_NODE) {
197  for (unsigned long jNode = 0; jNode < mNode-1; jNode++) {
198  surfaceElementConnectivity[3].push_back(0);
199  surfaceElementConnectivity[3].push_back(KindBound);
200  surfaceElementConnectivity[3].push_back((jNode + 1)*nNode);
201  surfaceElementConnectivity[3].push_back(jNode*nNode);
202  for (unsigned short i = 0; i < 6; i++)
203  surfaceElementConnectivity[3].push_back(0);
204  }
205  }
