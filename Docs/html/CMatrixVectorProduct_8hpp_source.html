37 /*!
38  * \class CMatrixVectorProduct
39  * \ingroup SpLinSys
40  * \brief Abstract base class for defining matrix-vector products
41  * \author J. Hicken.
42  *
43  * The Krylov-subspace solvers require only matrix-vector products and
44  * not the actual matrix/Jacobian. We need some way to indicate which
45  * function will perform the product. However, sometimes the
46  * functions that define the product will require different numbers
47  * and types of inputs. For example, the forward-difference
48  * approximation to a Jacobian-vector product requires the vector that
49  * defines the Jacobian and a perturbation parameter. The
50  * CMatrixVectorProduct class is used to derive child classes that can
51  * handle the different types of matrix-vector products and still be
52  * passed to a single implementation of the Krylov solvers.
53  * This abstraction may also be used to define matrix-free products.
54  */
55 template<class ScalarType>
56 class CMatrixVectorProduct {
57 public:
58  virtual ~CMatrixVectorProduct() = 0;
59  virtual void operator()(const CSysVector<ScalarType> & u, CSysVector<ScalarType> & v) const = 0;
60 };
61 template<class ScalarType>
62 CMatrixVectorProduct<ScalarType>::~CMatrixVectorProduct() {}
