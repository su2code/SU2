57 void CCubicSpline::SetSpline(const vector<su2double> &X, const vector<su2double> &Data) {
58 
59  C1DInterpolation::SetSpline(X,Data);
60 
61  const int N = x.size();
62 
63  /*--- Alias the vectors of coefficients to build the tridiagonal system. ---*/
64 
65  auto& lower = b; b.resize(N);
66  auto& main = c; c.resize(N);
67  auto& upper = d; d.resize(N);
68  vector<su2double> rhs(N);
69 
70  /*--- Main part of the tridiagonal system. ---*/
71 
72  for (int i=1; i<N-1; i++) {
73  lower[i] = x[i]-x[i-1];
74  main[i] = (x[i+1]-x[i-1])*2;
75  upper[i] = x[i+1]-x[i];
76  rhs[i] = 6*((y[i+1]-y[i])/upper[i] - (y[i]-y[i-1])/lower[i]);
77  }
78 
79  /*--- Start condition. ---*/
80 
81  if (startDer == SECOND) {
82  main[0] = 1.0;
83  upper[0] = 0;
84  rhs[0] = startVal;
85  }
86  else { // FIRST
87  main[0] = 2*lower[1];
88  upper[0] = lower[1];
89  rhs[0] = 6*((y[1]-y[0])/lower[1] - startVal);
90  }
91 
92  /*--- End condition. ---*/
93 
94  if (endDer == SECOND) {
95  main[N-1] = 1.0;
96  lower[N-1] = 0;
97  rhs[N-1] = endVal;
98  }
99  else { // FIRST
100  main[N-1] = 2*upper[N-2];
101  lower[N-1] = upper[N-2];
102  rhs[N-1] = 6*(endVal - (y[N-1]-y[N-2])/upper[N-2]);
103  }
104 
105  /*--- Solve system for 2nd derivative at the knots. ---*/
106 
107  CBlasStructure::tdma(lower, main, upper, rhs);
108  const auto& d2y = rhs;
109 
110  /*--- Compute the polynomial coefficients. ---*/
111 
112  for (int i=0; i<N-1; i++) {
113  su2double dx = x[i+1]-x[i];
114  c[i] = 0.5*d2y[i];
115  d[i] = (d2y[i+1]-d2y[i]) / (6*dx);
116  b[i] = (y[i+1]-y[i])/dx - (c[i]+d[i]*dx)*dx;
117  }
118 }

120 void CAkimaInterpolation::SetSpline(const vector<su2double> &X, const vector<su2double> &Data){
121 
122  C1DInterpolation::SetSpline(X,Data);
123 
124  const int n = X.size();
125  vector<su2double> h (n-1);
126  vector<su2double> p (n-1);
127 
128  /*---calculating finite differences (h) and gradients (p) ---*/
129  for (int i=0; i<n-1; i++){
130  h[i]=X[i+1]-X[i];
131  p[i]=(Data[i+1]-Data[i])/h[i];
132  }
133 
134  /*--- b,c,d are the akima spline's cofficient for the cubic equation ---*/
135  b.resize(n);
136  c.resize(n);
137  d.resize(n);
138 
139  b[0] = p[0];
140  b[1] = (p[0] + p[1])/2;
141  b[n-1] = p[n-2];
142  b[n-2] = (p[n-2]+p[n-3])/2;
143 
144  for (int i = 2; i < n-2; i++){
145  su2double w1=fabs(p[i+1]-p[i]) , w2=fabs(p[i-1]-p[i-2]);
146  if (w1+w2<0.0001) {
147  b[i] = (p[i-1]+p[i])/2 ;
148  }
149  else{
150  b[i] = (w1*p[i-1]+w2*p[i])/(w1+w2);
151  }
152  }
153 
154  for (int i = 0; i < n-1; i++){
155  c[i] = (3*p[i]-2*b[i]-b[i+1])/h[i] ;
156  d[i] = (b[i+1]+b[i]-2*p[i])/h[i]/h[i];
157  }
158  b.back() = c.back() = 0;
159 
160 }

162 vector<su2double> CorrectedInletValues(const vector<su2double> &Inlet_Interpolated ,
163  su2double Theta ,
164  unsigned short nDim,
165  const su2double *Coord,
166  unsigned short nVar_Turb,
167  INLET_INTERP_TYPE Interpolation_Type){
168 
169  unsigned short size_columns=Inlet_Interpolated.size()+nDim;
170  vector<su2double> Inlet_Values(size_columns);
171  su2double unit_r, unit_Theta, unit_m, Alpha, Phi;
172 
173  /*---For x,y,z,T,P columns---*/
174  for (int i=0;i<nDim;i++)
175  Inlet_Values[i] = Coord[i];
176 
177  for (int i=nDim;i<nDim+2;i++)
178  Inlet_Values[i] = Inlet_Interpolated[i-2];
179 
180  /*---For turbulence variables columns---*/
181  if (nVar_Turb == 1)
182  Inlet_Values[nDim+5] = Inlet_Interpolated[5];
183  else if (nVar_Turb == 2){
184  Inlet_Values[nDim+5] = Inlet_Interpolated[5];
185  Inlet_Values[nDim+6] = Inlet_Interpolated[6];
186  }
187 
188  /*--- Correct for Interpolation Type now ---*/
189  switch(Interpolation_Type){
190  case(INLET_INTERP_TYPE::VR_VTHETA):
191  unit_r = Inlet_Interpolated[nDim];
192  unit_Theta = Inlet_Interpolated[nDim+1];
193  break;
194 
195  case(INLET_INTERP_TYPE::ALPHA_PHI):
196  Alpha = Inlet_Interpolated[nDim]*PI_NUMBER/180;
197  Phi = Inlet_Interpolated[nDim+1]*PI_NUMBER/180;
198  unit_m = sqrt(1/(1+pow(tan(Alpha),2)));
199  unit_Theta = tan(Alpha)*unit_m;
200  unit_r=unit_m*sin(Phi);
201  break;
202 
203  default:
204  unit_Theta = unit_r = 0.0;
205  break;
206  }
207 
208  /*--- Converting from cylindrical to cartesian unit vectors ---*/
209  Inlet_Values[nDim+2] = unit_r*cos(Theta) - unit_Theta*sin(Theta);
210  Inlet_Values[nDim+3] = unit_r*sin(Theta) + unit_Theta*cos(Theta);
211  Inlet_Values[nDim+4] = sqrt(1-pow(unit_r,2)- pow(unit_Theta,2));
212 
213  return Inlet_Values;
214 }
