62 void CVolumetricMovement::UpdateGridCoord(CGeometry *geometry, CConfig *config) {
63 
64  unsigned short iDim;
65  unsigned long iPoint, total_index;
66  su2double new_coord;
67 
68  /*--- Update the grid coordinates using the solution of the linear system
69  after grid deformation (LinSysSol contains the x, y, z displacements). ---*/
70 
71  for (iPoint = 0; iPoint < nPoint; iPoint++)
72  for (iDim = 0; iDim < nDim; iDim++) {
73  total_index = iPoint*nDim + iDim;
74  new_coord = geometry->nodes->GetCoord(iPoint, iDim)+LinSysSol[total_index];
75  if (fabs(new_coord) < EPS*EPS) new_coord = 0.0;
76  geometry->nodes->SetCoord(iPoint, iDim, new_coord);
77  }
78 
79  /*--- LinSysSol contains the non-transformed displacements in the periodic halo cells.
80  * Hence we still need a communication of the transformed coordinates, otherwise periodicity
81  * is not maintained. ---*/
82 
83  geometry->InitiateComms(geometry, config, COORDINATES);
84  geometry->CompleteComms(geometry, config, COORDINATES);

117 void CVolumetricMovement::SetVolume_Deformation(CGeometry *geometry, CConfig *config, bool UpdateGeo, bool Derivative, bool ForwardProjectionDerivative) {
118 
119  unsigned long Tot_Iter = 0;
120  su2double MinVolume, MaxVolume;
121 
122  /*--- Retrieve number or iterations, tol, output, etc. from config ---*/
123 
124  auto Screen_Output = config->GetDeform_Output();
125  auto Nonlinear_Iter = config->GetGridDef_Nonlinear_Iter();
126 
127  /*--- Disable the screen output if we're running SU2_CFD ---*/
128 
129  if (config->GetKind_SU2() == SU2_COMPONENT::SU2_CFD && !Derivative) Screen_Output = false;
130  if (config->GetSmoothGradient()) Screen_Output=true;
131 
132  /*--- Set the number of nonlinear iterations to 1 if Derivative computation is enabled ---*/
133 
134  if (Derivative) Nonlinear_Iter = 1;
135 
136  /*--- Loop over the total number of grid deformation iterations. The surface
137  deformation can be divided into increments to help with stability. In
138  particular, the linear elasticity equations hold only for small deformations. ---*/
139 
140  for (auto iNonlinear_Iter = 0ul; iNonlinear_Iter < Nonlinear_Iter; iNonlinear_Iter++) {
141 
142  /*--- Initialize vector and sparse matrix ---*/
143 
144  LinSysSol.SetValZero();
145  LinSysRes.SetValZero();
146  StiffMatrix.SetValZero();
147 
148  /*--- Compute the stiffness matrix entries for all nodes/elements in the
149  mesh. FEA uses a finite element method discretization of the linear
150  elasticity equations (transfers element stiffnesses to point-to-point). ---*/
151 
152  MinVolume = SetFEAMethodContributions_Elem(geometry, config);
153 
154  /*--- Set the boundary and volume displacements (as prescribed by the
155  design variable perturbations controlling the surface shape)
156  as a Dirichlet BC. ---*/
157 
158  SetBoundaryDisplacements(geometry, config);
159 
160  /*--- Fix the location of any points in the domain, if requested. ---*/
161 
162  SetDomainDisplacements(geometry, config);
163 
164  /*--- Set the boundary derivatives (overrides the actual displacements) ---*/
165 
166  if (Derivative) { SetBoundaryDerivatives(geometry, config, ForwardProjectionDerivative); }
167 
168  /*--- Communicate any prescribed boundary displacements via MPI,
169  so that all nodes have the same solution and r.h.s. entries
170  across all partitions. ---*/
171 
172  CSysMatrixComms::Initiate(LinSysSol, geometry, config);
173  CSysMatrixComms::Complete(LinSysSol, geometry, config);
174 
175  CSysMatrixComms::Initiate(LinSysRes, geometry, config);
176  CSysMatrixComms::Complete(LinSysRes, geometry, config);
177 
178  /*--- Definition of the preconditioner matrix vector multiplication, and linear solver ---*/
179 
180  /*--- To keep legacy behavior ---*/
181  System.SetToleranceType(LinearToleranceType::RELATIVE);
182 
183  /*--- If we want no derivatives or the direct derivatives, we solve the system using the
184  * normal matrix vector product and preconditioner. For the mesh sensitivities using
185  * the discrete adjoint method we solve the system using the transposed matrix. ---*/
186  if (!Derivative || ((config->GetKind_SU2() == SU2_COMPONENT::SU2_CFD) && Derivative) || (config->GetSmoothGradient() && ForwardProjectionDerivative)) {
187 
188  Tot_Iter = System.Solve(StiffMatrix, LinSysRes, LinSysSol, geometry, config);
189 
190  } else if (Derivative && (config->GetKind_SU2() == SU2_COMPONENT::SU2_DOT)) {
191 
192  Tot_Iter = System.Solve_b(StiffMatrix, LinSysRes, LinSysSol, geometry, config);
193  }
194  su2double Residual = System.GetResidual();
195 
196  /*--- Update the grid coordinates and cell volumes using the solution
197  of the linear system (usol contains the x, y, z displacements). ---*/
198 
199  if (!Derivative) { UpdateGridCoord(geometry, config); }
200  else { UpdateGridCoord_Derivatives(geometry, config, ForwardProjectionDerivative); }
201  if (UpdateGeo) { UpdateDualGrid(geometry, config); }
202 
203  if (!Derivative) {
204  /*--- Check for failed deformation (negative volumes). ---*/
205 
206  ComputeDeforming_Element_Volume(geometry, MinVolume, MaxVolume, Screen_Output);
207 
208  /*--- Calculate amount of nonconvex elements ---*/
209 
210  ComputenNonconvexElements(geometry, Screen_Output);
211  }
212 
213  /*--- Set number of iterations in the mesh update. ---*/
214 
215  Set_nIterMesh(Tot_Iter);
216 
217  if (rank == MASTER_NODE && Screen_Output) {
218  cout << "Non-linear iter.: " << iNonlinear_Iter+1 << "/" << Nonlinear_Iter << ". Linear iter.: " << Tot_Iter << ". ";
219  if (nDim == 2) cout << "Min. area: " << MinVolume << ". Error: " << Residual << "." << endl;
220  else cout << "Min. volume: " << MinVolume << ". Error: " << Residual << "." << endl;
221  }
222 
223  }

227 void CVolumetricMovement::ComputeDeforming_Element_Volume(CGeometry *geometry, su2double &MinVolume, su2double &MaxVolume, bool Screen_Output) {
228 
229  unsigned long iElem, ElemCounter = 0, PointCorners[8];
230  su2double Volume = 0.0, CoordCorners[8][3];
231  unsigned short nNodes = 0, iNodes, iDim;
232  bool RightVol = true;
233 
234  if (rank == MASTER_NODE && Screen_Output)
235  cout << "Computing volumes of the grid elements." << endl;
236 
237  MaxVolume = -1E22; MinVolume = 1E22;
238 
239  /*--- Load up each triangle and tetrahedron to check for negative volumes. ---*/
240 
241  for (iElem = 0; iElem < geometry->GetnElem(); iElem++) {
242 
243  if (geometry->elem[iElem]->GetVTK_Type() == TRIANGLE) nNodes = 3;
244  if (geometry->elem[iElem]->GetVTK_Type() == QUADRILATERAL) nNodes = 4;
245  if (geometry->elem[iElem]->GetVTK_Type() == TETRAHEDRON) nNodes = 4;
246  if (geometry->elem[iElem]->GetVTK_Type() == PYRAMID) nNodes = 5;
247  if (geometry->elem[iElem]->GetVTK_Type() == PRISM) nNodes = 6;
248  if (geometry->elem[iElem]->GetVTK_Type() == HEXAHEDRON) nNodes = 8;
249 
250  for (iNodes = 0; iNodes < nNodes; iNodes++) {
251  PointCorners[iNodes] = geometry->elem[iElem]->GetNode(iNodes);
252  for (iDim = 0; iDim < nDim; iDim++) {
253  CoordCorners[iNodes][iDim] = geometry->nodes->GetCoord(PointCorners[iNodes], iDim);
254  }
255  }
256 
257  /*--- 2D elements ---*/
258 
259  if (nDim == 2) {
260  if (nNodes == 3) Volume = GetTriangle_Area(CoordCorners);
261  if (nNodes == 4) Volume = GetQuadrilateral_Area(CoordCorners);
262  }
263 
264  /*--- 3D Elementes ---*/
265 
266  if (nDim == 3) {
267  if (nNodes == 4) Volume = GetTetra_Volume(CoordCorners);
268  if (nNodes == 5) Volume = GetPyram_Volume(CoordCorners);
269  if (nNodes == 6) Volume = GetPrism_Volume(CoordCorners);
270  if (nNodes == 8) Volume = GetHexa_Volume(CoordCorners);
271  }
272 
273  RightVol = true;
274  if (Volume < 0.0) RightVol = false;
275 
276  MaxVolume = max(MaxVolume, Volume);
277  MinVolume = min(MinVolume, Volume);
278  geometry->elem[iElem]->SetVolume(Volume);
279 
280  if (!RightVol) ElemCounter++;
281 
282  }
283 
284 #ifdef HAVE_MPI
285  unsigned long ElemCounter_Local = ElemCounter; ElemCounter = 0;
286  su2double MaxVolume_Local = MaxVolume; MaxVolume = 0.0;
287  su2double MinVolume_Local = MinVolume; MinVolume = 0.0;
288  SU2_MPI::Allreduce(&ElemCounter_Local, &ElemCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
289  SU2_MPI::Allreduce(&MaxVolume_Local, &MaxVolume, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
290  SU2_MPI::Allreduce(&MinVolume_Local, &MinVolume, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
291 #endif
292 
293  /*--- Volume from 0 to 1 ---*/
294 
295  for (iElem = 0; iElem < geometry->GetnElem(); iElem++) {
296  Volume = geometry->elem[iElem]->GetVolume()/MaxVolume;
297  geometry->elem[iElem]->SetVolume(Volume);
298  }
299 
300  if ((ElemCounter != 0) && (rank == MASTER_NODE) && (Screen_Output))
301  cout <<"There are " << ElemCounter << " elements with negative volume.\n" << endl;
302

305 void CVolumetricMovement::ComputenNonconvexElements(CGeometry *geometry, bool Screen_Output) {
306  unsigned long iElem;
307  unsigned short iDim;
308  unsigned long nNonconvexElements = 0;
309 
310  /*--- Load up each tetrahedron to check for convex properties. ---*/
311  if (nDim == 2){
312  for (iElem = 0; iElem < geometry->GetnElem(); iElem++) {
313  su2double minCrossProduct = 1.e6, maxCrossProduct = -1.e6;
314 
315  const auto nNodes = geometry->elem[iElem]->GetnNodes();
316 
317  /*--- Get coordinates of corner points ---*/
318  unsigned short iNodes;
319  unsigned long PointCorners[8];
320  const su2double* CoordCorners[8];
321 
322  for (iNodes = 0; iNodes < nNodes; iNodes++) {
323  PointCorners[iNodes] = geometry->elem[iElem]->GetNode(iNodes);
324  CoordCorners[iNodes] = geometry->nodes->GetCoord(PointCorners[iNodes]);
325  }
326 
327  /*--- Determine whether element is convex ---*/
328  for (iNodes = 0; iNodes < nNodes; iNodes ++) {
329 
330  /*--- Calculate minimum and maximum angle between edge vectors adjacent to each node ---*/
331  su2double edgeVector_i[3], edgeVector_j[3];
332 
333  for (iDim = 0; iDim < nDim; iDim ++) {
334  if (iNodes == 0) {
335  edgeVector_i[iDim] = CoordCorners[nNodes-1][iDim] - CoordCorners[iNodes][iDim];
336  } else {
337  edgeVector_i[iDim] = CoordCorners[iNodes-1][iDim] - CoordCorners[iNodes][iDim];
338  }
339 
340  if (iNodes == nNodes-1) {
341  edgeVector_j[iDim] = CoordCorners[0][iDim] - CoordCorners[iNodes][iDim];
342  } else {
343  edgeVector_j[iDim] = CoordCorners[iNodes+1][iDim] - CoordCorners[iNodes][iDim];
344  }
345  }
346 
347  /*--- Calculate cross product of edge vectors ---*/
348  su2double crossProduct;
349  crossProduct = edgeVector_i[1]*edgeVector_j[0] - edgeVector_i[0]*edgeVector_j[1];
350 
351  if (crossProduct < minCrossProduct) minCrossProduct = crossProduct;
352  if (crossProduct > maxCrossProduct) maxCrossProduct = crossProduct;
353  }
354 
355  /*--- Element is nonconvex if cross product of at least one set of adjacent edges is negative ---*/
356  if (minCrossProduct < 0 && maxCrossProduct > 0){
357  nNonconvexElements++;
358  }
359  }
360  } else if (rank == MASTER_NODE) {
361  cout << "\nWARNING: Convexity is not checked for 3D elements (issue #1171).\n" << endl;
362  }
363 
364  unsigned long nNonconvexElements_Local = nNonconvexElements; nNonconvexElements = 0;
365  SU2_MPI::Allreduce(&nNonconvexElements_Local, &nNonconvexElements, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
366 
367  /*--- Set number of nonconvex elements in geometry ---*/
368  geometry->SetnNonconvexElements(nNonconvexElements);
369 }

373 void CVolumetricMovement::ComputeSolid_Wall_Distance(CGeometry *geometry, CConfig *config, su2double &MinDistance, su2double &MaxDistance) const {
374 
375  unsigned long nVertex_SolidWall, ii, jj, iVertex, iPoint, pointID;
376  unsigned short iMarker, iDim;
377  su2double dist, MaxDistance_Local, MinDistance_Local;
378  int rankID;
379 
380  /*--- Initialize min and max distance ---*/
381 
382  MaxDistance = -1E22; MinDistance = 1E22;
383 
384  /*--- Compute the total number of nodes on no-slip boundaries ---*/
385 
386  nVertex_SolidWall = 0;
387  for(iMarker=0; iMarker<config->GetnMarker_All(); ++iMarker) {
388  if(config->GetSolid_Wall(iMarker))
389  nVertex_SolidWall += geometry->GetnVertex(iMarker);
390  }
391 
392  /*--- Allocate the vectors to hold boundary node coordinates
393  and its local ID. ---*/
394 
395  vector<su2double> Coord_bound(nDim*nVertex_SolidWall);
396  vector<unsigned long> PointIDs(nVertex_SolidWall);
397 
398  /*--- Retrieve and store the coordinates of the no-slip boundary nodes
399  and their local point IDs. ---*/
400 
401  ii = 0; jj = 0;
402  for (iMarker=0; iMarker<config->GetnMarker_All(); ++iMarker) {
403  if (config->GetSolid_Wall(iMarker)) {
404  for (iVertex=0; iVertex<geometry->GetnVertex(iMarker); ++iVertex) {
405  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
406  PointIDs[jj++] = iPoint;
407  for (iDim=0; iDim<nDim; ++iDim)
408  Coord_bound[ii++] = geometry->nodes->GetCoord(iPoint, iDim);
409  }
410  }
411  }
412 
413  /*--- Build the ADT of the boundary nodes. ---*/
414 
415  CADTPointsOnlyClass WallADT(nDim, nVertex_SolidWall, Coord_bound.data(),
416  PointIDs.data(), true);
417 
418  /*--- Loop over all interior mesh nodes and compute the distances to each
419  of the no-slip boundary nodes. Store the minimum distance to the wall
420  for each interior mesh node. ---*/
421 
422  if( WallADT.IsEmpty() ) {
423 
424  /*--- No solid wall boundary nodes in the entire mesh.
425  Set the wall distance to zero for all nodes. ---*/
426 
427  for (iPoint=0; iPoint<geometry->GetnPoint(); ++iPoint)
428  geometry->nodes->SetWall_Distance(iPoint, 0.0);
429  }
430  else {
431 
432  /*--- Solid wall boundary nodes are present. Compute the wall
433  distance for all nodes. ---*/
434 
435  for(iPoint=0; iPoint<geometry->GetnPoint(); ++iPoint) {
436 
437  WallADT.DetermineNearestNode(geometry->nodes->GetCoord(iPoint), dist,
438  pointID, rankID);
439  geometry->nodes->SetWall_Distance(iPoint, dist);
440 
441  MaxDistance = max(MaxDistance, dist);
442 
443  /*--- To discard points on the surface we use > EPS ---*/
444 
445  if (sqrt(dist) > EPS) MinDistance = min(MinDistance, dist);
446 
447  }
448 
449  MaxDistance_Local = MaxDistance; MaxDistance = 0.0;
450  MinDistance_Local = MinDistance; MinDistance = 0.0;
451 
452 #ifdef HAVE_MPI
453  SU2_MPI::Allreduce(&MaxDistance_Local, &MaxDistance, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
454  SU2_MPI::Allreduce(&MinDistance_Local, &MinDistance, 1, MPI_DOUBLE, MPI_MIN, SU2_MPI::GetComm());
455 #else
456  MaxDistance = MaxDistance_Local;
457  MinDistance = MinDistance_Local;
458 #endif
459 
460  }

464 su2double CVolumetricMovement::SetFEAMethodContributions_Elem(CGeometry *geometry, CConfig *config) {
465 
466  unsigned short iVar, iDim, nNodes = 0, iNodes, StiffMatrix_nElem = 0;
467  unsigned long iElem, PointCorners[8];
468  su2double **StiffMatrix_Elem = nullptr, CoordCorners[8][3];
469  su2double MinVolume = 0.0, MaxVolume = 0.0, MinDistance = 0.0, MaxDistance = 0.0, ElemVolume = 0.0, ElemDistance = 0.0;
470 
471  bool Screen_Output = config->GetDeform_Output();
472 
473  /*--- Allocate maximum size (quadrilateral and hexahedron) ---*/
474 
475  if (nDim == 2) StiffMatrix_nElem = 8;
476  else StiffMatrix_nElem = 24;
477 
478  StiffMatrix_Elem = new su2double* [StiffMatrix_nElem];
479  for (iVar = 0; iVar < StiffMatrix_nElem; iVar++)
480  StiffMatrix_Elem[iVar] = new su2double [StiffMatrix_nElem];
481 
482  /*--- Compute min volume in the entire mesh. ---*/
483 
484  ComputeDeforming_Element_Volume(geometry, MinVolume, MaxVolume, Screen_Output);
485  if (rank == MASTER_NODE && Screen_Output) cout <<"Min. volume: "<< MinVolume <<", max. volume: "<< MaxVolume <<"." << endl;
486 
487  /*--- Compute the distance to the nearest surface if needed
488  as part of the stiffness calculation.. ---*/
489 
490  if ((config->GetDeform_Stiffness_Type() == SOLID_WALL_DISTANCE) ||
491  (config->GetDeform_Limit() < 1E6)) {
492  ComputeSolid_Wall_Distance(geometry, config, MinDistance, MaxDistance);
493  if (rank == MASTER_NODE && Screen_Output) cout <<"Min. distance: "<< MinDistance <<", max. distance: "<< MaxDistance <<"." << endl;
494  }
495 
496  /*--- Compute contributions from each element by forming the stiffness matrix (FEA) ---*/
497 
498  for (iElem = 0; iElem < geometry->GetnElem(); iElem++) {
499 
500  if (geometry->elem[iElem]->GetVTK_Type() == TRIANGLE) nNodes = 3;
501  if (geometry->elem[iElem]->GetVTK_Type() == QUADRILATERAL) nNodes = 4;
502  if (geometry->elem[iElem]->GetVTK_Type() == TETRAHEDRON) nNodes = 4;
503  if (geometry->elem[iElem]->GetVTK_Type() == PYRAMID) nNodes = 5;
504  if (geometry->elem[iElem]->GetVTK_Type() == PRISM) nNodes = 6;
505  if (geometry->elem[iElem]->GetVTK_Type() == HEXAHEDRON) nNodes = 8;
506 
507  for (iNodes = 0; iNodes < nNodes; iNodes++) {
508  PointCorners[iNodes] = geometry->elem[iElem]->GetNode(iNodes);
509  for (iDim = 0; iDim < nDim; iDim++) {
510  CoordCorners[iNodes][iDim] = geometry->nodes->GetCoord(PointCorners[iNodes], iDim);
511  }
512  }
513 
514  /*--- Extract Element volume and distance to compute the stiffness ---*/
515 
516  ElemVolume = geometry->elem[iElem]->GetVolume();
517 
518  if ((config->GetDeform_Stiffness_Type() == SOLID_WALL_DISTANCE)) {
519  ElemDistance = 0.0;
520  for (iNodes = 0; iNodes < nNodes; iNodes++)
521  ElemDistance += geometry->nodes->GetWall_Distance(PointCorners[iNodes]);
522  ElemDistance = ElemDistance/(su2double)nNodes;
523  }
524 
525  if (nDim == 2) SetFEA_StiffMatrix2D(geometry, config, StiffMatrix_Elem, PointCorners, CoordCorners, nNodes, ElemVolume, ElemDistance);
526  if (nDim == 3) SetFEA_StiffMatrix3D(geometry, config, StiffMatrix_Elem, PointCorners, CoordCorners, nNodes, ElemVolume, ElemDistance);
527 
528  AddFEA_StiffMatrix(geometry, StiffMatrix_Elem, PointCorners, nNodes);
529 
530  }
531 
532  /*--- Deallocate memory and exit ---*/
533 
534  for (iVar = 0; iVar < StiffMatrix_nElem; iVar++)
535  delete [] StiffMatrix_Elem[iVar];
536  delete [] StiffMatrix_Elem;
537 
538  return MinVolume;
539

542 su2double CVolumetricMovement::ShapeFunc_Triangle(su2double Xi, su2double Eta, su2double CoordCorners[8][3], su2double DShapeFunction[8][4]) {
543 
544  int i, j, k;
545  su2double c0, c1, xsj;
546  su2double xs[3][3], ad[3][3];
547 
548  /*--- Shape functions ---*/
549 
550  DShapeFunction[0][3] = Xi;
551  DShapeFunction[1][3] = Eta;
552  DShapeFunction[2][3] = 1-Xi-Eta;
553 
554  /*--- dN/d xi, dN/d eta ---*/
555 
556  DShapeFunction[0][0] = 1.0; DShapeFunction[0][1] = 0.0;
557  DShapeFunction[1][0] = 0.0; DShapeFunction[1][1] = 1.0;
558  DShapeFunction[2][0] = -1.0; DShapeFunction[2][1] = -1.0;
559 
560  /*--- Jacobian transformation ---*/
561 
562  for (i = 0; i < 2; i++) {
563  for (j = 0; j < 2; j++) {
564  xs[i][j] = 0.0;
565  for (k = 0; k < 3; k++) {
566  xs[i][j] = xs[i][j]+CoordCorners[k][j]*DShapeFunction[k][i];
567  }
568  }
569  }
570 
571  /*--- Adjoint to Jacobian ---*/
572
