257  /*!
258  * \brief Get the element container index and number of nodes of a given VTK type.
259  * \param[in] VTK_Type - Type of element.
260  * \param[out] EL_KIND - Element container index.
261  * \param[out] nNodes - Number of nodes.
262  */
263  static void GetElemKindAndNumNodes(unsigned short VTK_Type, int& EL_KIND, unsigned int& nNodes) {
264  /*--- if we need higher order quadrature rules overide some of the element kinds ---*/
265  switch (VTK_Type) {
266  case LINE: nNodes = 2; EL_KIND = EL_LINE; break;
267  case TRIANGLE: nNodes = 3; EL_KIND = EL_TRIA2; break;
268  case QUADRILATERAL: nNodes = 4; EL_KIND = EL_QUAD; break;
269  case TETRAHEDRON: nNodes = 4; EL_KIND = EL_TETRA2; break;
270  case PYRAMID: nNodes = 5; EL_KIND = EL_PYRAM2; break;
271  case PRISM: nNodes = 6; EL_KIND = EL_PRISM; break;
272  case HEXAHEDRON: nNodes = 8; EL_KIND = EL_HEXA; break;
273  default: assert(false); nNodes = 0; EL_KIND = -(1<<30); break;
274  }
275  }
