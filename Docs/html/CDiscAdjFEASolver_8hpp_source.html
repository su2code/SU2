45  /*!
46  * \brief A type to manage sensitivities of design variables.
47  */
48  struct SensData {
49  unsigned short size = 0;
50  su2double* val = nullptr; /*!< \brief Value of the variable. */
51  su2double* LocalSens = nullptr; /*!< \brief Local sensitivity (domain). */
52  su2double* GlobalSens = nullptr; /*!< \brief Global sensitivity (mpi). */
53  su2double* TotalSens = nullptr; /*!< \brief Total sensitivity (time domain). */
54 
55  su2double& operator[] (unsigned short i) { return val[i]; }
56  const su2double& operator[] (unsigned short i) const { return val[i]; }
57 
58  void resize(unsigned short n) {
59  clear();
60  size = n;
61  val = new su2double[n]();
62  LocalSens = new su2double[n]();
63  GlobalSens = new su2double[n]();
64  TotalSens = new su2double[n]();
65  }
66 
67  void clear() {
68  size = 0;
69  delete [] val;
70  delete [] LocalSens;
71  delete [] GlobalSens;
72  delete [] TotalSens;
73  }
74 
75  void Register() {
76  for (auto i = 0u; i < size; ++i) AD::RegisterInput(val[i]);
77  }
78 
79  void GetDerivative() {
80  for (auto i = 0u; i < size; ++i) LocalSens[i] = SU2_TYPE::GetDerivative(val[i]);
81 
82  SU2_MPI::Allreduce(LocalSens, GlobalSens, size, MPI_DOUBLE, MPI_SUM, SU2_MPI::GetComm());
83  }
84 
85  void UpdateTotal() {
86  for (auto i = 0u; i < size; ++i) TotalSens[i] += GlobalSens[i];
87  }
88 
89  ~SensData() { clear(); }
90  };
