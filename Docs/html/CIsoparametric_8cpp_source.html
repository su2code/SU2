49 void CIsoparametric::SetTransferCoeff(const CConfig* const* config) {
50 
51  const su2double matchingVertexTol = 1e-12; // 1um^2
52 
53  const int nProcessor = size;
54  const auto nMarkerInt = config[donorZone]->GetMarker_n_ZoneInterface()/2;
55  const auto nDim = donor_geometry->GetnDim();
56 
57  Buffer_Receive_nVertex_Donor = new unsigned long [nProcessor];
58 
59  /*--- Make space for donor info. ---*/
60 
61  targetVertices.resize(config[targetZone]->GetnMarker_All());
62 
63  /*--- Init stats. ---*/
64  MaxDistance = 0.0; ErrorCounter = 0;
65  unsigned long nGlobalVertexTarget = 0;
66 
67  /*--- Cycle over nMarkersInt interface to determine communication pattern. ---*/
68 
69  for (unsigned short iMarkerInt = 0; iMarkerInt < nMarkerInt; iMarkerInt++) {
70 
71  /* High level procedure:
72  * - Loop through vertices of the target grid;
73  * - Find nearest element;
74  * - Compute and set the transfer coefficients.
75  */
76 
77  /*--- On the donor side: find the tag of the boundary sharing the interface. ---*/
78  const auto markDonor = config[donorZone]->FindInterfaceMarker(iMarkerInt);
79 
80  /*--- On the target side: find the tag of the boundary sharing the interface. ---*/
81  const auto markTarget = config[targetZone]->FindInterfaceMarker(iMarkerInt);
82 
83  /*--- Checks if the zone contains the interface, if not continue to the next step. ---*/
84  if (!CheckInterfaceBoundary(markDonor, markTarget)) continue;
85 
86  unsigned long nVertexDonor = 0, nVertexTarget = 0;
87  if (markDonor != -1) nVertexDonor = donor_geometry->GetnVertex(markDonor);
88  if (markTarget != -1) nVertexTarget = target_geometry->GetnVertex(markTarget);
89 
90  /*--- Sets MaxLocalVertex_Donor, Buffer_Receive_nVertex_Donor. ---*/
91  Determine_ArraySize(markDonor, markTarget, nVertexDonor, nDim);
92 
93  const auto nGlobalVertexDonor = accumulate(Buffer_Receive_nVertex_Donor,
94  Buffer_Receive_nVertex_Donor+nProcessor, 0ul);
95 
96  Buffer_Send_Coord.resize(MaxLocalVertex_Donor, nDim);
97  Buffer_Send_GlobalPoint .resize(MaxLocalVertex_Donor);
98  Buffer_Receive_Coord.resize(nProcessor*MaxLocalVertex_Donor,nDim);
99  Buffer_Receive_GlobalPoint.resize(nProcessor * MaxLocalVertex_Donor);
100 
101  /*--- Collect coordinates and global point indices. ---*/
102  Collect_VertexInfo(markDonor, markTarget, nVertexDonor, nDim);
103  if (nVertexTarget) targetVertices[markTarget].resize(nVertexTarget);
104 
105  /*--- Compress the vertex information, and build a map of global point to "compressed
106  * index" to then reconstruct the donor elements in local index space. ---*/
107 
108  su2activematrix donorCoord(nGlobalVertexDonor, nDim);
109  vector<long> donorPoint(nGlobalVertexDonor);
110  vector<int> donorProc(nGlobalVertexDonor);
111  unordered_map<long, unsigned long> globalToLocalMap;
112 
113  auto iCount = 0ul;
114  for (int iProcessor = 0; iProcessor < nProcessor; ++iProcessor) {
115  auto offset = iProcessor * MaxLocalVertex_Donor;
116  for (auto iVertex = 0ul; iVertex < Buffer_Receive_nVertex_Donor[iProcessor]; ++iVertex) {
117  for (int iDim = 0; iDim < nDim; ++iDim)
118  donorCoord(iCount,iDim) = Buffer_Receive_Coord(offset+iVertex, iDim);
119  donorPoint[iCount] = Buffer_Receive_GlobalPoint[offset+iVertex];
120  donorProc[iCount] = iProcessor;
121  assert((globalToLocalMap.count(donorPoint[iCount]) == 0) && "Duplicate donor point found.");
122  globalToLocalMap[donorPoint[iCount]] = iCount;
123  ++iCount;
124  }
125  }
126  assert((iCount == nGlobalVertexDonor) && "Global donor point count mismatch.");
127 
128  /*--- Collect donor element (face) information. ---*/
129 
130  vector<unsigned long> allNumElem;
131  vector<unsigned short> elemNumNodes;
132  su2matrix<long> elemIdxNodes;
133 
134  const auto nGlobalElemDonor = Collect_ElementInfo(markDonor, nDim, true,
135  allNumElem, elemNumNodes, elemIdxNodes);
136 
137  /*--- Map the node to "local" indices and create a list of connected elements for each vertex. ---*/
138 
139  vector<vector<unsigned> > vertexElements(nGlobalVertexDonor);
140 
141  for (auto iElem = 0u; iElem < nGlobalElemDonor; ++iElem) {
142 
143  const auto nNode = elemNumNodes[iElem];
144 
145  for (auto iNode = 0u; iNode < nNode; ++iNode) {
146 
147  assert(globalToLocalMap.count(elemIdxNodes(iElem,iNode)) &&
148  "Unknown donor point referenced by donor element.");
149  const auto iVertex = globalToLocalMap.at(elemIdxNodes(iElem,iNode));
150  elemIdxNodes(iElem,iNode) = iVertex;
151 
152  vertexElements[iVertex].push_back(iElem);
153  }
154  }
155 
156  /*--- Compute transfer coefficients for each target point. ---*/
157  SU2_OMP_PARALLEL
158  {
159  su2double maxDist = 0.0;
160  unsigned long errorCount = 0, totalCount = 0;
161 
162  SU2_OMP_FOR_DYN(roundUpDiv(nVertexTarget,2*omp_get_max_threads()))
163  for (auto iVertexTarget = 0u; iVertexTarget < nVertexTarget; ++iVertexTarget) {
164 
165  auto& target_vertex = targetVertices[markTarget][iVertexTarget];
166  const auto iPoint = target_geometry->vertex[markTarget][iVertexTarget]->GetNode();
167 
168  if (!target_geometry->nodes->GetDomain(iPoint)) continue;
169  totalCount += 1;
170 
171  /*--- Coordinates of the target point. ---*/
172  const su2double* coord_i = target_geometry->nodes->GetCoord(iPoint);
173 
174  /*--- Find the closest donor vertex. ---*/
175  su2double minDist = 1e9;
176  unsigned iClosestVertex = 0;
177  for (auto iVertexDonor = 0u; iVertexDonor < nGlobalVertexDonor; ++iVertexDonor) {
178  su2double d = SquaredDistance(nDim, coord_i, donorCoord[iVertexDonor]);
179  if (d < minDist) {
180  minDist = d;
181  iClosestVertex = iVertexDonor;
182  }
183  }
184 
185  if (minDist < matchingVertexTol) {
186  /*--- Perfect match. ---*/
187  target_vertex.resize(1);
188  target_vertex.coefficient[0] = 1.0;
189  target_vertex.globalPoint[0] = donorPoint[iClosestVertex];
190  target_vertex.processor[0] = donorProc[iClosestVertex];
191  continue;
192  }
193 
194  /*--- Evaluate interpolation for the elements connected to the closest vertex. ---*/
195  DonorInfo donor;
196  donor.error = 2;
197  donor.distance = 1e9;
198  for (auto iElem : vertexElements[iClosestVertex]) {
199  /*--- Fetch element info. ---*/
200  DonorInfo candidate;
201  candidate.iElem = iElem;
202  const auto nNode = elemNumNodes[iElem];
203  su2double coords[4][3] = {{0.0}};
204 
205  for (auto iNode = 0u; iNode < nNode; ++iNode) {
206  const auto iVertex = elemIdxNodes(iElem, iNode);
207  for (auto iDim = 0u; iDim < nDim; ++iDim)
208  coords[iNode][iDim] = donorCoord(iVertex,iDim);
209  }
210 
211  /*--- Compute the interpolation coefficients. ---*/
212  switch (nNode) {
213  case 2: candidate.error = LineIsoparameters(coords, coord_i, candidate.isoparams); break;
214  case 3: candidate.error = TriangleIsoparameters(coords, coord_i, candidate.isoparams); break;
215  case 4: candidate.error = QuadrilateralIsoparameters(coords, coord_i, candidate.isoparams); break;
216  }
217 
218  /*--- Evaluate distance from target to final mapped point. ---*/
219  su2double finalCoord[3] = {0.0};
220  for (auto iDim = 0u; iDim < nDim; ++iDim)
221  for (auto iNode = 0u; iNode < nNode; ++iNode)
222  finalCoord[iDim] += coords[iNode][iDim] * candidate.isoparams[iNode];
223 
224  candidate.distance = Distance(nDim, coord_i, finalCoord);
225 
226  /*--- Detect a very bad candidate (NaN). ---*/
227  if (candidate.distance != candidate.distance) continue;
228 
229  /*--- Check if the candidate is an improvement, update donor if so. ---*/
230  if (candidate < donor) donor = candidate;
231  }
232 
233  if (donor.error > 1)
234  SU2_MPI::Error("Isoparametric interpolation failed, NaN detected.", CURRENT_FUNCTION);
235 
236  errorCount += donor.error;
237  maxDist = max(maxDist, donor.distance);
238 
239  const auto nNode = elemNumNodes[donor.iElem];
240 
241  target_vertex.resize(nNode);
242 
243  for (auto iNode = 0u; iNode < nNode; ++iNode) {
244  const auto iVertex = elemIdxNodes(donor.iElem, iNode);
245  target_vertex.coefficient[iNode] = donor.isoparams[iNode];
246  target_vertex.globalPoint[iNode] = donorPoint[iVertex];
247  target_vertex.processor[iNode] = donorProc[iVertex];
248  }
249 
250  }
251  END_SU2_OMP_FOR
252  SU2_OMP_CRITICAL
253  {
254  MaxDistance = max(MaxDistance, maxDist);
255  ErrorCounter += errorCount;
256  nGlobalVertexTarget += totalCount;
257  }
258  END_SU2_OMP_CRITICAL
259  }
260  END_SU2_OMP_PARALLEL
261 
262  } // end nMarkerInt loop
263 
264  /*--- Final reduction of statistics. ---*/
265  su2double tmp = MaxDistance;
266  unsigned long tmp1 = ErrorCounter, tmp2 = nGlobalVertexTarget;
267  SU2_MPI::Allreduce(&tmp, &MaxDistance, 1, MPI_DOUBLE, MPI_MAX, SU2_MPI::GetComm());
268  SU2_MPI::Allreduce(&tmp1, &ErrorCounter, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
269  SU2_MPI::Allreduce(&tmp2, &nGlobalVertexTarget, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
270 
271  ErrorRate = 100*su2double(ErrorCounter) / nGlobalVertexTarget;
272 
273 }

275 int CIsoparametric::LineIsoparameters(const su2double X[][3], const su2double *xj, su2double *isoparams) {
276 
277  /*--- Project the target point onto the line. ---*/
278 
279  su2double normal[2] = {0.0};
280  LineNormal(X, normal);
281  su2double xprj[2] = {0.0};
282  PointPlaneProjection<su2double,2>(xj, X[0], normal, xprj);
283 
284  su2double l01 = Distance(2, X[0], X[1]);
285  su2double l0j = Distance(2, X[0], xprj);
286  su2double lj1 = Distance(2, xprj, X[1]);
287 
288  /*--- Detect out of bounds point. ---*/
289 
290  const int outOfBounds = (l0j+lj1) > (2*l01);
291 
292  isoparams[0] = max(-0.5, min(lj1/l01, 1.5));
293  isoparams[1] = 1.0 - isoparams[0];
294 
295  return outOfBounds;
296 }

298 int CIsoparametric::TriangleIsoparameters(const su2double X[][3], const su2double *xj, su2double *isoparams) {
299 
300  /*--- The isoparameters are the solution to the determined system X^T * isoparams = xj.
301  * For which we solve the normal equations to avoid divisions by zero.
302  * This is consistent with the shape functions of the linear triangular element. ---*/
303 
304  const su2double extrapTol = -0.5;
305 
306  /*--- Project the target point onto the triangle. ---*/
307 
308  su2double normal[3] = {0.0}, xproj[3] = {0.0};
309  TriangleNormal(X, normal);
310  PointPlaneProjection<su2double,3>(xj, X[0], normal, xproj);
311 
312  su2double A[3][3] = {{0.0}}; // = X*X^T
313  for (int i = 0; i < 3; ++i) {
314 
315  for (int j = 0; j < 3; ++j)
316  for (int k = 0; k < 3; ++k)
317  A[i][j] += X[i][k] * X[j][k];
318 
319  isoparams[i] = 0.0; // use isoparams as rhs
320  for (int k = 0; k < 3; ++k)
321  isoparams[i] += X[i][k] * xproj[k];
322  }
323 
324  /*--- Solve system by in-place Gaussian elimination without pivoting. ---*/
325 
326  /*--- Transform system in Upper Matrix. ---*/
327  for (int i = 1; i < 3; ++i) {
328  for (int j = 0; j < i; ++j) {
329  su2double w = A[i][j] / A[j][j];
330  for (int k = j; k < 3; ++k)
331  A[i][k] -= w * A[j][k];
332  isoparams[i] -= w * isoparams[j];
333  }
334  }
335  /*--- Backwards substitution. ---*/
336  for (int i = 2; i >= 0; --i) {
337  for (int j = i+1; j < 3; ++j)
338  isoparams[i] -= A[i][j] * isoparams[j];
339  isoparams[i] /= A[i][i];
340  }
341 
342  /*--- Detect out of bounds point. ---*/
343  const int outOfBounds = (isoparams[0] < extrapTol) ||
344  (isoparams[1] < extrapTol) ||
345  (isoparams[2] < extrapTol);
346 
347  /*--- Mitigation. ---*/
348  if (outOfBounds) {
349  /*--- Clamp. ---*/
350  su2double sum = 0.0;
351  for (int i = 0; i < 3; ++i) {
352  isoparams[i] = max(isoparams[i], extrapTol);
353  sum += isoparams[i];
354  }
355  /*--- Enforce unit sum. ---*/
356  for (int i = 0; i < 3; ++i)
357  isoparams[i] /= sum;
358  }
359 
360  return outOfBounds;
361 }

363 int CIsoparametric::QuadrilateralIsoparameters(const su2double X[][3], const su2double *xj, su2double *isoparams) {
364 
365  /*--- The isoparameters are the shape functions (Ni) evaluated at xj, for that we need
366  * the corresponding Xi and Eta, which are obtained by solving the overdetermined
367  * nonlinear system r = xj - X^T * Ni(Xi,Eta) = 0 via the modified Marquardt method.
368  * It is not necessary to project the point as minimizing ||r|| is equivalent. ---*/
369 
370  const su2double extrapTol = 3.0;
371 
372  constexpr int NITER = 20;
373  const su2double tol = 1e-10, lambda = 0.05;
374 
375  su2double Xi = 0.0, Eta = 0.0, eps;
376 
377  /*--- Finding Xi and Eta is a "third order" effect that we do not
378  * differentiate (also because we need to iterate). ---*/
379 
380  const bool wasActive = AD::BeginPassive();
381 
382  for (int iter = 0; iter < NITER; ++iter) {
383 
384  /*--- Evaluate the residual. ---*/
385  su2double r[3] = {xj[0], xj[1], xj[2]};
386  su2double Ni[4] = {0.0};
387  CQUAD4::ShapeFunctions(Xi, Eta, Ni);
388  for (int i = 0; i < 3; ++i)
389  for (int j = 0; j < 4; ++j)
390  r[i] -= X[j][i] * Ni[j];
391 
392  /*--- Evaluate the residual Jacobian. ---*/
393  su2double dNi[4][2] = {{0.0}};
394  CQUAD4::ShapeFunctionJacobian(Xi, Eta, dNi);
395 
396  su2double jac[3][2] = {{0.0}};
397  for (int i = 0; i < 3; ++i)
398  for (int j = 0; j < 2; ++j)
399  for (int k = 0; k < 4; ++k)
400  jac[i][j] -= X[k][i] * dNi[k][j];
401 
402  /*--- Compute the correction (normal equations and Cramer's rule). ---*/
403  su2double A[2][2] = {{0.0}}, b[2] = {0.0};
404  for (int i = 0; i < 2; ++i) {
405  for (int j = i; j < 2; ++j)
406  for (int k = 0; k < 3; ++k)
407  A[i][j] += jac[k][i] * jac[k][j];
408 
409  A[i][i] *= (1.0+lambda);
410 
411  for (int k = 0; k < 3; ++k)
412  b[i] += jac[k][i] * r[k];
413  }
414  A[1][0] = A[0][1];
415 
416  su2double detA = 1.0 / (A[0][0]*A[1][1] - A[0][1]*A[1][0]);
417  su2double dXi = (b[0]*A[1][1] - b[1]*A[0][1]) * detA;
418  su2double dEta = (A[0][0]*b[1] - A[1][0]*b[0]) * detA;
419  Xi -= dXi;
420  Eta -= dEta;
421 
422  eps = fabs(dXi)+fabs(dEta);
423  if (eps < tol) break;
424  }
425 
426  AD::EndPassive(wasActive);
427 
428  int outOfBounds = 0;
429 
430  if (eps > 0.01) {
431  /*--- Iteration diverged, hard fallback. ---*/
432  Xi = Eta = 0.0;
433  outOfBounds = 1;
434  }
435  else {
436  /*--- Check bounds. ---*/
437  outOfBounds = (fabs(Xi) > extrapTol) || (fabs(Eta) > extrapTol);
438 
439  /*--- Mitigate by clamping coordinates. ---*/
440  if (outOfBounds) {
441  Xi = max(-extrapTol, min(Xi, extrapTol));
442  Eta = max(-extrapTol, min(Eta, extrapTol));
443  }
444  }
445 
446  /*--- Evaluate isoparameters at final Xi and Eta. ---*/
447  CQUAD4::ShapeFunctions(Xi, Eta, isoparams);
448 
449  return outOfBounds;
450 }
