97

99  Viscosity_Inf = config->GetViscosity_FreeStreamND();
100 
101  /*--- Factor_nu_Inf in [3.0, 5.0] ---*/

116  }
117

119  Factor_nu_ActDisk = config->GetNuFactor_Engine();
120  nu_tilde_ActDisk = Factor_nu_ActDisk*Viscosity_Inf/Density_Inf;

123  su2double Ji, Ji_3, fv1, cv1_3 = 7.1*7.1*7.1;

128  muT_Inf = Density_Inf*fv1*nu_tilde_Inf;

143  SlidingStateNodes.resize(nMarker);
144 
145  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++) {

147  SlidingState[iMarker].resize(nVertex[iMarker], nPrimVar+1) = nullptr;
148  SlidingStateNodes[iMarker].resize(nVertex[iMarker],0);

151

153  * due to arbitrary number of turbulence variables ---*/
154

156  for (unsigned long iMarker = 0; iMarker < nMarker; iMarker++)

208 void CTurbSASolver::Postprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh) {
209 
210  const su2double cv1_3 = 7.1*7.1*7.1, cR1 = 0.5, rough_const = 0.03;
211 
212  const bool neg_spalart_allmaras = config->GetSAParsedOptions().version == SA_OPTIONS::NEG;
213 
214  /*--- Compute eddy viscosity ---*/
215 
216  AD::StartNoSharedReading();
217 
218  SU2_OMP_FOR_STAT(omp_chunk_size)
219  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint ++) {
220 
221  const su2double rho = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
222  const su2double mu = solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
223 
224  const su2double nu = mu/rho;
225  const su2double nu_hat = nodes->GetSolution(iPoint,0);
226  const su2double roughness = geometry->nodes->GetRoughnessHeight(iPoint);
227  const su2double dist = geometry->nodes->GetWall_Distance(iPoint) + rough_const * roughness;
228 
229  su2double Ji = nu_hat/nu;
230  if (roughness > 1.0e-10)
231  Ji += cR1*roughness/(dist+EPS);
232 
233  const su2double Ji_3 = Ji*Ji*Ji;
234  const su2double fv1 = Ji_3/(Ji_3+cv1_3);
235 
236  su2double muT = rho*fv1*nu_hat;
237 
238  if (neg_spalart_allmaras) muT = max(muT,0.0);
239 
240  nodes->SetmuT(iPoint,muT);
241 
242  }
243  END_SU2_OMP_FOR
244 
245 
246  /*--- Compute turbulence index ---*/
247  if (config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE || config->GetSAParsedOptions().bc) {
248  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
249 
250  for (auto iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++){
251  if (config->GetViscous_Wall(iMarker)) {
252  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
253  for (auto iVertex = 0u; iVertex < geometry->nVertex[iMarker]; iVertex++) {
254  const auto iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
255 
256  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
257 
258  if (geometry->nodes->GetDomain(iPoint)) {
259  const auto jPoint = geometry->vertex[iMarker][iVertex]->GetNormal_Neighbor();
260 
261  su2double FrictionVelocity = 0.0;
262  /*--- Formulation varies for 2D and 3D problems: in 3D the friction velocity is assumed to be sqrt(mu * |Omega|)
263  (provided by the reference paper https://doi.org/10.2514/6.1992-439), whereas in 2D we have to use the
264  standard definition sqrt(c_f / rho) since Omega = 0. ---*/
265  if(nDim == 2){
266  su2double shearStress = 0.0;
267  for(auto iDim = 0u; iDim < nDim; iDim++) {
268  shearStress += pow(solver_container[FLOW_SOL]->GetCSkinFriction(iMarker, iVertex, iDim), 2.0);
269  }
270  shearStress = sqrt(shearStress);
271 
272  FrictionVelocity = sqrt(shearStress/flowNodes->GetDensity(iPoint));
273  } else {
274  su2double VorticityMag = max(GeometryToolbox::Norm(3, flowNodes->GetVorticity(iPoint)), 1e-12);
275  FrictionVelocity = sqrt(flowNodes->GetLaminarViscosity(iPoint)*VorticityMag);
276  }
277 
278  const su2double wall_dist = geometry->nodes->GetWall_Distance(jPoint);
279  const su2double Derivative = nodes->GetSolution(jPoint, 0) / wall_dist;
280  const su2double turbulence_index = Derivative / (FrictionVelocity * 0.41);
281 
282  nodes->SetTurbIndex(iPoint, turbulence_index);
283 
284  }
285  }
286  END_SU2_OMP_FOR
287  }
288  }
289  }
290 
291  AD::EndNoSharedReading();

308 void CTurbSASolver::Source_Residual(CGeometry *geometry, CSolver **solver_container,
309  CNumerics **numerics_container, CConfig *config, unsigned short iMesh) {
310 
311  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
312  const bool harmonic_balance = (config->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE);
313  const bool transition_BC = config->GetSAParsedOptions().bc;
314 
315  auto* flowNodes = su2staticcast_p<CFlowVariable*>(solver_container[FLOW_SOL]->GetNodes());
316 
317  /*--- Pick one numerics object per thread. ---*/
318  auto* numerics = numerics_container[SOURCE_FIRST_TERM + omp_get_thread_num()*MAX_TERMS];
319 
320  AD::StartNoSharedReading();
321 
322  /*--- Loop over all points. ---*/
323 
324  SU2_OMP_FOR_DYN(omp_chunk_size)
325  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
326 
327  /*--- Conservative variables w/o reconstruction ---*/
328 
329  numerics->SetPrimitive(flowNodes->GetPrimitive(iPoint), nullptr);
330 
331  /*--- Gradient of the primitive and conservative variables ---*/
332 
333  numerics->SetPrimVarGradient(flowNodes->GetGradient_Primitive(iPoint), nullptr);
334 
335  /*--- Set vorticity and strain rate magnitude ---*/
336 
337  numerics->SetVorticity(flowNodes->GetVorticity(iPoint), nullptr);
338 
339  numerics->SetStrainMag(flowNodes->GetStrainMag(iPoint), 0.0);
340 
341  /*--- Turbulent variables w/o reconstruction, and its gradient ---*/
342 
343  numerics->SetScalarVar(nodes->GetSolution(iPoint), nullptr);
344  numerics->SetScalarVarGradient(nodes->GetGradient(iPoint), nullptr);
345 
346  /*--- Set volume ---*/
347 
348  numerics->SetVolume(geometry->nodes->GetVolume(iPoint));
349 
350  /*--- Get Hybrid RANS/LES Type and set the appropriate wall distance ---*/
351 
352  if (config->GetKind_HybridRANSLES() == NO_HYBRIDRANSLES) {
353 
354  /*--- For the SA model, wall roughness is accounted by modifying the computed wall distance
355  * d_new = d + 0.03 k_s
356  * where k_s is the equivalent sand grain roughness height that is specified in cfg file.
357  * For smooth walls, wall roughness is zero and computed wall distance remains the same. */
358 
359  su2double modifiedWallDistance = geometry->nodes->GetWall_Distance(iPoint);
360 
361  modifiedWallDistance += 0.03*geometry->nodes->GetRoughnessHeight(iPoint);
362 
363  /*--- Set distance to the surface ---*/
364 
365  numerics->SetDistance(modifiedWallDistance, 0.0);
366 
367  /*--- Set the roughness of the closest wall. ---*/
368 
369  numerics->SetRoughness(geometry->nodes->GetRoughnessHeight(iPoint), 0.0 );
370 
371  } else {
372 
373  /*--- Set DES length scale ---*/
374 
375  numerics->SetDistance(nodes->GetDES_LengthScale(iPoint), 0.0);
376 
377  }
378 
379  /*--- Effective Intermittency ---*/
380 
381  if (config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE) {
382  numerics->SetIntermittencyEff(solver_container[TRANS_SOL]->GetNodes()->GetIntermittencyEff(iPoint));
383  numerics->SetIntermittency(solver_container[TRANS_SOL]->GetNodes()->GetSolution(iPoint, 0));
384  }
385 
386  /*--- Compute the source term ---*/
387 
388  auto residual = numerics->ComputeResidual(config);
389 
390  /*--- Store the intermittency ---*/
391 
392  if (transition_BC || config->GetKind_Trans_Model() != TURB_TRANS_MODEL::NONE) {
393  nodes->SetIntermittency(iPoint,numerics->GetIntermittencyEff());
394  }
395 
396  /*--- Subtract residual and the Jacobian ---*/
397 
398  LinSysRes.SubtractBlock(iPoint, residual);
399 
400  if (implicit) Jacobian.SubtractBlock2Diag(iPoint, residual.jacobian_i);
401 
402  }
403  END_SU2_OMP_FOR
404 
405  if (harmonic_balance) {
406 
407  SU2_OMP_FOR_STAT(omp_chunk_size)
408  for (unsigned long iPoint = 0; iPoint < nPointDomain; iPoint++) {
409 
410  su2double Volume = geometry->nodes->GetVolume(iPoint);
411 
412  /*--- Access stored harmonic balance source term ---*/
413 
414  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
415  su2double Source = nodes->GetHarmonicBalance_Source(iPoint,iVar);
416  LinSysRes(iPoint,iVar) += Source*Volume;
417  }
418  }
419  END_SU2_OMP_FOR
420  }
421 
422  AD::EndNoSharedReading();
423 
424 }

430 void CTurbSASolver::BC_HeatFlux_Wall(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
431  CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
432 
433  /*--- Evaluate nu tilde at the closest point to the surface using the wall functions. ---*/
434 
435  if (config->GetWall_Functions()) {
436  SU2_OMP_SAFE_GLOBAL_ACCESS(SetTurbVars_WF(geometry, solver_container, config, val_marker);)
437  return;
438  }
439 
440  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
441  bool rough_wall = false;
442  string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
443  WALL_TYPE WallType;
444  su2double Roughness_Height;
445  tie(WallType, Roughness_Height) = config->GetWallRoughnessProperties(Marker_Tag);
446  if (WallType == WALL_TYPE::ROUGH) rough_wall = true;
447 
448  /*--- The dirichlet condition is used only without wall function, otherwise the
449  convergence is compromised as we are providing nu tilde values for the
450  first point of the wall ---*/
451 
452  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
453  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
454 
455  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
456 
457  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
458 
459  if (geometry->nodes->GetDomain(iPoint)) {
460  if (!rough_wall) {
461  for (auto iVar = 0u; iVar < nVar; iVar++)
462  nodes->SetSolution_Old(iPoint,iVar,0.0);
463 
464  LinSysRes.SetBlock_Zero(iPoint);
465 
466  /*--- Includes 1 in the diagonal ---*/
467 
468  if (implicit) Jacobian.DeleteValsRowi(iPoint);
469  } else {
470  /*--- For rough walls, the boundary condition is given by
471  * (\frac{\partial \nu}{\partial n})_wall = \frac{\nu}{0.03*k_s}
472  * where \nu is the solution variable, $n$ is the wall normal direction
473  * and k_s is the equivalent sand grain roughness specified. ---*/
474 
475  /*--- Compute dual-grid area and boundary normal ---*/
476  su2double Normal[MAXNDIM] = {0.0};
477  for (auto iDim = 0u; iDim < nDim; iDim++)
478  Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
479 
480  su2double Area = GeometryToolbox::Norm(nDim, Normal);
481 
482  /*--- Get laminar_viscosity and density ---*/
483  su2double sigma = 2.0/3.0;
484  su2double laminar_viscosity = solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
485  su2double density = solver_container[FLOW_SOL]->GetNodes()->GetDensity(iPoint);
486 
487  su2double nu_total = (laminar_viscosity/density + nodes->GetSolution(iPoint,0));
488 
489  su2double coeff = (nu_total/sigma);
490  su2double RoughWallBC = nodes->GetSolution(iPoint,0)/(0.03*Roughness_Height);
491 
492  su2double Res_Wall;// = new su2double [nVar];
493  Res_Wall = coeff*RoughWallBC*Area;
494  LinSysRes.SubtractBlock(iPoint, &Res_Wall);
495 
496  su2double Jacobian_i = (laminar_viscosity*Area)/(0.03*Roughness_Height*sigma);
497  Jacobian_i += 2.0*RoughWallBC*Area/sigma;
498  if (implicit) Jacobian.AddVal2Diag(iPoint, -Jacobian_i);
499  }
500  }
501  }
502  END_SU2_OMP_FOR
503 }

512 void CTurbSASolver::BC_Inlet(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics,
513  CNumerics *visc_numerics, CConfig *config, unsigned short val_marker) {
514 
515  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
516 
517  /*--- Loop over all the vertices on this boundary marker ---*/
518 
519  SU2_OMP_FOR_STAT(OMP_MIN_SIZE)
520  for (auto iVertex = 0u; iVertex < geometry->nVertex[val_marker]; iVertex++) {
521 
522  const auto iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
523 
524  /*--- Check if the node belongs to the domain (i.e., not a halo node) ---*/
525 
526  if (geometry->nodes->GetDomain(iPoint)) {
527 
528  /*--- Normal vector for this vertex (negate for outward convention) ---*/
529 
530  su2double Normal[MAXNDIM] = {0.0};
531  for (auto iDim = 0u; iDim < nDim; iDim++)
532  Normal[iDim] = -geometry->vertex[val_marker][iVertex]->GetNormal(iDim);
533 
534  /*--- Allocate the value at the inlet ---*/
535 
536  auto V_inlet = solver_container[FLOW_SOL]->GetCharacPrimVar(val_marker, iVertex);
537 
538  /*--- Retrieve solution at the farfield boundary node ---*/
539 
540  auto V_domain = solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint);
541 
542  /*--- Set various quantities in the solver class ---*/
543 
544  conv_numerics->SetPrimitive(V_domain, V_inlet);
545 
546  /*--- Non-dimensionalize Inlet_TurbVars if Inlet-Files are used. ---*/
547  su2double Inlet_Vars[MAXNVAR];
548  Inlet_Vars[0] = Inlet_TurbVars[val_marker][iVertex][0];
549  if (config->GetInlet_Profile_From_File()) {
550  Inlet_Vars[0] *= config->GetDensity_Ref() / config->GetViscosity_Ref();
551  }
552 
553  /*--- Load the inlet turbulence variable (uniform by default). ---*/
554 
555  conv_numerics->SetScalarVar(nodes->GetSolution(iPoint), Inlet_Vars);
556 
557  /*--- Set various other quantities in the conv_numerics class ---*/
558 
559  conv_numerics->SetNormal(Normal);
560 
561  if (dynamic_grid)
562  conv_numerics->SetGridVel(geometry->nodes->GetGridVel(iPoint),
563  geometry->nodes->GetGridVel(iPoint));
564 
565  if (conv_numerics->GetBoundedScalar()) {
