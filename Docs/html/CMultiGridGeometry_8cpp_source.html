33 CMultiGridGeometry::CMultiGridGeometry(CGeometry *fine_grid, CConfig *config, unsigned short iMesh) : CGeometry() {
34 
35  nDim = fine_grid->GetnDim(); // Write the number of dimensions of the coarse grid.

37  /*--- Create a queue system to do the agglomeration
38  1st) More than two markers ---> Vertices (never agglomerate)
39  2nd) Two markers ---> Edges (agglomerate if same BC, never agglomerate if different BC)
40  3rd) One marker ---> Surface (always agglomarate)
41  4th) No marker ---> Internal Volume (always agglomarate) ---*/

44  i.e. consider up to neighbors of neighbors of neighbors.

222

224

242

244  iteration++;

247  and satisfies several geometrical criteria then the seed CV is accepted for agglomeration. ---*/
248

256

264  increase the priority of its neighbors) ---*/
265

269

271 
272  /*--- Determine if the CVPoint can be agglomerated ---*/
273

275  (fine_grid->nodes->GetDomain(CVPoint)) &&

277

279

283

285  nChildren++;
286 
287  /*--- Update the queue with the new control volume (remove the CV and
288  increase the priority of its neighbors) ---*/
289 
290  MGQueue_InnerCV.Update(CVPoint, fine_grid);

292  }
293 
294  }

297 
298  Suitable_Indirect_Neighbors.clear();
299  if (fine_grid->nodes->GetAgglomerate_Indirect(iPoint))
300  SetSuitableNeighbors(Suitable_Indirect_Neighbors, iPoint, Index_CoarseCV, fine_grid);
301 
302  /*--- Now we do a sweep over all the indirect nodes that can be added ---*/
303 
304  for (auto CVPoint : Suitable_Indirect_Neighbors) {
305 
306  /*--- The new point can be agglomerated ---*/
307 
308  if ((fine_grid->nodes->GetAgglomerate(CVPoint) == false) &&
309  (fine_grid->nodes->GetDomain(CVPoint))) {
310 
311  /*--- We set the value of the parent ---*/
312 
313  fine_grid->nodes->SetParent_CV(CVPoint, Index_CoarseCV);
314 
315  /*--- We set the indirect agglomeration information ---*/
316 
317  if (fine_grid->nodes->GetAgglomerate_Indirect(CVPoint))
318  nodes->SetAgglomerate_Indirect(Index_CoarseCV, true);
319 
320  /*--- We set the value of the child ---*/
321 
322  nodes->SetChildren_CV(Index_CoarseCV, nChildren, CVPoint);
323  nChildren++;
324 
325  /*--- Update the queue with the new control volume (remove the CV and
326  increase the priority of the neighbors) ---*/
327 
328  MGQueue_InnerCV.Update(CVPoint, fine_grid);
329 
330  }
331  }
332 
333  /*--- Update the number of control of childrens ---*/
334 
335  nodes->SetnChildren_CV(Index_CoarseCV, nChildren);
336  Index_CoarseCV++;
337  }
338  else {
339 
340  /*--- The seed point can not be agglomerated because of size, domain, streching, etc.
341  move the point to the lowest priority ---*/
342 
343  MGQueue_InnerCV.MoveCV(iPoint, -1);
344  }
345 
346  }

349 
350  for (auto iPoint = 0ul; iPoint < fine_grid->GetnPoint(); iPoint++) {
351  if ((fine_grid->nodes->GetAgglomerate(iPoint) == false) &&
352  (fine_grid->nodes->GetDomain(iPoint))) {
353 
354  fine_grid->nodes->SetParent_CV(iPoint, Index_CoarseCV);
355  if (fine_grid->nodes->GetAgglomerate_Indirect(iPoint))
356  nodes->SetAgglomerate_Indirect(Index_CoarseCV, true);
357  nodes->SetChildren_CV(Index_CoarseCV, 0, iPoint);
358  nodes->SetnChildren_CV(Index_CoarseCV, 1);
359  Index_CoarseCV++;
360  }
361  }

367  (only 1 neighbor), and merge their children CV's with the neighbor. ---*/
368 
369  SetPoint_Connectivity(fine_grid);

372

374

380  Identify the child CV from the finest grid and add it to the correct control volume.

382  (iCoarsePoint), use the new one (iCoarsePoint_Complete) ---*/

388  nodes->SetChildren_CV(iCoarsePoint_Complete, nChildren, iFinePoint);
389  nChildren++;

391  }
392

394

496 
497  }

499  }

546  }
547  }

553 bool CMultiGridGeometry::SetBoundAgglomeration(unsigned long CVPoint, short marker_seed,
554  const CGeometry *fine_grid, const CConfig *config) const {
555  bool agglomerate_CV = false;
556 
557  /*--- Basic condition, the point has not being previously agglomerated, it belongs to the domain,
558  and has passed some basic geometrical checks. ---*/
559 
560  if ((fine_grid->nodes->GetAgglomerate(CVPoint) == false) &&
561  (fine_grid->nodes->GetDomain(CVPoint)) &&
562  (GeometricalCheck(CVPoint, fine_grid, config))) {
563 
564  /*--- If the point belongs to a boundary, its type must be compatible with the seed marker. ---*/
565 
566  if (fine_grid->nodes->GetBoundary(CVPoint)) {
567 
568  /*--- Identify the markers of the vertex that we want to agglomerate ---*/
569 
570  int counter = 0;
571  unsigned short copy_marker[3] = {};
572  for (auto jMarker = 0u; jMarker < fine_grid->GetnMarker() && counter < 3; jMarker++) {
573  if (fine_grid->nodes->GetVertex(CVPoint, jMarker) != -1) {
574  copy_marker[counter] = jMarker;
575  counter++;
576  }
577  }
578 
579  /*--- The basic condition is that the aglomerated vertex must have the same physical marker,
580  but eventually a send-receive condition ---*/
581 
582  /*--- Only one marker in the vertex that is going to be aglomerated ---*/
583 
584  if (counter == 1) {
585 
586  /*--- We agglomerate if there is only a marker and is the same marker as the seed marker ---*/
587 
588  if (copy_marker[0] == marker_seed)
589  agglomerate_CV = true;
590 
591  /*--- If there is only one marker, but the marker is the SEND_RECEIVE ---*/
592 
593  if (config->GetMarker_All_KindBC(copy_marker[0]) == SEND_RECEIVE)
594  agglomerate_CV = true;
595 
596  }
597 
598  /*--- If there are two markers in the vertex that is going to be aglomerated ---*/
599 
600  if (counter == 2) {
601 
602  /*--- First we verify that the seed is a physical boundary ---*/
603 
604  if (config->GetMarker_All_KindBC(marker_seed) != SEND_RECEIVE) {
605 
606  /*--- Then we check that one of the marker is equal to the seed marker, and the other is send/receive ---*/
607 
608  if (((copy_marker[0] == marker_seed) && (config->GetMarker_All_KindBC(copy_marker[1]) == SEND_RECEIVE)) ||
609  ((config->GetMarker_All_KindBC(copy_marker[0]) == SEND_RECEIVE) && (copy_marker[1] == marker_seed)))
610  agglomerate_CV = true;
611  }
612 
613  }
614
