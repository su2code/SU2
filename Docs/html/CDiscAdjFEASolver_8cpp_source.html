30 CDiscAdjFEASolver::CDiscAdjFEASolver(CGeometry *geometry, CConfig *config, CSolver *direct_solver,
31  unsigned short Kind_Solver, unsigned short iMesh) : CSolver() {
32 
33  adjoint = true;
34 
35  unsigned long iPoint;
36 
37  const bool dynamic = (config->GetTime_Domain());
38 
39  nVar = dynamic? 3*direct_solver->GetnVar() : direct_solver->GetnVar();
40  nDim = geometry->GetnDim();
41 
42  /*-- Store some information about direct solver ---*/
43  this->KindDirect_Solver = Kind_Solver;
44  this->direct_solver = direct_solver;
45 
46  nMarker = config->GetnMarker_All();
47  nPoint = geometry->GetnPoint();
48  nPointDomain = geometry->GetnPointDomain();
49 
50  /*--- Define some auxiliary vectors related to the residual ---*/
51 
52  Residual_RMS.resize(nVar,1.0);
53  Residual_Max.resize(nVar,1.0);
54  Point_Max.resize(nVar,0);
55  Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
56 
57  /*--- Define some auxiliary vectors related to the residual for problems with a BGS strategy---*/
58 
59  if (config->GetMultizone_Residual()) {
60 
61  Residual_BGS.resize(nVar,1.0);
62  Residual_Max_BGS.resize(nVar,1.0);
63  Point_Max_BGS.resize(nVar,0);
64  Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
65  }
66 
67  /*--- Initialize the adjoint solution. ---*/
68 
69  vector<su2double> init(nVar,1e-16);
70  nodes = new CDiscAdjFEABoundVariable(init.data(), nPoint, nDim, nVar, config);
71  SetBaseClassPointerToNodes();
72 
73  /*--- Set which points are vertices and allocate boundary data. ---*/
74 
75  for (iPoint = 0; iPoint < nPoint; iPoint++)
76  for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
77  long iVertex = geometry->nodes->GetVertex(iPoint, iMarker);
78  if (iVertex >= 0) {
79  nodes->Set_isVertex(iPoint,true);
80  break;
81  }
82  }
83  nodes->AllocateBoundaryVariables(config);
84 
85  /*--- Initialize vector structures for multiple material definition ---*/
86 
87  nMPROP = config->GetnElasticityMod();
88 
89  /*--- For a material to be fully defined, we need to have the same number for all three parameters ---*/
90  bool checkDef = ((config->GetnElasticityMod() == config->GetnPoissonRatio()) &&
91  (config->GetnElasticityMod() == config->GetnMaterialDensity()) &&
92  (config->GetnMaterialDensity() == config->GetnPoissonRatio()));
93 
94  if (!checkDef){
95  SU2_MPI::Error("WARNING: For a material to be fully defined, E, Nu and Rho need to have the same dimensions.", CURRENT_FUNCTION);
96  }
97 
98  E.resize(nMPROP);
99  Nu.resize(nMPROP);
100  Rho.resize(nMPROP);
101  Rho_DL.resize(nMPROP);
102 
103  /*--- Initialize vector structures for multiple electric regions ---*/
104 
105  de_effects = config->GetDE_Effects();
106 
107  if (de_effects) {
108  nEField = config->GetnElectric_Field();
109  EField.resize(nEField);
110  }
111 
112  /*--- Initialize vector structures for structural-based design variables ---*/
113 
114  switch (config->GetDV_FEA()) {
115  case YOUNG_MODULUS:
116  case POISSON_RATIO:
117  case DENSITY_VAL:
118  case DEAD_WEIGHT:
119  case ELECTRIC_FIELD:
120  fea_dv = true;
121  break;
122  default:
123  fea_dv = false;
124  break;
125  }
126 
127  if (fea_dv) ReadDV(config);
128 
129 }

181 void CDiscAdjFEASolver::RegisterVariables(CGeometry *geometry, CConfig *config, bool reset){
182 
183  /*--- Register element-based values as input ---*/
184 
185  unsigned short iVar;
186 
187  if (KindDirect_Solver == RUNTIME_FEA_SYS) {
188 
189  const bool pseudo_static = config->GetPseudoStatic();
190 
191  for (iVar = 0; iVar < nMPROP; iVar++) {
192  E[iVar] = config->GetElasticyMod(iVar);
193  Nu[iVar] = config->GetPoissonRatio(iVar);
194  Rho[iVar] = pseudo_static? 0.0 : config->GetMaterialDensity(iVar);
195  Rho_DL[iVar] = config->GetMaterialDensity(iVar);
196  }
197 
198  /*--- Read the values of the electric field ---*/
199  if (de_effects) {
200  for (iVar = 0; iVar < nEField; iVar++)
201  EField[iVar] = config->Get_Electric_Field_Mod(iVar);
202  }
203 
204  /*--- Reset index, otherwise messes up other derivatives ---*/
205  if (fea_dv) {
206  for (iVar = 0; iVar < nDV; iVar++) AD::ResetInput(DV[iVar]);
207  }
208 
209  if (!reset) {
210  E.Register();
211  Nu.Register();
212  Rho.Register();
213  Rho_DL.Register();
214  if (de_effects) EField.Register();
215  if (fea_dv) DV.Register();
216 
217  /*--- Register the flow tractions ---*/
218  if (config->GetnMarker_Fluid_Load() > 0)
219  direct_solver->GetNodes()->RegisterFlowTraction();
220  }
221 
222  }
223 
224  /*--- Here it is possible to register other variables as input that influence the flow solution
225  * and thereby also the objective function. The adjoint values (i.e. the derivatives) can be
226  * extracted in the ExtractAdjointVariables routine. ---*/
227 }

239 void CDiscAdjFEASolver::ExtractAdjoint_Solution(CGeometry *geometry, CConfig *config, bool CrossTerm) {
240 
241  /*--- Set the old solution, for multi-zone problems this is done after computing the
242  * residuals, otherwise the per-zone-residuals do not make sense, as on entry Solution
243  * contains contributions from other zones but on extraction it does not. ---*/
244 
245  if (!config->GetMultizone_Problem()) nodes->Set_OldSolution();
246 
247  /*--- Extract and store the adjoint solution ---*/
248 
249  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
250  su2double Solution[MAXNVAR] = {0.0};
251  direct_solver->GetNodes()->GetAdjointSolution(iPoint,Solution);
252  nodes->SetSolution(iPoint,Solution);
253  }
254 
255  if (CrossTerm) return;
256 
257  /*--- Extract and store the adjoint solution at time n (including accel. and velocity) ---*/
258 
259  if (config->GetTime_Domain()) {
260  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
261  su2double Solution[MAXNVAR] = {0.0};
262  direct_solver->GetNodes()->GetAdjointSolution_time_n(iPoint,Solution);
263  nodes->Set_Solution_time_n(iPoint,Solution);
264  }
265  }
266 
267  /*--- Set the residuals ---*/
268 
269  SetResToZero();
270 
271  for (auto iPoint = 0ul; iPoint < nPointDomain; iPoint++) {
272  for (auto iVar = 0u; iVar < nVar; iVar++){
273  su2double residual = nodes->GetSolution(iPoint, iVar) - nodes->GetSolution_Old(iPoint, iVar);
274 
275  Residual_RMS[iVar] += residual*residual;
276  AddRes_Max(iVar,fabs(residual),geometry->nodes->GetGlobalIndex(iPoint),geometry->nodes->GetCoord(iPoint));
277  }
278  }
279 
280  SetResidual_RMS(geometry, config);
281 
282  SetIterLinSolver(direct_solver->System.GetIterations());
283  SetResLinSolver(direct_solver->System.GetResidual());
284 
285 }

313 void CDiscAdjFEASolver::SetAdjoint_Output(CGeometry *geometry, CConfig *config){
314 
315  const bool dynamic = config->GetTime_Domain();
316  const bool deform_mesh = (config->GetnMarker_Deform_Mesh() > 0);
317  const bool multizone = config->GetMultizone_Problem();
318 
319  su2double Solution[MAXNVAR] = {0.0};
320 
321  unsigned short iVar;
322 
323  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++){
324  for (iVar = 0; iVar < nVar; iVar++)
325  Solution[iVar] = nodes->GetSolution(iPoint,iVar);
326 
327  if (dynamic && !multizone) {
328  for (iVar = 0; iVar < nVar; iVar++)
329  Solution[iVar] += nodes->GetDual_Time_Derivative(iPoint,iVar);
330  }
331 
332  if (deform_mesh) {
333  for (iVar = 0; iVar < nDim; iVar++)
334  Solution[iVar] += nodes->GetSourceTerm_DispAdjoint(iPoint,iVar);
335 
336  if (dynamic) {
337  for (iVar = 0; iVar < nDim; iVar++)
338  Solution[nDim+iVar] += nodes->GetSourceTerm_VelAdjoint(iPoint,iVar);
339  }
340  }
341 
342  direct_solver->GetNodes()->SetAdjointSolution(iPoint,Solution);

355 void CDiscAdjFEASolver::SetSensitivity(CGeometry *geometry, CConfig *config, CSolver*){
356 
357  E.UpdateTotal();
358  Nu.UpdateTotal();
359  Rho.UpdateTotal();
360  Rho_DL.UpdateTotal();
361  if (de_effects) EField.UpdateTotal();
362  if (fea_dv) DV.UpdateTotal();
363 
364  /*--- Extract the topology optimization density sensitivities. ---*/
365 
366  direct_solver->ExtractAdjoint_Variables(geometry, config);
367 
368  /*--- Extract the geometric sensitivities ---*/
369 
370  const bool time_domain = config->GetTime_Domain();
371 
372  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++) {
373 
374  auto Coord = geometry->nodes->GetCoord(iPoint);
375 
376  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
377 
378  su2double Sensitivity = geometry->nodes->GetAdjointSolution(iPoint, iDim);
379  AD::ResetInput(Coord[iDim]);
380 
381  if (!time_domain) {
382  nodes->SetSensitivity(iPoint, iDim, Sensitivity);
383  } else {
384  nodes->SetSensitivity(iPoint, iDim, nodes->GetSensitivity(iPoint, iDim) + Sensitivity);
385  }
386  }

391 void CDiscAdjFEASolver::ReadDV(const CConfig *config) {
392 
393  string filename;
394  ifstream properties_file;
395 
396  /*--- Choose the filename of the design variable ---*/
397 
398  switch (config->GetDV_FEA()) {
399  case YOUNG_MODULUS:
400  filename = "dv_young.opt";
401  break;
402  case POISSON_RATIO:
403  filename = "dv_poisson.opt";
404  break;
405  case DENSITY_VAL:
406  case DEAD_WEIGHT:
407  filename = "dv_density.opt";
408  break;
409  case ELECTRIC_FIELD:
410  filename = "dv_efield.opt";
411  break;
412  default:
413  filename = "dv.opt";
414  break;
415  }
416 
417  if (rank == MASTER_NODE) cout << "Filename: " << filename << "." << endl;
418 
419  properties_file.open(filename.data(), ios::in);
420 
421  /*--- In case there is no file, all elements get the same property (0) ---*/
422 
423  if (properties_file.fail()) {
424 
425  if (rank == MASTER_NODE)
426  cout << "There is no design variable file." << endl;
427 
428  nDV = 1;
429  DV.resize(nDV);
430  DV[0] = 1.0;
431  }
432  else{
433 
434  string text_line;
435  vector<su2double> values;
436 
437  /*--- Skip the first line: it is the header ---*/
438  getline (properties_file, text_line);
439 
440  while (getline (properties_file, text_line)) {
441  istringstream point_line(text_line);
442 
443  unsigned long index;
444  su2double value;
445  point_line >> index >> value;
446 
447  values.push_back(value);
448  }
449 
450  nDV = values.size();
451  DV.resize(nDV);
452  unsigned short iDV = 0;
453  for (auto x : values) DV[iDV++] = x;
454 
455  }
456 
457 }
