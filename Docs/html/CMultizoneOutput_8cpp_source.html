30 CMultizoneOutput::CMultizoneOutput(const CConfig* driver_config, const CConfig* const* config, unsigned short nDim) :
31  COutput(driver_config, nDim, false) {
32 
33  unsigned short iZone = 0;
34 
35  nZone = driver_config->GetnZone();
36 
37  bgs_res_name = "BGS_RES";
38 
39  write_zone = false;
40 
41  /*--- If the zone output is disabled for every zone ---*/
42  for (iZone = 0; iZone < nZone; iZone++){
43  write_zone = config[iZone]->GetWrt_ZoneConv();
44  }
45 
46  if (nRequestedHistoryFields == 0){
47  requestedHistoryFields.emplace_back("ITER");
48  for (iZone = 0; iZone < nZone; iZone++){
49  requestedHistoryFields.emplace_back(bgs_res_name + "[" + PrintingToolbox::to_string(iZone) + "]");
50  requestedHistoryFields.emplace_back("AVG_RES[" + PrintingToolbox::to_string(iZone) + "]");
51  }
52  nRequestedHistoryFields = requestedHistoryFields.size();
53  }
54 
55  if (nRequestedScreenFields == 0){
56  if (config[ZONE_0]->GetTime_Domain()) requestedScreenFields.emplace_back("TIME_ITER");
57  requestedScreenFields.emplace_back("OUTER_ITER");
58  for (iZone = 0; iZone < nZone; iZone++){
59  requestedScreenFields.emplace_back("AVG_" + bgs_res_name + "[" + PrintingToolbox::to_string(iZone) + "]");
60  }
61  nRequestedScreenFields = requestedScreenFields.size();
62  }
63 
64  multiZoneHeaderString = "Multizone Summary";
65 
66  historyFilename = driver_config->GetCaseName();
67 
68  /*--- Add the correct file extension depending on the file format ---*/
69 
70  string hist_ext = ".csv";
71  if (driver_config->GetTabular_FileFormat() == TAB_OUTPUT::TAB_TECPLOT) hist_ext = ".dat";
72 
73  historyFilename += hist_ext;
74 
75  /*--- Set the default convergence field --- */
76 
77  if (convFields.empty()) convFields.emplace_back("AVG_BGS_RES[0]");
78 
79 }

81 void CMultizoneOutput::LoadMultizoneHistoryData(const COutput* const* output, const CConfig* const* config) {
82 
83  string nameMultizone, zoneIndex;
84  su2double comboValue = 0;
85 
86  for (unsigned short iZone = 0; iZone < nZone; iZone++) {
87  zoneIndex = "[" + PrintingToolbox::to_string(iZone) + "]";
88 
89  /*--- For all the variables per solver ---*/
90  for (const auto& item : output[iZone]->GetHistoryFields()) {
91  const auto& name = item.first;
92  if (name != "TIME_ITER" && name != "OUTER_ITER") {
93  nameMultizone = name + zoneIndex;
94  SetHistoryOutputValue(nameMultizone, item.second.value);
95  }
96  if (name == "COMBO") {
97  comboValue += item.second.value;
98  }
99  }
100 
101  /*-- Load the PerSurface values.- ---*/
102  for (const auto& item : output[iZone]->GetHistoryPerSurfaceFields()) {
103  const auto& name = item.first;
104  nameMultizone = name + zoneIndex;
105 
106  unsigned short iMarker = 0;
107  for (const auto& field : item.second) {
108  SetHistoryOutputPerSurfaceValue(nameMultizone, field.value, iMarker++);
109  }
110  }
111  }
112  SetHistoryOutputValue("COMBO", comboValue);
113 }

115 void CMultizoneOutput::SetMultizoneHistoryOutputFields(const COutput* const* output, const CConfig* const* config) {
116 
117  string name, header, group, zoneIndex;
118 
119  /*--- Set the fields ---*/
120  for (unsigned short iZone = 0; iZone < nZone; iZone++) {
121 
122  const auto& ZoneHistoryFields = output[iZone]->GetHistoryFields();
123  zoneIndex = "[" + PrintingToolbox::to_string(iZone) + "]";
124 
125  /*--- For all the variables per solver ---*/
126  for (const auto& nameSinglezone : output[iZone]->GetHistoryOutput_List()) {
127 
128  if (nameSinglezone != "TIME_ITER" && nameSinglezone != "OUTER_ITER") {
129 
130  const auto& field = ZoneHistoryFields.at(nameSinglezone);
131 
132  name = nameSinglezone + zoneIndex;
133  header = field.fieldName + zoneIndex;
134  group = field.outputGroup + zoneIndex;
135 
136  AddHistoryOutput(name, header, field.screenFormat, group, "", field.fieldType );
137  }
138  }
139 
140  /*--- Prepare Marker lists that are passed to 'AddHistoryOutputPerSurface'. ---*/
141  vector<string> Marker_Analyze;
142  for (unsigned short iMarker_Analyze = 0; iMarker_Analyze < config[iZone]->GetnMarker_Analyze(); iMarker_Analyze++) {
143  Marker_Analyze.push_back(config[iZone]->GetMarker_Analyze_TagBound(iMarker_Analyze));
144  }
145 
146  vector<string> Marker_Monitoring;
147  for (unsigned short iMarker_Monitoring = 0; iMarker_Monitoring < config[iZone]->GetnMarker_Monitoring(); iMarker_Monitoring++) {
148  Marker_Monitoring.push_back(config[iZone]->GetMarker_Monitoring_TagBound(iMarker_Monitoring));
149  }
150 
151  /*--- Add the PerSurface outputs. ---*/
152  const auto& ZoneHistoryPerSurfaceFields = output[iZone]->GetHistoryPerSurfaceFields();
153 
154  for (const auto& nameSinglezone : output[iZone]->GetHistoryOutputPerSurface_List()) {
155 
156  const auto& field = ZoneHistoryPerSurfaceFields.at(nameSinglezone);
157 
158  name = nameSinglezone + zoneIndex;
159 
160  /*--- Remove the unnecessary Marker name from the fieldName, i.e. "Avg_Massflow(inlet)"->"Avg_Massflow". ---*/
161  /*--- Note that index zero in 'field[0]' refers to a specific Marker. Some attributes remain constant over the markers
162  like the first part of the name, the screenFormat and the fieldType. ---*/
163  string baseheader;
164  const auto pos = field[0].fieldName.find('(');
165  if (pos != std::string::npos)
166  baseheader = field[0].fieldName.substr(0, pos);
167  else
168  SU2_MPI::Error("Cannot process PerSurface *_SURF history output: " + baseheader, CURRENT_FUNCTION);
169 
170  header = baseheader + zoneIndex;
171  /*--- Attach zone-index to the group after determining which group it is. ---*/
172  group = field[0].outputGroup;
173 
174  /*--- Determine whether Maker_Analyze/Monitoring has to be used. ---*/
175  auto* Marker = &Marker_Monitoring;
176  if ((group == "FLOW_COEFF_SURF") || (group == "SPECIES_COEFF_SURF"))
177  Marker = &Marker_Analyze;
178  else if (group != "AERO_COEFF_SURF" && group != "HEAT_SURF")
179  SU2_MPI::Error("Per Surface output group unknown: " + group, CURRENT_FUNCTION);
180 
181  group += zoneIndex;
182 
183  AddHistoryOutputPerSurface(name, header, field[0].screenFormat, group, *Marker, field[0].fieldType );
184  }
185  }
186  AddHistoryOutput("COMBO", "ComboObj", ScreenOutputFormat::SCIENTIFIC, "COMBO", "Combined obj. function value.", HistoryFieldType::COEFFICIENT);
187 }
