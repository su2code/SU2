34 CMultizoneDriver::CMultizoneDriver(char* confFile, unsigned short val_nZone, SU2_Comm MPICommunicator) :
35  CDriver(confFile, val_nZone, MPICommunicator, false) {
36 
37  /*--- Initialize the counter for TimeIter ---*/
38  TimeIter = 0;
39 
40  /*--- Structure for multizone convergence ---*/
41  init_res = new su2double*[nZone];
42  residual = new su2double*[nZone];
43  residual_rel = new su2double*[nZone];
44  nVarZone = new unsigned short[nZone];
45 
46  for (iZone = 0; iZone < nZone; iZone++){
47  nVarZone[iZone] = 0;
48  /*--- Account for all the solvers ---*/
49  for (unsigned short iSol = 0; iSol < MAX_SOLS; iSol++){
50  if (solver_container[iZone][INST_0][MESH_0][iSol] != nullptr)
51  nVarZone[iZone] += solver_container[iZone][INST_0][MESH_0][iSol]->GetnVar();
52  }
53  init_res[iZone] = new su2double[nVarZone[iZone]];
54  residual[iZone] = new su2double[nVarZone[iZone]];
55  residual_rel[iZone] = new su2double[nVarZone[iZone]];
56  /*--- Initialize the residual containers to 0.0 ---*/
57  for (unsigned short iVar = 0; iVar < nVarZone[iZone]; iVar++){
58  init_res[iZone][iVar] = 0.0;
59  residual[iZone][iVar] = 0.0;
60  residual_rel[iZone][iVar] = 0.0;
61  }
62  }
63 
64  /*----------------------------------------------------*/
65  /*------ Determine the properties of the problem -----*/
66  /*----------------------------------------------------*/
67 
68  bool structural_zone = false;
69  bool fluid_zone = false;
70  bool heat_zone = false;
71 
72  /*--- If there is at least a fluid and a structural zone ---*/
73  for (iZone = 0; iZone < nZone; iZone++){
74  switch (config_container[iZone]->GetKind_Solver()) {
75  case MAIN_SOLVER::EULER: case MAIN_SOLVER::NAVIER_STOKES: case MAIN_SOLVER::RANS:
76  case MAIN_SOLVER::NEMO_EULER: case MAIN_SOLVER::NEMO_NAVIER_STOKES:
77  case MAIN_SOLVER::INC_EULER: case MAIN_SOLVER::INC_NAVIER_STOKES: case MAIN_SOLVER::INC_RANS:
78  fluid_zone = true;
79  break;
80  case MAIN_SOLVER::FEM_ELASTICITY:
81  structural_zone = true;
82  break;
83  case MAIN_SOLVER::HEAT_EQUATION:
84  heat_zone = true;
85  break;
86  default:
87  break;
88  }
89  }
90 
91  fsi = false; cht = false;
92  /*--- If the problem has FSI properties ---*/
93  if (fluid_zone && structural_zone) fsi = true;
94  /*--- If the problem has CHT properties ---*/
95  if (fluid_zone && heat_zone) cht = true;
96 
97  /*----------------------------------------------------*/
98  /*- Define if a prefixed motion is imposed in a zone -*/
99  /*----------------------------------------------------*/
100 
101  prefixed_motion = new bool[nZone];
102  for (iZone = 0; iZone < nZone; iZone++){
103  switch (config_container[iZone]->GetKind_GridMovement()){
104  case RIGID_MOTION:
105  prefixed_motion[iZone] = true; break;
106  default:
107  prefixed_motion[iZone] = false; break;
108  }
109  if (config_container[iZone]->GetSurface_Movement(AEROELASTIC) ||
110  config_container[iZone]->GetSurface_Movement(AEROELASTIC_RIGID_MOTION) ||
111  config_container[iZone]->GetSurface_Movement(DEFORMING) ||
112  config_container[iZone]->GetSurface_Movement(EXTERNAL) ||
113  config_container[iZone]->GetSurface_Movement(EXTERNAL_ROTATION)){
114  prefixed_motion[iZone] = true;
115  }
116  }
117 
118 }

137 void CMultizoneDriver::StartSolver() {
138 
139  /*--- Find out the minimum of all references times and then set each zone to this (same) value.
140  To ensure that all zones run synchronously in time, be it a dimensional or non-dimensionalized one. ---*/
141 
142  su2double Time_Ref = std::numeric_limits<su2double>::max();
143  for (iZone = 0; iZone < nZone; iZone++)
144  Time_Ref = min(Time_Ref, config_container[iZone]->GetTime_Ref());
145 
146  for (iZone = 0; iZone < nZone; iZone++) {
147 
148  config_container[iZone]->SetTime_Ref(Time_Ref);
149 
150  /*--- Recompute some values as the reference time might has changed in iZone ---*/
151 
152  config_container[iZone]->SetDelta_UnstTimeND(config_container[iZone]->GetDelta_UnstTime() / Time_Ref);
153  config_container[iZone]->SetTotal_UnstTimeND(config_container[iZone]->GetTotal_UnstTime() / Time_Ref);
154  }
155 
156  StartTime = SU2_MPI::Wtime();
157 
158  driver_config->Set_StartTime(StartTime);
159 
160  /*--- Main external loop of the solver. Runs for the number of time steps required. ---*/
161 
162  if (rank == MASTER_NODE){
163  cout << endl <<"------------------------------ Begin Solver -----------------------------" << endl;
164  }
165 
166  if (rank == MASTER_NODE){
167  cout << endl <<"Simulation Run using the Multizone Driver" << endl;
168  if (driver_config->GetTime_Domain())
169  cout << "The simulation will run until time step " << driver_config->GetnTime_Iter() - driver_config->GetRestart_Iter() << "." << endl;
170  }
171 
172  /*--- Set the initial time iteration to the restart iteration. ---*/
173  if (driver_config->GetRestart() && driver_config->GetTime_Domain())
174  TimeIter = driver_config->GetRestart_Iter();
175 
176  /*--- Run the problem until the number of time iterations required is reached. ---*/
177  while ( TimeIter < driver_config->GetnTime_Iter() ) {
178 
179  /*--- Perform some preprocessing before starting the time-step simulation. ---*/
180 
181  Preprocess(TimeIter);
182 
183  /*--- Run a block iteration of the multizone problem. ---*/
184 
185  switch (driver_config->GetKind_MZSolver()){
186  case ENUM_MULTIZONE::MZ_BLOCK_GAUSS_SEIDEL: Run_GaussSeidel(); break; // Block Gauss-Seidel iteration
187  case ENUM_MULTIZONE::MZ_BLOCK_JACOBI: Run_Jacobi(); break; // Block-Jacobi iteration
188  }
189 
190  /*--- Update the solution for dual time stepping strategy ---*/
191 
192  Update();
193 
194  /*--- Monitor the computations after each iteration. ---*/
195 
196  StopCalc = Monitor(TimeIter);
197 
198  /*--- Output the solution in files. ---*/
199 
200  Output(TimeIter);
201 
202  /*--- If the convergence criteria has been met, terminate the simulation. ---*/
203 
204  if (StopCalc) break;
205 
206  TimeIter++;
207 
208  }
209

212 void CMultizoneDriver::Preprocess(unsigned long TimeIter) {
213 
214  /*--- Set the current time iteration in the config ---*/
215  driver_config->SetTimeIter(TimeIter);
216 
217  for (iZone = 0; iZone < nZone; iZone++){
218 
219  /*--- Set the value of the external iteration to TimeIter. -------------------------------------*/
220  /*--- TODO: This should be generalised for an homogeneous criteria throughout the code. --------*/
221  config_container[iZone]->SetTimeIter(TimeIter);
222 
223 
224  /*--- Store the current physical time in the config container, as
225  this can be used for verification / MMS. This should also be more
226  general once the drivers are more stable. ---*/
227 
228  if (driver_config->GetTime_Domain()) {
229  config_container[iZone]->SetPhysicalTime(static_cast<su2double>(TimeIter)*config_container[iZone]->GetDelta_UnstTimeND());
230  }
231  else {
232  config_container[iZone]->SetPhysicalTime(0.0);
233  }
234 
235  /*--- Set the initial condition for EULER/N-S/RANS ---------------------------------------------*/
236  /*--- For FSI, the initial conditions are set, after the mesh has been moved. --------------------------------------*/
237  if (!fsi && config_container[iZone]->GetFluidProblem()) {
238  solver_container[iZone][INST_0][MESH_0][FLOW_SOL]->SetInitialCondition(geometry_container[iZone][INST_0],
239  solver_container[iZone][INST_0],
240  config_container[iZone], TimeIter);
241  }
242  else if (!fsi && config_container[iZone]->GetHeatProblem()) {
243  /*--- Set the initial condition for HEAT equation ---------------------------------------------*/
244  solver_container[iZone][INST_0][MESH_0][HEAT_SOL]->SetInitialCondition(geometry_container[iZone][INST_0],
245  solver_container[iZone][INST_0],
246  config_container[iZone], TimeIter);
247  }
248  }
249 
250  SU2_MPI::Barrier(SU2_MPI::GetComm());
251 
252  /*--- Run a predictor step ---*/
253  for (iZone = 0; iZone < nZone; iZone++) {
254  if (config_container[iZone]->GetPredictor())
255  iteration_container[iZone][INST_0]->Predictor(output_container[iZone], integration_container, geometry_container,
256  solver_container, numerics_container, config_container, surface_movement,
257  grid_movement, FFDBox, iZone, INST_0);
258  }
259 
260  /*--- Perform a dynamic mesh update if required. ---*/
261 
262  DynamicMeshUpdate(TimeIter);
263 
264  /*--- Updating zone interface communication patterns for unsteady problems with pre-fixed motion in the config file ---*/
265  if (driver_config->GetTime_Domain()) {
266  for (iZone = 0; iZone < nZone; iZone++) {
267  for (unsigned short jZone = 0; jZone < nZone; jZone++){
268  if(jZone != iZone && interpolator_container[iZone][jZone] != nullptr && prefixed_motion[iZone])
269  interpolator_container[iZone][jZone]->SetTransferCoeff(config_container);
270  }
271  }
272  }

276 void CMultizoneDriver::Run_GaussSeidel() {
277 
278  unsigned short UpdateMesh;
279  bool DeformMesh = false;
280 
281  for (iZone = 0; iZone < nZone; iZone++) {
282  config_container[iZone]->SetOuterIter(0ul);
283  }
284 
285  /*--- Loop over the number of outer iterations ---*/
286  for (auto iOuter_Iter = 0ul; iOuter_Iter < driver_config->GetnOuter_Iter(); iOuter_Iter++){
287 
288  /*--- Loop over the number of zones (IZONE) ---*/
289  for (iZone = 0; iZone < nZone; iZone++){
290 
291  /*--- In principle, the mesh does not need to be updated ---*/
292  UpdateMesh = 0;
293 
294  /*--- Set the OuterIter ---*/
295  config_container[iZone]->SetOuterIter(iOuter_Iter);
296  config_container[iZone]->Set_StartTime(SU2_MPI::Wtime());
297  driver_config->SetOuterIter(iOuter_Iter);
298 
299  /*--- Transfer from all the remaining zones ---*/
300  for (auto jZone = 0u; jZone < nZone; jZone++){
301  /*--- The target zone is iZone ---*/
302  if (jZone != iZone){
303  DeformMesh = Transfer_Data(jZone, iZone);
304  if (DeformMesh) UpdateMesh+=1;
305  }
306  }
307  /*--- If a mesh update is required due to the transfer of data ---*/
308  if (UpdateMesh > 0) DynamicMeshUpdate(iZone, TimeIter);
309 
310  /*--- Iterate the zone as a block, either to convergence or to a max number of iterations ---*/
311  iteration_container[iZone][INST_0]->Solve(output_container[iZone], integration_container, geometry_container,
312  solver_container, numerics_container, config_container,
313  surface_movement, grid_movement, FFDBox, iZone, INST_0);
314 
315  /*--- A corrector step can help preventing numerical instabilities ---*/
316  Corrector(iZone);
317 
318  }
319 
320  if (OuterConvergence(iOuter_Iter)) break;
321 
322  }

393 bool CMultizoneDriver::OuterConvergence(unsigned long OuterIter) {
394 
395  /*--- Update the residual for the all the zones. ---*/
396 
397  for (iZone = 0; iZone < nZone; iZone++) {
398 
399  /*--- Account for all the solvers in this zone. ---*/
400 
401  auto solvers = solver_container[iZone][INST_0][MESH_0];
402 
403  for (unsigned short iSol = 0; iSol < MAX_SOLS; iSol++){
404  if (solvers[iSol] != nullptr) {
405  solvers[iSol]->ComputeResidual_Multizone(geometry_container[iZone][INST_0][MESH_0], config_container[iZone]);
406  solvers[iSol]->GetNodes()->Set_BGSSolution_k();
407  }
408  }
409 
410  /*--- Make sure that everything is loaded into the output container. ---*/
411 
412  output_container[iZone]->SetHistory_Output(geometry_container[iZone][INST_0][MESH_0], solvers, config_container[iZone]);
413 
414  }
415 
416  /*--- Print out the convergence data to screen and history file. ---*/
417 
418  driver_output->SetMultizoneHistory_Output(output_container, config_container, driver_config,
419  driver_config->GetTimeIter(), driver_config->GetOuterIter());
420 
421  return driver_output->GetConvergence();

425 void CMultizoneDriver::Update() {
426 
427  /*--- For enabling a consistent restart, we need to update the mesh with the interface information that introduces displacements --*/
428  /*--- Loop over the number of zones (IZONE) ---*/
429  for (iZone = 0; iZone < nZone; iZone++){
430 
431  unsigned short UpdateMesh = 0;
432 
433  /*--- Transfer from all the remaining zones (JZONE != IZONE)---*/
434  for (auto jZone = 0u; jZone < nZone; jZone++){
435  /*--- The target zone is iZone ---*/
436  if (jZone != iZone){
437  UpdateMesh += Transfer_Data(jZone, iZone);
438  }
439  }
440  /*--- If a mesh update is required due to the transfer of data ---*/
441  if (UpdateMesh > 0) DynamicMeshUpdate(iZone, TimeIter);
442 
443  iteration_container[iZone][INST_0]->Update(output_container[iZone], integration_container, geometry_container,
444  solver_container, numerics_container, config_container,
445  surface_movement, grid_movement, FFDBox, iZone, INST_0);
446 
447  /*--- Set the Convergence_FSI boolean to false for the next time step ---*/
448  for (unsigned short iSol = 0; iSol < MAX_SOLS-1; iSol++){
449  if (integration_container[iZone][INST_0][iSol] != nullptr){
450  integration_container[iZone][INST_0][iSol]->SetConvergence_FSI(false);
451  }
452  }
453  }

490 void CMultizoneDriver::DynamicMeshUpdate(unsigned long TimeIter) {
491 
492  bool AnyDeformMesh = false;
493 
494  for (iZone = 0; iZone < nZone; iZone++) {
495  const auto harmonic_balance = (config_container[iZone]->GetTime_Marching() == TIME_MARCHING::HARMONIC_BALANCE);
496  /*--- Dynamic mesh update ---*/
497  if ((config_container[iZone]->GetGrid_Movement()) && (!harmonic_balance) && (!fsi)) {
498  iteration_container[iZone][INST_0]->SetGrid_Movement(geometry_container[iZone][INST_0],surface_movement[iZone],
499  grid_movement[iZone][INST_0], solver_container[iZone][INST_0],
500  config_container[iZone], 0, TimeIter);
501  AnyDeformMesh = true;
502  }
503  }
504  /*--- Update the wall distances if the mesh was deformed. ---*/
505  if (AnyDeformMesh) {
506  CGeometry::ComputeWallDistance(config_container, geometry_container);
507  }

535 bool CMultizoneDriver::Transfer_Data(unsigned short donorZone, unsigned short targetZone) {
536 
537  bool UpdateMesh = false;
538 
539  /*--- Select the transfer method according to the magnitudes being transferred ---*/
540 
541  auto BroadcastData = [&](int donorSol, int targetSol) {
542  interface_container[donorZone][targetZone]->BroadcastData(
543  *interpolator_container[donorZone][targetZone].get(),
544  solver_container[donorZone][INST_0][MESH_0][donorSol],
545  solver_container[targetZone][INST_0][MESH_0][targetSol],
546  geometry_container[donorZone][INST_0][MESH_0],
547  geometry_container[targetZone][INST_0][MESH_0],
548  config_container[donorZone],
549  config_container[targetZone]);
550  };
551 
552  switch (interface_types[donorZone][targetZone]) {
553 
554  case SLIDING_INTERFACE:
555  BroadcastData(FLOW_SOL, FLOW_SOL);
556 
557  /*--- Additional transfer for turbulence variables. ---*/
558  if (config_container[targetZone]->GetKind_Solver() == MAIN_SOLVER::RANS ||
559  config_container[targetZone]->GetKind_Solver() == MAIN_SOLVER::INC_RANS) {
560  BroadcastData(TURB_SOL, TURB_SOL);
561  }
562 
563  /*--- Additional transfer for species variables. ---*/
564  if (config_container[targetZone]->GetKind_Species_Model() != SPECIES_MODEL::NONE) {
565  BroadcastData(SPECIES_SOL, SPECIES_SOL);
566  }
567  break;
568  case CONJUGATE_HEAT_FS:
569  BroadcastData(FLOW_SOL, HEAT_SOL);
570  break;
571  case CONJUGATE_HEAT_WEAKLY_FS:
572  BroadcastData(HEAT_SOL, HEAT_SOL);
573  break;
574  case CONJUGATE_HEAT_SF:
575  BroadcastData(HEAT_SOL, FLOW_SOL);
576  break;
577  case CONJUGATE_HEAT_WEAKLY_SF:
578  BroadcastData(HEAT_SOL, HEAT_SOL);
579  break;
580  case BOUNDARY_DISPLACEMENTS:
581  BroadcastData(FEA_SOL, MESH_SOL);
582  UpdateMesh = true;
583  break;
584  case FLOW_TRACTION:
585  BroadcastData(FLOW_SOL, FEA_SOL);
586  break;
587  case NO_TRANSFER:
588  case ZONES_ARE_EQUAL:
589  case NO_COMMON_INTERFACE:
590  break;
591  default:
592  if(rank == MASTER_NODE)
593  cout << "WARNING: One of the intended interface transfer routines is not "
594  << "known to the chosen driver and has not been executed." << endl;
595  break;
596  }
597 
598  return UpdateMesh;
599 }

601 bool CMultizoneDriver::Monitor(unsigned long TimeIter){
602 
603  /*--- Check whether the inner solver has converged --- */
604 
605  if (driver_config->GetTime_Domain() == NO){
606 
607  const auto OuterIter = driver_config->GetOuterIter();
608  const auto nOuterIter = driver_config->GetnOuter_Iter();
609 
610  const auto InnerConvergence = driver_output->GetConvergence();

622  }
