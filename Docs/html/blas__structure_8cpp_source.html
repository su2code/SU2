55 void CBlasStructure::gemm(const int M, const int N, const int K,
56  const su2double *A, const su2double *B, su2double *C,
57  const CConfig *config) {
58 
59  /* Initialize the variable for the timing, if profiling is active. */
60 #ifdef PROFILE
61  double timeGemm;
62  if( config ) config->GEMM_Tick(&timeGemm);
63 #endif
64 
65 #if (defined(CODI_REVERSE_TYPE) || defined(CODI_FORWARD_TYPE)) || !(defined(HAVE_LIBXSMM) || defined(HAVE_MKL) || defined(HAVE_BLAS))
66  /* Native implementation of the matrix product. This optimized implementation
67  assumes that the matrices are in column major order. This can be
68  accomplished by swapping N and M and A and B. This implementation is based
69  on https://github.com/flame/how-to-optimize-gemm. */
70  gemm_imp(N, M, K, B, A, C);
71 
72 #else
73 #ifdef HAVE_LIBXSMM
74 
75  /* The gemm function of libxsmm is used to carry out the multiplication.
76  Note that libxsmm_gemm expects the matrices in column major order. That's
77  why the in the calling sequence A and B and M and N are reversed. */
78  su2double alpha = 1.0;
79  su2double beta = 0.0;
80  char trans = 'N';
81 
82  libxsmm_dgemm(&trans, &trans, &N, &M, &K, &alpha, B, &N, A, &K, &beta, C, &N);
83 
84 #else // MKL and BLAS
85 
86  /* The standard blas routine dgemm is used for the multiplication.
87  Call dgemm without transposing the matrices. In that case dgemm expects
88  the matrices in column major order, see the comments for libxsmm. */
89  su2double alpha = 1.0;
90  su2double beta = 0.0;
91  char trans = 'N';
92 
93  dgemm_(&trans, &trans, &N, &M, &K, &alpha, B, &N, A, &K, &beta, C, &N);
94 
95 #endif
96 #endif
97 
98  /* Store the profiling information, if needed. */
99 #ifdef PROFILE
100  if( config ) config->GEMM_Tock(timeGemm, M, N, K);
101 #endif
102 }

104 /* Dense matrix vector multiplication, gemv functionality. */
105 void CBlasStructure::gemv(const int M, const int N, const su2double *A,
106  const su2double *x, su2double *y) {
107 
108 #if (defined (HAVE_BLAS) || defined(HAVE_MKL)) && !(defined(CODI_REVERSE_TYPE) || defined(CODI_FORWARD_TYPE))
109 
110  /* The standard blas routine dgemv is used for the multiplication.
111  Note that dgemv expects the matrices in column major order, while
112  A is in row major order. This can be solved by using the transpose
113  and switching M and N. */
114  su2double alpha = 1.0;
115  su2double beta = 0.0;
116  int inc = 1;
117  char trans = 'T';
118 
119  dgemv_(&trans, &N, &M, &alpha, A, &N, x, &inc, &beta, y, &inc);
120 
121 #else
122 
123  /* Native implementation of the matix vector product.
124  Initialize the elements of y to zero. */
125  for(int i=0; i<M; ++i) y[i] = 0.0;
126 
127  /* Carry out the matrix vector product. */
128  for(int k=0; k<M; ++k) {
129  const su2double *AA = A + k*N;
130  for(int l=0; l<N; ++l)
131  y[k] += AA[l]*x[l];
132  }
133 
134 #endif
135 }

144 /* Function, which perform the implementation of the gemm functionality. */
145 void CBlasStructure::gemm_imp(const int m, const int n, const int k,
146  const su2double *a, const su2double *b, su2double *c) {
147 
148  /* Initialize the elements of c to zero. */
149  for(int i=0; i<(m*n); ++i) c[i] = 0.0;
150 
151  /* Set the leading dimensions of the three matrices. */
152  const int lda = m;
153  const int ldb = k;
154  const int ldc = m;
155 
156  /* The full matrix multiplication is split in several blocks.
157  Loop over these blocks. */
158  for(int p=0; p<k; p+=kc) {
159  int pb = min(k-p, kc);
160  for(int j=0; j<n; j+=nc) {
161  int jb = min(n-j, nc);
162  for(int i=0; i<m; i+=mc) {
163  int ib = min(m-i, mc);
164 
165  /* Carry out the multiplication for this block. */
166  gemm_inner(ib, jb, pb, &A(i, p), lda, &B(p, j), ldb, &C(i, j), ldc);
167  }
168  }
169  }
170 }

182 /* Naive gemm implementation to handle arbitrary sized matrices. */
183 void CBlasStructure::gemm_arbitrary(int m, int n, int k, const su2double *a, int lda,
184  const su2double *b, int ldb, su2double *c, int ldc) {
185 
186  /* The order of these loops is tuned for column-major matrices. */
187  for (int p = 0; p < k; p++) {
188  for (int j = 0; j < n; j++) {
189  for (int i = 0; i < m; i++) {
190  C(i, j) += A(i, p) * B(p, j);
191  }
192  }
193  }
194 }
