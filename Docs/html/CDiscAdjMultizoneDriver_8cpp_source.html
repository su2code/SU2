162 
163  const bool time_domain = driver_config->GetTime_Domain();
164 
165  /*--- Main external loop of the solver. Runs for the number of time steps required. ---*/
166 
167  if (rank == MASTER_NODE) {
168  cout <<"\n------------------------------ Begin Solver -----------------------------" << endl;
169 
170  cout << "\nSimulation Run using the Discrete Adjoint Multizone Driver" << endl;
171 
172  if (time_domain)

174  }
175 
176  /*--- General setup. ---*/
177 
178  for (iZone = 0; iZone < nZone; iZone++) {
179 
180  wrt_sol_freq = min(wrt_sol_freq, config_container[iZone]->GetVolumeOutputFrequency(0));
181 
182  /*--- Set BGS_Solution_k to Solution, this is needed to restart
183  * correctly as the first OF gradient will overwrite the solution. ---*/
184 
185  Set_BGSSolution_k_To_Solution(iZone);
186 
187  /*--- Prepare Krylov or quasi-Newton methods. ---*/
188 
189  const auto nPoint = geometry_container[iZone][INST_0][MESH_0]->GetnPoint();
190  const auto nPointDomain = geometry_container[iZone][INST_0][MESH_0]->GetnPointDomain();
191  const auto nVar = GetTotalNumberOfVariables(iZone, true);
192 
193  if (config_container[iZone]->GetNewtonKrylov() &&
194  config_container[iZone]->GetnQuasiNewtonSamples() >= KrylovMinIters) {
195  AdjRHS[iZone].Initialize(nPoint, nPointDomain, nVar, nullptr);
196  AdjSol[iZone].Initialize(nPoint, nPointDomain, nVar, nullptr);
197  LinSolver[iZone].SetToleranceType(LinearToleranceType::RELATIVE);
198  }
199  else if (config_container[iZone]->GetnQuasiNewtonSamples() > 1) {
200  FixPtCorrector[iZone].resize(config_container[iZone]->GetnQuasiNewtonSamples(), nPoint, nVar, nPointDomain);
201  }
202  }
203 
204  /*--- Size and initialize the matrix of cross-terms. ---*/
205 
206  InitializeCrossTerms();
207 
208  /*--- Run time iterations. ---*/
209 
210  while (TimeIter < driver_config->GetnTime_Iter()) {
211 
212  Preprocess(TimeIter);
213 
214  Run();
215 
216  TimeIter++;
217  }

221 bool CDiscAdjMultizoneDriver::Iterate(unsigned short iZone, unsigned long iInnerIter, bool KrylovMode) {
222 
223  config_container[iZone]->SetInnerIter(iInnerIter);
224 
225  /*--- Evaluate the tape section belonging to solvers in iZone.
226  * Only evaluate TRANSFER terms on the last iteration or after convergence. ---*/
227 
228  eval_transfer = (eval_transfer || (iInnerIter == nInnerIter[iZone]-1)) && !KrylovMode;
229 
230  ComputeAdjoints(iZone, eval_transfer);
231 
232  /*--- Extracting adjoints for solvers in iZone w.r.t. to outputs in iZone (diagonal part). ---*/
233 
234  iteration_container[iZone][INST_0]->IterateDiscAdj(geometry_container, solver_container,
235  config_container, iZone, INST_0, false);
236 
237  /*--- Use QN driver to improve the solution. ---*/
238 
239  if (FixPtCorrector[iZone].size()) {
240  GetAllSolutions(iZone, true, FixPtCorrector[iZone].FPresult());
241  FixPtCorrector[iZone].compute();
242  if(iInnerIter) SetAllSolutions(iZone, true, FixPtCorrector[iZone]);
243  }
244 
245  /*--- This is done explicitly here for multizone cases, only in inner iterations and not when
246  * extracting cross terms so that the adjoint residuals in each zone still make sense. ---*/
247 
248  if (!KrylovMode) Set_SolutionOld_To_Solution(iZone);
249 
250  /*--- Print out the convergence data to screen and history file. ---*/
251 
252  return iteration_container[iZone][INST_0]->Monitor(output_container[iZone], integration_container, geometry_container,
253  solver_container, numerics_container, config_container,
254  surface_movement, grid_movement, FFDBox, iZone, INST_0);
255 }

257 void CDiscAdjMultizoneDriver::KrylovInnerIters(unsigned short iZone) {
258 
259  /*--- Use FGMRES to solve the adjoint system, the RHS is -External,
260  * the solution are the iZone adjoint variables + External,
261  * Recall that External also contains the OF gradient. ---*/
262 
263  GetAdjointRHS(iZone, AdjRHS[iZone]);
264 
265  Add_External_To_Solution(iZone);
266 
267  GetAllSolutions(iZone, true, AdjSol[iZone]);
268 
269  const bool monitor = config_container[iZone]->GetWrt_ZoneConv();
270  const auto product = AdjointProduct(this, iZone);
271 
272  /*--- Manipulate the screen output frequency to avoid printing garbage. ---*/
273  const auto wrtFreq = config_container[iZone]->GetScreen_Wrt_Freq(2);
274  config_container[iZone]->SetScreen_Wrt_Freq(2, nInnerIter[iZone]);
275  LinSolver[iZone].SetMonitoringFrequency(wrtFreq);
276 
277  Scalar eps = 1.0;
278  for (auto totalIter = nInnerIter[iZone]; totalIter >= KrylovMinIters && eps > KrylovTol;) {
279  Scalar eps_l = 0.0;
280  Scalar tol_l = KrylovTol / eps;
281  auto iter = min(totalIter-2ul, config_container[iZone]->GetnQuasiNewtonSamples()-2ul);
282  iter = LinSolver[iZone].FGMRES_LinSolver(AdjRHS[iZone], AdjSol[iZone], product, Identity(),
283  tol_l, iter, eps_l, monitor, config_container[iZone]);
284  totalIter -= iter+1;
285  eps *= eps_l;
286  }
287 
288  /*--- Store the solution and restore user settings. ---*/
289  SetAllSolutions(iZone, true, AdjSol[iZone]);
290  config_container[iZone]->SetScreen_Wrt_Freq(2, wrtFreq);
291 
292  /*--- Set the old solution such that iterating gives meaningful residuals. ---*/
293  AdjSol[iZone] += AdjRHS[iZone];
294  SetAllSolutionsOld(iZone, true, AdjSol[iZone]);
295 
296  /*--- Iterate to evaluate cross terms and residuals, this cannot happen within GMRES
297  * because the vectors it multiplies by the Jacobian are not the actual solution. ---*/
298  eval_transfer = true;
299  Iterate(iZone, product.iInnerIter);
300 
301  /*--- Set the solution as obtained from GMRES, otherwise it would be GMRES+Iterate once.
302  * This is set without the "External" (by adding RHS above) so that it can be added
303  * again in the next outer iteration with new contributions from other zones. ---*/
304  SetAllSolutions(iZone, true, AdjSol[iZone]);
305

308 void CDiscAdjMultizoneDriver::Run() {
309 
310  const unsigned long nOuterIter = driver_config->GetnOuter_Iter();
311  const bool time_domain = driver_config->GetTime_Domain();
312 
313  /*--- If the gradient of the objective function is 0 so are the adjoint variables.
314  * Unless in unsteady problems where there are other contributions to the RHS. ---*/
315 
316  const auto zeroGrad = EvaluateObjectiveFunctionGradient();
317 
318  if (zeroGrad && !time_domain) {
319  if (rank == MASTER_NODE) {
320  cout << "\nThe gradient of the objective function is numerically 0.";
321  cout << "\nThis implies that the adjoint variables are also 0.\n\n";
322  }
323  EvaluateSensitivities(0, true);
324  return;
325  }
326 
327  /*--- Loop over the number of outer iterations. ---*/
328 
329  for (unsigned long iOuterIter = 0, StopCalc = false; !StopCalc; iOuterIter++) {
330 
331  driver_config->SetOuterIter(iOuterIter);
332 
333  for (iZone = 0; iZone < nZone; iZone++)
334  config_container[iZone]->SetOuterIter(iOuterIter);
335 
336  /*--- For the adjoint iteration we need the derivatives of the iteration function with
337  * respect to the state (and possibly the mesh coordinate) variables.
338  * Since these derivatives do not change in the steady state case we only have to record
339  * if the current recording is different from them.
340  *
341  * To set the tape appropriately, the following recording methods are provided:
342  * (1) CLEAR_INDICES: All information from a previous recording is removed.
343  * (2) SOLUTION_VARIABLES: State variables of all solvers in a zone as input.
344  * (3) MESH_COORDS / MESH_DEFORM: Mesh coordinates as input.
345  * (4) SOLUTION_AND_MESH: Mesh coordinates and state variables as input.
346  *
347  * By default, all (state and mesh coordinate variables) will be declared as output,
348  * since it does not change the computational effort, just the memory consumption of the tape. ---*/
349 
350 
351  /*--- If we want to set up zone-specific tapes (retape), we do not need to record
352  * here. Otherwise, the whole tape of a coupled run will be created. ---*/
353 
354  if (RecordingState != RECORDING::SOLUTION_VARIABLES) {
355  SetRecording(RECORDING::CLEAR_INDICES, Kind_Tape::FULL_TAPE, ZONE_0);
356  SetRecording(RECORDING::SOLUTION_VARIABLES, Kind_Tape::FULL_TAPE, ZONE_0);
357  }
358 
359  /*-- Start loop over zones. ---*/
360 
361  for (iZone = 0; iZone < nZone; iZone++) {
362 
363  config_container[iZone]->Set_StartTime(SU2_MPI::Wtime());
364 
365  /*--- Start inner iterations from where we stopped in previous outer iteration. ---*/
366 
367  Set_Solution_To_BGSSolution_k(iZone);
368 
369  /*--- Inner loop to allow for multiple adjoint updates with respect to solvers in iZone. ---*/
370 
371  const bool restart = config_container[iZone]->GetRestart();
372  const bool no_restart = (iOuterIter > 0) || !restart;
373  eval_transfer = false;
374 
375  /*--- Reset QN driver for new inner iterations. ---*/
376 
377  if (FixPtCorrector[iZone].size()) {
378  FixPtCorrector[iZone].reset();
379  if(restart && (iOuterIter==1)) GetAllSolutions(iZone, true, FixPtCorrector[iZone]);
380  }
381 
382  if (!config_container[iZone]->GetNewtonKrylov() || !no_restart || nInnerIter[iZone] < KrylovMinIters) {
383 
384  /*--- Regular fixed-point, possibly with quasi-Newton method. ---*/
385 
386  for (unsigned long iInnerIter = 0; iInnerIter < nInnerIter[iZone]; iInnerIter++) {
387 
388  /*--- Add off-diagonal contribution (including the OF gradient) to Solution. ---*/
389 
390  if (no_restart || (iInnerIter > 0)) {
391  Add_External_To_Solution(iZone);
392  }
393  else {
394  /*--- If we restarted, Solution already has all contributions,
395  * we run only one inner iter to compute the cross terms. ---*/
396  eval_transfer = true;
397  }
398 
399  const bool converged = Iterate(iZone, iInnerIter);
400 
401  if (eval_transfer) break;
402 
403  eval_transfer = converged;
404  }
405  }
406  else {
407  /*--- Use Krylov method to drive inner iterations of this zone. ---*/
408 
409  KrylovInnerIters(iZone);
410  }
411 
412  /*--- Off-diagonal (coupling term) BGS update. ---*/
413 
414  // for (unsigned short jZone = 0; jZone < nZone; jZone++) {
415 
416  // if (jZone != iZone && interface_container[jZone][iZone] != nullptr) {
417 
418  // /*--- Extracting adjoints for solvers in jZone w.r.t. to the output of all solvers in iZone,
419  // * that is, for the cases iZone != jZone we are evaluating cross derivatives between zones. ---*/
420 
421  // config_container[jZone]->SetInnerIter(0);
422  // iteration_container[jZone][INST_0]->IterateDiscAdj(geometry_container, solver_container,
423  // config_container, jZone, INST_0, true);
424 
425  // /*--- Extract the cross-term performing a relaxed update of it and of the sum (External) for jZone. ---*/
426 
427  // Update_Cross_Term(iZone, jZone);
428  // }
429  // }
430 
431  /*--- Compute residual from Solution and Solution_BGS_k and update the latter. ---*/
432 
433  SetResidual_BGS(iZone);
434  Set_BGSSolution_k_To_Solution(iZone);
435  }
436 
437  /*--- Set the multizone output. ---*/
438 
439  driver_output->SetMultizoneHistory_Output(output_container, config_container, driver_config, TimeIter, iOuterIter);
440 
441  /*--- Check for convergence. ---*/
442 
443  StopCalc = driver_output->GetConvergence() || (iOuterIter == nOuterIter-1) ||
444  ((nZone==1) && output_container[ZONE_0]->GetConvergence());
445 
446  /*--- Clear the stored adjoint information to be ready for a new evaluation. ---*/
447 
448  AD::ClearAdjoints();
449 
450  /*--- Compute the geometrical sensitivities and write them to file, except for time_domain. ---*/
451 
452  if (time_domain) continue;
453 
454  if (StopCalc || ((iOuterIter % wrt_sol_freq == 0) && (iOuterIter != 0)))
455  EvaluateSensitivities(iOuterIter, StopCalc);
456  }
457 
458  if (time_domain) {
459  EvaluateSensitivities(TimeIter, (TimeIter+1) == driver_config->GetnTime_Iter());
460  }

464 bool CDiscAdjMultizoneDriver::EvaluateObjectiveFunctionGradient() {
465 
466  /*--- Evaluate the objective function gradient w.r.t. the solutions of all zones. ---*/
467 
468  SetRecording(RECORDING::CLEAR_INDICES, Kind_Tape::OBJECTIVE_FUNCTION_TAPE, ZONE_0);
469  SetRecording(RECORDING::SOLUTION_VARIABLES, Kind_Tape::OBJECTIVE_FUNCTION_TAPE, ZONE_0);
470  RecordingState = RECORDING::CLEAR_INDICES;
471 
472  AD::ClearAdjoints();
473  SetAdj_ObjFunction();
474  AD::ComputeAdjoint(OBJECTIVE_FUNCTION, START);
475 
476  /*--- Initialize External with the objective function gradient. ---*/
477 
478  su2double rhs_norm = 0.0;
479 
480  for (iZone = 0; iZone < nZone; iZone++) {
481 
482  iteration_container[iZone][INST_0]->IterateDiscAdj(geometry_container, solver_container,
483  config_container, iZone, INST_0, false);
484  Add_Solution_To_External(iZone);
485 
486  for (unsigned short iSol=0; iSol < MAX_SOLS; iSol++) {
487  auto solver = solver_container[iZone][INST_0][MESH_0][iSol];
488  if (solver && solver->GetAdjoint())
489  for (unsigned short iVar=0; iVar < solver->GetnVar(); ++iVar)
490  rhs_norm += solver->GetRes_RMS(iVar);
491  }
492  }
493 
494  return rhs_norm < EPS;
495 }

497 void CDiscAdjMultizoneDriver::EvaluateSensitivities(unsigned long Iter, bool force_writing) {
498 
499  /*--- SetRecording stores the computational graph on one iteration of the direct problem. Calling it with NONE
500  * as argument ensures that all information from a previous recording is removed. ---*/
501 
502  SetRecording(RECORDING::CLEAR_INDICES, Kind_Tape::FULL_TAPE, ZONE_0);
503 
504  /*--- Store the computational graph of one direct iteration with the mesh coordinates as input. ---*/
505 
506  SetRecording(RECORDING::MESH_COORDS, Kind_Tape::FULL_TAPE, ZONE_0);
507 
508  /*--- Initialize the adjoint of the output variables of the iteration with the adjoint solution
509  * of the current iteration. The values are passed to the AD tool. ---*/
510 
511  for (iZone = 0; iZone < nZone; iZone++) {
512 
513  Set_Solution_To_BGSSolution_k(iZone);
514 
515  Add_External_To_Solution(iZone);
516 
517  iteration_container[iZone][INST_0]->InitializeAdjoint(solver_container, geometry_container,
518  config_container, iZone, INST_0);
519  }
520 
521  /*--- Initialize the adjoint of the objective function with 1.0. ---*/
522 
523  SetAdj_ObjFunction();
524 
525  /*--- Interpret the stored information by calling the corresponding routine of the AD tool. ---*/
526 
527  AD::ComputeAdjoint();
528 
529  /*--- Extract the computed sensitivity values. ---*/
530 
531  for (iZone = 0; iZone < nZone; iZone++) {
532 
533  auto config = config_container[iZone];
534  auto solvers = solver_container[iZone][INST_0][MESH_0];
535  auto geometry = geometry_container[iZone][INST_0][MESH_0];
536 
537  int IDX_SOL = -1;
538  if (config->GetFluidProblem()) IDX_SOL = ADJFLOW_SOL;
539  else if (config->GetHeatProblem()) IDX_SOL = ADJHEAT_SOL;
540  else if (config->GetStructuralProblem()) IDX_SOL = ADJFEA_SOL;
541  else {
542  if (rank == MASTER_NODE)
543  cout << "WARNING: Sensitivities not set for one of the specified discrete adjoint solvers!" << endl;
544  continue;
545  }
546 
547  if (Has_Deformation(iZone))
548  solvers[ADJMESH_SOL]->SetSensitivity(geometry, config, solvers[IDX_SOL]);
549  else
550  solvers[IDX_SOL]->SetSensitivity(geometry, config);
551 
552  iteration_container[iZone][INST_0]->Postprocess(output_container[iZone], integration_container, geometry_container,
553  solver_container, numerics_container, config_container,
554  surface_movement, grid_movement, FFDBox, iZone, INST_0);
555  }
556 
557  /*--- Clear the stored adjoint information to be ready for a new evaluation. ---*/
558 
559  AD::ClearAdjoints();
560 
561  /*--- Output files (CMultizoneDriver::Output uses StopCalc to force file output). ---*/
562 
563  swap(StopCalc, force_writing);
564  Output(Iter);
565  swap(StopCalc, force_writing);
566 
567 }

569 void CDiscAdjMultizoneDriver::SetRecording(RECORDING kind_recording, Kind_Tape tape_type, unsigned short record_zone) {
570 
571  AD::Reset();
572 
573  /*--- Prepare for recording by resetting the solution to the initial converged solution. ---*/
574 
575  for(iZone = 0; iZone < nZone; iZone++) {
576  for (unsigned short iSol = 0; iSol < MAX_SOLS; iSol++) {
577  for (unsigned short iMesh = 0; iMesh <= config_container[iZone]->GetnMGLevels(); iMesh++) {
578  auto solver = solver_container[iZone][INST_0][iMesh][iSol];
579  if (solver && solver->GetAdjoint()) {
