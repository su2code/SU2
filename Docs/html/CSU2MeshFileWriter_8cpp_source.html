37 void CSU2MeshFileWriter::Write_Data(string val_filename) {
38 
39  ofstream output_file;
40 
41  /*--- We append the pre-defined suffix (extension) to the filename (prefix) ---*/
42  val_filename.append(fileExt);
43 
44  /*--- Only the FIRST node writes the header (it does not matter if that is the master). ---*/
45 
46  if (rank == 0) {
47  /*--- For multizone-cases this only works if the all zonal meshes are in one file.
48  If the meshes are separate for each zone another solution has to be found. ---*/
49  if (iZone==0) {
50  output_file.open(val_filename);
51  } else {
52  output_file.open(val_filename, ios::app);
53  }
54 
55  if (iZone==0 && nZone>1) {
56  output_file << "NZONE= " << nZone << endl;
57  }
58 
59  if (nZone > 1){
60  output_file << "IZONE= " << iZone+1 << endl;
61  }
62 
63  /*--- Write dimensions data. ---*/
64 
65  output_file << "NDIME= " << dataSorter->GetnDim() << endl;
66 
67  output_file << "NELEM= " << dataSorter->GetnElemGlobal() << endl;
68 
69  output_file.close();
70  }
71 
72  unsigned long nElem = 0, offset = 0;
73 
74  for (int iProcessor = 0; iProcessor < size; iProcessor++) {
75  if (rank == iProcessor) {
76  output_file.open(val_filename, ios::app);
77 
78  for (auto iElem = 0ul; iElem < dataSorter->GetnElem(TRIANGLE); iElem++) {
79  output_file << "5\t";
80  for (auto iNode = 0u; iNode < N_POINTS_TRIANGLE; ++iNode)
81  output_file << dataSorter->GetElem_Connectivity(TRIANGLE, iElem, iNode) - 1 << "\t";
82  output_file << nElem + offset << "\n"; nElem++;
83  }
84  for (auto iElem = 0ul; iElem < dataSorter->GetnElem(QUADRILATERAL); iElem++) {
85  output_file << "9\t";
86  for (auto iNode = 0u; iNode < N_POINTS_QUADRILATERAL; ++iNode)
87  output_file << dataSorter->GetElem_Connectivity(QUADRILATERAL, iElem, iNode) - 1 << "\t";
88  output_file << nElem + offset << "\n"; nElem++;
89  }
90  for (auto iElem = 0ul; iElem < dataSorter->GetnElem(TETRAHEDRON); iElem++) {
91  output_file << "10\t";
92  for (auto iNode = 0u; iNode < N_POINTS_TETRAHEDRON; ++iNode)
93  output_file << dataSorter->GetElem_Connectivity(TETRAHEDRON, iElem, iNode) - 1 << "\t";
94  output_file << nElem + offset << "\n"; nElem++;
95  }
96  for (auto iElem = 0ul; iElem < dataSorter->GetnElem(HEXAHEDRON); iElem++) {
97  output_file << "12\t";
98  for (auto iNode = 0u; iNode < N_POINTS_HEXAHEDRON; ++iNode)
99  output_file << dataSorter->GetElem_Connectivity(HEXAHEDRON, iElem, iNode) - 1 << "\t";
100  output_file << nElem + offset << "\n"; nElem++;
101  }
102  for (auto iElem = 0ul; iElem < dataSorter->GetnElem(PRISM); iElem++) {
103  output_file << "13\t";
104  for (auto iNode = 0u; iNode < N_POINTS_PRISM; ++iNode)
105  output_file << dataSorter->GetElem_Connectivity(PRISM, iElem, iNode) - 1 << "\t";
106  output_file << nElem + offset << "\n"; nElem++;
107  }
108 
109  for (auto iElem = 0ul; iElem < dataSorter->GetnElem(PYRAMID); iElem++) {
110  output_file << "14\t";
111  for (auto iNode = 0u; iNode < N_POINTS_PYRAMID; ++iNode)
112  output_file << dataSorter->GetElem_Connectivity(PYRAMID, iElem, iNode) - 1 << "\t";
113  output_file << nElem + offset << "\n"; nElem++;
114  }
115 
116  output_file.close();
117  }
118 
119  /*--- Communicate offset, implies a barrier. ---*/
120  SU2_MPI::Allreduce(&nElem, &offset, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
121  }
122 
123  /*--- Write the node coordinates. ---*/
124 
125  if (rank == 0) {
126  output_file.open(val_filename, ios::app);
127  output_file << "NPOIN= " << dataSorter->GetnPointsGlobal() << "\n";
128  output_file.close();
129  }
130 
131  unsigned long myPoint = 0; offset = 0;
132 
133  for (int iProcessor = 0; iProcessor < size; iProcessor++) {
134  if (rank == iProcessor) {
135  output_file.open(val_filename, ios::app);
136  output_file.precision(15);
137 
138  for (auto iPoint = 0ul; iPoint < dataSorter->GetnPoints(); iPoint++) {
139 
140  /*--- Loop over the coordinates and write the values to file. ---*/
141 
142  for (auto iDim = 0u; iDim < dataSorter->GetnDim(); iDim++) {
143  output_file << scientific << dataSorter->GetData(iDim, iPoint) << "\t";
144  }
145 
146  /*--- Write global index. ---*/
147 
148  output_file << iPoint + offset << "\n";
149  myPoint++;
150  }
151 
152  output_file.close();
153  }
154 
155  /*--- Communicate offset, implies a barrier. ---*/
156  SU2_MPI::Allreduce(&myPoint, &offset, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
157  }
158 
159  if (rank == MASTER_NODE) {
160 
161  output_file.open(val_filename, ios::app);
162 
163  /*--- Read the boundary information ---*/
164 
165  string str = "boundary";
166  if (nZone > 1) str += "_" + PrintingToolbox::to_string(iZone);
167  str += ".dat";
168 
169  ifstream input_file;
170  input_file.open(str);
171 
172  if (!input_file.is_open()) {
173  SU2_MPI::Error(string("Cannot find ") + str, CURRENT_FUNCTION);
174  }
175 
176  /*--- Read grid file with format SU2 ---*/
177 
178  string text_line;
179  while (getline(input_file, text_line)) {
180 
181  /*--- Write the physical boundaries ---*/
182 
183  auto position = text_line.find("NMARK=",0);
184 
185  if (position == string::npos) continue;
186 
187  text_line.erase(0,6);
188  const auto nMarker_ = atoi(text_line.c_str());
189  output_file << "NMARK= " << nMarker_ << endl;
190 
191  for (auto iMarker = 0; iMarker < nMarker_; iMarker++) {
192 
193  getline(input_file, text_line);
194  text_line.erase(0,11);
195  for (int iChar = 0; iChar < 20; iChar++) {
196  position = text_line.find(" ", 0);
197  if (position != string::npos) text_line.erase(position,1);
198  position = text_line.find("\r", 0);
199  if (position != string::npos) text_line.erase(position,1);
200  position = text_line.find("\n", 0);
201  if (position != string::npos) text_line.erase(position,1);
202  }
203  string Marker_Tag = text_line;
204 
205  /*--- Standart physical boundary ---*/
206 
207  getline (input_file, text_line);
208 
209  text_line.erase(0,13);
210  const auto nElem_Bound_ = atoi(text_line.c_str());
211  output_file << "MARKER_TAG= " << Marker_Tag << endl;
212  output_file << "MARKER_ELEMS= " << nElem_Bound_<< endl;
213  getline (input_file, text_line);
214 
215  text_line.erase(0,8);
216  const auto SendTo = atoi(text_line.c_str());
217 
218  if (Marker_Tag == "SEND_RECEIVE") {
219  output_file << "SEND_TO= " << SendTo << endl;
220  }
221 
222  for (auto iElem_Bound = 0; iElem_Bound < nElem_Bound_; iElem_Bound++) {
223 
224  getline(input_file, text_line);
225  istringstream bound_line(text_line);
226 
227  unsigned short VTK_Type;
228  bound_line >> VTK_Type;
229  output_file << VTK_Type;
230  unsigned long vnodes[4] = {0};
231 
232  switch (VTK_Type) {
233  case LINE:
234  case VERTEX:
235  bound_line >> vnodes[0]; bound_line >> vnodes[1];
236  output_file << "\t" << vnodes[0] << "\t" << vnodes[1] << "\n";
237  break;
238  case TRIANGLE:
239  bound_line >> vnodes[0]; bound_line >> vnodes[1]; bound_line >> vnodes[2];
240  output_file << "\t" << vnodes[0] << "\t" << vnodes[1] << "\t" << vnodes[2] << "\n";
241  break;
242  case QUADRILATERAL:
243  bound_line >> vnodes[0]; bound_line >> vnodes[1]; bound_line >> vnodes[2]; bound_line >> vnodes[3];
244  output_file << "\t" << vnodes[0] << "\t" << vnodes[1] << "\t" << vnodes[2] << "\t" << vnodes[3] << "\n";
245  break;
246  }
247  }
248  }
249  }
250 
251  output_file.close();
252  }
253 
254  SU2_MPI::Barrier(SU2_MPI::GetComm());
255 }
