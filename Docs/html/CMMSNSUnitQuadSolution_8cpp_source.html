129 void CMMSNSUnitQuadSolution::GetSolution(const su2double *val_coords,
130  const su2double val_t,
131  su2double *val_solution) const {
132 
133  /* Easier storage of the x- and y-coordinates. */
134  const su2double x = val_coords[0];
135  const su2double y = val_coords[1];
136 
137  /* Determine the solution for the density, velocity
138  components and pressure. */
139  const su2double LInv = 1.0/L;
140  const su2double PiLInv = PI_NUMBER*LInv;
141  const su2double PiL2Inv = PiLInv*LInv;
142 
143  const su2double rho = rho_0 + rho_x *sin(a_rhox *PiLInv*x)
144  + rho_y *cos(a_rhoy *PiLInv*y)
145  + rho_xy*cos(a_rhoxy*PiL2Inv*x*y);
146 
147  const su2double u = u_0 + u_x *sin(a_ux *PiLInv*x)
148  + u_y *cos(a_uy *PiLInv*y)
149  + u_xy*cos(a_uxy*PiL2Inv*x*y);
150 
151  const su2double v = v_0 + v_x *cos(a_vx *PiLInv*x)
152  + v_y *sin(a_vy *PiLInv*y)
153  + v_xy*cos(a_vxy*PiL2Inv*x*y);
154 
155  const su2double p = P_0 + P_x *cos(a_Px *PiLInv*x)
156  + P_y *sin(a_Py *PiLInv*y)
157  + P_xy*sin(a_Pxy*PiL2Inv*x*y);
158 
159  /* Compute the conservative variables from the primitive ones.
160  Note that the implementation below is valid for both 2D and 3D. */
161  val_solution[0] = rho;
162  val_solution[1] = rho*u;
163  val_solution[2] = rho*v;
164  val_solution[3] = 0.0;
165  val_solution[nDim+1] = p/(Gamma-1.0) + 0.5*rho*(u*u + v*v);
166 }
