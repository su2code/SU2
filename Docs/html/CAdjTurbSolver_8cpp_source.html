33 CAdjTurbSolver::CAdjTurbSolver(CGeometry *geometry, CConfig *config, unsigned short iMesh) : CSolver() {
34 
35  adjoint = true;
36 
37  nDim = geometry->GetnDim();

74  Jacobian_jj = new su2double* [nVar];

81 
82  /*--- Initialization of the structure of the whole Jacobian ---*/
83  Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config);
84  Jacobian.SetValZero();
85  LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
86  LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
87 
88  /*--- Far-Field values and initizalization ---*/
89  bool restart = config->GetRestart();
90 
91  nodes = new CAdjTurbVariable(0.0, nPoint, nDim, nVar, config);
92  SetBaseClassPointerToNodes();
93 
94  if (restart && (iMesh == MESH_0)) {
95  unsigned long index;
96  su2double dull_val;
97  string filename, AdjExt, text_line;
98  ifstream restart_file;
99 
100  /*--- Restart the solution from file information ---*/
101  string mesh_filename = config->GetSolution_AdjFileName();
102  filename = config->GetObjFunc_Extension(mesh_filename);
103 
104  restart_file.open(filename.data(), ios::in);
105 
106  /*--- In case there is no file ---*/
107  if (restart_file.fail()) {
108  SU2_MPI::Error(string("There is no adjoint restart file ") + filename, CURRENT_FUNCTION);
109  }
110 
111  /*--- Read all lines in the restart file ---*/
112  long iPoint_Local; unsigned long iPoint_Global = 0; unsigned long iPoint_Global_Local = 0;
113  unsigned short rbuf_NotMatching = 0, sbuf_NotMatching = 0;
114 
115  /*--- The first line is the header ---*/
116 
117  getline (restart_file, text_line);
118 
119  for (iPoint_Global = 0; iPoint_Global < geometry->GetGlobal_nPointDomain(); iPoint_Global++ ) {
120 
121  getline (restart_file, text_line);
122 
123  istringstream point_line(text_line);
124 
125  /*--- Retrieve local index. If this node from the restart file lives
126  on the current processor, we will load and instantiate the vars. ---*/
127 
128  iPoint_Local = geometry->GetGlobal_to_Local_Point(iPoint_Global);
129 
130  if (iPoint_Local > -1) {
131 
132  if (nDim == 2) point_line >> index >> dull_val >> dull_val >> dull_val >> dull_val >> dull_val >> dull_val >> Solution[0];
133  if (nDim == 3) point_line >> index >> dull_val >> dull_val >> dull_val >> dull_val >> dull_val >> dull_val >> dull_val >> dull_val >> Solution[0];
134  nodes->SetSolution(iPoint_Local,0,Solution[0]);
135  nodes->SetSolution_Old(iPoint_Local,0,Solution[0]);
136  iPoint_Global_Local++;
137  }
138 
139  }
140 
141  /*--- Detect a wrong solution file ---*/
142 
143  if (iPoint_Global_Local < nPointDomain) { sbuf_NotMatching = 1; }
144 
145 #ifndef HAVE_MPI
146  rbuf_NotMatching = sbuf_NotMatching;
147 #else
148  SU2_MPI::Allreduce(&sbuf_NotMatching, &rbuf_NotMatching, 1, MPI_UNSIGNED_SHORT, MPI_SUM, SU2_MPI::GetComm());
149 #endif
150  if (rbuf_NotMatching != 0) {
151  SU2_MPI::Error(string("The solution file ") + filename + string(" doesn't match with the mesh file!\n") +
152  string("It could be empty lines at the end of the file."), CURRENT_FUNCTION);
153  }
154 
155  /*--- Close the restart file ---*/
156  restart_file.close();
157  }
158 
159  /*--- MPI solution ---*/
160 
161  InitiateComms(geometry, config, SOLUTION);
162  CompleteComms(geometry, config, SOLUTION);
163

273 
274  unsigned long iEdge, iPoint, jPoint;

277  for (iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
278 
279  /*--- Points in edge ---*/
280  iPoint = geometry->edges->GetNode(iEdge,0);
281  jPoint = geometry->edges->GetNode(iEdge,1);
282 
283  /*--- Conservative variables w/o reconstruction ---*/
284  U_i = solver_container[FLOW_SOL]->GetNodes()->GetSolution(iPoint);
285  U_j = solver_container[FLOW_SOL]->GetNodes()->GetSolution(jPoint);
286  numerics->SetConservative(U_i, U_j);
287 
288  /*--- Set normal vectors and length ---*/
289  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
290 
291  /*--- Turbulent adjoint variables w/o reconstruction ---*/
292  TurbPsi_i = nodes->GetSolution(iPoint);
293  TurbPsi_j = nodes->GetSolution(jPoint);
294  numerics->SetTurbAdjointVar(TurbPsi_i, TurbPsi_j);
295 
296  /*--- Gradient of turbulent variables w/o reconstruction ---*/
297  const auto TurbVar_Grad_i = solver_container[TURB_SOL]->GetNodes()->GetGradient(iPoint);
298  const auto TurbVar_Grad_j = solver_container[TURB_SOL]->GetNodes()->GetGradient(jPoint);
299  numerics->SetScalarVarGradient(TurbVar_Grad_i, TurbVar_Grad_j);
300 
301  /*--- Set normal vectors and length ---*/
302  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
303 
304  numerics->ComputeResidual(Residual_i, Residual_j, Jacobian_ii, Jacobian_ij, Jacobian_ji, Jacobian_jj, config);
305 
306  /*--- Add and Subtract Residual ---*/
307  LinSysRes.AddBlock(iPoint, Residual_i);
308  LinSysRes.AddBlock(jPoint, Residual_j);
309  Jacobian.AddBlock2Diag(iPoint, Jacobian_ii);
310  Jacobian.AddBlock(iPoint, jPoint, Jacobian_ij);
311  Jacobian.AddBlock(jPoint, iPoint, Jacobian_ji);
312  Jacobian.AddBlock2Diag(jPoint, Jacobian_jj);

318 void CAdjTurbSolver::Viscous_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
319  CConfig *config, unsigned short iMesh, unsigned short iRKStep) {
320 
321  CNumerics* numerics = numerics_container[VISC_TERM];
322 
323  unsigned long iEdge, iPoint, jPoint;
324  su2double *Coord_i, *Coord_j;
325 
326  for (iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
327 
328  /*--- Points in edge ---*/
329  iPoint = geometry->edges->GetNode(iEdge,0);
330  jPoint = geometry->edges->GetNode(iEdge,1);
331 
332  /*--- Points coordinates, and set normal vectors and length ---*/
333  Coord_i = geometry->nodes->GetCoord(iPoint);
334  Coord_j = geometry->nodes->GetCoord(jPoint);
335  numerics->SetCoord(Coord_i, Coord_j);
336  numerics->SetNormal(geometry->edges->GetNormal(iEdge));

338  /*--- Conservative variables w/o reconstruction, turbulent variables w/o reconstruction,

342  numerics->SetTurbAdjointVar(nodes->GetSolution(iPoint), nodes->GetSolution(jPoint));
343 
344  /*--- Viscosity ---*/
345  numerics->SetLaminarViscosity(solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint),
346  solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(jPoint));
347 
348  /*--- Turbulent adjoint variables w/o reconstruction ---*/
349  numerics->SetTurbAdjointGradient(nodes->GetGradient(iPoint), nodes->GetGradient(jPoint));
350 
351  /*--- Compute residual in a non-conservative way, and update ---*/
352  numerics->ComputeResidual(Residual_i, Residual_j, Jacobian_ii, Jacobian_ij, Jacobian_ji, Jacobian_jj, config);
353 
354  /*--- Update adjoint viscous residual ---*/
355  LinSysRes.AddBlock(iPoint, Residual_i);
356  LinSysRes.AddBlock(jPoint, Residual_j);
357 
358  Jacobian.AddBlock2Diag(iPoint, Jacobian_ii);
359  Jacobian.AddBlock(iPoint, jPoint, Jacobian_ij);
360  Jacobian.AddBlock(jPoint, iPoint, Jacobian_ji);
361  Jacobian.AddBlock2Diag(jPoint, Jacobian_jj);
362

455 void CAdjTurbSolver::ImplicitEuler_Iteration(CGeometry *geometry, CSolver **solver_container, CConfig *config) {
456 
457  unsigned short iVar;
458  unsigned long iPoint, total_index;
459  su2double Delta, Vol;
460 
461  /*--- Set maximum residual to zero ---*/
462 
463  SetResToZero();
464 
465  /*--- Build implicit system ---*/
466 
467  for (iPoint = 0; iPoint < nPoint; iPoint++) {
468 
469  /*--- Read the volume ---*/
470 
471  Vol = geometry->nodes->GetVolume(iPoint);
472 
473  /*--- Modify matrix diagonal to assure diagonal dominance ---*/
474 
475  Delta = Vol / (config->GetCFLRedCoeff_AdjTurb()*solver_container[FLOW_SOL]->GetNodes()->GetDelta_Time(iPoint));
476 
477  Jacobian.AddVal2Diag(iPoint, Delta);
478 
479  /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
480 
481  for (iVar = 0; iVar < nVar; iVar++) {
482  total_index = iPoint*nVar+iVar;
483  LinSysRes[total_index] = -LinSysRes[total_index];
484  LinSysSol[total_index] = 0.0;
485  Residual_RMS[iVar] += LinSysRes[total_index]*LinSysRes[total_index];
486  AddRes_Max(iVar, fabs(LinSysRes[total_index]), geometry->nodes->GetGlobalIndex(iPoint), geometry->nodes->GetCoord(iPoint));
487  }
488 
489  }
490 
491  /*--- Initialize residual and solution at the ghost points ---*/
492 
493  for (iPoint = nPointDomain; iPoint < nPoint; iPoint++) {
494  for (iVar = 0; iVar < nVar; iVar++) {
495  total_index = iPoint*nVar + iVar;
496  LinSysRes[total_index] = 0.0;
497  LinSysSol[total_index] = 0.0;
498  }
499  }
500 
501  /*--- Solve or smooth the linear system ---*/
502 
503  System.Solve(Jacobian, LinSysRes, LinSysSol, geometry, config);
504 
505  /*--- Update solution (system written in terms of increments) ---*/
506 
507  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
508  for (iVar = 0; iVar < nVar; iVar++)
509  nodes->AddSolution(iPoint,iVar, LinSysSol[iPoint*nVar+iVar]);
510  }
511 
512  /*--- MPI solution ---*/
513 
514  InitiateComms(geometry, config, SOLUTION);
515  CompleteComms(geometry, config, SOLUTION);
516 
517  /*--- Compute the root mean square residual ---*/
518 
519  SetResidual_RMS(geometry, config);
520 
521 }
