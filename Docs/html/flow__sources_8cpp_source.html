61 CNumerics::ResidualType<> CSourceAxisymmetric_Flow::ComputeResidual(const CConfig* config) {
62 
63  su2double Pressure_i, Enthalpy_i, Velocity_i, sq_vel;
64  unsigned short iDim, iVar, jVar;
65 
66  if (Coord_i[1] > EPS) {
67 
68  yinv = 1.0/Coord_i[1];
69 
70  sq_vel = 0.0;
71  for (iDim = 0; iDim < nDim; iDim++) {
72  Velocity_i = U_i[iDim+1] / U_i[0];
73  sq_vel += Velocity_i *Velocity_i;
74  }
75 
76  Pressure_i = (Gamma-1.0)*U_i[0]*(U_i[nDim+1]/U_i[0]-0.5*sq_vel);
77  Enthalpy_i = (U_i[nDim+1] + Pressure_i) / U_i[0];
78 
79  residual[0] = yinv*Volume*U_i[2];
80  residual[1] = yinv*Volume*U_i[1]*U_i[2]/U_i[0];
81  residual[2] = yinv*Volume*(U_i[2]*U_i[2]/U_i[0]);
82  residual[3] = yinv*Volume*Enthalpy_i*U_i[2];
83 
84  /*--- Inviscid component of the source term. ---*/
85 
86  if (implicit) {
87  jacobian[0][0] = 0.0;
88  jacobian[0][1] = 0.0;
89  jacobian[0][2] = 1.0;
90  jacobian[0][3] = 0.0;
91 
92  jacobian[1][0] = -U_i[1]*U_i[2]/(U_i[0]*U_i[0]);
93  jacobian[1][1] = U_i[2]/U_i[0];
94  jacobian[1][2] = U_i[1]/U_i[0];
95  jacobian[1][3] = 0.0;
96 
97  jacobian[2][0] = -U_i[2]*U_i[2]/(U_i[0]*U_i[0]);
98  jacobian[2][1] = 0.0;
99  jacobian[2][2] = 2*U_i[2]/U_i[0];
100  jacobian[2][3] = 0.0;
101 
102  jacobian[3][0] = -Gamma*U_i[2]*U_i[3]/(U_i[0]*U_i[0]) + (Gamma-1)*U_i[2]*(U_i[1]*U_i[1]+U_i[2]*U_i[2])/(U_i[0]*U_i[0]*U_i[0]);
103  jacobian[3][1] = -(Gamma-1)*U_i[2]*U_i[1]/(U_i[0]*U_i[0]);
104  jacobian[3][2] = Gamma*U_i[3]/U_i[0] - 1/2*(Gamma-1)*( (U_i[1]*U_i[1]+U_i[2]*U_i[2])/(U_i[0]*U_i[0]) + 2*U_i[2]*U_i[2]/(U_i[0]*U_i[0]) );
105  jacobian[3][3] = Gamma*U_i[2]/U_i[0];
106 
107  for (iVar=0; iVar < nVar; iVar++)
108  for (jVar=0; jVar < nVar; jVar++)
109  jacobian[iVar][jVar] *= yinv*Volume;
110 
111  }
112 
113  /*--- Add the viscous terms if necessary. ---*/
114 
115  if (viscous) ResidualDiffusion();
116 
117  }
118 
119  else {
120 
121  for (iVar=0; iVar < nVar; iVar++)
122  residual[iVar] = 0.0;
123 
124  if (implicit) {
125  for (iVar=0; iVar < nVar; iVar++) {
126  for (jVar=0; jVar < nVar; jVar++)
127  jacobian[iVar][jVar] = 0.0;
128  }
129  }
130 
131  }
132 
133  return ResidualType<>(residual, jacobian, nullptr);
134 }

136 void CSourceAxisymmetric_Flow::ResidualDiffusion(){
137 
138  if (!rans){ turb_ke_i = 0.0; }
139 
140  su2double laminar_viscosity_i = V_i[nDim+5];
141  su2double eddy_viscosity_i = V_i[nDim+6];
142  su2double thermal_conductivity_i = V_i[nDim+7];
143  su2double heat_capacity_cp_i = V_i[nDim+8];
144 
145  su2double total_viscosity_i = laminar_viscosity_i + eddy_viscosity_i;
146  su2double total_conductivity_i = thermal_conductivity_i + heat_capacity_cp_i*eddy_viscosity_i/Prandtl_Turb;
147 
148  su2double u = U_i[1]/U_i[0];
149  su2double v = U_i[2]/U_i[0];
150 
151  residual[0] -= 0.0;
152  residual[1] -= Volume*(yinv*total_viscosity_i*(PrimVar_Grad_i[1][1]+PrimVar_Grad_i[2][0])
153  -TWO3*AuxVar_Grad_i[0][0]);
154  residual[2] -= Volume*(yinv*total_viscosity_i*2*(PrimVar_Grad_i[2][1]-v*yinv)
155  -TWO3*AuxVar_Grad_i[0][1]);
156  residual[3] -= Volume*(yinv*(total_viscosity_i*(u*(PrimVar_Grad_i[2][0]+PrimVar_Grad_i[1][1])
157  +v*TWO3*(2*PrimVar_Grad_i[2][1]-PrimVar_Grad_i[1][0]
158  -v*yinv+U_i[0]*turb_ke_i))
159  +total_conductivity_i*PrimVar_Grad_i[0][1])
160  -TWO3*(AuxVar_Grad_i[1][1]+AuxVar_Grad_i[2][0]));
161 }

164 CNumerics::ResidualType<> CSourceGeneralAxisymmetric_Flow::ComputeResidual(const CConfig* config) {
165  unsigned short iVar, jVar;
166 
167  if (Coord_i[1] > EPS) {
168 
169  yinv = 1.0/Coord_i[1];
170 
171  su2double Density_i = U_i[0];
172  su2double Velocity1_i = U_i[1]/U_i[0];
173  su2double Velocity2_i = U_i[2]/U_i[0];
174  su2double Energy_i = U_i[3]/U_i[0];
175 
176  su2double Pressure_i = V_j[3];
177  su2double Enthalpy_i = Energy_i + Pressure_i/Density_i;
178 
179  /*--- Inviscid component of the source term. ---*/
180 
181  residual[0] = yinv*Volume*U_i[2];
182  residual[1] = yinv*Volume*U_i[1]*Velocity2_i;
183  residual[2] = yinv*Volume*U_i[2]*Velocity2_i;
184  residual[3] = yinv*Volume*U_i[2]*Enthalpy_i;
185 
186  if (implicit) {
187 
188  su2double dPdrho_e_i = S_i[0];
189  su2double dPde_rho_i = S_i[1];
190 
191  jacobian[0][0] = 0.0;
192  jacobian[0][1] = 0.0;
193  jacobian[0][2] = 1.0;
194  jacobian[0][3] = 0.0;
195 
196  jacobian[1][0] = -Velocity1_i*Velocity2_i;
197  jacobian[1][1] = Velocity2_i;
198  jacobian[1][2] = Velocity1_i;
199  jacobian[1][3] = 0.0;
200 
201  jacobian[2][0] = -Velocity2_i*Velocity2_i;
202  jacobian[2][1] = 0.0;
203  jacobian[2][2] = 2*Velocity2_i;
204  jacobian[2][3] = 0.0;
205 
206  jacobian[3][0] = Velocity2_i*(dPdrho_e_i + dPde_rho_i/Density_i*(Velocity1_i*Velocity1_i
207  + Velocity2_i*Velocity2_i
208  - Energy_i) - Enthalpy_i);
209  jacobian[3][1] = -Velocity1_i*Velocity2_i/Density_i *dPde_rho_i;
210  jacobian[3][2] = Enthalpy_i - Velocity2_i*Velocity2_i/Density_i *dPde_rho_i;
211  jacobian[3][3] = Velocity2_i + Velocity2_i/Density_i *dPde_rho_i;
212 
213  for (iVar=0; iVar < nVar; iVar++)
214  for (jVar=0; jVar < nVar; jVar++)
215  jacobian[iVar][jVar] *= yinv*Volume;
216 
217  }
218 
219  /*--- Add the viscous terms if necessary. ---*/
220 
221  if (viscous) ResidualDiffusion();
222 
223  }
224 
225  else {
226 
227  for (iVar=0; iVar < nVar; iVar++)
228  residual[iVar] = 0.0;
229 
230  if (implicit) {
231  for (iVar=0; iVar < nVar; iVar++) {
232  for (jVar=0; jVar < nVar; jVar++)
233  jacobian[iVar][jVar] = 0.0;
234  }
235  }
236 
237  }
238 
239  return ResidualType<>(residual, jacobian, nullptr);
240 }

251 CNumerics::ResidualType<> CSourceIncAxisymmetric_Flow::ComputeResidual(const CConfig* config) {
252 
253  su2double yinv, Velocity_i[3];
254  unsigned short iDim, iVar, jVar;
255 
256  if (Coord_i[1] > EPS) {
257 
258  yinv = 1.0/Coord_i[1];
259 
260  /*--- Set primitive variables at points iPoint. ---*/
261 
262  Pressure_i = V_i[0];
263  Temp_i = V_i[nDim+1];
264  DensityInc_i = V_i[nDim+2];
265  BetaInc2_i = V_i[nDim+3];
266  Cp_i = V_i[nDim+7];
267  Enthalpy_i = Cp_i*Temp_i;
268 
269  for (iDim = 0; iDim < nDim; iDim++)
270  Velocity_i[iDim] = V_i[iDim+1];
271 
272  /*--- Inviscid component of the source term. ---*/
273 
274  residual[0] = yinv*Volume*DensityInc_i*Velocity_i[1];
275  residual[1] = yinv*Volume*DensityInc_i*Velocity_i[0]*Velocity_i[1];
276  residual[2] = yinv*Volume*DensityInc_i*Velocity_i[1]*Velocity_i[1];
277  residual[3] = yinv*Volume*DensityInc_i*Enthalpy_i*Velocity_i[1];
278 
279  if (implicit) {
280 
281  jacobian[0][0] = 0.0;
282  jacobian[0][1] = 0.0;
283  jacobian[0][2] = 1.0;
284  jacobian[0][3] = 0.0;
285 
286  jacobian[1][0] = 0.0;
287  jacobian[1][1] = Velocity_i[1];
288  jacobian[1][2] = Velocity_i[0];
289  jacobian[1][3] = 0.0;
290 
291  jacobian[2][0] = 0.0;
292  jacobian[2][1] = 0.0;
293  jacobian[2][2] = 2.0*Velocity_i[1];
294  jacobian[2][3] = 0.0;
295 
296  jacobian[3][0] = 0.0;
297  jacobian[3][1] = 0.0;
298  jacobian[3][2] = Enthalpy_i;
299  jacobian[3][3] = Cp_i*Velocity_i[1];
300 
301  for (iVar=0; iVar < nVar; iVar++)
302  for (jVar=0; jVar < nVar; jVar++)
303  jacobian[iVar][jVar] *= yinv*Volume*DensityInc_i;
304 
305  }
306 
307  /*--- Add the viscous terms if necessary. ---*/
308 
309  if (viscous) {
310 
311  Laminar_Viscosity_i = V_i[nDim+4];
312  Eddy_Viscosity_i = V_i[nDim+5];
313  Thermal_Conductivity_i = V_i[nDim+6];
314 
315  su2double total_viscosity;
316 
317  total_viscosity = (Laminar_Viscosity_i + Eddy_Viscosity_i);
318 
319  /*--- The full stress tensor is needed for variable density ---*/
320  ComputeStressTensor(nDim, tau, PrimVar_Grad_i+1, total_viscosity);
321 
322  /*--- Viscous terms. ---*/
323 
324  residual[0] -= 0.0;
325  residual[1] -= Volume*(yinv*tau[0][1] - TWO3*AuxVar_Grad_i[0][0]);
326  residual[2] -= Volume*(yinv*2.0*total_viscosity*PrimVar_Grad_i[2][1] -
327  yinv* yinv*2.0*total_viscosity*Velocity_i[1] -
328  TWO3*AuxVar_Grad_i[0][1]);
329  residual[3] -= Volume*yinv*Thermal_Conductivity_i*PrimVar_Grad_i[nDim+1][1];
330 
331  }
332 
333  } else {
334 
335  for (iVar=0; iVar < nVar; iVar++)
336  residual[iVar] = 0.0;
337 
338  if (implicit) {
339  for (iVar=0; iVar < nVar; iVar++) {
340  for (jVar=0; jVar < nVar; jVar++)
341  jacobian[iVar][jVar] = 0.0;
342  }
343  }
344 
345  }
346 
347  if (!energy) {
348  residual[nDim+1] = 0.0;
349  if (implicit) {
350  for (iVar = 0; iVar < nVar; iVar++) {
351  jacobian[iVar][nDim+1] = 0.0;
352  jacobian[nDim+1][iVar] = 0.0;
353  }
354  }
355  }
356 
357  return ResidualType<>(residual, jacobian, nullptr);
358 }

370 CNumerics::ResidualType<> CSourceBodyForce::ComputeResidual(const CConfig* config) {
371 
372  unsigned short iDim;
373  su2double Force_Ref = config->GetForce_Ref();
374 
375  /*--- Zero the continuity contribution ---*/
376 
377  residual[0] = 0.0;
378 
379  /*--- Momentum contribution ---*/
380 
381  for (iDim = 0; iDim < nDim; iDim++)
382  residual[iDim+1] = -Volume * U_i[0] * Body_Force_Vector[iDim] / Force_Ref;
383 
384  /*--- Energy contribution ---*/
385 
386  residual[nDim+1] = 0.0;
387  for (iDim = 0; iDim < nDim; iDim++)
388  residual[nDim+1] += -Volume * U_i[iDim+1] * Body_Force_Vector[iDim] / Force_Ref;
389 
390  return ResidualType<>(residual, jacobian, nullptr);

403 CNumerics::ResidualType<> CSourceIncBodyForce::ComputeResidual(const CConfig* config) {
404 
405  unsigned short iDim;
406  su2double DensityInc_0 = 0.0;
407  su2double Force_Ref = config->GetForce_Ref();
408  bool variable_density = (config->GetKind_DensityModel() == INC_DENSITYMODEL::VARIABLE);
409 
410  /*--- Check for variable density. If we have a variable density
411  problem, we should subtract out the hydrostatic pressure component. ---*/
412 
413  if (variable_density) DensityInc_0 = config->GetDensity_FreeStreamND();
414 
415  /*--- Zero the continuity contribution ---*/
416 
417  residual[0] = 0.0;
418 
419  /*--- Momentum contribution. Note that this form assumes we have
420  subtracted the operating density * gravity, i.e., removed the
421  hydrostatic pressure component (important for pressure BCs). ---*/
422 
423  for (iDim = 0; iDim < nDim; iDim++)
424  residual[iDim+1] = -Volume * (DensityInc_i - DensityInc_0) * Body_Force_Vector[iDim] / Force_Ref;
425 
426  /*--- Zero the temperature contribution ---*/
427 
428  residual[nDim+1] = 0.0;
429 
430  return ResidualType<>(residual, jacobian, nullptr);
431 }

447 CNumerics::ResidualType<> CSourceBoussinesq::ComputeResidual(const CConfig* config) {
448 
449  unsigned short iDim;
450  su2double Force_Ref = config->GetForce_Ref();
451  su2double T0 = config->GetTemperature_FreeStreamND();
452  su2double Beta = config->GetThermal_Expansion_CoeffND();
453 
454  /*--- Zero the continuity contribution ---*/
455 
456  residual[0] = 0.0;
457 
458  /*--- Momentum contribution. Note that this form assumes we have
459  subtracted the operating density * gravity, i.e., removed the
460  hydrostatic pressure component (important for pressure BCs). ---*/
461 
462  for (iDim = 0; iDim < nDim; iDim++)
463  residual[iDim+1] = Volume * DensityInc_i * ( Beta * (U_i[nDim+1] - T0)) * Gravity_Vector[iDim] / Force_Ref;
464 
465  /*--- Zero the energy contribution ---*/
466 
467  residual[nDim+1] = 0.0;
468 
469  return ResidualType<>(residual, jacobian, nullptr);

498 CNumerics::ResidualType<> CSourceRotatingFrame_Flow::ComputeResidual(const CConfig* config) {
499 
500  unsigned short iDim, iVar, jVar;
501  su2double Omega[MAXNDIM] = {0}, Momentum[MAXNDIM] = {0};
502 
503  bool implicit = (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT);
504 
505  /*--- Retrieve the angular velocity vector from config. ---*/
506 
507  for (iDim = 0; iDim < 3; iDim++){
508  Omega[iDim] = config->GetRotation_Rate(iDim)/config->GetOmega_Ref();
509  }
510 
511  /*--- Get the momentum vector at the current node. ---*/
512 
513  for (iDim = 0; iDim < nDim; iDim++)
514  Momentum[iDim] = U_i[iDim+1];
515 
516  /*--- Calculate rotating frame source term as ( Omega X Rho-U ) ---*/
517 
518  if (nDim == 2) {
519  residual[0] = 0.0;
520  residual[1] = (Omega[1]*Momentum[2] - Omega[2]*Momentum[1])*Volume;
521  residual[2] = (Omega[2]*Momentum[0] - Omega[0]*Momentum[2])*Volume;
522  residual[3] = 0.0;
523  } else {
524  residual[0] = 0.0;
525  residual[1] = (Omega[1]*Momentum[2] - Omega[2]*Momentum[1])*Volume;
526  residual[2] = (Omega[2]*Momentum[0] - Omega[0]*Momentum[2])*Volume;
527  residual[3] = (Omega[0]*Momentum[1] - Omega[1]*Momentum[0])*Volume;
528  residual[4] = 0.0;
529  }
530 
531  /*--- Calculate the source term Jacobian ---*/
532 
533  if (implicit) {
534  for (iVar = 0; iVar < nVar; iVar++)
535  for (jVar = 0; jVar < nVar; jVar++)
536  jacobian[iVar][jVar] = 0.0;
537  if (nDim == 2) {
538  jacobian[1][2] = -Omega[2]*Volume;
539  jacobian[2][1] = Omega[2]*Volume;
540  } else {
541  jacobian[1][2] = -Omega[2]*Volume;
542  jacobian[1][3] = Omega[1]*Volume;
543  jacobian[2][1] = Omega[2]*Volume;
544  jacobian[2][3] = -Omega[0]*Volume;
545  jacobian[3][1] = -Omega[1]*Volume;
546  jacobian[3][2] = Omega[0]*Volume;
547  }

567 CNumerics::ResidualType<> CSourceIncRotatingFrame_Flow::ComputeResidual(const CConfig* config) {
568 
569  unsigned short iDim, iVar, jVar;
570  su2double Momentum[MAXNDIM] = {0},
571  Velocity_i[MAXNDIM] = {0};
572 
573  /*--- Primitive variables plus momentum at the node (point i) ---*/
574 
575  DensityInc_i = V_i[nDim+2];
576 
577  for (iDim = 0; iDim < nDim; iDim++) {
578  Velocity_i[iDim] = V_i[iDim+1];
579  Momentum[iDim] = DensityInc_i*Velocity_i[iDim];
580  }

588  residual[3] = 0.0;
589  } else {
590  residual[0] = 0.0;
591  residual[1] = (Omega[1]*Momentum[2] - Omega[2]*Momentum[1])*Volume;
592  residual[2] = (Omega[2]*Momentum[0] - Omega[0]*Momentum[2])*Volume;
593  residual[3] = (Omega[0]*Momentum[1] - Omega[1]*Momentum[0])*Volume;
594  residual[4] = 0.0;
595  }
596 
597  /*--- Calculate the source term Jacobian ---*/
598 
599  if (implicit) {
600  for (iVar = 0; iVar < nVar; iVar++)
601  for (jVar = 0; jVar < nVar; jVar++)
602  jacobian[iVar][jVar] = 0.0;
603  if (nDim == 2) {
604  jacobian[1][2] = -DensityInc_i*Omega[2]*Volume;
605  jacobian[2][1] = DensityInc_i*Omega[2]*Volume;
606  } else {
607  jacobian[1][2] = -DensityInc_i*Omega[2]*Volume;
608  jacobian[1][3] = DensityInc_i*Omega[1]*Volume;
609  jacobian[2][1] = DensityInc_i*Omega[2]*Volume;
610  jacobian[2][3] = -DensityInc_i*Omega[0]*Volume;
611  jacobian[3][1] = -DensityInc_i*Omega[1]*Volume;
612  jacobian[3][2] = DensityInc_i*Omega[0]*Volume;
613  }
614  }
615

626 CNumerics::ResidualType<> CSourceVorticityConfinement::ComputeResidual(const CConfig* config) {
627  /*--- density, \rho ---*/
628  const su2double rho = V_i[nDim + 2];
629 
630  /*--- velocity, U = (u, v, w) ---*/
631  su2double U[3] = {V_i[1], V_i[2], 0.0};
632  if (nDim == 3) U[2] = V_i[3];
633  const su2double U_abs = max(GeometryToolbox::Norm(3, U), 1e-12);
634 
635  /*--- vorticity, \omega ---*/
636  const su2double omega[3] = {Vorticity_i[0], Vorticity_i[1], Vorticity_i[2]};
637 
638  /*--- grad of the mag of vorticity, \nabla|\omega| = AuxVar_Grad_i[0] ---*/
639  const su2double omega_abs_grad_abs = max(GeometryToolbox::Norm(3, AuxVar_Grad_i[0]), 1e-12);
640 
641  /*--- unit vector, n along \nabla|\omega| ---*/
642  const su2double n[3] = {
643  AuxVar_Grad_i[0][0] / omega_abs_grad_abs,
644  AuxVar_Grad_i[0][1] / omega_abs_grad_abs,
645  AuxVar_Grad_i[0][2] / omega_abs_grad_abs,
646  };
647 
648  /*--- n \cross \omega ---*/
649  su2double nXomega[3] = {0.0, 0.0, 0.0};
650  GeometryToolbox::CrossProduct(n, omega, nXomega);
651  const su2double nXomega_U = GeometryToolbox::DotProduct(3, nXomega, U);
652 
653  /*--- vorticity confinement parameter ---*/
654  const su2double vc_config = config->GetConfinement_Param();
655  su2double vc = vc_config * (1 + log10(pow((1 + (Volume / AvgVolume)), 1.0 / 3.0)));
656 
657  /*--- correction to vc near viscous wall ---*/
658  const bool viscous = config->GetViscous();
659  if (viscous) {
660  su2double U_infty[3] = {config->GetVelocity_FreeStreamND()[0], config->GetVelocity_FreeStreamND()[1], 0.0};
661  if (nDim == 3) U_infty[2] = config->GetVelocity_FreeStreamND()[2];
662  const su2double U_infty_abs = max(GeometryToolbox::Norm(3, U_infty), 1e-12);
663  const su2double L = config->GetLength_Reynolds();
664  const su2double mu = V_i[nDim + 5]; // viscosity
665  const su2double mu_t = V_i[nDim + 6]; // turbulent or eddy viscosity
666  const su2double mu_eff = mu + mu_t; // effective or total viscosity
667  const su2double Re = rho * U_infty_abs * L / mu_eff; // Reynolds number
668  const su2double delta = 5 * L / sqrt(Re); // boundary layer thickness
669 
670  if (dist_i <= 4 * delta) {
671  vc *= 0.0;
672  }
673  }
674 
675  /*--- source terms: S / rho ---*/
676  const su2double vc_Uabs_Volume = vc * U_abs * Volume;
677  const su2double S_u_rho = -nXomega[0] * vc_Uabs_Volume;
678  const su2double S_v_rho = -nXomega[1] * vc_Uabs_Volume;
679  const su2double S_w_rho = -nXomega[2] * vc_Uabs_Volume;
680  const su2double S_E_rho = -nXomega_U * vc_Uabs_Volume;
681 
682  residual[0] = 0.0;
683  residual[1] = rho * S_u_rho;
684  residual[2] = rho * S_v_rho;
685  if (nDim == 2) {
686  residual[3] = rho * S_E_rho;
687  }
688  if (nDim == 3) {
689  residual[3] = rho * S_w_rho;
690  residual[4] = rho * S_E_rho;
691  }
692 
693  const bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
694  if (implicit) {
695  for (unsigned iVar = 0; iVar < nVar; iVar++) {
696  for (unsigned jVar = 0; jVar < nVar; jVar++) {
697  jacobian[iVar][jVar] = 0.0;
698  }
699  }
700 
701  jacobian[1][0] = S_u_rho;
702  jacobian[2][0] = S_v_rho;
703  if (nDim == 2) {
704  jacobian[3][1] = S_u_rho;
705  jacobian[3][2] = S_v_rho;
706  }
707  if (nDim == 3) {
708  jacobian[3][0] = S_w_rho;
709  jacobian[4][1] = S_u_rho;
710  jacobian[4][2] = S_v_rho;
711  jacobian[4][3] = S_w_rho;
712  }
713  }
714 
715  return ResidualType<>(residual, jacobian, nullptr);
716 }

816

819 
820  for (unsigned short iVar = 0; iVar < nVar; iVar++) residual[iVar] = 0.0;
821

834  dot_product = GeometryToolbox::DotProduct(nDim, Streamwise_Coord_Vector, &V_i[1]);
835 
836  residual[nDim+1] = Volume * scalar_factor * dot_product;
837

839  if(turbulent) {
840 
841  /*--- Compute a scalar factor ---*/
842  scalar_factor = SPvals.Streamwise_Periodic_IntegratedHeatFlow / (SPvals.Streamwise_Periodic_MassFlow * sqrt(norm2_translation) * Prandtl_Turb);
843 
844  /*--- Compute scalar product between periodic translation vector and eddy viscosity gradient. ---*/
845  dot_product = GeometryToolbox::DotProduct(nDim, Streamwise_Coord_Vector, AuxVar_Grad_i[0]);
846 
847  residual[nDim+1] -= Volume * scalar_factor * dot_product;
848  } // if turbulent
849  } // if energy
