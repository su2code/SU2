35 template<class ScalarType>
36 FORCEINLINE ScalarType *CSysMatrix<ScalarType>::GetBlock_ILUMatrix(unsigned long block_i, unsigned long block_j) {
37  /*--- The position of the diagonal block is known which allows halving the search space. ---*/
38  const auto end = (block_j<block_i)? dia_ptr_ilu[block_i] : row_ptr_ilu[block_i+1];
39  for (auto index = (block_j<block_i)? row_ptr_ilu[block_i] : dia_ptr_ilu[block_i]; index < end; ++index)
40  if (col_ind_ilu[index] == block_j)
41  return &ILU_matrix[index*nVar*nVar];
42  return nullptr;
43 }

55 template<class T, bool alpha, bool beta, bool transp>
56 FORCEINLINE void gemv_impl(unsigned long n, unsigned long m, const T *a, const T *b, T *c) {
57  /*---
58  This is a templated version of GEMV with the constants as boolean
59  template parameters so that they can be optimized away at compilation.
60  This is still the traditional "row dot vector" method.
61  ---*/
62  if (!transp) {
63  for (auto i = 0ul; i < n; i++) {
64  if (!beta) c[i] = 0.0;
65  for (auto j = 0ul; j < m; j++)
66  c[i] += (alpha? 1 : -1) * a[i*m+j] * b[j];
67  }
68  } else {
69  if (!beta) for (auto j = 0ul; j < m; j++) c[j] = 0.0;
70  for (auto i = 0ul; i < n; i++)
71  for (auto j = 0ul; j < m; j++)
72  c[j] += (alpha? 1 : -1) * a[i*n+j] * b[i];
73  }
74 }

76 template<class T>
77 FORCEINLINE void gemm_impl(unsigned long n, const T *a, const T *b, T *c) {
78  /*--- Same deal as for GEMV but here only the type is templated. ---*/
79  unsigned long i, j, k;
80  for (i = 0; i < n; i++) {
81  for (j = 0; j < n; j++) {
82  c[i*n+j] = 0.0;
83  for (k = 0; k < n; k++)
84  c[i*n+j] += a[i*n+k] * b[k*n+j];
85  }
86  }
87 }
