32 CMMSNSTwoHalfCirclesSolution::CMMSNSTwoHalfCirclesSolution(unsigned short val_nDim,
33  unsigned short val_nVar,
34  unsigned short val_iMesh,
35  CConfig* config)
36  : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
37 
38  /*--- Write a message that the solution is initialized for the manufactured
39  solution for the Navier-Stokes equations between two half circles. ---*/
40  if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
41  cout << endl;
42  cout << "Warning: Fluid properties and solution are being " << endl;
43  cout << " initialized for the manufactured solution " << endl;
44  cout << " of the Navier-Stokes equations between two half circles!!!" << endl;
45  cout << endl << flush;
46  }
47 
48  /*--- Coefficients, needed to determine the solution. ---*/
49  const su2double Prandtl = config->GetPrandtl_Lam();
50 
51  RGas = config->GetGas_Constant();
52  Gamma = config->GetGamma();
53  Viscosity = config->GetMu_Constant();
54  Conductivity = Viscosity*Gamma*RGas/(Prandtl*(Gamma-1.0));
55 
56  /*--- Initialize TWall to the default value of 300 K (in case the outer wall
57  is not modelled as an isothermal wall) and try to retrieve the wall
58  temperature from the boundary conditions. ---*/
59  TWall = 300.0;
60  for(unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
61  if(config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL) {
62  const string Marker_Tag = config->GetMarker_All_TagBound(iMarker);
63  TWall = config->GetIsothermal_Temperature(Marker_Tag);
64  }
65  }
66 
67  /*--- Get the reference values for pressure, density and velocity. ---*/
68  Pressure_Ref = config->GetPressure_Ref();
69  Density_Ref = config->GetDensity_Ref();
70  Velocity_Ref = config->GetVelocity_Ref();
71 
72  /*--- The constants for the density and velocities. ---*/
73  rho_0 = 1.25;
74  u_0 = 135.78;
75  v_0 = -67.61;
76 
77  /*--- The constants for the temperature solution. ---*/
78  a_T1 = 1.05;
79  a_T2 = -0.85;
80 
81  /*--- Perform some sanity and error checks for this solution here. ---*/
82  if(config->GetTime_Marching() != TIME_MARCHING::STEADY)
83  SU2_MPI::Error("Steady mode must be selected for the MMS NS Two Half Circles case",
84  CURRENT_FUNCTION);
85 
86  if(Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::RANS &&
87  Kind_Solver != MAIN_SOLVER::FEM_EULER && Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
88  Kind_Solver != MAIN_SOLVER::FEM_LES) SU2_MPI::Error("Compressible flow equations must be selected for the MMS NS Two Half Circles case",
89  CURRENT_FUNCTION);
90 
91  if((Kind_Solver != MAIN_SOLVER::NAVIER_STOKES) &&
92  (Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES))
93  SU2_MPI::Error("Navier Stokes equations must be selected for the MMS NS Two Half Circles case",
94  CURRENT_FUNCTION);
95 
96  if((config->GetKind_FluidModel() != STANDARD_AIR) &&
97  (config->GetKind_FluidModel() != IDEAL_GAS))
98  SU2_MPI::Error("Standard air or ideal gas must be selected for the MMS NS Two Half Circles case",
99  CURRENT_FUNCTION);
100 
101  if(config->GetKind_ViscosityModel() != VISCOSITYMODEL::CONSTANT)
102  SU2_MPI::Error("Sutherland must be selected for viscosity for the MMS NS Two Half Circles case",
103  CURRENT_FUNCTION);
104 
105  if(config->GetKind_ConductivityModel() != CONDUCTIVITYMODEL::CONSTANT_PRANDTL)
106  SU2_MPI::Error("Constant Prandtl number must be selected for the MMS NS Two Half Circles case",
107  CURRENT_FUNCTION);
108 }

120 void CMMSNSTwoHalfCirclesSolution::GetSolution(const su2double *val_coords,
121  const su2double val_t,
122  su2double *val_solution) const {
123 
124  /* Easier storage of the x- and y-coordinates. */
125  const su2double x = val_coords[0];
126  const su2double y = val_coords[1];
127 
128  /* Determine the dimensional solution for the temperature. */
129  const su2double Pi = PI_NUMBER;
130  const su2double r = sqrt(x*x + y*y);
131  const su2double fact = (r-1.0)*(r-1.0)/(a_T1 + a_T2);
132 
133  su2double T = 0.25*TWall*(3.0 + fact*(a_T1*cos(Pi*(r-2.0))
134  + a_T2*cos(Pi*(r-2.0)*2.0)));
135 
136  /* Determine the dimensional solution for the velocities. */
137  su2double u = u_0*(r-1.0)*(2.0-r)*4.0;
138  su2double v = v_0*(r-1.0)*(2.0-r)*4.0;
139 
140  /* Compute the pressure from the density and temperature. */
141  su2double rho = rho_0;
142  su2double p = rho*RGas*T;
143 
144  /* Determine the non-dimensional solution. */
145  rho /= Density_Ref;
146  p /= Pressure_Ref;
147  u /= Velocity_Ref;
148  v /= Velocity_Ref;
149 
150  /* Determine the non-dimensional conserved variables.
151  Note that the implementation below is valid for both 2D and 3D. */
152  val_solution[0] = rho;
153  val_solution[1] = rho*u;
154  val_solution[2] = rho*v;
155  val_solution[3] = 0.0;
156  val_solution[nDim+1] = p/(Gamma-1.0) + 0.5*rho*(u*u + v*v);
157 }
