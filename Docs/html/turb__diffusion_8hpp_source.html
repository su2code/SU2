38 template <class FlowIndices>
39 class CAvgGrad_TurbSA final : public CAvgGrad_Scalar<FlowIndices> {
40 private:
41  using Base = CAvgGrad_Scalar<FlowIndices>;
42  using Base::Laminar_Viscosity_i;
43  using Base::Laminar_Viscosity_j;
44  using Base::Density_i;
45  using Base::Density_j;
46  using Base::ScalarVar_i;
47  using Base::ScalarVar_j;
48  using Base::Proj_Mean_GradScalarVar;
49  using Base::proj_vector_ij;
50  using Base::Flux;
51  using Base::Jacobian_i;
52  using Base::Jacobian_j;
53 
54  const su2double sigma = 2.0/3.0;
55 
56  /*!
57  * \brief Adds any extra variables to AD
58  */
59  void ExtraADPreaccIn() override {}

61  /*!
62  * \brief SA specific steps in the ComputeResidual method
63  * \param[in] config - Definition of the particular problem.
64  */
65  void FinishResidualCalc(const CConfig* config) override {
66  const bool implicit = config->GetKind_TimeIntScheme() == EULER_IMPLICIT;
67 
68  /*--- Compute mean effective viscosity ---*/
69 
70  const su2double nu_i = Laminar_Viscosity_i/Density_i;
71  const su2double nu_j = Laminar_Viscosity_j/Density_j;
72  const su2double nu_e = 0.5*(nu_i+nu_j+ScalarVar_i[0]+ScalarVar_j[0]);
73 
74  Flux[0] = nu_e*Proj_Mean_GradScalarVar[0]/sigma;
75 
76  /*--- For Jacobians -> Use of TSL approx. to compute derivatives of the gradients ---*/
77 
78  if (implicit) {
79  Jacobian_i[0][0] = (0.5*Proj_Mean_GradScalarVar[0]-nu_e*proj_vector_ij)/sigma;
80  Jacobian_j[0][0] = (0.5*Proj_Mean_GradScalarVar[0]+nu_e*proj_vector_ij)/sigma;
81  }
82  }
83

85  /*!
86  * \brief Constructor of the class.
87  * \param[in] val_nDim - Number of dimensions of the problem.

116  using Base::Jacobian_i;

119  const su2double sigma = 2.0/3.0;
120  const su2double cn1 = 16.0;
121 
122  /*!
123  * \brief Adds any extra variables to AD
124  */
125  void ExtraADPreaccIn() override {}

127  /*!
128  * \brief SA-neg specific steps in the ComputeResidual method
129  * \param[in] config - Definition of the particular problem.
130  */
131  void FinishResidualCalc(const CConfig* config) override {
132  const bool implicit = config->GetKind_TimeIntScheme() == EULER_IMPLICIT;
133 
134  /*--- Compute mean effective viscosity ---*/
135 
136  const su2double nu_i = Laminar_Viscosity_i/Density_i;
137  const su2double nu_j = Laminar_Viscosity_j/Density_j;
138 
139  const su2double nu_ij = 0.5*(nu_i+nu_j);
140  const su2double nu_tilde_ij = 0.5*(ScalarVar_i[0] + ScalarVar_j[0]);
141 
142  su2double nu_e;
143 
144  if (nu_tilde_ij > 0.0) {
145  nu_e = nu_ij + nu_tilde_ij;
146  }
147  else {
148  const su2double Xi = nu_tilde_ij/nu_ij;
149  const su2double fn = (cn1 + Xi*Xi*Xi)/(cn1 - Xi*Xi*Xi);
150  nu_e = nu_ij + fn*nu_tilde_ij;
151  }
152 
153  Flux[0] = nu_e*Proj_Mean_GradScalarVar[0]/sigma;
154 
155  /*--- For Jacobians -> Use of TSL approx. to compute derivatives of the gradients ---*/
156 
157  if (implicit) {
158  Jacobian_i[0][0] = (0.5*Proj_Mean_GradScalarVar[0]-nu_e*proj_vector_ij)/sigma;
159  Jacobian_j[0][0] = (0.5*Proj_Mean_GradScalarVar[0]+nu_e*proj_vector_ij)/sigma;
160  }
161  }
162 
163 public:
164  /*!
165  * \brief Constructor of the class.
166  * \param[in] val_nDim - Number of dimensions of the problem.
167  * \param[in] val_nVar - Number of variables of the problem.
168  * \param[in] correct_grad - Whether to correct gradient for skewness.

182 template <class FlowIndices>
183 class CAvgGrad_TurbSST final : public CAvgGrad_Scalar<FlowIndices> {
184 private:
185  using Base = CAvgGrad_Scalar<FlowIndices>;
186  using Base::Laminar_Viscosity_i;
187  using Base::Laminar_Viscosity_j;
188  using Base::Eddy_Viscosity_i;

200  const su2double sigma_k1; /*!< \brief Constants for the viscous terms, k-w (1), k-eps (2)*/
201  const su2double sigma_k2;
202  const su2double sigma_om1;
203  const su2double sigma_om2;
204 
205  su2double F1_i, F1_j; /*!< \brief Menter's first blending function */
206 
207  /*!

209  */

211  AD::SetPreaccIn(F1_i, F1_j);

214  /*!
215  * \brief SST specific steps in the ComputeResidual method
216  * \param[in] config - Definition of the particular problem.
217  */
218  void FinishResidualCalc(const CConfig* config) override {
219  const bool implicit = config->GetKind_TimeIntScheme() == EULER_IMPLICIT;
220 
221  /*--- Compute the blended constant for the viscous terms ---*/
222  const su2double sigma_kine_i = F1_i*sigma_k1 + (1.0 - F1_i)*sigma_k2;
223  const su2double sigma_kine_j = F1_j*sigma_k1 + (1.0 - F1_j)*sigma_k2;
224  const su2double sigma_omega_i = F1_i*sigma_om1 + (1.0 - F1_i)*sigma_om2;
225  const su2double sigma_omega_j = F1_j*sigma_om1 + (1.0 - F1_j)*sigma_om2;
226 
227  /*--- Compute mean effective dynamic viscosity ---*/
228  const su2double diff_i_kine = Laminar_Viscosity_i + sigma_kine_i*Eddy_Viscosity_i;
229  const su2double diff_j_kine = Laminar_Viscosity_j + sigma_kine_j*Eddy_Viscosity_j;
230  const su2double diff_i_omega = Laminar_Viscosity_i + sigma_omega_i*Eddy_Viscosity_i;
231  const su2double diff_j_omega = Laminar_Viscosity_j + sigma_omega_j*Eddy_Viscosity_j;
232 
233  const su2double diff_kine = 0.5*(diff_i_kine + diff_j_kine);
234  const su2double diff_omega = 0.5*(diff_i_omega + diff_j_omega);
235 
236  Flux[0] = diff_kine*Proj_Mean_GradScalarVar[0];
237  Flux[1] = diff_omega*Proj_Mean_GradScalarVar[1];
238 
239  /*--- For Jacobians -> Use of TSL (Thin Shear Layer) approx. to compute derivatives of the gradients ---*/
240  if (implicit) {
241  const su2double proj_on_rho_i = proj_vector_ij/Density_i;
242  Jacobian_i[0][0] = -diff_kine*proj_on_rho_i; Jacobian_i[0][1] = 0.0;
243  Jacobian_i[1][0] = 0.0; Jacobian_i[1][1] = -diff_omega*proj_on_rho_i;
244 
245  const su2double proj_on_rho_j = proj_vector_ij/Density_j;
246  Jacobian_j[0][0] = diff_kine*proj_on_rho_j; Jacobian_j[0][1] = 0.0;
247  Jacobian_j[1][0] = 0.0; Jacobian_j[1][1] = diff_omega*proj_on_rho_j;
248  }
249  }
250

252  /*!
253  * \brief Constructor of the class.
254  * \param[in] val_nDim - Number of dimensions of the problem.
255  * \param[in] val_nVar - Number of variables of the problem.
256  * \param[in] constants - Constants of the model.
257  * \param[in] correct_grad - Whether to correct gradient for skewness.
258  * \param[in] config - Definition of the particular problem.
259  */
