31 CDiscAdjMeshSolver::CDiscAdjMeshSolver(CGeometry *geometry, CConfig *config, CSolver *direct_solver) : CSolver() {
32 
33  nVar = geometry->GetnDim();
34  nDim = geometry->GetnDim();
35 
36  /*-- Store some information about direct solver ---*/
37  this->direct_solver = direct_solver;
38 
39  nPoint = geometry->GetnPoint();
40  nPointDomain = geometry->GetnPointDomain();
41 
42  omp_chunk_size = computeStaticChunkSize(nPoint, omp_get_max_threads(), OMP_MAX_SIZE);
43 
44  /*--- Define some auxiliary vectors related to the residual ---*/
45 
46  Residual_RMS.resize(nVar,1.0);
47  Residual_Max.resize(nVar,1.0);
48  Point_Max.resize(nVar,0);
49  Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
50 
51  /*--- Define some auxiliary vectors related to the residual for problems with a BGS strategy---*/
52 
53  if (config->GetMultizone_Residual()){
54 
55  Residual_BGS.resize(nVar,1.0);
56  Residual_Max_BGS.resize(nVar,1.0);
57  Point_Max_BGS.resize(nVar,0);
58  Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
59  }
60 
61  /*--- Initialize the node structure ---*/
62  nodes = new CDiscAdjMeshBoundVariable(nPoint,nDim,config);
63  SetBaseClassPointerToNodes();
64 
65  /*--- Set which points are vertices and allocate boundary data. ---*/
66 
67  vector<su2double> Solution(nVar,1e-16);
68 
69  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++) {
70 
71  nodes->SetSolution(iPoint,Solution.data());
72 
73  for (unsigned short iMarker = 0; iMarker < config->GetnMarker_All(); iMarker++) {
74  long iVertex = geometry->nodes->GetVertex(iPoint, iMarker);
75  if (iVertex >= 0) {
76  nodes->Set_isVertex(iPoint,true);
77  break;
78  }
79  }
80  }
81  static_cast<CDiscAdjMeshBoundVariable*>(nodes)->AllocateBoundaryVariables(config);
82 
83 }

116 void CDiscAdjMeshSolver::ExtractAdjoint_Solution(CGeometry *geometry, CConfig *config, bool CrossTerm){
117 
118  /*--- Extract the sensitivities of the mesh coordinates ---*/
119 
120  SU2_OMP_FOR_STAT(omp_chunk_size)
121  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++){
122 
123  /*--- Extract the adjoint solution from the original mesh coordinates ---*/
124 
125  su2double Solution[MAXNVAR] = {0.0};
126  direct_solver->GetNodes()->GetAdjoint_MeshCoord(iPoint,Solution);
127 
128  /*--- Store the adjoint solution (the container is reused) ---*/
129 
130  nodes->SetSolution(iPoint,Solution);
131 
132  }
133  END_SU2_OMP_FOR

156 void CDiscAdjMeshSolver::SetSensitivity(CGeometry *geometry, CConfig *config, CSolver *solver) {
157 
158  SU2_OMP_PARALLEL {
159 
160  const bool time_stepping = (config->GetTime_Marching() != TIME_MARCHING::STEADY);
161  const auto eps = config->GetAdjSharp_LimiterCoeff()*config->GetRefElemLength();
162 
163  /*--- Extract the sensitivities ---*/
164  ExtractAdjoint_Solution(geometry, config, false);
165 
166  /*--- Extract the adjoint variables: sensitivities of the boundary displacements ---*/
167  ExtractAdjoint_Variables(geometry, config);
168 
169  /*--- Store the sensitivities in the flow adjoint container ---*/
170 
171  SU2_OMP_FOR_STAT(omp_chunk_size)
172  for (auto iPoint = 0ul; iPoint < nPoint; iPoint++) {
173 
174  /*--- If sharp edge, set the sensitivity to 0 on that region ---*/
175  su2double limiter = 1.0;
176  if (config->GetSens_Remove_Sharp() && (geometry->nodes->GetSharpEdge_Distance(iPoint) < eps)) {
177  limiter = 0.0;
178  }
179 
180  for (auto iDim = 0u; iDim < nDim; iDim++) {
181  /*--- The sensitivity was extracted using ExtractAdjoint_Solution ---*/
182  const su2double Sensitivity = limiter * nodes->GetSolution(iPoint,iDim);
183 
184  /*--- Store the sensitivities ---*/
185  if (!time_stepping) {
186  solver->GetNodes()->SetSensitivity(iPoint, iDim, Sensitivity);
187  } else {
188  solver->GetNodes()->SetSensitivity(iPoint, iDim,
189  solver->GetNodes()->GetSensitivity(iPoint, iDim) + Sensitivity);
190  }
191  }
192  }
193  END_SU2_OMP_FOR
194 
195  solver->SetSurface_Sensitivity(geometry, config);
196 
197  }
198  END_SU2_OMP_PARALLEL
199 }
