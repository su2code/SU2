35 /*!
36  * \brief Base class for 1D interpolation.
37  * \ingroup LookUpInterp
38  */
39 class C1DInterpolation {
40 protected:
41  std::vector<su2double> x, y; /*!< \brief Data points. */
42 
43  /*!
44  * \brief Find containing interval.
45  * \param[in] xi - Point for which interpolation is required.
46  * \return The start index of the interval, or the size if the coordinate is out of bounds.
47  */
48  inline size_t lower_bound(su2double xi) const {
49 
50  if (xi <= x.front() || xi >= x.back()) return x.size();
51 
52  size_t lb = 0, ub = x.size()-1;
53 
54  while (ub-lb > 1) {
55  size_t mid = (lb+ub)/2;
56  auto& change = (xi < x[mid])? ub : lb;
57  change = mid;
58  }
59  return lb;
60  }
61 
62 public:
63  /*!
64  * \brief Virtual destructor of the C1DInterpolation class.
65  */
66  virtual ~C1DInterpolation() = default;
67 
68  /*!
69  * \brief Virtual method for setting the coefficients of the respective spline.
70  * \param[in] X - the x values.
71  * \param[in] Data - the f(x) values.
72  */
73  virtual void SetSpline(const std::vector<su2double> &X, const std::vector<su2double> &Data) {
74  assert(X.size() == Data.size());
75  x = X;
76  y = Data;
77  }
78 
79  /*!
80  * \brief Evaluate the value of the spline at a point.
81  */
82  virtual su2double EvaluateSpline(su2double Point_Interp) const = 0;
83  inline su2double operator() (su2double Point_Interp) const { return EvaluateSpline(Point_Interp); }
84 
85 };
