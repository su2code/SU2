32 CTGVSolution::CTGVSolution(unsigned short val_nDim,
33  unsigned short val_nVar,
34  unsigned short val_iMesh,
35  CConfig* config)
36  : CVerificationSolution(val_nDim, val_nVar, val_iMesh, config) {
37 
38  /*--- Write a message that the solution is initialized for the
39  Taylor-Green vortex test case. ---*/
40 
41  if ((rank == MASTER_NODE) && (val_iMesh == MESH_0)) {
42  cout << endl;
43  cout << "Warning: Fluid properties and solution are being " << endl;
44  cout << " initialized for the Taylor-Green vortex case!!!" << endl;
45  cout << endl << flush;
46  }
47 
48  /*--- Store TGV specific parameters here. ---*/
49 
50  tgvLength = 1.0; // Taylor-Green length scale.
51  tgvVelocity = 1.0; // Taylor-Green velocity.
52  tgvDensity = 1.0; // Taylor-Green density.
53  tgvPressure = 100.0; // Taylor-Green pressure.
54 
55  /*--- Useful coefficient in which Gamma is present. ---*/
56 
57  ovGm1 = 1.0/(config->GetGamma() - 1.0);
58 
59  /*--- Perform some sanity and error checks for this solution here. ---*/
60 
61  if((config->GetTime_Marching() != TIME_MARCHING::TIME_STEPPING) &&
62  (config->GetTime_Marching() != TIME_MARCHING::DT_STEPPING_1ST) &&
63  (config->GetTime_Marching() != TIME_MARCHING::DT_STEPPING_2ND))
64  SU2_MPI::Error("Unsteady mode must be selected for the Taylor Green Vortex",
65  CURRENT_FUNCTION);
66 
67  if(Kind_Solver != MAIN_SOLVER::EULER && Kind_Solver != MAIN_SOLVER::NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::RANS &&
68  Kind_Solver != MAIN_SOLVER::FEM_EULER && Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES && Kind_Solver != MAIN_SOLVER::FEM_RANS &&
69  Kind_Solver != MAIN_SOLVER::FEM_LES)
70  SU2_MPI::Error("Compressible flow equations must be selected for the Taylor Green Vortex",
71  CURRENT_FUNCTION);
72 
73  if((Kind_Solver != MAIN_SOLVER::NAVIER_STOKES) &&
74  (Kind_Solver != MAIN_SOLVER::FEM_NAVIER_STOKES))
75  SU2_MPI::Error("Navier Stokes equations must be selected for the Taylor Green Vortex",
76  CURRENT_FUNCTION);
77 
78  if((config->GetKind_FluidModel() != STANDARD_AIR) &&
79  (config->GetKind_FluidModel() != IDEAL_GAS))
80  SU2_MPI::Error("Standard air or ideal gas must be selected for the Taylor Green Vortex",
81  CURRENT_FUNCTION);
82 
83  if(config->GetKind_ViscosityModel() != VISCOSITYMODEL::CONSTANT)
84  SU2_MPI::Error("Constant viscosity must be selected for the Taylor Green Vortex",
85  CURRENT_FUNCTION);
86 
87  if(config->GetKind_ConductivityModel() != CONDUCTIVITYMODEL::CONSTANT_PRANDTL)
88  SU2_MPI::Error("Constant Prandtl number must be selected for the Taylor Green Vortex",
89  CURRENT_FUNCTION);
90 }

94 void CTGVSolution::GetSolution(const su2double *val_coords,
95  const su2double val_t,
96  su2double *val_solution) const {
97 
98  /* The initial conditions are set for the Taylor-Green vortex case, which
99  is a DNS case that features vortex breakdown into turbulence. These
100  particular settings are for the typical Re = 1600 case (M = 0.08) with
101  an initial temperature of 300 K. Note that this condition works in both
102  2D and 3D. */
103 
104  su2double val_coordsZ = 0.0;
105  if (nDim == 3) val_coordsZ = val_coords[2];
106 
107  /* Compute the primitive variables. */
108 
109  su2double rho = tgvDensity;
110  su2double u = tgvVelocity * (sin(val_coords[0]/tgvLength)*
111  cos(val_coords[1]/tgvLength)*
112  cos(val_coordsZ /tgvLength));
113  su2double v = -tgvVelocity * (cos(val_coords[0]/tgvLength)*
114  sin(val_coords[1]/tgvLength)*
115  cos(val_coordsZ /tgvLength));
116 
117  su2double factorA = cos(2.0*val_coordsZ/tgvLength) + 2.0;
118  su2double factorB = (cos(2.0*val_coords[0]/tgvLength) +
119  cos(2.0*val_coords[1]/tgvLength));
120 
121  su2double p = (tgvPressure +
122  tgvDensity*(pow(tgvVelocity,2.0)/16.0)*factorA*factorB);
123 
124  /* Compute the conservative variables. Note that both 2D and 3D
125  cases are treated correctly. */
126 
127  val_solution[0] = rho;
128  val_solution[1] = rho*u;
129  val_solution[2] = rho*v;
130  val_solution[3] = 0.0;
131  val_solution[nVar-1] = p*ovGm1 + 0.5*rho*(u*u + v*v);
132 }
