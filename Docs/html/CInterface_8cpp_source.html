62 void CInterface::BroadcastData(const CInterpolator& interpolator,
63  CSolver *donor_solution, CSolver *target_solution,
64  CGeometry *donor_geometry, CGeometry *target_geometry,
65  const CConfig *donor_config, const CConfig *target_config) {
66  static_assert(su2activematrix::Storage == StorageType::RowMajor,"");
67 
68  GetPhysical_Constants(donor_solution, target_solution, donor_geometry, target_geometry,
69  donor_config, target_config);
70 
71  /*--- Loop over interface markers. ---*/
72 
73  for (auto iMarkerInt = 0u; iMarkerInt < donor_config->GetMarker_n_ZoneInterface()/2; iMarkerInt++) {
74 
75  /*--- Check if this interface connects the two zones, if not continue. ---*/
76 
77  const auto markDonor = donor_config->FindInterfaceMarker(iMarkerInt);
78  const auto markTarget = target_config->FindInterfaceMarker(iMarkerInt);
79 
80  if(!CInterpolator::CheckInterfaceBoundary(markDonor, markTarget)) continue;
81 
82  /*--- Count donor vertices on this rank. ---*/
83 
84  int nLocalVertexDonor = 0;
85  if (markDonor >= 0) {
86  for (auto iVertex = 0ul; iVertex < donor_geometry->GetnVertex(markDonor); iVertex++) {
87  auto Point_Donor = donor_geometry->vertex[markDonor][iVertex]->GetNode();
88  /*--- Only domain points are donors. ---*/
89  nLocalVertexDonor += donor_geometry->nodes->GetDomain(Point_Donor);
90  }
91  }
92 
93  /*--- Gather donor counts and compute total sizes, and displacements (cumulative
94  * sums) to perform an Allgatherv of donor indices and variables. ---*/
95 
96  vector<int> nAllVertexDonor(size), nAllVarCounts(size), displIdx(size,0), displVar(size);
97  SU2_MPI::Allgather(&nLocalVertexDonor, 1, MPI_INT, nAllVertexDonor.data(), 1, MPI_INT, SU2_MPI::GetComm());
98 
99  for (int i = 0; i < size; ++i) {
100  nAllVarCounts[i] = nAllVertexDonor[i] * nVar;
101  if(i) displIdx[i] = displIdx[i-1] + nAllVertexDonor[i-1];
102  displVar[i] = displIdx[i] * nVar;
103  }
104 
105  /*--- Fill send buffers. ---*/
106 
107  vector<unsigned long> sendDonorIdx(nLocalVertexDonor);
108  su2activematrix sendDonorVar(nLocalVertexDonor, nVar);
109 
110  if (markDonor >= 0) {
111  for (auto iVertex = 0ul, iSend = 0ul; iVertex < donor_geometry->GetnVertex(markDonor); iVertex++) {
112  const auto iPoint = donor_geometry->vertex[markDonor][iVertex]->GetNode();
113 
114  /*--- If this processor owns the node. ---*/
115  if (donor_geometry->nodes->GetDomain(iPoint)) {
116 
117  GetDonor_Variable(donor_solution, donor_geometry, donor_config, markDonor, iVertex, iPoint);
118  for (auto iVar = 0u; iVar < nVar; iVar++) sendDonorVar(iSend, iVar) = Donor_Variable[iVar];
119 
120  sendDonorIdx[iSend] = donor_geometry->nodes->GetGlobalIndex(iPoint);
121  ++iSend;
122  }
123  }
124  }
125 
126  /*--- Gather data. ---*/
127 
128  const auto nGlobalVertexDonor = displIdx.back() + nAllVertexDonor.back();
129 
130  vector<unsigned long> donorIdx(nGlobalVertexDonor);
131  su2activematrix donorVar(nGlobalVertexDonor, nVar);
132 
133  SU2_MPI::Allgatherv(sendDonorIdx.data(), sendDonorIdx.size(), MPI_UNSIGNED_LONG, donorIdx.data(),
134  nAllVertexDonor.data(), displIdx.data(), MPI_UNSIGNED_LONG, SU2_MPI::GetComm());
135 
136  SU2_MPI::Allgatherv(sendDonorVar.data(), sendDonorVar.size(), MPI_DOUBLE, donorVar.data(),
137  nAllVarCounts.data(), displVar.data(), MPI_DOUBLE, SU2_MPI::GetComm());
138 
139  /*--- This rank does not need to do more work. ---*/
140  if (markTarget < 0) continue;
141 
142  /*--- Sort the donor information by index to then use binary searches. ---*/
143 
144  vector<size_t> order(donorIdx.size());
145  iota(order.begin(), order.end(), 0ul);
146  sort(order.begin(), order.end(), [&donorIdx](size_t i, size_t j) {return donorIdx[i] < donorIdx[j];} );
147 
148  /*--- inplace permutation. ---*/
149  for (size_t i = 0; i < order.size(); ++i) {
150  auto j = order[i];
151  while (j < i) j = order[j];
152  if (i == j) continue;
153  swap(donorIdx[i], donorIdx[j]);
154  for (auto iVar = 0u; iVar < nVar; ++iVar)
155  swap(donorVar(i,iVar), donorVar(j,iVar));
156  }
157 
158  /*--- Loop over target vertices. ---*/
159 
160  for (auto iVertex = 0ul; iVertex < target_geometry->GetnVertex(markTarget); iVertex++) {
161  const auto iPoint = target_geometry->vertex[markTarget][iVertex]->GetNode();
162 
163  if (!target_geometry->nodes->GetDomain(iPoint)) continue;
164 
165  auto& targetVertex = interpolator.targetVertices[markTarget][iVertex];
166  const auto nDonorPoints = targetVertex.nDonor();
167 
168  InitializeTarget_Variable(target_solution, markTarget, iVertex, nDonorPoints);
169 
170  /*--- For the number of donor points. ---*/
171  for (auto iDonorPoint = 0ul; iDonorPoint < nDonorPoints; iDonorPoint++) {
172 
173  /*--- Get the global index of the donor and the interpolation coefficient. ---*/
174 
175  const auto donorGlobalIndex = targetVertex.globalPoint[iDonorPoint];
176  const auto donorCoeff = targetVertex.coefficient[iDonorPoint];
177 
178  /*--- Find the index of the global donor point in the donor data. ---*/
179 
180  const auto idx = lower_bound(donorIdx.begin(), donorIdx.end(), donorGlobalIndex) - donorIdx.begin();
181  assert(idx < static_cast<long>(donorIdx.size()));
182 
183  /*--- Recover the Target_Variable from the buffer of variables. ---*/
184  RecoverTarget_Variable(donorVar[idx], donorCoeff);
185 
186  /*--- If the value is not directly aggregated in the previous function. ---*/
187  if (!valAggregated)
188  SetTarget_Variable(target_solution, target_geometry, target_config, markTarget, iVertex, iPoint);
189  }
190 
191  /*--- If we have aggregated the values in the function RecoverTarget_Variable, the set is outside the loop. ---*/
192  if (valAggregated)
193  SetTarget_Variable(target_solution, target_geometry, target_config, markTarget, iVertex, iPoint);
194  }
195  }
196 }

198 void CInterface::PreprocessAverage(CGeometry *donor_geometry, CGeometry *target_geometry,
199  const CConfig *donor_config, const CConfig *target_config,
200  unsigned short iMarkerInt){
201 
202  unsigned short nMarkerDonor, nMarkerTarget; // Number of markers on the interface, donor and target side
203  unsigned short iMarkerDonor, iMarkerTarget; // Variables for iteration over markers
204  unsigned short iSpan,jSpan, tSpan = 0, kSpan = 0, nSpanDonor, nSpanTarget, Donor_Flag = 0, Target_Flag = 0;
205  int Marker_Donor = -1, Marker_Target = -1;
206 
207  const su2double *SpanValuesDonor, *SpanValuesTarget;
208  su2double dist, test, dist2, test2;
209 
210  nMarkerDonor = donor_geometry->GetnMarker();
211  nMarkerTarget = target_geometry->GetnMarker();
212  //TODO turbo this approach only works if all the turboamchinery marker
213  // of all zones have the same amount of span wise sections.
214  //TODO turbo initialization needed for the MPI routine should be place somewhere else.
215  nSpanDonor = donor_config->GetnSpanWiseSections();
216  nSpanTarget = target_config->GetnSpanWiseSections();
217 
218  /*--- On the donor side ---*/
219  for (iMarkerDonor = 0; iMarkerDonor < nMarkerDonor; iMarkerDonor++){
220  /*--- If the tag GetMarker_All_MixingPlaneInterface equals the index we are looping at ---*/
221  if ( donor_config->GetMarker_All_MixingPlaneInterface(iMarkerDonor) == iMarkerInt ){
222  /*--- We have identified the local index of the Donor marker ---*/
223  /*--- Now we are going to store the average values that belong to Marker_Donor on each processor ---*/
224  /*--- Store the identifier for the structural marker ---*/
225  Marker_Donor = iMarkerDonor;
226  Donor_Flag = donor_config->GetMarker_All_TurbomachineryFlag(iMarkerDonor);
227  /*--- Exit the for loop: we have found the local index for Mixing-Plane interface ---*/
228  break;
229  }
230  else {
231  /*--- If the tag hasn't matched any tag within the donor markers ---*/
232  Marker_Donor = -1;
233  Donor_Flag = -1;
234  }
235  }
236 
237 #ifdef HAVE_MPI
238  auto BuffMarkerDonor = new int[size];
239  auto BuffDonorFlag = new int[size];
240  for (int iSize=0; iSize<size; iSize++){
241  BuffMarkerDonor[iSize] = -1;
242  BuffDonorFlag[iSize] = -1;
243  }
244 
245  SU2_MPI::Allgather(&Marker_Donor, 1 , MPI_INT, BuffMarkerDonor, 1, MPI_INT, SU2_MPI::GetComm());
246  SU2_MPI::Allgather(&Donor_Flag, 1 , MPI_INT, BuffDonorFlag, 1, MPI_INT, SU2_MPI::GetComm());
247 
248  Marker_Donor= -1;
249  Donor_Flag= -1;
250 
251  for (int iSize=0; iSize<size; iSize++){
252  if(BuffMarkerDonor[iSize] > 0.0){
253  Marker_Donor = BuffMarkerDonor[iSize];
254  Donor_Flag = BuffDonorFlag[iSize];
255  break;
256  }
257  }
258  delete [] BuffMarkerDonor;
259  delete [] BuffDonorFlag;
260 #endif
261 
262  /*--- On the target side we have to identify the marker as well ---*/
263 
264  for (iMarkerTarget = 0; iMarkerTarget < nMarkerTarget; iMarkerTarget++){
265  /*--- If the tag GetMarker_All_MixingPlaneInterface(iMarkerTarget) equals the index we are looping at ---*/
266  if ( target_config->GetMarker_All_MixingPlaneInterface(iMarkerTarget) == iMarkerInt ){
267  /*--- Store the identifier for the fluid marker ---*/
268 
269  // here i should then store it in the target zone
270 
271  Marker_Target = iMarkerTarget;
272  Target_Flag = target_config->GetMarker_All_TurbomachineryFlag(iMarkerTarget);
273  /*--- Exit the for loop: we have found the local index for iMarkerFSI on the FEA side ---*/
274  break;
275  }
276  else {
277  /*--- If the tag hasn't matched any tag within the Flow markers ---*/
278  Marker_Target = -1;
279  }
280  }
281 
282  if (Marker_Target != -1 && Marker_Donor != -1){
283 
284  SpanValuesDonor = donor_geometry->GetSpanWiseValue(Donor_Flag);
285  SpanValuesTarget = target_geometry->GetSpanWiseValue(Target_Flag);
286 
287 
288  for(iSpan = 1; iSpan <nSpanTarget-1; iSpan++){
289  dist = 10E+06;
290  dist2 = 10E+06;
291  for(jSpan = 0; jSpan < nSpanDonor;jSpan++){
292  test = abs(SpanValuesTarget[iSpan] - SpanValuesDonor[jSpan]);
293  test2 = abs(SpanValuesTarget[iSpan] - SpanValuesDonor[jSpan]);
294  if(test < dist && SpanValuesTarget[iSpan] > SpanValuesDonor[jSpan]){
295  dist = test;
296  kSpan = jSpan;
297  }
298  if(test2 < dist2){
299  dist2 = test2;
300  tSpan =jSpan;
301  }
302 
303  }
304  switch(donor_config->GetKind_MixingPlaneInterface()){
305  case MATCHING:
306  SpanLevelDonor[iSpan] = iSpan;
307  SpanValueCoeffTarget[iSpan] = 0.0;
308  break;
309  case NEAREST_SPAN:
310  SpanLevelDonor[iSpan] = tSpan;
311  SpanValueCoeffTarget[iSpan] = 0.0;
312  break;
313  case LINEAR_INTERPOLATION:
314  SpanLevelDonor[iSpan] = kSpan;
315  SpanValueCoeffTarget[iSpan] = (SpanValuesTarget[iSpan] - SpanValuesDonor[kSpan])
316  /(SpanValuesDonor[kSpan + 1] - SpanValuesDonor[kSpan]);
317  break;
318  default:
319  SU2_MPI::Error("MixingPlane interface option not implemented yet", CURRENT_FUNCTION);
320  break;
321 
322  }
323  }
324  }
325 
326 }
