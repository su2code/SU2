31 CTRIA1::CTRIA1() : CElementWithKnownSizes<NGAUSS,NNODE,NDIM>() {
32 
33  /*--- Gauss coordinates and weights ---*/
34 
35  GaussCoord[0][0] = 1.0/3.0; GaussCoord[0][1] = 1.0/3.0; GaussWeight(0) = 0.5;
36 
37  /*--- Store the values of the shape functions and their derivatives ---*/
38 
39  unsigned short iGauss;
40  su2double Xi, Eta, val_Ni;
41 
42  for (iGauss = 0; iGauss < NGAUSS; iGauss++) {
43 
44  Xi = GaussCoord[iGauss][0];
45  Eta = GaussCoord[iGauss][1];
46 
47  val_Ni = Xi; GaussPoint[iGauss].SetNi(val_Ni,0);
48  val_Ni = Eta; GaussPoint[iGauss].SetNi(val_Ni,1);
49  val_Ni = 1-Xi-Eta; GaussPoint[iGauss].SetNi(val_Ni,2);
50 
51  /*--- dN/d xi, dN/d eta ---*/
52 
53  dNiXj[iGauss][0][0] = 1.0; dNiXj[iGauss][0][1] = 0.0;
54  dNiXj[iGauss][1][0] = 0.0; dNiXj[iGauss][1][1] = 1.0;
55  dNiXj[iGauss][2][0] = -1.0; dNiXj[iGauss][2][1] = -1.0;
56 
57  }
58 
59  /*--- Shape functions evaluated at the nodes for extrapolation of the stresses at the Gaussian Points ---*/
60  /*--- The stress is constant over a TRIA1 element ---*/
61 
62  NodalExtrap[0][0] = 1.0;
63  NodalExtrap[1][0] = 1.0;
64  NodalExtrap[2][0] = 1.0;
65 
66 }

68 su2double CTRIA1::ComputeArea(const FrameType mode) const {
69 
70  unsigned short iDim;
71  su2double a[2] = {0.0,0.0}, b[2] = {0.0,0.0};
72  su2double Area = 0.0;
73 
74  /*--- Select the appropriate source for the nodal coordinates depending on the frame requested
75  for the gradient computation, REFERENCE (undeformed) or CURRENT (deformed) ---*/
76  const su2activematrix& Coord = (mode==REFERENCE) ? RefCoord : CurrentCoord;
77 
78  for (iDim = 0; iDim < NDIM; iDim++) {
79  a[iDim] = Coord[0][iDim]-Coord[2][iDim];
80  b[iDim] = Coord[1][iDim]-Coord[2][iDim];
81  }
82 
83  Area = 0.5*fabs(a[0]*b[1]-a[1]*b[0]);
84 
85  return Area;
86 
87 }
