225 /*!
226  * \brief Compute a chunk size based on totalWork and number of threads such that
227  * all threads get the same number of chunks (with limited size).
228  * \param[in] totalWork - e.g. total number of loop iterations.
229  * \param[in] numThreads - Number of threads that will share the work.
230  * \param[in] maxChunkSize - Upper bound for chunk size.
231  * \return The chunkSize.
232  */
233 inline size_t computeStaticChunkSize(size_t totalWork,
234  size_t numThreads,
235  size_t maxChunkSize)
236 {
237  if(!totalWork) return maxChunkSize;
238  size_t workPerThread = roundUpDiv(totalWork, numThreads);
239  size_t chunksPerThread = roundUpDiv(workPerThread, maxChunkSize);
240  return roundUpDiv(workPerThread, chunksPerThread);
241 }
