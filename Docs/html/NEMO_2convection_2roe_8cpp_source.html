92 CNumerics::ResidualType<> CUpwRoe_NEMO::ComputeResidual(const CConfig *config) {
93 
94  /*--- Face area (norm or the normal vector) ---*/
95  Area = GeometryToolbox::Norm(nDim, Normal);
96 
97  /*--- Unit Normal ---*/
98  for (unsigned short iDim = 0; iDim < nDim; iDim++)
99  UnitNormal[iDim] = Normal[iDim]/Area;
100 
101  /*--- Calculate Roe averaged variables ---*/
102  su2double R = sqrt(abs(V_j[RHO_INDEX]/V_i[RHO_INDEX]));
103 
104  for (unsigned short iVar = 0; iVar < nVar; iVar++)
105  RoeU[iVar] = (R*U_j[iVar] + U_i[iVar])/(R+1);
106 
107  for (unsigned short iVar = 0; iVar < nPrimVar; iVar++)
108  RoeV[iVar] = (R*V_j[iVar] + V_i[iVar])/(R+1);
109 
110  const auto& roe_eves = fluidmodel->ComputeSpeciesEve(RoeV[TVE_INDEX]);
111 
112  /*--- Calculate derivatives of pressure ---*/
113  fluidmodel->ComputedPdU(RoeV, roe_eves, RoedPdU);
114 
115  /*--- Calculate dual grid tangent vectors for P & invP ---*/
116  su2double l[MAXNDIM], m[MAXNDIM];
117  CreateBasis(UnitNormal,l,m);
118 
119  /*--- Compute the inviscid projected fluxes ---*/
120  GetInviscidProjFlux(U_i, V_i, Normal, ProjFlux_i);
121  GetInviscidProjFlux(U_j, V_j, Normal, ProjFlux_j);
122 
123  /*--- Compute projected P, invP, and Lambda ---*/
124  GetPMatrix (RoeU, RoeV, RoedPdU, UnitNormal, l, m, P_Tensor);
125  GetPMatrix_inv(RoeU, RoeV, RoedPdU, UnitNormal, l, m, invP_Tensor);
126 
127  /*--- Compute projected velocities ---*/
128  su2double ProjVelocity = GeometryToolbox::DotProduct(nDim, &RoeV[VEL_INDEX], UnitNormal);
129  su2double ProjVelocity_i = GeometryToolbox::DotProduct(nDim, &V_i[VEL_INDEX], UnitNormal);
130  su2double ProjVelocity_j = GeometryToolbox::DotProduct(nDim, &V_j[VEL_INDEX], UnitNormal);
131 
132  su2double RoeSoundSpeed = sqrt((1.0+RoedPdU[nSpecies+nDim])*
133  RoeV[P_INDEX]/RoeV[RHO_INDEX]);
134 
135  /*--- Calculate eigenvalues ---*/
136  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++)
137  Lambda[iSpecies] = ProjVelocity;
138 
139  for (unsigned short iDim = 0; iDim < nDim-1; iDim++)
140  Lambda[nSpecies+iDim] = ProjVelocity;
141 
142  Lambda[nSpecies+nDim-1] = ProjVelocity + RoeSoundSpeed;
143  Lambda[nSpecies+nDim] = ProjVelocity - RoeSoundSpeed;
144  Lambda[nSpecies+nDim+1] = ProjVelocity;
145 
146  /*--- Harten and Hyman (1983) entropy correction ---*/
147  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++)
148  Epsilon[iSpecies] = 4.0*max(0.0, max(Lambda[iSpecies]-ProjVelocity_i,
149  ProjVelocity_j-Lambda[iSpecies] ));
150  for (unsigned short iDim = 0; iDim < nDim-1; iDim++)
151  Epsilon[nSpecies+iDim] = 4.0*max(0.0, max(Lambda[iDim]-ProjVelocity_i,
152  ProjVelocity_j-Lambda[iDim] ));
153  Epsilon[nSpecies+nDim-1] = 4.0*max(0.0, max(Lambda[nSpecies+nDim-1]-(ProjVelocity_i+V_i[A_INDEX]),
154  (ProjVelocity_j+V_j[A_INDEX])-Lambda[nSpecies+nDim-1]));
155  Epsilon[nSpecies+nDim] = 4.0*max(0.0, max(Lambda[nSpecies+nDim]-(ProjVelocity_i-V_i[A_INDEX]),
156  (ProjVelocity_j-V_j[A_INDEX])-Lambda[nSpecies+nDim]));
157  Epsilon[nSpecies+nDim+1] = 4.0*max(0.0, max(Lambda[nSpecies+nDim+1]-ProjVelocity_i,
158  ProjVelocity_j-Lambda[nSpecies+nDim+1] ));
159  for (unsigned short iVar = 0; iVar < nVar; iVar++)
160  if ( fabs(Lambda[iVar]) < Epsilon[iVar] )
161  Lambda[iVar] = (Lambda[iVar]*Lambda[iVar] + Epsilon[iVar]*Epsilon[iVar])/(2.0*Epsilon[iVar]);
162  else
163  Lambda[iVar] = fabs(Lambda[iVar]);
164 
165  for (unsigned short iVar = 0; iVar < nVar; iVar++)
166  Lambda[iVar] = fabs(Lambda[iVar]);
167 
168  /*--- Calculate inviscid projected Jacobians ---*/
169  // Note: Scaling value is 0.5 because inviscid flux is based on 0.5*(Fc_i+Fc_j)
170  if (implicit){
171  GetInviscidProjJac(U_i, V_i, dPdU_i, Normal, 0.5, Jacobian_i);
172  GetInviscidProjJac(U_j, V_j, dPdU_j, Normal, 0.5, Jacobian_j);
173  }
174 
175  /*--- Difference of conserved variables at iPoint and jPoint ---*/
176  for (unsigned short iVar = 0; iVar < nVar; iVar++)
177  Diff_U[iVar] = U_j[iVar]-U_i[iVar];
178 
179  /*--- Roe's Flux approximation ---*/
180  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
181  Flux[iVar] = 0.5 * (ProjFlux_i[iVar] + ProjFlux_j[iVar]);
182  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
183 
184  /*--- Compute |Proj_ModJac_Tensor| = P x |Lambda| x inverse P ---*/
185  Proj_ModJac_Tensor_ij = 0.0;
186  for (unsigned short kVar = 0; kVar < nVar; kVar++)
187  Proj_ModJac_Tensor_ij += P_Tensor[iVar][kVar]*Lambda[kVar]*invP_Tensor[kVar][jVar];
188 
189  Flux[iVar] -= 0.5*Proj_ModJac_Tensor_ij*Diff_U[jVar]*Area;
190  if (implicit){
191  Jacobian_i[iVar][jVar] += 0.5*Proj_ModJac_Tensor_ij*Area;
192  Jacobian_j[iVar][jVar] -= 0.5*Proj_ModJac_Tensor_ij*Area;
193  }
194  }
195  }
196 
197  return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
198 }
