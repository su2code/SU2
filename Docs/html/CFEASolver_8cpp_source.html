236 void CFEASolver::HybridParallelInitialization(CGeometry* geometry) {
237 #ifdef HAVE_OMP
238  /*--- Get the element coloring. ---*/
239 
240  su2double parallelEff = 1.0;
241  const auto& coloring = geometry->GetElementColoring(&parallelEff);
242 
243  /*--- If the coloring is too bad use lock-guarded accesses
244  * to CSysMatrix/Vector in element loops instead. ---*/
245  LockStrategy = parallelEff < COLORING_EFF_THRESH;
246 
247  /*--- When using locks force a single color to reduce the color loop overhead. ---*/
248  if (LockStrategy && (coloring.getOuterSize()>1))
249  geometry->SetNaturalElementColoring();
250 
251  if (!coloring.empty()) {
252  /*--- We are not constrained by the color group size when using locks. ---*/
253  auto groupSize = LockStrategy? 1ul : geometry->GetElementColorGroupSize();
254  auto nColor = coloring.getOuterSize();
255  ElemColoring.reserve(nColor);
256 
257  for(auto iColor = 0ul; iColor < nColor; ++iColor)
258  ElemColoring.emplace_back(coloring.innerIdx(iColor), coloring.getNumNonZeros(iColor), groupSize);
259  }
260 
261  su2double minEff = 1.0;
262  SU2_MPI::Reduce(&parallelEff, &minEff, 1, MPI_DOUBLE, MPI_MIN, MASTER_NODE, SU2_MPI::GetComm());
263 
264  if (minEff < COLORING_EFF_THRESH) {
265  cout << "WARNING: The element coloring efficiency was " << minEff << ", a fallback strategy is in use.\n"
266  << " Better performance may be possible by reducing the number of threads per rank." << endl;
267  }
268 
269  if (LockStrategy) {
270  UpdateLocks.resize(nPoint);
271  for (unsigned long iPoint = 0; iPoint < nPoint; iPoint++)
272  omp_init_lock(&UpdateLocks[iPoint]);
273  }
274 
275  omp_chunk_size = computeStaticChunkSize(nPointDomain, omp_get_max_threads(), OMP_MAX_SIZE);
276 #else
277  ElemColoring[0] = DummyGridColor<>(nElement);
278 #endif
279 }

282 
283  const auto iZone = config->GetiZone();
284  const auto nZone = geometry->GetnZone();
285 
286  const bool topology_mode = config->GetTopology_Optimization();

297 
298  if (rank == MASTER_NODE) cout << "Filename: " << filename << "." << endl;
299 
300  ifstream properties_file;
301  properties_file.open(filename);
302 
303  /*--- In case there is no file, all elements get the same property (0) ---*/
304 
305  if (properties_file.fail()) {
306  if (rank == MASTER_NODE){
307  cout << "There is no element-based properties file." << endl;
308  cout << "The structural domain has uniform properties." << endl;
309 
310  if (topology_mode)
311  SU2_MPI::Error("Topology mode requires an element-based properties file.",CURRENT_FUNCTION);
312  }
313 
314  for (auto iElem = 0ul; iElem < nElement; iElem++){
315  element_properties[iElem] = new CElementProperty(FEA_TERM, 0, 0, 0);
316  }
317 
318  element_based = false;
319 
320  }
321  else{
322 
323  element_based = true;
324 
325  /*--- In case this is a parallel simulation, we need to perform the Global2Local index transformation first. ---*/
326 
327  unordered_map<unsigned long, unsigned long> Global2Local;
328 
329  for (auto iElem = 0ul; iElem < nElement; iElem++)
330  Global2Local[geometry->elem[iElem]->GetGlobalIndex()] = iElem;
331 
332  /*--- Read all lines in the restart file ---*/
333 
334  unsigned long iElem_Global_Local = 0, iElem_Global = 0; string text_line;
335 
336  /*--- The first line is the header ---*/
337 
338  getline (properties_file, text_line);
339 
340  for (iElem_Global = 0; iElem_Global < geometry->GetGlobal_nElemDomain(); iElem_Global++ ) {
341 
342  getline (properties_file, text_line);
343 
344  istringstream point_line(text_line);
345 
346  /*--- Retrieve local index. If this element from the restart file lives
347  only on a different processor, the value of iPoint_Local will be -1.
348  Otherwise, the local index for this node on the current processor
349  will be returned and used to instantiate the vars. ---*/
350 
351  auto it = Global2Local.find(iElem_Global);
352 
353  if (it != Global2Local.end()) {
354 
355  auto iElem_Local = it->second;
356 
357  unsigned long elProperties[4], index;
358 
359  if (config->GetAdvanced_FEAElementBased() || topology_mode){
360  point_line >> index >> elProperties[0] >> elProperties[1] >> elProperties[2] >> elProperties[3];
361  element_properties[iElem_Local] = new CElementProperty(
362  elProperties[0], elProperties[1], elProperties[2], elProperties[3]);
363  }
364  else{
365  point_line >> index >> elProperties[0];
366  element_properties[iElem_Local] = new CElementProperty(0, elProperties[0], 0, 0);
367  }
368 
369 
370  /*--- For backwards compatibility we only read a fifth column in topology mode ---*/
371  if (topology_mode) {
372  su2double elDensity;
373  point_line >> elDensity;
374  element_properties[iElem_Local]->SetDesignDensity(elDensity);
375  }
376 
377  iElem_Global_Local++;
378  }
379 
380  }
381 
382  /*--- Detect a wrong solution file ---*/
383 
384  if (iElem_Global_Local != nElement) {
385  SU2_MPI::Error(string("The properties file ") + filename + string(" doesn't match with the mesh file!\n") +
386  string("It could be empty lines at the end of the file."), CURRENT_FUNCTION);
387  }

556 void CFEASolver::Preprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, CNumerics **numerics,
557  unsigned short iMesh, unsigned long Iteration, unsigned short RunTime_EqSystem, bool Output) {
558 
559  const bool dynamic = config->GetTime_Domain();
560  const bool disc_adj_fem = (config->GetKind_Solver() == MAIN_SOLVER::DISC_ADJ_FEM);
561  const bool body_forces = config->GetDeadLoad();
562  const bool topology_mode = config->GetTopology_Optimization();
563 
564  /*
565  * For topology optimization we apply a filter on the design density field to avoid
566  * numerical issues (checkerboards), ensure mesh independence, and impose a length scale.
567  * This has to be done before computing the mass matrix and the dead load terms.
568  * This filter, and the volume fraction objective function, require the element volumes,
569  * so we ask "geometry" to compute them.
570  * This only needs to be done for the undeformed (initial) shape.
571  */
572  if (topology_mode && !topol_filter_applied) {
573  geometry->SetElemVolume();
574  FilterElementDensities(geometry, config);
575  topol_filter_applied = true;
576  }
577 
578  /*
579  * If the problem is dynamic we need a mass matrix, which will be constant along the calculation
580  * both for linear and nonlinear analysis, i.e. only computed once on the first time step.
581  *
582  * The same with the integration constants, as for now we consider the time step to be constant.
583  */
584  if (dynamic && (initial_calc || disc_adj_fem)) {
585  Compute_IntegrationConstants(config);
586  Compute_MassMatrix(geometry, numerics, config);
587  }
588 
589  /*
590  * If body forces are taken into account, we need to compute the term that goes into the residual,
591  * which will be constant along the calculation both for linear and nonlinear analysis.
592  *
593  * Only initialized once, at the first iteration of the first time step.
594  */
595  if (body_forces && (initial_calc || disc_adj_fem))
596  Compute_DeadLoad(geometry, numerics, config);
597 
598  /*--- Clear the linear system solution. ---*/
599  SU2_OMP_PARALLEL
600  {
601  LinSysSol.SetValZero();
602  }
603  END_SU2_OMP_PARALLEL
604 
605  /*--- Clear external forces. ---*/
606  nodes->Clear_SurfaceLoad_Res();
607 
608  /*--- Next call to Preprocessing will not be "initial_calc" and linear operations will not be repeated. ---*/
609  initial_calc = false;
610 
611 }
