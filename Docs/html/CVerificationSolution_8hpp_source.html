176  /*!
177  * \brief Increment the global maximum error for verification cases.
178  * \param[in] val_var - Index of the variable.
179  * \param[in] val_error - Value of the maximum error to store in the position <i>val_var</i>.
180  * \param[in] val_point - Value of the point index for the max error.
181  * \param[in] val_coord - Location (x, y, z) of the max error point.
182  */
183  void AddError_Max(unsigned short val_var, su2double val_error, unsigned long val_point, const su2double* val_coord) {
184  if (val_error > Error_Max[val_var]) {
185  Error_Max[val_var] = val_error;
186  Error_Point_Max[val_var] = val_point;
187  for (unsigned short iDim = 0; iDim < nDim; iDim++)
188  Error_Point_Max_Coord[val_var][iDim] = val_coord[iDim];
189  }
190  }
