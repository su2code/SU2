55 void CCentLax_AdjFlow::ComputeResidual (su2double *val_resconv_i, su2double *val_resvisc_i, su2double *val_resconv_j, su2double *val_resvisc_j,
56  su2double **val_Jacobian_ii, su2double **val_Jacobian_ij, su2double **val_Jacobian_ji, su2double **val_Jacobian_jj,
57  CConfig *config) {
58 
59  /*--- Mean value of the adjoint variables ---*/
60  MeanPsiRho = 0.5*(Psi_i[0]+Psi_j[0]);
61  for (iDim = 0; iDim < nDim; iDim++)
62  MeanPhi[iDim] = 0.5*(Psi_i[iDim+1]+Psi_j[iDim+1]);
63  MeanPsiE = 0.5*(Psi_i[nVar-1]+Psi_j[nVar-1]);
64 
65  /*--- Evaluation at point i ---*/
66  ProjVelocity_i = 0; ProjPhi = 0; ProjPhi_Vel = 0; sq_vel = 0;
67  for (iDim = 0; iDim < nDim; iDim++) {
68  Velocity_i[iDim] = U_i[iDim+1] / U_i[0];
69  ProjVelocity_i += Velocity_i[iDim]*Normal[iDim];
70  ProjPhi += MeanPhi[iDim]*Normal[iDim];
71  ProjPhi_Vel += MeanPhi[iDim]*Velocity_i[iDim];
72  sq_vel += 0.5*Velocity_i[iDim]*Velocity_i[iDim];
73  }
74  Area = GeometryToolbox::Norm(nDim, Normal);
75 
76  phis1 = ProjPhi + ProjVelocity_i*MeanPsiE;
77  phis2 = MeanPsiRho + ProjPhi_Vel + Enthalpy_i*MeanPsiE;
78 
79  /*--- Compute inviscid residual at point i ---*/
80  val_resconv_i[0] = ProjVelocity_i*MeanPsiRho - phis2*ProjVelocity_i + Gamma_Minus_One*phis1*sq_vel;
81  for (iDim = 0; iDim < nDim; iDim++)
82  val_resconv_i[iDim+1] = ProjVelocity_i*MeanPhi[iDim] + phis2*Normal[iDim] - Gamma_Minus_One*phis1*Velocity_i[iDim];
83  val_resconv_i[nVar-1] = ProjVelocity_i*MeanPsiE + Gamma_Minus_One*phis1;
84 
85  /*--- Flux contributions due to grid motion at point i ---*/
86  if (grid_movement) {
87  su2double ProjGridVel = 0.0;
88  for (iDim = 0; iDim < nDim; iDim++)
89  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
90  val_resconv_i[0] -= ProjGridVel*MeanPsiRho;
91  for (iDim = 0; iDim < nDim; iDim++)
92  val_resconv_i[iDim+1] -= ProjGridVel*MeanPhi[iDim];
93  val_resconv_i[nVar-1] -= ProjGridVel*MeanPsiE;
94  }
95 
96  /*--- Inviscid contribution to the implicit part ---*/
97  if (implicit) {
98  val_Jacobian_ii[0][0] = 0.0;
99  for (jDim = 0; jDim < nDim; jDim++)
100  val_Jacobian_ii[0][jDim+1] = -0.5*ProjVelocity_i*Velocity_i[jDim] + Gamma_Minus_One*sq_vel*0.5*Normal[jDim];
101  val_Jacobian_ii[0][nVar-1] = 0.5*ProjVelocity_i*(Gamma_Minus_One*sq_vel - Enthalpy_i);
102  for (iDim = 0; iDim < nDim; iDim++) {
103  val_Jacobian_ii[iDim+1][0] = 0.5*Normal[iDim];
104  for (jDim = 0; jDim < nDim; jDim++)
105  val_Jacobian_ii[iDim+1][jDim+1] = 0.5*Normal[iDim]*Velocity_i[jDim] - 0.5*Gamma_Minus_One*Velocity_i[iDim]*Normal[jDim];
106  val_Jacobian_ii[iDim+1][iDim+1] += 0.5*ProjVelocity_i;
107  val_Jacobian_ii[iDim+1][nVar-1] = 0.5*Enthalpy_i*Normal[iDim] - 0.5*Gamma_Minus_One*Velocity_i[iDim]*ProjVelocity_i;
108  }
109  val_Jacobian_ii[nVar-1][0] = 0;
110  for (jDim = 0; jDim < nDim; jDim++)
111  val_Jacobian_ii[nVar-1][jDim+1] = 0.5*Gamma_Minus_One*Normal[jDim];
112  val_Jacobian_ii[nVar-1][nVar-1] = 0.5*Gamma*ProjVelocity_i;
113 
114  for (iVar = 0; iVar < nVar; iVar++)
115  for (jVar = 0; jVar < nVar; jVar++)
116  val_Jacobian_ij[iVar][jVar] = val_Jacobian_ii[iVar][jVar];
117 
118  /*--- Jacobian contributions due to grid motion at point i ---*/
119  if (grid_movement) {
120  su2double ProjGridVel = 0.0;
121  for (iDim = 0; iDim < nDim; iDim++)
122  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
123  for (iVar = 0; iVar < nVar; iVar++) {
124  val_Jacobian_ii[iVar][iVar] -= 0.5*ProjGridVel;
125  val_Jacobian_ij[iVar][iVar] -= 0.5*ProjGridVel;
126  }
127  }
128  }
129 
130  /*--- Evaluation at point j ---*/
131  ProjVelocity_j = 0; ProjPhi_Vel = 0; sq_vel = 0;
132  for (iDim = 0; iDim < nDim; iDim++) {
133  Velocity_j[iDim] = U_j[iDim+1] / U_j[0];
134  ProjVelocity_j += Velocity_j[iDim]*Normal[iDim];
135  ProjPhi_Vel += MeanPhi[iDim]*Velocity_j[iDim];
136  sq_vel += 0.5*Velocity_j[iDim]*Velocity_j[iDim];
137  }
138 
139  phis1 = ProjPhi + ProjVelocity_j*MeanPsiE;
140  phis2 = MeanPsiRho + ProjPhi_Vel + Enthalpy_j*MeanPsiE;
141 
142  /*--- Compute inviscid residual at point j ---*/
143  val_resconv_j[0] = -(ProjVelocity_j*MeanPsiRho - phis2*ProjVelocity_j + Gamma_Minus_One*phis1*sq_vel);
144  for (iDim = 0; iDim < nDim; iDim++)
145  val_resconv_j[iDim+1] = -(ProjVelocity_j*MeanPhi[iDim] + phis2*Normal[iDim] - Gamma_Minus_One*phis1*Velocity_j[iDim]);
146  val_resconv_j[nVar-1] = -(ProjVelocity_j*MeanPsiE + Gamma_Minus_One*phis1);
147 
148  /*--- Flux contributions due to grid movement at point j ---*/
149  if (grid_movement) {
150  su2double ProjGridVel = 0.0;
151  for (iDim = 0; iDim < nDim; iDim++)
152  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
153  val_resconv_j[0] += ProjGridVel*MeanPsiRho;
154  for (iDim = 0; iDim < nDim; iDim++)
155  val_resconv_j[iDim+1] += ProjGridVel*MeanPhi[iDim];
156  val_resconv_j[nVar-1] += ProjGridVel*MeanPsiE;
157  }
158 
159  /*--- Inviscid contribution to the implicit part ---*/
160  if (implicit) {
161  val_Jacobian_jj[0][0] = 0.0;
162  for (jDim = 0; jDim < nDim; jDim++)
163  val_Jacobian_jj[0][jDim+1] = 0.5*ProjVelocity_j*Velocity_j[jDim] - Gamma_Minus_One*sq_vel*0.5*Normal[jDim];
164  val_Jacobian_jj[0][nVar-1] = -0.5*ProjVelocity_j*(Gamma_Minus_One*sq_vel - Enthalpy_j);
165  for (iDim = 0; iDim < nDim; iDim++) {
166  val_Jacobian_jj[iDim+1][0] = -0.5*Normal[iDim];
167  for (jDim = 0; jDim < nDim; jDim++)
168  val_Jacobian_jj[iDim+1][jDim+1] = -0.5*Normal[iDim]*Velocity_j[jDim] + 0.5*Gamma_Minus_One*Velocity_j[iDim]*Normal[jDim];
169  val_Jacobian_jj[iDim+1][iDim+1] -= 0.5*ProjVelocity_j;
170  val_Jacobian_jj[iDim+1][nVar-1] = -0.5*Enthalpy_j*Normal[iDim] + 0.5*Gamma_Minus_One*Velocity_j[iDim]*ProjVelocity_j;
171  }
172  val_Jacobian_jj[nVar-1][0] = 0;
173  for (jDim = 0; jDim < nDim; jDim++)
174  val_Jacobian_jj[nVar-1][jDim+1] = -0.5*Gamma_Minus_One*Normal[jDim];
175  val_Jacobian_jj[nVar-1][nVar-1] = -0.5*Gamma*ProjVelocity_j;
176 
177  for (iVar = 0; iVar < nVar; iVar++)
178  for (jVar = 0; jVar < nVar; jVar++)
179  val_Jacobian_ji[iVar][jVar] = val_Jacobian_jj[iVar][jVar];
180 
181  /*--- Jacobian contributions due to grid movement at point j ---*/
182  if (grid_movement) {
183  su2double ProjGridVel = 0.0;
184  for (iDim = 0; iDim < nDim; iDim++)
185  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
186  for (iVar = 0; iVar < nVar; iVar++) {
187  val_Jacobian_jj[iVar][iVar] += 0.5*ProjGridVel;
188  val_Jacobian_ji[iVar][iVar] += 0.5*ProjGridVel;
189  }
190  }
191  }
192 
193  /*--- Computes differences btw. variables ---*/
194  for (iVar = 0; iVar < nVar; iVar++)
195  Diff_Psi[iVar] = Psi_i[iVar]-Psi_j[iVar];
196 
197  /*--- Adjustment to projected velocity due to grid motion ---*/
198  if (grid_movement) {
199  su2double ProjGridVel = 0.0;
200  for (iDim = 0; iDim < nDim; iDim++)
201  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
202  ProjVelocity_i -= ProjGridVel;
203  ProjVelocity_j += ProjGridVel;
204  }
205 
206  /*--- Compute spectral radius ---*/
207  Local_Lambda_i = (fabs(ProjVelocity_i)+SoundSpeed_i*Area);
208  Local_Lambda_j = (fabs(ProjVelocity_j)+SoundSpeed_j*Area);
209  MeanLambda = 0.5*(Local_Lambda_i+Local_Lambda_j);
210 
211  /*--- Compute streching factor ---*/
212  Phi_i = pow(Lambda_i/(4.0*MeanLambda), Param_p);
213  Phi_j = pow(Lambda_j/(4.0*MeanLambda), Param_p);
214  StretchingFactor = 4.0*Phi_i*Phi_j/(Phi_i+Phi_j);
215 
216  sc2 = 3.0*(su2double(Neighbor_i)+su2double(Neighbor_j))/(su2double(Neighbor_i)*su2double(Neighbor_j));
217  Epsilon_0 = Param_Kappa_0*sc2*su2double(nDim)/3.0;
218 
219  /*--- Artifical dissipation evaluation ---*/
220  for (iVar = 0; iVar < nVar; iVar++) {
221  Residual = Epsilon_0*StretchingFactor*MeanLambda*Diff_Psi[iVar];
222  val_resvisc_i[iVar] = -Residual;
223  val_resvisc_j[iVar] = Residual;
224  }
225 
226  /*--- Contribution to implicit part ---*/
227  if (implicit) {
228  for (iVar = 0; iVar < nVar; iVar++) {
229  val_Jacobian_ii[iVar][iVar] -= Epsilon_0*StretchingFactor*MeanLambda;
230  val_Jacobian_ij[iVar][iVar] += Epsilon_0*StretchingFactor*MeanLambda;
231  val_Jacobian_ji[iVar][iVar] += Epsilon_0*StretchingFactor*MeanLambda;
232  val_Jacobian_jj[iVar][iVar] -= Epsilon_0*StretchingFactor*MeanLambda;
233  }
234  }

402  val_Jacobian_jj[iVar][iVar] += 0.5*ProjGridVel;
403  val_Jacobian_ji[iVar][iVar] += 0.5*ProjGridVel;
404  }
405  }
406  }
407 
408  /*--- Computes differences btw. variables and Laplacians ---*/
409 
410  for (iVar = 0; iVar < nVar; iVar++) {
411  Diff_Lapl[iVar] = Und_Lapl_i[iVar]-Und_Lapl_j[iVar];
412  Diff_Psi[iVar] = Psi_i[iVar]-Psi_j[iVar];
413  }
414 
415  /*--- Adjustment to projected velocity due to grid motion ---*/
416 
417  if (grid_movement) {
418  su2double ProjGridVel = 0.0;
419  for (iDim = 0; iDim < nDim; iDim++) {
420  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
421  }
422  ProjVelocity_i -= ProjGridVel;
423  ProjVelocity_j += ProjGridVel;
424  }
425 
426  /*--- Compute the spectral radius and stretching factor ---*/
427 
428  Local_Lambda_i = (fabs(ProjVelocity_i)+SoundSpeed_i*Area);
429  Local_Lambda_j = (fabs(ProjVelocity_j)+SoundSpeed_j*Area);
430  MeanLambda = 0.5*(Local_Lambda_i+Local_Lambda_j);
431 
432  Phi_i = pow(Lambda_i/(4.0*MeanLambda), Param_p);
433  Phi_j = pow(Lambda_j/(4.0*MeanLambda), Param_p);
434  StretchingFactor = 4.0*Phi_i*Phi_j/(Phi_i+Phi_j);
435 
436  su2double sc2 = 3.0*(su2double(Neighbor_i)+su2double(Neighbor_j))/(su2double(Neighbor_i)*su2double(Neighbor_j));
437  sc4 = sc2*sc2/4.0;
438  Epsilon_2 = Param_Kappa_2*0.5*(Sensor_i+Sensor_j)*sc2;
439  Epsilon_4 = max(0.0, Param_Kappa_4-Epsilon_2)*sc4;
440 
441  /*--- Compute viscous residual 1st- & 3rd-order dissipation ---*/
442 
443  for (iVar = 0; iVar < nVar; iVar++) {
444  Residual = (Epsilon_2*Diff_Psi[iVar]-Epsilon_4*Diff_Lapl[iVar])*StretchingFactor*MeanLambda;
445  val_resvisc_i[iVar] = -Residual;
446  val_resvisc_j[iVar] = Residual;
447  if (implicit) {
448  val_Jacobian_ii[iVar][iVar] -= Epsilon_2 + su2double(Neighbor_i+1)*Epsilon_4*StretchingFactor*MeanLambda;
449  val_Jacobian_ij[iVar][iVar] += Epsilon_2 + su2double(Neighbor_j+1)*Epsilon_4*StretchingFactor*MeanLambda;
450  val_Jacobian_ji[iVar][iVar] += Epsilon_2 + su2double(Neighbor_i+1)*Epsilon_4*StretchingFactor*MeanLambda;
451  val_Jacobian_jj[iVar][iVar] -= Epsilon_2 + su2double(Neighbor_j+1)*Epsilon_4*StretchingFactor*MeanLambda;
452  }
453  }

508 void CUpwRoe_AdjFlow::ComputeResidual (su2double *val_residual_i, su2double *val_residual_j, su2double **val_Jacobian_ii,
509  su2double **val_Jacobian_ij, su2double **val_Jacobian_ji, su2double **val_Jacobian_jj, CConfig *config) {
510 
511  /*--- Compute the area ---*/
512 
513  area = 0.0;
514  for (iDim = 0; iDim < nDim; iDim++)
515  area += Normal[iDim]*Normal[iDim];
516  area = sqrt(area);
517  rarea = 1.0 / area;
518 
519  /*--- Components of the normal & unit normal vector of the current face ---*/
520 
521  Sx = Normal[0];
522  Sy = Normal[1];
523  Sz = 0.0; if (nDim == 3) Sz = Normal[2];
524  nx = Sx * rarea;
525  ny = Sy * rarea;
526  nz = Sz * rarea;
527 
528  /*--- Flow variable states at point i (left, _l) and j (right, _r)---*/
529 
530  rho_l = V_i[nDim+2]; rho_r = V_j[nDim+2];
531  u_l = V_i[1]; v_l = V_i[2]; w_l = 0.0; if (nDim == 3) w_l = V_i[3];
532  u_r = V_j[1]; v_r = V_j[2]; w_r = 0.0; if (nDim == 3) w_r = V_j[3];
533  h_l = V_i[nDim+3]; h_r = V_j[nDim+3];
534 
535  /*--- One-half speed squared ---*/
536 
537  q_l = ONE2 * ((u_l*u_l) + (v_l*v_l) + (w_l*w_l));
538  q_r = ONE2 * ((u_r*u_r) + (v_r*v_r) + (w_r*w_r));
539 
540  /*--- Projected velocity ---*/
541 
542  Q_l = (u_l * Sx) + (v_l * Sy) + (w_l * Sz);
543  Q_r = (u_r * Sx) + (v_r * Sy) + (w_r * Sz);
544 
545  /*--- Mean adjoint variables ---*/
546 
547  psi1 = ONE2 * (Psi_i[0] + Psi_j[0]);
548  psi2 = ONE2 * (Psi_i[1] + Psi_j[1]);
549  psi3 = ONE2 * (Psi_i[2] + Psi_j[2]);
550  psi4 = 0.0; if (nDim == 3) psi4 = ONE2 * (Psi_i[3] + Psi_j[3]);
551  psi5 = ONE2 * (Psi_i[nVar-1] + Psi_j[nVar-1]);
552 
553  /*--- Left state ---*/
554 
555  l1psi = (Sx * psi2) + (Sy * psi3) + (Sz * psi4) + (Q_l * psi5);
556  l2psi = psi1 + (u_l * psi2) + (v_l * psi3) + (w_l * psi4) + (h_l * psi5);
557 
558  val_residual_i[0] = Q_l * psi1 - l2psi * Q_l + l1psi * Gamma_Minus_One * q_l;
559  val_residual_i[1] = Q_l * psi2 + l2psi * Sx - l1psi * Gamma_Minus_One * u_l;
560  val_residual_i[2] = Q_l * psi3 + l2psi * Sy - l1psi * Gamma_Minus_One * v_l;
561  if (nDim == 3) val_residual_i[3] = Q_l * psi4 + l2psi * Sz - l1psi * Gamma_Minus_One * w_l;
562  val_residual_i[nVar-1] = Q_l * psi5 + l1psi * Gamma_Minus_One;
563 
564  /*--- Right state ---*/
565 
566  l1psi = (Sx * psi2) + (Sy * psi3) + (Sz * psi4) + (Q_r * psi5);
567  l2psi = psi1 + (u_r * psi2) + (v_r * psi3) + (w_r * psi4) + (h_r * psi5);
568 
569  val_residual_j[0] = -(Q_r * psi1 - l2psi * Q_r + l1psi * Gamma_Minus_One * q_r);
570  val_residual_j[1] = -(Q_r * psi2 + l2psi * Sx - l1psi * Gamma_Minus_One * u_r);
571  val_residual_j[2] = -(Q_r * psi3 + l2psi * Sy - l1psi * Gamma_Minus_One * v_r);
572  if (nDim == 3) val_residual_j[3] = -(Q_r * psi4 + l2psi * Sz - l1psi * Gamma_Minus_One * w_r);
573  val_residual_j[nVar-1] = -(Q_r * psi5 + l1psi * Gamma_Minus_One);
574 
575 
576  /*--- f_{roe} = P^{-T} |lambda| P^T \delta \psi ---*/
577 
578  psi1_l = Psi_i[0];
579  psi2_l = Psi_i[1];
580  psi3_l = Psi_i[2];
581  psi4_l = 0.0; if (nDim == 3) psi4_l = Psi_i[3];
582  psi5_l = Psi_i[nVar-1];
583 
584  psi1_r = Psi_j[0];
585  psi2_r = Psi_j[1];
586  psi3_r = Psi_j[2];
587  psi4_r = 0.0; if (nDim == 3) psi4_r = Psi_j[3];
588  psi5_r = Psi_j[nVar-1];
589 
590  /*--- Roe averaging ---*/
591 
592  rrho_l = 1.0 / rho_l;
593  weight = sqrt(rho_r * rrho_l);
594  rweight1 = 1.0 / (1.0 + weight);
595  weight *= rweight1;
596 
597  h = h_l * rweight1 + weight * h_r;
598  u = u_l * rweight1 + weight * u_r;
599  v = v_l * rweight1 + weight * v_r;
600  w = w_l * rweight1 + weight * w_r;
601 
602  psi1 = ONE2 * (psi1_r - psi1_l);
603  psi2 = ONE2 * (psi2_r - psi2_l);
604  psi3 = ONE2 * (psi3_r - psi3_l);
605  psi4 = ONE2 * (psi4_r - psi4_l);
606  psi5 = ONE2 * (psi5_r - psi5_l);
607 
608  q2 = (u*u) + (v*v) + (w*w);
609  Q = (u * Sx) + (v * Sy) + (w * Sz);
610  vn = nx * u + ny * v + nz * w;
611  cc = Gamma_Minus_One * h - 0.5 * Gamma_Minus_One * q2;
612  c = sqrt(cc);
613 
614  /*--- Contribution to velocity projection due to grid movement ---*/
615 
616  if (grid_movement) {
617  su2double ProjGridVel = 0.0;
618  for (iDim = 0; iDim < nDim; iDim++)
619  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
620  Q -= ProjGridVel;
621  }
622 
623  /*--- Eigenvalues from the primal solution ---*/
624 
625  absQ = fabs(Q);
626  absQp = fabs(Q + c * area);
627  absQm = fabs(Q - c * area);
628 
629  alpha = ONE2 * Gamma_Minus_One * q2 / cc;
630  beta_u = psi2 + u * psi5;
631  beta_v = psi3 + v * psi5;
632  beta_w = psi4 + w * psi5;
633  eta = Gamma_Minus_One / cc;
634  l1psi = (nx * psi2) + (ny * psi3) + (nz * psi4) + (vn * psi5);
635  l2psi = psi1 + (u * psi2) + (v * psi3) + (w * psi4) + (h * psi5);
636  l1l2p = (l2psi + c * l1psi) * absQp;
637  l1l2m = (l2psi - c * l1psi) * absQm;
638 
639  /*--- adjoint flux computation in the x, y and z coordinate system ---*/
640 
641  Residual_Roe[0] = ((1.0-alpha)*l2psi - (1.0-alpha)*cc/Gamma_Minus_One*psi5
642  - u*beta_u*(1.0-(nx*nx)) - v*beta_v*(1.0-(ny*ny))
643  - w*beta_w*(1.0-(nz*nz)) + ny*nz*(w*beta_v + v*beta_w)
644  + nx*nz*(w*beta_u + u*beta_w) + ny*nx*(v*beta_u + u*beta_v) ) * absQ
645  - ONE2 / c * vn * (l1l2p - l1l2m) + ONE2 * alpha * (l1l2p + l1l2m);
646 
647  Residual_Roe[1] = (l2psi*u*eta - u*psi5 + beta_u*(1.0-(nx*nx))
648  - nx*(beta_v*ny + beta_w*nz) ) * absQ + ONE2*nx/c * (l1l2p - l1l2m )
649  - ONE2*eta*u * (l1l2p + l1l2m );
650 
651  Residual_Roe[2] = (l2psi*v*eta - v*psi5 + beta_v*(1.0-(ny*ny))
652  - ny*(beta_w*nz + beta_u*nx) ) * absQ + ONE2*ny/c * (l1l2p - l1l2m )
653  - ONE2*eta*v * (l1l2p + l1l2m );
654 
655  if (nDim == 3) Residual_Roe[3] = (l2psi*w*eta - w*psi5 + beta_w*(1.0-(nz*nz)) - nz*(beta_u*nx + beta_v*ny) ) * absQ
656  + ONE2*nz/c * (l1l2p - l1l2m ) - ONE2*eta*w * (l1l2p + l1l2m );
657 
658  Residual_Roe[nVar-1] = (psi5 - l2psi*eta) * absQ + ONE2*eta*(l1l2p + l1l2m);
659 
660  for (iVar = 0; iVar < nVar; iVar++) {
661  val_residual_i[iVar] += Residual_Roe[iVar];
662  val_residual_j[iVar] -= Residual_Roe[iVar];
663  }
664 
665  /*--- Flux contribution due to grid movement ---*/
666 
667  if (grid_movement) {
668  su2double ProjGridVel = 0.0;
669  for (iDim = 0; iDim < nDim; iDim++)
670  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
671  for (iVar = 0; iVar < nVar; iVar++) {
672  val_residual_i[iVar] -= ProjGridVel * 0.5*(Psi_i[iVar]+Psi_j[iVar]);
673  val_residual_j[iVar] += ProjGridVel * 0.5*(Psi_i[iVar]+Psi_j[iVar]);
674  }
675  }
676 
677  /*--- Implicit Contributions ---*/
678 
679  if (implicit) {
680 
681  /*--- Prepare variables for use in matrix routines ---*/
682 
683  RoeDensity = V_i[nDim+2]*sqrt(V_j[nDim+2]/V_i[nDim+2]);
684  RoeSoundSpeed = c;
685  UnitNormal[0] = nx; UnitNormal[1] = ny; if (nDim == 3 ) UnitNormal[2] = nz;
686  RoeVelocity[0] = u; RoeVelocity[1] = v; if (nDim == 3 ) RoeVelocity[2] = w;
687  Velocity_i[0] = u_l; Velocity_i[1] = v_l; if (nDim == 3 ) Velocity_i[2] = w_l;
688  Velocity_j[0] = u_r; Velocity_j[1] = v_r; if (nDim == 3 ) Velocity_j[2] = w_r;
689 
690  Pressure_i = V_i[nDim+1];
691  Density_i = V_i[nDim+2];
692  Enthalpy_i = V_i[nDim+3];
693  Energy_i = Enthalpy_i - Pressure_i/Density_i;
694 
695  Pressure_j = V_i[nDim+1];
696  Density_j = V_i[nDim+2];
697  Enthalpy_j = V_i[nDim+3];
698  Energy_j = Enthalpy_j - Pressure_j/Density_j;
699 
700  /*--- Jacobians of the inviscid flux, scaled by
701  0.5 because val_resconv ~ 0.5*(fc_i+fc_j)*Normal ---*/
702 
703  GetInviscidProjJac(Velocity_i, &Energy_i, Normal, 0.5, ProjFlux_i);
704  GetInviscidProjJac(Velocity_j, &Energy_j, Normal, 0.5, ProjFlux_j);
705 
706  /*--- Compute P, inverse P, and store eigenvalues ---*/
707 
708  GetPMatrix_inv(&RoeDensity, RoeVelocity, &RoeSoundSpeed, UnitNormal, invP_Tensor);
709  GetPMatrix(&RoeDensity, RoeVelocity, &RoeSoundSpeed, UnitNormal, P_Tensor);
710 
711  /*--- Flow eigenvalues ---*/
712 
713  for (iDim = 0; iDim < nDim; iDim++)
714  Lambda[iDim] = absQ;
715  Lambda[nVar-2] = absQp;
716  Lambda[nVar-1] = absQm;
717 
718  /*--- Roe's Flux approximation ---*/
719 
720  for (iVar = 0; iVar < nVar; iVar++) {
721  for (jVar = 0; jVar < nVar; jVar++) {
722  Proj_ModJac_Tensor_ij = 0.0;
723 
724  /*--- Compute |Proj_ModJac_Tensor| = P x |Lambda| x inverse P ---*/
725 
726  for (kVar = 0; kVar < nVar; kVar++)
727  Proj_ModJac_Tensor_ij += P_Tensor[iVar][kVar]*Lambda[kVar]*invP_Tensor[kVar][jVar];
728  Proj_ModJac_Tensor[iVar][jVar] = 0.5*Proj_ModJac_Tensor_ij*area;
729  }
730  }
731 
732  /*--- Transpose the matrices and store the Jacobians. Note the negative
733  sign for the ji and jj Jacobians bc the normal direction is flipped. ---*/
734 
735  for (iVar = 0; iVar < nVar; iVar++) {
736  for (jVar = 0; jVar < nVar; jVar++) {
737  val_Jacobian_ii[jVar][iVar] = ProjFlux_i[iVar][jVar] - Proj_ModJac_Tensor[iVar][jVar];
738  val_Jacobian_ij[jVar][iVar] = ProjFlux_i[iVar][jVar] + Proj_ModJac_Tensor[iVar][jVar];
739  val_Jacobian_ji[jVar][iVar] = -(ProjFlux_j[iVar][jVar] - Proj_ModJac_Tensor[iVar][jVar]);
740  val_Jacobian_jj[jVar][iVar] = -(ProjFlux_j[iVar][jVar] + Proj_ModJac_Tensor[iVar][jVar]);
741  }
742  }
743 
744  /*--- Jacobian contribution due to grid movement ---*/
745 
746  if (grid_movement) {
747  su2double ProjGridVel = 0.0;
748  for (iDim = 0; iDim < nDim; iDim++)
749  ProjGridVel += 0.5*(GridVel_i[iDim]+GridVel_j[iDim])*Normal[iDim];
750  for (iVar = 0; iVar < nVar; iVar++) {
751 
752  /*--- Adjust Jacobian main diagonal ---*/
753 
754  val_Jacobian_ii[iVar][iVar] -= 0.5*ProjGridVel;
755  val_Jacobian_ij[iVar][iVar] -= 0.5*ProjGridVel;
756  val_Jacobian_ji[iVar][iVar] += 0.5*ProjGridVel;
757  val_Jacobian_jj[iVar][iVar] += 0.5*ProjGridVel;
758  }
759  }
760 
761  }

823  B^{cv} = -\nabla \hat{nu}/\sigma + v ---*/

825  unsigned short iDim;
826  su2double proj_conv_flux_i = 0, proj_conv_flux_j = 0, proj_conv_flux_ij = 0;
827  su2double sigma = 2./3.;
828

832  proj_conv_flux_i += (ScalarVar_Grad_i[0][iDim]/sigma - Velocity_i[iDim])*Normal[iDim]; // projection of convective flux at iPoint
833  proj_conv_flux_j += (ScalarVar_Grad_j[0][iDim]/sigma - Velocity_j[iDim])*Normal[iDim]; // projection of convective flux at jPoint
834  }
