61 /*!
62  * \class CSourceBase_TurbSA
63  * \ingroup SourceDiscr
64  * \brief Class for integrating the source terms of the Spalart-Allmaras turbulence model equation.
65  * The variables that are subject to change in each variation/correction have their own class.
66  * \note Additional source terms (e.g. compressibility) are implemented as decorators.
67  */
68 template <class FlowIndices, class Omega, class ft2, class ModVort, class rFunc, class SourceTerms>
69 class CSourceBase_TurbSA : public CNumerics {
70  protected:
71 
72  /*--- Residual and Jacobian ---*/
73  su2double Residual, *Jacobian_i;
74  su2double Jacobian_Buffer; /*!< \brief Static storage for the Jacobian (which needs to be pointer for return type). */
75 
76  const FlowIndices idx; /*!< \brief Object to manage the access to the flow primitives. */
77  const SA_ParsedOptions options; /*!< \brief Struct with SA options. */
78 
79  bool transition_LM;
80 
81  public:
82  /*!
83  * \brief Constructor of the class.
84  * \param[in] nDim - Number of dimensions of the problem.
85  * \param[in] config - Definition of the particular problem.
86  */
87  CSourceBase_TurbSA(unsigned short nDim, const CConfig* config)
88  : CNumerics(nDim, 1, config),
89  idx(nDim, config->GetnSpecies()),
90  options(config->GetSAParsedOptions()),
91  transition_LM(config->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
92  /*--- Setup the Jacobian pointer, we need to return su2double** but we know
93  * the Jacobian is 1x1 so we use this trick to avoid heap allocation. ---*/
94  Jacobian_i = &Jacobian_Buffer;
95  }
96 
97 
98  /*!
99  * \brief Residual for source term integration.
100  * \param[in] config - Definition of the particular problem.
101  * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
102  */
103  ResidualType<> ComputeResidual(const CConfig* config) override {
104  const auto& density = V_i[idx.Density()];
105  const auto& laminar_viscosity = V_i[idx.LaminarViscosity()];
106 
107  AD::StartPreacc();
108  AD::SetPreaccIn(density, laminar_viscosity, StrainMag_i, ScalarVar_i[0], Volume, dist_i, roughness_i);
109  AD::SetPreaccIn(Vorticity_i, 3);
110  AD::SetPreaccIn(PrimVar_Grad_i + idx.Velocity(), nDim, nDim);
111  AD::SetPreaccIn(ScalarVar_Grad_i[0], nDim);
112 
113  /*--- Common auxiliary variables and constants of the model. ---*/
114  CSAVariables var;
115 
116  Residual = 0.0;
117  Jacobian_i[0] = 0.0;
118 
119  /*--- Evaluate Omega with a rotational correction term. ---*/
120 
121  Omega::get(Vorticity_i, nDim, PrimVar_Grad_i + idx.Velocity(), var);
122 
123  /*--- Dacles-Mariani et. al. rotation correction ("-R"). ---*/
124  if (options.rot) {
125  var.Omega += 2.0 * min(0.0, StrainMag_i - var.Omega);
126  }
127 
128  if (dist_i > 1e-10) {
129  /*--- Vorticity ---*/
130  var.S = var.Omega;
131 
132  var.dist_i_2 = pow(dist_i, 2);
133  const su2double nu = laminar_viscosity / density;
134  var.inv_k2_d2 = 1.0 / (var.k2 * var.dist_i_2);
135 
136  /*--- Modified values for roughness, roughness_i = 0 for smooth walls and Ji remains the same.
137  * Ref: Aupoix, B. and Spalart, P. R., "Extensions of the Spalart-Allmaras Turbulence Model to Account for Wall
138  * Roughness," International Journal of Heat and Fluid Flow, Vol. 24, 2003, pp. 454-462.
139  * See https://turbmodels.larc.nasa.gov/spalart.html#sarough for detailed explanation. ---*/
140  var.Ji = ScalarVar_i[0] / nu + var.cr1 * (roughness_i / (dist_i + EPS));
141  var.d_Ji = 1.0 / nu;
142 
143  const su2double Ji_2 = pow(var.Ji, 2);
144  const su2double Ji_3 = Ji_2 * var.Ji;
145 
146  var.fv1 = Ji_3 / (Ji_3 + var.cv1_3);
147  var.d_fv1 = 3 * Ji_2 * var.cv1_3 / (nu * pow(Ji_3 + var.cv1_3, 2));
148 
149  /*--- Using a modified relation so as to not change the Shat that depends on fv2.
150  * From NASA turb modeling resource and 2003 paper. ---*/
151  var.fv2 = 1 - ScalarVar_i[0] / (nu + ScalarVar_i[0] * var.fv1);
152  var.d_fv2 = -(1 / nu - Ji_2 * var.d_fv1) / pow(1 + var.Ji * var.fv1, 2);
153 
154  /*--- Compute ft2 term ---*/
155  ft2::get(var);
156 
157  /*--- Compute modified vorticity ---*/
158  ModVort::get(ScalarVar_i[0], nu, var);
159  var.inv_Shat = 1.0 / var.Shat;
160 
161  /*--- Compute auxiliary function r ---*/
162  rFunc::get(ScalarVar_i[0], var);
163 
164  var.g = var.r + var.cw2 * (pow(var.r, 6) - var.r);
165  var.g_6 = pow(var.g, 6);
166  var.glim = pow((1 + var.cw3_6) / (var.g_6 + var.cw3_6), 1.0 / 6.0);
167  var.fw = var.g * var.glim;
168 
169  var.d_g = var.d_r * (1 + var.cw2 * (6 * pow(var.r, 5) - 1));
170  var.d_fw = var.d_g * var.glim * (1 - var.g_6 / (var.g_6 + var.cw3_6));
171 
172  var.norm2_Grad = GeometryToolbox::SquaredNorm(nDim, ScalarVar_Grad_i[0]);
173 
174  if (options.bc) {
175  /*--- BC transition model (2020 revision). This should only be used with SA-noft2.
176  * TODO: Consider making this part of the "SourceTerms" template. ---*/
177  const su2double chi_1 = 0.002;
178  const su2double chi_2 = 50.0;
179 
180  /*--- Turbulence intensity is u'/U so we multiply by 100 to get percentage. ---*/
181  const su2double tu = 100.0 * config->GetTurbulenceIntensity_FreeStream();
182  const su2double nu_t = ScalarVar_i[0] * var.fv1;
183 
184  const su2double re_v = density * var.dist_i_2 / laminar_viscosity * var.Omega;
185  const su2double re_theta = re_v / 2.193;
186  /*--- Menter correlation. ---*/
187  const su2double re_theta_t = 803.73 * pow(tu + 0.6067, -1.027);
188 
189  const su2double term1 = sqrt(max(re_theta - re_theta_t, 0.0) / (chi_1 * re_theta_t));
190  const su2double term2 = sqrt(max((nu_t * chi_2) / nu, 0.0));
191  const su2double term_exponential = (term1 + term2);
192 
193  intermittency_eff_i = 1.0 - exp(-term_exponential);
194  var.intermittency = intermittency_eff_i;
195  var.interDestrFactor = 1;
196 
197  } else if (transition_LM){
198 
199  var.intermittency = intermittency_eff_i;
200  //var.intermittency = 1.0;
201  // Is wrong the reference from NASA?
202  // Original max(min(gamma, 0.5), 1.0) always gives 1 as result.
203  var.interDestrFactor = min(max(intermittency_i, 0.5), 1.0);
204 
205  } else {
206  /*--- Do not modify the production. ---*/
207  var.intermittency = 1.0;
208  var.interDestrFactor = 1.0;
209  }
210 
211  /*--- Compute production, destruction and cross production and jacobian ---*/
212  su2double Production = 0.0, Destruction = 0.0, CrossProduction = 0.0;
213  SourceTerms::get(ScalarVar_i[0], var, Production, Destruction, CrossProduction, Jacobian_i[0]);
214 
215  Residual = (Production - Destruction + CrossProduction) * Volume;
216  Jacobian_i[0] *= Volume;
217  }
218 
219  AD::SetPreaccOut(Residual);
220  AD::EndPreacc();
221 
222  return ResidualType<>(&Residual, &Jacobian_i, nullptr);
223  }
224 };

250 /*! \brief Edward. Here Omega is the Strain Rate. */
251 struct Edw {
252  template <class MatrixType>
253  static void get(const su2double*, unsigned short nDim, const MatrixType& velocity_grad, CSAVariables& var) {
254  su2double Sbar = 0.0;
255  for (unsigned short iDim = 0; iDim < nDim; ++iDim) {
256  for (unsigned short jDim = 0; jDim < nDim; ++jDim) {
257  Sbar += (velocity_grad[iDim][jDim] + velocity_grad[jDim][iDim]) * velocity_grad[iDim][jDim];
258  }
259  }
260  for (unsigned short iDim = 0; iDim < nDim; ++iDim) {
261  Sbar -= (2.0 / 3.0) * pow(velocity_grad[iDim][iDim], 2);
262  }
263  var.Omega = sqrt(max(Sbar, 0.0));
264  }
265 };

284 struct Nonzero {
285  static void get(CSAVariables& var) {
286  const su2double xsi2 = pow(var.Ji, 2);
287  var.ft2 = var.ct3 * exp(-var.ct4 * xsi2);
288  var.d_ft2 = -2.0 * var.ct4 * var.Ji * var.ft2 * var.d_Ji;
289  }

293 /*!
294  * \brief SA classes to compute the modified vorticity (\tilde{S}) and its derivative.
295  * \ingroup SourceDiscr
296  * \param[in] nue: SA variable.
297  * \param[in] nu: Laminar viscosity.
298  * \param[in,out] var: Common SA variables struct.
299  */
300 struct ModVort {
301 
302 /*! \brief Baseline. */
303 struct Bsl {
304  static void get(const su2double& nue, const su2double& nu, CSAVariables& var) {
305  const su2double Sbar = nue * var.fv2 * var.inv_k2_d2;
306  var.Shat = var.S + Sbar;
307  var.Shat = max(var.Shat, 1.0e-10);
308  if (var.Shat <= 1.0e-10) {
309  var.d_Shat = 0.0;
310  } else {
311  var.d_Shat = (var.fv2 + nue * var.d_fv2) * var.inv_k2_d2;
312  }
313  }
314 };
315 
316 /*! \brief Edward. */
317 struct Edw {
318  static void get(const su2double& nue, const su2double& nu, CSAVariables& var) {
319  var.Shat = max(var.S * ((1.0 / max(var.Ji, 1.0e-16)) + var.fv1), 1.0e-16);
320  var.Shat = max(var.Shat, 1.0e-10);
321  if (var.Shat <= 1.0e-10) {
322  var.d_Shat = 0.0;
323  } else {
324  var.d_Shat = -var.S * pow(var.Ji, -2) / nu + var.S * var.d_fv1;
325  }
326  }
327 };
328 
329 /*! \brief Negative. */
330 struct Neg {
331  static void get(const su2double& nue, const su2double& nu, CSAVariables& var) {
332  if (nue > 0.0) {
333  // Baseline solution
334  Bsl::get(nue, nu, var);
335  } else {
336  var.Shat = 1.0e-10;
337  var.d_Shat = 0.0;
338  }
339  /*--- Don't check whether Sbar <>= -cv2*S.
340  * Steven R. Allmaras, Forrester T. Johnson and Philippe R. Spalart -
341  * "Modifications and Clarifications for the Implementation of the Spalart-Allmaras Turbulence Model" eq. 12
342  * No need for Sbar ---*/
343  }
344 };
345 };

347 /*!
348  * \brief SA auxiliary function r and its derivative.
349  * \ingroup SourceDiscr
350  * \param[in] nue: SA variable.
351  * \param[in,out] var: Common SA variables struct.
352  */
353 struct r {
354 
355 /*! \brief Baseline. */
356 struct Bsl {
357  static void get(const su2double& nue, CSAVariables& var) {
358  var.r = min(nue * var.inv_Shat * var.inv_k2_d2, 10.0);
359  var.d_r = (var.Shat - nue * var.d_Shat) * pow(var.inv_Shat, 2) * var.inv_k2_d2;
360  if (var.r >= 10.0) var.d_r = 0.0;
361  }
362 };
363 
364 /*! \brief Edward. */
365 struct Edw {
366  static void get(const su2double& nue, CSAVariables& var) {
367  var.r = min(nue * var.inv_Shat * var.inv_k2_d2, 10.0);
368  var.r = tanh(var.r) / tanh(1.0);
369 
370  var.d_r = (var.Shat - nue * var.d_Shat) * pow(var.inv_Shat, 2) * var.inv_k2_d2;
371  var.d_r = (1 - pow(tanh(var.r), 2.0)) * (var.d_r) / tanh(1.0);
372  }
373 };

376 /*!
377  * \brief SA source terms classes: production, destruction and cross-productions term and their derivative.
378  * \ingroup SourceDiscr
379  * \param[in] nue: SA variable.
380  * \param[in] var: Common SA variables struct.
381  * \param[out] production: Production term.
382  * \param[out] destruction: Destruction term.
383  * \param[out] cross_production: CrossProduction term.
384  * \param[out] jacobian: Derivative of the combined source term wrt nue.
385  */
386 struct SourceTerms {
387 
388 /*! \brief Baseline (Original SA model). */
389 struct Bsl {
390  static void get(const su2double& nue, const CSAVariables& var, su2double& production, su2double& destruction,
391  su2double& cross_production, su2double& jacobian) {
392  ComputeProduction(nue, var, production, jacobian);
393  ComputeDestruction(nue, var, destruction, jacobian);
394  ComputeCrossProduction(nue, var, cross_production, jacobian);
395  }
396 
397  static void ComputeProduction(const su2double& nue, const CSAVariables& var, su2double& production,
398  su2double& jacobian) {
399  const su2double factor = var.intermittency * var.cb1;
400  production = factor * (1.0 - var.ft2) * var.Shat * nue;
401  jacobian += factor * (-var.Shat * nue * var.d_ft2 + (1.0 - var.ft2) * (nue * var.d_Shat + var.Shat));
402  }
403 
404  static void ComputeDestruction(const su2double& nue, const CSAVariables& var, su2double& destruction,
405  su2double& jacobian) {
406  const su2double cb1_k2 = var.cb1 / var.k2;
407  const su2double factor = var.cw1 * var.fw - cb1_k2 * var.ft2;
408  destruction = var.interDestrFactor * factor * pow(nue, 2) / var.dist_i_2;
409  jacobian -= var.interDestrFactor * ((var.cw1 * var.d_fw - cb1_k2 * var.d_ft2) * pow(nue, 2) + factor * 2 * nue) / var.dist_i_2;
410  }
411 
412  static void ComputeCrossProduction(const su2double& nue, const CSAVariables& var, su2double& cross_production,
413  su2double&) {
414  cross_production = var.cb2_sigma * var.norm2_Grad;
415  /*--- No contribution to the jacobian. ---*/
416  }
417 };
418 
419 /*! \brief Negative. */
420 struct Neg {
421  static void get(const su2double& nue, const CSAVariables& var, su2double& production, su2double& destruction,
422  su2double& cross_production, su2double& jacobian) {
423  if (nue > 0.0) {
424  Bsl::get(nue, var, production, destruction, cross_production, jacobian);
425  } else {
426  ComputeProduction(nue, var, production, jacobian);
427  ComputeDestruction(nue, var, destruction, jacobian);
428  ComputeCrossProduction(nue, var, cross_production, jacobian);
429  }
430  }
431 
432  static void ComputeProduction(const su2double& nue, const CSAVariables& var, su2double& production,
433  su2double& jacobian) {
434  const su2double dP_dnu = var.intermittency * var.cb1 * (1.0 - var.ct3) * var.S;
435  production = dP_dnu * nue;
436  jacobian += dP_dnu;
437  }
438 
439  static void ComputeDestruction(const su2double& nue, const CSAVariables& var, su2double& destruction,
440  su2double& jacobian) {
441  /*--- The destruction when nue < 0 is added instead of the usual subtraction, hence the negative sign. ---*/
442  const su2double dD_dnu = -var.cw1 * nue / var.dist_i_2;
443  destruction = dD_dnu * nue * var.interDestrFactor;
444  jacobian -= 2 * dD_dnu * var.interDestrFactor;
445  }
446 
447  static void ComputeCrossProduction(const su2double& nue, const CSAVariables& var, su2double& cross_production,
448  su2double& jacobian) {
449  Bsl::ComputeCrossProduction(nue, var, cross_production, jacobian);
450  }
451 };
452 };

458 /*!
459  * \class CCompressibilityCorrection
460  * \ingroup SourceDiscr
461  * \brief Mixing Layer Compressibility Correction (SA-comp).
462  */
463 template <class ParentClass>
464 class CCompressibilityCorrection final : public ParentClass {
465  private:
466  using ParentClass::Gamma;
467  using ParentClass::idx;
468  using ParentClass::nDim;
469  using ParentClass::PrimVar_Grad_i;
470  using ParentClass::ScalarVar_i;
471  using ParentClass::V_i;
472  using ParentClass::Volume;
473 
474  using ResidualType = typename ParentClass::template ResidualType<>;
475 
476  const su2double c5 = 3.5;
477 
478  public:
479  /*!
480  * \brief Constructor of the class.
481  * \param[in] nDim - Number of dimensions of the problem.
482  * \param[in] config - Definition of the particular problem.
483  */
484  CCompressibilityCorrection(unsigned short nDim, const CConfig* config)
485  : ParentClass(nDim, config) {}
486 
487  /*!
488  * \brief Residual for source term integration.
489  * \param[in] config - Definition of the particular problem.
490  * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
491  */
492  ResidualType ComputeResidual(const CConfig* config) override {
493  /*--- Residual from standard SA ---*/
494  ParentClass::ComputeResidual(config);
495 
496  /*--- Compressibility Correction term ---*/
497  const auto& pressure = V_i[idx.Pressure()];
498  const auto& density = V_i[idx.Density()];
499  const su2double sound_speed = sqrt(pressure * Gamma / density);
500  su2double aux_cc = 0;
501  for (unsigned short iDim = 0; iDim < nDim; ++iDim) {
502  for (unsigned short jDim = 0; jDim < nDim; ++jDim) {
503  aux_cc += pow(PrimVar_Grad_i[idx.Velocity() + iDim][jDim], 2);
504  }
505  }
506  const su2double d_CompCorrection = 2.0 * c5 * ScalarVar_i[0] / pow(sound_speed, 2) * aux_cc * Volume;
507  const su2double CompCorrection = 0.5 * ScalarVar_i[0] * d_CompCorrection;
508 
509  this->Residual -= CompCorrection;
510  this->Jacobian_i[0] -= d_CompCorrection;
511 
512  return ResidualType(&this->Residual, &this->Jacobian_i, nullptr);
513  }
514 };

549  if (use_ft2) {

588  const bool sustaining_terms = false;
589  const bool axisymmetric = false;

593  const su2double prod_lim_const;

602 
603  /*!
604  * \brief Get strain magnitude based on perturbed reynolds stress matrix.
605  * \param[in] turb_ke: turbulent kinetic energy of the node.
606  */
607  inline su2double PerturbedStrainMag(su2double turb_ke) const {
608  /*--- Compute norm of perturbed strain rate tensor. ---*/
609 
610  su2double perturbedStrainMag = 0;
611  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
612  for (unsigned short jDim = 0; jDim < nDim; jDim++) {
613  su2double StrainRate_ij = MeanPerturbedRSM[iDim][jDim] - TWO3 * turb_ke * delta[iDim][jDim];
614  StrainRate_ij = -StrainRate_ij * Density_i / (2 * Eddy_Viscosity_i);
615 
616  perturbedStrainMag += pow(StrainRate_ij, 2.0);
617  }
618  }
619  return sqrt(2.0 * perturbedStrainMag);
620  }
621 
622  /*!
623  * \brief Add contribution from convection and diffusion due to axisymmetric formulation to 2D residual
624  */
625  inline void ResidualAxisymmetricConvectionDiffusion(su2double alfa_blended, su2double zeta) {
626  if (Coord_i[1] < EPS) return;
627 
628  const su2double yinv = 1.0 / Coord_i[1];
629  const su2double rhov = Density_i * V_i[idx.Velocity() + 1];
630  const su2double& k = ScalarVar_i[0];
631  const su2double& w = ScalarVar_i[1];
632 
633  /*--- Compute blended constants ---*/
634  const su2double sigma_k_i = F1_i * sigma_k_1 + (1.0 - F1_i) * sigma_k_2;
635  const su2double sigma_w_i = F1_i * sigma_w_1 + (1.0 - F1_i) * sigma_w_2;
636 
637  /*--- Convection-Diffusion ---*/
638  const su2double cdk_axi = rhov * k - (Laminar_Viscosity_i + sigma_k_i * Eddy_Viscosity_i) * ScalarVar_Grad_i[0][1];
639  const su2double cdw_axi = rhov * w - (Laminar_Viscosity_i + sigma_w_i * Eddy_Viscosity_i) * ScalarVar_Grad_i[1][1];
640 
641  /*--- Add terms to the residuals ---*/
642 
643  Residual[0] -= yinv * Volume * cdk_axi;
644  Residual[1] -= yinv * Volume * cdw_axi;
645 
646  Jacobian_i[0][0] -= yinv * Volume * rhov;
647  Jacobian_i[0][1] -= 0.0;
648  Jacobian_i[1][0] -= 0.0;
649  Jacobian_i[1][1] -= yinv * Volume * rhov;
650 
651  }

710  /*!
711  * \brief Residual for source term integration.
712  * \param[in] config - Definition of the particular problem.
713  * \return A lightweight const-view (read-only) of the residual/flux and Jacobians.
714  */
715  ResidualType<> ComputeResidual(const CConfig* config) override {
716  AD::StartPreacc();
717  AD::SetPreaccIn(StrainMag_i);
718  AD::SetPreaccIn(ScalarVar_i, nVar);
719  AD::SetPreaccIn(ScalarVar_Grad_i, nVar, nDim);
720  AD::SetPreaccIn(Volume);
721  AD::SetPreaccIn(dist_i);
722  AD::SetPreaccIn(F1_i);
723  AD::SetPreaccIn(F2_i);
724  AD::SetPreaccIn(CDkw_i);
725  AD::SetPreaccIn(PrimVar_Grad_i, nDim + idx.Velocity(), nDim);
726  AD::SetPreaccIn(Vorticity_i, 3);
727  AD::SetPreaccIn(V_i[idx.Density()], V_i[idx.LaminarViscosity()], V_i[idx.EddyViscosity()]);
728  AD::SetPreaccIn(V_i[idx.Velocity() + 1]);
729 
730  Density_i = V_i[idx.Density()];
731  Laminar_Viscosity_i = V_i[idx.LaminarViscosity()];
732  Eddy_Viscosity_i = V_i[idx.EddyViscosity()];
733 
734  Residual[0] = 0.0;
735  Residual[1] = 0.0;
736  Jacobian_i[0][0] = 0.0;
737  Jacobian_i[0][1] = 0.0;
738  Jacobian_i[1][0] = 0.0;
739  Jacobian_i[1][1] = 0.0;
740 
741  /*--- Computation of blended constants for the source terms ---*/
742 
743  const su2double alfa_blended = F1_i * alfa_1 + (1.0 - F1_i) * alfa_2;
744  const su2double beta_blended = F1_i * beta_1 + (1.0 - F1_i) * beta_2;
745 
746  su2double eff_intermittency = 1.0;
747 
748  if (config->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
749  AD::SetPreaccIn(intermittency_eff_i);
750  eff_intermittency = intermittency_eff_i;
751  }
752 
753  if (dist_i > 1e-10) {
754 
755  su2double diverg = 0.0;
756  for (unsigned short iDim = 0; iDim < nDim; iDim++)
757  diverg += PrimVar_Grad_i[iDim + idx.Velocity()][iDim];
758  if (axisymmetric && Coord_i[1] > EPS) {
759  AD::SetPreaccIn(Coord_i[1]);
760  diverg += V_i[idx.Velocity() + 1] / Coord_i[1];
761  }
762 
763  /*--- If using UQ methodolgy, calculate production using perturbed Reynolds stress matrix ---*/
764 
765  const su2double VorticityMag = GeometryToolbox::Norm(3, Vorticity_i);
766  su2double P_Base = 0;
767 
768  /*--- Apply production term modifications ---*/
769  switch (sstParsedOptions.production) {
770  case SST_OPTIONS::UQ:
771  ComputePerturbedRSM(nDim, Eig_Val_Comp, uq_permute, uq_delta_b, uq_urlx, PrimVar_Grad_i + idx.Velocity(),
772  Density_i, Eddy_Viscosity_i, ScalarVar_i[0], MeanPerturbedRSM);
773  P_Base = PerturbedStrainMag(ScalarVar_i[0]);
774  break;
775 
776  case SST_OPTIONS::V:
777  P_Base = VorticityMag;
778  break;
779 
780  case SST_OPTIONS::KL:
781  P_Base = sqrt(StrainMag_i*VorticityMag);
782  break;
783 
784  default:
785  /*--- Base production term for SST-1994 and SST-2003 ---*/
786  P_Base = StrainMag_i;
787  break;
788  }
789 
790  /*--- Production limiter. ---*/
791  const su2double prod_limit = prod_lim_const * beta_star * Density_i * ScalarVar_i[1] * ScalarVar_i[0];
792 
793  su2double P = Eddy_Viscosity_i * pow(P_Base, 2);
794 
795  if (sstParsedOptions.version == SST_OPTIONS::V1994) {
796  /*--- INTRODUCE THE SST-V1994m BUG WHERE DIVERGENCE TERM WILL BE REMOVED ---*/
797  P -= 2.0 / 3.0 * Density_i * ScalarVar_i[0] * diverg;
798  }
799  su2double pk = max(0.0, min(P, prod_limit));
800 
801  const auto& eddy_visc_var = sstParsedOptions.version == SST_OPTIONS::V1994 ? VorticityMag : StrainMag_i;
802  const su2double zeta = max(ScalarVar_i[1], eddy_visc_var * F2_i / a1);
803 
804  /*--- Production limiter only for V2003, recompute for V1994. ---*/
805  su2double pw;
806  if (sstParsedOptions.version == SST_OPTIONS::V1994) {
807  /*--- INTRODUCE THE SST-V1994m BUG WHERE DIVERGENCE TERM WILL BE REMOVED ---*/
808  pw = alfa_blended * Density_i * max(pow(P_Base, 2) - 2.0 / 3.0 * zeta * diverg, 0.0);
809  } else {
810  pw = (alfa_blended * Density_i / Eddy_Viscosity_i) * pk;
811  }
812 
813  /*--- Sustaining terms, if desired. Note that if the production terms are
814  larger equal than the sustaining terms, the original formulation is
815  obtained again. This is in contrast to the version in literature
816  where the sustaining terms are simply added. This latter approach could
817  lead to problems for very big values of the free-stream turbulence
818  intensity. ---*/
819  if (sstParsedOptions.sust) {
820  const su2double sust_k = beta_star * Density_i * kAmb * omegaAmb;
821  const su2double sust_w = beta_blended * Density_i * omegaAmb * omegaAmb;
822  pk = max(pk, sust_k);
823  pw = max(pw, sust_w);
824  }
825 
826  /*--- Dissipation ---*/
827 
828  su2double dk = beta_star * Density_i * ScalarVar_i[1] * ScalarVar_i[0];
829  su2double dw = beta_blended * Density_i * ScalarVar_i[1] * ScalarVar_i[1];
830 
831  /*--- LM model coupling with production and dissipation term for k transport equation---*/
832  if (config->GetKind_Trans_Model() == TURB_TRANS_MODEL::LM) {
833  pk = pk * eff_intermittency;
834  dk = min(max(eff_intermittency, 0.1), 1.0) * dk;
835  }
836 
837  /*--- Add the production terms to the residuals. ---*/
838 
839  Residual[0] += pk * Volume;
840  Residual[1] += pw * Volume;
841 
842  /*--- Add the dissipation terms to the residuals.---*/
843 
844  Residual[0] -= dk * Volume;
845  Residual[1] -= dw * Volume;
846 
847  /*--- Cross diffusion ---*/
848 
849  Residual[1] += (1.0 - F1_i) * CDkw_i * Volume;
850 
851  /*--- Contribution due to 2D axisymmetric formulation ---*/
852 
853  if (axisymmetric) ResidualAxisymmetricConvectionDiffusion(alfa_blended, zeta);
854 
855  /*--- Implicit part ---*/
856 
857  Jacobian_i[0][0] = -beta_star * ScalarVar_i[1] * Volume;
858  Jacobian_i[0][1] = -beta_star * ScalarVar_i[0] * Volume;
859  Jacobian_i[1][0] = 0.0;
860  Jacobian_i[1][1] = -2.0 * beta_blended * ScalarVar_i[1] * Volume;
861  }
862 
863  AD::SetPreaccOut(Residual, nVar);
864  AD::EndPreacc();
865 
866  return ResidualType<>(Residual, Jacobian_i, nullptr);
867  }
