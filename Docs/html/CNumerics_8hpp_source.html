425  /*!
426  * \brief Compute the mean rate of strain matrix.
427  * \details The parameter primvargrad can be e.g. PrimVar_Grad_i or Mean_GradPrimVar.
428  * \param[in] nDim - 2 or 3
429  * \param[out] rateofstrain - Rate of strain matrix
430  * \param[in] velgrad - A velocity gradient matrix.
431  * \tparam Mat1 - any type that supports the [][] interface
432  * \tparam Mat2 - any type that supports the [][] interface
433  */
434  template<class Mat1, class Mat2>
435  FORCEINLINE static void ComputeMeanRateOfStrainMatrix(size_t nDim, Mat1& rateofstrain, const Mat2& velgrad){
436 
437  /* --- Calculate the rate of strain tensor, using mean velocity gradients --- */
438 
439  if (nDim == 3) {
440  rateofstrain[0][0] = velgrad[0][0];
441  rateofstrain[1][1] = velgrad[1][1];
442  rateofstrain[2][2] = velgrad[2][2];
443  rateofstrain[0][1] = 0.5 * (velgrad[0][1] + velgrad[1][0]);
444  rateofstrain[0][2] = 0.5 * (velgrad[0][2] + velgrad[2][0]);
445  rateofstrain[1][2] = 0.5 * (velgrad[1][2] + velgrad[2][1]);
446  rateofstrain[1][0] = rateofstrain[0][1];
447  rateofstrain[2][1] = rateofstrain[1][2];
448  rateofstrain[2][0] = rateofstrain[0][2];
449  }
450  else { // nDim==2
451  rateofstrain[0][0] = velgrad[0][0];
452  rateofstrain[1][1] = velgrad[1][1];
453  rateofstrain[2][2] = 0.0;
454  rateofstrain[0][1] = 0.5 * (velgrad[0][1] + velgrad[1][0]);
455  rateofstrain[0][2] = 0.0;
456  rateofstrain[1][2] = 0.0;
457  rateofstrain[1][0] = rateofstrain[0][1];
458  rateofstrain[2][1] = rateofstrain[1][2];
459  rateofstrain[2][0] = rateofstrain[0][2];
460  }
461  }

463  /*!
464  * \brief Compute the stress tensor from the velocity gradients.
465  * \details To obtain the Reynolds stress tensor +(u_i' u_j')~, divide the result
466  * of this function by (-rho). The argument density is only used if turb_ke is not 0.
467  * To select the velocity gradient components from a primitive variable gradient PrimVar_Grad_i,
468  * write PrimVar_Grad_i+1.
469  * If <code>nDim==2</code>, we use the same formula but only only access the entries [0][0]..[1][1] of
470  * stress and velgrad. If <code>reynolds3x3</code> is true, the other non-diagonal entries of stress
471  * set to zero, and <code>stress[2][2]</code> to some value.
472  * \param[in] nDim - Dimension of the flow problem, 2 or 3
473  * \param[out] stress - Stress tensor
474  * \param[in] velgrad - A velocity gradient matrix.
475  * \param[in] viscosity - Viscosity
476  * \param[in] density - Density
477  * \param[in] turb_ke - Turbulent kinetic energy, for the turbulent stress tensor
478  * \param[in] reynolds3x3 - If true, write to the third row and column of stress even if nDim==2.
479  * \tparam Mat1 - any type that supports the [][] interface
480  * \tparam Mat2 - any type that supports the [][] interface
481  */
482  template<class Mat1, class Mat2, class Scalar>
483  FORCEINLINE static void ComputeStressTensor(size_t nDim, Mat1& stress, const Mat2& velgrad,
484  Scalar viscosity, Scalar density=0.0,
485  Scalar turb_ke=0.0, bool reynolds3x3=false){
486  Scalar divVel = 0.0;
487  for (size_t iDim = 0; iDim < nDim; iDim++) {
488  divVel += velgrad[iDim][iDim];
489  }
490  Scalar pTerm = 2./3. * (divVel * viscosity + density * turb_ke);
491 
492  for (size_t iDim = 0; iDim < nDim; iDim++){
493  for (size_t jDim = 0; jDim < nDim; jDim++){
494  stress[iDim][jDim] = viscosity * (velgrad[iDim][jDim]+velgrad[jDim][iDim]);
495  }
496  stress[iDim][iDim] -= pTerm;
497  }

499  if(reynolds3x3 && nDim==2) { // fill the third row and column of Reynolds stress matrix
500  stress[0][2] = stress[1][2] = stress[2][0] = stress[2][1] = 0.0;
501  stress[2][2] = -pTerm;
502  }

505  /*!
506  * \brief Add a correction using a Quadratic Constitutive Relation to the stress tensor.
507  *
508  * See: Spalart, P. R., "Strategies for Turbulence Modelling and Simulation",
509  * International Journal of Heat and Fluid Flow, Vol. 21, 2000, pp. 252-263
510  *
511  * \param[in] nDim: 2D or 3D.
512  * \param[in] gradvel: Velocity gradients.
513  * \param[in,out] tau: Shear stress tensor.
514  */
515  template <class Mat1, class Mat2>
516  FORCEINLINE static void AddQCR(size_t nDim, const Mat1& gradvel, Mat2& tau) {
517  using Scalar = typename std::decay<decltype(gradvel[0][0])>::type;
518 
519  const Scalar c_cr1 = 0.3;
520 
521  /*--- Denominator Antisymmetric normalized rotation tensor ---*/
522 
523  Scalar factor = 0.0;
524  for (size_t iDim = 0; iDim < nDim; iDim++)
525  for (size_t jDim = 0; jDim < nDim; jDim++)
526  factor += gradvel[iDim][jDim] * gradvel[iDim][jDim];
527  factor = 1.0 / sqrt(max(factor,1E-10));
528 
529  /*--- Adding the QCR contribution ---*/
530 
531  Scalar tauQCR[MAXNDIM][MAXNDIM] = {{0.0}};
532 
533  for (size_t iDim = 0; iDim < nDim; iDim++){
534  for (size_t jDim = 0; jDim < nDim; jDim++){
535  for (size_t kDim = 0; kDim < nDim; kDim++){
536  Scalar O_ik = (gradvel[iDim][kDim] - gradvel[kDim][iDim]) * factor;
537  Scalar O_jk = (gradvel[jDim][kDim] - gradvel[kDim][jDim]) * factor;
538  tauQCR[iDim][jDim] += O_ik * tau[jDim][kDim] + O_jk * tau[iDim][kDim];
539  }
540  }
541  }
542 
543  for (size_t iDim = 0; iDim < nDim; iDim++)
544  for (size_t jDim = 0; jDim < nDim; jDim++)
545  tau[iDim][jDim] -= c_cr1 * tauQCR[iDim][jDim];
546  }

548  /*!
549  * \brief Perturb the Reynolds stress tensor based on parameters.
550  * \param[in] nDim - Dimension of the flow problem, 2 or 3.
551  * \param[in] uq_eigval_comp - Component 1C 2C 3C.
552  * \param[in] uq_permute - Whether to swap order of eigen vectors.
553  * \param[in] uq_delta_b - Delta_b parameter.
554  * \param[in] uq_urlx - Relaxation factor.
555  * \param[in] velgrad - A velocity gradient matrix.
556  * \param[in] density - Density.
557  * \param[in] viscosity - Eddy viscosity.
558  * \param[in] turb_ke: Turbulent kinetic energy.
559  * \param[out] MeanPerturbedRSM - Perturbed stress tensor.
560  */
561  template<class Mat1, class Mat2, class Scalar>
562  NEVERINLINE static void ComputePerturbedRSM(size_t nDim, size_t uq_eigval_comp, bool uq_permute, su2double uq_delta_b,
563  su2double uq_urlx, const Mat1& velgrad, Scalar density,
564  Scalar viscosity, Scalar turb_ke, Mat2& MeanPerturbedRSM) {
565  Scalar MeanReynoldsStress[3][3];
566  ComputeStressTensor(nDim, MeanReynoldsStress, velgrad, viscosity, density, turb_ke, true);
567  for (size_t iDim = 0; iDim < 3; iDim++)
568  for (size_t jDim = 0; jDim < 3; jDim++)
569  MeanReynoldsStress[iDim][jDim] /= -density;
570 
571  /* --- Calculate anisotropic part of Reynolds Stress tensor --- */
572 
573  Scalar A_ij[3][3];
574  for (size_t iDim = 0; iDim < 3; iDim++) {
575  for (size_t jDim = 0; jDim < 3; jDim++) {
576  A_ij[iDim][jDim] = .5 * MeanReynoldsStress[iDim][jDim] / turb_ke;
577  }
578  A_ij[iDim][iDim] -= 1.0/3.0;
579  }
580 
581  /* --- Get ordered eigenvectors and eigenvalues of A_ij --- */
582 
583  Scalar work[3], Eig_Vec[3][3], Eig_Val[3];
584  CBlasStructure::EigenDecomposition(A_ij, Eig_Vec, Eig_Val, 3, work);
585 
586  /* compute convex combination coefficients */
587  Scalar c1c = Eig_Val[2] - Eig_Val[1];
588  Scalar c2c = 2.0 * (Eig_Val[1] - Eig_Val[0]);
589  Scalar c3c = 3.0 * Eig_Val[0] + 1.0;
590 
591  /* define barycentric traingle corner points */
592  Scalar Corners[3][2];
593  Corners[0][0] = 1.0;
594  Corners[0][1] = 0.0;
595  Corners[1][0] = 0.0;
596  Corners[1][1] = 0.0;
597  Corners[2][0] = 0.5;
598  Corners[2][1] = 0.866025;
599 
600  /* define barycentric coordinates */
601  Scalar Barycentric_Coord[2];
602  Barycentric_Coord[0] = Corners[0][0] * c1c + Corners[1][0] * c2c + Corners[2][0] * c3c;
603  Barycentric_Coord[1] = Corners[0][1] * c1c + Corners[1][1] * c2c + Corners[2][1] * c3c;
604 
605  /* component 1C, 2C, 3C, converted to index of the "corners" */
606  Scalar New_Coord[2];
607  New_Coord[0] = Corners[uq_eigval_comp-1][0];
608  New_Coord[1] = Corners[uq_eigval_comp-1][1];
609 
610  /* calculate perturbed barycentric coordinates */
611  Barycentric_Coord[0] = Barycentric_Coord[0] + (uq_delta_b) * (New_Coord[0] - Barycentric_Coord[0]);
612  Barycentric_Coord[1] = Barycentric_Coord[1] + (uq_delta_b) * (New_Coord[1] - Barycentric_Coord[1]);
613 
614  /* rebuild c1c,c2c,c3c based on perturbed barycentric coordinates */
615  c3c = Barycentric_Coord[1] / Corners[2][1];
616  c1c = Barycentric_Coord[0] - Corners[2][0] * c3c;
617  c2c = 1 - c1c - c3c;
618 
619  /* build new anisotropy eigenvalues */
620  Eig_Val[0] = (c3c - 1) / 3.0;
621  Eig_Val[1] = 0.5 *c2c + Eig_Val[0];
622  Eig_Val[2] = c1c + Eig_Val[1];
623 
624  /* permute eigenvectors if required */
625  if (uq_permute) {
626  for (size_t jDim = 0; jDim < 3; jDim++)
627  swap(Eig_Vec[0][jDim], Eig_Vec[2][jDim]);
628  }
629 
630  CBlasStructure::EigenRecomposition(A_ij, Eig_Vec, Eig_Val, 3);
631 
632  /* compute perturbed Reynolds stress matrix; using under-relaxation factor (uq_urlx)*/
633  for (size_t iDim = 0; iDim < 3; iDim++) {
634  for (size_t jDim = 0; jDim < 3; jDim++) {
635  auto delta_ij = (jDim==iDim)? 1.0 : 0.0;
636  MeanPerturbedRSM[iDim][jDim] = 2.0 * turb_ke * (A_ij[iDim][jDim] + 1.0/3.0 * delta_ij);
637  MeanPerturbedRSM[iDim][jDim] = MeanReynoldsStress[iDim][jDim] +
638  uq_urlx*(MeanPerturbedRSM[iDim][jDim] - MeanReynoldsStress[iDim][jDim]);
639  }
640  }
641  }

643  /*!
644  * \brief Project average gradient onto normal (with or w/o correction) for viscous fluxes of scalar quantities.
645  * \param[in] nDim - Dimension of the flow problem, 2 or 3.
646  * \param[in] nVar - Number of variables.
647  * \param[in] normal - Area vector.
648  * \param[in] coord_i - Coordinate of point i.
649  * \param[in] coord_j - Coordinate of point j.
650  * \param[in] grad_i - Gradient at point i.
651  * \param[in] grad_j - Gradient at point j.
652  * \param[in] correct - Correct
653  * \param[in] var_i - Variable at point i.
654  * \param[in] var_j - Variable at point j.
655  * \param[out] projNormal - Average gradient projected on normal.
656  * \param[out] projCorrected - Corrected gradient projected on normal.
657  * \return (Edge_Vector DOT normal) / |Edge_Vector|^2.
658  */
659  template<class Vec1, class Vec2, class Mat>
660  FORCEINLINE static su2double ComputeProjectedGradient(int nDim, int nVar, const Vec1& normal,
661  const Vec1& coord_i, const Vec1& coord_j,
662  const Mat& grad_i, const Mat& grad_j,
663  bool correct,
664  const Vec2& var_i, const Vec2& var_j,
665  su2double* projNormal,
666  su2double* projCorrected) {
667  assert(nDim == 2 || nDim == 3);
668  nDim = (nDim > 2)? 3 : 2;
669  su2double edgeVec[MAXNDIM], dist_ij_2 = 0.0, proj_vector_ij = 0.0;
670 
671  for (int iDim = 0; iDim < nDim; iDim++) {
672  edgeVec[iDim] = coord_j[iDim] - coord_i[iDim];
673  dist_ij_2 += pow(edgeVec[iDim], 2);
674  proj_vector_ij += edgeVec[iDim] * normal[iDim];
675  }
676  proj_vector_ij /= max(dist_ij_2,EPS);
677 
678  /*--- Mean gradient approximation. ---*/
679  for (int iVar = 0; iVar < nVar; iVar++) {
680  projNormal[iVar] = 0.0;
681  su2double edgeProj = 0.0;
682 
683  for (int iDim = 0; iDim < nDim; iDim++) {
684  su2double meanGrad = 0.5 * (grad_i[iVar][iDim] + grad_j[iVar][iDim]);
685  projNormal[iVar] += meanGrad * normal[iDim];
686  if (correct) edgeProj += meanGrad * edgeVec[iDim];
687  }
688 
689  projCorrected[iVar] = projNormal[iVar];
690  if (correct) projCorrected[iVar] -= (edgeProj - (var_j[iVar]-var_i[iVar])) * proj_vector_ij;
691  }
692 
693  return proj_vector_ij;
694  }
