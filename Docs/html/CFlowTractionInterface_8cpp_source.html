42 void CFlowTractionInterface::Preprocess(const CConfig *flow_config) {
43 
44  /*--- Compute the constant factor to dimensionalize pressure and shear stress. ---*/
45  const su2double *Velocity_ND, *Velocity_Real;
46  su2double Density_ND, Density_Real, Velocity2_Real, Velocity2_ND;
47 
48  Velocity_Real = flow_config->GetVelocity_FreeStream();
49  Density_Real = flow_config->GetDensity_FreeStream();
50 
51  Velocity_ND = flow_config->GetVelocity_FreeStreamND();
52  Density_ND = flow_config->GetDensity_FreeStreamND();
53 
54  Velocity2_Real = 0.0;
55  Velocity2_ND = 0.0;
56  for (unsigned short iVar = 0; iVar < nVar; iVar++) {
57  Velocity2_Real += Velocity_Real[iVar]*Velocity_Real[iVar];
58  Velocity2_ND += Velocity_ND[iVar]*Velocity_ND[iVar];
59  }
60 
61  Physical_Constants[0] = Density_Real * Velocity2_Real / ( Density_ND * Velocity2_ND );
62 }

64 void CFlowTractionInterface::ComputeVertexAreas(const CConfig *config,
65  CGeometry *geometry,
66  CSolver *solution) {
67  constexpr size_t MAXNDIM = 3;
68  const auto nodes = solution->GetNodes();
69  const auto nDim = nVar;
70 
71  /*--- Compute current area associated with each vertex. ---*/
72 
73  vertexArea.clear();
74  unordered_set<short> processedMarkers({-1});
75 
76  for (auto iMarkerInt = 0; iMarkerInt < config->GetMarker_n_ZoneInterface()/2; ++iMarkerInt) {
77  /*--- Find the marker index associated with the pair. ---*/
78  const auto iMarker = config->FindInterfaceMarker(iMarkerInt);
79  /*--- The current mpi rank may not have this marker, or it may have been processed already. ---*/
80  if (processedMarkers.count(iMarker) > 0) continue;
81  processedMarkers.insert(iMarker);
82 
83  for (auto iElem = 0u; iElem < geometry->GetnElem_Bound(iMarker); ++iElem) {
84  /*--- Define the boundary element. ---*/
85  unsigned long nodeList[N_POINTS_MAXIMUM] = {0};
86  su2double coords[N_POINTS_MAXIMUM][MAXNDIM] = {{0.0}};
87  bool quad = geometry->bound[iMarker][iElem]->GetVTK_Type() == QUADRILATERAL;
88  auto nNode = quad? 4u : nDim;
89 
90  for (auto iNode = 0u; iNode < nNode; ++iNode) {
91  nodeList[iNode] = geometry->bound[iMarker][iElem]->GetNode(iNode);
92  for (auto iDim = 0u; iDim < nDim; ++iDim)
93  coords[iNode][iDim] = geometry->nodes->GetCoord(nodeList[iNode], iDim)+
94  nodes->GetSolution(nodeList[iNode],iDim);
95  }
96 
97  /*--- Compute the area contribution to each node. ---*/
98  su2double normal[MAXNDIM] = {0.0};
99 
100  switch (nNode) {
101  case 2: GeometryToolbox::LineNormal(coords, normal); break;
102  case 3: GeometryToolbox::TriangleNormal(coords, normal); break;
103  case 4: GeometryToolbox::QuadrilateralNormal(coords, normal); break;
104  }
105 
106  su2double area = GeometryToolbox::Norm(MAXNDIM, normal) / nNode;
107 
108  /*--- Update area of nodes. ---*/
109  for (auto iNode = 0u; iNode < nNode; ++iNode) {
110  auto iPoint = nodeList[iNode];
111  if (vertexArea.count(iPoint) == 0)
112  vertexArea[iPoint] = area;
113  else
114  vertexArea[iPoint] += area;
115  }
116  }
117  }
118 }

120 void CFlowTractionInterface::GetPhysical_Constants(CSolver *flow_solution, CSolver *struct_solution,
121  CGeometry *flow_geometry, CGeometry *struct_geometry,
122  const CConfig *flow_config, const CConfig *struct_config) {
123 
124  /*--- We have to clear the traction before applying it, because we are "adding" to node and not "setting" ---*/
125 
126  struct_solution->GetNodes()->Clear_FlowTraction();
127 
128  Preprocess(flow_config);
129 
130  if (!conservative) ComputeVertexAreas(struct_config, struct_geometry, struct_solution);
131 
132  /*--- Apply a ramp to the transfer of the fluid loads ---*/
133 
134  su2double ModAmpl = 0.0;
135  su2double CurrentTime = struct_config->GetCurrent_DynTime();
136 
137  bool Ramp_Load = struct_config->GetRamp_Load();
138  su2double Ramp_Time = struct_config->GetRamp_Time();
139 
140  ModAmpl = struct_solution->Compute_LoadCoefficient(CurrentTime, Ramp_Time, struct_config);
141 
142  Physical_Constants[1] = ModAmpl;
143 
144  /*--- For static FSI, we cannot apply the ramp like this ---*/
145  if ((!flow_config->GetTime_Domain())){
146  Physical_Constants[1] = 1.0;
147  if (Ramp_Load){
148  CurrentTime = static_cast<su2double>(struct_config->GetOuterIter());
149  Ramp_Time = static_cast<su2double>(struct_config->GetnIterFSI_Ramp() - 1);
150 
151  ModAmpl = struct_solution->Compute_LoadCoefficient(CurrentTime, Ramp_Time, struct_config);
152  Physical_Constants[1] = ModAmpl;
153  }
154  }
155 
156  /*--- Store the force coefficient ---*/
157  struct_solution->SetForceCoeff(Physical_Constants[1]);
158 
159 }

161 void CFlowTractionInterface::GetDonor_Variable(CSolver *flow_solution, CGeometry *flow_geometry,
162  const CConfig *flow_config, unsigned long Marker_Flow,
163  unsigned long Vertex_Flow, unsigned long Point_Struct) {
164 
165  const auto Point_Flow = flow_geometry->vertex[Marker_Flow][Vertex_Flow]->GetNode();
166 
167  // Get the normal at the vertex: this normal goes inside the fluid domain.
168  const su2double* Normal_Flow = flow_geometry->vertex[Marker_Flow][Vertex_Flow]->GetNormal();
169 
170  // If we do not want integrated tractions, i.e. forces, we will need to divide by area.
171  su2double oneOnArea = 1.0;
172  if (!conservative) oneOnArea = 1.0 / GeometryToolbox::Norm(nVar, Normal_Flow);
173 
174  // Retrieve the values of pressure
175 
176  CVariable* flow_nodes = flow_solution->GetNodes();
177  su2double Pinf = flow_solution->GetPressure_Inf();
178  su2double Pn = flow_nodes->GetPressure(Point_Flow);
179 
180  // Calculate tn in the fluid nodes for the inviscid term --> Units of force (non-dimensional).
181 
182  for (auto iVar = 0u; iVar < nVar; iVar++)
183  Donor_Variable[iVar] = -(Pn-Pinf)*Normal_Flow[iVar];
184 
185  // Calculate tn in the fluid nodes for the viscous term
186 
187  if (flow_config->GetViscous()) {
188 
189  su2double Viscosity = flow_nodes->GetLaminarViscosity(Point_Flow);
190 
191  su2double tau[3][3];
192  CNumerics::ComputeStressTensor(nVar, tau, flow_nodes->GetVelocityGradient(Point_Flow), Viscosity);
193  for (auto iVar = 0u; iVar < nVar; iVar++) {
194  for (auto jVar = 0u; jVar < nVar; jVar++) {
195  // Viscous component in the tn vector --> Units of force (non-dimensional).
196  Donor_Variable[iVar] += tau[iVar][jVar] * Normal_Flow[jVar];
197  }
198  }
199 
200  }
201 
202  // Redimensionalize and take into account ramp transfer of the loads
203  for (auto iVar = 0u; iVar < nVar; iVar++) {
204  Donor_Variable[iVar] *= Physical_Constants[0] * Physical_Constants[1] * oneOnArea;
205  }
206 
207 }

209 void CFlowTractionInterface::SetTarget_Variable(CSolver *fea_solution, CGeometry *fea_geometry,
210  const CConfig *fea_config, unsigned long Marker_Struct,
211  unsigned long Vertex_Struct, unsigned long Point_Struct) {
212 
213  /*--- Add the Flow traction, if consistent interpolation is in use the traction needs to be integrated. ---*/
214  if (!conservative) {
215  assert(vertexArea.count(Point_Struct));
216  const auto area = vertexArea.at(Point_Struct);
217  for (auto iVar = 0u; iVar < nVar; ++iVar) Target_Variable[iVar] *= area;
218  }
219 
220  fea_solution->GetNodes()->Add_FlowTraction(Point_Struct, Target_Variable);
221 }
