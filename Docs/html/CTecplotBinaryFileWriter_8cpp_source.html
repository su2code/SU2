174  dataSorter->GetNodeEnd(rank) < (unsigned long)dataSorter->GetElem_Connectivity(TETRAHEDRON, 0, i))
175  halo_nodes.insert(dataSorter->GetElem_Connectivity(TETRAHEDRON, 0, i));

178  if ((unsigned long)dataSorter->GetElem_Connectivity(HEXAHEDRON, 0, i) <= dataSorter->GetNodeBegin(rank) ||
179  dataSorter->GetNodeEnd(rank) < (unsigned long)dataSorter->GetElem_Connectivity(HEXAHEDRON, 0, i))
180  halo_nodes.insert(dataSorter->GetElem_Connectivity(HEXAHEDRON, 0, i));
181 
182  for (unsigned long i = 0; i < nParallel_Pris * N_POINTS_PRISM; ++i)
183  if ((unsigned long)dataSorter->GetElem_Connectivity(PRISM, 0, i) <= dataSorter->GetNodeBegin(rank) ||
184  dataSorter->GetNodeEnd(rank) < (unsigned long)dataSorter->GetElem_Connectivity(PRISM, 0, i))
185  halo_nodes.insert(dataSorter->GetElem_Connectivity(PRISM, 0, i));
186 
187  for (unsigned long i = 0; i < nParallel_Pyra * N_POINTS_PYRAMID; ++i)
188  if ((unsigned long)dataSorter->GetElem_Connectivity(PYRAMID, 0, i) <= dataSorter->GetNodeBegin(rank) ||
189  dataSorter->GetNodeEnd(rank) < (unsigned long)dataSorter->GetElem_Connectivity(PYRAMID, 0, i))
190  halo_nodes.insert(dataSorter->GetElem_Connectivity(PYRAMID, 0, i));

192  /* Sorted list of halo nodes for this MPI rank. */
193  sorted_halo_nodes.assign(halo_nodes.begin(), halo_nodes.end());
194 
195  /* Have to include all nodes our cells refer to or TecIO will barf, so add the halo node count to the number of local nodes. */
196  int64_t partition_num_nodes = dataSorter->GetNodeEnd(rank) - dataSorter->GetNodeBegin(rank) + static_cast<int64_t>(halo_nodes.size());
197  int64_t partition_num_cells = nParallel_Tetr + nParallel_Hexa + nParallel_Pris + nParallel_Pyra;
198 
199  /*--- We effectively tack the halo nodes onto the end of the node list for this partition.
200  TecIO will later replace them with references to nodes in neighboring partitions. */
201  size_t num_halo_nodes = sorted_halo_nodes.size();
202  vector<int64_t> halo_node_local_numbers(max((size_t)1, num_halo_nodes)); /* Min size 1 to avoid crashes when we access these vectors below. */
203  vector<int32_t> neighbor_partitions(max((size_t)1, num_halo_nodes));
204  vector<int64_t> neighbor_nodes(max((size_t)1, num_halo_nodes));
205  for(int64_t i = 0; i < static_cast<int64_t>(num_halo_nodes); ++i) {
206  halo_node_local_numbers[i] = dataSorter->GetNodeEnd(rank) - dataSorter->GetNodeBegin(rank) + i + 1;
207  int owning_rank;
208  unsigned long node_number;
209  node_partitioner.GetOwningRankAndNodeNumber(sorted_halo_nodes[i], owning_rank, node_number);
210  neighbor_partitions[i] = owning_rank + 1; /* Partition numbers are 1-based. */
211  neighbor_nodes[i] = static_cast<int64_t>(node_number);
212  }
213  err = tecFEPartitionCreate64(file_handle, zone, rank + 1, partition_num_nodes, partition_num_cells,
214  static_cast<int64_t>(num_halo_nodes), &halo_node_local_numbers[0], &neighbor_partitions[0], &neighbor_nodes[0], 0, NULL);
215  if (err) cout << rank << ": Error creating Tecplot zone partition." << endl;
216 
217  /* Gather halo node data. First, tell each rank how many nodes' worth of data we need from them. */
218  for (size_t i = 0; i < num_halo_nodes; ++i)
219  ++num_nodes_to_receive[neighbor_partitions[i] - 1];
220  vector<int> num_nodes_to_send(size);
221  SU2_MPI::Alltoall(&num_nodes_to_receive[0], 1, MPI_INT, &num_nodes_to_send[0], 1, MPI_INT, SU2_MPI::GetComm());
222 
223  /* Now send the global node numbers whose data we need,
224  and receive the same from all other ranks.
225  Each rank has globally consecutive node numbers,
226  so we can just parcel out sorted_halo_nodes for send. */
227  vector<int> nodes_to_send_displacements(size);
228  vector<int> nodes_to_receive_displacements(size);
229  nodes_to_send_displacements[0] = 0;
230  nodes_to_receive_displacements[0] = 0;
231  for(int iRank = 1; iRank < size; ++iRank) {
232  nodes_to_send_displacements[iRank] = nodes_to_send_displacements[iRank - 1] + num_nodes_to_send[iRank - 1];
233  nodes_to_receive_displacements[iRank] = nodes_to_receive_displacements[iRank - 1] + num_nodes_to_receive[iRank - 1];
234  }
235  int total_num_nodes_to_send = nodes_to_send_displacements[size - 1] + num_nodes_to_send[size - 1];
236  vector<unsigned long> nodes_to_send(max(1, total_num_nodes_to_send));
237 
238  /* The terminology gets a bit confusing here. We're sending the node numbers
239  (sorted_halo_nodes) whose data we need to receive, and receiving
240  lists of nodes whose data we need to send. */
241  if (sorted_halo_nodes.empty()) sorted_halo_nodes.resize(1); /* Avoid crash. */
242  SU2_MPI::Alltoallv(&sorted_halo_nodes[0], &num_nodes_to_receive[0], &nodes_to_receive_displacements[0], MPI_UNSIGNED_LONG,
243  &nodes_to_send[0], &num_nodes_to_send[0], &nodes_to_send_displacements[0], MPI_UNSIGNED_LONG,
244  SU2_MPI::GetComm());
245 
246  /* Now actually send and receive the data */
247  vector<passivedouble> data_to_send(max(1, total_num_nodes_to_send * (int)fieldNames.size()));
248  halo_var_data.resize(max((size_t)1, fieldNames.size() * num_halo_nodes));
249  vector<int> num_values_to_send(size);
250  vector<int> values_to_send_displacements(size);
251  vector<int> num_values_to_receive(size);
252  size_t index = 0;
253  for(int iRank = 0; iRank < size; ++iRank) {
254  /* We send and receive GlobalField_Counter values per node. */
255  num_values_to_send[iRank] = num_nodes_to_send[iRank] * fieldNames.size();
256  values_to_send_displacements[iRank] = nodes_to_send_displacements[iRank] * fieldNames.size();
257  num_values_to_receive[iRank] = num_nodes_to_receive[iRank] * fieldNames.size();
258  values_to_receive_displacements[iRank] = nodes_to_receive_displacements[iRank] * fieldNames.size();
259  for(iVar = 0; iVar < fieldNames.size(); ++iVar)
260  for(int iNode = 0; iNode < num_nodes_to_send[iRank]; ++iNode) {
261  unsigned long node_offset = nodes_to_send[nodes_to_send_displacements[iRank] + iNode] - dataSorter->GetNodeBegin(rank) - 1;
262  data_to_send[index++] =dataSorter->GetData(iVar,node_offset);
263  }
264  }
265  CBaseMPIWrapper::Alltoallv(&data_to_send[0], &num_values_to_send[0], &values_to_send_displacements[0], MPI_DOUBLE,
266  &halo_var_data[0], &num_values_to_receive[0], &values_to_receive_displacements[0], MPI_DOUBLE,
267  SU2_MPI::GetComm());
268  }
269  else {
270  /* Zone will be gathered to and output by MASTER_NODE */
271  int32_t partition_owner = MASTER_NODE;
272  err = tecZoneMapPartitionsToMPIRanks(file_handle, zone, 1, &partition_owner);
273  }

369  /**
370  * Each rank writes node numbers relative to the partition it is outputting (starting with node number 1).
371  * Ghost (halo) nodes identified above are numbered sequentially just beyond the end of the actual, local nodes.
372  * Note that beg_node and end_node refer to zero-based node numbering, but Conn_* contain one-based node numbers.
373  */
374 #define MAKE_LOCAL(n) dataSorter->GetNodeBegin(rank) < (unsigned long)n && (unsigned long)n <= dataSorter->GetNodeEnd(rank) \
375  ? (int64_t)((unsigned long)n - dataSorter->GetNodeBegin(rank)) \
376  : GetHaloNodeNumber(n, dataSorter->GetNodeEnd(rank) - dataSorter->GetNodeBegin(rank), sorted_halo_nodes)

378  for (iElem = 0; err == 0 && iElem < nParallel_Tetr; iElem++) {
379  nodes[0] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(TETRAHEDRON, iElem, 0));
380  nodes[1] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(TETRAHEDRON, iElem, 1));
381  nodes[2] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(TETRAHEDRON, iElem, 2));
382  nodes[3] = nodes[2];
383  nodes[4] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(TETRAHEDRON, iElem, 3));
384  nodes[5] = nodes[4];
385  nodes[6] = nodes[4];
386  nodes[7] = nodes[4];

388  if (err) cout << rank << ": Error outputting Tecplot node values." << endl;
389  }
390 
391  for (iElem = 0; err == 0 && iElem < nParallel_Hexa; iElem++) {
392  nodes[0] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(HEXAHEDRON, iElem, 0));
393  nodes[1] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(HEXAHEDRON, iElem, 1));
394  nodes[2] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(HEXAHEDRON, iElem, 2));
395  nodes[3] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(HEXAHEDRON, iElem, 3));
396  nodes[4] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(HEXAHEDRON, iElem, 4));
397  nodes[5] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(HEXAHEDRON, iElem, 5));
398  nodes[6] = MAKE_LOCAL(dataSorter->GetElem_Connectivity(HEXAHEDRON, iElem, 6));
