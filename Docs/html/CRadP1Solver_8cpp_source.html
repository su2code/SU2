36 CRadP1Solver::CRadP1Solver(CGeometry* geometry, CConfig *config) : CRadSolver(geometry, config) {
37 
38  unsigned short iVar;
39  unsigned short direct_diff = config->GetDirectDiff();
40  bool multizone = config->GetMultizone_Problem();
41 
42  nDim = geometry->GetnDim();
43  nPoint = geometry->GetnPoint();
44  nPointDomain = geometry->GetnPointDomain();
45  nVar = 1;
46 
47  /*--- Initialize nVarGrad for deallocation ---*/
48 
49  nVarGrad = nVar;
50 
51  Residual = new su2double[nVar];
52  Solution = new su2double[nVar];
53 
54  Res_Visc = new su2double[nVar];
55 
56  /*--- Define some structures for locating max residuals ---*/
57 
58  Residual_RMS.resize(nVar,0.0);
59  Residual_Max.resize(nVar,0.0);
60  Point_Max.resize(nVar,0);
61  Point_Max_Coord.resize(nVar,nDim) = su2double(0.0);
62 
63  /*--- Jacobians and vector structures for implicit computations ---*/
64 
65  if (config->GetKind_TimeIntScheme_Radiation() == EULER_IMPLICIT) {
66 
67  Jacobian_i = new su2double* [nVar];
68  Jacobian_j = new su2double* [nVar];
69  for (iVar = 0; iVar < nVar; iVar++) {
70  Jacobian_i[iVar] = new su2double [nVar];
71  Jacobian_j[iVar] = new su2double [nVar];
72  }
73 
74  if (rank == MASTER_NODE) cout << "Initialize Jacobian structure (P1 radiation equation)." << endl;
75  Jacobian.Initialize(nPoint, nPointDomain, nVar, nVar, true, geometry, config);
76 
77  }
78 
79  /*--- Solution and residual vectors ---*/
80 
81  LinSysSol.Initialize(nPoint, nPointDomain, nVar, 0.0);
82  LinSysRes.Initialize(nPoint, nPointDomain, nVar, 0.0);
83 
84  /*--- Read farfield conditions from config ---*/
85  Temperature_Inf = config->GetTemperature_FreeStreamND();
86 
87  /*--- Initialize the secondary values for direct derivative approxiations ---*/
88 
89  switch(direct_diff){
90  case NO_DERIVATIVE: case D_DENSITY:
91  case D_PRESSURE: case D_VISCOSITY:
92  case D_MACH: case D_AOA:
93  case D_SIDESLIP: case D_REYNOLDS:
94  case D_TURB2LAM: case D_DESIGN:
95  /*--- Not necessary here ---*/
96  break;
97  case D_TEMPERATURE:
98  SU2_TYPE::SetDerivative(Temperature_Inf, 1.0);
99  break;
100  default:
101  break;
102  }
103 
104  SetTemperature_Inf(Temperature_Inf);
105 
106  /*--- Initialize the BGS residuals. ---*/
107  if (multizone){
108  Residual_BGS.resize(nVar,1.0);
109  Residual_Max_BGS.resize(nVar,1.0);
110  Point_Max_BGS.resize(nVar,0);
111  Point_Max_Coord_BGS.resize(nVar,nDim) = su2double(0.0);
112  }
113 
114  /*--- Always instantiate and initialize the variable to a zero value. ---*/
115  su2double init_val;
116  switch(config->GetKind_P1_Init()){
117  case P1_INIT::ZERO: init_val = 0.0; break;
118  case P1_INIT::TEMPERATURE: init_val = 4.0*STEFAN_BOLTZMANN*pow(config->GetInc_Temperature_Init(),4.0); break;
119  default: init_val = 0.0; break;
120  }
121 
122  nodes = new CRadP1Variable(init_val, nPoint, nDim, nVar, config);
123  SetBaseClassPointerToNodes();
124 
125  /*--- Initialize the structure for volumetric heat source ---*/
126  if(config->GetHeatSource()){
127  SetVolumetricHeatSource(geometry, config);
128  }
129 
130 }

161 void CRadP1Solver::Postprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh) {
162 
163  unsigned long iPoint;
164  su2double Energy, Temperature;
165  su2double SourceTerm, SourceTerm_Derivative;
166 
167  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
168 
169  /*--- Retrieve the radiative energy ---*/
170  Energy = nodes->GetSolution(iPoint, 0);
171 
172  /*--- Retrieve temperature from the flow solver ---*/
173  Temperature = solver_container[FLOW_SOL]->GetNodes()->GetTemperature(iPoint);
174 
175  /*--- Compute the divergence of the radiative flux ---*/
176  SourceTerm = Absorption_Coeff*(Energy - 4.0*STEFAN_BOLTZMANN*pow(Temperature,4.0));
177 
178  /*--- Compute the derivative of the source term with respect to the temperature ---*/
179  SourceTerm_Derivative = - 16.0*Absorption_Coeff*STEFAN_BOLTZMANN*pow(Temperature,3.0);
180 
181  /*--- Store the source term and its derivative ---*/
182  nodes->SetRadiative_SourceTerm(iPoint, 0, SourceTerm);
183  nodes->SetRadiative_SourceTerm(iPoint, 1, SourceTerm_Derivative);
184 
185  }

189 void CRadP1Solver::Viscous_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
190  CConfig *config, unsigned short iMesh, unsigned short iRKStep) {
191 
192  CNumerics* numerics = numerics_container[VISC_TERM];
193 
194  unsigned long iEdge, iPoint, jPoint;
195 
196  for (iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
197 
198  /*--- Points in edge ---*/
199 
200  iPoint = geometry->edges->GetNode(iEdge,0);
201  jPoint = geometry->edges->GetNode(iEdge,1);
202 
203  /*--- Points coordinates, and normal vector ---*/
204 
205  numerics->SetCoord(geometry->nodes->GetCoord(iPoint),
206  geometry->nodes->GetCoord(jPoint));
207  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
208 
209  /*--- Radiation variables w/o reconstruction, and its gradients ---*/
210 
211  numerics->SetRadVar(nodes->GetSolution(iPoint), nodes->GetSolution(jPoint));
212  numerics->SetRadVarGradient(nodes->GetGradient(iPoint), nodes->GetGradient(jPoint));
213 
214  /*--- Compute residual, and Jacobians ---*/
215 
216  numerics->ComputeResidual(Residual, Jacobian_i, Jacobian_j, config);
217 
218  /*--- Add and subtract residual, and update Jacobian ---*/
219 
220  LinSysRes.SubtractBlock(iPoint, Residual);
221  LinSysRes.AddBlock(jPoint, Residual);
222  Jacobian.UpdateBlocksSub(iEdge, iPoint, jPoint, Jacobian_i, Jacobian_j);
223 
224  }

260 void CRadP1Solver::BC_Isothermal_Wall(CGeometry *geometry, CSolver **solver_container, CNumerics *conv_numerics, CNumerics *visc_numerics, CConfig *config,
261  unsigned short val_marker) {
262 
263  unsigned short iVar, jVar;
264  unsigned long iVertex, iPoint;
265 
266  su2double Theta, Ib_w, Radiative_Energy;
267  su2double *Normal, Area, Wall_Emissivity;
268  su2double Radiative_Heat_Flux;
269  su2double Twall;
270 
271  bool implicit = (config->GetKind_TimeIntScheme_Flow() == EULER_IMPLICIT);
272 
273  /*--- Identify the boundary by string name ---*/
274  string Marker_Tag = config->GetMarker_All_TagBound(val_marker);
275 
276  /*--- Get the specified wall emissivity from config ---*/
277  Wall_Emissivity = config->GetWall_Emissivity(Marker_Tag);
278 
279  /*--- Compute the constant for the wall theta ---*/
280  Theta = Wall_Emissivity / (2.0*(2.0 - Wall_Emissivity));
281 
282  /*--- Retrieve the specified wall temperature ---*/
283  Twall = config->GetIsothermal_Temperature(Marker_Tag)/config->GetTemperature_Ref();
284 
285  /*--- Loop over all of the vertices on this boundary marker ---*/
286 
287  for (iVertex = 0; iVertex < geometry->nVertex[val_marker]; iVertex++) {
288  iPoint = geometry->vertex[val_marker][iVertex]->GetNode();
289 
290  /*--- Check if the node belongs to the domain (i.e, not a halo node) ---*/
291 
292  if (geometry->nodes->GetDomain(iPoint)) {
293 
294  /*--- Compute dual-grid area and boundary normal ---*/
295  Normal = geometry->vertex[val_marker][iVertex]->GetNormal();
296 
297  Area = GeometryToolbox::Norm(nDim, Normal);
298 
299  // Weak application of the boundary condition
300 
301  /*--- Initialize the viscous residuals to zero ---*/
302  for (iVar = 0; iVar < nVar; iVar++) {
303  Res_Visc[iVar] = 0.0;
304  if (implicit) {
305  for (jVar = 0; jVar < nVar; jVar++)
306  Jacobian_i[iVar][jVar] = 0.0;
307  }
308  }
309 
310  /*--- Apply a weak boundary condition for the radiative transfer equation. ---*/
311 
312  /*--- Compute the blackbody intensity at the wall. ---*/
313  Ib_w = 4.0*STEFAN_BOLTZMANN*pow(Twall,4.0);
314 
315  /*--- Compute the radiative heat flux. ---*/
316  Radiative_Energy = nodes->GetSolution(iPoint, 0);
317  Radiative_Heat_Flux = 1.0*Theta*(Ib_w - Radiative_Energy);
318 
319  /*--- Compute the Viscous contribution to the residual ---*/
320  Res_Visc[0] = Radiative_Heat_Flux*Area;
321 
322  /*--- Apply to the residual vector ---*/
323  LinSysRes.SubtractBlock(iPoint, Res_Visc);
324 
325  /*--- Compute the Jacobian contribution. ---*/
326  if (implicit) {
327  Jacobian_i[0][0] = - Theta;
328  Jacobian.SubtractBlock2Diag(iPoint, Jacobian_i);
329  }
330  }
331  }

485 void CRadP1Solver::ImplicitEuler_Iteration(CGeometry *geometry, CSolver **solver_container, CConfig *config) {
486 
487  unsigned short iVar;
488  unsigned long iPoint, total_index, IterLinSol = 0;
489  su2double Vol;
490  su2double Delta;
491 
492  /*--- Set maximum residual to zero ---*/
493 
494  SetResToZero();
495 
496  /*--- Build implicit system ---*/
497 
498  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
499 
500  /*--- Read the volume ---*/
501 
502  Vol = geometry->nodes->GetVolume(iPoint);
503 
504  /*--- Modify matrix diagonal to assure diagonal dominance ---*/
505 
506  if (nodes->GetDelta_Time(iPoint) != 0.0) {
507  Delta = Vol / nodes->GetDelta_Time(iPoint);
508  Jacobian.AddVal2Diag(iPoint, Delta);
509  }
510  else {
511  Jacobian.SetVal2Diag(iPoint, 1.0);
512  for (iVar = 0; iVar < nVar; iVar++) {
513  total_index = iPoint*nVar + iVar;
514  LinSysRes[total_index] = 0.0;
515  }
516  }
517 
518  /*--- Right hand side of the system (-Residual) and initial guess (x = 0) ---*/
519 
520  for (iVar = 0; iVar < nVar; iVar++) {
521  total_index = iPoint*nVar+iVar;
522  LinSysRes[total_index] = - (LinSysRes[total_index]);
523  LinSysSol[total_index] = 0.0;
524  Residual_RMS[iVar] += LinSysRes[total_index]*LinSysRes[total_index];
525  AddRes_Max(iVar, fabs(LinSysRes[total_index]), geometry->nodes->GetGlobalIndex(iPoint), geometry->nodes->GetCoord(iPoint));
526  }
527  }
528 
529  /*--- Initialize residual and solution at the ghost points ---*/
530 
531  for (iPoint = nPointDomain; iPoint < nPoint; iPoint++) {
532  for (iVar = 0; iVar < nVar; iVar++) {
533  total_index = iPoint*nVar + iVar;
534  LinSysRes[total_index] = 0.0;
535  LinSysSol[total_index] = 0.0;
536  }
537  }
538 
539  /*--- Solve or smooth the linear system ---*/
540 
541  IterLinSol = System.Solve(Jacobian, LinSysRes, LinSysSol, geometry, config);
542 
543  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
544  for (iVar = 0; iVar < nVar; iVar++) {
545  nodes->AddSolution(iPoint, iVar, LinSysSol[iPoint*nVar+iVar]);
546  }
547  }
548 
549  /*--- The the number of iterations of the linear solver ---*/
550 
551  SetIterLinSolver(IterLinSol);
552 
553  /*--- MPI solution ---*/
554 
555  InitiateComms(geometry, config, SOLUTION);
556  CompleteComms(geometry, config, SOLUTION);
557 
558  /*--- Compute the root mean square residual ---*/
559 
560  SetResidual_RMS(geometry, config);
561 
562 }

564 void CRadP1Solver::SetTime_Step(CGeometry *geometry, CSolver **solver_container, CConfig *config,
565  unsigned short iMesh, unsigned long Iteration) {
566 
567  unsigned short iMarker;
568  unsigned long iEdge, iVertex, iPoint = 0, jPoint = 0;
569  su2double Area, Vol, Lambda;
570  su2double Global_Delta_Time = 1E6, Local_Delta_Time = 0.0, K_v = 0.25;
571  su2double CFL = config->GetCFL_Rad();
572  su2double GammaP1 = 1.0 / (3.0*(Absorption_Coeff + Scattering_Coeff));
573  const su2double* Normal;
574 
575  /*--- Compute spectral radius based on thermal conductivity ---*/
576 
577  Min_Delta_Time = 1.E6; Max_Delta_Time = 0.0;
578 
579  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
580  nodes->SetMax_Lambda_Visc(iPoint, 0.0);
581  }
582 
583  /*--- Loop interior edges ---*/
584 
585  for (iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
586 
587  iPoint = geometry->edges->GetNode(iEdge,0);
588  jPoint = geometry->edges->GetNode(iEdge,1);
589 
590  /*--- Get the edge's normal vector to compute the edge's area ---*/
591  Normal = geometry->edges->GetNormal(iEdge);
592  Area = GeometryToolbox::Norm(nDim, Normal);
593 
594  /*--- Viscous contribution ---*/
595 
596  Lambda = GammaP1*Area*Area;
597  if (geometry->nodes->GetDomain(iPoint)) nodes->AddMax_Lambda_Visc(iPoint, Lambda);
598  if (geometry->nodes->GetDomain(jPoint)) nodes->AddMax_Lambda_Visc(jPoint, Lambda);
599 
600  }
601 
602  /*--- Loop boundary edges ---*/
603 
604  for (iMarker = 0; iMarker < geometry->GetnMarker(); iMarker++) {
605  for (iVertex = 0; iVertex < geometry->GetnVertex(iMarker); iVertex++) {
606 
607  /*--- Point identification, Normal vector and area ---*/
608 
609  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
610  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
611  Area = GeometryToolbox::Norm(nDim, Normal);
612 
613  /*--- Viscous contribution ---*/
614 
615  Lambda = GammaP1*Area*Area;
616  if (geometry->nodes->GetDomain(iPoint)) nodes->AddMax_Lambda_Visc(iPoint, Lambda);
617 
618  }
619  }
620 
621  /*--- Each element uses their own speed, steady state simulation ---*/
622 
623  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
624 
625  Vol = geometry->nodes->GetVolume(iPoint);
626 
627  if (Vol != 0.0) {
628 
629  /*--- Time step setting method ---*/
630 
631  Local_Delta_Time = CFL*K_v*Vol*Vol/ nodes->GetMax_Lambda_Visc(iPoint);
632
