37 void CBSplineBlending::SetOrder(short val_order, short n_controlpoints){
38 
39  unsigned short iKnot;
40 
41  Order = val_order;
42  Degree = Order - 1;
43  nControl = n_controlpoints;
44 
45  KnotSize = Order + nControl;
46 
47  U.resize(KnotSize, 0.0);
48 
49  /*--- Set up the knot vectors for open uniform B-Splines ---*/
50 
51  /*--- Note: the first knots are zero now.---*/
52 
53  /*--- The next knots are equidistantly distributed in [0,1] ---*/
54 
55  for (iKnot = 0; iKnot < nControl - Order; iKnot++){
56  U[Order + iKnot] = su2double(iKnot+1)/su2double(nControl - Order + 1);
57  }
58  for (iKnot = nControl - Order; iKnot < nControl; iKnot++){
59  U[Order + iKnot] = 1.0;
60  }
61 
62  /*--- Allocate the temporary vectors for the basis evaluation ---*/
63 
64  N.resize(Order, vector<su2double>(Order, 0.0));
65 }

67 su2double CBSplineBlending::GetBasis(short val_i, su2double val_t){
68 
69  /*--- Evaluation is based on the algorithm from "The NURBS Book (Les Piegl and Wayne Tiller)" ---*/
70 
71  /*--- Special cases ---*/
72 
73  if ((val_i == 0 && val_t == U[0]) || (val_i == (short)U.size()-1 && val_t == U.back())) {return 1.0;}
74 
75  /*--- Local property of BSplines ---*/
76 
77  if ((val_t < U[val_i]) || (val_t >= U[val_i+Order])){ return 0.0;}
78 
79  unsigned short j,k;
80  su2double saved, temp;
81 
82  for (j = 0; j < Order; j++){
83  if ((val_t >= U[val_i+j]) && (val_t < U[val_i+j+1])) N[j][0] = 1.0;
84  else N[j][0] = 0;
85  }
86 
87  for (k = 1; k < Order; k++){
88  if (N[0][k-1] == 0.0) saved = 0.0;
89  else saved = ((val_t - U[val_i])*N[0][k-1])/(U[val_i+k] - U[val_i]);
90  for (j = 0; j < Order-k; j++){
91  if (N[j+1][k-1] == 0.0){
92  N[j][k] = saved; saved = 0.0;
93  } else {
94  temp = N[j+1][k-1]/(U[val_i+j+k+1] - U[val_i+j+1]);
95  N[j][k] = saved+(U[val_i+j+k+1] - val_t)*temp;
96  saved = (val_t - U[val_i+j+1])*temp;
97  }
98  }
99  }
100  return N[0][Order-1];
101 }

103 su2double CBSplineBlending::GetDerivative(short val_i, su2double val_t, short val_order_der){
104 
105  if ((val_t < U[val_i]) || (val_t >= U[val_i+Order])){ return 0.0;}
106 
107  /*--- Evaluate the i+p basis functions up to the order p (stored in the matrix N). ---*/
108 
109  GetBasis(val_i, val_t);
110 
111  /*--- Use the recursive definition for the derivative (hardcoded for 1st and 2nd derivative). ---*/
112 
113  if (val_order_der == 0){ return N[0][Order-1];}
114 
115  if (val_order_der == 1){
116  return (Order-1.0)/(1e-10 + U[val_i+Order-1] - U[val_i] )*N[0][Order-2]
117  - (Order-1.0)/(1e-10 + U[val_i+Order] - U[val_i+1])*N[1][Order-2];
118  }
119 
120  if (val_order_der == 2 && Order > 2){
121  const su2double left = (Order-2.0)/(1e-10 + U[val_i+Order-2] - U[val_i]) *N[0][Order-3]
122  - (Order-2.0)/(1e-10 + U[val_i+Order-1] - U[val_i+1])*N[1][Order-3];
123 
124  const su2double right = (Order-2.0)/(1e-10 + U[val_i+Order-1] - U[val_i+1])*N[1][Order-3]
125  - (Order-2.0)/(1e-10 + U[val_i+Order] - U[val_i+2])*N[2][Order-3];
126 
127  return (Order-1.0)/(1e-10 + U[val_i+Order-1] - U[val_i] )*left
128  - (Order-1.0)/(1e-10 + U[val_i+Order] - U[val_i+1])*right;
129  }
130 
131  /*--- Higher order derivatives are not implemented, so we exit if they are requested. ---*/
132 
133  if (val_order_der > 2){
134  SU2_MPI::Error("Higher order derivatives for BSplines are not implemented.", CURRENT_FUNCTION);
135  }
136  return 0.0;
137 }
