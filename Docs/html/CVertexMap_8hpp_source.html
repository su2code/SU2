34 /*!
35  * \class CVertexMap
36  * \ingroup Containers
37  * \brief A lookup type map, maps indices in a large range to indices in a smaller one.
38  *
39  * The usage case is:
40  * 1 - Initialize via Reset for the large range size.
41  * 2 - Use SetIsVertex to define what large range indices have small range correspondence.
42  * 3 - Call Build, vertices are given a sequential index in increasing order of iPoint,
43  * not in order of calls to SetIsVertex. Keep this in mind when allocating vertex data.
44  * 4 - Use GetVertexIndex to convert a point index to vertex index.
45  *
46  * Only consider using this class if you have no reasonable way to keep track of vertices.
47  * For example if you need to assume every point might be a vertex, but you do not want to
48  * allocate data for the entire set of points.
49  *
50  * \note For efficiency use the smallest type that can fit the maximum number of vertices.
51  */
52 template<typename T = unsigned long>
53 class CVertexMap {
54  static_assert(std::is_unsigned<T>::value && std::is_integral<T>::value,
55  "Vertex map requires an unsigned integral type (e.g. unsigned).");
56 
57  private:
58  su2vector<T> Map; /*!< \brief Map from range 0-(nPoint-1) to 1-nVertex. */
59  bool isValid = false; /*!< \brief Set to true when it is safe to use the accessors. */
60  T nVertex = 0; /*!< \brief Number of vertices. */
61 
62  public:
63  /*!
64  * \brief Check if the current mapping is valid.
65  */
66  inline bool GetIsValid() const { return isValid; }
67 
68  /*!
69  * \brief Get the number of vertices currently mapped.
70  */
71  inline T GetnVertex() const { return nVertex; }
72 
73  /*!
74  * \brief Reset the map for size nPoint, marks every point as not-a-vertex.
75  */
76  void Reset(unsigned long nPoint) {
77  Map.resize(nPoint) = 0;
78  nVertex = 0;
79  isValid = true;
80  }
81 
82  /*!
83  * \brief Set the vertex status of a point.
84  */
85  inline bool SetIsVertex(unsigned long iPoint, bool isVertex) {
86  /*--- Invalidate map if change is requested as that destroys it. ---*/
87  if (isVertex != bool(Map(iPoint))) {
88  isValid = false;
89  Map(iPoint) = T(isVertex);
90  }
91  return isValid;
92  }
93 
94  /*!
95  * \brief Get wheter a point is marked as vertex.
96  */
97  inline bool GetIsVertex(unsigned long iPoint) const {
98  return (Map(iPoint) != 0);
99  }
100 
101  /*!
102  * \brief Build the point to vertex map.
103  */
104  T Build() {
105  if (!isValid) {
106  /*--- The map is 1 based, the accessors correct
107  accordingly when returning vertex indices. ---*/
108  nVertex = 0;
109 
110  for (unsigned long iPoint = 0; iPoint < Map.size(); ++iPoint)
111  if (Map(iPoint)!=0)
112  Map(iPoint) = ++nVertex;
113 
114  isValid = true;
115  }
116  return nVertex;
117  }
118 
119  /*!
120  * \brief Convert a point index to vertex index.
121  * \param[in,out] iVertex - On entry point index, on exit vertex index.
122  * \return True if conversion is successful (i.e. point is vertex).
123  */
124  inline bool GetVertexIndex(unsigned long &iVertex) const {
125  assert(isValid && "Vertex map is not in valid state.");
126  iVertex = Map(iVertex);
127  if(iVertex==0) return false; // not a vertex
128  iVertex--; // decrement for 0 based
129  return true; // is a vertex
130  }
131 
132 };
