280  factor = 1.0/Area_Monitored;
281 
282  Weight_ObjFunc = Weight_ObjFunc*factor;
283  config->SetWeight_ObjFunc(iMarker_Monitoring, Weight_ObjFunc);
284 
285  }
286  }
287 
288  /*--- MPI solution ---*/
289 
290  InitiateComms(geometry, config, SOLUTION);
291  CompleteComms(geometry, config, SOLUTION);

310 void CAdjNSSolver::Preprocessing(CGeometry *geometry, CSolver **solver_container, CConfig *config, unsigned short iMesh, unsigned short iRKStep, unsigned short RunTime_EqSystem, bool Output) {
311 
312  unsigned long iPoint, nonPhysicalPoints = 0;
313  su2double SharpEdge_Distance;
314  bool physical = true;
315 
316  /*--- Retrieve information about the spatial and temporal integration for the
317  adjoint equations (note that the flow problem may use different methods). ---*/
318 
319  bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
320  bool limiter = (config->GetKind_SlopeLimit_AdjFlow() != LIMITER::NONE);
321  bool center_jst = (config->GetKind_Centered_AdjFlow() == CENTERED::JST);
322  bool fixed_cl = config->GetFixed_CL_Mode();
323  bool eval_dof_dcx = config->GetEval_dOF_dCX();
324 
325  /*--- Update the objective function coefficient to guarantee zero gradient. ---*/
326 
327  if (fixed_cl && eval_dof_dcx) { SetFarfield_AoA(geometry, solver_container, config, iMesh, Output); }
328 
329  /*--- Residual initialization ---*/
330 
331  for (iPoint = 0; iPoint < nPoint; iPoint ++) {
332 
333  /*--- Get the distance form a sharp edge ---*/
334 
335  SharpEdge_Distance = geometry->nodes->GetSharpEdge_Distance(iPoint);
336 
337  /*--- Set the primitive variables compressible
338  adjoint variables ---*/
339 
340  physical = nodes->SetPrimVar(iPoint,SharpEdge_Distance, false, config);
341 
342  /* Check for non-realizable states for reporting. */
343 
344  if (!physical) nonPhysicalPoints++;
345 
346  /*--- Initialize the convective residual vector ---*/
347 
348  if (!Output) LinSysRes.SetBlock_Zero(iPoint);
349 
350  }
351 
352  /*--- Compute gradients adj for solution reconstruction and viscous term ---*/
353 
354  if (config->GetReconstructionGradientRequired()) {
355  if (config->GetKind_Gradient_Method_Recon() == GREEN_GAUSS)
356  SetSolution_Gradient_GG(geometry, config, true);
357  if (config->GetKind_Gradient_Method_Recon() == LEAST_SQUARES)
358  SetSolution_Gradient_LS(geometry, config, true);
359  if (config->GetKind_Gradient_Method_Recon() == WEIGHTED_LEAST_SQUARES)
360  SetSolution_Gradient_LS(geometry, config, true);
361  }
362  if (config->GetKind_Gradient_Method() == GREEN_GAUSS) SetSolution_Gradient_GG(geometry, config);
363  if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) SetSolution_Gradient_LS(geometry, config);
364 
365  /*--- Limiter computation (upwind reconstruction) ---*/
366 
367  if (limiter && !Output) SetSolution_Limiter(geometry, config);
368 
369  /*--- Compute gradients adj for viscous term coupling ---*/
370 
371  if ((config->GetKind_Solver() == MAIN_SOLVER::ADJ_RANS) && (!config->GetFrozen_Visc_Cont())) {
372  if (config->GetKind_Gradient_Method() == GREEN_GAUSS) solver_container[ADJTURB_SOL]->SetSolution_Gradient_GG(geometry, config);
373  if (config->GetKind_Gradient_Method() == WEIGHTED_LEAST_SQUARES) solver_container[ADJTURB_SOL]->SetSolution_Gradient_LS(geometry, config);
374  }
375 
376  /*--- Artificial dissipation for centered schemes ---*/
377 
378  if (center_jst && (iMesh == MESH_0)) {
379  SetCentered_Dissipation_Sensor(geometry, config);
380  SetUndivided_Laplacian(geometry, config);
381  }
382 
383  /*--- Initialize the Jacobian for implicit integration ---*/
384 
385  if (implicit) Jacobian.SetValZero();
386 
387  /*--- Error message ---*/
388 
389  if (config->GetComm_Level() == COMM_FULL) {
390 #ifdef HAVE_MPI
391  unsigned long MyErrorCounter = nonPhysicalPoints; nonPhysicalPoints = 0;
392  SU2_MPI::Allreduce(&MyErrorCounter, &nonPhysicalPoints, 1, MPI_UNSIGNED_LONG, MPI_SUM, SU2_MPI::GetComm());
393 #endif
394  if (iMesh == MESH_0) config->SetNonphysical_Points(nonPhysicalPoints);
395  }

399 void CAdjNSSolver::Viscous_Residual(CGeometry *geometry, CSolver **solver_container, CNumerics **numerics_container,
400  CConfig *config, unsigned short iMesh, unsigned short iRKStep) {
401 
402  CNumerics* numerics = numerics_container[VISC_TERM];
403 
404  unsigned long iPoint, jPoint, iEdge;
405 
406  bool implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
407 
408  for (iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
409 
410  /*--- Points in edge, coordinates and normal vector---*/
411 
412  iPoint = geometry->edges->GetNode(iEdge,0);
413  jPoint = geometry->edges->GetNode(iEdge,1);
414 
415  numerics->SetCoord(geometry->nodes->GetCoord(iPoint), geometry->nodes->GetCoord(jPoint));
416  numerics->SetNormal(geometry->edges->GetNormal(iEdge));
417 
418  /*--- Primitive variables w/o reconstruction and adjoint variables w/o reconstruction---*/
419 
420  numerics->SetPrimitive(solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(iPoint),
421  solver_container[FLOW_SOL]->GetNodes()->GetPrimitive(jPoint));
422 
423  numerics->SetAdjointVar(nodes->GetSolution(iPoint), nodes->GetSolution(jPoint));
424 
425  /*--- Gradient and limiter of Adjoint Variables ---*/
426 
427  numerics->SetAdjointVarGradient(nodes->GetGradient(iPoint), nodes->GetGradient(jPoint));
428 
429  /*--- Compute residual ---*/
430 
431  numerics->ComputeResidual(Residual_i, Residual_j, Jacobian_ii, Jacobian_ij, Jacobian_ji, Jacobian_jj, config);
432 
433  /*--- Update adjoint viscous residual ---*/
434 
435  LinSysRes.SubtractBlock(iPoint, Residual_i);
436  LinSysRes.AddBlock(jPoint, Residual_j);
437 
438  if (implicit) {
439  Jacobian.SubtractBlock2Diag(iPoint, Jacobian_ii);
440  Jacobian.SubtractBlock(iPoint, jPoint, Jacobian_ij);
441  Jacobian.AddBlock(jPoint, iPoint, Jacobian_ji);
442  Jacobian.AddBlock2Diag(jPoint, Jacobian_jj);
443  }
444 
445  }

509  }
510 
511  /*--- If turbulence computation we must add some coupling terms to the NS adjoint eq. ---*/
512 
513  if ((config->GetKind_Solver() == MAIN_SOLVER::ADJ_RANS) && (!config->GetFrozen_Visc_Cont())) {
514 
515  for (iEdge = 0; iEdge < geometry->GetnEdge(); iEdge++) {
516

518

520  jPoint = geometry->edges->GetNode(iEdge,1);

531  solver_container[FLOW_SOL]->GetNodes()->GetGradient_Primitive(jPoint));
532

534

536  solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(jPoint));
537

539 
540  second_numerics->SetScalarVar(solver_container[TURB_SOL]->GetNodes()->GetSolution(iPoint),
541  solver_container[TURB_SOL]->GetNodes()->GetSolution(jPoint));
542 
543  /*--- Turbulent adjoint variables w/o reconstruction ---*/
544 
545  second_numerics->SetTurbAdjointVar(solver_container[ADJTURB_SOL]->GetNodes()->GetSolution(iPoint),
546  solver_container[ADJTURB_SOL]->GetNodes()->GetSolution(jPoint));
547 
548  /*--- Set distance to the surface ---*/
549 
550  second_numerics->SetDistance(geometry->nodes->GetWall_Distance(iPoint), geometry->nodes->GetWall_Distance(jPoint));
551 
552  /*--- Update adjoint viscous residual ---*/
553 
554  second_numerics->ComputeResidual(Residual, config);
555 
556  LinSysRes.AddBlock(iPoint, Residual);
557  LinSysRes.SubtractBlock(jPoint, Residual);
558  }
559 
560  }
561 
562  // WARNING: The rotating frame source term has been placed in the second
563  // source term container since the section below is commented. This needs a
564  // permanent fix asap!
565 
566  if (rotating_frame) {
567 
568  /*--- Loop over all points ---*/
569  for (iPoint = 0; iPoint < nPointDomain; iPoint++) {
570 
571  /*--- Load the adjoint variables ---*/
572  second_numerics->SetAdjointVar(nodes->GetSolution(iPoint),
573  nodes->GetSolution(iPoint));
574 
575  /*--- Load the volume of the dual mesh cell ---*/
576  second_numerics->SetVolume(geometry->nodes->GetVolume(iPoint));
577 
578  /*--- Compute the adjoint rotating frame source residual ---*/
579  second_numerics->ComputeResidual(Residual, Jacobian_i, config);
580 
581  /*--- Add the source residual to the total ---*/
582  LinSysRes.AddBlock(iPoint, Residual);
583 
584  /*--- Add the implicit Jacobian contribution ---*/
585  if (implicit) Jacobian.AddBlock2Diag(iPoint, Jacobian_i);
586 
587  }
588  }
589 }

591 void CAdjNSSolver::Viscous_Sensitivity(CGeometry *geometry, CSolver **solver_container, CNumerics *numerics, CConfig *config) {
592 
593  unsigned long iVertex, iPoint;
594  unsigned short iDim, jDim, iMarker, iPos, jPos;
595  su2double *d = nullptr, div_phi, *Normal = nullptr, Area,
596  normal_grad_psi5, normal_grad_T, sigma_partial, Laminar_Viscosity = 0.0, heat_flux_factor, temp_sens = 0.0,
597  *Psi = nullptr, *U = nullptr, Enthalpy, gradPsi5_v, psi5_tau_partial, psi5_tau_grad_vel, source_v_1, Density,
598  Pressure = 0.0, div_vel, val_turb_ke, vartheta, vartheta_partial, psi5_p_div_vel, Omega[3], rho_v[3] = {0.0,0.0,0.0},
599  CrossProduct[3], r, ru, rv, rw, rE, p, T, dp_dr, dp_dru,
600  dp_drv, dp_drw, dp_drE, dH_dr, dH_dru, dH_drv, dH_drw, dH_drE, H, D[3][3], Dd[3], Mach_Inf, eps, scale = 1.0,
601  RefVel2, RefDensity, Mach2Vel, *Velocity_Inf, factor;
602 
603  su2double *USens = new su2double[nVar];
604  su2double *UnitNormal = new su2double[nDim];
605  su2double *normal_grad_vel = new su2double[nDim];
606  su2double *tang_deriv_psi5 = new su2double[nDim];
607  su2double *tang_deriv_T = new su2double[nDim];
608  su2double **Sigma = new su2double* [nDim];
609 
610  for (iDim = 0; iDim < nDim; iDim++)
611  Sigma[iDim] = new su2double [nDim];
612 
613  su2double *normal_grad_gridvel = new su2double[nDim];
614  su2double *normal_grad_v_ux =new su2double[nDim];
615  su2double **Sigma_Psi5v = new su2double* [nDim];
616  for (iDim = 0; iDim < nDim; iDim++)
617  Sigma_Psi5v[iDim] = new su2double [nDim];
618  su2double **tau = new su2double* [nDim];
619  for (iDim = 0; iDim < nDim; iDim++)
620  tau[iDim] = new su2double [nDim];
621  su2double *Velocity = new su2double[nDim];
622 
623  bool rotating_frame = config->GetRotating_Frame();
624  bool grid_movement = config->GetGrid_Movement();
625  su2double RefArea = config->GetRefArea();
626  su2double Mach_Motion = config->GetMach_Motion();
627  unsigned short ObjFunc = config->GetKind_ObjFunc();
628  su2double Gas_Constant = config->GetGas_ConstantND();
629  su2double Cp = (Gamma / Gamma_Minus_One) * Gas_Constant;
630  su2double Prandtl_Lam = config->GetPrandtl_Lam();
631 
632  if (config->GetSystemMeasurements() == US) scale = 1.0/12.0;
633  else scale = 1.0;
634 
635  /*--- Compute non-dimensional factor. For dynamic meshes, use the motion Mach
636  number as a reference value for computing the force coefficients.
637  Otherwise, use the freestream values,
638  which is the standard convention. ---*/
639 
640  if (grid_movement) {
641  Mach2Vel = sqrt(Gamma*Gas_Constant*config->GetTemperature_FreeStreamND());
642  RefVel2 = (Mach_Motion*Mach2Vel)*(Mach_Motion*Mach2Vel);
643  }
644  else {
645  Velocity_Inf = config->GetVelocity_FreeStreamND();
646  RefVel2 = 0.0;
647  for (iDim = 0; iDim < nDim; iDim++)
648  RefVel2 += Velocity_Inf[iDim]*Velocity_Inf[iDim];
649  }
650 
651  RefDensity = config->GetDensity_FreeStreamND();
652 
653  factor = 1.0;
654  /*-- For multi-objective problems these scaling factors are applied before solution ---*/
655  if (config->GetnObj()==1) {
656  factor = 1.0/(0.5*RefDensity*RefArea*RefVel2);
657 
658  if ((ObjFunc == INVERSE_DESIGN_HEATFLUX) ||
659  (ObjFunc == TOTAL_HEATFLUX) || (ObjFunc == MAXIMUM_HEATFLUX) ||
660  (ObjFunc == SURFACE_MASSFLOW))
661  factor = 1.0;
662 
663  if ((ObjFunc == SURFACE_TOTAL_PRESSURE) || (ObjFunc == SURFACE_STATIC_PRESSURE))
664  factor = 1.0/Area_Monitored;
665 
666  }
667 
668 
669  /*--- Compute gradient of the grid velocity, if applicable ---*/
670 
671  if (grid_movement)
672  SetGridVel_Gradient(geometry, config);
673 
674  Total_Sens_Geo = 0.0;
675  Total_Sens_Mach = 0.0;
676  Total_Sens_AoA = 0.0;
677  Total_Sens_Press = 0.0;
678  Total_Sens_Temp = 0.0;
679 
680  for (iMarker = 0; iMarker < nMarker; iMarker++) {
681 
682  Sens_Geo[iMarker] = 0.0;
683 
684  if ((config->GetMarker_All_KindBC(iMarker) == HEAT_FLUX) ||
685  (config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL)) {
686 
687  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
688 
689  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
690 
691  if (geometry->nodes->GetDomain(iPoint)) {
692 
693  const auto PsiVar_Grad = nodes->GetGradient(iPoint);
694  const auto PrimVar_Grad = solver_container[FLOW_SOL]->GetNodes()->GetGradient_Primitive(iPoint);
695 
696  Laminar_Viscosity = solver_container[FLOW_SOL]->GetNodes()->GetLaminarViscosity(iPoint);
697 
698  heat_flux_factor = Cp * Laminar_Viscosity / Prandtl_Lam;
699 
700  /*--- Compute face area and the unit normal to the surface ---*/
701 
702  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
703  Area = GeometryToolbox::Norm(nDim, Normal);
704  for (iDim = 0; iDim < nDim; iDim++) { UnitNormal[iDim] = Normal[iDim] / Area; }
705 
706  /*--- Compute the sensitivity related to the temperature ---*/
707 
708 
709  normal_grad_psi5 = 0.0; normal_grad_T = 0.0;
710  for (iDim = 0; iDim < nDim; iDim++) {
711  normal_grad_psi5 += PsiVar_Grad[nVar-1][iDim]*UnitNormal[iDim];
712  normal_grad_T += PrimVar_Grad[0][iDim]*UnitNormal[iDim];
713  }
714 
715  temp_sens = 0.0;
716  if (config->GetMarker_All_KindBC(iMarker) == HEAT_FLUX) {
717 
718  /*--- Heat Flux Term: temp_sens = (\partial_tg \psi_5)\cdot (k \partial_tg T) ---*/
719 
720  for (iDim = 0; iDim < nDim; iDim++) {
721  tang_deriv_psi5[iDim] = PsiVar_Grad[nVar-1][iDim] - normal_grad_psi5*UnitNormal[iDim];
722  tang_deriv_T[iDim] = PrimVar_Grad[0][iDim] - normal_grad_T*UnitNormal[iDim];
723  }
724  for (iDim = 0; iDim < nDim; iDim++)
725  temp_sens += heat_flux_factor * tang_deriv_psi5[iDim] * tang_deriv_T[iDim];
726 
727  } else if (config->GetMarker_All_KindBC(iMarker) == ISOTHERMAL) {
728 
729  /*--- Isothermal Term: temp_sens = - k * \partial_n(\psi_5) * \partial_n(T) ---*/
730 
731  temp_sens = - heat_flux_factor * normal_grad_psi5 * normal_grad_T;
732 
733  }
734 
735 
736  /*--- Term: sigma_partial = \Sigma_{ji} n_i \partial_n v_j ---*/
737 
738  div_phi = 0.0;
739  for (iDim = 0; iDim < nDim; iDim++) {
740  div_phi += PsiVar_Grad[iDim+1][iDim];
741  for (jDim = 0; jDim < nDim; jDim++)
742  Sigma[iDim][jDim] = Laminar_Viscosity * (PsiVar_Grad[iDim+1][jDim]+PsiVar_Grad[jDim+1][iDim]);
743  }
744  for (iDim = 0; iDim < nDim; iDim++)
745  Sigma[iDim][iDim] -= TWO3*Laminar_Viscosity * div_phi;
746 
747 
748  for (iDim = 0; iDim < nDim; iDim++) {
749  normal_grad_vel[iDim] = 0.0;
750  for (jDim = 0; jDim < nDim; jDim++)
751  normal_grad_vel[iDim] += PrimVar_Grad[iDim+1][jDim]*UnitNormal[jDim];
752  }
753 
754  sigma_partial = 0.0;
755  for (iDim = 0; iDim < nDim; iDim++)
756  for (jDim = 0; jDim < nDim; jDim++)
757  sigma_partial += UnitNormal[iDim]*Sigma[iDim][jDim]*normal_grad_vel[jDim];
758 
759  /*--- Compute additional terms in the surface sensitivity for
760  moving walls in a rotating frame or dynamic mesh problem. ---*/
761 
762  if (grid_movement) {
763 
764  Psi = nodes->GetSolution(iPoint);
765  U = solver_container[FLOW_SOL]->GetNodes()->GetSolution(iPoint);
766  Density = U[0];
767  Pressure = solver_container[FLOW_SOL]->GetNodes()->GetPressure(iPoint);
768  Enthalpy = solver_container[FLOW_SOL]->GetNodes()->GetEnthalpy(iPoint);
769 
770  /*--- Turbulent kinetic energy ---*/
771  /// TODO: This does not seem to be consistent with the primal treatment.
772  if (config->GetKind_Turb_Model() == TURB_MODEL::SST)
773  val_turb_ke = solver_container[TURB_SOL]->GetNodes()->GetSolution(iPoint,0);
774  else
775  val_turb_ke = 0.0;
776 
777  CNumerics::ComputeStressTensor(nDim, tau, PrimVar_Grad+1, Laminar_Viscosity, Density, val_turb_ke);
778 
779  /*--- Form normal_grad_gridvel = \partial_n (u_omega) ---*/
780 
781  const auto GridVel_Grad = geometry->nodes->GetGridVel_Grad(iPoint);
782  for (iDim = 0; iDim < nDim; iDim++) {
783  normal_grad_gridvel[iDim] = 0.0;
784  for (jDim = 0; jDim < nDim; jDim++)
785  normal_grad_gridvel[iDim] += GridVel_Grad[iDim][jDim]*UnitNormal[jDim];
786  }
787 
788  /*--- Form normal_grad_v_ux = \partial_n (v - u_omega) ---*/
789 
790  for (iDim = 0; iDim < nDim; iDim++)
791  normal_grad_v_ux[iDim] = normal_grad_vel[iDim] - normal_grad_gridvel[iDim];
792 
793  /*--- Form Sigma_Psi5v ---*/
794 
795  div_vel = 0.0;
796  for (iDim = 0 ; iDim < nDim; iDim++) {
797  Velocity[iDim] = U[iDim+1]/Density;
798  div_vel += PrimVar_Grad[iDim+1][iDim];
799  }
800 
801  gradPsi5_v = 0.0;
802  for (iDim = 0; iDim < nDim; iDim++) {
803  gradPsi5_v += PsiVar_Grad[nDim+1][iDim]*Velocity[iDim];
804  for (jDim = 0; jDim < nDim; jDim++)
805  Sigma_Psi5v[iDim][jDim] = Laminar_Viscosity * (PsiVar_Grad[nDim+1][iDim]*Velocity[jDim]+PsiVar_Grad[nDim+1][jDim]*Velocity[iDim]);
806  }
807  for (iDim = 0; iDim < nDim; iDim++)
808  Sigma_Psi5v[iDim][iDim] -= TWO3*Laminar_Viscosity * gradPsi5_v;
809 
810 
811  /*--- Now compute terms of the surface sensitivity ---*/
812 
813  /*--- Form vartheta_partial = \vartheta * \partial_n (v - u_x) . n ---*/
814  vartheta = Density*Psi[0] + Density*Enthalpy*Psi[nDim+1];
815  for (iDim = 0; iDim < nDim; iDim++) {
816  vartheta += U[iDim+1]*Psi[iDim+1];
817  }
818  vartheta_partial = 0.0;
819  for (iDim = 0; iDim < nDim; iDim++)
820  vartheta_partial += vartheta * normal_grad_v_ux[iDim] * UnitNormal[iDim];
821 
822  /*--- Form sigma_partial = n_i ( \Sigma_Phi_{ij} + \Sigma_Psi5v_{ij} ) \partial_n (v - u_x)_j ---*/
823 
824  sigma_partial = 0.0;
825  for (iDim = 0; iDim < nDim; iDim++)
826  for (jDim = 0; jDim < nDim; jDim++)
827  sigma_partial += UnitNormal[iDim]*(Sigma[iDim][jDim]+Sigma_Psi5v[iDim][jDim])*normal_grad_v_ux[jDim];
828 
829  /*--- Form psi5_tau_partial = \Psi_5 * \partial_n (v - u_x)_i * tau_{ij} * n_j ---*/
830 
831  psi5_tau_partial = 0.0;
832  for (iDim = 0; iDim < nDim; iDim++)
833  for (jDim = 0; jDim < nDim; jDim++)
834  psi5_tau_partial -= Psi[nDim+1]*normal_grad_v_ux[iDim]*tau[iDim][jDim]*UnitNormal[jDim];
835 
836  /*--- Form psi5_p_div_vel = ---*/
837 
838  psi5_p_div_vel = -Psi[nDim+1]*Pressure*div_vel;
839 
840  /*--- Form psi5_tau_grad_vel = \Psi_5 * tau_{ij} : \nabla( v ) ---*/
841 
842  psi5_tau_grad_vel = 0.0;
843  for (iDim = 0; iDim < nDim; iDim++)
844  for (jDim = 0; jDim < nDim; jDim++)
845  psi5_tau_grad_vel += Psi[nDim+1]*tau[iDim][jDim]*PrimVar_Grad[iDim+1][jDim];
846 
847  /*--- Retrieve the angular velocity vector ---*/
848 
849  source_v_1 = 0.0;
850  if (rotating_frame) {
851 
852  for (iDim = 0; iDim < 3; iDim++){
853  Omega[iDim] = config->GetRotation_Rate(iDim)/config->GetOmega_Ref();
854  }
855 
856  /*--- Calculate momentum source terms as: rho * ( Omega X V ) ---*/
857 
858  for (iDim = 0; iDim < nDim; iDim++)
859  rho_v[iDim] = U[iDim+1];
860  if (nDim == 2) rho_v[2] = 0.0;
861 
862  CrossProduct[0] = Omega[1]*rho_v[2] - Omega[2]*rho_v[1];
863  CrossProduct[1] = Omega[2]*rho_v[0] - Omega[0]*rho_v[2];
864  CrossProduct[2] = Omega[0]*rho_v[1] - Omega[1]*rho_v[0];
865 
866 
867  for (iDim = 0; iDim < nDim; iDim++) {
868  source_v_1 += Psi[iDim+1]*CrossProduct[iDim];
869  }
870  }
871 
872  /*--- For simplicity, store all additional terms within sigma_partial ---*/
873 
874  sigma_partial = sigma_partial + vartheta_partial + psi5_tau_partial + psi5_p_div_vel + psi5_tau_grad_vel + source_v_1;
875 
876  }
877 
878  /*--- Compute sensitivity for each surface point ---*/
879 
880  CSensitivity[iMarker][iVertex] = (sigma_partial - temp_sens) * Area * scale * factor;
881 
882  /*--- If sharp edge, set the sensitivity to 0 on that region ---*/
883 
884  if (config->GetSens_Remove_Sharp()) {
885  eps = config->GetVenkat_LimiterCoeff()*config->GetRefElemLength();
886  if ( geometry->nodes->GetSharpEdge_Distance(iPoint) < config->GetAdjSharp_LimiterCoeff()*eps )
887  CSensitivity[iMarker][iVertex] = 0.0;
888  }
889 
890  Sens_Geo[iMarker] -= CSensitivity[iMarker][iVertex];
891 
892  }
893  }
894 
895  Total_Sens_Geo += Sens_Geo[iMarker];
896 
897  }
898  }
899 
900  /*--- Farfield Sensitivity (Mach, AoA, Press, Temp), only for compressible flows ---*/
901 
902  for (iMarker = 0; iMarker < nMarker; iMarker++) {
903 
904  if (config->GetMarker_All_KindBC(iMarker) == FAR_FIELD || config->GetMarker_All_KindBC(iMarker) == INLET_FLOW ||
905  config->GetMarker_All_KindBC(iMarker) == SUPERSONIC_INLET || config->GetMarker_All_KindBC(iMarker) == SUPERSONIC_OUTLET ||
906  config->GetMarker_All_KindBC(iMarker) == ENGINE_INFLOW ) {
907 
908  Sens_Mach[iMarker] = 0.0;
909  Sens_AoA[iMarker] = 0.0;
910  Sens_Press[iMarker] = 0.0;
911  Sens_Temp[iMarker] = 0.0;
912  Sens_BPress[iMarker] = 0.0;
913 
914  for (iVertex = 0; iVertex < geometry->nVertex[iMarker]; iVertex++) {
915  iPoint = geometry->vertex[iMarker][iVertex]->GetNode();
916 
917  if (geometry->nodes->GetDomain(iPoint)) {
918  Psi = nodes->GetSolution(iPoint);
919  U = solver_container[FLOW_SOL]->GetNodes()->GetSolution(iPoint);
920  Normal = geometry->vertex[iMarker][iVertex]->GetNormal();
921 
922  Mach_Inf = config->GetMach();
923  if (grid_movement) Mach_Inf = config->GetMach_Motion();
924 
925  r = U[0]; ru = U[1]; rv = U[2];
926  if (nDim == 2) { rw = 0.0; rE = U[3]; }
927  else { rw = U[3]; rE = U[4]; }
928  p = Gamma_Minus_One*(rE-(ru*ru + rv*rv + rw*rw)/(2*r));
929 
930  Area = GeometryToolbox::Norm(nDim, Normal);
931  for (iDim = 0; iDim < nDim; iDim++) UnitNormal[iDim] = -Normal[iDim]/Area;
932 
933  H = (rE + p)/r;
934 
935  dp_dr = Gamma_Minus_One*(ru*ru + rv*rv + rw*rw)/(2*r*r);
936  dp_dru = -Gamma_Minus_One*ru/r;
937  dp_drv = -Gamma_Minus_One*rv/r;
938  if (nDim == 2) { dp_drw = 0.0; dp_drE = Gamma_Minus_One; }
939  else { dp_drw = -Gamma_Minus_One*rw/r; dp_drE = Gamma_Minus_One; }
940 
941  dH_dr = (-H + dp_dr)/r; dH_dru = dp_dru/r; dH_drv = dp_drv/r;
942  if (nDim == 2) { dH_drw = 0.0; dH_drE = (1 + dp_drE)/r; }
943  else { dH_drw = dp_drw/r; dH_drE = (1 + dp_drE)/r; }
944 
945  if (nDim == 2) {
946  Jacobian_j[0][0] = 0.0;
947  Jacobian_j[1][0] = Area*UnitNormal[0];
948  Jacobian_j[2][0] = Area*UnitNormal[1];
949  Jacobian_j[3][0] = 0.0;
950 
951  Jacobian_j[0][1] = (-(ru*ru)/(r*r) + dp_dr)*Area*UnitNormal[0] + (-(ru*rv)/(r*r))*Area*UnitNormal[1];
952  Jacobian_j[1][1] = (2*ru/r + dp_dru)*Area*UnitNormal[0] + (rv/r)*Area*UnitNormal[1];
953  Jacobian_j[2][1] = (dp_drv)*Area*UnitNormal[0] + (ru/r)*Area*UnitNormal[1];
954  Jacobian_j[3][1] = (dp_drE)*Area*UnitNormal[0];
955 
956  Jacobian_j[0][2] = (-(ru*rv)/(r*r))*Area*UnitNormal[0] + (-(rv*rv)/(r*r) + dp_dr)*Area*UnitNormal[1];
957  Jacobian_j[1][2] = (rv/r)*Area*UnitNormal[0] + (dp_dru)*Area*UnitNormal[1];
958  Jacobian_j[2][2] = (ru/r)*Area*UnitNormal[0] + (2*rv/r + dp_drv)*Area*UnitNormal[1];
959  Jacobian_j[3][2] = (dp_drE)*Area*UnitNormal[1];
960 
961  Jacobian_j[0][3] = (ru*dH_dr)*Area*UnitNormal[0] + (rv*dH_dr)*Area*UnitNormal[1];
962  Jacobian_j[1][3] = (H + ru*dH_dru)*Area*UnitNormal[0] + (rv*dH_dru)*Area*UnitNormal[1];
963  Jacobian_j[2][3] = (ru*dH_drv)*Area*UnitNormal[0] + (H + rv*dH_drv)*Area*UnitNormal[1];
964  Jacobian_j[3][3] = (ru*dH_drE)*Area*UnitNormal[0] + (rv*dH_drE)*Area*UnitNormal[1];
965  }
966  else {
967  Jacobian_j[0][0] = 0.0;
968  Jacobian_j[1][0] = Area*UnitNormal[0];
969  Jacobian_j[2][0] = Area*UnitNormal[1];
970  Jacobian_j[3][0] = Area*UnitNormal[2];
971  Jacobian_j[4][0] = 0.0;
972 
973  Jacobian_j[0][1] = (-(ru*ru)/(r*r) + dp_dr)*Area*UnitNormal[0] + (-(ru*rv)/(r*r))*Area*UnitNormal[1] + (-(ru*rw)/(r*r))*Area*UnitNormal[2];
974  Jacobian_j[1][1] = (2*ru/r + dp_dru)*Area*UnitNormal[0] + (rv/r)*Area*UnitNormal[1] + (rw/r)*Area*UnitNormal[2];
975  Jacobian_j[2][1] = (dp_drv)*Area*UnitNormal[0] + (ru/r)*Area*UnitNormal[1];
976  Jacobian_j[3][1] = (dp_drw)*Area*UnitNormal[0] + (ru/r)*Area*UnitNormal[2];
977  Jacobian_j[4][1] = (dp_drE)*Area*UnitNormal[0];
978 
979  Jacobian_j[0][2] = (-(ru*rv)/(r*r))*Area*UnitNormal[0] + (-(rv*rv)/(r*r) + dp_dr)*Area*UnitNormal[1] + (-(rv*rw)/(r*r))*Area*UnitNormal[2];
980  Jacobian_j[1][2] = (rv/r)*Area*UnitNormal[0] + (dp_dru)*Area*UnitNormal[1];
981  Jacobian_j[2][2] = (ru/r)*Area*UnitNormal[0] + (2*rv/r + dp_drv)*Area*UnitNormal[1] + (rw/r)*Area*UnitNormal[2];
982  Jacobian_j[3][2] = (dp_drw)*Area*UnitNormal[1] + (rv/r)*Area*UnitNormal[2];
983  Jacobian_j[4][2] = (dp_drE)*Area*UnitNormal[1];
984 
985  Jacobian_j[0][3] = (-(ru*rw)/(r*r))*Area*UnitNormal[0] + (-(rv*rw)/(r*r))*Area*UnitNormal[1] + (-(rw*rw)/(r*r) + dp_dr)*Area*UnitNormal[2];
986  Jacobian_j[1][3] = (rw/r)*Area*UnitNormal[0] + (dp_dru)*Area*UnitNormal[2];
987  Jacobian_j[2][3] = (rw/r)*Area*UnitNormal[1] + (dp_drv)*Area*UnitNormal[2];
988  Jacobian_j[3][3] = (ru/r)*Area*UnitNormal[0] + (rv/r)*Area*UnitNormal[1] + (2*rw/r + dp_drw)*Area*UnitNormal[2];
989  Jacobian_j[4][3] = (dp_drE)*Area*UnitNormal[2];
