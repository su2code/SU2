30 CPrimalGridFEM::CPrimalGridFEM(unsigned long val_elemGlobalID, unsigned short val_VTK_Type,
31  unsigned short val_nPolyGrid, unsigned short val_nPolySol,
32  unsigned short val_nDOFsGrid, unsigned short val_nDOFsSol,
33  unsigned long val_offDOfsSol, std::istringstream &elem_line)
34  : CPrimalGrid(true, val_nDOFsGrid, nFacesOfElementType(val_VTK_Type))
35 {
36  /*--- Store the integer data in the member variables of this object. ---*/
37  VTK_Type = val_VTK_Type;
38  nFaces = nFacesOfElementType(VTK_Type);
39 
40  nPolyGrid = val_nPolyGrid;
41  nPolySol = val_nPolySol;
42  nDOFsGrid = val_nDOFsGrid;
43  nDOFsSol = val_nDOFsSol;
44 
45  elemIDGlobal = val_elemGlobalID;
46  offsetDOFsSolGlobal = val_offDOfsSol;
47 
48  /*--- Read face structure of the element from elem_line. ---*/
49 
50  for(unsigned short i=0; i<nDOFsGrid; i++)
51  elem_line >> Nodes[i];
52 
53  /*--- If a linear element is used, the node numbering for non-simplices
54  must be adapted. The reason is that compatability with the original
55  SU2 format is maintained for linear elements, but for the FEM solver
56  the nodes of the elements are stored row-wise. ---*/
57  if(nPolyGrid == 1){
58  switch( VTK_Type ) {
59 
60  case QUADRILATERAL:
61  std::swap(Nodes[2], Nodes[3]);
62  break;
63 
64  case HEXAHEDRON:
65  std::swap(Nodes[2], Nodes[3]);
66  std::swap(Nodes[6], Nodes[7]);
67  break;
68 
69  case PYRAMID:
70  std::swap(Nodes[2], Nodes[3]);
71  break;
72  }
73  }
74 }

99 void CPrimalGridFEM::GetLocalCornerPointsAllFaces(unsigned short elementType,
100  unsigned short nPoly,
101  unsigned short nDOFs,
102  unsigned short &numFaces,
103  unsigned short nPointsPerFace[],
104  unsigned long faceConn[6][4]) {
105 
106  /*--- Determine the element type and set the face data accordingly.
107  The faceConn values are local to the element. ---*/
108 
109  numFaces = nFacesOfElementType(elementType);
110  unsigned short nn2, nn3, nn4;
111  switch( elementType ) {
112  case TRIANGLE:
113  nPointsPerFace[0] = 2; faceConn[0][0] = 0; faceConn[0][1] = nPoly;
114  nPointsPerFace[1] = 2; faceConn[1][0] = nPoly; faceConn[1][1] = nDOFs -1;
115  nPointsPerFace[2] = 2; faceConn[2][0] = nDOFs -1; faceConn[2][1] = 0;
116  break;
117 
118  case QUADRILATERAL:
119  nn2 = nPoly*(nPoly+1);
120  nPointsPerFace[0] = 2; faceConn[0][0] = 0; faceConn[0][1] = nPoly;
121  nPointsPerFace[1] = 2; faceConn[1][0] = nPoly; faceConn[1][1] = nDOFs -1;
122  nPointsPerFace[2] = 2; faceConn[2][0] = nDOFs -1; faceConn[2][1] = nn2;
123  nPointsPerFace[3] = 2; faceConn[3][0] = nn2; faceConn[3][1] = 0;
124  break;
125 
126  case TETRAHEDRON:
127  nn2 = (nPoly+1)*(nPoly+2)/2 -1; nn3 = nDOFs -1;
128  nPointsPerFace[0] = 3; faceConn[0][0] = 0; faceConn[0][1] = nPoly; faceConn[0][2] = nn2;
129  nPointsPerFace[1] = 3; faceConn[1][0] = 0; faceConn[1][1] = nn3; faceConn[1][2] = nPoly;
130  nPointsPerFace[2] = 3; faceConn[2][0] = 0; faceConn[2][1] = nn2; faceConn[2][2] = nn3;
131  nPointsPerFace[3] = 3; faceConn[3][0] = nPoly; faceConn[3][1] = nn3; faceConn[3][2] = nn2;
132  break;
133 
134  case PYRAMID:
135  nn2 = (nPoly+1)*(nPoly+1) -1; nn3 = nn2 - nPoly;
136  nPointsPerFace[0] = 4; faceConn[0][0] = 0; faceConn[0][1] = nPoly; faceConn[0][2] = nn2; faceConn[0][3] = nn3;
137  nPointsPerFace[1] = 3; faceConn[1][0] = 0; faceConn[1][1] = nDOFs -1; faceConn[1][2] = nPoly;
138  nPointsPerFace[2] = 3; faceConn[2][0] = nn3; faceConn[2][1] = nn2; faceConn[2][2] = nDOFs -1;
139  nPointsPerFace[3] = 3; faceConn[3][0] = 0; faceConn[3][1] = nn3; faceConn[3][2] = nDOFs -1;
140  nPointsPerFace[4] = 3; faceConn[4][0] = nPoly; faceConn[4][1] = nDOFs -1; faceConn[4][2] = nn2;
141  break;
142 
143  case PRISM:
144  nn2 = (nPoly+1)*(nPoly+2)/2; nn3 = nPoly*nn2; --nn2;
145  nPointsPerFace[0] = 3; faceConn[0][0] = 0; faceConn[0][1] = nPoly; faceConn[0][2] = nn2;
146  nPointsPerFace[1] = 3; faceConn[1][0] = nn3; faceConn[1][1] = nn2+nn3; faceConn[1][2] = nPoly+nn3;
147  nPointsPerFace[2] = 4; faceConn[2][0] = 0; faceConn[2][1] = nn3; faceConn[2][2] = nPoly+nn3; faceConn[2][3] = nPoly;
148  nPointsPerFace[3] = 4; faceConn[3][0] = 0; faceConn[3][1] = nn2; faceConn[3][2] = nn2+nn3; faceConn[3][3] = nn3;
149  nPointsPerFace[4] = 4; faceConn[4][0] = nPoly; faceConn[4][1] = nPoly+nn3; faceConn[4][2] = nn2+nn3; faceConn[4][3] = nn2;
150  break;
151 
152  case HEXAHEDRON:
153  nn2 = (nPoly+1)*(nPoly+1); nn4 = nPoly*nn2; --nn2; nn3 = nn2 - nPoly;
154  nPointsPerFace[0] = 4; faceConn[0][0] = 0; faceConn[0][1] = nPoly; faceConn[0][2] = nn2; faceConn[0][3] = nn3;
155  nPointsPerFace[1] = 4; faceConn[1][0] = nn4; faceConn[1][1] = nn3+nn4; faceConn[1][2] = nn2+nn4; faceConn[1][3] = nPoly+nn4;
156  nPointsPerFace[2] = 4; faceConn[2][0] = 0; faceConn[2][1] = nn4; faceConn[2][2] = nPoly+nn4; faceConn[2][3] = nPoly;
157  nPointsPerFace[3] = 4; faceConn[3][0] = nn3; faceConn[3][1] = nn2; faceConn[3][2] = nn2+nn4; faceConn[3][3] = nn3+nn4;
158  nPointsPerFace[4] = 4; faceConn[4][0] = 0; faceConn[4][1] = nn3; faceConn[4][2] = nn3+nn4; faceConn[4][3] = nn4;
159  nPointsPerFace[5] = 4; faceConn[5][0] = nPoly; faceConn[5][1] = nPoly+nn4; faceConn[5][2] = nn2+nn4; faceConn[5][3] = nn2;
160  break;
161  }
162 }

164 void CPrimalGridFEM::GetCornerPointsAllFaces(unsigned short &numFaces,
165  unsigned short nPointsPerFace[],
166  unsigned long faceConn[6][4]) const {
167 
168  /*--- Get the corner points of the faces local to the element. ---*/
169 
170  GetLocalCornerPointsAllFaces(VTK_Type, nPolyGrid, nDOFsGrid,
171  numFaces, nPointsPerFace, faceConn);
172 
173  /*--- Convert the local values of faceConn to global values. ---*/
174 
175  for(unsigned short i=0; i<numFaces; ++i) {
176  for(unsigned short j=0; j<nPointsPerFace[i]; ++j) {
177  unsigned long nn = faceConn[i][j];
178  faceConn[i][j] = Nodes[nn];
179  }
180  }
