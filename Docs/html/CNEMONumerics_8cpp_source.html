83 void CNEMONumerics::GetInviscidProjFlux(const su2double *val_U,
84  const su2double *val_V,
85  const su2double *val_normal,
86  su2double *val_Proj_Flux) {
87  unsigned short iSpecies, iVar;
88  su2double rho, u, v, w, rhoEve, P, H;
89  const su2double *rhos;
90 
91  /*--- Initialize vectors ---*/
92  for (iVar = 0; iVar < nVar; iVar++)
93  val_Proj_Flux[iVar] = 0.0;
94 
95  /*--- Rename for convienience ---*/
96  rho = val_V[RHO_INDEX];
97  u = val_V[VEL_INDEX];
98  v = val_V[VEL_INDEX+1];
99  w = val_V[VEL_INDEX+2];
100  P = val_V[P_INDEX];
101  H = val_V[H_INDEX];
102  rhoEve = val_U[nSpecies+nDim+1];
103  rhos = &val_V[RHOS_INDEX];
104 
105  if (nDim == 2) {
106 
107  /*--- iDim = 0 (x-direction) ---*/
108  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
109  val_Proj_Flux[iSpecies] = (rhos[iSpecies]*u) * val_normal[0];
110  val_Proj_Flux[nSpecies] = (rho*u*u + P) * val_normal[0];
111  val_Proj_Flux[nSpecies+1] = (rho*u*v) * val_normal[0];
112  val_Proj_Flux[nSpecies+2] = (rho*u*H) * val_normal[0];
113  val_Proj_Flux[nSpecies+3] = (rhoEve*u) * val_normal[0];
114 
115  /*---- iDim = 1 (y-direction) ---*/
116  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
117  val_Proj_Flux[iSpecies] += (rhos[iSpecies]*v) * val_normal[1];
118  val_Proj_Flux[nSpecies] += (rho*v*u) * val_normal[1];
119  val_Proj_Flux[nSpecies+1] += (rho*v*v + P) * val_normal[1];
120  val_Proj_Flux[nSpecies+2] += (rho*v*H) * val_normal[1];
121  val_Proj_Flux[nSpecies+3] += (rhoEve*v) * val_normal[1];
122  }
123  else {
124 
125  /*--- iDim = 0 (x-direction) ---*/
126  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
127  val_Proj_Flux[iSpecies] = (rhos[iSpecies]*u) * val_normal[0];
128  val_Proj_Flux[nSpecies] = (rho*u*u + P) * val_normal[0];
129  val_Proj_Flux[nSpecies+1] = (rho*u*v) * val_normal[0];
130  val_Proj_Flux[nSpecies+2] = (rho*u*w) * val_normal[0];
131  val_Proj_Flux[nSpecies+3] = (rho*u*H) * val_normal[0];
132  val_Proj_Flux[nSpecies+4] = (rhoEve*u) * val_normal[0];
133 
134  /*--- iDim = 0 (y-direction) ---*/
135  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
136  val_Proj_Flux[iSpecies] += (rhos[iSpecies]*v) * val_normal[1];
137  val_Proj_Flux[nSpecies] += (rho*v*u) * val_normal[1];
138  val_Proj_Flux[nSpecies+1] += (rho*v*v + P) * val_normal[1];
139  val_Proj_Flux[nSpecies+2] += (rho*v*w) * val_normal[1];
140  val_Proj_Flux[nSpecies+3] += (rho*v*H) * val_normal[1];
141  val_Proj_Flux[nSpecies+4] += (rhoEve*v) * val_normal[1];
142 
143  /*--- iDim = 0 (z-direction) ---*/
144  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++)
145  val_Proj_Flux[iSpecies] += (rhos[iSpecies]*w) * val_normal[2];
146  val_Proj_Flux[nSpecies] += (rho*w*u) * val_normal[2];
147  val_Proj_Flux[nSpecies+1] += (rho*w*v) * val_normal[2];
148  val_Proj_Flux[nSpecies+2] += (rho*w*w + P) * val_normal[2];
149  val_Proj_Flux[nSpecies+3] += (rho*w*H) * val_normal[2];
150  val_Proj_Flux[nSpecies+4] += (rhoEve*w) * val_normal[2];
151  }
152 }

154 void CNEMONumerics::GetInviscidProjJac(const su2double *val_U, const su2double *val_V,
155  const su2double *val_dPdU, const su2double *val_normal,
156  const su2double val_scale,
157  su2double **val_Proj_Jac_Tensor) {
158 
159  const su2double *rhos;
160 
161  rhos = &val_V[RHOS_INDEX];
162 
163  /*--- Initialize the Jacobian tensor ---*/
164  for (unsigned short iVar = 0; iVar < nVar; iVar++)
165  for (unsigned short jVar = 0; jVar < nVar; jVar++)
166  val_Proj_Jac_Tensor[iVar][jVar] = 0.0;
167 
168  /*--- Rename for convenience ---*/
169  su2double rho = val_V[RHO_INDEX];
170  su2double H = val_V[H_INDEX];
171  su2double rhoEve = val_U[nSpecies+nDim+1];
172 
173  su2double u[MAXNDIM];
174  for (unsigned short iDim = 0; iDim < nDim; iDim++)
175  u[iDim] = val_V[VEL_INDEX+iDim];
176 
177  /*--- Calculate projected velocity ---*/
178  su2double proj_vel = GeometryToolbox::DotProduct(nDim, u, val_normal);
179 
180  /*--- Species density rows ---*/
181  for (unsigned short iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
182  for (unsigned short jSpecies = 0; jSpecies < nSpecies; jSpecies++) {
183  val_Proj_Jac_Tensor[iSpecies][jSpecies] += -(rhos[iSpecies]/rho) * proj_vel;
184  }
185  val_Proj_Jac_Tensor[iSpecies][iSpecies] += proj_vel;
186  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
187  val_Proj_Jac_Tensor[iSpecies][nSpecies+iDim] += (rhos[iSpecies]/rho) * val_normal[iDim];
188  val_Proj_Jac_Tensor[nSpecies+iDim][iSpecies] += val_dPdU[iSpecies]*val_normal[iDim] - proj_vel*u[iDim];
189  }
190  val_Proj_Jac_Tensor[nSpecies+nDim][iSpecies] += (val_dPdU[iSpecies]-H) * proj_vel;
191  val_Proj_Jac_Tensor[nSpecies+nDim+1][iSpecies] += -proj_vel * rhoEve/rho;
192  }
193 
194  /*--- Momentum rows ---*/
195  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
196  for (unsigned short jDim = 0; jDim < nDim; jDim++) {
197  val_Proj_Jac_Tensor[nSpecies+iDim][nSpecies+jDim] += val_dPdU[nSpecies+jDim]*val_normal[iDim] + u[iDim]*val_normal[jDim];
198  }
199  val_Proj_Jac_Tensor[nSpecies+iDim][nSpecies+iDim] += proj_vel;
200  val_Proj_Jac_Tensor[nSpecies+iDim][nSpecies+nDim] += val_dPdU[nSpecies+nDim]*val_normal[iDim];
201  val_Proj_Jac_Tensor[nSpecies+iDim][nSpecies+nDim+1] += val_dPdU[nSpecies+nDim+1]*val_normal[iDim];
202  }
203 
204  /*--- Total energy row ---*/
205  for (unsigned short iDim = 0; iDim < nDim; iDim++)
206  val_Proj_Jac_Tensor[nSpecies+nDim][nSpecies+iDim] += val_dPdU[nSpecies+iDim]*proj_vel + H*val_normal[iDim];
207  val_Proj_Jac_Tensor[nSpecies+nDim][nSpecies+nDim] += (1+val_dPdU[nSpecies+nDim])*proj_vel;
208  val_Proj_Jac_Tensor[nSpecies+nDim][nSpecies+nDim+1] += val_dPdU[nSpecies+nDim+1] *proj_vel;
209 
210  /*--- Vib.-el. energy row ---*/
211  for (unsigned short iDim = 0; iDim < nDim; iDim++)
212  val_Proj_Jac_Tensor[nSpecies+nDim+1][nSpecies+iDim] = rhoEve/rho*val_normal[iDim];
213  val_Proj_Jac_Tensor[nSpecies+nDim+1][nSpecies+nDim+1] = proj_vel;
214 
215  for (unsigned short iVar = 0; iVar < nVar; iVar++)
216  for (unsigned short jVar = 0; jVar < nVar; jVar++)
217  val_Proj_Jac_Tensor[iVar][jVar] = val_scale * val_Proj_Jac_Tensor[iVar][jVar];
218 }

220 void CNEMONumerics::GetViscousProjFlux(const su2double *val_primvar,
221  const su2double* const* val_gradprimvar,
222  su2double *val_eve,
223  const su2double *val_normal,
224  const su2double *val_diffusioncoeff,
225  su2double val_lam_viscosity,
226  su2double val_eddy_viscosity,
227  su2double val_therm_conductivity,
228  su2double val_therm_conductivity_ve,
229  const CConfig *config) {
230 
231  if (ionization) {
232  SU2_MPI::Error("NEED TO IMPLEMENT IONIZED FUNCTIONALITY!!!",CURRENT_FUNCTION);
233  }
234 
235  // Requires a slightly non-standard primitive vector:
236  // Assumes - V = [Y1, ... , Yn, T, Tve, ... ]
237  // and gradient GV = [GY1, ... , GYn, GT, GTve, ... ]
238  // rather than the standard V = [r1, ... , rn, T, Tve, ... ]
239 
240  su2activematrix Flux_Tensor(nVar,nDim);
241 
242  /*--- Initialize ---*/
243  for (auto iVar = 0; iVar < nVar; iVar++) {
244  Proj_Flux_Tensor[iVar] = 0.0;
245  for (auto iDim = 0; iDim < nDim; iDim++)
246  Flux_Tensor[iVar][iDim] = 0.0;
247  }
248 
249  /*--- Rename variables for convenience ---*/
250  const auto& Ds = val_diffusioncoeff;
251  const su2double mu = val_lam_viscosity+val_eddy_viscosity;
252  su2double ktr = val_therm_conductivity;
253  su2double kve = val_therm_conductivity_ve;
254  const su2double rho = val_primvar[RHO_INDEX];
255  const su2double T = val_primvar[T_INDEX];
256  const su2double Tve = val_primvar[TVE_INDEX];
257  const auto& V = val_primvar;
258  const auto& GV = val_gradprimvar;
259  const auto& hs = fluidmodel->ComputeSpeciesEnthalpy(T, Tve, val_eve);
260 
261  /*--- Pre-compute mixture quantities ---*/ //TODO
262  su2double Vector[MAXNDIM] = {0.0};
263  for (auto iDim = 0; iDim < nDim; iDim++) {
264  for (auto iSpecies = 0; iSpecies < nHeavy; iSpecies++) {
265  Vector[iDim] += rho*Ds[iSpecies]*GV[RHOS_INDEX+iSpecies][iDim];
266  }
267  }
268 
269  /*--- Compute the viscous stress tensor ---*/
270  ComputeStressTensor(nDim,tau,val_gradprimvar+VEL_INDEX, mu);
271 
272  /*--- Populate entries in the viscous flux vector ---*/
273  for (auto iDim = 0; iDim < nDim; iDim++) {
274 
275  /*--- Species diffusion velocity ---*/
276  for (auto iSpecies = 0; iSpecies < nHeavy; iSpecies++) {
277  Flux_Tensor[iSpecies][iDim] = rho*Ds[iSpecies]*GV[RHOS_INDEX+iSpecies][iDim]
278  - V[RHOS_INDEX+iSpecies]*Vector[iDim];
279  }
280 
281  /*--- Shear-stress/momentum related terms ---*/
282  Flux_Tensor[nSpecies+nDim][iDim] = 0.0;
283  for (auto jDim = 0; jDim < nDim; jDim++) {
284  Flux_Tensor[nSpecies+jDim][iDim] = tau[iDim][jDim];
285  Flux_Tensor[nSpecies+nDim][iDim] += tau[iDim][jDim]*val_primvar[VEL_INDEX+jDim];
286  }
287 
288  /*--- Diffusion terms ---*/
289  for (auto iSpecies = 0; iSpecies < nHeavy; iSpecies++) {
290  Flux_Tensor[nSpecies+nDim][iDim] += Flux_Tensor[iSpecies][iDim] * hs[iSpecies];
291  Flux_Tensor[nSpecies+nDim+1][iDim] += Flux_Tensor[iSpecies][iDim] * val_eve[iSpecies];
292  }
293 
294  /*--- Heat transfer terms ---*/
295  Flux_Tensor[nSpecies+nDim][iDim] += ktr*GV[T_INDEX][iDim] + kve*GV[TVE_INDEX][iDim];
296  Flux_Tensor[nSpecies+nDim+1][iDim] += kve*GV[TVE_INDEX][iDim];
297  }
298 
299  for (auto iVar = 0; iVar < nVar; iVar++) {
300  for (auto iDim = 0; iDim < nDim; iDim++) {
301  Proj_Flux_Tensor[iVar] += Flux_Tensor[iVar][iDim]*val_normal[iDim];
302  }
303  }
304 }

355 void CNEMONumerics::GetPMatrix(const su2double *U, const su2double *V, const su2double *val_dPdU,
356  const su2double *val_normal, const su2double *l, const su2double *m,
357  su2double **val_p_tensor) const {
358 
359  // P matrix is equivalent to the L matrix in Gnoffo
360  unsigned short iSpecies, iDim, iVar, jVar;
361  su2double sqvel, rho, a, a2, eve;
362  su2double vU, vV, vW;
363 
364  /*--- Initialize the P matrix to zero ---*/
365  for (iVar = 0; iVar < nVar; iVar++)
366  for (jVar = 0; jVar < nVar; jVar++)
367  val_p_tensor[iVar][jVar] = 0.0;
368 
369  /*--- Pre-compute useful quantities ---*/
370  sqvel = 0.0;
371  rho = V[RHO_INDEX];
372  eve = U[nSpecies+nDim+1]/rho;
373  vU = 0.0; vV = 0.0; vW = 0.0;
374  for (iDim = 0; iDim < nDim; iDim++) {
375  vU += V[VEL_INDEX+iDim] * val_normal[iDim];
376  vV += V[VEL_INDEX+iDim] * l[iDim];
377  vW += V[VEL_INDEX+iDim] * m[iDim];
378  sqvel += V[VEL_INDEX+iDim] * V[VEL_INDEX+iDim];
379  }
380  a = V[A_INDEX];
381  a2 = V[A_INDEX]*V[A_INDEX];
382 
383  if(nDim == 2) {
384  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
385  val_p_tensor[iSpecies][iSpecies] += 1.0/a2;
386  val_p_tensor[iSpecies][nSpecies] += 0.0;
387  val_p_tensor[iSpecies][nSpecies+1] += V[RHOS_INDEX+iSpecies] / (2.0*rho*a2);
388  val_p_tensor[iSpecies][nSpecies+2] += V[RHOS_INDEX+iSpecies] / (2.0*rho*a2);
389  val_p_tensor[iSpecies][nSpecies+3] += 0.0;
390 
391  val_p_tensor[nSpecies][iSpecies] += V[VEL_INDEX] / a2;
392  val_p_tensor[nSpecies+1][iSpecies] += V[VEL_INDEX+1] / a2;
393  val_p_tensor[nSpecies+2][iSpecies] += (val_dPdU[nSpecies+nDim]*sqvel-val_dPdU[iSpecies]) /
394  (val_dPdU[nSpecies+nDim]*a2);
395  val_p_tensor[nSpecies+3][iSpecies] += 0.0;
396  }
397 
398  for (iDim = 0; iDim < nDim; iDim++){
399  val_p_tensor[nSpecies+iDim][nSpecies] += l[iDim];
400  val_p_tensor[nSpecies+iDim][nSpecies+1] += (V[VEL_INDEX+iDim]+a*val_normal[iDim]) / (2.0*a2);
401  val_p_tensor[nSpecies+iDim][nSpecies+2] += (V[VEL_INDEX+iDim]-a*val_normal[iDim]) / (2.0*a2);
402  val_p_tensor[nSpecies+iDim][nSpecies+3] += 0.0;
403  }
404 
405  val_p_tensor[nSpecies+2][nSpecies] += vV;
406  val_p_tensor[nSpecies+2][nSpecies+1] += ((V[H_INDEX])+a*vU) / (2.0*a2);
407  val_p_tensor[nSpecies+2][nSpecies+2] += ((V[H_INDEX])-a*vU) / (2.0*a2);
408  val_p_tensor[nSpecies+2][nSpecies+3] += -val_dPdU[nSpecies+nDim+1] / (val_dPdU[nSpecies+nDim]*a2);
409 
410  val_p_tensor[nSpecies+3][nSpecies] += 0.0;
411  val_p_tensor[nSpecies+3][nSpecies+1] += eve / (2.0*a2);
412  val_p_tensor[nSpecies+3][nSpecies+2] += eve / (2.0*a2);
413  val_p_tensor[nSpecies+3][nSpecies+3] += 1.0 / a2;
414 
415  } else {
416 
417  for (iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
418  val_p_tensor[iSpecies][iSpecies] = 1.0/a2;
419  val_p_tensor[iSpecies][nSpecies] = 0.0;
420  val_p_tensor[iSpecies][nSpecies+1] = 0.0;
421  val_p_tensor[iSpecies][nSpecies+2] = V[RHOS_INDEX+iSpecies] / (2.0*rho*a2);
422  val_p_tensor[iSpecies][nSpecies+3] = V[RHOS_INDEX+iSpecies] / (2.0*rho*a2);
423  val_p_tensor[iSpecies][nSpecies+4] = 0.0;
424 
425  val_p_tensor[nSpecies][iSpecies] = V[VEL_INDEX] / a2;
426  val_p_tensor[nSpecies+1][iSpecies] = V[VEL_INDEX+1] / a2;
427  val_p_tensor[nSpecies+2][iSpecies] = V[VEL_INDEX+2] / a2;
428  val_p_tensor[nSpecies+3][iSpecies] = (val_dPdU[nSpecies+3]*sqvel-val_dPdU[iSpecies])
429  / (val_dPdU[nSpecies+3]*a2);
430  val_p_tensor[nSpecies+4][iSpecies] = 0.0;
431  }
432 
433  for (iDim = 0; iDim < nDim; iDim++){
434  val_p_tensor[nSpecies+iDim][nSpecies] = l[iDim];
435  val_p_tensor[nSpecies+iDim][nSpecies+1] = m[iDim];
436  val_p_tensor[nSpecies+iDim][nSpecies+2] = (V[VEL_INDEX+iDim]+a*val_normal[iDim]) / (2.0*a2);

478

483  val_invp_tensor[iSpecies][jSpecies] += -(V[RHOS_INDEX+iSpecies]/rho) * val_dPdU[jSpecies];
484  }

513  val_invp_tensor[nSpecies+3][nSpecies+1] += val_dPdU[nSpecies+nDim] * V[VEL_INDEX+1] * eve;
514  val_invp_tensor[nSpecies+3][nSpecies+2] += -val_dPdU[nSpecies+nDim] * eve;
515  val_invp_tensor[nSpecies+3][nSpecies+3] += a2 - val_dPdU[nSpecies+nDim+1]*eve;
