103 void CAvgGrad_Base::CorrectGradient(su2double** GradPrimVar,
104  const su2double* val_PrimVar_i,
105  const su2double* val_PrimVar_j,
106  const su2double* val_edge_vector,
107  const su2double val_dist_ij_2,
108  const unsigned short val_nPrimVar) {
109  for (unsigned short iVar = 0; iVar < val_nPrimVar; iVar++) {
110  Proj_Mean_GradPrimVar_Edge[iVar] = 0.0;
111  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
112  Proj_Mean_GradPrimVar_Edge[iVar] += GradPrimVar[iVar][iDim]*val_edge_vector[iDim];
113  }
114  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
115  GradPrimVar[iVar][iDim] -= (Proj_Mean_GradPrimVar_Edge[iVar] -
116  (val_PrimVar_j[iVar]-val_PrimVar_i[iVar]))*val_edge_vector[iDim] / val_dist_ij_2;
117  }

121 void CAvgGrad_Base::SetStressTensor(const su2double *val_primvar,
122  const su2double* const *val_gradprimvar,
123  const su2double val_turb_ke,
124  const su2double val_laminar_viscosity,
125  const su2double val_eddy_viscosity) {
126 
127  const su2double Density = val_primvar[nDim+2];
128 
129  /* --- If UQ methodology is used, use the perturbed Reynolds stress tensor
130  * for the turbulent part of tau. Otherwise both the laminar and turbulent
131  * parts of tau can be computed with the total viscosity. --- */
132 
133  if (sstParsedOptions.uq) {
134  // laminar part
135  ComputeStressTensor(nDim, tau, val_gradprimvar+1, val_laminar_viscosity);
136  // add turbulent part which was perturbed
137  for (unsigned short iDim = 0 ; iDim < nDim; iDim++)
138  for (unsigned short jDim = 0 ; jDim < nDim; jDim++)
139  tau[iDim][jDim] += (-Density) * MeanPerturbedRSM[iDim][jDim];
140  } else {
141  const su2double total_viscosity = val_laminar_viscosity + val_eddy_viscosity;
142  // turb_ke is not considered in the stress tensor, see #797
143  ComputeStressTensor(nDim, tau, val_gradprimvar+1, total_viscosity, Density, su2double(0.0));
144  }

147 void CAvgGrad_Base::AddTauWall(const su2double *UnitNormal,
148  const su2double TauWall) {
149 
150  /*--- Compute the wall shear stress as the magnitude of the
151  tangential projection of the shear stress tensor. ---*/
152 
153  su2double TauTangent[MAXNDIM];
154  GeometryToolbox::TangentProjection(nDim, tau, UnitNormal, TauTangent);
155 
156  su2double WallShearStress = GeometryToolbox::Norm(nDim, TauTangent);
157  su2double Scale = TauWall / WallShearStress;
158 
159  /*--- Scale the stress tensor by the ratio of the wall shear stress
160  (from wall functions) to the one computed above. ---*/
161 
162  for (auto iDim = 0u; iDim < nDim; iDim++)
163  for (auto jDim = 0u; jDim < nDim; jDim++)
164  tau[iDim][jDim] *= Scale;
165 }

167 void CAvgGrad_Base::SetTauJacobian(const su2double *val_Mean_PrimVar,
168  const su2double val_laminar_viscosity,
169  const su2double val_eddy_viscosity,
170  const su2double val_dist_ij,
171  const su2double *val_normal) {
172 
173  /*--- QCR and wall functions are **not** accounted for here ---*/
174 
175  const su2double Density = val_Mean_PrimVar[nDim+2];
176  const su2double total_viscosity = val_laminar_viscosity + val_eddy_viscosity;
177  const su2double xi = total_viscosity/(Density*val_dist_ij);
178 
179  for (unsigned short iDim = 0; iDim < nDim; iDim++) {
180  for (unsigned short jDim = 0; jDim < nDim; jDim++) {
181  // Jacobian w.r.t. momentum
182  tau_jacobian_i[iDim][jDim+1] = -xi*(delta[iDim][jDim] + val_normal[iDim]*val_normal[jDim]/3.0);
183  }
184  // Jacobian w.r.t. density
185  tau_jacobian_i[iDim][0] = 0;
186  for (unsigned short jDim = 0; jDim < nDim; jDim++) {
187  tau_jacobian_i[iDim][0] -= tau_jacobian_i[iDim][jDim+1]*val_Mean_PrimVar[jDim+1];
188  }
189  // Jacobian w.r.t. energy
190  tau_jacobian_i[iDim][nDim+1] = 0;
191  }
192 }

212  const su2double *val_normal) {

221  Flux_Tensor[2][0] = tau[0][1];

223  heat_flux_vector[0];
224  Flux_Tensor[0][1] = 0.0;

229  } else {

232  Flux_Tensor[2][0] = tau[0][1];
233  Flux_Tensor[3][0] = tau[0][2];
234  Flux_Tensor[4][0] = tau[0][0]*val_primvar[1] + tau[0][1]*val_primvar[2] + tau[0][2]*val_primvar[3] +
235  heat_flux_vector[0];
236  Flux_Tensor[0][1] = 0.0;
237  Flux_Tensor[1][1] = tau[1][0];
238  Flux_Tensor[2][1] = tau[1][1];
239  Flux_Tensor[3][1] = tau[1][2];
240  Flux_Tensor[4][1] = tau[1][0]*val_primvar[1] + tau[1][1]*val_primvar[2] + tau[1][2]*val_primvar[3] +
241  heat_flux_vector[1];

258 void CAvgGrad_Base::GetViscousProjJacs(const su2double *val_Mean_PrimVar,
259  const su2double val_dS,
260  const su2double *val_Proj_Visc_Flux,
261  su2double **val_Proj_Jac_Tensor_i,
262  su2double **val_Proj_Jac_Tensor_j) {
263 
264  const su2double Density = val_Mean_PrimVar[nDim+2];
265  const su2double factor = 0.5/Density;
266 
267  if (nDim == 2) {
268 
269  val_Proj_Jac_Tensor_i[0][0] = 0.0;
270  val_Proj_Jac_Tensor_i[0][1] = 0.0;
271  val_Proj_Jac_Tensor_i[0][2] = 0.0;
272  val_Proj_Jac_Tensor_i[0][3] = 0.0;
273  val_Proj_Jac_Tensor_i[1][0] = val_dS*tau_jacobian_i[0][0];
274  val_Proj_Jac_Tensor_i[1][1] = val_dS*tau_jacobian_i[0][1];
275  val_Proj_Jac_Tensor_i[1][2] = val_dS*tau_jacobian_i[0][2];
276  val_Proj_Jac_Tensor_i[1][3] = val_dS*tau_jacobian_i[0][3];
277  val_Proj_Jac_Tensor_i[2][0] = val_dS*tau_jacobian_i[1][0];
278  val_Proj_Jac_Tensor_i[2][1] = val_dS*tau_jacobian_i[1][1];
279  val_Proj_Jac_Tensor_i[2][2] = val_dS*tau_jacobian_i[1][2];
280  val_Proj_Jac_Tensor_i[2][3] = val_dS*tau_jacobian_i[1][3];
281  const su2double contraction = tau_jacobian_i[0][0]*val_Mean_PrimVar[1] +
282  tau_jacobian_i[1][0]*val_Mean_PrimVar[2];
283  val_Proj_Jac_Tensor_i[3][0] = val_dS*(contraction - heat_flux_jac_i[0]);
284  val_Proj_Jac_Tensor_i[3][1] = -val_dS*(tau_jacobian_i[0][0] + heat_flux_jac_i[1]);
285  val_Proj_Jac_Tensor_i[3][2] = -val_dS*(tau_jacobian_i[1][0] + heat_flux_jac_i[2]);
286  val_Proj_Jac_Tensor_i[3][3] = -val_dS*heat_flux_jac_i[3];
287 
288  for (unsigned short iVar = 0; iVar < nVar; iVar++)
289  for (unsigned short jVar = 0; jVar < nVar; jVar++)
290  val_Proj_Jac_Tensor_j[iVar][jVar] = -val_Proj_Jac_Tensor_i[iVar][jVar];
291 
292  const su2double proj_viscousflux_vel= val_Proj_Visc_Flux[1]*val_Mean_PrimVar[1] +
293  val_Proj_Visc_Flux[2]*val_Mean_PrimVar[2];
294  val_Proj_Jac_Tensor_i[3][0] -= factor*proj_viscousflux_vel;
295  val_Proj_Jac_Tensor_j[3][0] -= factor*proj_viscousflux_vel;
296  val_Proj_Jac_Tensor_i[3][1] += factor*val_Proj_Visc_Flux[1];
297  val_Proj_Jac_Tensor_j[3][1] += factor*val_Proj_Visc_Flux[1];
298  val_Proj_Jac_Tensor_i[3][2] += factor*val_Proj_Visc_Flux[2];
299  val_Proj_Jac_Tensor_j[3][2] += factor*val_Proj_Visc_Flux[2];
300 
301 
302  } else {
303 
304  val_Proj_Jac_Tensor_i[0][0] = 0.0;
305  val_Proj_Jac_Tensor_i[0][1] = 0.0;
306  val_Proj_Jac_Tensor_i[0][2] = 0.0;
307  val_Proj_Jac_Tensor_i[0][3] = 0.0;
308  val_Proj_Jac_Tensor_i[0][4] = 0.0;
309  val_Proj_Jac_Tensor_i[1][0] = val_dS*tau_jacobian_i[0][0];
310  val_Proj_Jac_Tensor_i[1][1] = val_dS*tau_jacobian_i[0][1];
311  val_Proj_Jac_Tensor_i[1][2] = val_dS*tau_jacobian_i[0][2];
312  val_Proj_Jac_Tensor_i[1][3] = val_dS*tau_jacobian_i[0][3];
313  val_Proj_Jac_Tensor_i[1][4] = val_dS*tau_jacobian_i[0][4];
314  val_Proj_Jac_Tensor_i[2][0] = val_dS*tau_jacobian_i[1][0];
315  val_Proj_Jac_Tensor_i[2][1] = val_dS*tau_jacobian_i[1][1];
316  val_Proj_Jac_Tensor_i[2][2] = val_dS*tau_jacobian_i[1][2];
317  val_Proj_Jac_Tensor_i[2][3] = val_dS*tau_jacobian_i[1][3];
318  val_Proj_Jac_Tensor_i[2][4] = val_dS*tau_jacobian_i[1][4];

320  val_Proj_Jac_Tensor_i[3][1] = val_dS*tau_jacobian_i[2][1];
321  val_Proj_Jac_Tensor_i[3][2] = val_dS*tau_jacobian_i[2][2];
322  val_Proj_Jac_Tensor_i[3][3] = val_dS*tau_jacobian_i[2][3];
323  val_Proj_Jac_Tensor_i[3][4] = val_dS*tau_jacobian_i[2][4];
324  const su2double contraction = tau_jacobian_i[0][0]*val_Mean_PrimVar[1] +
325  tau_jacobian_i[1][0]*val_Mean_PrimVar[2] +
326  tau_jacobian_i[2][0]*val_Mean_PrimVar[3];
327  val_Proj_Jac_Tensor_i[4][0] = val_dS*(contraction - heat_flux_jac_i[0]);
328  val_Proj_Jac_Tensor_i[4][1] = -val_dS*(tau_jacobian_i[0][0] + heat_flux_jac_i[1]);

348 
349  }

359 CNumerics::ResidualType<> CAvgGrad_Flow::ComputeResidual(const CConfig* config) {
360 
361  implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
362 
363  AD::StartPreacc();
364  AD::SetPreaccIn(V_i, nDim+9); AD::SetPreaccIn(V_j, nDim+9);
365  AD::SetPreaccIn(Coord_i, nDim); AD::SetPreaccIn(Coord_j, nDim);
366  AD::SetPreaccIn(PrimVar_Grad_i, nDim+1, nDim);
367  AD::SetPreaccIn(PrimVar_Grad_j, nDim+1, nDim);
368  AD::SetPreaccIn(turb_ke_i); AD::SetPreaccIn(turb_ke_j);
369  AD::SetPreaccIn(TauWall_i); AD::SetPreaccIn(TauWall_j);
370  AD::SetPreaccIn(Normal, nDim);
371 
372  unsigned short iVar, jVar, iDim;
373 
374  /*--- Normalized normal vector ---*/
375 
376  Area = GeometryToolbox::Norm(nDim, Normal);
377 
378  for (iDim = 0; iDim < nDim; iDim++)
379  UnitNormal[iDim] = Normal[iDim]/Area;
380 
381  PrimVar_i = V_i;
382  PrimVar_j = V_j;
383 
384  for (iVar = 0; iVar < nPrimVar; iVar++) {
385  Mean_PrimVar[iVar] = 0.5*(PrimVar_i[iVar]+PrimVar_j[iVar]);
386  }
387 
388  /*--- Compute vector going from iPoint to jPoint ---*/
389 
390  dist_ij_2 = 0.0;
391  for (iDim = 0; iDim < nDim; iDim++) {
392  Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
393  dist_ij_2 += Edge_Vector[iDim]*Edge_Vector[iDim];
394  }
395 
396  /*--- Laminar and Eddy viscosity ---*/
397 
398  Laminar_Viscosity_i = V_i[nDim+5]; Laminar_Viscosity_j = V_j[nDim+5];
399  Eddy_Viscosity_i = V_i[nDim+6]; Eddy_Viscosity_j = V_j[nDim+6];
400 
401  /*--- Mean Viscosities and turbulent kinetic energy---*/
402 
403  Mean_Laminar_Viscosity = 0.5*(Laminar_Viscosity_i + Laminar_Viscosity_j);
404  Mean_Eddy_Viscosity = 0.5*(Eddy_Viscosity_i + Eddy_Viscosity_j);
405  Mean_turb_ke = 0.5*(turb_ke_i + turb_ke_j);
406 
407  /*--- Mean gradient approximation ---*/
408 
409  for (iVar = 0; iVar < nDim+1; iVar++) {
410  for (iDim = 0; iDim < nDim; iDim++) {
411  Mean_GradPrimVar[iVar][iDim] = 0.5*(PrimVar_Grad_i[iVar][iDim] + PrimVar_Grad_j[iVar][iDim]);
412  }
413  }
414 
415  /*--- Projection of the mean gradient in the direction of the edge ---*/
416 
417  if (correct_gradient && dist_ij_2 != 0.0) {
418  CorrectGradient(Mean_GradPrimVar, PrimVar_i, PrimVar_j, Edge_Vector,
419  dist_ij_2, nDim+1);
420  }
421 
422  /*--- Wall shear stress values (wall functions) only used if present for one but not both points (xor) ---*/
423 
424  const int scale = (TauWall_i > 0.0) ^ (TauWall_j > 0.0);
425  Mean_TauWall = (max(TauWall_i,0.0) + max(TauWall_j,0.0)) * scale;
426 
427  /*--- If using UQ methodology, set Reynolds Stress tensor and perform perturbation ---*/
428 
429  if (sstParsedOptions.uq) {
430  ComputePerturbedRSM(nDim, Eig_Val_Comp, uq_permute, uq_delta_b, uq_urlx,
431  Mean_GradPrimVar+1, Mean_PrimVar[nDim+2], Mean_Eddy_Viscosity,
432  Mean_turb_ke, MeanPerturbedRSM);
433  }
434 
435  /*--- Get projected flux tensor (viscous residual) ---*/
436 
437  SetStressTensor(Mean_PrimVar, Mean_GradPrimVar, Mean_turb_ke,
438  Mean_Laminar_Viscosity, Mean_Eddy_Viscosity);
439  if (config->GetSAParsedOptions().qcr2000) AddQCR(nDim, &Mean_GradPrimVar[1], tau);
440  if (Mean_TauWall > 0) AddTauWall(UnitNormal, Mean_TauWall);
441 
442  SetHeatFluxVector(Mean_GradPrimVar, Mean_Laminar_Viscosity,
443  Mean_Eddy_Viscosity);
444 
445  GetViscousProjFlux(Mean_PrimVar, Normal);
446 
447  /*--- Compute the implicit part ---*/
448

544 CNumerics::ResidualType<> CAvgGradInc_Flow::ComputeResidual(const CConfig* config) {
545 
546  implicit = (config->GetKind_TimeIntScheme() == EULER_IMPLICIT);
547 
548  AD::StartPreacc();
549  AD::SetPreaccIn(V_i, nDim+9); AD::SetPreaccIn(V_j, nDim+9);
550  AD::SetPreaccIn(Coord_i, nDim); AD::SetPreaccIn(Coord_j, nDim);
551  AD::SetPreaccIn(PrimVar_Grad_i, nVar, nDim);
552  AD::SetPreaccIn(PrimVar_Grad_j, nVar, nDim);
553  AD::SetPreaccIn(turb_ke_i); AD::SetPreaccIn(turb_ke_j);
554  AD::SetPreaccIn(TauWall_i); AD::SetPreaccIn(TauWall_j);
555  AD::SetPreaccIn(Normal, nDim);
556 
557  unsigned short iVar, jVar, iDim;
558 
559  /*--- Normalized normal vector ---*/
560 
561  Area = GeometryToolbox::Norm(nDim, Normal);
562 
563  for (iDim = 0; iDim < nDim; iDim++)
564  UnitNormal[iDim] = Normal[iDim]/Area;
565 
566  PrimVar_i = V_i;
567  PrimVar_j = V_j;
568 
569  for (iVar = 0; iVar < nPrimVar; iVar++) {
570  Mean_PrimVar[iVar] = 0.5*(PrimVar_i[iVar]+PrimVar_j[iVar]);
571  }
572 
573  /*--- Compute vector going from iPoint to jPoint ---*/
574 
575  dist_ij_2 = 0.0;
576  for (iDim = 0; iDim < nDim; iDim++) {
577  Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
578  dist_ij_2 += Edge_Vector[iDim]*Edge_Vector[iDim];
579  }
580 
581  /*--- Density and transport properties ---*/
582 
583  Laminar_Viscosity_i = V_i[nDim+4]; Laminar_Viscosity_j = V_j[nDim+4];
584  Eddy_Viscosity_i = V_i[nDim+5]; Eddy_Viscosity_j = V_j[nDim+5];
585  Thermal_Conductivity_i = V_i[nDim+6]; Thermal_Conductivity_j = V_j[nDim+6];
586 
587  /*--- Mean transport properties ---*/
588 
589  Mean_Laminar_Viscosity = 0.5*(Laminar_Viscosity_i + Laminar_Viscosity_j);
590  Mean_Eddy_Viscosity = 0.5*(Eddy_Viscosity_i + Eddy_Viscosity_j);
591  Mean_turb_ke = 0.5*(turb_ke_i + turb_ke_j);
592  Mean_Thermal_Conductivity = 0.5*(Thermal_Conductivity_i + Thermal_Conductivity_j);
593 
594  /*--- Mean gradient approximation ---*/
595 
596  for (iVar = 0; iVar < nVar; iVar++)
597  for (iDim = 0; iDim < nDim; iDim++)
598  Mean_GradPrimVar[iVar][iDim] = 0.5*(PrimVar_Grad_i[iVar][iDim] + PrimVar_Grad_j[iVar][iDim]);
599 
600  /*--- Projection of the mean gradient in the direction of the edge ---*/
601 
602  if (correct_gradient && dist_ij_2 != 0.0) {
603  CorrectGradient(Mean_GradPrimVar, PrimVar_i, PrimVar_j, Edge_Vector,
604  dist_ij_2, nVar);
605  }
606 
607  /*--- Wall shear stress values (wall functions) only used if present for one but not both points (xor) ---*/
608 
609  const int scale = (TauWall_i > 0.0) ^ (TauWall_j > 0.0);
610  Mean_TauWall = (max(TauWall_i,0.0) + max(TauWall_j,0.0)) * scale;
611 
612  /*--- If using UQ methodology, set Reynolds Stress tensor and perform perturbation ---*/
613 
614  if (sstParsedOptions.uq) {
615  ComputePerturbedRSM(nDim, Eig_Val_Comp, uq_permute, uq_delta_b, uq_urlx,
616  Mean_GradPrimVar+1, Mean_PrimVar[nDim+2], Mean_Eddy_Viscosity,
617  Mean_turb_ke, MeanPerturbedRSM);
618  }
619 
620  /*--- Get projected flux tensor (viscous residual) ---*/
621  SetStressTensor(Mean_PrimVar, Mean_GradPrimVar, Mean_turb_ke,
622  Mean_Laminar_Viscosity, Mean_Eddy_Viscosity);
623  if (config->GetSAParsedOptions().qcr2000) AddQCR(nDim, &Mean_GradPrimVar[1], tau);
624  if (Mean_TauWall > 0) AddTauWall(UnitNormal, Mean_TauWall);
625 
626  GetViscousIncProjFlux(Mean_GradPrimVar, Normal, Mean_Thermal_Conductivity);
627 
628  /*--- Implicit part ---*/
629 
630  if (implicit) {
631 
632  if (dist_ij_2 == 0.0) {
633  for (iVar = 0; iVar < nVar; iVar++) {
634  for (jVar = 0; jVar < nVar; jVar++) {
635  Jacobian_i[iVar][jVar] = 0.0;
636  Jacobian_j[iVar][jVar] = 0.0;
637  }
638  }
639  } else {
640 
641  const su2double dist_ij = sqrt(dist_ij_2);
642  SetIncTauJacobian(Mean_Laminar_Viscosity, Mean_Eddy_Viscosity, dist_ij, UnitNormal);
643 
644  GetViscousIncProjJacs(Area, Jacobian_i, Jacobian_j);
645 
646  /*--- Include the temperature equation Jacobian. ---*/
647  su2double proj_vector_ij = 0.0;
648  for (iDim = 0; iDim < nDim; iDim++) {
649  proj_vector_ij += (Coord_j[iDim]-Coord_i[iDim])*Normal[iDim];
650  }
651  proj_vector_ij = proj_vector_ij/dist_ij_2;
652  Jacobian_i[nDim+1][nDim+1] = -Mean_Thermal_Conductivity*proj_vector_ij;
653  Jacobian_j[nDim+1][nDim+1] = Mean_Thermal_Conductivity*proj_vector_ij;
654  }
655 
656  }
657 
658  if (!energy) {
659  Proj_Flux_Tensor[nDim+1] = 0.0;
660  if (implicit) {
661  for (iVar = 0; iVar < nVar; iVar++) {
662  Jacobian_i[iVar][nDim+1] = 0.0;
663  Jacobian_j[iVar][nDim+1] = 0.0;
664 
665  Jacobian_i[nDim+1][iVar] = 0.0;
666  Jacobian_j[nDim+1][iVar] = 0.0;
667  }
668  }
669  }
670 
671  AD::SetPreaccOut(Proj_Flux_Tensor, nVar);
672  AD::EndPreacc();
673 
674  return ResidualType<>(Proj_Flux_Tensor, Jacobian_i, Jacobian_j);
675

919  Mean_Thermal_Conductivity = 0.5*(Thermal_Conductivity_i + Thermal_Conductivity_j);

923 
924  for (iVar = 0; iVar < nDim+1; iVar++) {
925  for (iDim = 0; iDim < nDim; iDim++) {
926  Mean_GradPrimVar[iVar][iDim] = 0.5*(PrimVar_Grad_i[iVar][iDim] + PrimVar_Grad_j[iVar][iDim]);
927  }
928  }
929

931

933  CorrectGradient(Mean_GradPrimVar, PrimVar_i, PrimVar_j, Edge_Vector,

935  }

938 
939  const int scale = (TauWall_i > 0.0) ^ (TauWall_j > 0.0);
940  Mean_TauWall = (max(TauWall_i,0.0) + max(TauWall_j,0.0)) * scale;
941

943 
944  if (sstParsedOptions.uq) {
945  ComputePerturbedRSM(nDim, Eig_Val_Comp, uq_permute, uq_delta_b, uq_urlx,
946  Mean_GradPrimVar+1, Mean_PrimVar[nDim+2], Mean_Eddy_Viscosity,
947  Mean_turb_ke, MeanPerturbedRSM);
948  }
949 
950  /*--- Get projected flux tensor (viscous residual) ---*/
951 
952  SetStressTensor(Mean_PrimVar, Mean_GradPrimVar, Mean_turb_ke,
953  Mean_Laminar_Viscosity, Mean_Eddy_Viscosity);
954  if (config->GetSAParsedOptions().qcr2000) AddQCR(nDim, &Mean_GradPrimVar[1], tau);
955  if (Mean_TauWall > 0) AddTauWall(UnitNormal, Mean_TauWall);
956 
957  SetHeatFluxVector(Mean_GradPrimVar, Mean_Laminar_Viscosity,
958  Mean_Eddy_Viscosity, Mean_Thermal_Conductivity, Mean_Cp);
959 
960  GetViscousProjFlux(Mean_PrimVar, Normal);
961 
962  /*--- Compute the implicit part ---*/
963 
964  if (implicit) {
965 
966  if (dist_ij_2 == 0.0) {
967  for (iVar = 0; iVar < nVar; iVar++) {
968  for (jVar = 0; jVar < nVar; jVar++) {
969  Jacobian_i[iVar][jVar] = 0.0;
970  Jacobian_j[iVar][jVar] = 0.0;
971  }
972  }
973  } else {
974  const su2double dist_ij = sqrt(dist_ij_2);
975 
976  SetTauJacobian(Mean_PrimVar, Mean_Laminar_Viscosity, Mean_Eddy_Viscosity, dist_ij, UnitNormal);
977 
978  SetHeatFluxJacobian(Mean_PrimVar, Mean_SecVar, Mean_Eddy_Viscosity,
979  Mean_Thermal_Conductivity, Mean_Cp, dist_ij);
980 
981  GetViscousProjJacs(Mean_PrimVar, Area, Proj_Flux_Tensor, Jacobian_i, Jacobian_j);
982  }
983 
984  }
985 
986  AD::SetPreaccOut(Proj_Flux_Tensor, nVar);
987  AD::EndPreacc();
988 
989  return ResidualType<>(Proj_Flux_Tensor, Jacobian_i, Jacobian_j);
