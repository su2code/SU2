103  /*!
104  * \brief Helper function, prunes (by setting to zero) small interpolation coefficients,
105  * i.e. <= tolerance*max(abs(coeffs)). The vector is re-scaled such that sum(coeffs)==1.
106  * \param[in] tolerance - Relative pruning tolerance.
107  * \param[in] size - Size of the coefficient vector.
108  * \param[in,out] coeffs - Iterator to start of vector of interpolation coefficients.
109  * \return Number of non-zero coefficients after pruning and correction factor.
110  */
111  template<typename Float, typename Int, class ForwardIt>
112  static pair<Int,Float> PruneSmallCoefficients(Float tolerance, Int size, ForwardIt coeffs) {
113 
114  /*--- Determine the pruning threshold. ---*/
115  Float thresh = 0.0;
116  auto end = coeffs;
117  for (Int i = 0; i < size; ++i)
118  thresh = max(thresh, fabs(*(end++)));
119  thresh *= tolerance;
120 
121  /*--- Prune and count non-zeros. ---*/
122  Int numNonZeros = 0;
123  Float coeffSum = 0.0;
124  for (auto it = coeffs; it != end; ++it) {
125  if (fabs(*it) > thresh) { // keep
126  coeffSum += *it;
127  ++numNonZeros;
128  }
129  else { *it = 0.0; } // prune
130  }
131 
132  /*--- Correct remaining coefficients, sum must be 1 for conservation. ---*/
133  Float correction = 1.0 / coeffSum;
134  while (coeffs != end) *(coeffs++) *= correction;
135 
136  return make_pair(numNonZeros, correction);
137  }
