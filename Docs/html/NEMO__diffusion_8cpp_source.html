103 CNumerics::ResidualType<> CAvgGrad_NEMO::ComputeResidual(const CConfig *config) {
104 
105  /*--- Normalized normal vector ---*/
106  Area = GeometryToolbox::Norm(nDim, Normal);
107 
108  for (auto iDim = 0; iDim < nDim; iDim++)
109  UnitNormal[iDim] = Normal[iDim]/Area;
110 
111  /*--- Mean transport coefficients ---*/
112  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++)
113  Mean_Diffusion_Coeff[iSpecies] = 0.5*(Diffusion_Coeff_i[iSpecies] +
114  Diffusion_Coeff_j[iSpecies]);
115  Mean_Laminar_Viscosity = 0.5*(Laminar_Viscosity_i +
116  Laminar_Viscosity_j);
117  Mean_Eddy_Viscosity = 0.5*(Eddy_Viscosity_i +
118  Eddy_Viscosity_j);
119  Mean_Thermal_Conductivity = 0.5*(Thermal_Conductivity_i +
120  Thermal_Conductivity_j);
121  Mean_Thermal_Conductivity_ve = 0.5*(Thermal_Conductivity_ve_i +
122  Thermal_Conductivity_ve_j);
123 
124  /*--- Mean gradient approximation ---*/
125  // Mass fraction
126  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
127  PrimVar_i[iSpecies] = V_i[iSpecies]/V_i[RHO_INDEX];
128  PrimVar_j[iSpecies] = V_j[iSpecies]/V_j[RHO_INDEX];
129  Mean_PrimVar[iSpecies] = 0.5*(PrimVar_i[iSpecies] + PrimVar_j[iSpecies]);
130  for (auto iDim = 0; iDim < nDim; iDim++) {
131  Mean_GradPrimVar[iSpecies][iDim] = 0.5*(1.0/V_i[RHO_INDEX] * (PrimVar_Grad_i[iSpecies][iDim] -
132  PrimVar_i[iSpecies] * PrimVar_Grad_i[RHO_INDEX][iDim]) +
133  1.0/V_j[RHO_INDEX] * (PrimVar_Grad_j[iSpecies][iDim] -
134  PrimVar_j[iSpecies] * PrimVar_Grad_j[RHO_INDEX][iDim]));
135  }
136  }
137 
138  for (auto iVar = nSpecies; iVar < nPrimVar; iVar++) {
139  PrimVar_i[iVar] = V_i[iVar];
140  PrimVar_j[iVar] = V_j[iVar];
141  Mean_PrimVar[iVar] = 0.5*(PrimVar_i[iVar]+PrimVar_j[iVar]);
142  }
143  for (auto iVar = nSpecies; iVar < nPrimVarGrad; iVar++) {
144  for (auto iDim = 0; iDim < nDim; iDim++) {
145  Mean_GradPrimVar[iVar][iDim] = 0.5*(PrimVar_Grad_i[iVar][iDim] +
146  PrimVar_Grad_j[iVar][iDim]);
147  }
148  }
149  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
150  Mean_Eve[iSpecies] = 0.5*(eve_i[iSpecies] + eve_j[iSpecies]);
151  Mean_Cvve[iSpecies] = 0.5*(Cvve_i[iSpecies] + Cvve_j[iSpecies]);
152  }
153 
154  /*--- Get projected flux tensor ---*/
155  GetViscousProjFlux(Mean_PrimVar, Mean_GradPrimVar, Mean_Eve, Normal,
156  Mean_Diffusion_Coeff, Mean_Laminar_Viscosity, Mean_Eddy_Viscosity,
157  Mean_Thermal_Conductivity, Mean_Thermal_Conductivity_ve,
158  config);
159 
160 
161  /*--- Update viscous residual ---*/
162  for (auto iVar = 0; iVar < nVar; iVar++)
163  Flux[iVar] = Proj_Flux_Tensor[iVar];
164 
165  /*--- Compute the implicit part ---*/
166  if (implicit) {
167  dist_ij = 0.0;
168  for (auto iDim = 0; iDim < nDim; iDim++)
169  dist_ij += (Coord_j[iDim]-Coord_i[iDim])*(Coord_j[iDim]-Coord_i[iDim]);
170  dist_ij = sqrt(dist_ij);
171 
172  for (auto iVar = 0; iVar < nVar; iVar++) {
173  for (auto jVar = 0; jVar < nVar; jVar++) {
174  Jacobian_i[iVar][jVar] = 0.0;
175  Jacobian_j[iVar][jVar] = 0.0;
176  }
177  }
178  GetViscousProjJacs(Mean_PrimVar, Mean_Eve, Mean_Cvve,
179  Mean_Diffusion_Coeff, Mean_Laminar_Viscosity,Mean_Eddy_Viscosity,
180  Mean_Thermal_Conductivity, Mean_Thermal_Conductivity_ve,
181  dist_ij, UnitNormal, Area, Proj_Flux_Tensor,
182  Jacobian_i, Jacobian_j, config);
183  }
184 
185  return ResidualType<>(Flux, Jacobian_i, Jacobian_j);
186 }

258 CNumerics::ResidualType<> CAvgGradCorrected_NEMO::ComputeResidual(const CConfig *config) {
259 
260  /*--- Normalized normal vector ---*/
261  Area = GeometryToolbox::Norm(nDim, Normal);
262 
263  for (auto iDim = 0; iDim < nDim; iDim++)
264  UnitNormal[iDim] = Normal[iDim]/Area;
265 
266  /*--- Compute vector going from iPoint to jPoint ---*/
267  for (auto iDim = 0; iDim < nDim; iDim++) {
268  Edge_Vector[iDim] = Coord_j[iDim]-Coord_i[iDim];
269  }
270  su2double dist_ij_2 = GeometryToolbox::SquaredNorm(nDim, Edge_Vector);
271 
272  /*--- Make a local copy of the primitive variables ---*/
273  // NOTE: We are transforming the species density terms to species mass fractions
274  // Mass fraction
275  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
276  PrimVar_i[iSpecies] = V_i[iSpecies]/V_i[RHO_INDEX];
277  PrimVar_j[iSpecies] = V_j[iSpecies]/V_j[RHO_INDEX];
278  Mean_PrimVar[iSpecies] = 0.5*(PrimVar_i[iSpecies] + PrimVar_j[iSpecies]);
279  for (auto iDim = 0; iDim < nDim; iDim++) {
280  Mean_GradPrimVar[iSpecies][iDim] = 0.5*(1.0/V_i[RHO_INDEX] *
281  (PrimVar_Grad_i[iSpecies][iDim] -
282  PrimVar_i[iSpecies] *
283  PrimVar_Grad_i[RHO_INDEX][iDim]) +
284  1.0/V_j[RHO_INDEX] *
285  (PrimVar_Grad_j[iSpecies][iDim] -
286  PrimVar_j[iSpecies] *
287  PrimVar_Grad_j[RHO_INDEX][iDim]));
288  }
289  }
290  for (auto iVar = nSpecies; iVar < nPrimVar; iVar++) {
291  PrimVar_i[iVar] = V_i[iVar];
292  PrimVar_j[iVar] = V_j[iVar];
293  Mean_PrimVar[iVar] = 0.5*(PrimVar_i[iVar]+PrimVar_j[iVar]);
294  }
295  for (auto iVar = nSpecies; iVar < nPrimVarGrad; iVar++) {
296  for (iDim = 0; iDim < nDim; iDim++) {
297  Mean_GradPrimVar[iVar][iDim] = 0.5*(PrimVar_Grad_i[iVar][iDim] +
298  PrimVar_Grad_j[iVar][iDim]);
299  }
300  }
301  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++) {
302  Mean_Eve[iSpecies] = 0.5*(eve_i[iSpecies] + eve_j[iSpecies]);
303  Mean_Cvve[iSpecies] = 0.5*(Cvve_i[iSpecies] + Cvve_j[iSpecies]);
304  }
305 
306  /*--- Mean transport coefficients ---*/
307  for (auto iSpecies = 0; iSpecies < nSpecies; iSpecies++)
308  Mean_Diffusion_Coeff[iSpecies] = 0.5*(Diffusion_Coeff_i[iSpecies] +
309  Diffusion_Coeff_j[iSpecies]);
310  Mean_Laminar_Viscosity = 0.5*(Laminar_Viscosity_i +
311  Laminar_Viscosity_j);
312  Mean_Eddy_Viscosity = 0.5*(Eddy_Viscosity_i +
313  Eddy_Viscosity_j);
314  Mean_Thermal_Conductivity = 0.5*(Thermal_Conductivity_i +
315  Thermal_Conductivity_j);
316  Mean_Thermal_Conductivity_ve = 0.5*(Thermal_Conductivity_ve_i +
317  Thermal_Conductivity_ve_j);
318 
319  /*--- Projection of the mean gradient in the direction of the edge ---*/
320  for (auto iVar = 0; iVar < nPrimVarGrad; iVar++) {
321  Proj_Mean_GradPrimVar_Edge[iVar] = GeometryToolbox::DotProduct(nDim, Mean_GradPrimVar[iVar], Edge_Vector);
322  for (auto iDim = 0; iDim < nDim; iDim++) {
323  Mean_GradPrimVar[iVar][iDim] -= (Proj_Mean_GradPrimVar_Edge[iVar] -
324  (PrimVar_j[iVar]-PrimVar_i[iVar]))*Edge_Vector[iDim] / dist_ij_2;
325  }
326  }
327 
328  /*--- Get projected flux tensor ---*/
329  GetViscousProjFlux(Mean_PrimVar, Mean_GradPrimVar, Mean_Eve,
330  Normal, Mean_Diffusion_Coeff,
331  Mean_Laminar_Viscosity,
332  Mean_Eddy_Viscosity,
333  Mean_Thermal_Conductivity,
334  Mean_Thermal_Conductivity_ve,
335  config);
336 
337  /*--- Update viscous residual ---*/
338  for (auto iVar = 0; iVar < nVar; iVar++)
339  Flux[iVar] = Proj_Flux_Tensor[iVar];
340 
341  /*--- Compute the implicit part ---*/
342  if (implicit) {
343  dist_ij = sqrt(dist_ij_2);
344 
345  for (auto iVar = 0; iVar < nVar; iVar++) {
346  for (unsigned short jVar = 0; jVar < nVar; jVar++) {
347  Jacobian_i[iVar][jVar] = 0.0;
348  Jacobian_j[iVar][jVar] = 0.0;
349  }
350  }
351  GetViscousProjJacs(Mean_PrimVar, Mean_Eve, Mean_Cvve,
352  Mean_Diffusion_Coeff, Mean_Laminar_Viscosity, Mean_Eddy_Viscosity,
353  Mean_Thermal_Conductivity, Mean_Thermal_Conductivity_ve,
354  dist_ij, UnitNormal, Area, Proj_Flux_Tensor,
355  Jacobian_i, Jacobian_j, config);
356 
357  }
358 
359  return ResidualType<>(Flux, Jacobian_j, Jacobian_j);
360 }
