112 /*!
113  * \brief Tangential projection.
114  */
115 template<size_t nDim>
116 FORCEINLINE VectorDbl<nDim> tangentProjection(const MatrixDbl<nDim>& tensor,
117  const VectorDbl<nDim>& unitVector) {
118  VectorDbl<nDim> proj;
119  for (size_t iDim = 0; iDim < nDim; ++iDim)
120  proj(iDim) = dot(tensor[iDim], unitVector);
121 
122  Double normalProj = dot(proj, unitVector);
123 
124  for (size_t iDim = 0; iDim < nDim; ++iDim)
125  proj(iDim) -= normalProj * unitVector(iDim);
126 
127  return proj;
128 }

217 /*!
218  * \brief Distance vector, from point i to point j.
219  */
220 template<size_t nDim, class Container>
221 FORCEINLINE VectorDbl<nDim> distanceVector(Int iPoint, Int jPoint,
222  const Container& coords) {
223  auto coord_i = gatherVariables<nDim>(iPoint, coords);
224  auto coord_j = gatherVariables<nDim>(jPoint, coords);
225  VectorDbl<nDim> vector_ij;
226  for (size_t iDim = 0; iDim < nDim; ++iDim) {
227  vector_ij(iDim) = coord_j(iDim) - coord_i(iDim);
228  }
229  return vector_ij;
230 }
